221  public override void PrepareTriangulation(ITriangulatable t)
222  {
223  base.PrepareTriangulation(t);
224 
225  double xmin;
226  double ymin;
227 
228  double xmax = xmin = Points[0].X;
229  double ymax = ymin = Points[0].Y;
230 
231  // Calculate bounds. Should be combined with the sorting
232  foreach (TriangulationPoint p in Points)
233  {
234  if (p.X > xmax)
235  {
236  xmax = p.X;
237  }
238 
239  if (p.X < xmin)
240  {
241  xmin = p.X;
242  }
243 
244  if (p.Y > ymax)
245  {
246  ymax = p.Y;
247  }
248 
249  if (p.Y < ymin)
250  {
251  ymin = p.Y;
252  }
253  }
254 
255  double deltaX = Alpha * (xmax - xmin);
256  double deltaY = Alpha * (ymax - ymin);
257  TriangulationPoint p1 = new TriangulationPoint(xmax + deltaX, ymin - deltaY);
258  TriangulationPoint p2 = new TriangulationPoint(xmin - deltaX, ymin - deltaY);
259 
260  Head = p1;
261  Tail = p2;
262 
263  // long time = System.nanoTime();
264  // Sort the points along y-axis
265  Points.Sort(comparator);
266 
267  // logger.info( "Triangulation setup [{}ms]", ( System.nanoTime() - time ) / 1e6 );
268  }
