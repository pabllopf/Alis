416  public int EdgeIndex(TriangulationPoint p1, TriangulationPoint p2)
417  {
418  int i1 = Points.IndexOf(p1);
419  int i2 = Points.IndexOf(p2);
420 
421  // Points of this triangle in the edge p1-p2
422  bool a = i1 == 0 || i2 == 0;
423  bool b = i1 == 1 || i2 == 1;
424  bool c = i1 == 2 || i2 == 2;
425 
426  if (b && c)
427  {
428  return 0;
429  }
430 
431  if (a && c)
432  {
433  return 1;
434  }
435 
436  if (a && b)
437  {
438  return 2;
439  }
440 
441  return -1;
442  }
