84  public Quaterniond(double pitch, double yaw, double roll)
85  {
86  yaw *= 0.5;
87  pitch *= 0.5;
88  roll *= 0.5;
89 
90  double c1 = Math.Cos(yaw);
91  double c2 = Math.Cos(pitch);
92  double c3 = Math.Cos(roll);
93  double s1 = Math.Sin(yaw);
94  double s2 = Math.Sin(pitch);
95  double s3 = Math.Sin(roll);
96 
97  W = c1 * c2 * c3 - s1 * s2 * s3;
98  Xyz.X = s1 * s2 * c3 + c1 * c2 * s3;
99  Xyz.Y = s1 * c2 * c3 - c1 * s2 * s3;

159  {

199  public Vector3d ToEulerAngles()
200  {
201  /*
202  reference
203  http://en.wikipedia.org/wiki/Conversion_between_qernions_and_Euler_angles
204  http://www.euclideanspace.com/maths/geometry/rotations/conversions/qernionToEuler/
205  */
206 
207  Quaterniond q = this;
208 
209  q.Normalize();
210 
211  double singularityTest = q.X * q.Y - q.W * q.Z;
212  double yawY = 2d * (q.W * q.X + q.Y * q.Z);
213  double yawX = 1d - 2d * (q.Z * q.Z + q.X * q.X);
214 
215  // Threshold for the singularities found at the north/south poles.
216  // TODO: Think about how this threshold should change with the added precision
217  const double SINGULARITY_THRESHOLD = 0.4999995d;
218 
219  Vector3d eulerAngles;
220 
221  if (singularityTest < -SINGULARITY_THRESHOLD)
222  {
223  eulerAngles.Z = -Math.PI / 2; // -90 degrees
224  eulerAngles.Y = Math.Atan2(yawY, yawX);
225  eulerAngles.X = MathHelper.NormalizeRadians(-eulerAngles.Y - 2d * Math.Atan2(q.Y, q.W));
226  }
227  else if (singularityTest > SINGULARITY_THRESHOLD)
228  {
229  eulerAngles.Z = Math.PI / 2; // 90 degrees
230  eulerAngles.Y = Math.Atan2(yawY, yawX);
231  eulerAngles.X = MathHelper.NormalizeRadians(eulerAngles.Y - 2d * Math.Atan2(q.Y, q.W));
232  }
233  else
234  {
235  eulerAngles.Z = Math.Asin(2d * singularityTest);
236  eulerAngles.X = Math.Atan2(yawY, yawX);
237  eulerAngles.Y = Math.Atan2(-2d * (q.W * q.Y + q.Z * q.X), 1d - 2d * (q.Y * q.Y + q.Z * q.Z));
238  }
239 
240  return eulerAngles;
241  }

578  public static void FromMatrix(in Matrix3d matrix, out Quaterniond result)
579  {
580  double trace = matrix.Trace;
581 
582  if (trace > 0)
583  {
584  double s = Math.Sqrt(trace + 1) * 2;
585  double invS = 1.0 / s;
586 
587  result.W = s * 0.25;
588  result.Xyz.X = (matrix.Row2.Y - matrix.Row1.Z) * invS;
589  result.Xyz.Y = (matrix.Row0.Z - matrix.Row2.X) * invS;
590  result.Xyz.Z = (matrix.Row1.X - matrix.Row0.Y) * invS;
591  }
592  else
593  {
594  double m00 = matrix.Row0.X, m11 = matrix.Row1.Y, m22 = matrix.Row2.Z;
595 
596  if (m00 > m11 && m00 > m22)
597  {
598  double s = Math.Sqrt(1 + m00 - m11 - m22) * 2;
599  double invS = 1.0 / s;
600 
601  result.W = (matrix.Row2.Y - matrix.Row1.Z) * invS;
602  result.Xyz.X = s * 0.25;
603  result.Xyz.Y = (matrix.Row0.Y + matrix.Row1.X) * invS;
604  result.Xyz.Z = (matrix.Row0.Z + matrix.Row2.X) * invS;
605  }
606  else if (m11 > m22)
607  {
608  double s = Math.Sqrt(1 + m11 - m00 - m22) * 2;
609  double invS = 1.0 / s;
610 
611  result.W = (matrix.Row0.Z - matrix.Row2.X) * invS;
612  result.Xyz.X = (matrix.Row0.Y + matrix.Row1.X) * invS;
613  result.Xyz.Y = s * 0.25;
614  result.Xyz.Z = (matrix.Row1.Z + matrix.Row2.Y) * invS;
615  }
616  else
617  {
618  double s = Math.Sqrt(1 + m22 - m00 - m11) * 2;
619  double invS = 1.0 / s;
620 
621  result.W = (matrix.Row1.X - matrix.Row0.Y) * invS;
622  result.Xyz.X = (matrix.Row0.Z + matrix.Row2.X) * invS;
623  result.Xyz.Y = (matrix.Row1.Z + matrix.Row2.Y) * invS;
624  result.Xyz.Z = s * 0.25;
625  }
626  }
627  }

636  [Pure]
637  public static Quaterniond Slerp(Quaterniond q1, Quaterniond q2, double blend)
638  {
639  // if either input is zero, return the other.
640  if (q1.LengthSquared == 0.0d)
641  {
642  if (q2.LengthSquared == 0.0d)
643  {
644  return Identity;
645  }
646 
647  return q2;
648  }
649 
650  if (q2.LengthSquared == 0.0d)
651  {
652  return q1;
653  }
654 
655  double cosHalfAngle = q1.W * q2.W + Vector3d.Dot(q1.Xyz, q2.Xyz);
656 
657  if (cosHalfAngle >= 1.0d || cosHalfAngle <= -1.0d)
658  {
659  // angle = 0.0d, so just return one input.
660  return q1;
661  }
662 
663  if (cosHalfAngle < 0.0d)
664  {
665  q2.Xyz = -q2.Xyz;
666  q2.W = -q2.W;
667  cosHalfAngle = -cosHalfAngle;
668  }
669 
670  double blendA;
671  double blendB;
672  if (cosHalfAngle < 0.99d)
673  {
674  // do proper slerp for big angles
675  double halfAngle = Math.Acos(cosHalfAngle);
676  double sinHalfAngle = Math.Sin(halfAngle);
677  double oneOverSinHalfAngle = 1.0d / sinHalfAngle;
678  blendA = Math.Sin(halfAngle * (1.0d - blend)) * oneOverSinHalfAngle;
679  blendB = Math.Sin(halfAngle * blend) * oneOverSinHalfAngle;
680  }
681  else
682  {
683  // do lerp if angle is really small.
684  blendA = 1.0d - blend;
685  blendB = blend;
686  }
687 
688  Quaterniond result = new Quaterniond(blendA * q1.Xyz + blendB * q2.Xyz, blendA * q1.W + blendB * q2.W);
689  if (result.LengthSquared > 0.0d)
690  {
691  return Normalize(result);
692  }
693 
694  return Identity;
695  }
