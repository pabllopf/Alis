98  private void AddPair(ref FixtureProxy proxyA, ref FixtureProxy proxyB)
99  {
100  Fixture fixtureA = proxyA.Fixture;
101  Fixture fixtureB = proxyB.Fixture;
102 
103  int indexA = proxyA.ChildIndex;
104  int indexB = proxyB.ChildIndex;
105 
106  Body bodyA = fixtureA.Body;
107  Body bodyB = fixtureB.Body;
108 
109  // Are the fixtures on the same body?
110  if (bodyA == bodyB)
111  {
112  return;
113  }
114 
115  // TODO_ERIN use a hash table to remove a potential bottleneck when both
116  // bodies have a lot of contacts.
117  // Does a contact already exist?
118  ContactEdge edge = bodyB.ContactList;
119  while (edge != null)
120  {
121  if (edge.Other == bodyA)
122  {
123  Fixture fA = edge.Contact.FixtureA;
124  Fixture fB = edge.Contact.FixtureB;
125  int iA = edge.Contact.ChildIndexA;
126  int iB = edge.Contact.ChildIndexB;
127 
128  if (fA == fixtureA && fB == fixtureB && iA == indexA && iB == indexB)
129  {
130  // A contact already exists.
131  return;
132  }
133 
134  if (fA == fixtureB && fB == fixtureA && iA == indexB && iB == indexA)
135  {
136  // A contact already exists.
137  return;
138  }
139  }
140 
141  edge = edge.Next;
142  }
143 
144  // Does a joint override collision? Is at least one body dynamic?
145  if (!bodyB.ShouldCollide(bodyA))
146  {
147  return;
148  }
149 
150  //Check default filter
151  if (!ShouldCollide(fixtureA, fixtureB))
152  {
153  return;
154  }
155 
156  // Check user filtering.
157  if (ContactFilter != null && !ContactFilter(fixtureA, fixtureB))
158  {
159  return;
160  }
161 
162  //Velcro: BeforeCollision delegate
163  if (fixtureA.BeforeCollision != null && !fixtureA.BeforeCollision(fixtureA, fixtureB))
164  {
165  return;
166  }
167 
168  if (fixtureB.BeforeCollision != null && !fixtureB.BeforeCollision(fixtureB, fixtureA))
169  {
170  return;
171  }
172 
173  // Call the factory.
174  Contact c = Contact.Create(fixtureA, indexA, fixtureB, indexB);
175  if (c == null)
176  {
177  return;
178  }
179 
180  // Contact creation may swap fixtures.
181  fixtureA = c.FixtureA;
182  fixtureB = c.FixtureB;
183  indexA = c.ChildIndexA;
184  indexB = c.ChildIndexB;
185  bodyA = fixtureA.Body;
186  bodyB = fixtureB.Body;
187 
188  // Insert into the world.
189  c.Previous = null;
190  c.Next = ContactList;
191  if (ContactList != null)
192  {
193  ContactList.Previous = c;
194  }
195 
196  ContactList = c;
197 
198  // Connect to island graph.
199 
200  // Connect to body A
201  c.NodeA.Contact = c;
202  c.NodeA.Other = bodyB;
203 
204  c.NodeA.Prev = null;
205  c.NodeA.Next = bodyA.ContactList;
206  if (bodyA.ContactList != null)
207  {
208  bodyA.ContactList.Prev = c.NodeA;
209  }
210 
211  bodyA.ContactList = c.NodeA;
212 
213  // Connect to body B
214  c.NodeB.Contact = c;
215  c.NodeB.Other = bodyA;
216 
217  c.NodeB.Prev = null;
218  c.NodeB.Next = bodyB.ContactList;
219  if (bodyB.ContactList != null)
220  {
221  bodyB.ContactList.Prev = c.NodeB;
222  }
223 
224  bodyB.ContactList = c.NodeB;
225  ++ContactCounter;
226  }

240  internal void Remove(Contact c)
241  {
242  if (c.FixtureA == null || c.FixtureB == null)
243  {
244  return;
245  }
246 
247  Fixture fixtureA = c.FixtureA;
248  Fixture fixtureB = c.FixtureB;
249 
250  //Velcro: When contacts are removed, we invoke OnSeparation
251  if (c.IsTouching)
252  {
253  //Report the separation to both participants:
254  fixtureA.OnSeparation?.Invoke(fixtureA, fixtureB, c);
255 
256  //Reverse the order of the reported fixtures. The first fixture is always the one that the user subscribed to.
257  fixtureB.OnSeparation?.Invoke(fixtureB, fixtureA, c);
258 
259  //The generic handler
260  EndContact?.Invoke(c);
261  }
262 
263  Body bodyA = fixtureA.Bodyprivate;
264  Body bodyB = fixtureB.Bodyprivate;
265 
266  // Remove from the world.
267  if (c.Previous != null)
268  {
269  c.Previous.Next = c.Next;
270  }
271 
272  if (c.Next != null)
273  {
274  c.Next.Previous = c.Previous;
275  }
276 
277  if (c == ContactList)
278  {
279  ContactList = c.Next;
280  }
281 
282  // Remove from body 1
283  if (c.NodeA.Prev != null)
284  {
285  c.NodeA.Prev.Next = c.NodeA.Next;
286  }
287 
288  if (c.NodeA.Next != null)
289  {
290  c.NodeA.Next.Prev = c.NodeA.Prev;
291  }
292 
293  if (c.NodeA == bodyA.ContactList)
294  {
295  bodyA.ContactList = c.NodeA.Next;
296  }
297 
298  // Remove from body 2
299  if (c.NodeB.Prev != null)
300  {
301  c.NodeB.Prev.Next = c.NodeB.Next;
302  }
303 
304  if (c.NodeB.Next != null)
305  {
306  c.NodeB.Next.Prev = c.NodeB.Prev;
307  }
308 
309  if (c.NodeB == bodyB.ContactList)
310  {
311  bodyB.ContactList = c.NodeB.Next;
312  }
313 
314  // Call the factory.
315  c.Destroy();
316  --ContactCounter;
317  }

323  internal void Collide()
324  {
325  // Update awake contacts.
326 
327  Contact c = ContactList;
328 
329  while (c != null)
330  {
331  Fixture fixtureA = c.FixtureA;
332  Fixture fixtureB = c.FixtureB;
333  int indexA = c.ChildIndexA;
334  int indexB = c.ChildIndexB;
335  Body bodyA = fixtureA.Bodyprivate;
336  Body bodyB = fixtureB.Bodyprivate;
337 
338  //Velcro: Do no try to collide disabled bodies
339  if (!bodyA.Enabled || !bodyB.Enabled)
340  {
341  c = c.Next;
342  continue;
343  }
344 
345  // Is this contact flagged for filtering?
346  if (c.FilterFlag)
347  {
348  // Should these bodies collide?
349  if (!bodyB.ShouldCollide(bodyA))
350  {
351  Contact cNuke = c;
352  c = cNuke.Next;
353  Remove(cNuke);
354  continue;
355  }
356 
357  // Check default filtering
358  if (!ShouldCollide(fixtureA, fixtureB))
359  {
360  Contact cNuke = c;
361  c = cNuke.Next;
362  Remove(cNuke);
363  continue;
364  }
365 
366  // Check user filtering.
367  if (ContactFilter != null && !ContactFilter(fixtureA, fixtureB))
368  {
369  Contact cNuke = c;
370  c = cNuke.Next;
371  Remove(cNuke);
372  continue;
373  }
374 
375  // Clear the filtering flag.
376  c.Flags &= ~ContactFlags.FilterFlag;
377  }
378 
379  bool activeA = bodyA.Awake && bodyA.BodyType != BodyType.Static;
380  bool activeB = bodyB.Awake && bodyB.BodyType != BodyType.Static;
381 
382  // At least one body must be awake and it must be dynamic or kinematic.
383  if (!activeA && !activeB)
384  {
385  c = c.Next;
386  continue;
387  }
388 
389  int proxyIdA = fixtureA.Proxies[indexA].ProxyId;
390  int proxyIdB = fixtureB.Proxies[indexB].ProxyId;
391  bool overlap = BroadPhase.TestOverlap(proxyIdA, proxyIdB);
392 
393  // Here we destroy contacts that cease to overlap in the broad-phase.
394  if (!overlap)
395  {
396  Contact cNuke = c;
397  c = cNuke.Next;
398  Remove(cNuke);
399  continue;
400  }
401 
402  // The contact persists.
403  c.Update(this);
404  c = c.Next;
405  }
406  }

414  private static bool ShouldCollide(Fixture fixtureA, Fixture fixtureB)
415  {
416  if (Settings.UseFpeCollisionCategories)
417  {
418  if (fixtureA.CollisionGroup == fixtureB.CollisionGroup &&
419  fixtureA.CollisionGroup != 0 && fixtureB.CollisionGroup != 0)
420  {
421  return false;
422  }
423 
424  if (((fixtureA.CollisionCategories & fixtureB.CollidesWith) ==
425  Category.None) &
426  ((fixtureB.CollisionCategories & fixtureA.CollidesWith) ==
427  Category.None))
428  {
429  return false;
430  }
431 
432  return true;
433  }
434 
435  if (fixtureA.CollisionGroup == fixtureB.CollisionGroup &&
436  fixtureA.CollisionGroup != 0)
437  {
438  return fixtureA.CollisionGroup > 0;
439  }
440 
441  bool collide = (fixtureA.CollidesWith & fixtureB.CollisionCategories) != 0 &&
442  (fixtureA.CollisionCategories & fixtureB.CollidesWith) != 0;
443 
444  return collide;
445  }
446  }
