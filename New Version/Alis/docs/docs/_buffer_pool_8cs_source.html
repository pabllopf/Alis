289  private void EnlargeBufferIfRequired(int count)
290  {
291  // we cannot fit the data into the existing buffer, time for a new buffer
292  if (count > _buffer.Length - _ms.Position)
293  {
294  int position = (int) _ms.Position;
295 
296  // double the buffer size
297  long newSize = (long) _buffer.Length * 2;
298 
299  // make sure the new size is big enough
300  long requiredSize = (long) count + _buffer.Length - position;
301 
302  if (requiredSize > int.MaxValue)
303  {
304  throw new WebSocketBufferOverflowException(
305  $"Tried to create a buffer ({requiredSize:#,##0} bytes) that was larger than the max allowed size ({int.MaxValue:#,##0})");
306  }
307 
308  if (requiredSize > newSize)
309  {
310  // compute the power of two larger than requiredSize. so 40000 => 65536
311  long candidateSize = (long) Math.Pow(2, Math.Ceiling(Math.Log(requiredSize) / Math.Log(2)));
312  if (candidateSize > int.MaxValue)
313  {
314  newSize = requiredSize;
315  }
316  else
317  {
318  newSize = candidateSize;
319  }
320  }
321 
322  byte[] newBuffer = new byte[newSize];
323  Buffer.BlockCopy(_buffer, 0, newBuffer, 0, position);
324  _ms = new MemoryStream(newBuffer, 0, newBuffer.Length, true, true)
325  {
326  Position = position
327  };
328 
329  _buffer = newBuffer;
330  }
331  }
