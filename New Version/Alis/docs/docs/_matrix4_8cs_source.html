602  [Pure]
603  public Quaternion ExtractRotation(bool rowNormalize = true)
604  {
605  Vector3 row0 = Row0.Xyz;
606  Vector3 row1 = Row1.Xyz;
607  Vector3 row2 = Row2.Xyz;
608 
609  if (rowNormalize)
610  {
611  row0 = row0.Normalized();
612  row1 = row1.Normalized();
613  row2 = row2.Normalized();
614  }
615 
616  // code below adapted from Blender
617  Quaternion q = default(Quaternion);
618  double trace = 0.25 * (row0[0] + row1[1] + row2[2] + 1.0);
619 
620  if (trace > 0)
621  {
622  double sq = Math.Sqrt(trace);
623 
624  q.W = (float) sq;
625  sq = 1.0 / (4.0 * sq);
626  q.X = (float) ((row1[2] - row2[1]) * sq);
627  q.Y = (float) ((row2[0] - row0[2]) * sq);
628  q.Z = (float) ((row0[1] - row1[0]) * sq);
629  }
630  else if (row0[0] > row1[1] && row0[0] > row2[2])
631  {
632  double sq = 2.0 * Math.Sqrt(1.0 + row0[0] - row1[1] - row2[2]);
633 
634  q.X = (float) (0.25 * sq);
635  sq = 1.0 / sq;
636  q.W = (float) ((row2[1] - row1[2]) * sq);
637  q.Y = (float) ((row1[0] + row0[1]) * sq);
638  q.Z = (float) ((row2[0] + row0[2]) * sq);
639  }
640  else if (row1[1] > row2[2])
641  {
642  double sq = 2.0 * Math.Sqrt(1.0 + row1[1] - row0[0] - row2[2]);
643 
644  q.Y = (float) (0.25 * sq);
645  sq = 1.0 / sq;
646  q.W = (float) ((row2[0] - row0[2]) * sq);
647  q.X = (float) ((row1[0] + row0[1]) * sq);
648  q.Z = (float) ((row2[1] + row1[2]) * sq);
649  }
650  else
651  {
652  double sq = 2.0 * Math.Sqrt(1.0 + row2[2] - row0[0] - row1[1]);
653 
654  q.Z = (float) (0.25 * sq);
655  sq = 1.0 / sq;
656  q.W = (float) ((row1[0] - row0[1]) * sq);
657  q.X = (float) ((row2[0] + row0[2]) * sq);
658  q.Y = (float) ((row2[1] + row1[2]) * sq);
659  }
660 
661  q.Normalize();
662  return q;
663  }

677  public static void CreateFromAxisAngle(Vector3 axis, float angle, out Matrix4 result)
678  {
679  // normalize and create a local copy of the vector.
680  axis.Normalize();
681  float axisX = axis.X, axisY = axis.Y, axisZ = axis.Z;
682 
683  // calculate angles
684  float cos = MathF.Cos(-angle);
685  float sin = MathF.Sin(-angle);
686  float t = 1.0f - cos;
687 
688  // do the conversion math once
689  float tXX = t * axisX * axisX;
690  float tXY = t * axisX * axisY;
691  float tXZ = t * axisX * axisZ;
692  float tYY = t * axisY * axisY;
693  float tYZ = t * axisY * axisZ;
694  float tZZ = t * axisZ * axisZ;
695 
696  float sinX = sin * axisX;
697  float sinY = sin * axisY;
698  float sinZ = sin * axisZ;
699 
700  result.Row0.X = tXX + cos;
701  result.Row0.Y = tXY - sinZ;
702  result.Row0.Z = tXZ + sinY;
703  result.Row0.W = 0;
704  result.Row1.X = tXY + sinZ;
705  result.Row1.Y = tYY + cos;
706  result.Row1.Z = tYZ - sinX;
707  result.Row1.W = 0;
708  result.Row2.X = tXZ - sinY;
709  result.Row2.Y = tYZ + sinX;
710  result.Row2.Z = tZZ + cos;
711  result.Row2.W = 0;
712  result.Row3 = Vector4.UnitW;
713  }

733  public static void CreateFromQuaternion(in Quaternion q, out Matrix4 result)
734  {
735  // Adapted from https://en.wikipedia.org/wiki/Quaternions_and_spatial_rotation#Quaternion-derived_rotation_matrix
736  // with the caviat that opentk uses row-major matrices so the matrix we create is transposed
737  float sqx = q.X * q.X;
738  float sqy = q.Y * q.Y;
739  float sqz = q.Z * q.Z;
740  float sqw = q.W * q.W;
741 
742  float xy = q.X * q.Y;
743  float xz = q.X * q.Z;
744  float xw = q.X * q.W;
745 
746  float yz = q.Y * q.Z;
747  float yw = q.Y * q.W;
748 
749  float zw = q.Z * q.W;
750 
751  float s2 = 2f / (sqx + sqy + sqz + sqw);
752 
753  result.Row0.X = 1f - s2 * (sqy + sqz);
754  result.Row1.Y = 1f - s2 * (sqx + sqz);
755  result.Row2.Z = 1f - s2 * (sqx + sqy);
756 
757  result.Row0.Y = s2 * (xy + zw);
758  result.Row1.X = s2 * (xy - zw);
759 
760  result.Row2.X = s2 * (xz + yw);
761  result.Row0.Z = s2 * (xz - yw);
762 
763  result.Row2.Y = s2 * (yz - xw);
764  result.Row1.Z = s2 * (yz + xw);
765 
766  result.Row0.W = 0;
767  result.Row1.W = 0;
768  result.Row2.W = 0;
769  result.Row3 = new Vector4(0, 0, 0, 1);
770  }
