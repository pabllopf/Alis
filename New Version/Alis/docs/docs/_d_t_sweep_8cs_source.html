68  public static void Triangulate(DtSweepContext tcx)
69  {
70  tcx.CreateAdvancingFront();
71 
72  Sweep(tcx);
73 
74  // Finalize triangulation
75  if (tcx.TriangulationMode == TriangulationMode.Polygon)
76  {
77  FinalizationPolygon(tcx);
78  }
79  else
80  {
81  FinalizationConvexHull(tcx);
82  }
83 
84  tcx.Done();
85  }

88  private static void Sweep(DtSweepContext tcx)
89  {
90  List<TriangulationPoint> points = tcx.Points;
91 
92  for (int i = 1; i < points.Count; i++)
93  {
94  TriangulationPoint point = points[i];
95 
96  AdvancingFrontNode node = PointEvent(tcx, point);
97 
98  if (point.HasEdges)
99  {
100  foreach (DtSweepConstraint e in point.Edges)
101  {
102  EdgeEvent(tcx, e, node);
103  }
104  }
105 
106  tcx.Update(null);
107  }
108  }

111  private static void FinalizationConvexHull(DtSweepContext tcx)
112  {
113  DelaunayTriangle t1, t2;
114 
115  AdvancingFrontNode n1 = tcx.AFront.Head.Next;
116  AdvancingFrontNode n2 = n1.Next;
117 
118  TurnAdvancingFrontConvex(tcx, n1, n2);
119 
120  // TODO: implement ConvexHull for lower right and left boundary
121 
122  // Lets remove triangles connected to the two "algorithm" points
123 
124  // XXX: When the first the nodes are points in a triangle we need to do a flip before
125  // removing triangles or we will lose a valid triangle.
126  // Same for last three nodes!
127  // !!! If I implement ConvexHull for lower right and left boundary this fix should not be
128  // needed and the removed triangles will be added again by default
129  n1 = tcx.AFront.Tail.Prev;
130  if (n1.Triangle.Contains(n1.Next.Point) && n1.Triangle.Contains(n1.Prev.Point))
131  {
132  t1 = n1.Triangle.NeighborAcross(n1.Point);
133  RotateTrianglePair(n1.Triangle, n1.Point, t1, t1.OppositePoint(n1.Triangle, n1.Point));
134  tcx.MapTriangleToNodes(n1.Triangle);
135  tcx.MapTriangleToNodes(t1);
136  }
137 
138  n1 = tcx.AFront.Head.Next;
139  if (n1.Triangle.Contains(n1.Prev.Point) && n1.Triangle.Contains(n1.Next.Point))
140  {
141  t1 = n1.Triangle.NeighborAcross(n1.Point);
142  RotateTrianglePair(n1.Triangle, n1.Point, t1, t1.OppositePoint(n1.Triangle, n1.Point));
143  tcx.MapTriangleToNodes(n1.Triangle);
144  tcx.MapTriangleToNodes(t1);
145  }
146 
147  // Lower right boundary
148  TriangulationPoint first = tcx.AFront.Head.Point;
149  n2 = tcx.AFront.Tail.Prev;
150  t1 = n2.Triangle;
151  TriangulationPoint p1 = n2.Point;
152  n2.Triangle = null;
153  do
154  {
155  tcx.RemoveFromList(t1);
156  p1 = t1.PointCcw(p1);
157  if (p1 == first)
158  {
159  break;
160  }
161 
162  t2 = t1.NeighborCcw(p1);
163  t1.Clear();
164  t1 = t2;
165  } while (true);
166 
167  // Lower left boundary
168  first = tcx.AFront.Head.Next.Point;
169  p1 = t1.PointCw(tcx.AFront.Head.Point);
170  t2 = t1.NeighborCw(tcx.AFront.Head.Point);
171  t1.Clear();
172  t1 = t2;
173  while (p1 != first) //TODO: Port note. This was do while before.
174  {
175  tcx.RemoveFromList(t1);
176  p1 = t1.PointCcw(p1);
177  t2 = t1.NeighborCcw(p1);
178  t1.Clear();
179  t1 = t2;
180  }
181 
182  // Remove current head and tail node now that we have removed all triangles attached
183  // to them. Then set new head and tail node points
184  tcx.AFront.Head = tcx.AFront.Head.Next;
185  tcx.AFront.Head.Prev = null;
186  tcx.AFront.Tail = tcx.AFront.Tail.Prev;
187  tcx.AFront.Tail.Next = null;
188 
189  tcx.FinalizeTriangulation();
190  }

193  private static void TurnAdvancingFrontConvex(DtSweepContext tcx, AdvancingFrontNode b, AdvancingFrontNode c)
194  {
195  AdvancingFrontNode first = b;
196  while (c != tcx.AFront.Tail)
197  {
198  if (TriangulationUtil.Orient2d(b.Point, c.Point, c.Next.Point) == Orientation.Ccw)
199  {
200  // [b,c,d] Concave - fill around c
201  Fill(tcx, c);
202  c = c.Next;
203  }
204  else
205  {
206  // [b,c,d] Convex
207  if (b != first && TriangulationUtil.Orient2d(b.Prev.Point, b.Point, c.Point) == Orientation.Ccw)
208  {
209  // [a,b,c] Concave - fill around b
210  Fill(tcx, b);
211  b = b.Prev;
212  }
213  else
214  {
215  // [a,b,c] Convex - nothing to fill
216  b = c;
217  c = c.Next;
218  }
219  }
220  }

245  private static AdvancingFrontNode PointEvent(DtSweepContext tcx, TriangulationPoint point)
246  {
247  AdvancingFrontNode node = tcx.LocateNode(point);
248  AdvancingFrontNode newNode = NewFrontTriangle(tcx, point, node);
249 
250  // Only need to check +epsilon since point never have smaller
251  // x value than node due to how we fetch nodes from the front
252  if (point.X <= node.Point.X + TriangulationUtil.Epsilon)
253  {
254  Fill(tcx, node);
255  }
256 
257  tcx.AddNode(newNode);
258 
259  FillAdvancingFront(tcx, newNode);
260  return newNode;

296  {
297  try
298  {
299  tcx.EdgeEvent.ConstrainedEdge = edge;
300  tcx.EdgeEvent.Right = edge.P.X > edge.Q.X;
301 
302  if (IsEdgeSideOfTriangle(node.Triangle, edge.P, edge.Q))

313  }
314  catch (PointOnEdgeException e)
315  {
316  Debug.WriteLine("Skipping Edge: {0}", e.Message);

578  private static void EdgeEvent(DtSweepContext tcx, TriangulationPoint ep, TriangulationPoint eq,
579  DelaunayTriangle triangle, TriangulationPoint point)
580  {
581  if (IsEdgeSideOfTriangle(triangle, ep, eq))
582  {
583  return;
584  }
585 
586  TriangulationPoint p1 = triangle.PointCcw(point);
587  Orientation o1 = TriangulationUtil.Orient2d(eq, p1, ep);
588  if (o1 == Orientation.Collinear)
589  {
590  if (triangle.Contains(eq, p1))
591  {
592  triangle.MarkConstrainedEdge(eq, p1);
593 
594  // We are modifying the constraint maybe it would be better to
595  // not change the given constraint and just keep a variable for the new constraint
596  tcx.EdgeEvent.ConstrainedEdge.Q = p1;
597  triangle = triangle.NeighborAcross(point);
598  EdgeEvent(tcx, ep, p1, triangle, p1);
599  }
600  else
601  {
602  throw new PointOnEdgeException("EdgeEvent - Point on constrained edge not supported yet");
603  }
604 
605  if (tcx.IsDebugEnabled)
606  {
607  Debug.WriteLine("EdgeEvent - Point on constrained edge");
608  }
609 
610  return;
611  }
612 
613  TriangulationPoint p2 = triangle.PointCw(point);
614  Orientation o2 = TriangulationUtil.Orient2d(eq, p2, ep);
615  if (o2 == Orientation.Collinear)
616  {
617  if (triangle.Contains(eq, p2))
618  {
619  triangle.MarkConstrainedEdge(eq, p2);
620 
621  // We are modifying the constraint maybe it would be better to
622  // not change the given constraint and just keep a variable for the new constraint
623  tcx.EdgeEvent.ConstrainedEdge.Q = p2;
624  triangle = triangle.NeighborAcross(point);
625  EdgeEvent(tcx, ep, p2, triangle, p2);
626  }
627  else
628  {
629  throw new PointOnEdgeException("EdgeEvent - Point on constrained edge not supported yet");
630  }
631 
632  if (tcx.IsDebugEnabled)
633  {
634  Debug.WriteLine("EdgeEvent - Point on constrained edge");
635  }
636 
637  return;
638  }
639 
640  if (o1 == o2)
641  {
642  // Need to decide if we are rotating CW or CCW to get to a triangle
643  // that will cross edge
644  if (o1 == Orientation.Cw)
645  {
646  triangle = triangle.NeighborCcw(point);
647  }
648  else
649  {
650  triangle = triangle.NeighborCw(point);
651  }
652 
653  EdgeEvent(tcx, ep, eq, triangle, point);
654  }
655  else
656  {
657  // This triangle crosses constraint so lets flippin start!
658  FlipEdgeEvent(tcx, ep, eq, triangle, point);
659  }
660  }

713  }
714  else
715  {

721  // XXX: I think one of the triangles should be legalized here?
722  }
723  }

727  {
728  Console.WriteLine("[FLIP] - flipping and continuing with triangle still crossing edge");

736  }

972  private static double Angle(TriangulationPoint origin, TriangulationPoint pa, TriangulationPoint pb)
973  {
974  /* Complex plane
975  * ab = cosA +i*sinA
976  * ab = (ax + ay*i)(bx + by*i) = (ax*bx + ay*by) + i(ax*by-ay*bx)
977  * atan2(y,x) computes the principal value of the argument function
978  * applied to the complex number x+iy
979  * Where x = ax*bx + ay*by
980  * y = ax*by - ay*bx
981  */
982  double px = origin.X;
983  double py = origin.Y;
984  double ax = pa.X - px;
985  double ay = pa.Y - py;
986  double bx = pb.X - px;
987  double by = pb.Y - py;
988  double x = ax * by - ay * bx;
989  double y = ax * bx + ay * by;
990  double angle = Math.Atan2(x, y);
991  return angle;

1180  {

1182  // violate the Delaunay condition

1222
