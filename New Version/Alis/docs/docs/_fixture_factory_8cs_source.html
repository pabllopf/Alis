222  public static List<Fixture> AttachCompoundPolygon(List<Vertices> list, float density, Body body)
223  {
224  List<Fixture> res = new List<Fixture>(list.Count);
225 
226  //Then we create several fixtures using the body
227  foreach (Vertices vertices in list)
228  {
229  if (vertices.Count == 2)
230  {
231  EdgeShape shape = new EdgeShape(vertices[0], vertices[1]);
232  res.Add(body.AddFixture(shape));
233  }
234  else
235  {
236  PolygonShape shape = new PolygonShape(vertices, density);
237  res.Add(body.AddFixture(shape));
238  }
239  }
240 
241  return res;
242  }

269  public static List<Fixture> AttachSolidArc(float density, float radians, int sides, float radius, Body body)
270  {
271  Vertices arc = PolygonUtils.CreateArc(radians, sides, radius);
272  arc.Rotate((MathConstants.Pi - radians) / 2);
273 
274  //Close the arc
275  arc.Add(arc[0]);
276 
277  List<Vertices> triangles = Triangulate.ConvexPartition(arc, TriangulationAlgorithm.Earclip);
278 
279  return AttachCompoundPolygon(triangles, density, body);
