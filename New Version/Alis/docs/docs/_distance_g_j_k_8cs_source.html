71  public static void ComputeDistance(ref DistanceInput input, out DistanceOutput output, out SimplexCache cache)
72  {
73  cache = new SimplexCache();
74 
75  ++GjkCalls;
76 
77  // Initialize the simplex.
78  Simplex simplex = new Simplex();
79  simplex.ReadCache(ref cache, ref input.ProxyA, ref input.TransformA, ref input.ProxyB,
80  ref input.TransformB);
81 
82  // These store the vertices of the last simplex so that we
83  // can check for duplicates and prevent cycling.
84  FixedArray3<int> saveA = new FixedArray3<int>();
85  FixedArray3<int> saveB = new FixedArray3<int>();
86 
87  // Main iteration loop.
88  int iter = 0;
89 
90  //Velcro: Moved the max iterations to settings
91  while (iter < Settings.MaxGjkIterations)
92  {
93  // Copy simplex so we can identify duplicates.
94  int saveCount = simplex.Count;
95  for (int i = 0; i < saveCount; ++i)
96  {
97  saveA[i] = simplex.V[i].IndexA;
98  saveB[i] = simplex.V[i].IndexB;
99  }
100 
101  switch (simplex.Count)
102  {
103  case 1:
104  break;
105  case 2:
106  simplex.Solve2();
107  break;
108  case 3:
109  simplex.Solve3();
110  break;
111  default:
112  Debug.Assert(false);
113  break;
114  }
115 
116  // If we have 3 points, then the origin is in the corresponding triangle.
117  if (simplex.Count == 3)
118  {
119  break;
120  }
121 
122  // Get search direction.
123  Vector2 d = simplex.GetSearchDirection();
124 
125  // Ensure the search direction is numerically fit.
126  if (d.LengthSquared() < MathConstants.Epsilon * MathConstants.Epsilon)
127  {
128  // The origin is probably contained by a line segment
129  // or triangle. Thus the shapes are overlapped.
130 
131  // We can't return zero here even though there may be overlap.
132  // In case the simplex is a point, segment, or triangle it is difficult
133  // to determine if the origin is contained in the CSO or very close to it.
134  break;
135  }
136 
137  // Compute a tentative new simplex vertex using support points.
138  SimplexVertex vertex = simplex.V[simplex.Count];
139  vertex.IndexA = input.ProxyA.GetSupport(MathUtils.MulT(input.TransformA.Q, -d));
140  vertex.Wa = MathUtils.Mul(ref input.TransformA, input.ProxyA.Vertices[vertex.IndexA]);
141 
142  vertex.IndexB = input.ProxyB.GetSupport(MathUtils.MulT(input.TransformB.Q, d));
143  vertex.Wb = MathUtils.Mul(ref input.TransformB, input.ProxyB.Vertices[vertex.IndexB]);
144  vertex.W = vertex.Wb - vertex.Wa;
145  simplex.V[simplex.Count] = vertex;
146 
147  // Iteration count is equated to the number of support point calls.
148  ++iter;
149 
150  ++GjkIters;
151 
152  // Check for duplicate support points. This is the main termination criteria.
153  bool duplicate = false;
154  for (int i = 0; i < saveCount; ++i)
155  {
156  if (vertex.IndexA == saveA[i] && vertex.IndexB == saveB[i])
157  {
158  duplicate = true;
159  break;
160  }
161  }
162 
163  // If we found a duplicate support point we must exit to avoid cycling.
164  if (duplicate)
165  {
166  break;
167  }
168 
169  // New vertex is OK and needed.
170  ++simplex.Count;
171  }
172 
173  GjkMaxIters = Math.Max(GjkMaxIters, iter);
174 
175  // Prepare output.
176  simplex.GetWitnessPoints(out output.PointA, out output.PointB);
177  output.Distance = (output.PointA - output.PointB).Length();
178  output.Iterations = iter;
179 
180  // Cache the simplex.
181  simplex.WriteCache(ref cache);
182 
183  // Apply radii if requested.
184  if (input.UseRadii)
185  {
186  float rA = input.ProxyA.Radius;
187  float rB = input.ProxyB.Radius;
188 
189  if (output.Distance > rA + rB && output.Distance > MathConstants.Epsilon)
190  {
191  // Shapes are still no overlapped.
192  // Move the witness points to the outer surface.
193  output.Distance -= rA + rB;
194  Vector2 normal = output.PointB - output.PointA;
195  normal = Vector2.Normalize(normal);
196  output.PointA += rA * normal;
197  output.PointB -= rB * normal;
198  }
199  else
200  {
201  // Shapes are overlapped when radii are considered.
202  // Move the witness points to the middle.
203  Vector2 p = 0.5f * (output.PointA + output.PointB);
204  output.PointA = p;
205  output.PointB = p;
206  output.Distance = 0.0f;
207  }
208  }
209  }

220  public static bool ShapeCast(ref ShapeCastInput input, out ShapeCastOutput output)
221  {
222  output = new ShapeCastOutput
223  {
224  Iterations = 0,
225  Lambda = 1.0f,
226  Normal = Vector2.Zero,
227  Point = Vector2.Zero
228  };
229 
230  DistanceProxy proxyA = input.ProxyA;
231  DistanceProxy proxyB = input.ProxyB;
232 
233  float radiusA = MathUtils.Max(proxyA.Radius, Settings.PolygonRadius);
234  float radiusB = MathUtils.Max(proxyB.Radius, Settings.PolygonRadius);
235  float radius = radiusA + radiusB;
236 
237  Transform xfA = input.TransformA;
238  Transform xfB = input.TransformB;
239 
240  Vector2 r = input.TranslationB;
241  Vector2 n = new Vector2(0.0f, 0.0f);
242  float lambda = 0.0f;
243 
244  // Initial simplex
245  Simplex simplex = new Simplex
246  {
247  Count = 0
248  };
249 
250  // Get simplex vertices as an array.
251  //SimplexVertex vertices = simplex.V.Value0; //Velcro: we don't need this as we have an indexer instead
252 
253  // Get support point in -r direction
254  int indexA = proxyA.GetSupport(MathUtils.MulT(xfA.Q, -r));
255  Vector2 wA = MathUtils.Mul(ref xfA, proxyA.GetVertex(indexA));
256  int indexB = proxyB.GetSupport(MathUtils.MulT(xfB.Q, r));
257  Vector2 wB = MathUtils.Mul(ref xfB, proxyB.GetVertex(indexB));
258  Vector2 v = wA - wB;
259 
260  // Sigma is the target distance between polygons
261  float sigma = MathUtils.Max(Settings.PolygonRadius, radius - Settings.PolygonRadius);
262  float tolerance = 0.5f * Settings.LinearSlop;
263 
264  // Main iteration loop.
265  int iter = 0;
266 
267  //Velcro: We have moved the max iterations into settings
268  while (iter < Settings.MaxGjkIterations && v.Length() - sigma > tolerance)
269  {
270  Debug.Assert(simplex.Count < 3);
271 
272  output.Iterations += 1;
273 
274  // Support in direction -v (A - B)
275  indexA = proxyA.GetSupport(MathUtils.MulT(xfA.Q, -v));
276  wA = MathUtils.Mul(ref xfA, proxyA.GetVertex(indexA));
277  indexB = proxyB.GetSupport(MathUtils.MulT(xfB.Q, v));
278  wB = MathUtils.Mul(ref xfB, proxyB.GetVertex(indexB));
279  Vector2 p = wA - wB;
280 
281  // -v is a normal at p
282  v = Vector2.Normalize(v);
283 
284  // Intersect ray with plane
285  float vp = MathUtils.Dot(ref v, ref p);
286  float vr = MathUtils.Dot(ref v, ref r);
287  if (vp - sigma > lambda * vr)
288  {
289  if (vr <= 0.0f)
290  {
291  return false;
292  }
293 
294  lambda = (vp - sigma) / vr;
295  if (lambda > 1.0f)
296  {
297  return false;
298  }
299 
300  n = -v;
301  simplex.Count = 0;
302  }
303 
304  // Reverse simplex since it works with B - A.
305  // Shift by lambda * r because we want the closest point to the current clip point.
306  // Note that the support point p is not shifted because we want the plane equation
307  // to be formed in unshifted space.
308  SimplexVertex vertex = simplex.V[simplex.Count];
309  vertex.IndexA = indexB;
310  vertex.Wa = wB + lambda * r;
311  vertex.IndexB = indexA;
312  vertex.Wb = wA;
313  vertex.W = vertex.Wb - vertex.Wa;
314  vertex.A = 1.0f;
315  simplex.V[simplex.Count] = vertex; //Velcro: we have to copy the value back
316  simplex.Count += 1;
317 
318  switch (simplex.Count)
319  {
320  case 1:
321  break;
322 
323  case 2:
324  simplex.Solve2();
325  break;
326 
327  case 3:
328  simplex.Solve3();
329  break;
330 
331  default:
332  Debug.Assert(false);
333  break;
334  }
335 
336  // If we have 3 points, then the origin is in the corresponding triangle.
337  if (simplex.Count == 3)
338  {
339  // Overlap
340  return false;
341  }
342 
343  // Get search direction.
344  v = simplex.GetClosestPoint();
345 
346  // Iteration count is equated to the number of support point calls.
347  ++iter;
348  }
349 
350  if (iter == 0)
351  {
352  // Initial overlap
353  return false;
354  }
355 
356  // Prepare output.
357  simplex.GetWitnessPoints(out _, out Vector2 pointB);
358 
359  if (v.LengthSquared() > 0.0f)
360  {
361  n = -v;
362  n = Vector2.Normalize(n);
363  }
364 
365  output.Point = pointB + radiusA * n;
366  output.Normal = n;
367  output.Lambda = lambda;
368  output.Iterations = iter;
369  return true;
370  }
371  }
