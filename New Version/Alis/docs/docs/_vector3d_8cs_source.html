979  public static void Transform(in Vector3d vec, in Quaterniond quat, out Vector3d result)
980  {
981  // Since vec.W == 0, we can optimize quat * vec * quat^-1 as follows:
982  // vec + 2.0 * cross(quat.xyz, cross(quat.xyz, vec) + quat.w * vec)
983  Vector3d xyz = quat.Xyz;
984  Cross(in xyz, in vec, out Vector3d temp);
985  Multiply(in vec, quat.W, out Vector3d temp2);
986  Add(in temp, in temp2, out temp);
987  Cross(in xyz, in temp, out temp2);
988  Multiply(in temp2, 2f, out temp2);
989  Add(in vec, in temp2, out result);
990  }
