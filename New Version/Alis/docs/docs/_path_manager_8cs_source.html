74  public static void ConvertPathToPolygon(Path path, Body body, float density, int subdivisions)
75  {
76  if (!path.Closed)
77  {
78  throw new Exception("The path must be closed to convert to a polygon.");
79  }
80 
81  List<Vector2> verts = path.GetVertices(subdivisions);
82 
83  List<Vertices> decomposedVerts =
84  Triangulate.ConvexPartition(new Vertices(verts), TriangulationAlgorithm.Bayazit);
85 
86  foreach (Vertices item in decomposedVerts)
87  {
88  body.AddFixture(new PolygonShape(item, density));
89  }
90  }

163  public static List<RevoluteJoint> AttachBodiesWithRevoluteJoint(World world, List<Body> bodies,
164  Vector2 localAnchorA, Vector2 localAnchorB, bool connectFirstAndLast, bool collideConnected)
165  {
166  List<RevoluteJoint> joints = new List<RevoluteJoint>(bodies.Count + 1);
167 
168  for (int i = 1; i < bodies.Count; i++)
169  {
170  RevoluteJoint joint = new RevoluteJoint(bodies[i], bodies[i - 1], localAnchorA, localAnchorB)
171  {
172  CollideConnected = collideConnected
173  };
174  world.AddJoint(joint);
175  joints.Add(joint);
176  }
177 
178  if (connectFirstAndLast)
179  {
180  RevoluteJoint lastjoint =
181  new RevoluteJoint(bodies[0], bodies[bodies.Count - 1], localAnchorA, localAnchorB)
182  {
183  CollideConnected = collideConnected
184  };
185  world.AddJoint(lastjoint);
186  joints.Add(lastjoint);
187  }
188 
189  return joints;
190  }
