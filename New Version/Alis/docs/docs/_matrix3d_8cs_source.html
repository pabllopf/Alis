395  [Pure]
396  public Quaterniond ExtractRotation(bool rowNormalize = true)
397  {
398  Vector3d row0 = Row0;
399  Vector3d row1 = Row1;
400  Vector3d row2 = Row2;
401 
402  if (rowNormalize)
403  {
404  row0 = row0.Normalized();
405  row1 = row1.Normalized();
406  row2 = row2.Normalized();
407  }
408 
409  // code below adapted from Blender
410  Quaterniond q = default(Quaterniond);
411  double trace = 0.25 * (row0[0] + row1[1] + row2[2] + 1.0);
412 
413  if (trace > 0)
414  {
415  double sq = Math.Sqrt(trace);
416 
417  q.W = sq;
418  sq = 1.0 / (4.0 * sq);
419  q.X = (row1[2] - row2[1]) * sq;
420  q.Y = (row2[0] - row0[2]) * sq;
421  q.Z = (row0[1] - row1[0]) * sq;
422  }
423  else if (row0[0] > row1[1] && row0[0] > row2[2])
424  {
425  double sq = 2.0 * Math.Sqrt(1.0 + row0[0] - row1[1] - row2[2]);
426 
427  q.X = 0.25 * sq;
428  sq = 1.0 / sq;
429  q.W = (row2[1] - row1[2]) * sq;
430  q.Y = (row1[0] + row0[1]) * sq;
431  q.Z = (row2[0] + row0[2]) * sq;
432  }
433  else if (row1[1] > row2[2])
434  {
435  double sq = 2.0 * Math.Sqrt(1.0 + row1[1] - row0[0] - row2[2]);
436 
437  q.Y = 0.25 * sq;
438  sq = 1.0 / sq;
439  q.W = (row2[0] - row0[2]) * sq;
440  q.X = (row1[0] + row0[1]) * sq;
441  q.Z = (row2[1] + row1[2]) * sq;
442  }
443  else
444  {
445  double sq = 2.0 * Math.Sqrt(1.0 + row2[2] - row0[0] - row1[1]);
446 
447  q.Z = 0.25 * sq;
448  sq = 1.0 / sq;
449  q.W = (row1[0] - row0[1]) * sq;
450  q.X = (row2[0] + row0[2]) * sq;
451  q.Y = (row2[1] + row1[2]) * sq;
452  }
453 
454  q.Normalize();
455  return q;
456  }

464  public static void CreateFromAxisAngle(Vector3d axis, double angle, out Matrix3d result)
465  {
466  // normalize and create a local copy of the vector.
467  axis.Normalize();
468  double axisX = axis.X, axisY = axis.Y, axisZ = axis.Z;
469 
470  // calculate angles
471  double cos = Math.Cos(-angle);
472  double sin = Math.Sin(-angle);
473  double t = 1.0f - cos;
474 
475  // do the conversion math once
476  double tXX = t * axisX * axisX;
477  double tXY = t * axisX * axisY;
478  double tXZ = t * axisX * axisZ;
479  double tYY = t * axisY * axisY;
480  double tYZ = t * axisY * axisZ;
481  double tZZ = t * axisZ * axisZ;
482 
483  double sinX = sin * axisX;
484  double sinY = sin * axisY;
485  double sinZ = sin * axisZ;
486 
487  result.Row0.X = tXX + cos;
488  result.Row0.Y = tXY - sinZ;
489  result.Row0.Z = tXZ + sinY;
490  result.Row1.X = tXY + sinZ;
491  result.Row1.Y = tYY + cos;
492  result.Row1.Z = tYZ - sinX;
493  result.Row2.X = tXZ - sinY;
494  result.Row2.Y = tYZ + sinX;
495  result.Row2.Z = tZZ + cos;
496  }

516  public static void CreateFromQuaternion(in Quaterniond q, out Matrix3d result)
517  {
518  // Adapted from https://en.wikipedia.org/wiki/Quaternions_and_spatial_rotation#Quaternion-derived_rotation_matrix
519  // with the caviat that opentk uses row-major matrices so the matrix we create is transposed
520  double sqx = q.X * q.X;
521  double sqy = q.Y * q.Y;
522  double sqz = q.Z * q.Z;
523  double sqw = q.W * q.W;
524 
525  double xy = q.X * q.Y;
526  double xz = q.X * q.Z;
527  double xw = q.X * q.W;
528 
529  double yz = q.Y * q.Z;
530  double yw = q.Y * q.W;
531 
532  double zw = q.Z * q.W;
533 
534  double s2 = 2d / (sqx + sqy + sqz + sqw);
535 
536  result.Row0.X = 1d - s2 * (sqy + sqz);
537  result.Row1.Y = 1d - s2 * (sqx + sqz);
538  result.Row2.Z = 1d - s2 * (sqx + sqy);
539 
540  result.Row0.Y = s2 * (xy + zw);
541  result.Row1.X = s2 * (xy - zw);
542 
543  result.Row2.X = s2 * (xz + yw);
544  result.Row0.Z = s2 * (xz - yw);
545 
546  result.Row2.Y = s2 * (yz - xw);
547  result.Row1.Z = s2 * (yz + xw);
548  }

775  double leftM11 = left.Row0.X;
776  double leftM12 = left.Row0.Y;
777  double leftM13 = left.Row0.Z;
778  double leftM21 = left.Row1.X;
779  double leftM22 = left.Row1.Y;
780  double leftM23 = left.Row1.Z;
781  double leftM31 = left.Row2.X;
782  double leftM32 = left.Row2.Y;
783  double leftM33 = left.Row2.Z;
784  double rightM11 = right.Row0.X;
785  double rightM12 = right.Row0.Y;
786  double rightM13 = right.Row0.Z;
787  double rightM21 = right.Row1.X;
788  double rightM22 = right.Row1.Y;
789  double rightM23 = right.Row1.Z;
790  double rightM31 = right.Row2.X;
791  double rightM32 = right.Row2.Y;
792  double rightM33 = right.Row2.Z;

811  public static void Invert(in Matrix3d matrix, out Matrix3d result)
812  {
813  // Original implementation can be found here:
814  // https://github.com/niswegmann/small-matrix-inverse/blob/6eac02b84ad06870692abaf828638a391548502c/invert3x3_c.h
815  double row0X = matrix.Row0.X, row0Y = matrix.Row0.Y, row0Z = matrix.Row0.Z;
816  double row1X = matrix.Row1.X, row1Y = matrix.Row1.Y, row1Z = matrix.Row1.Z;
817  double row2X = matrix.Row2.X, row2Y = matrix.Row2.Y, row2Z = matrix.Row2.Z;
818 
819  // Compute the elements needed to calculate the determinant
820  // so that we can throw without writing anything to the out parameter.
821  double invRow0X = +row1Y * row2Z - row1Z * row2Y;
822  double invRow1X = -row1X * row2Z + row1Z * row2X;
823  double invRow2X = +row1X * row2Y - row1Y * row2X;
824 
825  // Compute determinant:
826  double det = row0X * invRow0X + row0Y * invRow1X + row0Z * invRow2X;
827 
828  if (det == 0f)
829  {
830  throw new InvalidOperationException("Matrix is singular and cannot be inverted.");
831  }
832 
833  // Compute adjoint:
834  result.Row0.X = invRow0X;
835  result.Row0.Y = -row0Y * row2Z + row0Z * row2Y;
836  result.Row0.Z = +row0Y * row1Z - row0Z * row1Y;
837  result.Row1.X = invRow1X;
838  result.Row1.Y = +row0X * row2Z - row0Z * row2X;
839  result.Row1.Z = -row0X * row1Z + row0Z * row1X;
840  result.Row2.X = invRow2X;
841  result.Row2.Y = -row0X * row2Y + row0Y * row2X;
842  result.Row2.Z = +row0X * row1Y - row0Y * row1X;
843 
844  // Multiply adjoint with reciprocal of determinant:
845  det = 1.0f / det;
846 
847  result.Row0.X *= det;
848  result.Row0.Y *= det;
849  result.Row0.Z *= det;
850  result.Row1.X *= det;
851  result.Row1.Y *= det;
852  result.Row1.Z *= det;
853  result.Row2.X *= det;
854  result.Row2.Y *= det;
855  result.Row2.Z *= det;
856  }
