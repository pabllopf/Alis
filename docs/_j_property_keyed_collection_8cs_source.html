228  public bool Compare(JPropertyKeyedCollection other)
229  {
230  if (this == other)
231  {
232  return true;
233  }
234

236  // ignore order when comparing properties
237  Dictionary<string, JToken>? d1 = _dictionary;
238  Dictionary<string, JToken>? d2 = other._dictionary;
239 
240  if (d1 == null && d2 == null)
241  {
242  return true;
243  }
244 
245  if (d1 == null)
246  {
247  return (d2!.Count == 0);
248  }
249

252  return (d1.Count == 0);
253  }

256  {

261  {
262  if (!d2.TryGetValue(keyAndProperty.Key, out JToken secondValue))
263  {
264  return false;
265  }
266 
267  JProperty p1 = (JProperty)keyAndProperty.Value;
268  JProperty p2 = (JProperty)secondValue;
269 
270  if (p1.Value == null)
271  {
272  return (p2.Value == null);
273  }
274 
275  if (!p1.Value.DeepEquals(p2.Value))
276  {
277  return false;
278  }
279  }
280
