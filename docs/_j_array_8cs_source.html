118  public new static JArray Load(JsonReader reader, JsonLoadSettings? settings)
119  {
120  if (reader.TokenType == JsonToken.None)
121  {
122  if (!reader.Read())
123  {
124  throw JsonReaderException.Create(reader, "Error reading JArray from JsonReader.");
125  }
126  }
127 
128  reader.MoveToContent();
129 
130  if (reader.TokenType != JsonToken.StartArray)
131  {
132  throw JsonReaderException.Create(reader, "Error reading JArray from JsonReader. Current JsonReader item is not an array: {0}".FormatWith(CultureInfo.InvariantCulture, reader.TokenType));
133  }
134 
135  JArray a = new JArray();
136  a.SetLineInfo(reader as IJsonLineInfo, settings);
137 
138  a.ReadTokenFrom(reader, settings);
139 
140  return a;
141  }
