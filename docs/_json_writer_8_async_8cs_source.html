41  internal Task AutoCompleteAsync(JsonToken tokenBeingWritten, CancellationToken cancellationToken)
42  {
43  State oldState = _currentState;
44 
45  // gets new state based on the current state and what is being written
46  State newState = StateArray[(int)tokenBeingWritten][(int)oldState];
47 
48  if (newState == State.Error)
49  {
50  throw JsonWriterException.Create(this, "Token {0} in state {1} would result in an invalid JSON object.".FormatWith(CultureInfo.InvariantCulture, tokenBeingWritten.ToString(), oldState.ToString()), null);
51  }
52 
53  _currentState = newState;
54 
55  if (_formatting == Formatting.Indented)
56  {
57  switch (oldState)
58  {
59  case State.Start:
60  break;
61  case State.Property:
62  return WriteIndentSpaceAsync(cancellationToken);
63  case State.ArrayStart:
64  case State.ConstructorStart:
65  return WriteIndentAsync(cancellationToken);
66  case State.Array:
67  case State.Constructor:
68  return tokenBeingWritten == JsonToken.Comment ? WriteIndentAsync(cancellationToken) : AutoCompleteAsync(cancellationToken);
69  case State.Object:
70  switch (tokenBeingWritten)
71  {
72  case JsonToken.Comment:
73  break;
74  case JsonToken.PropertyName:
75  return AutoCompleteAsync(cancellationToken);
76  default:
77  return WriteValueDelimiterAsync(cancellationToken);
78  }
79 
80  break;
81  default:
82  if (tokenBeingWritten == JsonToken.PropertyName)
83  {
84  return WriteIndentAsync(cancellationToken);
85  }
86 
87  break;
88  }
89  }
90  else if (tokenBeingWritten != JsonToken.Comment)
91  {
92  switch (oldState)
93  {
94  case State.Object:
95  case State.Array:
96  case State.Constructor:
97  return WriteValueDelimiterAsync(cancellationToken);
98  }
99  }
100 
101  return AsyncUtils.CompletedTask;
102  }
103

107  await WriteIndentAsync(cancellationToken).ConfigureAwait(false);
108  }

278  internal Task InternalWriteEndAsync(JsonContainerType type, CancellationToken cancellationToken)
279  {
280  if (cancellationToken.IsCancellationRequested)
281  {
282  return cancellationToken.FromCanceled();
283  }
284 
285  int levelsToComplete = CalculateLevelsToComplete(type);
286  while (levelsToComplete-- > 0)
287  {
288  JsonToken token = GetCloseTokenForType(Pop());
289 
290  Task t;
291  if (_currentState == State.Property)
292  {
293  t = WriteNullAsync(cancellationToken);
294  if (!t.IsCompletedSucessfully())
295  {
296  return AwaitProperty(t, levelsToComplete, token, cancellationToken);
297  }
298  }
299 
300  if (_formatting == Formatting.Indented)
301  {
302  if (_currentState != State.ObjectStart && _currentState != State.ArrayStart)
303  {
304  t = WriteIndentAsync(cancellationToken);
305  if (!t.IsCompletedSucessfully())
306  {
307  return AwaitIndent(t, levelsToComplete, token, cancellationToken);
308  }
309  }
310  }
311 
312  t = WriteEndAsync(token, cancellationToken);
313  if (!t.IsCompletedSucessfully())
314  {
315  return AwaitEnd(t, levelsToComplete, cancellationToken);
316  }
317 
318  UpdateCurrentState();
319  }
320 
321  return AsyncUtils.CompletedTask;

676  public Task WriteTokenAsync(JsonToken token, object? value, CancellationToken cancellationToken = default)
677  {
678  if (cancellationToken.IsCancellationRequested)
679  {
680  return cancellationToken.FromCanceled();
681  }
682 
683  switch (token)
684  {
685  case JsonToken.None:
686  // read to next
687  return AsyncUtils.CompletedTask;
688  case JsonToken.StartObject:
689  return WriteStartObjectAsync(cancellationToken);
690  case JsonToken.StartArray:
691  return WriteStartArrayAsync(cancellationToken);
692  case JsonToken.StartConstructor:
693  ValidationUtils.ArgumentNotNull(value, nameof(value));
694  return WriteStartConstructorAsync(value.ToString(), cancellationToken);
695  case JsonToken.PropertyName:
696  ValidationUtils.ArgumentNotNull(value, nameof(value));
697  return WritePropertyNameAsync(value.ToString(), cancellationToken);
698  case JsonToken.Comment:
699  return WriteCommentAsync(value?.ToString(), cancellationToken);
700  case JsonToken.Integer:
701  ValidationUtils.ArgumentNotNull(value, nameof(value));
702  return
703 #if HAVE_BIG_INTEGER
704  value is BigInteger integer ? WriteValueAsync(integer, cancellationToken) :
705 #endif
706  WriteValueAsync(Convert.ToInt64(value, CultureInfo.InvariantCulture), cancellationToken);
707  case JsonToken.Float:
708  ValidationUtils.ArgumentNotNull(value, nameof(value));
709  if (value is decimal dec)
710  {
711  return WriteValueAsync(dec, cancellationToken);
712  }
713 
714  if (value is double doub)
715  {
716  return WriteValueAsync(doub, cancellationToken);
717  }
718 
719  if (value is float f)
720  {
721  return WriteValueAsync(f, cancellationToken);
722  }
723 
724  return WriteValueAsync(Convert.ToDouble(value, CultureInfo.InvariantCulture), cancellationToken);
725  case JsonToken.String:
726  ValidationUtils.ArgumentNotNull(value, nameof(value));
727  return WriteValueAsync(value.ToString(), cancellationToken);
728  case JsonToken.Boolean:
729  ValidationUtils.ArgumentNotNull(value, nameof(value));
730  return WriteValueAsync(Convert.ToBoolean(value, CultureInfo.InvariantCulture), cancellationToken);
731  case JsonToken.Null:
732  return WriteNullAsync(cancellationToken);
733  case JsonToken.Undefined:
734  return WriteUndefinedAsync(cancellationToken);
735  case JsonToken.EndObject:
736  return WriteEndObjectAsync(cancellationToken);
737  case JsonToken.EndArray:
738  return WriteEndArrayAsync(cancellationToken);
739  case JsonToken.EndConstructor:
740  return WriteEndConstructorAsync(cancellationToken);
741  case JsonToken.Date:
742  ValidationUtils.ArgumentNotNull(value, nameof(value));
743  if (value is DateTimeOffset offset)
744  {
745  return WriteValueAsync(offset, cancellationToken);
746  }
747 
748  return WriteValueAsync(Convert.ToDateTime(value, CultureInfo.InvariantCulture), cancellationToken);
749  case JsonToken.Raw:
750  return WriteRawValueAsync(value?.ToString(), cancellationToken);
751  case JsonToken.Bytes:
752  ValidationUtils.ArgumentNotNull(value, nameof(value));
753  if (value is Guid guid)
754  {
755  return WriteValueAsync(guid, cancellationToken);
756  }
757 
758  return WriteValueAsync((byte[]?)value, cancellationToken);
759  default:
760  throw MiscellaneousUtils.CreateArgumentOutOfRangeException(nameof(token), token, "Unexpected token type.");
761  }
762  }

764  internal virtual async Task WriteTokenAsync(JsonReader reader, bool writeChildren, bool writeDateConstructorAsDate, bool writeComments, CancellationToken cancellationToken)
765  {
766  int initialDepth = CalculateWriteTokenInitialDepth(reader);
767 
768  do
769  {
770  // write a JValue date when the constructor is for a date
771  if (writeDateConstructorAsDate && reader.TokenType == JsonToken.StartConstructor && string.Equals(reader.Value?.ToString(), "Date", StringComparison.Ordinal))
772  {
773  await WriteConstructorDateAsync(reader, cancellationToken).ConfigureAwait(false);
774  }
775  else
776  {
777  if (writeComments || reader.TokenType != JsonToken.Comment)
778  {
779  await WriteTokenAsync(reader.TokenType, reader.Value, cancellationToken).ConfigureAwait(false);
780  }
781  }
782  } while (
783  // stop if we have reached the end of the token being read
784  initialDepth - 1 < reader.Depth - (JsonTokenUtils.IsEndToken(reader.TokenType) ? 1 : 0)
785  && writeChildren
786  && await reader.ReadAsync(cancellationToken).ConfigureAwait(false));
787 
788  if (IsWriteTokenIncomplete(reader, writeChildren, initialDepth))
789  {
790  throw JsonWriterException.Create(this, "Unexpected end when reading token.", null);
791  }
792  }
793 
794  // For internal use, when we know the writer does not offer true async support (e.g. when backed
795  // by a StringWriter) and therefore async write methods are always in practice just a less efficient
796  // path through the sync version.
797  internal async Task WriteTokenSyncReadingAsync(JsonReader reader, CancellationToken cancellationToken)
798  {
799  int initialDepth = CalculateWriteTokenInitialDepth(reader);
800 
801  do
802  {
803  // write a JValue date when the constructor is for a date
804  if (reader.TokenType == JsonToken.StartConstructor && string.Equals(reader.Value?.ToString(), "Date", StringComparison.Ordinal))
805  {
806  WriteConstructorDate(reader);
807  }
808  else
809  {
810  WriteToken(reader.TokenType, reader.Value);
811  }
812  } while (
813  // stop if we have reached the end of the token being read
814  initialDepth - 1 < reader.Depth - (JsonTokenUtils.IsEndToken(reader.TokenType) ? 1 : 0)
815  && await reader.ReadAsync(cancellationToken).ConfigureAwait(false));
816 
817  if (initialDepth < CalculateWriteTokenFinalDepth(reader))
818  {
819  throw JsonWriterException.Create(this, "Unexpected end when reading token.", null);
820  }
821  }

823  private async Task WriteConstructorDateAsync(JsonReader reader, CancellationToken cancellationToken)
824  {
825  if (!await reader.ReadAsync(cancellationToken).ConfigureAwait(false))
826  {
827  throw JsonWriterException.Create(this, "Unexpected end when reading date constructor.", null);
828  }
829  if (reader.TokenType != JsonToken.Integer)
830  {
831  throw JsonWriterException.Create(this, "Unexpected token when reading date constructor. Expected Integer, got " + reader.TokenType, null);
832  }
833 
834  DateTime date = DateTimeUtils.ConvertJavaScriptTicksToDateTime((long)reader.Value!);
835 
836  if (!await reader.ReadAsync(cancellationToken).ConfigureAwait(false))
837  {
838  throw JsonWriterException.Create(this, "Unexpected end when reading date constructor.", null);
839  }
840  if (reader.TokenType != JsonToken.EndConstructor)
841  {
842  throw JsonWriterException.Create(this, "Unexpected token when reading date constructor. Expected EndConstructor, got " + reader.TokenType, null);
843  }
844 
845  await WriteValueAsync(date, cancellationToken).ConfigureAwait(false);
846  }
