201  private static bool FilterMembers(MemberInfo member)
202  {
203  if (member is PropertyInfo property)
204  {
205  if (ReflectionUtils.IsIndexedProperty(property))
206  {
207  return false;
208  }
209 
210  return !ReflectionUtils.IsByRefLikeType(property.PropertyType);
211  }
212  else if (member is FieldInfo field)
213  {
214  return !ReflectionUtils.IsByRefLikeType(field.FieldType);
215  }

225  protected virtual List<MemberInfo> GetSerializableMembers(Type objectType)
226  {
227  bool ignoreSerializableAttribute;
228 #if HAVE_BINARY_SERIALIZATION
229  ignoreSerializableAttribute = IgnoreSerializableAttribute;
230 #else
231  ignoreSerializableAttribute = true;
232 #endif
233 
234  MemberSerialization memberSerialization = JsonTypeReflector.GetObjectMemberSerialization(objectType, ignoreSerializableAttribute);
235 
236  // Exclude index properties
237  // Do not filter ByRef types here because accessing FieldType/PropertyType can trigger additonal assembly loads
238  IEnumerable<MemberInfo> allMembers = ReflectionUtils.GetFieldsAndProperties(objectType, BindingFlags.Public | BindingFlags.NonPublic | BindingFlags.Instance | BindingFlags.Static)
239  .Where(m => m is PropertyInfo p ? !ReflectionUtils.IsIndexedProperty(p) : true);
240 
241  List<MemberInfo> serializableMembers = new List<MemberInfo>();
242 
243  if (memberSerialization != MemberSerialization.Fields)
244  {
245 #if HAVE_DATA_CONTRACTS
246  DataContractAttribute? dataContractAttribute = JsonTypeReflector.GetDataContractAttribute(objectType);
247 #endif
248 
249 #pragma warning disable 618
250  // Exclude index properties and ByRef types
251  List<MemberInfo> defaultMembers = ReflectionUtils.GetFieldsAndProperties(objectType, DefaultMembersSearchFlags)
252  .Where(FilterMembers).ToList();
253 #pragma warning restore 618
254 
255  foreach (MemberInfo member in allMembers)
256  {
257  // exclude members that are compiler generated if set
258  if (SerializeCompilerGeneratedMembers || !member.IsDefined(typeof(CompilerGeneratedAttribute), true))
259  {
260  if (defaultMembers.Contains(member))
261  {
262  // add all members that are found by default member search
263  serializableMembers.Add(member);
264  }
265  else
266  {
267  // add members that are explicitly marked with JsonProperty/DataMember attribute
268  // or are a field if serializing just fields
269  if (JsonTypeReflector.GetAttribute<JsonPropertyAttribute>(member) != null)
270  {
271  serializableMembers.Add(member);
272  }
273  else if (JsonTypeReflector.GetAttribute<JsonRequiredAttribute>(member) != null)
274  {
275  serializableMembers.Add(member);
276  }
277 #if HAVE_DATA_CONTRACTS
278  else if (dataContractAttribute != null && JsonTypeReflector.GetAttribute<DataMemberAttribute>(member) != null)
279  {
280  serializableMembers.Add(member);
281  }
282 #endif
283  else if (memberSerialization == MemberSerialization.Fields && member.MemberType() == MemberTypes.Field)
284  {
285  serializableMembers.Add(member);
286  }
287  }
288  }
289  }
290 
291 #if HAVE_DATA_CONTRACTS
292  // don't include EntityKey on entities objects... this is a bit hacky
293  if (objectType.AssignableToTypeName("System.Data.Objects.DataClasses.EntityObject", false, out _))
294  {
295  serializableMembers = serializableMembers.Where(ShouldSerializeEntityMember).ToList();
296  }
297 #endif
298  // don't include TargetSite on non-serializable exceptions
299  // MemberBase is problematic to serialize. Large, self referencing instances, etc
300  if (typeof(Exception).IsAssignableFrom(objectType))
301  {
302  serializableMembers = serializableMembers.Where(m => !string.Equals(m.Name, "TargetSite", StringComparison.Ordinal)).ToList();
303  }
304  }
305  else
306  {
307  // serialize all fields
308  foreach (MemberInfo member in allMembers)
309  {
310  if (member is FieldInfo field && !field.IsStatic)
311  {
312  serializableMembers.Add(member);
313  }
314  }
315  }
316 
317  return serializableMembers;
318  }

340  protected virtual JsonObjectContract CreateObjectContract(Type objectType)
341  {
342  JsonObjectContract contract = new JsonObjectContract(objectType);
343  InitializeContract(contract);
344 
345  bool ignoreSerializableAttribute;
346 #if HAVE_BINARY_SERIALIZATION
347  ignoreSerializableAttribute = IgnoreSerializableAttribute;
348 #else
349  ignoreSerializableAttribute = true;
350 #endif
351 
352  contract.MemberSerialization = JsonTypeReflector.GetObjectMemberSerialization(contract.NonNullableUnderlyingType, ignoreSerializableAttribute);
353  contract.Properties.AddRange(CreateProperties(contract.NonNullableUnderlyingType, contract.MemberSerialization));
354 
355  Func<string, string>? extensionDataNameResolver = null;
356 
357  JsonObjectAttribute? attribute = JsonTypeReflector.GetCachedAttribute<JsonObjectAttribute>(contract.NonNullableUnderlyingType);
358  if (attribute != null)
359  {
360  contract.ItemRequired = attribute._itemRequired;
361  contract.ItemNullValueHandling = attribute._itemNullValueHandling;
362  contract.MissingMemberHandling = attribute._missingMemberHandling;
363 
364  if (attribute.NamingStrategyType != null)
365  {
366  NamingStrategy namingStrategy = JsonTypeReflector.GetContainerNamingStrategy(attribute)!;
367  extensionDataNameResolver = s => namingStrategy.GetDictionaryKey(s);
368  }
369  }
370 
371  if (extensionDataNameResolver == null)
372  {
373  extensionDataNameResolver = ResolveExtensionDataName;
374  }
375 
376  contract.ExtensionDataNameResolver = extensionDataNameResolver;
377 
378  if (contract.IsInstantiable)
379  {
380  ConstructorInfo? overrideConstructor = GetAttributeConstructor(contract.NonNullableUnderlyingType);
381 
382  // check if a JsonConstructorAttribute has been defined and use that
383  if (overrideConstructor != null)
384  {
385  contract.OverrideCreator = JsonTypeReflector.ReflectionDelegateFactory.CreateParameterizedConstructor(overrideConstructor);
386  contract.CreatorParameters.AddRange(CreateConstructorParameters(overrideConstructor, contract.Properties));
387  }
388  else if (contract.MemberSerialization == MemberSerialization.Fields)
389  {
390 #if HAVE_BINARY_FORMATTER
391  // mimic DataContractSerializer behaviour when populating fields by overriding default creator to create an uninitialized object
392  // note that this is only possible when the application is fully trusted so fall back to using the default constructor (if available) in partial trust
393  if (JsonTypeReflector.FullyTrusted)
394  {
395  contract.DefaultCreator = contract.GetUninitializedObject;
396  }
397 #endif
398  }
399  else if (contract.DefaultCreator == null || contract.DefaultCreatorNonPublic)
400  {
401  ConstructorInfo? constructor = GetParameterizedConstructor(contract.NonNullableUnderlyingType);
402  if (constructor != null)
403  {
404  contract.ParameterizedCreator = JsonTypeReflector.ReflectionDelegateFactory.CreateParameterizedConstructor(constructor);
405  contract.CreatorParameters.AddRange(CreateConstructorParameters(constructor, contract.Properties));
406  }
407  }
408  else if (contract.NonNullableUnderlyingType.IsValueType())
409  {
410  // value types always have default constructor
411  // check whether there is a constructor that matches with non-writable properties on value type
412  ConstructorInfo? constructor = GetImmutableConstructor(contract.NonNullableUnderlyingType, contract.Properties);
413  if (constructor != null)
414  {
415  contract.OverrideCreator = JsonTypeReflector.ReflectionDelegateFactory.CreateParameterizedConstructor(constructor);
416  contract.CreatorParameters.AddRange(CreateConstructorParameters(constructor, contract.Properties));
417  }
418  }
419  }
420 
421  MemberInfo extensionDataMember = GetExtensionDataMemberForType(contract.NonNullableUnderlyingType);
422  if (extensionDataMember != null)
423  {
424  SetExtensionDataDelegates(contract, extensionDataMember);
425  }
426 
427  // serializing DirectoryInfo without ISerializable will stackoverflow
428  // https://github.com/JamesNK/Newtonsoft.Json/issues/1541
429  if (Array.IndexOf(BlacklistedTypeNames, objectType.FullName) != -1)
430  {
431  contract.OnSerializingCallbacks.Add(ThrowUnableToSerializeError);
432  }
433 
434  return contract;
435  }

442  private MemberInfo GetExtensionDataMemberForType(Type type)
443  {
444  IEnumerable<MemberInfo> members = GetClassHierarchyForType(type).SelectMany(baseType =>
445  {
446  IList<MemberInfo> m = new List<MemberInfo>();
447  m.AddRange(baseType.GetProperties(BindingFlags.Public | BindingFlags.NonPublic | BindingFlags.Instance | BindingFlags.DeclaredOnly));
448  m.AddRange(baseType.GetFields(BindingFlags.Public | BindingFlags.NonPublic | BindingFlags.Instance | BindingFlags.DeclaredOnly));
449 
450  return m;
451  });
452 
453  MemberInfo extensionDataMember = members.LastOrDefault(m =>
454  {
455  MemberTypes memberType = m.MemberType();
456  if (memberType != MemberTypes.Property && memberType != MemberTypes.Field)
457  {
458  return false;
459  }
460 
461  // last instance of attribute wins on type if there are multiple
462  if (!m.IsDefined(typeof(JsonExtensionDataAttribute), false))
463  {
464  return false;
465  }
466 
467  if (!ReflectionUtils.CanReadMemberValue(m, true))
468  {
469  throw new JsonException("Invalid extension data attribute on '{0}'. Member '{1}' must have a getter.".FormatWith(CultureInfo.InvariantCulture, GetClrTypeFullName(m.DeclaringType), m.Name));
470  }
471 
472  Type t = ReflectionUtils.GetMemberUnderlyingType(m);
473 
474  if (ReflectionUtils.ImplementsGenericDefinition(t, typeof(IDictionary<,>), out Type? dictionaryType))
475  {
476  Type keyType = dictionaryType.GetGenericArguments()[0];
477  Type valueType = dictionaryType.GetGenericArguments()[1];
478 
479  if (keyType.IsAssignableFrom(typeof(string)) && valueType.IsAssignableFrom(typeof(JToken)))
480  {
481  return true;
482  }
483  }
484 
485  throw new JsonException("Invalid extension data attribute on '{0}'. Member '{1}' type must implement IDictionary<string, JToken>.".FormatWith(CultureInfo.InvariantCulture, GetClrTypeFullName(m.DeclaringType), m.Name));
486  });
487 
488  return extensionDataMember;
489  }

491  private static void SetExtensionDataDelegates(JsonObjectContract contract, MemberInfo member)
492  {
493  JsonExtensionDataAttribute? extensionDataAttribute = ReflectionUtils.GetAttribute<JsonExtensionDataAttribute>(member);
494  if (extensionDataAttribute == null)
495  {
496  return;
497  }
498 
499  Type t = ReflectionUtils.GetMemberUnderlyingType(member);
500 
501  ReflectionUtils.ImplementsGenericDefinition(t, typeof(IDictionary<,>), out Type? dictionaryType);
502 
503  Type keyType = dictionaryType!.GetGenericArguments()[0];
504  Type valueType = dictionaryType!.GetGenericArguments()[1];
505 
506  Type createdType;
507 
508  // change type to a class if it is the base interface so it can be instantiated if needed
509  if (ReflectionUtils.IsGenericDefinition(t, typeof(IDictionary<,>)))
510  {
511  createdType = typeof(Dictionary<,>).MakeGenericType(keyType, valueType);
512  }
513  else
514  {
515  createdType = t;
516  }
517 
518  Func<object, object?> getExtensionDataDictionary = JsonTypeReflector.ReflectionDelegateFactory.CreateGet<object>(member);
519 
520  if (extensionDataAttribute.ReadData)
521  {
522  Action<object, object?>? setExtensionDataDictionary = (ReflectionUtils.CanSetMemberValue(member, true, false))
523  ? JsonTypeReflector.ReflectionDelegateFactory.CreateSet<object>(member)
524  : null;
525  Func<object> createExtensionDataDictionary = JsonTypeReflector.ReflectionDelegateFactory.CreateDefaultConstructor<object>(createdType);
526  MethodInfo? setMethod = t.GetProperty("Item", BindingFlags.Public | BindingFlags.Instance, null, valueType, new[] { keyType }, null)?.GetSetMethod();
527  if (setMethod == null)
528  {
529  // Item is explicitly implemented and non-public
530  // get from dictionary interface
531  setMethod = dictionaryType!.GetProperty("Item", BindingFlags.Public | BindingFlags.Instance, null, valueType, new[] { keyType }, null)?.GetSetMethod();
532  }
533 
534  MethodCall<object, object?> setExtensionDataDictionaryValue = JsonTypeReflector.ReflectionDelegateFactory.CreateMethodCall<object>(setMethod!);
535 
536  ExtensionDataSetter extensionDataSetter = (o, key, value) =>
537  {
538  object? dictionary = getExtensionDataDictionary(o);
539  if (dictionary == null)
540  {
541  if (setExtensionDataDictionary == null)
542  {
543  throw new JsonSerializationException("Cannot set value onto extension data member '{0}'. The extension data collection is null and it cannot be set.".FormatWith(CultureInfo.InvariantCulture, member.Name));
544  }
545 
546  dictionary = createExtensionDataDictionary();
547  setExtensionDataDictionary(o, dictionary);
548  }
549 
550  setExtensionDataDictionaryValue(dictionary, key, value);
551  };
552 
553  contract.ExtensionDataSetter = extensionDataSetter;
554  }
555 
556  if (extensionDataAttribute.WriteData)
557  {
558  Type enumerableWrapper = typeof(EnumerableDictionaryWrapper<,>).MakeGenericType(keyType, valueType);
559  ConstructorInfo constructors = enumerableWrapper.GetConstructors().First();
560  ObjectConstructor<object> createEnumerableWrapper = JsonTypeReflector.ReflectionDelegateFactory.CreateParameterizedConstructor(constructors);
561 
562  ExtensionDataGetter extensionDataGetter = o =>
563  {
564  object? dictionary = getExtensionDataDictionary(o);
565  if (dictionary == null)
566  {
567  return null;
568  }
569 
570  return (IEnumerable<KeyValuePair<object, object>>)createEnumerableWrapper(dictionary);
571  };
572 
573  contract.ExtensionDataGetter = extensionDataGetter;
574  }
575 
576  contract.ExtensionDataValueType = valueType;
577  }

630  {
631  IEnumerable<ConstructorInfo> constructors = objectType.GetConstructors();
632  IEnumerator<ConstructorInfo> en = constructors.GetEnumerator();
633  if (en.MoveNext())
634  {
635  ConstructorInfo constructor = en.Current;
636  if (!en.MoveNext())
637  {
638  ParameterInfo[] parameters = constructor.GetParameters();
639  if (parameters.Length > 0)
640  {
641  foreach (ParameterInfo parameterInfo in parameters)
642  {
643  JsonProperty? memberProperty = MatchProperty(memberProperties, parameterInfo.Name, parameterInfo.ParameterType);
644  if (memberProperty == null || memberProperty.Writable)
645  {
646  return null;
647  }
648  }
649 
650  return constructor;
651  }
652  }
653  }

687  protected virtual IList<JsonProperty> CreateConstructorParameters(ConstructorInfo constructor, JsonPropertyCollection memberProperties)
688  {
689  ParameterInfo[] constructorParameters = constructor.GetParameters();
690 
691  JsonPropertyCollection parameterCollection = new JsonPropertyCollection(constructor.DeclaringType);
692 
693  foreach (ParameterInfo parameterInfo in constructorParameters)
694  {
695  if (parameterInfo.Name == null)
696  {
697  continue;
698  }
699 
700  JsonProperty? matchingMemberProperty = MatchProperty(memberProperties, parameterInfo.Name, parameterInfo.ParameterType);
701 
702  // ensure that property will have a name from matching property or from parameterinfo
703  // parameterinfo could have no name if generated by a proxy (I'm looking at you Castle)
704  if (matchingMemberProperty != null || parameterInfo.Name != null)
705  {
706  JsonProperty property = CreatePropertyFromConstructorParameter(matchingMemberProperty, parameterInfo);
707 
708  if (property != null)
709  {
710  parameterCollection.AddProperty(property);
711  }
712  }
713  }
714 
715  return parameterCollection;
716  }

718  private JsonProperty? MatchProperty(JsonPropertyCollection properties, string name, Type type)
719  {
720  // it is possible to generate a member with a null name using Reflection.Emit
721  // protect against an ArgumentNullException from GetClosestMatchProperty by testing for null here
722  if (name == null)
723  {
724  return null;
725  }
726 
727  JsonProperty? property = properties.GetClosestMatchProperty(name);
728  // must match type as well as name
729  if (property == null || property.PropertyType != type)
730  {
731  return null;
732  }
733 
734  return property;
735  }

743  protected virtual JsonProperty CreatePropertyFromConstructorParameter(JsonProperty? matchingMemberProperty, ParameterInfo parameterInfo)
744  {
745  JsonProperty property = new JsonProperty();
746  property.PropertyType = parameterInfo.ParameterType;
747  property.AttributeProvider = new ReflectionAttributeProvider(parameterInfo);
748 
749  SetPropertySettingsFromAttributes(property, parameterInfo, parameterInfo.Name, parameterInfo.Member.DeclaringType, MemberSerialization.OptOut, out _);
750 
751  property.Readable = false;
752  property.Writable = true;
753 
754  // "inherit" values from matching member property if unset on parameter
755  if (matchingMemberProperty != null)
756  {
757  property.PropertyName = (property.PropertyName != parameterInfo.Name) ? property.PropertyName : matchingMemberProperty.PropertyName;
758  property.Converter = property.Converter ?? matchingMemberProperty.Converter;
759 
760  if (!property._hasExplicitDefaultValue && matchingMemberProperty._hasExplicitDefaultValue)
761  {
762  property.DefaultValue = matchingMemberProperty.DefaultValue;
763  }
764 
765  property._required = property._required ?? matchingMemberProperty._required;
766  property.IsReference = property.IsReference ?? matchingMemberProperty.IsReference;
767  property.NullValueHandling = property.NullValueHandling ?? matchingMemberProperty.NullValueHandling;
768  property.DefaultValueHandling = property.DefaultValueHandling ?? matchingMemberProperty.DefaultValueHandling;
769  property.ReferenceLoopHandling = property.ReferenceLoopHandling ?? matchingMemberProperty.ReferenceLoopHandling;
770  property.ObjectCreationHandling = property.ObjectCreationHandling ?? matchingMemberProperty.ObjectCreationHandling;
771  property.TypeNameHandling = property.TypeNameHandling ?? matchingMemberProperty.TypeNameHandling;
772  }
773 
774  return property;
775  }
