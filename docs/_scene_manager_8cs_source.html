116  public static void Load(string name)
117  {
118  Span<Scene> span = current.scenes.Span;
119  for (int i = 0; i < span.Length; i++)
120  {
121  if (span[i] != null && span[i].Name.Equals(name))
122  {
123  Logger.Log(string.Format(ChangeScene, name, current.currentScene.Name));
124  current.currentScene.IsActive = false;
