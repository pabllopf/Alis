113  public JsonDictionaryContract(Type underlyingType)
114  : base(underlyingType)
115  {
116  ContractType = JsonContractType.Dictionary;
117 
118  Type? keyType;
119  Type? valueType;
120 
121  if (ReflectionUtils.ImplementsGenericDefinition(underlyingType, typeof(IDictionary<,>), out _genericCollectionDefinitionType))
122  {
123  keyType = _genericCollectionDefinitionType.GetGenericArguments()[0];
124  valueType = _genericCollectionDefinitionType.GetGenericArguments()[1];
125 
126  if (ReflectionUtils.IsGenericDefinition(UnderlyingType, typeof(IDictionary<,>)))
127  {
128  CreatedType = typeof(Dictionary<,>).MakeGenericType(keyType, valueType);
129  }
130  else if (underlyingType.IsGenericType())
131  {
132  // ConcurrentDictionary<,> + IDictionary setter + null value = error
133  // wrap to use generic setter
134  // https://github.com/JamesNK/Newtonsoft.Json/issues/1582
135  Type typeDefinition = underlyingType.GetGenericTypeDefinition();
136  if (typeDefinition.FullName == JsonTypeReflector.ConcurrentDictionaryTypeName)
137  {
138  ShouldCreateWrapper = true;
139  }
140  }
141 
142 #if HAVE_READ_ONLY_COLLECTIONS
143  IsReadOnlyOrFixedSize = ReflectionUtils.InheritsGenericDefinition(underlyingType, typeof(ReadOnlyDictionary<,>));
144 #endif
145 
146  }
147 #if HAVE_READ_ONLY_COLLECTIONS
148  else if (ReflectionUtils.ImplementsGenericDefinition(underlyingType, typeof(IReadOnlyDictionary<,>), out _genericCollectionDefinitionType))
149  {
150  keyType = _genericCollectionDefinitionType.GetGenericArguments()[0];
151  valueType = _genericCollectionDefinitionType.GetGenericArguments()[1];
152 
153  if (ReflectionUtils.IsGenericDefinition(UnderlyingType, typeof(IReadOnlyDictionary<,>)))
154  {
155  CreatedType = typeof(ReadOnlyDictionary<,>).MakeGenericType(keyType, valueType);
156  }
157 
158  IsReadOnlyOrFixedSize = true;
159  }
160 #endif
161  else
162  {
163  ReflectionUtils.GetDictionaryKeyValueTypes(UnderlyingType, out keyType, out valueType);
164 
165  if (UnderlyingType == typeof(IDictionary))
166  {
167  CreatedType = typeof(Dictionary<object, object>);
168  }
169  }
170 
171  if (keyType != null && valueType != null)
172  {
173  _parameterizedConstructor = CollectionUtils.ResolveEnumerableCollectionConstructor(
174  CreatedType,
175  typeof(KeyValuePair<,>).MakeGenericType(keyType, valueType),
176  typeof(IDictionary<,>).MakeGenericType(keyType, valueType));
177 
178 #if HAVE_FSHARP_TYPES
179  if (!HasParameterizedCreatorInternal && underlyingType.Name == FSharpUtils.FSharpMapTypeName)
180  {
181  FSharpUtils.EnsureInitialized(underlyingType.Assembly());
182  _parameterizedCreator = FSharpUtils.Instance.CreateMap(keyType, valueType);
183  }
184 #endif
185  }
186 
187  if (!typeof(IDictionary).IsAssignableFrom(CreatedType))
188  {
189  ShouldCreateWrapper = true;
190  }
191 
192  DictionaryKeyType = keyType;
193  DictionaryValueType = valueType;
194 
195 #if (NET20 || NET35)
196  if (DictionaryValueType != null && ReflectionUtils.IsNullableType(DictionaryValueType))
197  {
198  // bug in .NET 2.0 & 3.5 that Dictionary<TKey, Nullable<TValue>> throws an error when adding null via IDictionary[key] = object
199  // wrapper will handle calling Add(T) instead
200  if (ReflectionUtils.InheritsGenericDefinition(CreatedType, typeof(Dictionary<,>), out _))
201  {
202  ShouldCreateWrapper = true;
203  }
204  }
205 #endif
206 
207  if (DictionaryKeyType != null &&
208  DictionaryValueType != null &&
209  ImmutableCollectionsUtils.TryBuildImmutableForDictionaryContract(
210  underlyingType,
211  DictionaryKeyType,
212  DictionaryValueType,
213  out Type? immutableCreatedType,
214  out ObjectConstructor<object>? immutableParameterizedCreator))
215  {
216  CreatedType = immutableCreatedType;
217  _parameterizedCreator = immutableParameterizedCreator;
218  IsReadOnlyOrFixedSize = true;
219  }
220  }

222  internal IWrappedDictionary CreateWrapper(object dictionary)
223  {
224  if (_genericWrapperCreator == null)
225  {
226  _genericWrapperType = typeof(DictionaryWrapper<,>).MakeGenericType(DictionaryKeyType, DictionaryValueType);
227 
228  ConstructorInfo genericWrapperConstructor = _genericWrapperType.GetConstructor(new[] { _genericCollectionDefinitionType! });
229  _genericWrapperCreator = JsonTypeReflector.ReflectionDelegateFactory.CreateParameterizedConstructor(genericWrapperConstructor);
230  }
231 
232  return (IWrappedDictionary)_genericWrapperCreator(dictionary);
233  }

240 
241  _genericTemporaryDictionaryCreator = JsonTypeReflector.ReflectionDelegateFactory.CreateDefaultConstructor<object>(temporaryDictionaryType);
242  }
243 
244  return (IDictionary)_genericTemporaryDictionaryCreator();
245  }
