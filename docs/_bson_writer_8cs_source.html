206  internal void AddToken(BsonToken token)
207  {
208  if (_parent != null)
209  {
210  if (_parent is BsonObject bo)
211  {
212  bo.Add(_propertyName, token);
213  _propertyName = null;
214  }
215  else
216  {
217  ((BsonArray)_parent).Add(token);
218  }
219  }
220  else
221  {
222  if (token.Type != BsonType.Object && token.Type != BsonType.Array)
223  {
224  throw JsonWriterException.Create(this, "Error writing {0} value. BSON must start with an Object or Array.".FormatWith(CultureInfo.InvariantCulture, token.Type), null);
225  }
226 
227  _parent = token;
228  _root = token;
229  }
230  }
