220  private void Render(Vector2 vector2)
221  {
222  image = Image.LoadPixelData<Rgba32>(Project.VideoGame.PreviewRender(), 512, 512);
223  imageSharpTexture = new ImageSharpTexture(image, true);
224 
225  unsafe

228  {

230  if (!image.TryGetSinglePixelSpan(out Span<Rgba32> pixelSpan))
231  {
232  throw new VeldridException("Unable to get image pixelspan.");
233  }
234  fixed (void* pin = &MemoryMarshal.GetReference(pixelSpan))
235  {
236  imGuiController.graphicsDevice.UpdateTexture(
237  texture,
238  (IntPtr)pin,
239  (uint)(imageSharpTexture.PixelSizeInBytes * image.Width * image.Height),
240  0,
241  0,
242  0,
243  (uint)image.Width,
244  (uint)image.Height,
245  1,
246  (uint)level,
247  0);
248  }
249  }
250  }
251 
252  intPtr = imGuiController.GetOrCreateImGuiBinding(imGuiController.graphicsDevice.ResourceFactory, texture);
