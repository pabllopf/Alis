218  private void Render(Vector2 vector2)
219  {
220  image = Image.LoadPixelData<Rgba32>(Project.Current.VideoGame.PreviewRender(), 512, 512);
221  imageSharpTexture = new ImageSharpTexture(image, true);
222 
223  unsafe
224  {
225  for (int level = 0; level < imageSharpTexture.MipLevels; level++)
226  {
227  Image<Rgba32> image = imageSharpTexture.Images[level];
228  if (!image.TryGetSinglePixelSpan(out Span<Rgba32> pixelSpan))
229  {
230  throw new VeldridException("Unable to get image pixelspan.");
231  }
232  fixed (void* pin = &MemoryMarshal.GetReference(pixelSpan))
233  {
234  imGuiController.graphicsDevice.UpdateTexture(
235  texture,
236  (IntPtr)pin,
237  (uint)(imageSharpTexture.PixelSizeInBytes * image.Width * image.Height),
238  0,
239  0,
240  0,
241  (uint)image.Width,
242  (uint)image.Height,
243  1,
244  (uint)level,
245  0);
246  }
247  }
248  }
249 
250  intPtr = imGuiController.GetOrCreateImGuiBinding(imGuiController.graphicsDevice.ResourceFactory, texture);
251  ImGui.Image(intPtr, vector2);
252  }
