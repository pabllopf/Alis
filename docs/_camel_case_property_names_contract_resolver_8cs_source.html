65 
66  // for backwards compadibility the CamelCasePropertyNamesContractResolver shares contracts between instances
67  StructMultiKey<Type, Type> key = new StructMultiKey<Type, Type>(GetType(), type);
68  Dictionary<StructMultiKey<Type, Type>, JsonContract>? cache = _contractCache;
69  if (cache == null || !cache.TryGetValue(key, out JsonContract contract))
70  {
71  contract = CreateContract(type);

73  // avoid the possibility of modifying the cache dictionary while another thread is accessing it
74  lock (TypeContractCacheLock)

78  ? new Dictionary<StructMultiKey<Type, Type>, JsonContract>(cache)
79  : new Dictionary<StructMultiKey<Type, Type>, JsonContract>();
80  updatedCache[key] = contract;
81 
82  _contractCache = updatedCache;
83  }
