368  private void UpdateImGuiInput(InputSnapshot snapshot)
369  {
370  ImGuiIOPtr io = ImGui.GetIO();
371 
372  Vector2 mousePosition = snapshot.MousePosition;
373 
374  // Determine if any of the mouse buttons were pressed during this snapshot period, even if they are no longer held.
375  bool leftPressed = false;
376  bool middlePressed = false;
377  bool rightPressed = false;
378  foreach (MouseEvent me in snapshot.MouseEvents)
379  {
380  if (me.Down)
381  {
382  switch (me.MouseButton)
383  {
384  case MouseButton.Left:
385  leftPressed = true;
386  break;
387  case MouseButton.Middle:
388  middlePressed = true;
389  break;
390  case MouseButton.Right:
391  rightPressed = true;
392  break;
393  }
394  }
395  }
396 
397  io.MouseDown[0] = leftPressed || snapshot.IsMouseDown(MouseButton.Left);
398  io.MouseDown[1] = rightPressed || snapshot.IsMouseDown(MouseButton.Right);
399  io.MouseDown[2] = middlePressed || snapshot.IsMouseDown(MouseButton.Middle);
400  io.MousePos = mousePosition;
401  io.MouseWheel = snapshot.WheelDelta;
402 
403  IReadOnlyList<char> keyCharPresses = snapshot.KeyCharPresses;
404  for (int i = 0; i < keyCharPresses.Count; i++)
405  {
406  char c = keyCharPresses[i];
407  io.AddInputCharacter(c);
408  }
409 
410  IReadOnlyList<KeyEvent> keyEvents = snapshot.KeyEvents;
411  for (int i = 0; i < keyEvents.Count; i++)
412  {
413  KeyEvent keyEvent = keyEvents[i];
414  io.KeysDown[(int)keyEvent.Key] = keyEvent.Down;
415  if (keyEvent.Key == Key.ControlLeft)
416  {
417  _controlDown = keyEvent.Down;
418  }
419  if (keyEvent.Key == Key.ShiftLeft)
420  {
421  _shiftDown = keyEvent.Down;
422  }
423  if (keyEvent.Key == Key.AltLeft)
424  {
425  _altDown = keyEvent.Down;
426  }
427  if (keyEvent.Key == Key.WinLeft)
428  {
429  _winKeyDown = keyEvent.Down;
430  }
431  }
432 
433  io.KeyCtrl = _controlDown;
434  io.KeyAlt = _altDown;
435  io.KeyShift = _shiftDown;
436  io.KeySuper = _winKeyDown;
437  }

465  private void RenderImDrawData(ImDrawDataPtr draw_data, GraphicsDevice gd, CommandList cl)
466  {
467  uint vertexOffsetInVertices = 0;
468  uint indexOffsetInElements = 0;
469 
470  if (draw_data.CmdListsCount == 0)
471  {
472  return;
473  }
474 
475  uint totalVBSize = (uint)(draw_data.TotalVtxCount * Unsafe.SizeOf<ImDrawVert>());
476  if (totalVBSize > _vertexBuffer.SizeInBytes)
477  {
478  gd.DisposeWhenIdle(_vertexBuffer);
479  _vertexBuffer = gd.ResourceFactory.CreateBuffer(new BufferDescription((uint)(totalVBSize * 1.5f), BufferUsage.VertexBuffer | BufferUsage.Dynamic));
480  }
481 
482  uint totalIBSize = (uint)(draw_data.TotalIdxCount * sizeof(ushort));
483  if (totalIBSize > _indexBuffer.SizeInBytes)
484  {
485  gd.DisposeWhenIdle(_indexBuffer);
486  _indexBuffer = gd.ResourceFactory.CreateBuffer(new BufferDescription((uint)(totalIBSize * 1.5f), BufferUsage.IndexBuffer | BufferUsage.Dynamic));
487  }
488 
489  for (int i = 0; i < draw_data.CmdListsCount; i++)
490  {
491  ImDrawListPtr cmd_list = draw_data.CmdListsRange[i];
492 
493  cl.UpdateBuffer(
494  _vertexBuffer,
495  vertexOffsetInVertices * (uint)Unsafe.SizeOf<ImDrawVert>(),
496  cmd_list.VtxBuffer.Data,
497  (uint)(cmd_list.VtxBuffer.Size * Unsafe.SizeOf<ImDrawVert>()));
498 
499  cl.UpdateBuffer(
500  _indexBuffer,
501  indexOffsetInElements * sizeof(ushort),
502  cmd_list.IdxBuffer.Data,
503  (uint)(cmd_list.IdxBuffer.Size * sizeof(ushort)));
504 
505  vertexOffsetInVertices += (uint)cmd_list.VtxBuffer.Size;
506  indexOffsetInElements += (uint)cmd_list.IdxBuffer.Size;
507  }
508 
509  // Setup orthographic projection matrix into our constant buffer
510  ImGuiIOPtr io = ImGui.GetIO();
511  Matrix4x4 mvp = Matrix4x4.CreateOrthographicOffCenter(
512  0f,
513  io.DisplaySize.X,
514  io.DisplaySize.Y,
515  0.0f,
516  -1.0f,
517  1.0f);
518 
519  _gd.UpdateBuffer(_projMatrixBuffer, 0, ref mvp);
520 
521  cl.SetVertexBuffer(0, _vertexBuffer);
522  cl.SetIndexBuffer(_indexBuffer, IndexFormat.UInt16);
523  cl.SetPipeline(_pipeline);
524  cl.SetGraphicsResourceSet(0, _mainResourceSet);
525 
526  draw_data.ScaleClipRects(io.DisplayFramebufferScale);
527 
528  // Render command lists
529  int vtx_offset = 0;
530  int idx_offset = 0;
531  for (int n = 0; n < draw_data.CmdListsCount; n++)
532  {
533  ImDrawListPtr cmd_list = draw_data.CmdListsRange[n];
534  for (int cmd_i = 0; cmd_i < cmd_list.CmdBuffer.Size; cmd_i++)
535  {
536  ImDrawCmdPtr pcmd = cmd_list.CmdBuffer[cmd_i];
537  if (pcmd.UserCallback != IntPtr.Zero)
538  {
539  throw new NotImplementedException();
540  }
541  else
542  {
543  if (pcmd.TextureId != IntPtr.Zero)
544  {
545  if (pcmd.TextureId == _fontAtlasID)
546  {
547  cl.SetGraphicsResourceSet(1, _fontTextureResourceSet);
548  }
549  else
550  {
551  cl.SetGraphicsResourceSet(1, GetImageResourceSet(pcmd.TextureId));
552  }
553  }
554 
555  cl.SetScissorRect(
556  0,
557  (uint)pcmd.ClipRect.X,
558  (uint)pcmd.ClipRect.Y,
559  (uint)(pcmd.ClipRect.Z - pcmd.ClipRect.X),
560  (uint)(pcmd.ClipRect.W - pcmd.ClipRect.Y));
561 
562  cl.DrawIndexed(pcmd.ElemCount, 1, (uint)idx_offset, vtx_offset, 0);
563  }
564 
565  idx_offset += (int)pcmd.ElemCount;
566  }
567  vtx_offset += cmd_list.VtxBuffer.Size;
568  }
569  }
