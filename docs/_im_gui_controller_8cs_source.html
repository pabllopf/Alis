365  private void UpdateImGuiInput(InputSnapshot snapshot)
366  {
367  ImGuiIOPtr io = ImGui.GetIO();
368 
369  Vector2 mousePosition = snapshot.MousePosition;
370 
371  // Determine if any of the mouse buttons were pressed during this snapshot period, even if they are no longer held.
372  bool leftPressed = false;
373  bool middlePressed = false;
374  bool rightPressed = false;
375  foreach (MouseEvent me in snapshot.MouseEvents)
376  {
377  if (me.Down)
378  {
379  switch (me.MouseButton)
380  {
381  case MouseButton.Left:
382  leftPressed = true;
383  break;
384  case MouseButton.Middle:
385  middlePressed = true;
386  break;
387  case MouseButton.Right:
388  rightPressed = true;
389  break;
390  }
391  }
392  }
393 
394  io.MouseDown[0] = leftPressed || snapshot.IsMouseDown(MouseButton.Left);
395  io.MouseDown[1] = rightPressed || snapshot.IsMouseDown(MouseButton.Right);
396  io.MouseDown[2] = middlePressed || snapshot.IsMouseDown(MouseButton.Middle);
397  io.MousePos = mousePosition;
398  io.MouseWheel = snapshot.WheelDelta;
399 
400  IReadOnlyList<char> keyCharPresses = snapshot.KeyCharPresses;
401  for (int i = 0; i < keyCharPresses.Count; i++)
402  {
403  char c = keyCharPresses[i];
404  io.AddInputCharacter(c);
405  }
406 
407  IReadOnlyList<KeyEvent> keyEvents = snapshot.KeyEvents;
408  for (int i = 0; i < keyEvents.Count; i++)
409  {
410  KeyEvent keyEvent = keyEvents[i];
411  io.KeysDown[(int)keyEvent.Key] = keyEvent.Down;
412  if (keyEvent.Key == Key.ControlLeft)
413  {
414  _controlDown = keyEvent.Down;
415  }
416  if (keyEvent.Key == Key.ShiftLeft)
417  {
418  _shiftDown = keyEvent.Down;
419  }
420  if (keyEvent.Key == Key.AltLeft)
421  {
422  _altDown = keyEvent.Down;
423  }
424  if (keyEvent.Key == Key.WinLeft)
425  {
426  _winKeyDown = keyEvent.Down;
427  }
428  }
429 
430  io.KeyCtrl = _controlDown;
431  io.KeyAlt = _altDown;
432  io.KeyShift = _shiftDown;
433  io.KeySuper = _winKeyDown;
434  }

461  private void RenderImDrawData(ImDrawDataPtr draw_data, GraphicsDevice gd, CommandList cl)
462  {
463  uint vertexOffsetInVertices = 0;
464  uint indexOffsetInElements = 0;
465 
466  if (draw_data.CmdListsCount == 0)
467  {
468  return;
469  }
470 
471  uint totalVBSize = (uint)(draw_data.TotalVtxCount * Unsafe.SizeOf<ImDrawVert>());
472  if (totalVBSize > _vertexBuffer.SizeInBytes)
473  {
474  gd.DisposeWhenIdle(_vertexBuffer);
475  _vertexBuffer = gd.ResourceFactory.CreateBuffer(new BufferDescription((uint)(totalVBSize * 1.5f), BufferUsage.VertexBuffer | BufferUsage.Dynamic));
476  }
477 
478  uint totalIBSize = (uint)(draw_data.TotalIdxCount * sizeof(ushort));
479  if (totalIBSize > _indexBuffer.SizeInBytes)
480  {
481  gd.DisposeWhenIdle(_indexBuffer);
482  _indexBuffer = gd.ResourceFactory.CreateBuffer(new BufferDescription((uint)(totalIBSize * 1.5f), BufferUsage.IndexBuffer | BufferUsage.Dynamic));
483  }
484 
485  for (int i = 0; i < draw_data.CmdListsCount; i++)
486  {
487  ImDrawListPtr cmd_list = draw_data.CmdListsRange[i];
488 
489  cl.UpdateBuffer(
490  _vertexBuffer,
491  vertexOffsetInVertices * (uint)Unsafe.SizeOf<ImDrawVert>(),
492  cmd_list.VtxBuffer.Data,
493  (uint)(cmd_list.VtxBuffer.Size * Unsafe.SizeOf<ImDrawVert>()));
494 
495  cl.UpdateBuffer(
496  _indexBuffer,
497  indexOffsetInElements * sizeof(ushort),
498  cmd_list.IdxBuffer.Data,
499  (uint)(cmd_list.IdxBuffer.Size * sizeof(ushort)));
500 
501  vertexOffsetInVertices += (uint)cmd_list.VtxBuffer.Size;
502  indexOffsetInElements += (uint)cmd_list.IdxBuffer.Size;
503  }
504 
505  // Setup orthographic projection matrix into our constant buffer
506  ImGuiIOPtr io = ImGui.GetIO();
507  Matrix4x4 mvp = Matrix4x4.CreateOrthographicOffCenter(
508  0f,
509  io.DisplaySize.X,
510  io.DisplaySize.Y,
511  0.0f,
512  -1.0f,
513  1.0f);
514 
515  _gd.UpdateBuffer(_projMatrixBuffer, 0, ref mvp);
516 
517  cl.SetVertexBuffer(0, _vertexBuffer);
518  cl.SetIndexBuffer(_indexBuffer, IndexFormat.UInt16);
519  cl.SetPipeline(_pipeline);
520  cl.SetGraphicsResourceSet(0, _mainResourceSet);
521 
522  draw_data.ScaleClipRects(io.DisplayFramebufferScale);
523 
524  // Render command lists
525  int vtx_offset = 0;
526  int idx_offset = 0;
527  for (int n = 0; n < draw_data.CmdListsCount; n++)
528  {
529  ImDrawListPtr cmd_list = draw_data.CmdListsRange[n];
530  for (int cmd_i = 0; cmd_i < cmd_list.CmdBuffer.Size; cmd_i++)
531  {
532  ImDrawCmdPtr pcmd = cmd_list.CmdBuffer[cmd_i];
533  if (pcmd.UserCallback != IntPtr.Zero)
534  {
535  throw new NotImplementedException();
536  }
537  else
538  {
539  if (pcmd.TextureId != IntPtr.Zero)
540  {
541  if (pcmd.TextureId == _fontAtlasID)
542  {
543  cl.SetGraphicsResourceSet(1, _fontTextureResourceSet);
544  }
545  else
546  {
547  cl.SetGraphicsResourceSet(1, GetImageResourceSet(pcmd.TextureId));
548  }
549  }
550 
551  cl.SetScissorRect(
552  0,
553  (uint)pcmd.ClipRect.X,
554  (uint)pcmd.ClipRect.Y,
555  (uint)(pcmd.ClipRect.Z - pcmd.ClipRect.X),
556  (uint)(pcmd.ClipRect.W - pcmd.ClipRect.Y));
557 
558  cl.DrawIndexed(pcmd.ElemCount, 1, (uint)idx_offset, vtx_offset, 0);
559  }
560 
561  idx_offset += (int)pcmd.ElemCount;
562  }
563  vtx_offset += cmd_list.VtxBuffer.Size;
564  }
