80  public static ReflectionObject Create(Type t, MethodBase? creator, params string[] memberNames)
81  {
82  ReflectionDelegateFactory delegateFactory = JsonTypeReflector.ReflectionDelegateFactory;
83 
84  ObjectConstructor<object>? creatorConstructor = null;
85  if (creator != null)
86  {
87  creatorConstructor = delegateFactory.CreateParameterizedConstructor(creator);
88  }
89  else
90  {
91  if (ReflectionUtils.HasDefaultConstructor(t, false))
92  {
93  Func<object> ctor = delegateFactory.CreateDefaultConstructor<object>(t);
94 
95  creatorConstructor = args => ctor();
96  }
97  }
98 
99  ReflectionObject d = new ReflectionObject(creatorConstructor);
100 
101  foreach (string memberName in memberNames)
102  {
103  MemberInfo[] members = t.GetMember(memberName, BindingFlags.Instance | BindingFlags.Public);
104  if (members.Length != 1)
105  {
106  throw new ArgumentException("Expected a single member with the name '{0}'.".FormatWith(CultureInfo.InvariantCulture, memberName));
107  }
108 
109  MemberInfo member = members.Single();
110 
111  ReflectionMember reflectionMember = new ReflectionMember();
112 
113  switch (member.MemberType())
114  {
115  case MemberTypes.Field:
116  case MemberTypes.Property:
117  if (ReflectionUtils.CanReadMemberValue(member, false))
118  {
119  reflectionMember.Getter = delegateFactory.CreateGet<object>(member);
120  }
121 
122  if (ReflectionUtils.CanSetMemberValue(member, false, false))
123  {
124  reflectionMember.Setter = delegateFactory.CreateSet<object>(member);
125  }
126  break;
127  case MemberTypes.Method:
128  MethodInfo method = (MethodInfo)member;
129  if (method.IsPublic)
130  {
131  ParameterInfo[] parameters = method.GetParameters();
132  if (parameters.Length == 0 && method.ReturnType != typeof(void))
133  {
134  MethodCall<object, object?> call = delegateFactory.CreateMethodCall<object>(method);
135  reflectionMember.Getter = target => call(target);
136  }
137  else if (parameters.Length == 1 && method.ReturnType == typeof(void))
138  {
139  MethodCall<object, object?> call = delegateFactory.CreateMethodCall<object>(method);
140  reflectionMember.Setter = (target, arg) => call(target, arg);
141  }
142  }
143  break;
144  default:
145  throw new ArgumentException("Unexpected member type '{0}' for member '{1}'.".FormatWith(CultureInfo.InvariantCulture, member.MemberType(), member.Name));
146  }
147 
148  reflectionMember.MemberType = ReflectionUtils.GetMemberUnderlyingType(member);
149 
150  d.Members[memberName] = reflectionMember;
151  }
152 
153  return d;
154  }
155  }
