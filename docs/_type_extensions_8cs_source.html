488  private static bool TestAccessibility(MemberInfo member, BindingFlags bindingFlags)
489  {
490  if (member is FieldInfo f)
491  {
492  return TestAccessibility(f, bindingFlags);
493  }
494  else if (member is MethodBase m)
495  {
496  return TestAccessibility(m, bindingFlags);
497  }
498  else if (member is PropertyInfo p)

506  private static bool TestAccessibility(FieldInfo member, BindingFlags bindingFlags)
507  {
508  bool visibility = (member.IsPublic && bindingFlags.HasFlag(BindingFlags.Public)) ||
509  (!member.IsPublic && bindingFlags.HasFlag(BindingFlags.NonPublic));
510 
511  bool instance = (member.IsStatic && bindingFlags.HasFlag(BindingFlags.Static)) ||
512  (!member.IsStatic && bindingFlags.HasFlag(BindingFlags.Instance));
513 
514  return visibility && instance;

581  public static bool AssignableToTypeName(this Type type, string fullTypeName, bool searchInterfaces, [NotNullWhen(true)]out Type? match)
582  {
583  Type current = type;
584 
585  while (current != null)
586  {
587  if (string.Equals(current.FullName, fullTypeName, StringComparison.Ordinal))
588  {
589  match = current;
590  return true;
591  }
592 
593  current = current.BaseType();
594  }
595 
596  if (searchInterfaces)
597  {
598  foreach (Type i in type.GetInterfaces())
599  {
600  if (string.Equals(i.Name, fullTypeName, StringComparison.Ordinal))
601  {
602  match = type;
603  return true;
604  }
605  }
606  }
607 
608  match = null;
609  return false;
610  }
