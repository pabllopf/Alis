220 #if HAVE_BIG_INTEGER
221  private static int CompareBigInteger(BigInteger i1, object i2)
222  {
223  int result = i1.CompareTo(ConvertUtils.ToBigInteger(i2));
224 
225  if (result != 0)
226  {
227  return result;
228  }
229 
230  // converting a fractional number to a BigInteger will lose the fraction
231  // check for fraction if result is two numbers are equal
232  if (i2 is decimal d1)
233  {
234  return (0m).CompareTo(Math.Abs(d1 - Math.Truncate(d1)));
235  }
236  else if (i2 is double || i2 is float)
237  {
238  double d = Convert.ToDouble(i2, CultureInfo.InvariantCulture);
239  return (0d).CompareTo(Math.Abs(d - Math.Truncate(d)));
240  }
241 
242  return result;
243  }

246  internal static int Compare(JTokenType valueType, object? objA, object? objB)
247  {
248  if (objA == objB)
249  {
250  return 0;
251  }
252  if (objB == null)
253  {
254  return 1;
255  }
256  if (objA == null)
257  {
258  return -1;
259  }
260 
261  switch (valueType)
262  {
263  case JTokenType.Integer:
264  {
265 #if HAVE_BIG_INTEGER
266  if (objA is BigInteger integerA)
267  {
268  return CompareBigInteger(integerA, objB);
269  }
270  if (objB is BigInteger integerB)
271  {
272  return -CompareBigInteger(integerB, objA);
273  }
274 #endif
275  if (objA is ulong || objB is ulong || objA is decimal || objB is decimal)
276  {
277  return Convert.ToDecimal(objA, CultureInfo.InvariantCulture).CompareTo(Convert.ToDecimal(objB, CultureInfo.InvariantCulture));
278  }
279  else if (objA is float || objB is float || objA is double || objB is double)
280  {
281  return CompareFloat(objA, objB);
282  }
283  else
284  {
285  return Convert.ToInt64(objA, CultureInfo.InvariantCulture).CompareTo(Convert.ToInt64(objB, CultureInfo.InvariantCulture));
286  }
287  }
288  case JTokenType.Float:
289  {
290 #if HAVE_BIG_INTEGER
291  if (objA is BigInteger integerA)
292  {
293  return CompareBigInteger(integerA, objB);
294  }
295  if (objB is BigInteger integerB)
296  {
297  return -CompareBigInteger(integerB, objA);
298  }
299 #endif
300  if (objA is ulong || objB is ulong || objA is decimal || objB is decimal)
301  {
302  return Convert.ToDecimal(objA, CultureInfo.InvariantCulture).CompareTo(Convert.ToDecimal(objB, CultureInfo.InvariantCulture));
303  }
304  return CompareFloat(objA, objB);
305  }
306  case JTokenType.Comment:
307  case JTokenType.String:
308  case JTokenType.Raw:
309  string s1 = Convert.ToString(objA, CultureInfo.InvariantCulture);
310  string s2 = Convert.ToString(objB, CultureInfo.InvariantCulture);
311 
312  return string.CompareOrdinal(s1, s2);
313  case JTokenType.Boolean:
314  bool b1 = Convert.ToBoolean(objA, CultureInfo.InvariantCulture);
315  bool b2 = Convert.ToBoolean(objB, CultureInfo.InvariantCulture);
316 
317  return b1.CompareTo(b2);
318  case JTokenType.Date:
319 #if HAVE_DATE_TIME_OFFSET
320  if (objA is DateTime dateA)
321  {
322 #else
323  DateTime dateA = (DateTime)objA;
324 #endif
325  DateTime dateB;
326 
327 #if HAVE_DATE_TIME_OFFSET
328  if (objB is DateTimeOffset offsetB)
329  {
330  dateB = offsetB.DateTime;
331  }
332  else
333 #endif
334  {
335  dateB = Convert.ToDateTime(objB, CultureInfo.InvariantCulture);
336  }
337 
338  return dateA.CompareTo(dateB);
339 #if HAVE_DATE_TIME_OFFSET
340  }
341  else
342  {
343  DateTimeOffset offsetA = (DateTimeOffset)objA;
344  if (!(objB is DateTimeOffset offsetB))
345  {
346  offsetB = new DateTimeOffset(Convert.ToDateTime(objB, CultureInfo.InvariantCulture));
347  }
348 
349  return offsetA.CompareTo(offsetB);
350  }
351 #endif
352  case JTokenType.Bytes:
353  if (!(objB is byte[] bytesB))
354  {
355  throw new ArgumentException("Object must be of type byte[].");
356  }
357 
358  byte[]? bytesA = objA as byte[];
359  MiscellaneousUtils.Assert(bytesA != null);
360 
361  return MiscellaneousUtils.ByteArrayCompare(bytesA!, bytesB);
362  case JTokenType.Guid:
363  if (!(objB is Guid))
364  {
365  throw new ArgumentException("Object must be of type Guid.");
366  }
367 
368  Guid guid1 = (Guid)objA;
369  Guid guid2 = (Guid)objB;
370 
371  return guid1.CompareTo(guid2);
372  case JTokenType.Uri:
373  Uri? uri2 = objB as Uri;
374  if (uri2 == null)
375  {
376  throw new ArgumentException("Object must be of type Uri.");
377  }
378 
379  Uri uri1 = (Uri)objA;
380 
381  return Comparer<string>.Default.Compare(uri1.ToString(), uri2.ToString());
382  case JTokenType.TimeSpan:
383  if (!(objB is TimeSpan))
384  {
385  throw new ArgumentException("Object must be of type TimeSpan.");
386  }
387 
388  TimeSpan ts1 = (TimeSpan)objA;
389  TimeSpan ts2 = (TimeSpan)objB;
390 
391  return ts1.CompareTo(ts2);
392  default:
393  throw MiscellaneousUtils.CreateArgumentOutOfRangeException(nameof(valueType), valueType, "Unexpected value type: {0}".FormatWith(CultureInfo.InvariantCulture, valueType));
394  }
395  }

397  private static int CompareFloat(object objA, object objB)
398  {
399  double d1 = Convert.ToDouble(objA, CultureInfo.InvariantCulture);
400  double d2 = Convert.ToDouble(objB, CultureInfo.InvariantCulture);
401 
402  // take into account possible floating point errors
403  if (MathUtils.ApproxEquals(d1, d2))
404  {
405  return 0;
406  }
407 
408  return d1.CompareTo(d2);
409  }

416  if (operation == ExpressionType.Add || operation == ExpressionType.AddAssign)

441  result = i1 + i2;

446  return true;

449  result = i1 * i2;
450  return true;
451  case ExpressionType.Divide:
452  case ExpressionType.DivideAssign:
453  result = i1 / i2;
454  return true;
455  }
456  }
457  else
458 #endif

460  {
461  if (objA == null || objB == null)
462  {
463  result = null;
464  return true;
465  }
466 
467  decimal d1 = Convert.ToDecimal(objA, CultureInfo.InvariantCulture);
468  decimal d2 = Convert.ToDecimal(objB, CultureInfo.InvariantCulture);
469 
470  switch (operation)
471  {
472  case ExpressionType.Add:
473  case ExpressionType.AddAssign:
474  result = d1 + d2;
475  return true;
476  case ExpressionType.Subtract:
477  case ExpressionType.SubtractAssign:
478  result = d1 - d2;
479  return true;
480  case ExpressionType.Multiply:
481  case ExpressionType.MultiplyAssign:
482  result = d1 * d2;
483  return true;
484  case ExpressionType.Divide:
485  case ExpressionType.DivideAssign:
486  result = d1 / d2;
487  return true;
488  }
489  }
490  else if (objA is float || objB is float || objA is double || objB is double)
491  {
492  if (objA == null || objB == null)
493  {
494  result = null;
495  return true;
496  }
497 
498  double d1 = Convert.ToDouble(objA, CultureInfo.InvariantCulture);
499  double d2 = Convert.ToDouble(objB, CultureInfo.InvariantCulture);
500 
501  switch (operation)
502  {
503  case ExpressionType.Add:
504  case ExpressionType.AddAssign:
505  result = d1 + d2;
506  return true;
507  case ExpressionType.Subtract:
508  case ExpressionType.SubtractAssign:
509  result = d1 - d2;
510  return true;
511  case ExpressionType.Multiply:
512  case ExpressionType.MultiplyAssign:
513  result = d1 * d2;
514  return true;
515  case ExpressionType.Divide:
516  case ExpressionType.DivideAssign:
517  result = d1 / d2;
518  return true;
519  }
520  }
521  else if (objA is int || objA is uint || objA is long || objA is short || objA is ushort || objA is sbyte || objA is byte ||
522  objB is int || objB is uint || objB is long || objB is short || objB is ushort || objB is sbyte || objB is byte)
523  {
524  if (objA == null || objB == null)
525  {
526  result = null;
527  return true;
528  }
529 
530  long l1 = Convert.ToInt64(objA, CultureInfo.InvariantCulture);
531  long l2 = Convert.ToInt64(objB, CultureInfo.InvariantCulture);
532 
533  switch (operation)
534  {
535  case ExpressionType.Add:
536  case ExpressionType.AddAssign:
537  result = l1 + l2;
538  return true;
539  case ExpressionType.Subtract:
540  case ExpressionType.SubtractAssign:
541  result = l1 - l2;
542  return true;
543  case ExpressionType.Multiply:
544  case ExpressionType.MultiplyAssign:
545  result = l1 * l2;
546  return true;
547  case ExpressionType.Divide:
548  case ExpressionType.DivideAssign:
549  result = l1 / l2;
550  return true;
551  }
552  }
553

602  private static JTokenType GetValueType(JTokenType? current, object? value)
603  {
604  if (value == null)
605  {
606  return JTokenType.Null;
607  }
608 #if HAVE_ADO_NET
609  else if (value == DBNull.Value)
610  {
611  return JTokenType.Null;
612  }
613 #endif
614  else if (value is string)
615  {
616  return GetStringValueType(current);
617  }
618  else if (value is long || value is int || value is short || value is sbyte
619  || value is ulong || value is uint || value is ushort || value is byte)
620  {
621  return JTokenType.Integer;
622  }
623  else if (value is Enum)
624  {
625  return JTokenType.Integer;
626  }
627 #if HAVE_BIG_INTEGER
628  else if (value is BigInteger)
629  {
630  return JTokenType.Integer;
631  }
632 #endif
633  else if (value is double || value is float || value is decimal)
634  {
635  return JTokenType.Float;
636  }
637  else if (value is DateTime)
638  {
639  return JTokenType.Date;
640  }
641 #if HAVE_DATE_TIME_OFFSET
642  else if (value is DateTimeOffset)
643  {
644  return JTokenType.Date;
645  }
646 #endif
647  else if (value is byte[])
648  {
649  return JTokenType.Bytes;
650  }
651  else if (value is bool)
652  {
653  return JTokenType.Boolean;
654  }
655  else if (value is Guid)
656  {
657  return JTokenType.Guid;
658  }
659  else if (value is Uri)
660  {
661  return JTokenType.Uri;
662  }
663  else if (value is TimeSpan)
664  {
665  return JTokenType.TimeSpan;
666  }
667 
668  throw new ArgumentException("Could not determine JSON object type for type {0}.".FormatWith(CultureInfo.InvariantCulture, value.GetType()));

721  public override void WriteTo(JsonWriter writer, params JsonConverter[] converters)
722  {
723  if (converters != null && converters.Length > 0 && _value != null)
724  {
725  JsonConverter? matchingConverter = JsonSerializer.GetMatchingConverter(converters, _value.GetType());
726  if (matchingConverter != null && matchingConverter.CanWrite)
727  {
728  matchingConverter.WriteJson(writer, _value, JsonSerializer.CreateDefault());
729  return;
730  }
731  }
732 
733  switch (_valueType)
734  {
735  case JTokenType.Comment:
736  writer.WriteComment(_value?.ToString());
737  return;
738  case JTokenType.Raw:
739  writer.WriteRawValue(_value?.ToString());
740  return;
741  case JTokenType.Null:
742  writer.WriteNull();
743  return;
744  case JTokenType.Undefined:
745  writer.WriteUndefined();
746  return;
747  case JTokenType.Integer:
748  if (_value is int i)
749  {
750  writer.WriteValue(i);
751  }
752  else if (_value is long l)
753  {
754  writer.WriteValue(l);
755  }
756  else if (_value is ulong ul)
757  {
758  writer.WriteValue(ul);
759  }
760 #if HAVE_BIG_INTEGER
761  else if (_value is BigInteger integer)
762  {
763  writer.WriteValue(integer);
764  }
765 #endif
766  else
767  {
768  writer.WriteValue(Convert.ToInt64(_value, CultureInfo.InvariantCulture));
769  }
770  return;
771  case JTokenType.Float:
772  if (_value is decimal dec)
773  {
774  writer.WriteValue(dec);
775  }
776  else if (_value is double d)
777  {
778  writer.WriteValue(d);
779  }
780  else if (_value is float f)
781  {
782  writer.WriteValue(f);
783  }
784  else
785  {
786  writer.WriteValue(Convert.ToDouble(_value, CultureInfo.InvariantCulture));
787  }
788  return;
789  case JTokenType.String:
790  writer.WriteValue(_value?.ToString());
791  return;
792  case JTokenType.Boolean:
793  writer.WriteValue(Convert.ToBoolean(_value, CultureInfo.InvariantCulture));
794  return;
795  case JTokenType.Date:
796 #if HAVE_DATE_TIME_OFFSET
797  if (_value is DateTimeOffset offset)
798  {
799  writer.WriteValue(offset);
800  }
801  else
802 #endif
803  {
804  writer.WriteValue(Convert.ToDateTime(_value, CultureInfo.InvariantCulture));
805  }
806  return;
807  case JTokenType.Bytes:
808  writer.WriteValue((byte[]?)_value);
809  return;
810  case JTokenType.Guid:
811  writer.WriteValue((_value != null) ? (Guid?)_value : null);
812  return;
813  case JTokenType.TimeSpan:
814  writer.WriteValue((_value != null) ? (TimeSpan?)_value : null);
815  return;
816  case JTokenType.Uri:
817  writer.WriteValue((Uri?)_value);
818  return;
819  }
820 
821  throw MiscellaneousUtils.CreateArgumentOutOfRangeException(nameof(Type), _valueType, "Unexpected token type.");
822  }
