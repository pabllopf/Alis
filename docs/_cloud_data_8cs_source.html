130  {

134  ListFolderResult cloudList = client.Files.ListFolderAsync(pathOfCloud, true).Result;
135  List<Metadata> listFiles = cloudList.Entries.ToList().FindAll(i => i.IsFile).FindAll(i => extensions.Any(j => i.Name.Contains(j)));
136 
137  for (int i = 0; i < listFiles.Count; i++)
138  {
139  string path = pathToDownload + Path.GetDirectoryName(listFiles[i].PathLower);
140  string pathWithFile = pathToDownload + listFiles[i].PathLower;
141  if (!Directory.Exists(path))
142  {
143  Directory.CreateDirectory(path);
144  }
145 
146  if (File.Exists(pathWithFile))
147  {
148  File.Delete(pathWithFile);
149  }
150

152  FileStream file = File.Create(pathWithFile);
153  stream.CopyTo(file);

155  }

157  return;
