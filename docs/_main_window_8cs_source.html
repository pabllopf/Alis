190  private int Update()
191  {
192  Logger.Log("Updating the Main Window.");
193 
194  RgbaFloat clearColor = new RgbaFloat(1.0f, 1.0f, 1.0f, 1.0f);
195  Logger.Log("Define the background color. " + clearColor.ToString() + ".");
196 
197  WidgetManager widgetManager = new WidgetManager(info, imGuiController);
198  Logger.Log("Create widget manager");
199 
200  while (window.Exists)
201  {
202  snapshot = window.PumpEvents();
203  imGuiController.Update(deltaSeconds, snapshot);
204 
205 #if DEBUG
206  //ImGui.ShowDemoWindow();
207 #endif
208  widgetManager.Update();
209 
210  RangeAccessor<System.Numerics.Vector4> colors = ImGui.GetStyle().Colors;
211  if (dark.Equals(colors[(int)ImGuiCol.WindowBg]))
212  {
213  LoadDarkStyle();
214  }
215 
216  if (clasic.Equals(colors[(int)ImGuiCol.WindowBg]))
217  {
218  LoadClasicStyle();
219  }
220 
221 
222  if (light.Equals(colors[(int)ImGuiCol.WindowBg]))
223  {
224  LoadLightStyle();
225  }
226 
227  commandList.Begin();
228  commandList.SetFramebuffer(graphicsDevice.MainSwapchain.Framebuffer);
229  commandList.ClearColorTarget(0, clearColor);
230  imGuiController.Render(graphicsDevice, commandList);
231  commandList.End();
232 
233  graphicsDevice.SubmitCommands(commandList);
234  graphicsDevice.SwapBuffers(graphicsDevice.MainSwapchain);
235  }
236 
237  return 1;
238  }
