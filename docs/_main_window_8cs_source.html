174  private int Update()
175  {
176  Logger.Log("Updating the Main Window.");
177 
178  RgbaFloat clearColor = new RgbaFloat(1.0f, 1.0f, 1.0f, 1.0f);
179  Logger.Log("Define the background color. " + clearColor.ToString() + ".");
180 
181  WidgetManager widgetManager = new WidgetManager(info, imGuiController);
182  Logger.Log("Create widget manager");
183 
184  while (window.Exists)
185  {
186  snapshot = window.PumpEvents();
187  imGuiController.Update(deltaSeconds, snapshot);
188 
189 #if DEBUG
190 #endif
192  widgetManager.Update();
193 
194  commandList.Begin();
195  commandList.SetFramebuffer(graphicsDevice.MainSwapchain.Framebuffer);
196  commandList.ClearColorTarget(0, clearColor);
197  imGuiController.Render(graphicsDevice, commandList);
198  commandList.End();
199 
200  graphicsDevice.SubmitCommands(commandList);
201  graphicsDevice.SwapBuffers(graphicsDevice.MainSwapchain);
202  }
203 
204  return 1;
205  }
