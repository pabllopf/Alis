74  public override bool Read()
75  {
76  if (CurrentState != State.Start)
77  {
78  if (_current == null)
79  {
80  return false;
81  }
82 
83  if (_current is JContainer container && _parent != container)
84  {
85  return ReadInto(container);
86  }
87  else
88  {
89  return ReadOver(_current);
90  }
91  }
92 
93  // The current value could already be the root value if it is a comment
94  if (_current == _root)
95  {
96  return false;
97  }
98 
99  _current = _root;
100  SetToken(_current);
101  return true;
102  }

104  private bool ReadOver(JToken t)
105  {
106  if (t == _root)
107  {
108  return ReadToEnd();
109  }
110 
111  JToken? next = t.Next;
112  if ((next == null || next == t) || t == t.Parent!.Last)
113  {
114  if (t.Parent == null)
115  {
116  return ReadToEnd();
117  }
118 
119  return SetEnd(t.Parent);
120  }
121  else
122  {
123  _current = next;
124  SetToken(_current);
125  return true;
126  }
127  }

136  private JsonToken? GetEndToken(JContainer c)
137  {

185  private void SetToken(JToken token)
186  {
187  switch (token.Type)
188  {
189  case JTokenType.Object:
190  SetToken(JsonToken.StartObject);
191  break;
192  case JTokenType.Array:
193  SetToken(JsonToken.StartArray);
194  break;
195  case JTokenType.Constructor:
196  SetToken(JsonToken.StartConstructor, ((JConstructor)token).Name);
197  break;
198  case JTokenType.Property:
199  SetToken(JsonToken.PropertyName, ((JProperty)token).Name);
200  break;
201  case JTokenType.Comment:
202  SetToken(JsonToken.Comment, ((JValue)token).Value);
203  break;
204  case JTokenType.Integer:
205  SetToken(JsonToken.Integer, ((JValue)token).Value);
206  break;
207  case JTokenType.Float:
208  SetToken(JsonToken.Float, ((JValue)token).Value);
209  break;
210  case JTokenType.String:
211  SetToken(JsonToken.String, ((JValue)token).Value);
212  break;
213  case JTokenType.Boolean:
214  SetToken(JsonToken.Boolean, ((JValue)token).Value);
215  break;
216  case JTokenType.Null:
217  SetToken(JsonToken.Null, ((JValue)token).Value);
218  break;
219  case JTokenType.Undefined:
220  SetToken(JsonToken.Undefined, ((JValue)token).Value);
221  break;
222  case JTokenType.Date:
223  {
224  object? v = ((JValue)token).Value;
225  if (v is DateTime dt)
226  {
227  v = DateTimeUtils.EnsureDateTime(dt, DateTimeZoneHandling);
228  }
229 
230  SetToken(JsonToken.Date, v);
231  break;
232  }
233  case JTokenType.Raw:
234  SetToken(JsonToken.Raw, ((JValue)token).Value);
235  break;
236  case JTokenType.Bytes:
237  SetToken(JsonToken.Bytes, ((JValue)token).Value);
238  break;
239  case JTokenType.Guid:
240  SetToken(JsonToken.String, SafeToString(((JValue)token).Value));
241  break;
242  case JTokenType.Uri:
243  {
244  object? v = ((JValue)token).Value;
245  SetToken(JsonToken.String, v is Uri uri ? uri.OriginalString : SafeToString(v));
246  break;
247  }
248  case JTokenType.TimeSpan:
249  SetToken(JsonToken.String, SafeToString(((JValue)token).Value));
250  break;
251  default:
252  throw MiscellaneousUtils.CreateArgumentOutOfRangeException(nameof(token.Type), token.Type, "Unexpected JTokenType.");
253  }
254  }
