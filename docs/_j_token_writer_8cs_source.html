487  internal override void WriteToken(JsonReader reader, bool writeChildren, bool writeDateConstructorAsDate, bool writeComments)
488  {
489  // cloning the token rather than reading then writing it doesn't lose some type information, e.g. Guid, byte[], etc
490  if (reader is JTokenReader tokenReader && writeChildren && writeDateConstructorAsDate && writeComments)
491  {
492  if (tokenReader.TokenType == JsonToken.None)
493  {
494  if (!tokenReader.Read())
495  {
496  return;
497  }
498  }
499 
500  JToken value = tokenReader.CurrentToken!.CloneToken();
501 
502  if (_parent != null)
503  {
504  _parent.Add(value);
505  _current = _parent.Last;
506 
507  // if the writer was in a property then move out of it and up to its parent object
508  if (_parent.Type == JTokenType.Property)
509  {
510  _parent = _parent.Parent;
511  InternalWriteValue(JsonToken.Null);
512  }
513  }
514  else
515  {
516  _current = value;
517 
518  if (_token == null && _value == null)
519  {
520  _token = value as JContainer;
521  _value = value as JValue;
522  }
523  }
524 
525  tokenReader.Skip();
526  }
527  else
528  {
529  base.WriteToken(reader, writeChildren, writeDateConstructorAsDate, writeComments);
530  }
531  }
532  }
