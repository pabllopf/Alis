37  public class KeyValuePairConverter : JsonConverter
38  {
39  private const string KeyName = "Key";
40  private const string ValueName = "Value";

42  private static readonly ThreadSafeStore<Type, ReflectionObject> ReflectionObjectPerType = new ThreadSafeStore<Type, ReflectionObject>(InitializeReflectionObject);
43 
44  private static ReflectionObject InitializeReflectionObject(Type t)
45  {
46  IList<Type> genericArguments = t.GetGenericArguments();
47  Type keyType = genericArguments[0];
48  Type valueType = genericArguments[1];
49 
50  return ReflectionObject.Create(t, t.GetConstructor(new[] { keyType, valueType }), KeyName, ValueName);
51  }

59  public override void WriteJson(JsonWriter writer, object? value, JsonSerializer serializer)
60  {
61  if (value == null)
62  {
63  writer.WriteNull();
64  return;
65  }
66 
67  ReflectionObject reflectionObject = ReflectionObjectPerType.Get(value.GetType());
68 
69  DefaultContractResolver? resolver = serializer.ContractResolver as DefaultContractResolver;
70 
71  writer.WriteStartObject();
72  writer.WritePropertyName((resolver != null) ? resolver.GetResolvedPropertyName(KeyName) : KeyName);
73  serializer.Serialize(writer, reflectionObject.GetValue(value, KeyName), reflectionObject.GetType(KeyName));
74  writer.WritePropertyName((resolver != null) ? resolver.GetResolvedPropertyName(ValueName) : ValueName);
75  serializer.Serialize(writer, reflectionObject.GetValue(value, ValueName), reflectionObject.GetType(ValueName));
76  writer.WriteEndObject();
77  }

87  public override object? ReadJson(JsonReader reader, Type objectType, object? existingValue, JsonSerializer serializer)
88  {
89  if (reader.TokenType == JsonToken.Null)
90  {
91  if (!ReflectionUtils.IsNullableType(objectType))
92  {
93  throw JsonSerializationException.Create(reader, "Cannot convert null value to KeyValuePair.");
94  }
95 
96  return null;
97  }
98 
99  object? key = null;
100  object? value = null;
101 
102  reader.ReadAndAssert();
103 
104  Type t = ReflectionUtils.IsNullableType(objectType)
105  ? Nullable.GetUnderlyingType(objectType)
106  : objectType;
107 
108  ReflectionObject reflectionObject = ReflectionObjectPerType.Get(t);
109  JsonContract keyContract = serializer.ContractResolver.ResolveContract(reflectionObject.GetType(KeyName));
110  JsonContract valueContract = serializer.ContractResolver.ResolveContract(reflectionObject.GetType(ValueName));
111 
112  while (reader.TokenType == JsonToken.PropertyName)
113  {
114  string propertyName = reader.Value!.ToString();
115  if (string.Equals(propertyName, KeyName, StringComparison.OrdinalIgnoreCase))
116  {
117  reader.ReadForTypeAndAssert(keyContract, false);
118 
119  key = serializer.Deserialize(reader, keyContract.UnderlyingType);
120  }
121  else if (string.Equals(propertyName, ValueName, StringComparison.OrdinalIgnoreCase))
122  {
123  reader.ReadForTypeAndAssert(valueContract, false);
124 
125  value = serializer.Deserialize(reader, valueContract.UnderlyingType);
126  }
127  else
128  {
129  reader.Skip();
130  }
131 
132  reader.ReadAndAssert();
133  }
134 
135  return reflectionObject.Creator!(key, value);
136  }
137 
145  public override bool CanConvert(Type objectType)
146  {
147  Type t = (ReflectionUtils.IsNullableType(objectType))
148  ? Nullable.GetUnderlyingType(objectType)
149  : objectType;
150 
151  if (t.IsValueType() && t.IsGenericType())
152  {
153  return (t.GetGenericTypeDefinition() == typeof(KeyValuePair<,>));
154  }
155 
156  return false;
157  }
158  }
