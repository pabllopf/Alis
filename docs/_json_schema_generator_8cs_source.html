228  private JsonSchema GenerateInternal(Type type, Required valueRequired, bool required)
229  {
230  ValidationUtils.ArgumentNotNull(type, nameof(type));
231 
232  string resolvedId = GetTypeId(type, false);
233  string explicitId = GetTypeId(type, true);
234 
235  if (!StringUtils.IsNullOrEmpty(resolvedId))
236  {
237  JsonSchema resolvedSchema = _resolver.GetSchema(resolvedId);
238  if (resolvedSchema != null)
239  {
240  // resolved schema is not null but referencing member allows nulls
241  // change resolved schema to allow nulls. hacky but what are ya gonna do?
242  if (valueRequired != Required.Always && !HasFlag(resolvedSchema.Type, JsonSchemaType.Null))
243  {
244  resolvedSchema.Type |= JsonSchemaType.Null;
245  }
246  if (required && resolvedSchema.Required != true)
247  {
248  resolvedSchema.Required = true;
249  }
250 
251  return resolvedSchema;
252  }
253  }
254 
255  // test for unresolved circular reference
256  if (_stack.Any(tc => tc.Type == type))
257  {
258  throw new JsonException("Unresolved circular reference for type '{0}'. Explicitly define an Id for the type using a JsonObject/JsonArray attribute or automatically generate a type Id using the UndefinedSchemaIdHandling property.".FormatWith(CultureInfo.InvariantCulture, type));
259  }
260 
261  JsonContract contract = ContractResolver.ResolveContract(type);
262  JsonConverter converter = contract.Converter ?? contract.InternalConverter;
263 
264  Push(new TypeSchema(type, new JsonSchema()));
265 
266  if (explicitId != null)
267  {
268  CurrentSchema.Id = explicitId;
269  }
270 
271  if (required)
272  {
273  CurrentSchema.Required = true;
274  }
275  CurrentSchema.Title = GetTitle(type);
276  CurrentSchema.Description = GetDescription(type);
277 
278  if (converter != null)
279  {
280  CurrentSchema.Type = JsonSchemaType.Any;
281  }
282  else
283  {
284  switch (contract.ContractType)
285  {
286  case JsonContractType.Object:
287  CurrentSchema.Type = AddNullType(JsonSchemaType.Object, valueRequired);
288  CurrentSchema.Id = GetTypeId(type, false);
289  GenerateObjectSchema(type, (JsonObjectContract)contract);
290  break;
291  case JsonContractType.Array:
292  CurrentSchema.Type = AddNullType(JsonSchemaType.Array, valueRequired);
293 
294  CurrentSchema.Id = GetTypeId(type, false);
295 
296  JsonArrayAttribute arrayAttribute = JsonTypeReflector.GetCachedAttribute<JsonArrayAttribute>(type);
297  bool allowNullItem = (arrayAttribute == null || arrayAttribute.AllowNullItems);
298 
299  Type collectionItemType = ReflectionUtils.GetCollectionItemType(type);
300  if (collectionItemType != null)
301  {
302  CurrentSchema.Items = new List<JsonSchema>();
303  CurrentSchema.Items.Add(GenerateInternal(collectionItemType, (!allowNullItem) ? Required.Always : Required.Default, false));
304  }
305  break;
306  case JsonContractType.Primitive:
307  CurrentSchema.Type = GetJsonSchemaType(type, valueRequired);
308 
309  if (CurrentSchema.Type == JsonSchemaType.Integer && type.IsEnum() && !type.IsDefined(typeof(FlagsAttribute), true))
310  {
311  CurrentSchema.Enum = new List<JToken>();
312 
313  EnumInfo enumValues = EnumUtils.GetEnumValuesAndNames(type);
314  for (int i = 0; i < enumValues.Names.Length; i++)
315  {
316  ulong v = enumValues.Values[i];
317  JToken value = JToken.FromObject(Enum.ToObject(type, v));
318 
319  CurrentSchema.Enum.Add(value);
320  }
321  }
322  break;
323  case JsonContractType.String:
324  JsonSchemaType schemaType = (!ReflectionUtils.IsNullable(contract.UnderlyingType))
325  ? JsonSchemaType.String
326  : AddNullType(JsonSchemaType.String, valueRequired);
327 
328  CurrentSchema.Type = schemaType;
329  break;
330  case JsonContractType.Dictionary:
331  CurrentSchema.Type = AddNullType(JsonSchemaType.Object, valueRequired);
332 
333  Type keyType;
334  Type valueType;
335  ReflectionUtils.GetDictionaryKeyValueTypes(type, out keyType, out valueType);
336 
337  if (keyType != null)
338  {
339  JsonContract keyContract = ContractResolver.ResolveContract(keyType);
340 
341  // can be converted to a string
342  if (keyContract.ContractType == JsonContractType.Primitive)
343  {
344  CurrentSchema.AdditionalProperties = GenerateInternal(valueType, Required.Default, false);
345  }
346  }
347  break;
348 #if HAVE_BINARY_SERIALIZATION
349  case JsonContractType.Serializable:
350  CurrentSchema.Type = AddNullType(JsonSchemaType.Object, valueRequired);
351  CurrentSchema.Id = GetTypeId(type, false);
352  GenerateISerializableContract(type, (JsonISerializableContract)contract);
353  break;
354 #endif
355 #if HAVE_DYNAMIC
356  case JsonContractType.Dynamic:
357 #endif
358  case JsonContractType.Linq:
359  CurrentSchema.Type = JsonSchemaType.Any;
360  break;
361  default:
362  throw new JsonException("Unexpected contract type: {0}".FormatWith(CultureInfo.InvariantCulture, contract));
363  }
364  }
365 
366  return Pop().Schema;
367  }

384  private void GenerateObjectSchema(Type type, JsonObjectContract contract)
385  {
386  CurrentSchema.Properties = new Dictionary<string, JsonSchema>();
387  foreach (JsonProperty property in contract.Properties)
388  {
389  if (!property.Ignored)
390  {
391  bool optional = property.NullValueHandling == NullValueHandling.Ignore ||
392  HasFlag(property.DefaultValueHandling.GetValueOrDefault(), DefaultValueHandling.Ignore) ||
393  property.ShouldSerialize != null ||
394  property.GetIsSpecified != null;
395 
396  JsonSchema propertySchema = GenerateInternal(property.PropertyType, property.Required, !optional);
397 
398  if (property.DefaultValue != null)
399  {
400  propertySchema.Default = JToken.FromObject(property.DefaultValue);
401  }
402 
403  CurrentSchema.Properties.Add(property.PropertyName, propertySchema);
404  }
405  }
406 
407  if (type.IsSealed())
408  {
409  CurrentSchema.AllowAdditionalProperties = false;
410  }
411  }

420  internal static bool HasFlag(JsonSchemaType? value, JsonSchemaType flag)
421  {
422  // default value is Any
423  if (value == null)
424  {
425  return true;
426  }
427 
428  bool match = ((value & flag) == flag);
429  if (match)
430  {
431  return true;
432  }
433 
434  // integer is a subset of float
435  if (flag == JsonSchemaType.Integer && (value & JsonSchemaType.Float) == JsonSchemaType.Float)
436  {
437  return true;
438  }
439 
440  return false;
441  }

443  private JsonSchemaType GetJsonSchemaType(Type type, Required valueRequired)
444  {
445  JsonSchemaType schemaType = JsonSchemaType.None;
446  if (valueRequired != Required.Always && ReflectionUtils.IsNullable(type))
447  {
448  schemaType = JsonSchemaType.Null;
449  if (ReflectionUtils.IsNullableType(type))
450  {
451  type = Nullable.GetUnderlyingType(type);
452  }
453  }
454 
455  PrimitiveTypeCode typeCode = ConvertUtils.GetTypeCode(type);
456 
457  switch (typeCode)
458  {
459  case PrimitiveTypeCode.Empty:
460  case PrimitiveTypeCode.Object:
461  return schemaType | JsonSchemaType.String;
462 #if HAVE_DB_NULL_TYPE_CODE
463  case PrimitiveTypeCode.DBNull:
464  return schemaType | JsonSchemaType.Null;
465 #endif
466  case PrimitiveTypeCode.Boolean:
467  return schemaType | JsonSchemaType.Boolean;
468  case PrimitiveTypeCode.Char:
469  return schemaType | JsonSchemaType.String;
470  case PrimitiveTypeCode.SByte:
471  case PrimitiveTypeCode.Byte:
472  case PrimitiveTypeCode.Int16:
473  case PrimitiveTypeCode.UInt16:
474  case PrimitiveTypeCode.Int32:
475  case PrimitiveTypeCode.UInt32:
476  case PrimitiveTypeCode.Int64:
477  case PrimitiveTypeCode.UInt64:
478 #if HAVE_BIG_INTEGER
479  case PrimitiveTypeCode.BigInteger:
480 #endif
481  return schemaType | JsonSchemaType.Integer;
482  case PrimitiveTypeCode.Single:
483  case PrimitiveTypeCode.Double:
484  case PrimitiveTypeCode.Decimal:
485  return schemaType | JsonSchemaType.Float;
486  // convert to string?
487  case PrimitiveTypeCode.DateTime:
488 #if HAVE_DATE_TIME_OFFSET
489  case PrimitiveTypeCode.DateTimeOffset:
490 #endif
491  return schemaType | JsonSchemaType.String;
492  case PrimitiveTypeCode.String:
493  case PrimitiveTypeCode.Uri:
494  case PrimitiveTypeCode.Guid:
495  case PrimitiveTypeCode.TimeSpan:
496  case PrimitiveTypeCode.Bytes:
497  return schemaType | JsonSchemaType.String;
498  default:
499  throw new JsonException("Unexpected type code '{0}' for type '{1}'.".FormatWith(CultureInfo.InvariantCulture, typeCode, type));
500  }
501  }
502  }
