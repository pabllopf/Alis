72  private void WriteTokenInternal(BsonToken t)
73  {
74  switch (t.Type)
75  {
76  case BsonType.Object:
77  {
78  BsonObject value = (BsonObject)t;
79  _writer.Write(value.CalculatedSize);
80  foreach (BsonProperty property in value)
81  {
82  _writer.Write((sbyte)property.Value.Type);
83  WriteString((string)property.Name.Value, property.Name.ByteCount, null);
84  WriteTokenInternal(property.Value);
85  }
86  _writer.Write((byte)0);
87  }
88  break;
89  case BsonType.Array:
90  {
91  BsonArray value = (BsonArray)t;
92  _writer.Write(value.CalculatedSize);
93  ulong index = 0;
94  foreach (BsonToken c in value)
95  {
96  _writer.Write((sbyte)c.Type);
97  WriteString(index.ToString(CultureInfo.InvariantCulture), MathUtils.IntLength(index), null);
98  WriteTokenInternal(c);
99  index++;
100  }
101  _writer.Write((byte)0);
102  }
103  break;
104  case BsonType.Integer:
105  {
106  BsonValue value = (BsonValue)t;
107  _writer.Write(Convert.ToInt32(value.Value, CultureInfo.InvariantCulture));
108  }
109  break;
110  case BsonType.Long:
111  {
112  BsonValue value = (BsonValue)t;
113  _writer.Write(Convert.ToInt64(value.Value, CultureInfo.InvariantCulture));
114  }
115  break;
116  case BsonType.Number:
117  {
118  BsonValue value = (BsonValue)t;
119  _writer.Write(Convert.ToDouble(value.Value, CultureInfo.InvariantCulture));
120  }
121  break;
122  case BsonType.String:
123  {
124  BsonString value = (BsonString)t;
125  WriteString((string)value.Value, value.ByteCount, value.CalculatedSize - 4);
126  }
127  break;
128  case BsonType.Boolean:
129  _writer.Write(t == BsonBoolean.True);
130  break;
131  case BsonType.Null:
132  case BsonType.Undefined:
133  break;
134  case BsonType.Date:
135  {
136  BsonValue value = (BsonValue)t;
137 
138  long ticks = 0;
139 
140  if (value.Value is DateTime dateTime)
141  {
142  if (DateTimeKindHandling == DateTimeKind.Utc)
143  {
144  dateTime = dateTime.ToUniversalTime();
145  }
146  else if (DateTimeKindHandling == DateTimeKind.Local)
147  {
148  dateTime = dateTime.ToLocalTime();
149  }
150 
151  ticks = DateTimeUtils.ConvertDateTimeToJavaScriptTicks(dateTime, false);
152  }
153 #if HAVE_DATE_TIME_OFFSET
154  else
155  {
156  DateTimeOffset dateTimeOffset = (DateTimeOffset)value.Value;
157  ticks = DateTimeUtils.ConvertDateTimeToJavaScriptTicks(dateTimeOffset.UtcDateTime, dateTimeOffset.Offset);
158  }
159 #endif
160 
161  _writer.Write(ticks);
162  }
163  break;
164  case BsonType.Binary:
165  {
166  BsonBinary value = (BsonBinary)t;
167 
168  byte[] data = (byte[])value.Value;
169  _writer.Write(data.Length);
170  _writer.Write((byte)value.BinaryType);
171  _writer.Write(data);
172  }
173  break;
174  case BsonType.Oid:
175  {
176  BsonValue value = (BsonValue)t;
177 
178  byte[] data = (byte[])value.Value;
179  _writer.Write(data);
180  }
181  break;
182  case BsonType.Regex:
183  {
184  BsonRegex value = (BsonRegex)t;
185 
186  WriteString((string)value.Pattern.Value, value.Pattern.ByteCount, null);
187  WriteString((string)value.Options.Value, value.Options.ByteCount, null);
188  }
189  break;
190  default:
191  throw new ArgumentOutOfRangeException(nameof(t), "Unexpected token when writing BSON: {0}".FormatWith(CultureInfo.InvariantCulture, t.Type));
192  }
193  }

195  private void WriteString(string s, int byteCount, int? calculatedlengthPrefix)
196  {
197  if (calculatedlengthPrefix != null)
198  {
199  _writer.Write(calculatedlengthPrefix.GetValueOrDefault());

207  public void WriteUtf8Bytes(string s, int byteCount)
208  {
209  if (s != null)
210  {
211  if (byteCount <= 256)
212  {
213  if (_largeByteBuffer == null)
214  {
215  _largeByteBuffer = new byte[256];
216  }
217 
218  Encoding.GetBytes(s, 0, s.Length, _largeByteBuffer, 0);
219  _writer.Write(_largeByteBuffer, 0, byteCount);
220  }
221  else
222  {
223  byte[] bytes = Encoding.GetBytes(s);
224  _writer.Write(bytes);
225  }
226  }

243  private int CalculateSize(BsonToken t)
244  {
245  switch (t.Type)
246  {
247  case BsonType.Object:
248  {
249  BsonObject value = (BsonObject)t;
250 
251  int bases = 4;
252  foreach (BsonProperty p in value)
253  {
254  int size = 1;
255  size += CalculateSize(p.Name);
256  size += CalculateSize(p.Value);
257 
258  bases += size;
259  }
260  bases += 1;
261  value.CalculatedSize = bases;
262  return bases;
263  }
264  case BsonType.Array:
265  {
266  BsonArray value = (BsonArray)t;
267 
268  int size = 4;
269  ulong index = 0;
270  foreach (BsonToken c in value)
271  {
272  size += 1;
273  size += CalculateSize(MathUtils.IntLength(index));
274  size += CalculateSize(c);
275  index++;
276  }
277  size += 1;
278  value.CalculatedSize = size;
279 
280  return value.CalculatedSize;
281  }
282  case BsonType.Integer:
283  return 4;
284  case BsonType.Long:
285  return 8;
286  case BsonType.Number:
287  return 8;
288  case BsonType.String:
289  {
290  BsonString value = (BsonString)t;
291  string s = (string)value.Value;
292  value.ByteCount = (s != null) ? Encoding.GetByteCount(s) : 0;
293  value.CalculatedSize = CalculateSizeWithLength(value.ByteCount, value.IncludeLength);
294 
295  return value.CalculatedSize;
296  }
297  case BsonType.Boolean:
298  return 1;
299  case BsonType.Null:
300  case BsonType.Undefined:
301  return 0;
302  case BsonType.Date:
303  return 8;
304  case BsonType.Binary:
305  {
306  BsonBinary value = (BsonBinary)t;
307 
308  byte[] data = (byte[])value.Value;
309  value.CalculatedSize = 4 + 1 + data.Length;
310 
311  return value.CalculatedSize;
312  }
313  case BsonType.Oid:
314  return 12;
315  case BsonType.Regex:
316  {
317  BsonRegex value = (BsonRegex)t;
318  int size = 0;
319  size += CalculateSize(value.Pattern);
320  size += CalculateSize(value.Options);
321  value.CalculatedSize = size;
322 
323  return value.CalculatedSize;
324  }
325  default:
326  throw new ArgumentOutOfRangeException(nameof(t), "Unexpected token when writing BSON: {0}".FormatWith(CultureInfo.InvariantCulture, t.Type));
327  }
328  }
