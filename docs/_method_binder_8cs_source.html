60  {
61  0x0001, 0x0FE2, 0x0D54, 0x0FFA,

73  private static bool CanConvertPrimitive(Type from, Type to)
74  {
75  if (from == to)
76  {
77  // same type
78  return true;
79  }
80 
81  int fromMask = 0;
82  int toMask = 0;
83 
84  for (int i = 0; i < PrimitiveTypes.Length; i++)
85  {
86  if (PrimitiveTypes[i] == from)
87  {
88  fromMask = WideningMasks[i];
89  }
90  else if (PrimitiveTypes[i] == to)
91  {
92  toMask = 1 << i;
93  }
94 
95  if (fromMask != 0 && toMask != 0)
96  {
97  break;
98  }
99  }
100 
101  return (fromMask & toMask) != 0;
102  }

112  private static bool FilterParameters(ParameterInfo[] parameters, IList<Type> types, bool enableParamArray)
113  {
114  ValidationUtils.ArgumentNotNull(parameters, nameof(parameters));
115  ValidationUtils.ArgumentNotNull(types, nameof(types));
116 
117  if (parameters.Length == 0)
118  {
119  // fast check for parameterless methods
120  return types.Count == 0;
121  }
122  if (parameters.Length > types.Count)
123  {
124  // not all declared parameters were specified (optional parameters are not supported)
125  return false;
126  }
127 
128  // check if the last parameter is ParamArray
129  Type? paramArrayType = null;
130 
131  if (enableParamArray)
132  {
133  ParameterInfo lastParam = parameters[parameters.Length - 1];
134  if (lastParam.ParameterType.IsArray && lastParam.IsDefined(typeof(ParamArrayAttribute)))
135  {
136  paramArrayType = lastParam.ParameterType.GetElementType();
137  }
138  }
139 
140  if (paramArrayType == null && parameters.Length != types.Count)
141  {
142  // when there's no ParamArray, number of parameters should match
143  return false;
144  }
145 
146  for (int i = 0; i < types.Count; i++)
147  {
148  Type paramType = (paramArrayType != null && i >= parameters.Length - 1) ? paramArrayType : parameters[i].ParameterType;
149 
150  if (paramType == types[i])
151  {
152  // exact match with provided type
153  continue;
154  }
155 
156  if (paramType == typeof(object))
157  {
158  // parameter of type object matches anything
159  continue;
160  }
161 
162  if (paramType.IsPrimitive())
163  {
164  if (!types[i].IsPrimitive() || !CanConvertPrimitive(types[i], paramType))
165  {
166  // primitive parameter can only be assigned from compatible primitive type
167  return false;
168  }
169  }
170  else
171  {
172  if (!paramType.IsAssignableFrom(types[i]))
173  {
174  return false;
175  }
176  }
177  }
178 
179  return true;
180  }

186  private class ParametersMatchComparer : IComparer<ParameterInfo[]>
187  {

191  public ParametersMatchComparer(IList<Type> types, bool enableParamArray)

194 
195  _types = types;
196  _enableParamArray = enableParamArray;
197  }

199  public int Compare(ParameterInfo[] parameters1, ParameterInfo[] parameters2)
200  {
201  ValidationUtils.ArgumentNotNull(parameters1, nameof(parameters1));
202  ValidationUtils.ArgumentNotNull(parameters2, nameof(parameters2));
203 
204  // parameterless method wins
205  if (parameters1.Length == 0)
206  {
207  return -1;
208  }
209  if (parameters2.Length == 0)
210  {
211  return 1;
212  }
213 
214  Type? paramArrayType1 = null, paramArrayType2 = null;
215 
216  if (_enableParamArray)
217  {
218  ParameterInfo lastParam1 = parameters1[parameters1.Length - 1];
219  if (lastParam1.ParameterType.IsArray && lastParam1.IsDefined(typeof(ParamArrayAttribute)))
220  {
221  paramArrayType1 = lastParam1.ParameterType.GetElementType();
222  }
223 
224  ParameterInfo lastParam2 = parameters2[parameters2.Length - 1];
225  if (lastParam2.ParameterType.IsArray && lastParam2.IsDefined(typeof(ParamArrayAttribute)))
226  {
227  paramArrayType2 = lastParam2.ParameterType.GetElementType();
228  }
229 
230  // A method using params always loses to one not using params
231  if (paramArrayType1 != null && paramArrayType2 == null)
232  {
233  return 1;
234  }
235  if (paramArrayType2 != null && paramArrayType1 == null)
236  {
237  return -1;
238  }
239  }
240 
241  for (int i = 0; i < _types.Count; i++)
242  {
243  Type type1 = (paramArrayType1 != null && i >= parameters1.Length - 1) ? paramArrayType1 : parameters1[i].ParameterType;
244  Type type2 = (paramArrayType2 != null && i >= parameters2.Length - 1) ? paramArrayType2 : parameters2[i].ParameterType;
245 
246  if (type1 == type2)
247  {
248  // exact match between parameter types doesn't change score
249  continue;
250  }
251 
252  // exact match with source type decides winner immediately
253  if (type1 == _types[i])
254  {
255  return -1;
256  }
257  if (type2 == _types[i])
258  {
259  return 1;
260  }
261 
262  int r = ChooseMorePreciseType(type1, type2);
263  if (r != 0)
264  {
265  // winner decided
266  return r;
267  }
268  }
269 
270  return 0;
271  }
272 
273  private static int ChooseMorePreciseType(Type type1, Type type2)
274  {
275  if (type1.IsByRef || type2.IsByRef)
276  {
277  if (type1.IsByRef && type2.IsByRef)
278  {
279  type1 = type1.GetElementType();
280  type2 = type2.GetElementType();
281  }
282  else if (type1.IsByRef)
283  {
284  type1 = type1.GetElementType();
285  if (type1 == type2)
286  {
287  return 1;
288  }
289  }
290  else
291  {
292  type2 = type2.GetElementType();
293  if (type2 == type1)
294  {
295  return -1;
296  }
297  }
298  }
299 
300  bool c1FromC2, c2FromC1;
301 
302  if (type1.IsPrimitive() && type2.IsPrimitive())
303  {
304  c1FromC2 = CanConvertPrimitive(type2, type1);
305  c2FromC1 = CanConvertPrimitive(type1, type2);
306  }
307  else
308  {
309  c1FromC2 = type1.IsAssignableFrom(type2);
310  c2FromC1 = type2.IsAssignableFrom(type1);
311  }
312 
313  if (c1FromC2 == c2FromC1)
314  {
315  return 0;
316  }
317 
318  return c1FromC2 ? 1 : -1;
319  }

329  public static TMethod SelectMethod<TMethod>(IEnumerable<TMethod> candidates, IList<Type> types) where TMethod : MethodBase
330  {
331  ValidationUtils.ArgumentNotNull(candidates, nameof(candidates));
332  ValidationUtils.ArgumentNotNull(types, nameof(types));
333 
334  // ParamArrays are not supported by ReflectionDelegateFactory
335  // They will be treated like ordinary array arguments
336  const bool enableParamArray = false;
337 
338  return candidates
339  .Where(m => FilterParameters(m.GetParameters(), types, enableParamArray))
340  .OrderBy(m => m.GetParameters(), new ParametersMatchComparer(types, enableParamArray))
341  .FirstOrDefault();
342  }
