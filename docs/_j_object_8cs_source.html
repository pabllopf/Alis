149  internal override void ValidateToken(JToken o, JToken? existing)
150  {
151  ValidationUtils.ArgumentNotNull(o, nameof(o));
152 
153  if (o.Type != JTokenType.Property)
154  {
155  throw new ArgumentException("Can not add {0} to {1}.".FormatWith(CultureInfo.InvariantCulture, o.GetType(), GetType()));
156  }

158  JProperty newProperty = (JProperty)o;
159 
160  if (existing != null)
161  {
162  JProperty existingProperty = (JProperty)existing;
163 
164  if (newProperty.Name == existingProperty.Name)
165  {
166  return;
167  }
168  }
169

171  {
172  throw new ArgumentException("Can not add property {0} to {1}. Property with the same name already exists on object.".FormatWith(CultureInfo.InvariantCulture, newProperty.Name, GetType()));
173  }

176  internal override void MergeItem(object content, JsonMergeSettings? settings)
177  {
178  if (!(content is JObject o))
179  {
180  return;
181  }
182 
183  foreach (KeyValuePair<string, JToken?> contentItem in o)
184  {
185  JProperty? existingProperty = Property(contentItem.Key, settings?.PropertyNameComparison ?? StringComparison.Ordinal);
186 
187  if (existingProperty == null)
188  {
189  Add(contentItem.Key, contentItem.Value);
190  }
191  else if (contentItem.Value != null)
192  {
193  if (!(existingProperty.Value is JContainer existingContainer) || existingContainer.Type != contentItem.Value.Type)
194  {
195  if (!IsNull(contentItem.Value) || settings?.MergeNullValueHandling == MergeNullValueHandling.Merge)
196  {
197  existingProperty.Value = contentItem.Value;
198  }
199  }
200  else
201  {
202  existingContainer.Merge(contentItem.Value, settings);
203  }
204  }
205  }

407  public new static JObject Load(JsonReader reader, JsonLoadSettings? settings)
408  {
409  ValidationUtils.ArgumentNotNull(reader, nameof(reader));
410 
411  if (reader.TokenType == JsonToken.None)
412  {
413  if (!reader.Read())
414  {
415  throw JsonReaderException.Create(reader, "Error reading JObject from JsonReader.");
416  }
417  }
418 
419  reader.MoveToContent();
420 
421  if (reader.TokenType != JsonToken.StartObject)
422  {
423  throw JsonReaderException.Create(reader, "Error reading JObject from JsonReader. Current JsonReader item is not an object: {0}".FormatWith(CultureInfo.InvariantCulture, reader.TokenType));
424  }
425 
426  JObject o = new JObject();
427  o.SetLineInfo(reader as IJsonLineInfo, settings);
428 
429  o.ReadTokenFrom(reader, settings);
430 
431  return o;
432  }

656  void ICollection<KeyValuePair<string, JToken?>>.CopyTo(KeyValuePair<string, JToken?>[] array, int arrayIndex)
657  {
658  if (array == null)
659  {
660  throw new ArgumentNullException(nameof(array));
661  }
662  if (arrayIndex < 0)
663  {
664  throw new ArgumentOutOfRangeException(nameof(arrayIndex), "arrayIndex is less than 0.");
665  }
666  if (arrayIndex >= array.Length && arrayIndex != 0)
667  {
668  throw new ArgumentException("arrayIndex is equal to or greater than the length of array.");
669  }
670  if (Count > array.Length - arrayIndex)
671  {
672  throw new ArgumentException("The number of elements in the source JObject is greater than the available space from arrayIndex to the end of the destination array.");
673  }
674 
675  int index = 0;
676  foreach (JProperty property in _properties)
677  {
678  array[arrayIndex + index] = new KeyValuePair<string, JToken?>(property.Name, property.Value);
679  index++;
680  }
