276  public void Awake()
277  {
278  if (limitJustOneProcessor)
279  {
280  Span<GameObject> span = gameObjects.Span;
281  for (int i = 0; i < span.Length; i++)
282  {
283  if (span[i] != null && span[i].IsActive)
284  {
285  span[i].Awake();
286  }
287  }
288  }
289  else
290  {
291  for (int i = 0; i < Processor; i++)
292  {
293  int init = i * SizeBlocksOfTasks;
294  int end = ((i + 1) * SizeBlocksOfTasks) > lastIndex ? lastIndex : ((i + 1) * SizeBlocksOfTasks);
295 
296  if (init <= end)
297  {
298  tasks[i] = Task.Run(() =>
299  {
300  Span<GameObject> span = gameObjects.Span;
301  for (int j = init; j < end; j++)
302  {
303  if (span[j] != null)
304  {
305  span[j].Awake();
306  }
307  }
308  });
309  }
310  }
311 
312  Task.WaitAll(tasks);
313  }
