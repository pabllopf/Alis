334  private void Push(JsonContainerType value)
335  {
336  UpdateScopeWithFinishedValue();
337 
338  if (_currentPosition.Type == JsonContainerType.None)
339  {
340  _currentPosition = new JsonPosition(value);
341  }
342  else
343  {
344  if (_stack == null)
345  {
346  _stack = new List<JsonPosition>();
347  }
348 
349  _stack.Add(_currentPosition);
350  _currentPosition = new JsonPosition(value);
351 
352  // this is a little hacky because Depth increases when first property/value is written but only testing here is faster/simpler
353  if (_maxDepth != null && Depth + 1 > _maxDepth && !_hasExceededMaxDepth)
354  {
355  _hasExceededMaxDepth = true;
356  throw JsonReaderException.Create(this, "The reader's MaxDepth of {0} has been exceeded.".FormatWith(CultureInfo.InvariantCulture, _maxDepth));
357  }
358  }
359  }
360 
361  private JsonContainerType Pop()
362  {
363  JsonPosition oldPosition;
364  if (_stack != null && _stack.Count > 0)
365  {
366  oldPosition = _currentPosition;
367  _currentPosition = _stack[_stack.Count - 1];
368  _stack.RemoveAt(_stack.Count - 1);
369  }
370  else
371  {
372  oldPosition = _currentPosition;
373  _currentPosition = new JsonPosition();
374  }
375 
376  if (_maxDepth != null && Depth <= _maxDepth)
377  {
378  _hasExceededMaxDepth = false;
379  }
380 
381  return oldPosition.Type;
382  }

399  public virtual int? ReadAsInt32()
400  {
401  JsonToken t = GetContentToken();
402 
403  switch (t)
404  {
405  case JsonToken.None:
406  case JsonToken.Null:
407  case JsonToken.EndArray:
408  return null;
409  case JsonToken.Integer:
410  case JsonToken.Float:
411  object v = Value!;
412  if (v is int i)
413  {
414  return i;
415  }
416 
417 #if HAVE_BIG_INTEGER
418  if (v is BigInteger value)
419  {
420  i = (int)value;
421  }
422  else
423 #endif
424  {
425  try
426  {
427  i = Convert.ToInt32(v, CultureInfo.InvariantCulture);
428  }
429  catch (Exception ex)
430  {
431  // handle error for large integer overflow exceptions
432  throw JsonReaderException.Create(this, "Could not convert to integer: {0}.".FormatWith(CultureInfo.InvariantCulture, v), ex);
433  }
434  }
435 
436  SetToken(JsonToken.Integer, i, false);
437  return i;
438  case JsonToken.String:
439  string? s = (string?)Value;
440  return ReadInt32String(s);
441  }
442 
443  throw JsonReaderException.Create(this, "Error reading integer. Unexpected token: {0}.".FormatWith(CultureInfo.InvariantCulture, t));
444  }

511  public virtual byte[]? ReadAsBytes()
512  {
513  JsonToken t = GetContentToken();
514 
515  switch (t)
516  {
517  case JsonToken.StartObject:
518  {
519  ReadIntoWrappedTypeObject();
520 
521  byte[]? data = ReadAsBytes();
522  ReaderReadAndAssert();
523 
524  if (TokenType != JsonToken.EndObject)
525  {
526  throw JsonReaderException.Create(this, "Error reading bytes. Unexpected token: {0}.".FormatWith(CultureInfo.InvariantCulture, TokenType));
527  }
528 
529  SetToken(JsonToken.Bytes, data, false);
530  return data;
531  }
532  case JsonToken.String:
533  {
534  // attempt to convert possible base 64 or GUID string to bytes
535  // GUID has to have format 00000000-0000-0000-0000-000000000000
536  string s = (string)Value!;
537 
538  byte[] data;
539 
540  if (s.Length == 0)
541  {
542  data = CollectionUtils.ArrayEmpty<byte>();
543  }
544  else if (ConvertUtils.TryConvertGuid(s, out Guid g1))
545  {
546  data = g1.ToByteArray();
547  }
548  else
549  {
550  data = Convert.FromBase64String(s);
551  }
552 
553  SetToken(JsonToken.Bytes, data, false);
554  return data;
555  }
556  case JsonToken.None:
557  case JsonToken.Null:
558  case JsonToken.EndArray:
559  return null;
560  case JsonToken.Bytes:
561  if (Value is Guid g2)
562  {
563  byte[] data = g2.ToByteArray();
564  SetToken(JsonToken.Bytes, data, false);
565  return data;
566  }
567 
568  return (byte[]?)Value;
569  case JsonToken.StartArray:
570  return ReadArrayIntoByteArray();
571  }
572 
573  throw JsonReaderException.Create(this, "Error reading bytes. Unexpected token: {0}.".FormatWith(CultureInfo.InvariantCulture, t));
574  }

576  internal byte[] ReadArrayIntoByteArray()
577  {
578  List<byte> buffer = new List<byte>();
579 
580  while (true)
581  {
582  if (!Read())
583  {
584  SetToken(JsonToken.None);
585  }
586 
587  if (ReadArrayElementIntoByteArrayReportDone(buffer))
588  {
589  byte[] d = buffer.ToArray();
590  SetToken(JsonToken.Bytes, d, false);
591  return d;
592  }
593  }
594  }

596  private bool ReadArrayElementIntoByteArrayReportDone(List<byte> buffer)
597  {
598  switch (TokenType)
599  {
600  case JsonToken.None:
601  throw JsonReaderException.Create(this, "Unexpected end when reading bytes.");
602  case JsonToken.Integer:
603  buffer.Add(Convert.ToByte(Value, CultureInfo.InvariantCulture));
604  return false;
605  case JsonToken.EndArray:
606  return true;
607  case JsonToken.Comment:
608  return false;
609  default:
610  throw JsonReaderException.Create(this, "Unexpected token when reading bytes: {0}.".FormatWith(CultureInfo.InvariantCulture, TokenType));
611  }

998  protected void SetToken(JsonToken newToken, object? value, bool updateIndex)
999  {
1000  _tokenType = newToken;
1001  _value = value;
1002 
1003  switch (newToken)
1004  {
1005  case JsonToken.StartObject:
1006  _currentState = State.ObjectStart;
1007  Push(JsonContainerType.Object);
1008  break;
1009  case JsonToken.StartArray:
1010  _currentState = State.ArrayStart;
1011  Push(JsonContainerType.Array);
1012  break;
1013  case JsonToken.StartConstructor:
1014  _currentState = State.ConstructorStart;
1015  Push(JsonContainerType.Constructor);
