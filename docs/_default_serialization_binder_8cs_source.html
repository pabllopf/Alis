56  private Type GetTypeFromTypeNameKey(StructMultiKey<string?, string> typeNameKey)
57  {
58  string? assemblyName = typeNameKey.Value1;
59  string typeName = typeNameKey.Value2;
60 
61  if (assemblyName != null)
62  {
63  Assembly assembly;
64 
65 #if !(DOTNET || PORTABLE40 || PORTABLE)
66  // look, I don't like using obsolete methods as much as you do but this is the only way
67  // Assembly.Load won't check the GAC for a partial name
68 #pragma warning disable 618,612
69  assembly = Assembly.LoadWithPartialName(assemblyName);
70 #pragma warning restore 618,612
71 #elif DOTNET || PORTABLE
72  assembly = Assembly.Load(new AssemblyName(assemblyName));
73 #else
74  assembly = Assembly.Load(assemblyName);
75 #endif
76 
77 #if HAVE_APP_DOMAIN
78  if (assembly == null)
79  {
80  // will find assemblies loaded with Assembly.LoadFile outside of the main directory
81  Assembly[] loadedAssemblies = AppDomain.CurrentDomain.GetAssemblies();
82  foreach (Assembly a in loadedAssemblies)
83  {
84  // check for both full name or partial name match
85  if (a.FullName == assemblyName || a.GetName().Name == assemblyName)
86  {
87  assembly = a;
88  break;
89  }
90  }
91  }
92 #endif
93 
94  if (assembly == null)
95  {
96  throw new JsonSerializationException("Could not load assembly '{0}'.".FormatWith(CultureInfo.InvariantCulture, assemblyName));
97  }
98 
99  Type? type = assembly.GetType(typeName);
100  if (type == null)
101  {
102  // if generic type, try manually parsing the type arguments for the case of dynamically loaded assemblies
103  // example generic typeName format: System.Collections.Generic.Dictionary`2[[System.String, mscorlib, Version=2.0.0.0, Culture=neutral, PublicKeyToken=b77a5c561934e089],[System.String, mscorlib, Version=2.0.0.0, Culture=neutral, PublicKeyToken=b77a5c561934e089]]
104  if (typeName.IndexOf('`') >= 0)
105  {
106  try
107  {
108  type = GetGenericTypeFromTypeName(typeName, assembly);
109  }
110  catch (Exception ex)
111  {
112  throw new JsonSerializationException("Could not find type '{0}' in assembly '{1}'.".FormatWith(CultureInfo.InvariantCulture, typeName, assembly.FullName), ex);
113  }
114  }
115 
116  if (type == null)
117  {
118  throw new JsonSerializationException("Could not find type '{0}' in assembly '{1}'.".FormatWith(CultureInfo.InvariantCulture, typeName, assembly.FullName));
119  }
120  }
121 
122  return type;
123  }
124  else
125  {
126  return Type.GetType(typeName);
127  }
128  }

130  private Type? GetGenericTypeFromTypeName(string typeName, Assembly assembly)
131  {
132  Type? type = null;
133  int openBracketIndex = typeName.IndexOf('[');
134  if (openBracketIndex >= 0)
135  {
136  string genericTypeDefName = typeName.Substring(0, openBracketIndex);
137  Type genericTypeDef = assembly.GetType(genericTypeDefName);
138  if (genericTypeDef != null)
139  {
140  List<Type> genericTypeArguments = new List<Type>();
141  int scope = 0;
142  int typeArgStartIndex = 0;
143  int endIndex = typeName.Length - 1;
144  for (int i = openBracketIndex + 1; i < endIndex; ++i)
145  {
146  char current = typeName[i];
147  switch (current)
148  {
149  case '[':
150  if (scope == 0)
151  {
152  typeArgStartIndex = i + 1;
153  }
154  ++scope;
155  break;
156  case ']':
157  --scope;
158  if (scope == 0)
159  {
160  string typeArgAssemblyQualifiedName = typeName.Substring(typeArgStartIndex, i - typeArgStartIndex);
161 
162  StructMultiKey<string?, string> typeNameKey = ReflectionUtils.SplitFullyQualifiedTypeName(typeArgAssemblyQualifiedName);
163  genericTypeArguments.Add(GetTypeByName(typeNameKey));
164  }
165  break;
166  }
167  }
168 
169  type = genericTypeDef.MakeGenericType(genericTypeArguments.ToArray());
170  }
171  }
172 
173  return type;
174  }
