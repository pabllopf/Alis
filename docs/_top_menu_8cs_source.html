391  private static void DirectoryCopy(string sourceDirName, string destDirName, bool copySubDirs)
392  {
393  // Get the subdirectories for the specified directory.
394  DirectoryInfo dir = new DirectoryInfo(sourceDirName);
395 
396  if (!dir.Exists)
397  {
398  throw new DirectoryNotFoundException(
399  "Source directory does not exist or could not be found: "
400  + sourceDirName);
401  }
402 
403  DirectoryInfo[] dirs = dir.GetDirectories();
404 
405  // If the destination directory doesn't exist, create it.
406  Directory.CreateDirectory(destDirName);
407 
408  // Get the files in the directory and copy them to the new location.
409  FileInfo[] files = dir.GetFiles();
410  foreach (FileInfo file in files)
411  {
412  string tempPath = Path.Combine(destDirName, file.Name);
413  file.CopyTo(tempPath, true);
414  }
415 
416  // If copying subdirectories, copy them and their contents to new location.
417  if (copySubDirs)
418  {
419  foreach (DirectoryInfo subdir in dirs)
420  {
421  string tempPath = Path.Combine(destDirName, subdir.Name);
422  DirectoryCopy(subdir.FullName, tempPath, copySubDirs);
423  }
424  }
425  }

428  private void RunCommand(string message, string fileName, string commandBuild, string WorkingDirectory)
429  {
430  BottomMenu.Current.Loading(true, message);
431 
432  System.Diagnostics.Process buildProcess = new System.Diagnostics.Process();
433 
434  buildProcess.StartInfo.FileName = fileName;
435  buildProcess.StartInfo.WorkingDirectory = WorkingDirectory;
436  buildProcess.StartInfo.CreateNoWindow = false;
437  buildProcess.StartInfo.RedirectStandardInput = true;
438  buildProcess.StartInfo.RedirectStandardOutput = true;
439  buildProcess.StartInfo.UseShellExecute = false;
440  buildProcess.Start();
441  buildProcess.StandardInput.WriteLine(commandBuild);
442  buildProcess.StandardInput.Flush();
443  buildProcess.StandardInput.Close();
444  buildProcess.WaitForExit();
445 
446  string[] sentences = buildProcess.StandardOutput.ReadToEnd().Split("\n");
447  for (int i = 0; i < sentences.Length; i++)
448  {
449  if (sentences[i].Contains("warning"))
450  {
451  Console.Current.Warning(sentences[i]);
452  continue;
453  }
454 
455  if (sentences[i].Contains("error") || sentences[i].Contains("failed") || sentences[i].Contains("exception"))
456  {
457  Console.Current.Error(sentences[i]);
458  continue;
459  }
460 
461  Console.Current.Log(sentences[i], false);
462  }
463 
464  BottomMenu.Current.Loading(false, "");
465  }
