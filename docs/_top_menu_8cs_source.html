395  private static void DirectoryCopy(string sourceDirName, string destDirName, bool copySubDirs)
396  {
397  // Get the subdirectories for the specified directory.
398  DirectoryInfo dir = new DirectoryInfo(sourceDirName);
399 
400  if (!dir.Exists)
401  {
402  throw new DirectoryNotFoundException(
403  "Source directory does not exist or could not be found: "
404  + sourceDirName);
405  }
406 
407  DirectoryInfo[] dirs = dir.GetDirectories();
408 
409  // If the destination directory doesn't exist, create it.
410  Directory.CreateDirectory(destDirName);
411 
412  // Get the files in the directory and copy them to the new location.
413  FileInfo[] files = dir.GetFiles();
414  foreach (FileInfo file in files)
415  {
416  string tempPath = Path.Combine(destDirName, file.Name);
417  file.CopyTo(tempPath, true);
418  }
419 
420  // If copying subdirectories, copy them and their contents to new location.
421  if (copySubDirs)
422  {
423  foreach (DirectoryInfo subdir in dirs)
424  {
425  string tempPath = Path.Combine(destDirName, subdir.Name);
426  DirectoryCopy(subdir.FullName, tempPath, copySubDirs);
427  }
428  }
429  }

432  private void RunCommand(string message, string fileName, string commandBuild, string WorkingDirectory)
433  {
434  BottomMenu.Current.Loading(true, message);
435 
436  System.Diagnostics.Process buildProcess = new System.Diagnostics.Process();
437 
438  buildProcess.StartInfo.FileName = fileName;
439  buildProcess.StartInfo.WorkingDirectory = WorkingDirectory;
440  buildProcess.StartInfo.CreateNoWindow = false;
441  buildProcess.StartInfo.RedirectStandardInput = true;
442  buildProcess.StartInfo.RedirectStandardOutput = true;
443  buildProcess.StartInfo.UseShellExecute = false;
444  buildProcess.Start();
445  buildProcess.StandardInput.WriteLine(commandBuild);
446  buildProcess.StandardInput.Flush();
447  buildProcess.StandardInput.Close();
448  buildProcess.WaitForExit();
449 
450  string[] sentences = buildProcess.StandardOutput.ReadToEnd().Split("\n");
451  for (int i = 0; i < sentences.Length; i++)
452  {
453  if (sentences[i].Contains("warning"))
454  {
455  Console.Current.Warning(sentences[i]);
456  continue;
457  }
458 
459  if (sentences[i].Contains("error") || sentences[i].Contains("failed") || sentences[i].Contains("exception"))
460  {
461  Console.Current.Error(sentences[i]);
462  continue;
463  }
464 
465  Console.Current.Log(sentences[i], false);
466  }
467 
468  BottomMenu.Current.Loading(false, "");
469  }
