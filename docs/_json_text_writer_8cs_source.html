321  protected override void WriteIndent()
322  {
323  // levels of indentation multiplied by the indent count
324  int currentIndentCount = Top * _indentation;
325 
326  int newLineLen = SetIndentChars();
327 
328  _writer.Write(_indentChars, 0, newLineLen + Math.Min(currentIndentCount, IndentCharBufferSize));
329 
330  while ((currentIndentCount -= IndentCharBufferSize) > 0)
331  {
332  _writer.Write(_indentChars, newLineLen, Math.Min(currentIndentCount, IndentCharBufferSize));
333  }
334  }
335 
336  private int SetIndentChars()
337  {
338  // Set _indentChars to be a newline followed by IndentCharBufferSize indent characters.
339  string writerNewLine = _writer.NewLine;
340  int newLineLen = writerNewLine.Length;
341  bool match = _indentChars != null && _indentChars.Length == IndentCharBufferSize + newLineLen;
342  if (match)
343  {
344  for (int i = 0; i != newLineLen; ++i)
345  {
346  if (writerNewLine[i] != _indentChars![i])
347  {
348  match = false;
349  break;
350  }
351  }
352  }
353 
354  if (!match)
355  {
356  // If we're here, either _indentChars hasn't been set yet, or _writer.NewLine
357  // has been changed, or _indentChar has been changed.
358  _indentChars = (writerNewLine + new string(_indentChar, IndentCharBufferSize)).ToCharArray();
359  }
360 
361  return newLineLen;
362  }

836  private int WriteNumberToBuffer(ulong value, bool negative)
837  {
838  if (value <= uint.MaxValue)

840  // avoid the 64 bit division if possible
841  return WriteNumberToBuffer((uint)value, negative);
842  }
