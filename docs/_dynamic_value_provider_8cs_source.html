52  public DynamicValueProvider(MemberInfo memberInfo)
53  {

55  _memberInfo = memberInfo;
56  }

63  public void SetValue(object target, object? value)
64  {
65  try
66  {
67  if (_setter == null)
68  {
69  _setter = DynamicReflectionDelegateFactory.Instance.CreateSet<object>(_memberInfo);
70  }
71 
72 #if DEBUG
73  // dynamic method doesn't check whether the type is 'legal' to set
74  // add this check for unit tests
75  if (value == null)
76  {
77  if (!ReflectionUtils.IsNullable(ReflectionUtils.GetMemberUnderlyingType(_memberInfo)))
78  {
79  throw new JsonSerializationException("Incompatible value. Cannot set {0} to null.".FormatWith(CultureInfo.InvariantCulture, _memberInfo));
80  }
81  }
82  else if (!ReflectionUtils.GetMemberUnderlyingType(_memberInfo).IsAssignableFrom(value.GetType()))
83  {
84  throw new JsonSerializationException("Incompatible value. Cannot set {0} to type {1}.".FormatWith(CultureInfo.InvariantCulture, _memberInfo, value.GetType()));
85  }
86 #endif
87 
88  _setter(target, value);
89  }
90  catch (Exception ex)
91  {
92  throw new JsonSerializationException("Error setting value to '{0}' on '{1}'.".FormatWith(CultureInfo.InvariantCulture, _memberInfo.Name, target.GetType()), ex);
93  }
94  }

101  public object? GetValue(object target)
102  {
103  try
104  {
105  if (_getter == null)
106  {
107  _getter = DynamicReflectionDelegateFactory.Instance.CreateGet<object>(_memberInfo);
108  }
109 
110  return _getter(target);
111  }
112  catch (Exception ex)
113  {
114  throw new JsonSerializationException("Error getting value from '{0}' on '{1}'.".FormatWith(CultureInfo.InvariantCulture, _memberInfo.Name, target.GetType()), ex);
115  }
116  }
117  }
