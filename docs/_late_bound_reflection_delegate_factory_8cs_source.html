42  public override ObjectConstructor<object> CreateParameterizedConstructor(MethodBase method)
43  {
44  ValidationUtils.ArgumentNotNull(method, nameof(method));
45 
46  if (method is ConstructorInfo c)
47  {
48  // don't convert to method group to avoid medium trust issues
49  // https://github.com/JamesNK/Newtonsoft.Json/issues/476
50  return a => c.Invoke(a);
51  }
52 
53  return a => method.Invoke(null, a);
54  }
55 
56  public override MethodCall<T, object?> CreateMethodCall<T>(MethodBase method)
57  {
58  ValidationUtils.ArgumentNotNull(method, nameof(method));
59 
60  if (method is ConstructorInfo c)
61  {
62  return (o, a) => c.Invoke(a);
63  }
64 
65  return (o, a) => method.Invoke(o, a);
66  }

68  public override Func<T> CreateDefaultConstructor<T>(Type type)
69  {
70  ValidationUtils.ArgumentNotNull(type, nameof(type));
71 
72  if (type.IsValueType())
73  {
74  return () => (T)Activator.CreateInstance(type);
75  }
76 
77  ConstructorInfo constructorInfo = ReflectionUtils.GetDefaultConstructor(type, true);
78 
79  return () => (T)constructorInfo.Invoke(null);
80  }
