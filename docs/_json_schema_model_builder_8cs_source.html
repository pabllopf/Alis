57  public JsonSchemaNode AddSchema(JsonSchemaNode existingNode, JsonSchema schema)
58  {
59  string newId;
60  if (existingNode != null)
61  {
62  if (existingNode.Schemas.Contains(schema))
63  {
64  return existingNode;
65  }
66 
67  newId = JsonSchemaNode.GetId(existingNode.Schemas.Union(new[] { schema }));
68  }
69  else
70  {
71  newId = JsonSchemaNode.GetId(new[] { schema });
72  }
73 
74  if (_nodes.Contains(newId))
75  {
76  return _nodes[newId];
77  }
78 
79  JsonSchemaNode currentNode = (existingNode != null)
80  ? existingNode.Combine(schema)
81  : new JsonSchemaNode(schema);
82 
83  _nodes.Add(currentNode);
84 
85  AddProperties(schema.Properties, currentNode.Properties);
86 
87  AddProperties(schema.PatternProperties, currentNode.PatternProperties);
88 
89  if (schema.Items != null)
90  {
91  for (int i = 0; i < schema.Items.Count; i++)
92  {
93  AddItem(currentNode, i, schema.Items[i]);
94  }
95  }
96 
97  if (schema.AdditionalItems != null)
98  {
99  AddAdditionalItems(currentNode, schema.AdditionalItems);
100  }
101 
102  if (schema.AdditionalProperties != null)
103  {
104  AddAdditionalProperties(currentNode, schema.AdditionalProperties);
105  }
106 
107  if (schema.Extends != null)
108  {
109  foreach (JsonSchema jsonSchema in schema.Extends)
110  {
111  currentNode = AddSchema(currentNode, jsonSchema);
112  }
113  }
114 
115  return currentNode;
116  }

136  public void AddItem(JsonSchemaNode parentNode, int index, JsonSchema schema)
137  {
138  JsonSchemaNode existingItemNode = (parentNode.Items.Count > index)
139  ? parentNode.Items[index]
140  : null;
141 
142  JsonSchemaNode newItemNode = AddSchema(existingItemNode, schema);
143 
144  if (!(parentNode.Items.Count > index))
145  {
146  parentNode.Items.Add(newItemNode);
147  }
148  else
149  {

151  }
152  }

164  private JsonSchemaModel BuildNodeModel(JsonSchemaNode node)
165  {
166  if (_nodeModels.TryGetValue(node, out JsonSchemaModel model))
167  {
168  return model;
169  }
170 
171  model = JsonSchemaModel.Create(node.Schemas);
172  _nodeModels[node] = model;
173 
174  foreach (KeyValuePair<string, JsonSchemaNode> property in node.Properties)
175  {
176  if (model.Properties == null)
177  {
178  model.Properties = new Dictionary<string, JsonSchemaModel>();
179  }
180 
181  model.Properties[property.Key] = BuildNodeModel(property.Value);
182  }
183  foreach (KeyValuePair<string, JsonSchemaNode> property in node.PatternProperties)
184  {
185  if (model.PatternProperties == null)
186  {
187  model.PatternProperties = new Dictionary<string, JsonSchemaModel>();
188  }
189 
190  model.PatternProperties[property.Key] = BuildNodeModel(property.Value);
191  }
192  foreach (JsonSchemaNode t in node.Items)
193  {
194  if (model.Items == null)
195  {
196  model.Items = new List<JsonSchemaModel>();
197  }
198 
199  model.Items.Add(BuildNodeModel(t));
200  }
201  if (node.AdditionalProperties != null)
202  {
203  model.AdditionalProperties = BuildNodeModel(node.AdditionalProperties);
204  }
205  if (node.AdditionalItems != null)
206  {
207  model.AdditionalItems = BuildNodeModel(node.AdditionalItems);
208  }
209 
210  return model;
211  }
212  }
