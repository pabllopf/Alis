324  internal JToken EnsureParentToken(JToken? item, bool skipParentCheck)
325  {
326  if (item == null)
327  {
328  return JValue.CreateNull();
329  }
330 
331  if (skipParentCheck)
332  {
333  return item;
334  }
335 
336  // to avoid a token having multiple parents or creating a recursive loop, create a copy if...
337  // the item already has a parent
338  // the item is being added to itself
339  // the item is being added to the root parent of itself
340  if (item.Parent != null || item == this || (item.HasValues && Root == item))
341  {
342  item = item.CloneToken();
343  }
344 
345  return item;
346  }

350  internal virtual void InsertItem(int index, JToken? item, bool skipParentCheck)
351  {
352  IList<JToken> children = ChildrenTokens;
353 
354  if (index > children.Count)
355  {
356  throw new ArgumentOutOfRangeException(nameof(index), "Index must be within the bounds of the List.");
357  }
358 
359  CheckReentrancy();
360 
361  item = EnsureParentToken(item, skipParentCheck);
362 
363  JToken? previous = (index == 0) ? null : children[index - 1];
364  // haven't inserted new token yet so next token is still at the inserting index
365  JToken? next = (index == children.Count) ? null : children[index];
366 
367  ValidateToken(item, null);
368 
369  item.Parent = this;
370 
371  item.Previous = previous;
372  if (previous != null)
373  {
374  previous.Next = item;
375  }
376 
377  item.Next = next;
378  if (next != null)
379  {
380  next.Previous = item;
381  }
382 
383  children.Insert(index, item);
384 
385 #if HAVE_COMPONENT_MODEL
386  if (_listChanged != null)
387  {
388  OnListChanged(new ListChangedEventArgs(ListChangedType.ItemAdded, index));
389  }
390 #endif
391 #if HAVE_INOTIFY_COLLECTION_CHANGED
392  if (_collectionChanged != null)
393  {
394  OnCollectionChanged(new NotifyCollectionChangedEventArgs(NotifyCollectionChangedAction.Add, item, index));
395  }
396 #endif
397  }

400  {
401  IList<JToken> children = ChildrenTokens;
402 
403  if (index < 0)
404  {
405  throw new ArgumentOutOfRangeException(nameof(index), "Index is less than 0.");
406  }

413 
414  JToken item = children[index];
415  JToken? previous = (index == 0) ? null : children[index - 1];

420  previous.Next = next;
421  }

423  {

468  {

749  internal void ReadContentFrom(JsonReader r, JsonLoadSettings? settings)
750  {
751  ValidationUtils.ArgumentNotNull(r, nameof(r));
752  IJsonLineInfo? lineInfo = r as IJsonLineInfo;
753 
754  JContainer? parent = this;
755 
756  do
757  {
758  if (parent is JProperty p && p.Value != null)
759  {
760  if (parent == this)
761  {
762  return;
763  }
764 
765  parent = parent.Parent;
766  }
767 
768  MiscellaneousUtils.Assert(parent != null);
769 
770  switch (r.TokenType)
771  {
772  case JsonToken.None:
773  // new reader. move to actual content
774  break;
775  case JsonToken.StartArray:
776  JArray a = new JArray();
777  a.SetLineInfo(lineInfo, settings);
778  parent.Add(a);
779  parent = a;
780  break;
781 
782  case JsonToken.EndArray:
783  if (parent == this)
784  {
785  return;
786  }
787 
788  parent = parent.Parent;
789  break;
790  case JsonToken.StartObject:
791  JObject o = new JObject();
792  o.SetLineInfo(lineInfo, settings);
793  parent.Add(o);
794  parent = o;
795  break;
796  case JsonToken.EndObject:
797  if (parent == this)
798  {
799  return;
800  }
801 
802  parent = parent.Parent;
803  break;
804  case JsonToken.StartConstructor:
805  JConstructor constructor = new JConstructor(r.Value!.ToString());
806  constructor.SetLineInfo(lineInfo, settings);
807  parent.Add(constructor);
808  parent = constructor;
809  break;
810  case JsonToken.EndConstructor:
811  if (parent == this)
812  {
813  return;
814  }
815 
816  parent = parent.Parent;
817  break;
818  case JsonToken.String:
819  case JsonToken.Integer:
820  case JsonToken.Float:
821  case JsonToken.Date:
822  case JsonToken.Boolean:
823  case JsonToken.Bytes:
824  JValue v = new JValue(r.Value);
825  v.SetLineInfo(lineInfo, settings);
826  parent.Add(v);
827  break;
828  case JsonToken.Comment:
829  if (settings != null && settings.CommentHandling == CommentHandling.Load)
830  {
831  v = JValue.CreateComment(r.Value!.ToString());
832  v.SetLineInfo(lineInfo, settings);
833  parent.Add(v);
834  }
835  break;
836  case JsonToken.Null:
837  v = JValue.CreateNull();
838  v.SetLineInfo(lineInfo, settings);
839  parent.Add(v);
840  break;
841  case JsonToken.Undefined:
842  v = JValue.CreateUndefined();
843  v.SetLineInfo(lineInfo, settings);
844  parent.Add(v);
845  break;
846  case JsonToken.PropertyName:
847  JProperty? property = ReadProperty(r, settings, lineInfo, parent);
848  if (property != null)
849  {
850  parent = property;
851  }
852  else
853  {
854  r.Skip();
855  }
856  break;
857  default:
858  throw new InvalidOperationException("The JsonReader should not be on a token of type {0}.".FormatWith(CultureInfo.InvariantCulture, r.TokenType));
859  }
860  } while (r.Read());
861  }

863  private static JProperty? ReadProperty(JsonReader r, JsonLoadSettings? settings, IJsonLineInfo? lineInfo, JContainer parent)
864  {
865  DuplicatePropertyNameHandling duplicatePropertyNameHandling = settings?.DuplicatePropertyNameHandling ?? DuplicatePropertyNameHandling.Replace;
866 
867  JObject parentObject = (JObject)parent;
868  string propertyName = r.Value!.ToString();
869  JProperty? existingPropertyWithName = parentObject.Property(propertyName, StringComparison.Ordinal);
870  if (existingPropertyWithName != null)
871  {
872  if (duplicatePropertyNameHandling == DuplicatePropertyNameHandling.Ignore)
873  {
874  return null;
875  }
876  else if (duplicatePropertyNameHandling == DuplicatePropertyNameHandling.Error)
877  {
878  throw JsonReaderException.Create(r, "Property with the name '{0}' already exists in the current JSON object.".FormatWith(CultureInfo.InvariantCulture, propertyName));
879  }
880  }
881 
882  JProperty property = new JProperty(propertyName);
883  property.SetLineInfo(lineInfo, settings);
884  // handle multiple properties with the same name in JSON
885  if (existingPropertyWithName == null)
886  {
887  parent.Add(property);
888  }
889  else
890  {
891  existingPropertyWithName.Replace(property);
892  }
893 
894  return property;
895  }

1128  internal static void MergeEnumerableContent(JContainer target, IEnumerable content, JsonMergeSettings? settings)
1129  {
1130  switch (settings?.MergeArrayHandling ?? MergeArrayHandling.Concat)
1131  {
1132  case MergeArrayHandling.Concat:
1133  foreach (JToken item in content)
1134  {
1135  target.Add(item);
1136  }
1137  break;
1138  case MergeArrayHandling.Union:
1139 #if HAVE_HASH_SET
1140  HashSet<JToken> items = new HashSet<JToken>(target, EqualityComparer);
1141 
1142  foreach (JToken item in content)
1143  {
1144  if (items.Add(item))
1145  {
1146  target.Add(item);
1147  }
1148  }
1149 #else
1150  Dictionary<JToken, bool> items = new Dictionary<JToken, bool>(EqualityComparer);
1151  foreach (JToken t in target)
1152  {
1153  items[t] = true;
1154  }
1155 
1156  foreach (JToken item in content)
1157  {
1158  if (!items.ContainsKey(item))
1159  {
1160  items[item] = true;
1161  target.Add(item);
1162  }
1163  }
1164 #endif
1165  break;
1166  case MergeArrayHandling.Replace:
1167  if (target == content)
1168  {
1169  break;
1170  }
1171  target.ClearItems();
1172  foreach (JToken item in content)
1173  {
1174  target.Add(item);
1175  }
1176  break;
1177  case MergeArrayHandling.Merge:
1178  int i = 0;
1179  foreach (object targetItem in content)
1180  {
1181  if (i < target.Count)
1182  {
1183  JToken? sourceItem = target[i];
1184 
1185  if (sourceItem is JContainer existingContainer)
1186  {
1187  existingContainer.Merge(targetItem, settings);
1188  }
1189  else
1190  {
1191  if (targetItem != null)
1192  {
1193  JToken contentValue = CreateFromContent(targetItem);
1194  if (contentValue.Type != JTokenType.Null)
1195  {
1196  target[i] = contentValue;
1197  }
1198  }
1199  }
1200  }
1201  else
1202  {
1203  target.Add(targetItem);
1204  }
1205 
1206  i++;
1207  }
1208  break;
1209  default:
1210  throw new ArgumentOutOfRangeException(nameof(settings), "Unexpected merge array handling when merging JSON.");
1211  }
1212  }
1213  }
