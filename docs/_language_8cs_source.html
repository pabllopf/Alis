58  public static Encoding GetEncoding(string filename)
59  {
60  var bom = new byte[4];
61  using (var file = new FileStream(filename, FileMode.Open, FileAccess.Read))
62  {
63  file.Read(bom, 0, 4);
64  }
65 
66  switch (bom[0])
67  {
68  case 0x2b when bom[1] == 0x2f && bom[2] == 0x76:
69  return Encoding.UTF7;
70  case 0xef when bom[1] == 0xbb && bom[2] == 0xbf:
71  return Encoding.UTF8;
72  case 0xff when bom[1] == 0xfe && bom[2] == 0 && bom[3] == 0:
73  return Encoding.UTF32;
74  case 0xff when bom[1] == 0xfe:
75  return Encoding.Unicode;
76  case 0xfe when bom[1] == 0xff:
77  return Encoding.BigEndianUnicode;
78  case 0 when bom[1] == 0 && bom[2] == 0xfe && bom[3] == 0xff:
79  return new UTF32Encoding(true, true);
80  default:
81  return Encoding.ASCII;
82  }
