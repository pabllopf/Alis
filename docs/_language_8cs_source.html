67  public static Encoding GetEncoding(string filename)
68  {
69  var bom = new byte[4];
70  using (var file = new FileStream(filename, FileMode.Open, FileAccess.Read))
71  {
72  file.Read(bom, 0, 4);
73  }
74 
75  return bom[0] switch
76  {
77  0x2b when bom[1] == 0x2f && bom[2] == 0x76 => Encoding.UTF7,
78  0xef when bom[1] == 0xbb && bom[2] == 0xbf => Encoding.UTF8,
79  0xff when bom[1] == 0xfe && bom[2] == 0 && bom[3] == 0 => Encoding.UTF32,
80  0xff when bom[1] == 0xfe => Encoding.Unicode,
81  0xfe when bom[1] == 0xff => Encoding.BigEndianUnicode,
82  0 when bom[1] == 0 && bom[2] == 0xfe && bom[3] == 0xff => new UTF32Encoding(true, true),
83  _ => Encoding.ASCII,
84  };
85  }
