68  public static Encoding GetEncoding(string filename)
69  {
70  var bom = new byte[4];
71  using (var file = new FileStream(filename, FileMode.Open, FileAccess.Read))
72  {
73  file.Read(bom, 0, 4);
74  }
75 
76  switch (bom[0])
77  {
78  case 0x2b when bom[1] == 0x2f && bom[2] == 0x76:
79  return Encoding.UTF7;
80  case 0xef when bom[1] == 0xbb && bom[2] == 0xbf:
81  return Encoding.UTF8;
82  case 0xff when bom[1] == 0xfe && bom[2] == 0 && bom[3] == 0:
83  return Encoding.UTF32;
84  case 0xff when bom[1] == 0xfe:
85  return Encoding.Unicode;
86  case 0xfe when bom[1] == 0xff:
87  return Encoding.BigEndianUnicode;
88  case 0 when bom[1] == 0 && bom[2] == 0xfe && bom[3] == 0xff:
89  return new UTF32Encoding(true, true);
90  default:
91  return Encoding.ASCII;
92  }
