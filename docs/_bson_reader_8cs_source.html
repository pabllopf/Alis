186  public override bool Read()
187  {
188  try
189  {
190  bool success;
191 
192  switch (_bsonReaderState)
193  {
194  case BsonReaderState.Normal:
195  success = ReadNormal();
196  break;
197  case BsonReaderState.ReferenceStart:
198  case BsonReaderState.ReferenceRef:
199  case BsonReaderState.ReferenceId:
200  success = ReadReference();
201  break;
202  case BsonReaderState.CodeWScopeStart:
203  case BsonReaderState.CodeWScopeCode:
204  case BsonReaderState.CodeWScopeScope:
205  case BsonReaderState.CodeWScopeScopeObject:
206  case BsonReaderState.CodeWScopeScopeEnd:
207  success = ReadCodeWScope();
208  break;
209  default:
210  throw JsonReaderException.Create(this, "Unexpected state: {0}".FormatWith(CultureInfo.InvariantCulture, _bsonReaderState));
211  }
212 
213  if (!success)
214  {
215  SetToken(JsonToken.None);
216  return false;
217  }
218 
219  return true;
220  }
221  catch (EndOfStreamException)
222  {
223  SetToken(JsonToken.None);
224  return false;
225  }
226  }

246  private bool ReadCodeWScope()
247  {
248  switch (_bsonReaderState)
249  {
250  case BsonReaderState.CodeWScopeStart:
251  SetToken(JsonToken.PropertyName, "$code");
252  _bsonReaderState = BsonReaderState.CodeWScopeCode;
253  return true;
254  case BsonReaderState.CodeWScopeCode:
255  // total CodeWScope size - not used
256  ReadInt32();
257 
258  SetToken(JsonToken.String, ReadLengthString());
259  _bsonReaderState = BsonReaderState.CodeWScopeScope;
260  return true;
261  case BsonReaderState.CodeWScopeScope:
262  if (CurrentState == State.PostValue)
263  {
264  SetToken(JsonToken.PropertyName, "$scope");
265  return true;
266  }
267  else
268  {
269  SetToken(JsonToken.StartObject);
270  _bsonReaderState = BsonReaderState.CodeWScopeScopeObject;
271 
272  ContainerContext newContext = new ContainerContext(BsonType.Object);
273  PushContext(newContext);
274  newContext.Length = ReadInt32();
275 
276  return true;
277  }
278  case BsonReaderState.CodeWScopeScopeObject:
279  bool result = ReadNormal();
280  if (result && TokenType == JsonToken.EndObject)
281  {
282  _bsonReaderState = BsonReaderState.CodeWScopeScopeEnd;
283  }
284 
285  return result;
286  case BsonReaderState.CodeWScopeScopeEnd:
287  SetToken(JsonToken.EndObject);
288  _bsonReaderState = BsonReaderState.Normal;
289  return true;

295  private bool ReadReference()
296  {
297  switch (CurrentState)
298  {
299  case State.ObjectStart:
300  {
301  SetToken(JsonToken.PropertyName, JsonTypeReflector.RefPropertyName);
302  _bsonReaderState = BsonReaderState.ReferenceRef;
303  return true;
304  }
305  case State.Property:
306  {
307  if (_bsonReaderState == BsonReaderState.ReferenceRef)
308  {
309  SetToken(JsonToken.String, ReadLengthString());
310  return true;
311  }
312  else if (_bsonReaderState == BsonReaderState.ReferenceId)
313  {
314  SetToken(JsonToken.Bytes, ReadBytes(12));
315  return true;
316  }
317  else
318  {
319  throw JsonReaderException.Create(this, "Unexpected state when reading BSON reference: " + _bsonReaderState);
320  }
321  }
322  case State.PostValue:
323  {
324  if (_bsonReaderState == BsonReaderState.ReferenceRef)
325  {
326  SetToken(JsonToken.PropertyName, JsonTypeReflector.IdPropertyName);
327  _bsonReaderState = BsonReaderState.ReferenceId;
328  return true;
329  }
330  else if (_bsonReaderState == BsonReaderState.ReferenceId)
331  {
332  SetToken(JsonToken.EndObject);
333  _bsonReaderState = BsonReaderState.Normal;
334  return true;
335  }
336  else
337  {
338  throw JsonReaderException.Create(this, "Unexpected state when reading BSON reference: " + _bsonReaderState);
339  }
340  }
341  default:
342  throw JsonReaderException.Create(this, "Unexpected state when reading BSON reference: " + CurrentState);
343  }
344  }

346  private bool ReadNormal()
347  {
348  switch (CurrentState)
349  {
350  case State.Start:
351  {
352  JsonToken token = (!_readRootValueAsArray) ? JsonToken.StartObject : JsonToken.StartArray;
353  BsonType type = (!_readRootValueAsArray) ? BsonType.Object : BsonType.Array;
354 
355  SetToken(token);
356  ContainerContext newContext = new ContainerContext(type);
357  PushContext(newContext);
358  newContext.Length = ReadInt32();
359  return true;
360  }
361  case State.Complete:
362  case State.Closed:
363  return false;
364  case State.Property:
365  {
366  ReadType(_currentElementType);
367  return true;
368  }
369  case State.ObjectStart:
370  case State.ArrayStart:
371  case State.PostValue:
372  ContainerContext context = _currentContext;
373  if (context == null)
374  {
375  if (SupportMultipleContent)
376  {
377  goto case State.Start;
378  }
379 
380  return false;
381  }
382 
383  int lengthMinusEnd = context.Length - 1;
384 
385  if (context.Position < lengthMinusEnd)
386  {
387  if (context.Type == BsonType.Array)
388  {
389  ReadElement();
390  ReadType(_currentElementType);
391  return true;
392  }
393  else
394  {
395  SetToken(JsonToken.PropertyName, ReadElement());
396  return true;
397  }
398  }
399  else if (context.Position == lengthMinusEnd)
400  {
401  if (ReadByte() != 0)
402  {
403  throw JsonReaderException.Create(this, "Unexpected end of object byte value.");
404  }
405 
406  PopContext();
407  if (_currentContext != null)
408  {
409  MovePosition(context.Length);
410  }
411 
412  JsonToken endToken = (context.Type == BsonType.Object) ? JsonToken.EndObject : JsonToken.EndArray;
413  SetToken(endToken);
414  return true;
415  }
416  else
417  {
418  throw JsonReaderException.Create(this, "Read past end of current container context.");
419  }
420  case State.ConstructorStart:
421  break;
422  case State.Constructor:
423  break;
424  case State.Error:
425  break;
426  case State.Finished:
427  break;
428  default:
429  throw new ArgumentOutOfRangeException();
430  }
431 
432  return false;
433  }

460  private void ReadType(BsonType type)
461  {
462  switch (type)
463  {
464  case BsonType.Number:
465  double d = ReadDouble();
466 
467  if (_floatParseHandling == FloatParseHandling.Decimal)
468  {
469  SetToken(JsonToken.Float, Convert.ToDecimal(d, CultureInfo.InvariantCulture));
470  }
471  else
472  {
473  SetToken(JsonToken.Float, d);
474  }
475  break;
476  case BsonType.String:
477  case BsonType.Symbol:
478  SetToken(JsonToken.String, ReadLengthString());
479  break;
480  case BsonType.Object:
481  {
482  SetToken(JsonToken.StartObject);
483 
484  ContainerContext newContext = new ContainerContext(BsonType.Object);
485  PushContext(newContext);
486  newContext.Length = ReadInt32();
487  break;
488  }
489  case BsonType.Array:
490  {
491  SetToken(JsonToken.StartArray);
492 
493  ContainerContext newContext = new ContainerContext(BsonType.Array);
494  PushContext(newContext);
495  newContext.Length = ReadInt32();
496  break;
497  }
498  case BsonType.Binary:
499  BsonBinaryType binaryType;
500  byte[] data = ReadBinary(out binaryType);
501 
502  object value = (binaryType != BsonBinaryType.Uuid)
503  ? data
504  : (object)new Guid(data);
505 
506  SetToken(JsonToken.Bytes, value);
507  break;
508  case BsonType.Undefined:
509  SetToken(JsonToken.Undefined);
510  break;
511  case BsonType.Oid:
512  byte[] oid = ReadBytes(12);
513  SetToken(JsonToken.Bytes, oid);
514  break;
515  case BsonType.Boolean:
516  bool b = Convert.ToBoolean(ReadByte());
517  SetToken(JsonToken.Boolean, b);
518  break;
519  case BsonType.Date:
520  long ticks = ReadInt64();
521  DateTime utcDateTime = DateTimeUtils.ConvertJavaScriptTicksToDateTime(ticks);
522 
523  DateTime dateTime;
524  switch (DateTimeKindHandling)
525  {
526  case DateTimeKind.Unspecified:
527  dateTime = DateTime.SpecifyKind(utcDateTime, DateTimeKind.Unspecified);
528  break;
529  case DateTimeKind.Local:
530  dateTime = utcDateTime.ToLocalTime();
531  break;
532  default:
533  dateTime = utcDateTime;
534  break;
535  }
536 
537  SetToken(JsonToken.Date, dateTime);
538  break;
539  case BsonType.Null:
540  SetToken(JsonToken.Null);
541  break;
542  case BsonType.Regex:
543  string expression = ReadString();
544  string modifiers = ReadString();
545 
546  string regex = @"/" + expression + @"/" + modifiers;
547  SetToken(JsonToken.String, regex);
548  break;
549  case BsonType.Reference:
550  SetToken(JsonToken.StartObject);
551  _bsonReaderState = BsonReaderState.ReferenceStart;
552  break;
553  case BsonType.Code:
554  SetToken(JsonToken.String, ReadLengthString());
555  break;
556  case BsonType.CodeWScope:
557  SetToken(JsonToken.StartObject);
558  _bsonReaderState = BsonReaderState.CodeWScopeStart;
559  break;
560  case BsonType.Integer:
561  SetToken(JsonToken.Integer, (long)ReadInt32());
562  break;
563  case BsonType.TimeStamp:
564  case BsonType.Long:
565  SetToken(JsonToken.Integer, ReadInt64());
566  break;
567  default:
568  throw new ArgumentOutOfRangeException(nameof(type), "Unexpected BsonType value: " + type);
569  }
570  }

589  private string ReadString()
590  {
591  EnsureBuffers();
592 
593  StringBuilder builder = null;
594 
595  int totalBytesRead = 0;
596  // used in case of left over multibyte characters in the buffer
597  int offset = 0;
598  while (true)
599  {
600  int count = offset;
601  byte b;
602  while (count < MaxCharBytesSize && (b = _reader.ReadByte()) > 0)
603  {
604  _byteBuffer[count++] = b;
605  }
606  int byteCount = count - offset;
607  totalBytesRead += byteCount;
608 
609  if (count < MaxCharBytesSize && builder == null)
610  {
611  // pref optimization to avoid reading into a string builder
612  // if string is smaller than the buffer then return it directly
613  int length = Encoding.UTF8.GetChars(_byteBuffer, 0, byteCount, _charBuffer, 0);
614 
615  MovePosition(totalBytesRead + 1);
616  return new string(_charBuffer, 0, length);
617  }
618  else
619  {
620  // calculate the index of the end of the last full character in the buffer
621  int lastFullCharStop = GetLastFullCharStop(count - 1);
622 
623  int charCount = Encoding.UTF8.GetChars(_byteBuffer, 0, lastFullCharStop + 1, _charBuffer, 0);
624 
625  if (builder == null)
626  {
627  builder = new StringBuilder(MaxCharBytesSize * 2);
628  }
629 
630  builder.Append(_charBuffer, 0, charCount);
631 
632  if (lastFullCharStop < byteCount - 1)
633  {
634  offset = byteCount - lastFullCharStop - 1;
635  // copy left over multi byte characters to beginning of buffer for next iteration
636  Array.Copy(_byteBuffer, lastFullCharStop + 1, _byteBuffer, 0, offset);
637  }
638  else
639  {
640  // reached end of string
641  if (count < MaxCharBytesSize)
642  {
643  MovePosition(totalBytesRead + 1);
644  return builder.ToString();
645  }
646 
647  offset = 0;
648  }
649  }
650  }
651  }

665  private string GetString(int length)
666  {
667  if (length == 0)
668  {
669  return string.Empty;
670  }
671 
672  EnsureBuffers();
673 
674  StringBuilder builder = null;
675 
676  int totalBytesRead = 0;
677 
678  // used in case of left over multibyte characters in the buffer
679  int offset = 0;
680  do
681  {
682  int count = ((length - totalBytesRead) > MaxCharBytesSize - offset)
683  ? MaxCharBytesSize - offset
684  : length - totalBytesRead;
685 
686  int byteCount = _reader.Read(_byteBuffer, offset, count);
687 
688  if (byteCount == 0)
689  {
690  throw new EndOfStreamException("Unable to read beyond the end of the stream.");
691  }
692 
693  totalBytesRead += byteCount;
694 
695  // Above, byteCount is how many bytes we read this time.
696  // Below, byteCount is how many bytes are in the _byteBuffer.
697  byteCount += offset;
698 
699  if (byteCount == length)
700  {
701  // pref optimization to avoid reading into a string builder
702  // first iteration and all bytes read then return string directly
703  int charCount = Encoding.UTF8.GetChars(_byteBuffer, 0, byteCount, _charBuffer, 0);
704  return new string(_charBuffer, 0, charCount);
705  }
706  else
707  {
708  int lastFullCharStop = GetLastFullCharStop(byteCount - 1);
709 
710  if (builder == null)
711  {
712  builder = new StringBuilder(length);
713  }
714 
715  int charCount = Encoding.UTF8.GetChars(_byteBuffer, 0, lastFullCharStop + 1, _charBuffer, 0);
716  builder.Append(_charBuffer, 0, charCount);
717 
718  if (lastFullCharStop < byteCount - 1)
719  {
720  offset = byteCount - lastFullCharStop - 1;
721  // copy left over multi byte characters to beginning of buffer for next iteration
722  Array.Copy(_byteBuffer, lastFullCharStop + 1, _byteBuffer, 0, offset);
723  }
724  else
725  {
726  offset = 0;
727  }
728  }
729  } while (totalBytesRead < length);
730 
731  return builder.ToString();
732  }

734  private int GetLastFullCharStop(int start)
735  {
736  int lookbackPos = start;
737  int bis = 0;
738  while (lookbackPos >= 0)
739  {
740  bis = BytesInSequence(_byteBuffer[lookbackPos]);
741  if (bis == 0)
742  {
743  lookbackPos--;
744  continue;
745  }
746  else if (bis == 1)
747  {
748  break;
749  }
750  else
751  {
752  lookbackPos--;
753  break;
754  }
755  }
756  if (bis == start - lookbackPos)
757  {
758  //Full character.
759  return start;
760  }
761  else
762  {
763  return lookbackPos;
764  }
765  }
