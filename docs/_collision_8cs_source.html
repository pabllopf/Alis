103  public override void Update()
104  {
105  if (rectangle != null && update)
106  {
107  if (!Render.Current.Collisions.Contains(rectangle))
108  {
109  Render.Current.AddCollision(rectangle);
110  }
111 
112  rectangle.Position = new Vector2f(GetGameObject().Transform.Position.X, GetGameObject().Transform.Position.Y);
113  foreach (RectangleShape rectangleShape in Render.Current.Collisions)
114  {
115  if (!rectangleShape.Equals(rectangle))
116  {
117  var leftTop = new Vector2f(rectangle.GetGlobalBounds().Left, rectangle.GetGlobalBounds().Top);
118  var rightTop = new Vector2f(rectangle.GetGlobalBounds().Left + rectangle.GetGlobalBounds().Width, rectangle.GetGlobalBounds().Top);
119  var downLeft = new Vector2f(rectangle.GetGlobalBounds().Left, rectangle.GetGlobalBounds().Top + rectangle.GetGlobalBounds().Height);
120  var downRight = new Vector2f(rectangle.GetGlobalBounds().Left + rectangle.GetGlobalBounds().Width, rectangle.GetGlobalBounds().Top + rectangle.GetGlobalBounds().Height);
121 
122  var midleTop = new Vector2f(((rightTop.X - leftTop.X) / 2) + leftTop.X, leftTop.Y);
123  var midleDown = new Vector2f(((downRight.X - downLeft.X) / 2) + downLeft.X, downLeft.Y);
124  var midleLeft = new Vector2f(downLeft.X, ((downLeft.Y - leftTop.Y) / 2) + leftTop.Y);
125  var midleRight = new Vector2f(downRight.X, ((downRight.Y - rightTop.Y) / 2) + rightTop.Y);
126 
127 
128  if (Render.Current.Collisions.Find(i => (i != rectangle) && i.GetGlobalBounds().Intersects(rectangle.GetGlobalBounds())) != null)
129  {
130  GetGameObject().Components.ForEach(i => i.OnCollionStay(this));
131 
132  if (!isTrigger)
133  {
134  if (rectangleShape.GetGlobalBounds().Contains(midleTop.X, midleTop.Y))
135  {
136  this.GetGameObject().Transform.CanGoUp = false;
137  }
138

140  {

142  }
143 
144  if (rectangleShape.GetGlobalBounds().Contains(midleLeft.X, midleLeft.Y))
145  {
146  this.GetGameObject().Transform.CanGoLeft = false;
147  }
148 
149  if (rectangleShape.GetGlobalBounds().Contains(midleRight.X, midleRight.Y))
150  {
151  this.GetGameObject().Transform.CanGoRight = false;
152  }
153  }
154  }
155  else
156  {
157  if (!rectangleShape.GetGlobalBounds().Contains(midleTop.X, midleTop.Y))
158  {
159  this.GetGameObject().Transform.CanGoUp = true;
160  }
161 
162  if (!rectangleShape.GetGlobalBounds().Contains(midleDown.X, midleDown.Y))
