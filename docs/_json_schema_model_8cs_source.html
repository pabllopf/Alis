70  public static JsonSchemaModel Create(IList<JsonSchema> schemata)
71  {

82  private static void Combine(JsonSchemaModel model, JsonSchema schema)
83  {
84  // Version 3 of the Draft JSON Schema has the default value of Not Required
85  model.Required = model.Required || (schema.Required ?? false);
86  model.Type = model.Type & (schema.Type ?? JsonSchemaType.Any);
87 
88  model.MinimumLength = MathUtils.Max(model.MinimumLength, schema.MinimumLength);
89  model.MaximumLength = MathUtils.Min(model.MaximumLength, schema.MaximumLength);
90 
91  // not sure what is the best way to combine divisibleBy
92  model.DivisibleBy = MathUtils.Max(model.DivisibleBy, schema.DivisibleBy);
93 
94  model.Minimum = MathUtils.Max(model.Minimum, schema.Minimum);
95  model.Maximum = MathUtils.Max(model.Maximum, schema.Maximum);
96  model.ExclusiveMinimum = model.ExclusiveMinimum || (schema.ExclusiveMinimum ?? false);
97  model.ExclusiveMaximum = model.ExclusiveMaximum || (schema.ExclusiveMaximum ?? false);
98 
99  model.MinimumItems = MathUtils.Max(model.MinimumItems, schema.MinimumItems);
100  model.MaximumItems = MathUtils.Min(model.MaximumItems, schema.MaximumItems);
101  model.PositionalItemsValidation = model.PositionalItemsValidation || schema.PositionalItemsValidation;
102  model.AllowAdditionalProperties = model.AllowAdditionalProperties && schema.AllowAdditionalProperties;
103  model.AllowAdditionalItems = model.AllowAdditionalItems && schema.AllowAdditionalItems;
104  model.UniqueItems = model.UniqueItems || schema.UniqueItems;
105  if (schema.Enum != null)
106  {
107  if (model.Enum == null)
108  {
109  model.Enum = new List<JToken>();
110  }
111 
112  model.Enum.AddRangeDistinct(schema.Enum, JToken.EqualityComparer);
113  }
114  model.Disallow = model.Disallow | (schema.Disallow ?? JsonSchemaType.None);
115 
116  if (schema.Pattern != null)
117  {
118  if (model.Patterns == null)
119  {
120  model.Patterns = new List<string>();
121  }
122 
123  model.Patterns.AddDistinct(schema.Pattern);
124  }
125  }
126  }
