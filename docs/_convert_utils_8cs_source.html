201  public static PrimitiveTypeCode GetTypeCode(Type t, out bool isEnum)
202  {
203  if (TypeCodeMap.TryGetValue(t, out PrimitiveTypeCode typeCode))
204  {
205  isEnum = false;
206  return typeCode;
207  }
208 
209  if (t.IsEnum())
210  {
211  isEnum = true;
212  return GetTypeCode(Enum.GetUnderlyingType(t));
213  }
214 
215  // performance?
216  if (ReflectionUtils.IsNullableType(t))
217  {
218  Type nonNullable = Nullable.GetUnderlyingType(t);
219  if (nonNullable.IsEnum())
220  {
221  Type nullableUnderlyingType = typeof(Nullable<>).MakeGenericType(Enum.GetUnderlyingType(nonNullable));
222  isEnum = true;
223  return GetTypeCode(nullableUnderlyingType);
224  }
225  }
226 
227  isEnum = false;
228  return PrimitiveTypeCode.Object;
229  }

262  private static Func<object?, object?>? CreateCastConverter(StructMultiKey<Type, Type> t)
263  {
264  Type initialType = t.Value1;
265  Type targetType = t.Value2;
266  MethodInfo castMethodInfo = targetType.GetMethod("op_Implicit", new[] { initialType })
267  ?? targetType.GetMethod("op_Explicit", new[] { initialType });
268 
269  if (castMethodInfo == null)
270  {
271  return null;
272  }
273 
274  MethodCall<object?, object?> call = JsonTypeReflector.ReflectionDelegateFactory.CreateMethodCall<object?>(castMethodInfo);
275 
276  return o => call(null, o);
277  }

364  internal enum ConvertResult
365  {
366  Success = 0,
367  CannotConvertNull = 1,
368  NotInstantiableType = 2,
369  NoValidConversion = 3

372  public static object Convert(object initialValue, CultureInfo culture, Type targetType)
373  {
374  switch (TryConvertInternal(initialValue, culture, targetType, out object? value))
375  {
376  case ConvertResult.Success:
377  return value!;
378  case ConvertResult.CannotConvertNull:
379  throw new Exception("Can not convert null {0} into non-nullable {1}.".FormatWith(CultureInfo.InvariantCulture, initialValue.GetType(), targetType));
380  case ConvertResult.NotInstantiableType:
381  throw new ArgumentException("Target type {0} is not a value type or a non-abstract class.".FormatWith(CultureInfo.InvariantCulture, targetType), nameof(targetType));
382  case ConvertResult.NoValidConversion:
383  throw new InvalidOperationException("Can not convert from {0} to {1}.".FormatWith(CultureInfo.InvariantCulture, initialValue.GetType(), targetType));
384  default:
385  throw new InvalidOperationException("Unexpected conversion result.");
386  }
387  }
388 
389  private static bool TryConvert(object? initialValue, CultureInfo culture, Type targetType, out object? value)
390  {
391  try
392  {
393  if (TryConvertInternal(initialValue, culture, targetType, out value) == ConvertResult.Success)
394  {
395  return true;
396  }
397 
398  value = null;
399  return false;
400  }
401  catch
402  {
403  value = null;
404  return false;
405  }
406  }
407 
408  private static ConvertResult TryConvertInternal(object? initialValue, CultureInfo culture, Type targetType, out object? value)
409  {
410  if (initialValue == null)
411  {
412  throw new ArgumentNullException(nameof(initialValue));
413  }
414 
415  if (ReflectionUtils.IsNullableType(targetType))
416  {
417  targetType = Nullable.GetUnderlyingType(targetType);
418  }
419 
420  Type initialType = initialValue.GetType();
421 
422  if (targetType == initialType)
423  {
424  value = initialValue;
425  return ConvertResult.Success;
426  }
427 
428  // use Convert.ChangeType if both types are IConvertible
429  if (IsConvertible(initialValue.GetType()) && IsConvertible(targetType))
430  {
431  if (targetType.IsEnum())
432  {
433  if (initialValue is string)
434  {
435  value = Enum.Parse(targetType, initialValue.ToString(), true);
436  return ConvertResult.Success;
437  }
438  else if (IsInteger(initialValue))
439  {
440  value = Enum.ToObject(targetType, initialValue);
441  return ConvertResult.Success;
442  }
443  }
444 
445  value = System.Convert.ChangeType(initialValue, targetType, culture);
446  return ConvertResult.Success;
447  }
448 
449 #if HAVE_DATE_TIME_OFFSET
450  if (initialValue is DateTime dt && targetType == typeof(DateTimeOffset))
451  {
452  value = new DateTimeOffset(dt);
453  return ConvertResult.Success;
454  }
455 #endif
456 
457  if (initialValue is byte[] bytes && targetType == typeof(Guid))
458  {
459  value = new Guid(bytes);
460  return ConvertResult.Success;
461  }
462 
463  if (initialValue is Guid guid && targetType == typeof(byte[]))
464  {
465  value = guid.ToByteArray();
466  return ConvertResult.Success;
467  }
468 
469  if (initialValue is string s)
470  {
471  if (targetType == typeof(Guid))
472  {
473  value = new Guid(s);
474  return ConvertResult.Success;
475  }
476  if (targetType == typeof(Uri))
477  {
478  value = new Uri(s, UriKind.RelativeOrAbsolute);
479  return ConvertResult.Success;
480  }
481  if (targetType == typeof(TimeSpan))
482  {
483  value = ParseTimeSpan(s);
484  return ConvertResult.Success;
485  }
486  if (targetType == typeof(byte[]))
487  {
488  value = System.Convert.FromBase64String(s);
489  return ConvertResult.Success;
490  }
491  if (targetType == typeof(Version))
492  {
493  if (VersionTryParse(s, out Version? result))
494  {
495  value = result;
496  return ConvertResult.Success;
497  }
498  value = null;
499  return ConvertResult.NoValidConversion;
500  }
501  if (typeof(Type).IsAssignableFrom(targetType))
502  {
503  value = Type.GetType(s, true);
504  return ConvertResult.Success;
505  }
506  }
507 
508 #if HAVE_BIG_INTEGER
509  if (targetType == typeof(BigInteger))
510  {
511  value = ToBigInteger(initialValue);
512  return ConvertResult.Success;
513  }
514  if (initialValue is BigInteger integer)
515  {
516  value = FromBigInteger(integer, targetType);
517  return ConvertResult.Success;
518  }
519 #endif
520 
521 #if HAVE_TYPE_DESCRIPTOR
522  // see if source or target types have a TypeConverter that converts between the two
523  TypeConverter toConverter = TypeDescriptor.GetConverter(initialType);
524 
525  if (toConverter != null && toConverter.CanConvertTo(targetType))
526  {
527  value = toConverter.ConvertTo(null, culture, initialValue, targetType);
528  return ConvertResult.Success;
529  }
530 
531  TypeConverter fromConverter = TypeDescriptor.GetConverter(targetType);
532 
533  if (fromConverter != null && fromConverter.CanConvertFrom(initialType))
534  {
535  value = fromConverter.ConvertFrom(null, culture, initialValue);
536  return ConvertResult.Success;
537  }
538 #endif
539 #if HAVE_ADO_NET
540  // handle DBNull
541  if (initialValue == DBNull.Value)
542  {
543  if (ReflectionUtils.IsNullable(targetType))
544  {
545  value = EnsureTypeAssignable(null, initialType, targetType);
546  return ConvertResult.Success;
547  }
548 
549  // cannot convert null to non-nullable
550  value = null;
551  return ConvertResult.CannotConvertNull;
552  }
553 #endif
554 
555  if (targetType.IsInterface() || targetType.IsGenericTypeDefinition() || targetType.IsAbstract())
556  {
557  value = null;
558  return ConvertResult.NotInstantiableType;
559  }
560 
561  value = null;
562  return ConvertResult.NoValidConversion;
563  }

599  private static object? EnsureTypeAssignable(object? value, Type initialType, Type targetType)
600  {
601  if (value != null)
602  {
603  Type valueType = value.GetType();
604 
605  if (targetType.IsAssignableFrom(valueType))
606  {
607  return value;
608  }
609 
610  Func<object?, object?>? castConverter = CastConverters.Get(new StructMultiKey<Type, Type>(valueType, targetType));
611  if (castConverter != null)
612  {
613  return castConverter(value);
614  }
615  }
616  else
617  {
618  if (ReflectionUtils.IsNullable(targetType))
619  {
620  return null;
621  }
622  }
623 
624  throw new ArgumentException("Could not cast or convert from {0} to {1}.".FormatWith(CultureInfo.InvariantCulture, initialType?.ToString() ?? "{null}", targetType));
625  }

664  public static ParseResult Int32TryParse(char[] chars, int start, int length, out int value)
665  {
666  value = 0;
667 
668  if (length == 0)
669  {
670  return ParseResult.Invalid;
671  }
672 
673  bool isNegative = (chars[start] == '-');
674 
675  if (isNegative)
676  {
677  // text just a negative sign
678  if (length == 1)
679  {
680  return ParseResult.Invalid;
681  }
682 
683  start++;
684  length--;
685  }
686 
687  int end = start + length;
688 
689  // Int32.MaxValue and MinValue are 10 chars
690  // Or is 10 chars and start is greater than two
691  // Need to improve this!
692  if (length > 10 || (length == 10 && chars[start] - '0' > 2))
693  {
694  // invalid result takes precedence over overflow
695  for (int i = start; i < end; i++)
696  {
697  int c = chars[i] - '0';
698 
699  if (c < 0 || c > 9)
700  {
701  return ParseResult.Invalid;
702  }
703  }
704 
705  return ParseResult.Overflow;
706  }
707 
708  for (int i = start; i < end; i++)
709  {
710  int c = chars[i] - '0';
711 
712  if (c < 0 || c > 9)
713  {
714  return ParseResult.Invalid;
715  }
716 
717  int newValue = (10 * value) - c;
718 
719  // overflow has caused the number to loop around
720  if (newValue > value)
721  {
722  i++;
723 
724  // double check the rest of the string that there wasn't anything invalid
725  // invalid result takes precedence over overflow result
726  for (; i < end; i++)
727  {
728  c = chars[i] - '0';
729 
730  if (c < 0 || c > 9)
731  {
732  return ParseResult.Invalid;
733  }
734  }
735 
736  return ParseResult.Overflow;
737  }
738 
739  value = newValue;
740  }
741 
742  // go from negative to positive to avoids overflow
743  // negative can be slightly bigger than positive
744  if (!isNegative)
745  {
746  // negative integer can be one bigger than positive
747  if (value == int.MinValue)
748  {
749  return ParseResult.Overflow;
750  }
751 
752  value = -value;
753  }
754 
755  return ParseResult.Success;
756  }

932  public static double PackDouble(bool negative, ulong val, int scale)
933  {
934  // handle zero value
935  if (val == 0)
936  {
937  return negative ? -0.0 : 0.0;
938  }
939 
940  // normalize the mantissa
941  int exp = 64;
942 
943  if ((val & 0xFFFFFFFF00000000) == 0)
944  {
945  val <<= 32;
946  exp -= 32;
947  }
948  if ((val & 0xFFFF000000000000) == 0)
949  {
950  val <<= 16;
951  exp -= 16;
952  }
953  if ((val & 0xFF00000000000000) == 0)
954  {
955  val <<= 8;
956  exp -= 8;
957  }
958  if ((val & 0xF000000000000000) == 0)
959  {
960  val <<= 4;
961  exp -= 4;
962  }
963  if ((val & 0xC000000000000000) == 0)
964  {
965  val <<= 2;
966  exp -= 2;
967  }
968  if ((val & 0x8000000000000000) == 0)
969  {
970  val <<= 1;
971  exp -= 1;
972  }
973 
974  if (scale < 0)
975  {
976  scale = -scale;
977 
978  // check scale bounds
979  if (scale >= 22 * 16)
980  {
981  // underflow
982  return negative ? -0.0 : 0.0;
983  }
984 
985  // perform scaling
986  int index = scale & 15;
987  if (index != 0)
988  {
989  exp -= MultExp64Power10[index - 1] - 1;
990  val = Mul64Lossy(val, MultVal64Power10Inv[index - 1], ref exp);
991  }
992 
993  index = scale >> 4;
994  if (index != 0)
995  {
996  exp -= MultExp64Power10By16[index - 1] - 1;
997  val = Mul64Lossy(val, MultVal64Power10By16Inv[index - 1], ref exp);
998  }
999  }
1000  else
1001  {
1002  // check scale bounds
1003  if (scale >= 22 * 16)
1004  {
1005  // overflow
1006  return negative ? double.NegativeInfinity : double.PositiveInfinity;
1007  }
1008 
1009  // perform scaling
1010  int index = scale & 15;
1011  if (index != 0)
1012  {
1013  exp += MultExp64Power10[index - 1];
1014  val = Mul64Lossy(val, MultVal64Power10[index - 1], ref exp);
1015  }
1016 
1017  index = scale >> 4;
1018  if (index != 0)
1019  {
1020  exp += MultExp64Power10By16[index - 1];
1021  val = Mul64Lossy(val, MultVal64Power10By16[index - 1], ref exp);
1022  }
1023  }
1024 
1025  // round & scale down
1026 
1027  if ((val & (1 << 10)) != 0)
1028  {
1029  // IEEE round to even
1030  ulong tmp = val + ((1UL << 10) - 1 + ((val >> 11) & 1));
1031  if (tmp < val)
1032  {
1033  // overflow
1034  tmp = (tmp >> 1) | 0x8000000000000000;
1035  exp++;
1036  }
1037  val = tmp;
1038  }
1039

1041 
1042  exp += 0x3FE;
1043

1045 
1046  if (exp <= 0)
1047  {
1048  if (exp == -52 && (val >= 0x8000000000000058))

1051  val = 0x0000000000000001;
1052  }
1053  else if (exp <= -52)
1054  {
1055  // underflow
1056  val = 0;
1057  }
1058  else
1059  {
1060  // denormalized value
1061  val >>= (-exp + 12);
1062  }
1063  }
1064  else if (exp >= 0x7FF)
1065  {
1066  // overflow
1067  val = 0x7FF0000000000000;
1068  }
