55  public override void WriteJson(JsonWriter writer, object? value, JsonSerializer serializer)

68  private byte[] GetByteArray(object value)
69  {
70 #if HAVE_LINQ
71  if (value.GetType().FullName == BinaryTypeName)
72  {
73  EnsureReflectionObject(value.GetType());
74  MiscellaneousUtils.Assert(_reflectionObject != null);
75 
76  return (byte[])_reflectionObject.GetValue(value, BinaryToArrayName)!;
77  }
78 #endif
79 #if HAVE_ADO_NET
80  if (value is SqlBinary binary)
81  {
82  return binary.Value;
83  }
84 #endif
85 
86  throw new JsonSerializationException("Unexpected value type when writing binary: {0}".FormatWith(CultureInfo.InvariantCulture, value.GetType()));
87  }

107  public override object? ReadJson(JsonReader reader, Type objectType, object? existingValue, JsonSerializer serializer)
108  {
109  if (reader.TokenType == JsonToken.Null)
110  {
111  if (!ReflectionUtils.IsNullable(objectType))
112  {
113  throw JsonSerializationException.Create(reader, "Cannot convert null value to {0}.".FormatWith(CultureInfo.InvariantCulture, objectType));
114  }
115 
116  return null;
117  }
118 
119  byte[] data;
120 
121  if (reader.TokenType == JsonToken.StartArray)
122  {
123  data = ReadByteArray(reader);
124  }
125  else if (reader.TokenType == JsonToken.String)
126  {
127  // current token is already at base64 string
128  // unable to call ReadAsBytes so do it the old fashion way
129  string encodedData = reader.Value!.ToString();
130  data = Convert.FromBase64String(encodedData);
131  }
132  else
133  {
134  throw JsonSerializationException.Create(reader, "Unexpected token parsing binary. Expected String or StartArray, got {0}.".FormatWith(CultureInfo.InvariantCulture, reader.TokenType));
135  }
136 
137  Type t = (ReflectionUtils.IsNullableType(objectType))
138  ? Nullable.GetUnderlyingType(objectType)
139  : objectType;
140 
141 #if HAVE_LINQ
142  if (t.FullName == BinaryTypeName)
143  {
144  EnsureReflectionObject(t);
145  MiscellaneousUtils.Assert(_reflectionObject != null);
146 
147  return _reflectionObject.Creator!(data);
148  }
149 #endif
150 
151 #if HAVE_ADO_NET
152  if (t == typeof(SqlBinary))
153  {
154  return new SqlBinary(data);
155  }
156 #endif
157 
158  throw JsonSerializationException.Create(reader, "Unexpected object type when writing binary: {0}".FormatWith(CultureInfo.InvariantCulture, objectType));
159  }
160 
161  private byte[] ReadByteArray(JsonReader reader)
162  {
163  List<byte> byteList = new List<byte>();
164 
165  while (reader.Read())
166  {
167  switch (reader.TokenType)
168  {
169  case JsonToken.Integer:
170  byteList.Add(Convert.ToByte(reader.Value, CultureInfo.InvariantCulture));
171  break;
172  case JsonToken.EndArray:
173  return byteList.ToArray();
174  case JsonToken.Comment:
175  // skip
176  break;
177  default:
178  throw JsonSerializationException.Create(reader, "Unexpected token when reading bytes: {0}".FormatWith(CultureInfo.InvariantCulture, reader.TokenType));
179  }
180  }
181 
182  throw JsonSerializationException.Create(reader, "Unexpected end when reading bytes.");
183  }
