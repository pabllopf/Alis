55  private void ReferenceOrWriteSchema(JsonSchema schema)
56  {
57  if (schema.Id != null && _resolver.GetSchema(schema.Id) != null)
58  {
59  _writer.WriteStartObject();
60  _writer.WritePropertyName(JsonTypeReflector.RefPropertyName);
61  _writer.WriteValue(schema.Id);
62  _writer.WriteEndObject();
63  }
64  else
65  {
66  WriteSchema(schema);
67  }
68  }

70  public void WriteSchema(JsonSchema schema)
71  {
72  ValidationUtils.ArgumentNotNull(schema, nameof(schema));
73 
74  if (!_resolver.LoadedSchemas.Contains(schema))
75  {
76  _resolver.LoadedSchemas.Add(schema);
77  }
78 
79  _writer.WriteStartObject();
80  WritePropertyIfNotNull(_writer, JsonSchemaConstants.IdPropertyName, schema.Id);
81  WritePropertyIfNotNull(_writer, JsonSchemaConstants.TitlePropertyName, schema.Title);
82  WritePropertyIfNotNull(_writer, JsonSchemaConstants.DescriptionPropertyName, schema.Description);
83  WritePropertyIfNotNull(_writer, JsonSchemaConstants.RequiredPropertyName, schema.Required);
84  WritePropertyIfNotNull(_writer, JsonSchemaConstants.ReadOnlyPropertyName, schema.ReadOnly);
85  WritePropertyIfNotNull(_writer, JsonSchemaConstants.HiddenPropertyName, schema.Hidden);
86  WritePropertyIfNotNull(_writer, JsonSchemaConstants.TransientPropertyName, schema.Transient);
87  if (schema.Type != null)
88  {
89  WriteType(JsonSchemaConstants.TypePropertyName, _writer, schema.Type.GetValueOrDefault());
90  }
91  if (!schema.AllowAdditionalProperties)
92  {
93  _writer.WritePropertyName(JsonSchemaConstants.AdditionalPropertiesPropertyName);
94  _writer.WriteValue(schema.AllowAdditionalProperties);
95  }
96  else
97  {
98  if (schema.AdditionalProperties != null)
99  {
100  _writer.WritePropertyName(JsonSchemaConstants.AdditionalPropertiesPropertyName);
101  ReferenceOrWriteSchema(schema.AdditionalProperties);
102  }
103  }
104  if (!schema.AllowAdditionalItems)
105  {
106  _writer.WritePropertyName(JsonSchemaConstants.AdditionalItemsPropertyName);
107  _writer.WriteValue(schema.AllowAdditionalItems);
108  }
109  else
110  {
111  if (schema.AdditionalItems != null)
112  {
113  _writer.WritePropertyName(JsonSchemaConstants.AdditionalItemsPropertyName);
114  ReferenceOrWriteSchema(schema.AdditionalItems);
115  }
116  }
117  WriteSchemaDictionaryIfNotNull(_writer, JsonSchemaConstants.PropertiesPropertyName, schema.Properties);
118  WriteSchemaDictionaryIfNotNull(_writer, JsonSchemaConstants.PatternPropertiesPropertyName, schema.PatternProperties);
119  WriteItems(schema);
120  WritePropertyIfNotNull(_writer, JsonSchemaConstants.MinimumPropertyName, schema.Minimum);
121  WritePropertyIfNotNull(_writer, JsonSchemaConstants.MaximumPropertyName, schema.Maximum);
122  WritePropertyIfNotNull(_writer, JsonSchemaConstants.ExclusiveMinimumPropertyName, schema.ExclusiveMinimum);
123  WritePropertyIfNotNull(_writer, JsonSchemaConstants.ExclusiveMaximumPropertyName, schema.ExclusiveMaximum);
124  WritePropertyIfNotNull(_writer, JsonSchemaConstants.MinimumLengthPropertyName, schema.MinimumLength);
125  WritePropertyIfNotNull(_writer, JsonSchemaConstants.MaximumLengthPropertyName, schema.MaximumLength);
126  WritePropertyIfNotNull(_writer, JsonSchemaConstants.MinimumItemsPropertyName, schema.MinimumItems);
127  WritePropertyIfNotNull(_writer, JsonSchemaConstants.MaximumItemsPropertyName, schema.MaximumItems);
128  WritePropertyIfNotNull(_writer, JsonSchemaConstants.DivisibleByPropertyName, schema.DivisibleBy);
129  WritePropertyIfNotNull(_writer, JsonSchemaConstants.FormatPropertyName, schema.Format);
130  WritePropertyIfNotNull(_writer, JsonSchemaConstants.PatternPropertyName, schema.Pattern);
131  if (schema.Enum != null)
132  {
133  _writer.WritePropertyName(JsonSchemaConstants.EnumPropertyName);
134  _writer.WriteStartArray();
135  foreach (JToken token in schema.Enum)
136  {
137  token.WriteTo(_writer);
138  }
139  _writer.WriteEndArray();
140  }
141  if (schema.Default != null)
142  {
143  _writer.WritePropertyName(JsonSchemaConstants.DefaultPropertyName);
144  schema.Default.WriteTo(_writer);
145  }
146  if (schema.Disallow != null)
147  {
148  WriteType(JsonSchemaConstants.DisallowPropertyName, _writer, schema.Disallow.GetValueOrDefault());
149  }
150  if (schema.Extends != null && schema.Extends.Count > 0)
151  {
152  _writer.WritePropertyName(JsonSchemaConstants.ExtendsPropertyName);
153  if (schema.Extends.Count == 1)
154  {
155  ReferenceOrWriteSchema(schema.Extends[0]);
156  }
157  else
158  {
159  _writer.WriteStartArray();
160  foreach (JsonSchema jsonSchema in schema.Extends)
161  {
162  ReferenceOrWriteSchema(jsonSchema);
163  }
164  _writer.WriteEndArray();
165  }
166  }
167  _writer.WriteEndObject();
168  }

170  private void WriteSchemaDictionaryIfNotNull(JsonWriter writer, string propertyName, IDictionary<string, JsonSchema> properties)
171  {
172  if (properties != null)
173  {
174  writer.WritePropertyName(propertyName);
175  writer.WriteStartObject();
176  foreach (KeyValuePair<string, JsonSchema> property in properties)
177  {
178  writer.WritePropertyName(property.Key);
179  ReferenceOrWriteSchema(property.Value);
180  }
181  writer.WriteEndObject();
182  }
183  }
184 
185  private void WriteItems(JsonSchema schema)
186  {
187  if (schema.Items == null && !schema.PositionalItemsValidation)
188  {
189  return;
190  }
191 
192  _writer.WritePropertyName(JsonSchemaConstants.ItemsPropertyName);
193 
194  if (!schema.PositionalItemsValidation)
195  {
196  if (schema.Items != null && schema.Items.Count > 0)
197  {
198  ReferenceOrWriteSchema(schema.Items[0]);
199  }
200  else
201  {
202  _writer.WriteStartObject();
203  _writer.WriteEndObject();
204  }
205  return;
206  }
207 
208  _writer.WriteStartArray();
209  if (schema.Items != null)
210  {
211  foreach (JsonSchema itemSchema in schema.Items)
212  {
213  ReferenceOrWriteSchema(itemSchema);
214  }
215  }
216  _writer.WriteEndArray();
217  }

219  private void WriteType(string propertyName, JsonWriter writer, JsonSchemaType type)
220  {
221  if (Enum.IsDefined(typeof(JsonSchemaType), type))
222  {
223  writer.WritePropertyName(propertyName);
224  writer.WriteValue(JsonSchemaBuilder.MapType(type));
225  }
226  else
227  {
228  IEnumerator<JsonSchemaType> en = EnumUtils.GetFlagsValues(type).Where(v => v != JsonSchemaType.None).GetEnumerator();
229  if (en.MoveNext())
230  {
231  writer.WritePropertyName(propertyName);
232  JsonSchemaType first = en.Current;
233  if (en.MoveNext())
234  {
235  writer.WriteStartArray();
236  writer.WriteValue(JsonSchemaBuilder.MapType(first));
237  do
238  {
239  writer.WriteValue(JsonSchemaBuilder.MapType(en.Current));
240  } while (en.MoveNext());
241  writer.WriteEndArray();
242  }
243  else
244  {
245  writer.WriteValue(JsonSchemaBuilder.MapType(first));
246  }
247  }
248  }
249  }
250 
251  private void WritePropertyIfNotNull(JsonWriter writer, string propertyName, object value)
252  {
253  if (value != null)
254  {
255  writer.WritePropertyName(propertyName);
256  writer.WriteValue(value);
257  }
258  }
259  }
