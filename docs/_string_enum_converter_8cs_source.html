178  public override void WriteJson(JsonWriter writer, object? value, JsonSerializer serializer)
179  {
180  if (value == null)
181  {
182  writer.WriteNull();
183  return;
184  }
185 
186  Enum e = (Enum)value;
187 
188  if (!EnumUtils.TryToString(e.GetType(), value, NamingStrategy, out string? enumName))
189  {
190  if (!AllowIntegerValues)
191  {
192  throw JsonSerializationException.Create(null, writer.ContainerPath, "Integer value {0} is not allowed.".FormatWith(CultureInfo.InvariantCulture, e.ToString("D")), null);
193  }
194 
195  // enum value has no name so write number
196  writer.WriteValue(value);
197  }
198  else
199  {
200  writer.WriteValue(enumName);
201  }
202  }

212  public override object? ReadJson(JsonReader reader, Type objectType, object? existingValue, JsonSerializer serializer)
213  {
214  if (reader.TokenType == JsonToken.Null)
215  {
216  if (!ReflectionUtils.IsNullableType(objectType))
217  {
218  throw JsonSerializationException.Create(reader, "Cannot convert null value to {0}.".FormatWith(CultureInfo.InvariantCulture, objectType));
219  }
220 
221  return null;
222  }
223 
224  bool isNullable = ReflectionUtils.IsNullableType(objectType);
225  Type t = isNullable ? Nullable.GetUnderlyingType(objectType) : objectType;
226 
227  try
228  {
229  if (reader.TokenType == JsonToken.String)
230  {
231  string? enumText = reader.Value?.ToString();
232 
233  if (StringUtils.IsNullOrEmpty(enumText) && isNullable)
234  {
235  return null;
236  }
237 
238  return EnumUtils.ParseEnum(t, NamingStrategy, enumText!, !AllowIntegerValues);
239  }
240 
241  if (reader.TokenType == JsonToken.Integer)
242  {
243  if (!AllowIntegerValues)
244  {
245  throw JsonSerializationException.Create(reader, "Integer value {0} is not allowed.".FormatWith(CultureInfo.InvariantCulture, reader.Value));
246  }
247 
248  return ConvertUtils.ConvertOrCast(reader.Value, CultureInfo.InvariantCulture, t);
249  }
250  }
251  catch (Exception ex)
252  {
253  throw JsonSerializationException.Create(reader, "Error converting value {0} to type '{1}'.".FormatWith(CultureInfo.InvariantCulture, MiscellaneousUtils.ToString(reader.Value), objectType), ex);
254  }
255 
256  // we don't actually expect to get here.
257  throw JsonSerializationException.Create(reader, "Unexpected token {0} when parsing enum.".FormatWith(CultureInfo.InvariantCulture, reader.TokenType));
258  }
259 
267  public override bool CanConvert(Type objectType)
268  {
269  Type t = (ReflectionUtils.IsNullableType(objectType))
270  ? Nullable.GetUnderlyingType(objectType)
271  : objectType;
272 
273  return t.IsEnum();
274  }
