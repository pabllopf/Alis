67  public void Populate(JsonReader reader, object target)
68  {
69  ValidationUtils.ArgumentNotNull(target, nameof(target));
70 
71  Type objectType = target.GetType();
72 
73  JsonContract contract = Serializer._contractResolver.ResolveContract(objectType);
74 
75  if (!reader.MoveToContent())
76  {
77  throw JsonSerializationException.Create(reader, "No JSON content found.");
78  }
79 
80  if (reader.TokenType == JsonToken.StartArray)
81  {
82  if (contract.ContractType == JsonContractType.Array)
83  {
84  JsonArrayContract arrayContract = (JsonArrayContract)contract;
85 
86  PopulateList((arrayContract.ShouldCreateWrapper) ? arrayContract.CreateWrapper(target) : (IList)target, reader, arrayContract, null, null);
87  }
88  else
89  {
90  throw JsonSerializationException.Create(reader, "Cannot populate JSON array onto type '{0}'.".FormatWith(CultureInfo.InvariantCulture, objectType));
91  }
92  }
93  else if (reader.TokenType == JsonToken.StartObject)
94  {
95  reader.ReadAndAssert();
96 
97  string? id = null;
98  if (Serializer.MetadataPropertyHandling != MetadataPropertyHandling.Ignore
99  && reader.TokenType == JsonToken.PropertyName
100  && string.Equals(reader.Value!.ToString(), JsonTypeReflector.IdPropertyName, StringComparison.Ordinal))
101  {
102  reader.ReadAndAssert();
103  id = reader.Value?.ToString();
104  reader.ReadAndAssert();
105  }
106 
107  if (contract.ContractType == JsonContractType.Dictionary)
108  {
109  JsonDictionaryContract dictionaryContract = (JsonDictionaryContract)contract;
110  PopulateDictionary((dictionaryContract.ShouldCreateWrapper) ? dictionaryContract.CreateWrapper(target) : (IDictionary)target, reader, dictionaryContract, null, id);
111  }
112  else if (contract.ContractType == JsonContractType.Object)
113  {
114  PopulateObject(target, reader, (JsonObjectContract)contract, null, id);
115  }
116  else
117  {
118  throw JsonSerializationException.Create(reader, "Cannot populate JSON object onto type '{0}'.".FormatWith(CultureInfo.InvariantCulture, objectType));
119  }
120  }
121  else
122  {
123  throw JsonSerializationException.Create(reader, "Unexpected initial token '{0}' when populating object. Expected JSON object or array.".FormatWith(CultureInfo.InvariantCulture, reader.TokenType));
124  }
125  }

142  public object? Deserialize(JsonReader reader, Type? objectType, bool checkAdditionalContent)
143  {
144  if (reader == null)
145  {
146  throw new ArgumentNullException(nameof(reader));
147  }
148 
149  JsonContract? contract = GetContractSafe(objectType);
150 
151  try
152  {
153  JsonConverter? converter = GetConverter(contract, null, null, null);
154 
155  if (reader.TokenType == JsonToken.None && !reader.ReadForType(contract, converter != null))
156  {
157  if (contract != null && !contract.IsNullable)
158  {
159  throw JsonSerializationException.Create(reader, "No JSON content found and type '{0}' is not nullable.".FormatWith(CultureInfo.InvariantCulture, contract.UnderlyingType));
160  }
161 
162  return null;
163  }
164 
165  object? deserializedValue;
166 
167  if (converter != null && converter.CanRead)
168  {
169  deserializedValue = DeserializeConvertable(converter, reader, objectType!, null);
170  }
171  else
172  {
173  deserializedValue = CreateValueInternal(reader, objectType, contract, null, null, null, null);
174  }
175 
176  if (checkAdditionalContent)
177  {
178  while (reader.Read())
179  {
180  if (reader.TokenType != JsonToken.Comment)
181  {
182  throw JsonSerializationException.Create(reader, "Additional text found in JSON string after finishing deserializing object.");
183  }
184  }
185  }
186 
187  return deserializedValue;
188  }
189  catch (Exception ex)
190  {
191  if (IsErrorHandled(null, contract, null, reader as IJsonLineInfo, reader.Path, ex))
192  {
193  HandleError(reader, false, 0);
194  return null;
195  }
196  else
197  {
198  // clear context in case serializer is being used inside a converter
199  // if the converter wraps the error then not clearing the context will cause this error:
200  // "Current error context error is different to requested error."
201  ClearErrorContext();
202  throw;
203  }
204  }
205  }

217  private JToken? CreateJToken(JsonReader reader, JsonContract? contract)
218  {
219  ValidationUtils.ArgumentNotNull(reader, nameof(reader));
220 
221  if (contract != null)
222  {
223  if (contract.UnderlyingType == typeof(JRaw))
224  {
225  return JRaw.Create(reader);
226  }
227  if (reader.TokenType == JsonToken.Null
228  && !(contract.UnderlyingType == typeof(JValue) || contract.UnderlyingType == typeof(JToken)))
229  {
230  return null;
231  }
232  }
233 
234  JToken? token;
235  using (JTokenWriter writer = new JTokenWriter())
236  {
237  writer.WriteToken(reader);
238  token = writer.Token;
239  }
240 
241  return token;
242  }

244  private JToken CreateJObject(JsonReader reader)
245  {
246  ValidationUtils.ArgumentNotNull(reader, nameof(reader));
247 
248  // this is needed because we've already read inside the object, looking for metadata properties
249  using (JTokenWriter writer = new JTokenWriter())
250  {
251  writer.WriteStartObject();
252 
253  do
254  {
255  if (reader.TokenType == JsonToken.PropertyName)
256  {
257  string propertyName = (string)reader.Value!;
258  if (!reader.ReadAndMoveToContent())
259  {
260  break;
261  }
262 
263  if (CheckPropertyName(reader, propertyName))
264  {
265  continue;
266  }
267 
268  writer.WritePropertyName(propertyName);
269  writer.WriteToken(reader, true, true, false);
270  }
271  else if (reader.TokenType == JsonToken.Comment)
272  {
273  // eat
274  }
275  else
276  {
277  writer.WriteEndObject();
278  return writer.Token!;
279  }
280  } while (reader.Read());
281 
282  throw JsonSerializationException.Create(reader, "Unexpected end when deserializing object.");
283  }

286  private object? CreateValueInternal(JsonReader reader, Type? objectType, JsonContract? contract, JsonProperty? member, JsonContainerContract? containerContract, JsonProperty? containerMember, object? existingValue)
287  {
288  if (contract != null && contract.ContractType == JsonContractType.Linq)
289  {
290  return CreateJToken(reader, contract);
291  }
292 
293  do
294  {
295  switch (reader.TokenType)
296  {
297  // populate a typed object or generic dictionary/array
298  // depending upon whether an objectType was supplied
299  case JsonToken.StartObject:
300  return CreateObject(reader, objectType, contract, member, containerContract, containerMember, existingValue);
301  case JsonToken.StartArray:
302  return CreateList(reader, objectType, contract, member, existingValue, null);
303  case JsonToken.Integer:
304  case JsonToken.Float:
305  case JsonToken.Boolean:
306  case JsonToken.Date:
307  case JsonToken.Bytes:
308  return EnsureType(reader, reader.Value, CultureInfo.InvariantCulture, contract, objectType);
309  case JsonToken.String:
310  string s = (string)reader.Value!;
311 
312  // string that needs to be returned as a byte array should be base 64 decoded
313  if (objectType == typeof(byte[]))
314  {
315  return Convert.FromBase64String(s);
316  }
317 
318  // convert empty string to null automatically for nullable types
319  if (CoerceEmptyStringToNull(objectType, contract, s))
320  {
321  return null;
322  }
323 
324  return EnsureType(reader, s, CultureInfo.InvariantCulture, contract, objectType);
325  case JsonToken.StartConstructor:
326  string constructorName = reader.Value!.ToString();
327 
328  return EnsureType(reader, constructorName, CultureInfo.InvariantCulture, contract, objectType);
329  case JsonToken.Null:
330  case JsonToken.Undefined:
331 #if HAVE_ADO_NET
332  if (objectType == typeof(DBNull))
333  {
334  return DBNull.Value;
335  }
336 #endif
337 
338  return EnsureType(reader, reader.Value, CultureInfo.InvariantCulture, contract, objectType);
339  case JsonToken.Raw:
340  return new JRaw((string?)reader.Value);
341  case JsonToken.Comment:
342  // ignore
343  break;
344  default:
345  throw JsonSerializationException.Create(reader, "Unexpected token while deserializing object: " + reader.TokenType);
346  }
347  } while (reader.Read());
348 
349  throw JsonSerializationException.Create(reader, "Unexpected end when deserializing object.");
350  }

354  return StringUtils.IsNullOrEmpty(s) && objectType != null && objectType != typeof(string) && objectType != typeof(object) && contract != null && contract.IsNullable;

358  {
359  switch (contract.ContractType)

381  private JsonConverter? GetConverter(JsonContract? contract, JsonConverter? memberConverter, JsonContainerContract? containerContract, JsonProperty? containerProperty)
382  {
383  JsonConverter? converter = null;
384  if (memberConverter != null)
385  {
386  // member attribute converter
387  converter = memberConverter;
388  }
389  else if (containerProperty?.ItemConverter != null)
390  {
391  converter = containerProperty.ItemConverter;
392  }
393  else if (containerContract?.ItemConverter != null)
394  {
395  converter = containerContract.ItemConverter;
396  }
397  else if (contract != null)
398  {
399  if (contract.Converter != null)
400  {
401  // class attribute converter
402  converter = contract.Converter;
403  }
404  else if (Serializer.GetMatchingConverter(contract.UnderlyingType) is JsonConverter matchingConverter)
405  {
406  // passed in converters
407  converter = matchingConverter;
408  }
409  else if (contract.InternalConverter != null)
410  {
411  // internally specified converter
412  converter = contract.InternalConverter;
413  }
414  }
415  return converter;
416  }

418  private object? CreateObject(JsonReader reader, Type? objectType, JsonContract? contract, JsonProperty? member, JsonContainerContract? containerContract, JsonProperty? containerMember, object? existingValue)
419  {
420  string? id;
421  Type? resolvedObjectType = objectType;
422 
423  if (Serializer.MetadataPropertyHandling == MetadataPropertyHandling.Ignore)
424  {
425  // don't look for metadata properties
426  reader.ReadAndAssert();
427  id = null;
428  }
429  else if (Serializer.MetadataPropertyHandling == MetadataPropertyHandling.ReadAhead)
430  {
431  if (!(reader is JTokenReader tokenReader))
432  {
433  JToken t = JToken.ReadFrom(reader);
434  tokenReader = (JTokenReader)t.CreateReader();
435  tokenReader.Culture = reader.Culture;
436  tokenReader.DateFormatString = reader.DateFormatString;
437  tokenReader.DateParseHandling = reader.DateParseHandling;
438  tokenReader.DateTimeZoneHandling = reader.DateTimeZoneHandling;
439  tokenReader.FloatParseHandling = reader.FloatParseHandling;
440  tokenReader.SupportMultipleContent = reader.SupportMultipleContent;
441 
442  // start
443  tokenReader.ReadAndAssert();
444 
445  reader = tokenReader;
446  }
447 
448  if (ReadMetadataPropertiesToken(tokenReader, ref resolvedObjectType, ref contract, member, containerContract, containerMember, existingValue, out object? newValue, out id))
449  {
450  return newValue;
451  }
452  }
453  else
454  {
455  reader.ReadAndAssert();
456  if (ReadMetadataProperties(reader, ref resolvedObjectType, ref contract, member, containerContract, containerMember, existingValue, out object? newValue, out id))
457  {
458  return newValue;
459  }
460  }
461 
462  if (HasNoDefinedType(contract))
463  {
464  return CreateJObject(reader);
465  }
466 
467  MiscellaneousUtils.Assert(resolvedObjectType != null);
468  MiscellaneousUtils.Assert(contract != null);
469 
470  switch (contract.ContractType)
471  {
472  case JsonContractType.Object:
473  {
474  bool createdFromNonDefaultCreator = false;
475  JsonObjectContract objectContract = (JsonObjectContract)contract;
476  object targetObject;
477  // check that if type name handling is being used that the existing value is compatible with the specified type
478  if (existingValue != null && (resolvedObjectType == objectType || resolvedObjectType.IsAssignableFrom(existingValue.GetType())))
479  {
480  targetObject = existingValue;
481  }
482  else
483  {
484  targetObject = CreateNewObject(reader, objectContract, member, containerMember, id, out createdFromNonDefaultCreator);
485  }
486 
487  // don't populate if read from non-default creator because the object has already been read
488  if (createdFromNonDefaultCreator)
489  {
490  return targetObject;
491  }
492 
493  return PopulateObject(targetObject, reader, objectContract, member, id);
494  }
495  case JsonContractType.Primitive:
496  {
497  JsonPrimitiveContract primitiveContract = (JsonPrimitiveContract)contract;
498  // if the content is inside $value then read past it
499  if (Serializer.MetadataPropertyHandling != MetadataPropertyHandling.Ignore
500  && reader.TokenType == JsonToken.PropertyName
501  && string.Equals(reader.Value!.ToString(), JsonTypeReflector.ValuePropertyName, StringComparison.Ordinal))
502  {
503  reader.ReadAndAssert();
504 
505  // the token should not be an object because the $type value could have been included in the object
506  // without needing the $value property
507  if (reader.TokenType == JsonToken.StartObject)
508  {
509  throw JsonSerializationException.Create(reader, "Unexpected token when deserializing primitive value: " + reader.TokenType);
510  }
511 
512  object? value = CreateValueInternal(reader, resolvedObjectType, primitiveContract, member, null, null, existingValue);
513 
514  reader.ReadAndAssert();
515  return value;
516  }
517  break;
518  }
519  case JsonContractType.Dictionary:
520  {
521  JsonDictionaryContract dictionaryContract = (JsonDictionaryContract)contract;
522  object targetDictionary;
523 
524  if (existingValue == null)
525  {
526  IDictionary dictionary = CreateNewDictionary(reader, dictionaryContract, out bool createdFromNonDefaultCreator);
527 
528  if (createdFromNonDefaultCreator)
529  {
530  if (id != null)
531  {
532  throw JsonSerializationException.Create(reader, "Cannot preserve reference to readonly dictionary, or dictionary created from a non-default constructor: {0}.".FormatWith(CultureInfo.InvariantCulture, contract.UnderlyingType));
533  }
534 
535  if (contract.OnSerializingCallbacks.Count > 0)
536  {
537  throw JsonSerializationException.Create(reader, "Cannot call OnSerializing on readonly dictionary, or dictionary created from a non-default constructor: {0}.".FormatWith(CultureInfo.InvariantCulture, contract.UnderlyingType));
538  }
539 
540  if (contract.OnErrorCallbacks.Count > 0)
541  {
542  throw JsonSerializationException.Create(reader, "Cannot call OnError on readonly list, or dictionary created from a non-default constructor: {0}.".FormatWith(CultureInfo.InvariantCulture, contract.UnderlyingType));
543  }
544 
545  if (!dictionaryContract.HasParameterizedCreatorInternal)
546  {
547  throw JsonSerializationException.Create(reader, "Cannot deserialize readonly or fixed size dictionary: {0}.".FormatWith(CultureInfo.InvariantCulture, contract.UnderlyingType));
548  }
549  }
550 
551  PopulateDictionary(dictionary, reader, dictionaryContract, member, id);
552 
553  if (createdFromNonDefaultCreator)
554  {
555  ObjectConstructor<object> creator = (dictionaryContract.OverrideCreator ?? dictionaryContract.ParameterizedCreator)!;
556 
557  return creator(dictionary);
558  }
559  else if (dictionary is IWrappedDictionary wrappedDictionary)
560  {
561  return wrappedDictionary.UnderlyingDictionary;
562  }
563 
564  targetDictionary = dictionary;
565  }
566  else
567  {
568  targetDictionary = PopulateDictionary(dictionaryContract.ShouldCreateWrapper || !(existingValue is IDictionary) ? dictionaryContract.CreateWrapper(existingValue) : (IDictionary)existingValue, reader, dictionaryContract, member, id);
569  }
570 
571  return targetDictionary;
572  }
573 #if HAVE_DYNAMIC
574  case JsonContractType.Dynamic:
575  JsonDynamicContract dynamicContract = (JsonDynamicContract)contract;
576  return CreateDynamic(reader, dynamicContract, member, id);
577 #endif
578 #if HAVE_BINARY_SERIALIZATION
579  case JsonContractType.Serializable:
580  JsonISerializableContract serializableContract = (JsonISerializableContract)contract;
581  return CreateISerializable(reader, serializableContract, member, id);
582 #endif
583  }

586  @"To fix this error either change the JSON to a {1} or change the deserialized type so that it is a normal .NET type (e.g. not a primitive type like integer, not a collection type like an array or List<T>) that can be deserialized from a JSON object. 592  private bool ReadMetadataPropertiesToken(JTokenReader reader, ref Type? objectType, ref JsonContract? contract, JsonProperty? member, JsonContainerContract? containerContract, JsonProperty? containerMember, object? existingValue, [NotNullWhen(true)]out object? newValue, out string? id)
593  {
594  id = null;
595  newValue = null;
596 
597  if (reader.TokenType == JsonToken.StartObject)
598  {
599  JObject current = (JObject)reader.CurrentToken!;
600 
601  JProperty? refProperty = current.Property(JsonTypeReflector.RefPropertyName, StringComparison.Ordinal);
602  if (refProperty != null)
603  {
604  JToken refToken = refProperty.Value;
605  if (refToken.Type != JTokenType.String && refToken.Type != JTokenType.Null)
606  {
607  throw JsonSerializationException.Create(refToken, refToken.Path, "JSON reference {0} property must have a string or null value.".FormatWith(CultureInfo.InvariantCulture, JsonTypeReflector.RefPropertyName), null);
608  }
609 
610  string? reference = (string?)refProperty;
611 
612  if (reference != null)
613  {
614  JToken? additionalContent = refProperty.Next ?? refProperty.Previous;
615  if (additionalContent != null)
616  {
617  throw JsonSerializationException.Create(additionalContent, additionalContent.Path, "Additional content found in JSON reference object. A JSON reference object should only have a {0} property.".FormatWith(CultureInfo.InvariantCulture, JsonTypeReflector.RefPropertyName), null);
618  }
619 
620  newValue = Serializer.GetReferenceResolver().ResolveReference(this, reference);
621 
622  if (TraceWriter != null && TraceWriter.LevelFilter >= TraceLevel.Info)
623  {
624  TraceWriter.Trace(TraceLevel.Info, JsonPosition.FormatMessage(reader, reader.Path, "Resolved object reference '{0}' to {1}.".FormatWith(CultureInfo.InvariantCulture, reference, newValue.GetType())), null);
625  }
626 
627  reader.Skip();
628  return true;
629  }
630  }
631  JToken? typeToken = current[JsonTypeReflector.TypePropertyName];
632  if (typeToken != null)
633  {
634  string? qualifiedTypeName = (string?)typeToken;
635  JsonReader typeTokenReader = typeToken.CreateReader();
636  typeTokenReader.ReadAndAssert();
637  ResolveTypeName(typeTokenReader, ref objectType, ref contract, member, containerContract, containerMember, qualifiedTypeName!);
638 
639  JToken? valueToken = current[JsonTypeReflector.ValuePropertyName];
640  if (valueToken != null)
641  {
642  while (true)
643  {
644  reader.ReadAndAssert();
645  if (reader.TokenType == JsonToken.PropertyName)
646  {
647  if ((string)reader.Value! == JsonTypeReflector.ValuePropertyName)
648  {
649  return false;
650  }
651  }
652 
653  reader.ReadAndAssert();
654  reader.Skip();
655  }
656  }
657  }
658  JToken? idToken = current[JsonTypeReflector.IdPropertyName];
659  if (idToken != null)
660  {
661  id = (string?)idToken;
662  }
663  JToken? valuesToken = current[JsonTypeReflector.ArrayValuesPropertyName];
664  if (valuesToken != null)
665  {
666  JsonReader listReader = valuesToken.CreateReader();
667  listReader.ReadAndAssert();
668  newValue = CreateList(listReader, objectType, contract, member, existingValue, id);
669 
670  reader.Skip();
671  return true;
672  }
673  }
674 
675  reader.ReadAndAssert();
676  return false;
677  }

679  private bool ReadMetadataProperties(JsonReader reader, ref Type? objectType, ref JsonContract? contract, JsonProperty? member, JsonContainerContract? containerContract, JsonProperty? containerMember, object? existingValue, out object? newValue, out string? id)
680  {
681  id = null;
682  newValue = null;
683 
684  if (reader.TokenType == JsonToken.PropertyName)
685  {
686  string propertyName = reader.Value!.ToString();
687 
688  if (propertyName.Length > 0 && propertyName[0] == '$')
689  {
690  // read metadata properties
691  // $type, $id, $ref, etc
692  bool metadataProperty;
693 
694  do
695  {
696  propertyName = reader.Value!.ToString();
697 
698  if (string.Equals(propertyName, JsonTypeReflector.RefPropertyName, StringComparison.Ordinal))
699  {
700  reader.ReadAndAssert();
701  if (reader.TokenType != JsonToken.String && reader.TokenType != JsonToken.Null)
702  {
703  throw JsonSerializationException.Create(reader, "JSON reference {0} property must have a string or null value.".FormatWith(CultureInfo.InvariantCulture, JsonTypeReflector.RefPropertyName));
704  }
705 
706  string? reference = reader.Value?.ToString();
707 
708  reader.ReadAndAssert();
709 
710  if (reference != null)
711  {
712  if (reader.TokenType == JsonToken.PropertyName)
713  {
714  throw JsonSerializationException.Create(reader, "Additional content found in JSON reference object. A JSON reference object should only have a {0} property.".FormatWith(CultureInfo.InvariantCulture, JsonTypeReflector.RefPropertyName));
715  }
716 
717  newValue = Serializer.GetReferenceResolver().ResolveReference(this, reference);
718 
719  if (TraceWriter != null && TraceWriter.LevelFilter >= TraceLevel.Info)
720  {
721  TraceWriter.Trace(TraceLevel.Info, JsonPosition.FormatMessage(reader as IJsonLineInfo, reader.Path, "Resolved object reference '{0}' to {1}.".FormatWith(CultureInfo.InvariantCulture, reference, newValue!.GetType())), null);
722  }
723 
724  return true;
