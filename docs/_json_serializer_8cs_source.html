651  private static void ApplySerializerSettings(JsonSerializer serializer, JsonSerializerSettings settings)
652  {
653  if (!CollectionUtils.IsNullOrEmpty(settings.Converters))
654  {
655  // insert settings converters at the beginning so they take precedence
656  // if user wants to remove one of the default converters they will have to do it manually
657  for (int i = 0; i < settings.Converters.Count; i++)
658  {
659  serializer.Converters.Insert(i, settings.Converters[i]);
660  }
661  }
662 
663  // serializer specific
664  if (settings._typeNameHandling != null)
665  {
666  serializer.TypeNameHandling = settings.TypeNameHandling;
667  }

669  {

919  private void SetupReader(JsonReader reader, out CultureInfo? previousCulture, out DateTimeZoneHandling? previousDateTimeZoneHandling, out DateParseHandling? previousDateParseHandling, out FloatParseHandling? previousFloatParseHandling, out int? previousMaxDepth, out string? previousDateFormatString)
920  {

922  {
923  previousCulture = reader.Culture;
924  reader.Culture = _culture;
925  }
926  else
927  {
928  previousCulture = null;
929  }

932  {
933  previousDateTimeZoneHandling = reader.DateTimeZoneHandling;
934  reader.DateTimeZoneHandling = _dateTimeZoneHandling.GetValueOrDefault();
935  }
936  else

939  }

945  }

949  }
950

955  }
956  else

959  }
960 
961  if (_maxDepthSet && reader.MaxDepth != _maxDepth)

964  reader.MaxDepth = _maxDepth;
965  }
966  else

990  private void ResetReader(JsonReader reader, CultureInfo? previousCulture, DateTimeZoneHandling? previousDateTimeZoneHandling, DateParseHandling? previousDateParseHandling, FloatParseHandling? previousFloatParseHandling, int? previousMaxDepth, string? previousDateFormatString)
991  {
992  // reset reader back to previous options
993  if (previousCulture != null)
994  {
995  reader.Culture = previousCulture;
996  }
997  if (previousDateTimeZoneHandling != null)
998  {
999  reader.DateTimeZoneHandling = previousDateTimeZoneHandling.GetValueOrDefault();
1000  }
1001  if (previousDateParseHandling != null)
1002  {
1003  reader.DateParseHandling = previousDateParseHandling.GetValueOrDefault();
1004  }
1005  if (previousFloatParseHandling != null)
1006  {
1007  reader.FloatParseHandling = previousFloatParseHandling.GetValueOrDefault();
1008  }
1009  if (_maxDepthSet)
1010  {
1011  reader.MaxDepth = previousMaxDepth;
1012  }
1013  if (_dateFormatStringSet)
1014  {
1015  reader.DateFormatString = previousDateFormatString;
1016  }
