91  public new static async Task<JProperty> LoadAsync(JsonReader reader, JsonLoadSettings? settings, CancellationToken cancellationToken = default)
92  {
93  if (reader.TokenType == JsonToken.None)
94  {
95  if (!await reader.ReadAsync(cancellationToken).ConfigureAwait(false))
96  {
97  throw JsonReaderException.Create(reader, "Error reading JProperty from JsonReader.");
98  }
99  }
100 
101  await reader.MoveToContentAsync(cancellationToken).ConfigureAwait(false);
102 
103  if (reader.TokenType != JsonToken.PropertyName)
104  {
105  throw JsonReaderException.Create(reader, "Error reading JProperty from JsonReader. Current JsonReader item is not a property: {0}".FormatWith(CultureInfo.InvariantCulture, reader.TokenType));
106  }
107 
108  JProperty p = new JProperty((string)reader.Value!);
109  p.SetLineInfo(reader as IJsonLineInfo, settings);
110 
111  await p.ReadTokenFromAsync(reader, settings, cancellationToken).ConfigureAwait(false);
