101  // Build the trapezoidal map and query graph
105  private void Process()
106  {
107  foreach (Edge edge in edgeList)
108  {
109  List<Trapezoid> traps = queryGraph.FollowEdge(edge);
110 
111  // Remove trapezoids from trapezoidal Map
112  foreach (Trapezoid t in traps)
113  {
114  trapezoidalMap.Map.Remove(t);
115 
116  bool cp = t.Contains(edge.P);
117  bool cq = t.Contains(edge.Q);
118  Trapezoid[] tList;
119 
120  if (cp && cq)
121  {
122  tList = trapezoidalMap.Case1(t, edge);
123  queryGraph.Case1(t.Sink, edge, tList);
124  }
125  else if (cp && !cq)
126  {
127  tList = trapezoidalMap.Case2(t, edge);
128  queryGraph.Case2(t.Sink, edge, tList);
129  }
130  else if (!cp && !cq)
131  {
132  tList = trapezoidalMap.Case3(t, edge);
133  queryGraph.Case3(t.Sink, edge, tList);
134  }
135  else
136  {
137  tList = trapezoidalMap.Case4(t, edge);
138  queryGraph.Case4(t.Sink, edge, tList);
139  }
140 
141  // Add new trapezoids to map
142  foreach (Trapezoid y in tList)
143  {
144  trapezoidalMap.Map.Add(y);
145  }
146  }
147 
148  trapezoidalMap.Clear();
149  }
150 
151  // Mark outside trapezoids
152  foreach (Trapezoid t in trapezoidalMap.Map)
153  {
154  MarkOutside(t);
155  }
156 
157  // Collect interior trapezoids
158  foreach (Trapezoid t in trapezoidalMap.Map)
159  {
160  if (t.Inside)
161  {
162  Trapezoids.Add(t);
163  t.AddPoints();
164  }
165  }
166 
167  // Generate the triangles
168  CreateMountains();
169  }

171  // Build a list of x-monotone mountains
175  private void CreateMountains()
176  {
177  foreach (Edge edge in edgeList)
178  {
179  if (edge.MPoints.Count > 2)
180  {
181  MonotoneMountain mountain = new MonotoneMountain();
182 
183  // Sorting is a perfromance hit. Literature says this can be accomplised in
184  // linear time, although I don't see a way around using traditional methods
185  // when using a randomized incremental algorithm
186 
187  // Insertion sort is one of the fastest algorithms for sorting arrays containing
188  // fewer than ten elements, or for lists that are already mostly sorted.
189 
190  List<Point> points = new List<Point>(edge.MPoints);
191  points.Sort((p1, p2) => p1.X.CompareTo(p2.X));
192 
193  foreach (Point p in points)
194  {
195  mountain.Add(p);
196  }
197 
198  // Triangulate monotone mountain
199  mountain.Process();
200 
201  // Extract the triangles into a single list
202  foreach (List<Point> t in mountain.Triangles)
203  {
204  Triangles.Add(t);
205  }
206 
207  xMonoPoly.Add(mountain);
208  }
209  }
210  }

225  // Create segments and connect end points; update edge event pointer
231  private List<Edge> InitEdges(List<Point> points)
232  {
233  List<Edge> edges = new List<Edge>();
234 
235  for (int i = 0; i < points.Count - 1; i++)
236  {
237  edges.Add(new Edge(points[i], points[i + 1]));
238  }
239 
240  edges.Add(new Edge(points[0], points[points.Count - 1]));
241  return OrderSegments(edges);
242  }

249  private List<Edge> OrderSegments(List<Edge> edgeInput)
250  {
251  // Ignore vertical segments!
252  List<Edge> edges = new List<Edge>();
253 
254  foreach (Edge e in edgeInput)
255  {
256  Point p = ShearTransform(e.P);
257  Point q = ShearTransform(e.Q);
258 
259  // Point p must be to the left of point q
260  if (p.X > q.X)
261  {
262  edges.Add(new Edge(q, p));
263  }
264  else if (p.X < q.X)
265  {
266  edges.Add(new Edge(p, q));
267  }
268  }
269 
270  // Randomized triangulation improves performance
271  // See Seidel's paper, or O'Rourke's book, p. 57
272  Shuffle(edges);
273  return edges;
274  }
275 
281  private static void Shuffle<T>(IList<T> list)
282  {
283  int n = list.Count;
284  while (n > 1)
285  {
286  n--;
287  int k = RandomUtils.GetInt32(n + 1);
288  T value = list[k];
289  list[k] = list[n];
290  list[n] = value;
291  }
292  }
293 
294  // Prevents any two distinct endpoints from lying on a common vertical line, and avoiding
295  // the degenerate case. See Mark de Berg et al, Chapter 6.3
301  private Point ShearTransform(Point point) => new Point(point.X + sheer * point.Y, point.Y);
302  }
