416  internal void Update(ContactManager contactManager)
417  {
418  //Velcro: Do not try and update destroyed contacts
419  if (FixtureA == null || FixtureB == null)
420  {
421  return;
422  }
423 
424  Manifold oldManifold = Manifold;
425 
426  // Re-enable this contact.
427  Flags |= ContactFlags.EnabledFlag;
428 
429  bool touching;
430  bool wasTouching = IsTouching;
431 
432  bool sensor = FixtureA.IsSensor || FixtureB.IsSensor;
433 
434  Body bodyA = FixtureA.Body;
435  Body bodyB = FixtureB.Body;
436 
437  Transform xfA = bodyA.Xf;
438  Transform xfB = bodyB.Xf;
439 
440  // Is this contact a sensor?
441  if (sensor)
442  {
443  Shape shapeA = FixtureA.Shape;
444  Shape shapeB = FixtureB.Shape;
445  touching = Narrowphase.Collision.TestOverlap(shapeA, ChildIndexA, shapeB, ChildIndexB, ref xfA,
446  ref xfB);
447 
448  // Sensors don't generate manifolds.
449  manifold.PointCount = 0;
450  }
451  else
452  {
453  Evaluate(ref manifold, ref xfA, ref xfB);
454  touching = Manifold.PointCount > 0;
455 
456  // Match old contact ids to new contact ids and copy the
457  // stored impulses to warm start the solver.
458  for (int i = 0; i < Manifold.PointCount; ++i)
459  {
460  ManifoldPoint mp2 = Manifold.Points[i];
461  mp2.NormalImpulse = 0.0f;
462  mp2.TangentImpulse = 0.0f;
463  ContactId id2 = mp2.Id;
464 
465  for (int j = 0; j < oldManifold.PointCount; ++j)
466  {
467  ManifoldPoint mp1 = oldManifold.Points[j];
468 
469  if (mp1.Id.Key == id2.Key)
470  {
471  mp2.NormalImpulse = mp1.NormalImpulse;
472  mp2.TangentImpulse = mp1.TangentImpulse;
473  break;
474  }
475  }
476 
477  manifold.Points[i] = mp2;
478  }
479 
480  if (touching != wasTouching)
481  {
482  bodyA.Awake = true;
483  bodyB.Awake = true;
484  }
485  }
486 
487  if (touching)
488  {
489  Flags |= ContactFlags.TouchingFlag;
490  }
491  else
492  {
493  Flags &= ~ContactFlags.TouchingFlag;
494  }
495 
496  if ((wasTouching == false) && touching)
497  {
498  //Velcro: Report the collision to both fixtures:
499  FixtureA.OnCollision?.Invoke(FixtureA, FixtureB, this);
500  FixtureB.OnCollision?.Invoke(FixtureB, FixtureA, this);
501 
502  //Velcro: Report the collision to both bodies:
503  bodyA.OnCollision?.Invoke(FixtureA, FixtureB, this);
504  bodyB.OnCollision?.Invoke(FixtureB, FixtureA, this);
505 
506  // Call BeginContact. It can disable the contact as well.
507  contactManager.BeginContact?.Invoke(this);
508 
509  // Velcro: If the user disabled the contact (needed to exclude it in TOI solver) at any point by
510  // any of the callbacks, we need to mark it as not touching and call any separation
511  // callbacks for fixtures that didn't explicitly disable the collision.
512  if (!Enabled)
513  {
514  touching = false;
515  }
516  }
517 
518  if (wasTouching && !touching)
519  {
520  //Report the separation to both fixtures:
521  FixtureA.OnSeparation?.Invoke(FixtureA, FixtureB, this);
522  FixtureB.OnSeparation?.Invoke(FixtureB, FixtureA, this);
523 
524  //Report the separation to both bodies:
525  bodyA.OnSeparation?.Invoke(FixtureA, FixtureB, this);
526  bodyB.OnSeparation?.Invoke(FixtureB, FixtureA, this);
527 
528  contactManager.EndContact?.Invoke(this);
529  }
530 
531  if (!sensor && touching)
532  {
533  contactManager.PreSolve?.Invoke(this, ref oldManifold);
534  }
535  }

541  private void Evaluate(ref Manifold manifold, ref Transform transformA, ref Transform transformB)
542  {
543  switch (type)
544  {
545  case ContactType.Polygon:
546  CollidePolygon.CollidePolygons(ref manifold, (PolygonShape) FixtureA.Shape, ref transformA,
547  (PolygonShape) FixtureB.Shape, ref transformB);
548  break;
549  case ContactType.PolygonAndCircle:
550  CollideCircle.CollidePolygonAndCircle(ref manifold, (PolygonShape) FixtureA.Shape, ref transformA,
551  (CircleShape) FixtureB.Shape, ref transformB);
552  break;
553  case ContactType.EdgeAndCircle:
554  CollideEdge.CollideEdgeAndCircle(ref manifold, (EdgeShape) FixtureA.Shape, ref transformA,
555  (CircleShape) FixtureB.Shape, ref transformB);
556  break;
557  case ContactType.EdgeAndPolygon:
558  CollideEdge.CollideEdgeAndPolygon(ref manifold, (EdgeShape) FixtureA.Shape, ref transformA,
559  (PolygonShape) FixtureB.Shape, ref transformB);
560  break;
561  case ContactType.ChainAndCircle:
562  ChainShape chain = (ChainShape) FixtureA.Shape;
563  chain.GetChildEdge(Edge, ChildIndexA);
564  CollideEdge.CollideEdgeAndCircle(ref manifold, Edge, ref transformA, (CircleShape) FixtureB.Shape,
565  ref transformB);
566  break;
567  case ContactType.ChainAndPolygon:
568  ChainShape loop2 = (ChainShape) FixtureA.Shape;
569  loop2.GetChildEdge(Edge, ChildIndexA);
570  CollideEdge.CollideEdgeAndPolygon(ref manifold, Edge, ref transformA,
571  (PolygonShape) FixtureB.Shape, ref transformB);
572  break;
573  case ContactType.Circle:
574  CollideCircle.CollideCircles(ref manifold, (CircleShape) FixtureA.Shape, ref transformA,
575  (CircleShape) FixtureB.Shape, ref transformB);
576  break;
577  default:
578  throw new ArgumentException("You are using an unsupported contact type.");
579  }
580  }

590  internal static Contact Create(Fixture fixtureA, int indexA, Fixture fixtureB, int indexB)
591  {
592  ShapeType type1 = fixtureA.Shape.ShapeType;
593  ShapeType type2 = fixtureB.Shape.ShapeType;
594 
595  //Debug.Assert((ShapeType.Unknown < type1) && (type1 < ShapeType.TypeCount));
596  //Debug.Assert((ShapeType.Unknown < type2) && (type2 < ShapeType.TypeCount));
597 
598  Contact c;
599  Queue<Contact> pool = World.Current.ContactPool;
600  if (pool.Count > 0)
601  {
602  c = pool.Dequeue();
603  if ((type1 >= type2 || ((type1 == ShapeType.Edge) && (type2 == ShapeType.Polygon))) &&
604  !((type2 == ShapeType.Edge) && (type1 == ShapeType.Polygon)))
605  {
606  c.Reset(fixtureA, indexA, fixtureB, indexB);
607  }
608  else
609  {
610  c.Reset(fixtureB, indexB, fixtureA, indexA);
611  }
612  }
613  else
614  {
615  // Edge+Polygon is non-symmetrical due to the way Erin handles collision type registration.
616  if ((type1 >= type2 || ((type1 == ShapeType.Edge) && (type2 == ShapeType.Polygon))) &&
617  !((type2 == ShapeType.Edge) && (type1 == ShapeType.Polygon)))
618  {
619  c = new Contact(fixtureA, indexA, fixtureB, indexB);
620  }
621  else
622  {
623  c = new Contact(fixtureB, indexB, fixtureA, indexA);
624  }
625  }
626 
627  c.type = Registers[(int) type1, (int) type2];
628 
629  return c;
630  }
