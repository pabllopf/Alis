341  internal override void InitVelocityConstraints(ref SolverData data)
342  {
343  indexA = BodyA.IslandIndex;
344  indexB = BodyB.IslandIndex;
345  localCenterA = BodyA.Sweep.LocalCenter;
346  localCenterB = BodyB.Sweep.LocalCenter;
347  invMassA = BodyA.InvMass;
348  invMassB = BodyB.InvMass;
349  invIa = BodyA.InvI;
350  invIb = BodyB.InvI;
351 
352  Vector2 cA = data.Positions[indexA].C;
353  float aA = data.Positions[indexA].A;
354  Vector2 vA = data.Velocities[indexA].V;
355  float wA = data.Velocities[indexA].W;
356 
357  Vector2 cB = data.Positions[indexB].C;
358  float aB = data.Positions[indexB].A;
359  Vector2 vB = data.Velocities[indexB].V;
360  float wB = data.Velocities[indexB].W;
361 
362  Rot qA = new Rot(aA), qB = new Rot(aB);
363 
364  rA = MathUtils.Mul(ref qA, localAnchorA - localCenterA);
365  rB = MathUtils.Mul(ref qB, localAnchorB - localCenterB);
366  u = cB + rB - cA - rA;
367 
368  // Handle singularity.
369  currentLength = u.Length();
370  if (currentLength > Settings.LinearSlop)
371  {
372  u *= 1.0f / currentLength;
373  }
374  else
375  {
376  u = Vector2.Zero;
377  mass = 0.0f;
378  impulse = 0.0f;
379  lowerImpulse = 0.0f;
380  upperImpulse = 0.0f;
381  }
382 
383  float crAu = MathUtils.Cross(rA, u);
384  float crBu = MathUtils.Cross(rB, u);
385  float invMass = invMassA + invIa * crAu * crAu + invMassB + invIb * crBu * crBu;
386  mass = invMass != 0.0f ? 1.0f / invMass : 0.0f;
387 
388  if (stiffness > 0.0f && minLength < maxLength)
389  {
390  // soft
391  float c = currentLength - length;
392 
393  float d = damping;
394  float k = stiffness;
395 
396  // magic formulas
397  float h = data.Step.DeltaTime;
398 
399  // gamma = 1 / (h * (d + h * k))
400  // the extra factor of h in the denominator is since the lambda is an impulse, not a force
401  gamma = h * (d + h * k);
402  gamma = gamma != 0.0f ? 1.0f / gamma : 0.0f;
403  bias = c * h * k * gamma;
404 
405  invMass += gamma;
406  softMass = invMass != 0.0f ? 1.0f / invMass : 0.0f;
407  }
408  else
409  {
410  // rigid
411  gamma = 0.0f;
412  bias = 0.0f;
413  softMass = mass;
414  }
415 
416  if (data.Step.WarmStarting)
417  {
418  // Scale the impulse to support a variable time step.
419  impulse *= data.Step.DeltaTimeRatio;
420  lowerImpulse *= data.Step.DeltaTimeRatio;
421  upperImpulse *= data.Step.DeltaTimeRatio;
422 
423  Vector2 p = (impulse + lowerImpulse - upperImpulse) * u;
424  vA -= invMassA * p;
425  wA -= invIa * MathUtils.Cross(ref rA, ref p);
426  vB += invMassB * p;
427  wB += invIb * MathUtils.Cross(ref rB, ref p);
428  }
429  else
430  {
431  impulse = 0.0f;
432  }
433 
434  data.Velocities[indexA].V = vA;
435  data.Velocities[indexA].W = wA;
436  data.Velocities[indexB].V = vB;
437  data.Velocities[indexB].W = wB;
438  }

444  internal override void SolveVelocityConstraints(ref SolverData data)
445  {
446  Vector2 vA = data.Velocities[indexA].V;
447  float wA = data.Velocities[indexA].W;
448  Vector2 vB = data.Velocities[indexB].V;
449  float wB = data.Velocities[indexB].W;
450 
451  if (minLength < maxLength)
452  {
453  if (stiffness > 0.0f)
454  {
455  // Cdot = dot(u, v + cross(w, r))
456  Vector2 vpA = vA + MathUtils.Cross(wA, rA);
457  Vector2 vpB = vB + MathUtils.Cross(wB, rB);
458  float cdot = MathUtils.Dot(u, vpB - vpA);
459 
460  float impulse = -softMass * (cdot + bias + gamma * this.impulse);
461  this.impulse += impulse;
462 
463  Vector2 p = impulse * u;
464  vA -= invMassA * p;
465  wA -= invIa * MathUtils.Cross(rA, p);
466  vB += invMassB * p;
467  wB += invIb * MathUtils.Cross(rB, p);
468  }
469 
470  // lower
471  {
472  float c = currentLength - minLength;
473  float bias = MathUtils.Max(0.0f, c) * data.Step.InvertedDeltaTime;
474 
475  Vector2 vpA = vA + MathUtils.Cross(wA, rA);
476  Vector2 vpB = vB + MathUtils.Cross(wB, rB);
477  float cdot = MathUtils.Dot(u, vpB - vpA);
478 
479  float impulse = -mass * (cdot + bias);
480  float oldImpulse = lowerImpulse;
481  lowerImpulse = MathUtils.Max(0.0f, lowerImpulse + impulse);
482  impulse = lowerImpulse - oldImpulse;
483  Vector2 p = impulse * u;
484 
485  vA -= invMassA * p;
486  wA -= invIa * MathUtils.Cross(rA, p);
487  vB += invMassB * p;
488  wB += invIb * MathUtils.Cross(rB, p);
489  }
490 
491  // upper
492  {
493  float c = maxLength - currentLength;
494  float bias = MathUtils.Max(0.0f, c) * data.Step.InvertedDeltaTime;
495 
496  Vector2 vpA = vA + MathUtils.Cross(wA, rA);
497  Vector2 vpB = vB + MathUtils.Cross(wB, rB);
498  float cdot = MathUtils.Dot(u, vpA - vpB);
499 
500  float impulse = -mass * (cdot + bias);
501  float oldImpulse = upperImpulse;
502  upperImpulse = MathUtils.Max(0.0f, upperImpulse + impulse);
503  impulse = upperImpulse - oldImpulse;
504  Vector2 p = -impulse * u;
505 
506  vA -= invMassA * p;
507  wA -= invIa * MathUtils.Cross(rA, p);
508  vB += invMassB * p;
509  wB += invIb * MathUtils.Cross(rB, p);
510  }
511  }
512  else
513  {
514  // Equal limits
515 
516  // Cdot = dot(u, v + cross(w, r))
517  Vector2 vpA = vA + MathUtils.Cross(wA, rA);
518  Vector2 vpB = vB + MathUtils.Cross(wB, rB);
519  float cdot = MathUtils.Dot(u, vpB - vpA);
520 
521  float impulse = -mass * cdot;
522  this.impulse += impulse;
523 
524  Vector2 p = impulse * u;
525  vA -= invMassA * p;
526  wA -= invIa * MathUtils.Cross(rA, p);
527  vB += invMassB * p;
528  wB += invIb * MathUtils.Cross(rB, p);
529  }
530 
531  data.Velocities[indexA].V = vA;
532  data.Velocities[indexA].W = wA;
533  data.Velocities[indexB].V = vB;
534  data.Velocities[indexB].W = wB;
535  }

542  internal override bool SolvePositionConstraints(ref SolverData data)
543  {
544  Vector2 cA = data.Positions[indexA].C;
545  float aA = data.Positions[indexA].A;
546  Vector2 cB = data.Positions[indexB].C;
547  float aB = data.Positions[indexB].A;
548 
549  Rot qA = new Rot(aA), qB = new Rot(aB);
550 
551  Vector2 rA = MathUtils.Mul(qA, localAnchorA - localCenterA);
552  Vector2 rB = MathUtils.Mul(qB, localAnchorB - localCenterB);
553  Vector2 u = cB + rB - cA - rA;
554 
555  float length = MathUtils.Normalize(ref u);
556  float c;
557  if (minLength == maxLength)
558  {
559  c = length - minLength;
560  }
561  else if (length < minLength)
562  {
563  c = length - minLength;
564  }
565  else if (maxLength < length)
566  {
567  c = length - maxLength;
568  }
569  else
570  {
571  return true;
572  }
573 
574  float impulse = -mass * c;
575  Vector2 p = impulse * u;
576 
577  cA -= invMassA * p;
578  aA -= invIa * MathUtils.Cross(rA, p);
579  cB += invMassB * p;
580  aB += invIb * MathUtils.Cross(rB, p);
581 
582  data.Positions[indexA].C = cA;
583  data.Positions[indexA].A = aA;
584  data.Positions[indexB].C = cB;
585  data.Positions[indexB].A = aB;
586 
587  return MathUtils.Abs(c) < Settings.LinearSlop;
588  }
