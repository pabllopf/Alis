361  internal override void InitVelocityConstraints(ref SolverData data)
362  {
363  indexA = BodyA.IslandIndex;
364  indexB = BodyB.IslandIndex;
365  localCenterA = BodyA.Sweep.LocalCenter;
366  localCenterB = BodyB.Sweep.LocalCenter;
367  invMassA = BodyA.InvMass;
368  invMassB = BodyB.InvMass;
369  invIa = BodyA.InvI;
370  invIb = BodyB.InvI;
371 
372  Vector2F cA = data.Positions[indexA].C;
373  float aA = data.Positions[indexA].A;
374  Vector2F vA = data.Velocities[indexA].V;
375  float wA = data.Velocities[indexA].W;
376 
377  Vector2F cB = data.Positions[indexB].C;
378  float aB = data.Positions[indexB].A;
379  Vector2F vB = data.Velocities[indexB].V;
380  float wB = data.Velocities[indexB].W;
381 
382  Rotation qA = new Rotation(aA), qB = new Rotation(aB);
383 
384  rA = MathUtils.Mul(ref qA, localAnchorA - localCenterA);
385  rB = MathUtils.Mul(ref qB, localAnchorB - localCenterB);
386  u = cB + rB - cA - rA;
387 
388  // Handle singularity.
389  currentLength = u.Length();
390  if (currentLength > Settings.LinearSlop)
391  {
392  u *= 1.0f / currentLength;
393  }
394  else
395  {
396  u = Vector2F.Zero;
397  mass = 0.0f;
398  impulse = 0.0f;
399  lowerImpulse = 0.0f;
400  upperImpulse = 0.0f;
401  }
402 
403  float crAu = MathUtils.Cross(rA, u);
404  float crBu = MathUtils.Cross(rB, u);
405  float invMass = invMassA + invIa * crAu * crAu + invMassB + invIb * crBu * crBu;
406  mass = invMass != 0.0f ? 1.0f / invMass : 0.0f;
407 
408  if ((stiffness > 0.0f) && (minLength < maxLength))
409  {
410  // soft
411  float c = currentLength - lengthPrivate;
412 
413  float d = damping;
414  float k = stiffness;
415 
416  // magic formulas
417  float h = data.Step.DeltaTime;
418 
419  // gamma = 1 / (h * (d + h * k))
420  // the extra factor of h in the denominator is since the lambda is an impulse, not a force
421  gamma = h * (d + h * k);
422  gamma = gamma != 0.0f ? 1.0f / gamma : 0.0f;
423  bias = c * h * k * gamma;
424 
425  invMass += gamma;
426  softMass = invMass != 0.0f ? 1.0f / invMass : 0.0f;
427  }
428  else
429  {
430  // rigid
431  gamma = 0.0f;
432  bias = 0.0f;
433  softMass = mass;
434  }
435 
436  if (data.Step.WarmStarting)
437  {
438  // Scale the impulse to support a variable time step.
439  impulse *= data.Step.DeltaTimeRatio;
440  lowerImpulse *= data.Step.DeltaTimeRatio;
441  upperImpulse *= data.Step.DeltaTimeRatio;
442 
443  Vector2F p = (impulse + lowerImpulse - upperImpulse) * u;
444  vA -= invMassA * p;
445  wA -= invIa * MathUtils.Cross(ref rA, ref p);
446  vB += invMassB * p;
447  wB += invIb * MathUtils.Cross(ref rB, ref p);
448  }
449  else
450  {
451  impulse = 0.0f;
452  }
453 
454  data.Velocities[indexA].V = vA;
455  data.Velocities[indexA].W = wA;
456  data.Velocities[indexB].V = vB;
457  data.Velocities[indexB].W = wB;
458  }

464  internal override void SolveVelocityConstraints(ref SolverData data)
465  {
466  Vector2F vA = data.Velocities[indexA].V;
467  float wA = data.Velocities[indexA].W;
468  Vector2F vB = data.Velocities[indexB].V;
469  float wB = data.Velocities[indexB].W;
470 
471  if (minLength < maxLength)
472  {
473  if (stiffness > 0.0f)
474  {
475  // Cdot = dot(u, v + cross(w, r))
476  Vector2F vpA = vA + MathUtils.Cross(wA, rA);
477  Vector2F vpB = vB + MathUtils.Cross(wB, rB);
478  float cdot = MathUtils.Dot(u, vpB - vpA);
479 
480  float impulse = -softMass * (cdot + bias + gamma * this.impulse);
481  this.impulse += impulse;
482 
483  Vector2F p = impulse * u;
484  vA -= invMassA * p;
485  wA -= invIa * MathUtils.Cross(rA, p);
486  vB += invMassB * p;
487  wB += invIb * MathUtils.Cross(rB, p);
488  }
489 
490  // lower
491  {
492  float c = currentLength - minLength;
493  float bias = MathUtils.Max(0.0f, c) * data.Step.InvertedDeltaTime;
494 
495  Vector2F vpA = vA + MathUtils.Cross(wA, rA);
496  Vector2F vpB = vB + MathUtils.Cross(wB, rB);
497  float cdot = MathUtils.Dot(u, vpB - vpA);
498 
499  float impulse = -mass * (cdot + bias);
500  float oldImpulse = lowerImpulse;
501  lowerImpulse = MathUtils.Max(0.0f, lowerImpulse + impulse);
502  impulse = lowerImpulse - oldImpulse;
503  Vector2F p = impulse * u;
504 
505  vA -= invMassA * p;
506  wA -= invIa * MathUtils.Cross(rA, p);
507  vB += invMassB * p;
508  wB += invIb * MathUtils.Cross(rB, p);
509  }
510 
511  // upper
512  {
513  float c = maxLength - currentLength;
514  float bias = MathUtils.Max(0.0f, c) * data.Step.InvertedDeltaTime;
515 
516  Vector2F vpA = vA + MathUtils.Cross(wA, rA);
517  Vector2F vpB = vB + MathUtils.Cross(wB, rB);
518  float cdot = MathUtils.Dot(u, vpA - vpB);
519 
520  float impulse = -mass * (cdot + bias);
521  float oldImpulse = upperImpulse;
522  upperImpulse = MathUtils.Max(0.0f, upperImpulse + impulse);
523  impulse = upperImpulse - oldImpulse;
524  Vector2F p = -impulse * u;
525 
526  vA -= invMassA * p;
527  wA -= invIa * MathUtils.Cross(rA, p);
528  vB += invMassB * p;
529  wB += invIb * MathUtils.Cross(rB, p);
530  }
531  }
532  else
533  {
534  // Equal limits
535 
536  // Cdot = dot(u, v + cross(w, r))
537  Vector2F vpA = vA + MathUtils.Cross(wA, rA);
538  Vector2F vpB = vB + MathUtils.Cross(wB, rB);
539  float cdot = MathUtils.Dot(u, vpB - vpA);
540 
541  float impulse = -mass * cdot;
542  this.impulse += impulse;
543 
544  Vector2F p = impulse * u;
545  vA -= invMassA * p;
546  wA -= invIa * MathUtils.Cross(rA, p);
547  vB += invMassB * p;
548  wB += invIb * MathUtils.Cross(rB, p);
549  }
550 
551  data.Velocities[indexA].V = vA;
552  data.Velocities[indexA].W = wA;
553  data.Velocities[indexB].V = vB;
554  data.Velocities[indexB].W = wB;
555  }

562  internal override bool SolvePositionConstraints(ref SolverData data)
563  {
564  Vector2F cA = data.Positions[indexA].C;
565  float aA = data.Positions[indexA].A;
566  Vector2F cB = data.Positions[indexB].C;
567  float aB = data.Positions[indexB].A;
568 
569  Rotation qA = new Rotation(aA), qB = new Rotation(aB);
570 
571  Vector2F rA = MathUtils.Mul(qA, localAnchorA - localCenterA);
572  Vector2F rB = MathUtils.Mul(qB, localAnchorB - localCenterB);
573  Vector2F u = cB + rB - cA - rA;
574 
575  float length = MathUtils.Normalize(ref u);
576  float c;
577  if (minLength == maxLength)
578  {
579  c = length - minLength;
580  }
581  else if (length < minLength)
582  {
583  c = length - minLength;
584  }
585  else if (maxLength < length)
586  {
587  c = length - maxLength;
588  }
589  else
590  {
591  return true;
592  }
593 
594  float impulse = -mass * c;
595  Vector2F p = impulse * u;
596 
597  cA -= invMassA * p;
598  aA -= invIa * MathUtils.Cross(rA, p);
599  cB += invMassB * p;
600  aB += invIb * MathUtils.Cross(rB, p);
601 
602  data.Positions[indexA].C = cA;
603  data.Positions[indexA].A = aA;
604  data.Positions[indexB].C = cB;
605  data.Positions[indexB].A = aB;
606 
607  return MathUtils.Abs(c) < Settings.LinearSlop;
608  }
609  }
