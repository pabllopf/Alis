197  public bool RayCast(ref RayCastInput input, out RayCastOutput output, bool doInteriorCheck = true)
198  {
199  // From Real-time Collision Detection, p179.
200 
201  output = new RayCastOutput();
202 
203  float tmin = -MathConstants.MaxFloat;
204  float tmax = MathConstants.MaxFloat;
205 
206  Vector2 p = input.Point1;
207  Vector2 d = input.Point2 - input.Point1;
208  Vector2 absD = MathUtils.Abs(d);
209 
210  Vector2 normal = Vector2.Zero;
211 
212  for (int i = 0; i < 2; ++i)
213  {
214  float absDI = i == 0 ? absD.X : absD.Y;
215  float lowerBoundI = i == 0 ? LowerBound.X : LowerBound.Y;
216  float upperBoundI = i == 0 ? UpperBound.X : UpperBound.Y;
217  float pI = i == 0 ? p.X : p.Y;
218 
219  if (absDI < MathConstants.Epsilon)
220  {
221  // Parallel.
222  if (pI < lowerBoundI || upperBoundI < pI)
223  {
224  return false;
225  }
226  }
227  else
228  {
229  float dI = i == 0 ? d.X : d.Y;
230 
231  float invD = 1.0f / dI;
232  float t1 = (lowerBoundI - pI) * invD;
233  float t2 = (upperBoundI - pI) * invD;
234 
235  // Sign of the normal vector.
236  float s = -1.0f;
237 
238  if (t1 > t2)
239  {
240  MathUtils.Swap(ref t1, ref t2);
241  s = 1.0f;
242  }
243 
244  // Push the min up
245  if (t1 > tmin)
246  {
247  if (i == 0)
248  {
249  normal.X = s;
250  }
251  else
252  {
253  normal.Y = s;
254  }
255 
256  tmin = t1;
257  }
258 
259  // Pull the max down
260  tmax = Math.Min(tmax, t2);
261 
262  if (tmin > tmax)
263  {
264  return false;
265  }
266  }
267  }
268 
269  // Does the ray start inside the box?
270  // Does the ray intersect beyond the max fraction?
271  if (doInteriorCheck && (tmin < 0.0f || input.MaxFraction < tmin))
272  {
273  return false;
274  }
275 
276  // Intersection.
277  output.Fraction = tmin;
278  output.Normal = normal;
279  return true;
280  }
281  }
