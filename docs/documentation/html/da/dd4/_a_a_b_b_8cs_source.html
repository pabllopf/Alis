198  public bool RayCast(ref RayCastInput input, out RayCastOutput output, bool doInteriorCheck = true)
199  {
200  // From Real-time Collision Detection, p179.
201 
202  output = new RayCastOutput();
203 
204  float tmin = -float.MaxValue;
205  float tmax = float.MaxValue;
206 
207  Vector2F p = input.Point1;
208  Vector2F d = input.Point2 - input.Point1;
209  Vector2F absD = MathUtils.Abs(d);
210 
211  Vector2F normal = Vector2F.Zero;
212 
213  for (int i = 0; i < 2; ++i)
214  {
215  float absDi = i == 0 ? absD.X : absD.Y;
216  float lowerBoundI = i == 0 ? LowerBound.X : LowerBound.Y;
217  float upperBoundI = i == 0 ? UpperBound.X : UpperBound.Y;
218  float pI = i == 0 ? p.X : p.Y;
219 
220  if (absDi < Constant.Epsilon)
221  {
222  // Parallel.
223  if (pI < lowerBoundI || upperBoundI < pI)
224  {
225  return false;
226  }
227  }
228  else
229  {
230  float dI = i == 0 ? d.X : d.Y;
231 
232  float invD = 1.0f / dI;
233  float t1 = (lowerBoundI - pI) * invD;
234  float t2 = (upperBoundI - pI) * invD;
235 
236  // Sign of the normal vector.
237  float s = -1.0f;
238 
239  if (t1 > t2)
240  {
241  MathUtils.Swap(ref t1, ref t2);
242  s = 1.0f;
243  }
244 
245  // Push the min up
246  if (t1 > tmin)
247  {
248  if (i == 0)
249  {
250  normal.X = s;
251  }
252  else
253  {
254  normal.Y = s;
255  }
256 
257  tmin = t1;
258  }
259 
260  // Pull the max down
261  tmax = Math.Min(tmax, t2);
262 
263  if (tmin > tmax)
264  {
265  return false;
266  }
267  }
268  }
269 
270  // Does the ray start inside the box?
271  // Does the ray intersect beyond the max fraction?
272  if (doInteriorCheck && (tmin < 0.0f || input.Fraction < tmin))
273  {
274  return false;
275  }
276 
277  // Intersection.
278  output.Fraction = tmin;
279  output.Normal = normal;
280  return true;
281  }
282  }
