39  // Linear constraint (point-to-line)
40  // d = pB - pA = xB + rB - xA - rA
41  // C = dot(ay, d)
42  // Cdot = dot(d, cross(wA, ay)) + dot(ay, vB + cross(wB, rB) - vA - cross(wA, rA))
43  // = -dot(ay, vA) - dot(cross(d + rA, ay), wA) + dot(ay, vB) + dot(cross(rB, ay), vB)
44  // J = [-ay, -cross(d + rA, ay), ay, cross(rB, ay)]
45 
46  // Spring linear constraint
47  // C = dot(ax, d)
48  // Cdot = = -dot(ax, vA) - dot(cross(d + rA, ax), wA) + dot(ax, vB) + dot(cross(rB, ax), vB)
49  // J = [-ax -cross(d+rA, ax) ax cross(rB, ax)]
50 
51  // Motor rotational constraint
52  // Cdot = wB - wA
53  // J = [0 0 -1 0 0 1]

431  float wB = bB.AngularVelocity;

434  MathUtils.Dot(axis, vB + MathUtils.Cross(wB, rB) - vA - MathUtils.Cross(wA, rA));
435  return speed;

588  internal override void InitVelocityConstraints(ref SolverData data)
589  {
590  indexA = BodyA.IslandIndex;
591  indexB = BodyB.IslandIndex;
592  localCenterA = BodyA.Sweep.LocalCenter;
593  localCenterB = BodyB.Sweep.LocalCenter;
594  invMassA = BodyA.InvMass;
595  invMassB = BodyB.InvMass;
596  invIa = BodyA.InvI;
597  invIb = BodyB.InvI;
598 
599  float mA = invMassA, mB = invMassB;
600  float iA = invIa, iB = invIb;
601 
602  Vector2F cA = data.Positions[indexA].C;
603  float aA = data.Positions[indexA].A;
604  Vector2F vA = data.Velocities[indexA].V;
605  float wA = data.Velocities[indexA].W;
606 
607  Vector2F cB = data.Positions[indexB].C;
608  float aB = data.Positions[indexB].A;
609  Vector2F vB = data.Velocities[indexB].V;
610  float wB = data.Velocities[indexB].W;
611 
612  Rotation qA = new Rotation(aA), qB = new Rotation(aB);
613 
614  // Compute the effective masses.
615  Vector2F rA = MathUtils.Mul(qA, localAnchorA - localCenterA);
616  Vector2F rB = MathUtils.Mul(qB, localAnchorB - localCenterB);
617  Vector2F d = cB + rB - cA - rA;
618 
619  // Point to line constraint
620  {
621  ay = MathUtils.Mul(qA, localYAxisA);
622  sAy = MathUtils.Cross(d + rA, ay);
623  sBy = MathUtils.Cross(rB, ay);
624 
625  mass = mA + mB + iA * sAy * sAy + iB * sBy * sBy;
626 
627  if (mass > 0.0f)
628  {
629  mass = 1.0f / mass;
630  }
631  }
632 
633  // Spring constraint
634  ax = MathUtils.Mul(qA, localXAxisA);
635  sAx = MathUtils.Cross(d + rA, ax);
636  sBx = MathUtils.Cross(rB, ax);
637 
638  float invMass = mA + mB + iA * sAx * sAx + iB * sBx * sBx;
639  if (invMass > 0.0f)
640  {
641  axialMass = 1.0f / invMass;
642  }
643  else
644  {
645  axialMass = 0.0f;
646  }
647 
648  springMass = 0.0f;
649  bias = 0.0f;
650  gamma = 0.0f;
651 
652  if ((stiffness > 0.0f) && (invMass > 0.0f))
653  {
654  springMass = 1.0f / invMass;
655 
656  float c = MathUtils.Dot(d, ax);
657 
658  // magic formulas
659  float h = data.Step.DeltaTime;
660  gamma = h * (damping + h * stiffness);
661  if (gamma > 0.0f)
662  {
663  gamma = 1.0f / gamma;
664  }
665 
666  bias = c * h * stiffness * gamma;
667 
668  springMass = invMass + gamma;
669  if (springMass > 0.0f)
670  {
671  springMass = 1.0f / springMass;
672  }
673  }
674  else
675  {
676  springImpulse = 0.0f;
677  }
678 
679  if (enableLimit)
680  {
681  translation = MathUtils.Dot(ax, d);
682  }
683  else
684  {
685  lowerImpulse = 0.0f;
686  upperImpulse = 0.0f;
687  }
688 
689  if (enableMotor)
690  {
691  motorMass = iA + iB;
692  if (motorMass > 0.0f)
693  {
694  motorMass = 1.0f / motorMass;
695  }
696  }
697  else
698  {
699  motorMass = 0.0f;
700  motorImpulse = 0.0f;
701  }
702 
703  if (data.Step.WarmStarting)
704  {
705  // Account for variable time step.
706  impulse *= data.Step.DeltaTimeRatio;
707  springImpulse *= data.Step.DeltaTimeRatio;
708  motorImpulse *= data.Step.DeltaTimeRatio;
709 
710  float axialImpulse = springImpulse + lowerImpulse - upperImpulse;
711  Vector2F p = impulse * ay + axialImpulse * ax;
712  float la = impulse * sAy + axialImpulse * sAx + motorImpulse;
713  float lb = impulse * sBy + axialImpulse * sBx + motorImpulse;
714 
715  vA -= invMassA * p;
716  wA -= invIa * la;
717 
718  vB += invMassB * p;
719  wB += invIb * lb;
720  }
721  else
722  {
723  impulse = 0.0f;
724  springImpulse = 0.0f;
725  motorImpulse = 0.0f;
726  lowerImpulse = 0.0f;
727  upperImpulse = 0.0f;
728  }
729 
730  data.Velocities[indexA].V = vA;
731  data.Velocities[indexA].W = wA;
732  data.Velocities[indexB].V = vB;
733  data.Velocities[indexB].W = wB;
734  }

740  internal override void SolveVelocityConstraints(ref SolverData data)
741  {
742  float mA = invMassA, mB = invMassB;
743  float iA = invIa, iB = invIb;
744 
745  Vector2F vA = data.Velocities[indexA].V;
746  float wA = data.Velocities[indexA].W;
747  Vector2F vB = data.Velocities[indexB].V;
748  float wB = data.Velocities[indexB].W;
749 
750  // Solve spring constraint
751  {
752  float cdot = MathUtils.Dot(ax, vB - vA) + sBx * wB - sAx * wA;
753  float impulse = -springMass * (cdot + bias + gamma * springImpulse);
754  springImpulse += impulse;
755 
756  Vector2F p = impulse * ax;
757  float la = impulse * sAx;
758  float lb = impulse * sBx;
759 
760  vA -= mA * p;
761  wA -= iA * la;
762 
763  vB += mB * p;
764  wB += iB * lb;
765  }
766 
767  // Solve rotational motor constraint
768  {
769  float cdot = wB - wA - motorSpeed;
770  float impulse = -motorMass * cdot;
771 
772  float oldImpulse = motorImpulse;
773  float maxImpulse = data.Step.DeltaTime * maxMotorTorque;
774  motorImpulse = MathUtils.Clamp(motorImpulse + impulse, -maxImpulse, maxImpulse);
775  impulse = motorImpulse - oldImpulse;
776 
777  wA -= iA * impulse;
778  wB += iB * impulse;
779  }
780 
781  if (enableLimit)
782  {
783  // Lower limit
784  {
785  float c = translation - lowerTranslation;
786  float cdot = MathUtils.Dot(ax, vB - vA) + sBx * wB - sAx * wA;
787  float impulse = -axialMass * (cdot + MathUtils.Max(c, 0.0f) * data.Step.InvertedDeltaTime);
788  float oldImpulse = lowerImpulse;
789  lowerImpulse = MathUtils.Max(lowerImpulse + impulse, 0.0f);
790  impulse = lowerImpulse - oldImpulse;
791 
792  Vector2F p = impulse * ax;
793  float la = impulse * sAx;
794  float lb = impulse * sBx;
795 
796  vA -= mA * p;
797  wA -= iA * la;
798  vB += mB * p;
799  wB += iB * lb;
800  }
801 
802  // Upper limit
803  // Note: signs are flipped to keep C positive when the constraint is satisfied.
804  // This also keeps the impulse positive when the limit is active.
805  {
806  float c = upperTranslation - translation;
807  float cdot = MathUtils.Dot(ax, vA - vB) + sAx * wA - sBx * wB;
808  float impulse = -axialMass * (cdot + MathUtils.Max(c, 0.0f) * data.Step.InvertedDeltaTime);
809  float oldImpulse = upperImpulse;
810  upperImpulse = MathUtils.Max(upperImpulse + impulse, 0.0f);
811  impulse = upperImpulse - oldImpulse;
812 
813  Vector2F p = impulse * ax;
814  float la = impulse * sAx;
815  float lb = impulse * sBx;
816 
817  vA += mA * p;
818  wA += iA * la;
819  vB -= mB * p;
820  wB -= iB * lb;
821  }
822  }
823 
824  // Solve point to line constraint
825  {
826  float cdot = MathUtils.Dot(ay, vB - vA) + sBy * wB - sAy * wA;
827  float impulse = -mass * cdot;
828  this.impulse += impulse;
829 
830  Vector2F p = impulse * ay;
831  float la = impulse * sAy;
832  float lb = impulse * sBy;
833 
834  vA -= mA * p;
835  wA -= iA * la;
836 
837  vB += mB * p;
838  wB += iB * lb;
839  }
840 
841  data.Velocities[indexA].V = vA;
842  data.Velocities[indexA].W = wA;
843  data.Velocities[indexB].V = vB;
844  data.Velocities[indexB].W = wB;
845  }

852  internal override bool SolvePositionConstraints(ref SolverData data)
853  {
854  Vector2F cA = data.Positions[indexA].C;
855  float aA = data.Positions[indexA].A;
856  Vector2F cB = data.Positions[indexB].C;
857  float aB = data.Positions[indexB].A;
858 
859  float linearError = 0.0f;
860 
861  if (enableLimit)
862  {
863  Rotation qA = new Rotation(aA), qB = new Rotation(aB);
864 
865  Vector2F rA = MathUtils.Mul(qA, localAnchorA - localCenterA);
866  Vector2F rB = MathUtils.Mul(qB, localAnchorB - localCenterB);
867  Vector2F d = cB - cA + rB - rA;
868 
869  Vector2F ax = MathUtils.Mul(qA, localXAxisA);
870  float sAx = MathUtils.Cross(d + rA, this.ax);
871  float sBx = MathUtils.Cross(rB, this.ax);
872 
873  float c = 0.0f;
874  float translation = MathUtils.Dot(ax, d);
875  if (MathUtils.Abs(upperTranslation - lowerTranslation) < 2.0f * Settings.LinearSlop)
876  {
877  c = translation;
878  }
879  else if (translation <= lowerTranslation)
880  {
881  c = MathUtils.Min(translation - lowerTranslation, 0.0f);
882  }
883  else if (translation >= upperTranslation)
884  {
885  c = MathUtils.Max(translation - upperTranslation, 0.0f);
886  }
887 
888  if (c != 0.0f)
889  {
890  float invMass = invMassA + invMassB + invIa * sAx * sAx + invIb * sBx * sBx;
891  float impulse = 0.0f;
892  if (invMass != 0.0f)
893  {
894  impulse = -c / invMass;
895  }
896 
897  Vector2F p = impulse * ax;
898  float la = impulse * sAx;
899  float lb = impulse * sBx;
900 
901  cA -= invMassA * p;
902  aA -= invIa * la;
903  cB += invMassB * p;
904  aB += invIb * lb;
905 
906  linearError = MathUtils.Abs(c);
907  }
908  }
909 
910  // Solve perpendicular constraint
911  {
912  Rotation qA = new Rotation(aA), qB = new Rotation(aB);
913 
914  Vector2F rA = MathUtils.Mul(qA, localAnchorA - localCenterA);
915  Vector2F rB = MathUtils.Mul(qB, localAnchorB - localCenterB);
916  Vector2F d = cB - cA + rB - rA;
917 
918  Vector2F ay = MathUtils.Mul(qA, localYAxisA);
919 
920  float sAy = MathUtils.Cross(d + rA, ay);
921  float sBy = MathUtils.Cross(rB, ay);
922 
923  float c = MathUtils.Dot(d, ay);
924 
925  float invMass = invMassA + invMassB + invIa * this.sAy * this.sAy + invIb * this.sBy * this.sBy;
926 
927  float impulse = 0.0f;
928  if (invMass != 0.0f)
929  {
930  impulse = -c / invMass;
931  }
932 
933  Vector2F p = impulse * ay;
934  float la = impulse * sAy;
935  float lb = impulse * sBy;
936 
937  cA -= invMassA * p;
938  aA -= invIa * la;
939  cB += invMassB * p;
940  aB += invIb * lb;
941 
942  linearError = MathUtils.Max(linearError, MathUtils.Abs(c));
943  }
944 
945  data.Positions[indexA].C = cA;
946  data.Positions[indexA].A = aA;
947  data.Positions[indexB].C = cB;
948  data.Positions[indexB].A = aB;
949 
950  return linearError <= Settings.LinearSlop;
951  }
952  }
