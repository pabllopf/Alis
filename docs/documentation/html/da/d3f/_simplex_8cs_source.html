64  internal void ReadCache(ref SimplexCache cache, ref DistanceProxy proxyA, ref Transform transformA,
65  ref DistanceProxy proxyB, ref Transform transformB)
66  {
67  Debug.Assert(cache.Count <= 3);
68 
69  // Copy data from cache.
70  Count = cache.Count;
71  for (int i = 0; i < Count; ++i)
72  {
73  SimplexVertex v = V[i];
74  v.IndexA = cache.IndexA[i];
75  v.IndexB = cache.IndexB[i];
76  Vector2 wALocal = proxyA.Vertices[v.IndexA];
77  Vector2 wBLocal = proxyB.Vertices[v.IndexB];
78  v.Wa = MathUtils.Mul(ref transformA, wALocal);
79  v.Wb = MathUtils.Mul(ref transformB, wBLocal);
80  v.W = v.Wb - v.Wa;
81  v.A = 0.0f;
82  V[i] = v;
83  }
84 
85  // Compute the new simplex metric, if it is substantially different than
86  // old metric then flush the simplex.
87  if (Count > 1)
88  {
89  float metric1 = cache.Metric;
90  float metric2 = GetMetric();
91  if (metric2 < 0.5f * metric1 || 2.0f * metric1 < metric2 || metric2 < MathConstants.Epsilon)
92  {
93  // Reset the simplex.
94  Count = 0;
95  }
96  }
97 
98  // If the cache is empty or invalid ...
99  if (Count == 0)
100  {
101  SimplexVertex v = V[0];
102  v.IndexA = 0;
103  v.IndexB = 0;
104  Vector2 wALocal = proxyA.Vertices[0];
105  Vector2 wBLocal = proxyB.Vertices[0];
106  v.Wa = MathUtils.Mul(ref transformA, wALocal);
107  v.Wb = MathUtils.Mul(ref transformB, wBLocal);
108  v.W = v.Wb - v.Wa;
109  v.A = 1.0f;
110  V[0] = v;
111  Count = 1;
112  }
113  }

134  internal Vector2 GetSearchDirection()
135  {
136  switch (Count)
137  {
138  case 1:
139  return -V[0].W;
140 
141  case 2:

144  float sgn = MathUtils.Cross(e12, -V[0].W);

235  case 1:

238  case 2:

250  // Solve a line segment using barycentric coordinates.
251  //
252  // p = a1 * w1 + a2 * w2
253  // a1 + a2 = 1
254  //
255  // The vector from the origin to the closest point on the line is
256  // perpendicular to the line.
257  // e12 = w2 - w1
258  // dot(p, e) = 0
259  // a1 * dot(w1, e) + a2 * dot(w2, e) = 0
260  //
261  // 2-by-2 linear system
262  // [1 1 ][a1] = [1]
263  // [w1.e12 w2.e12][a2] = [0]
264  //
265  // Define
266  // d12_1 = dot(w2, e12)
267  // d12_2 = -dot(w1, e12)
268  // d12 = d12_1 + d12_2
269  //
270  // Solution
271  // a1 = d12_1 / d12
272  // a2 = d12_2 / d12
273 
277  internal void Solve2()
278  {
279  Vector2 w1 = V[0].W;
280  Vector2 w2 = V[1].W;
281  Vector2 e12 = w2 - w1;
282 
283  // w1 region
284  float d122 = -Vector2.Dot(w1, e12);
285  if (d122 <= 0.0f)
286  {
287  // a2 <= 0, so we clamp it to 0
288  V.Value0.A = 1.0f;
289  Count = 1;
290  return;
291  }
292 
293  // w2 region
294  float d121 = Vector2.Dot(w2, e12);
295  if (d121 <= 0.0f)
296  {
297  // a1 <= 0, so we clamp it to 0
298  V.Value1.A = 1.0f;
299  Count = 1;
300  V.Value0 = V.Value1;
301  return;
302  }
303 
304  // Must be in e12 region.
305  float invD12 = 1.0f / (d121 + d122);
306  V.Value0.A = d121 * invD12;
307  V.Value1.A = d122 * invD12;
308  Count = 2;
309  }

319  internal void Solve3()
320  {
321  Vector2 w1 = V[0].W;
322  Vector2 w2 = V[1].W;
323  Vector2 w3 = V[2].W;
324 
325  // Edge12
326  // [1 1 ][a1] = [1]
327  // [w1.e12 w2.e12][a2] = [0]
328  // a3 = 0
329  Vector2 e12 = w2 - w1;
330  float w1E12 = Vector2.Dot(w1, e12);
331  float w2E12 = Vector2.Dot(w2, e12);
332  float d121 = w2E12;
333  float d122 = -w1E12;
334 
335  // Edge13
336  // [1 1 ][a1] = [1]
337  // [w1.e13 w3.e13][a3] = [0]
338  // a2 = 0
339  Vector2 e13 = w3 - w1;
340  float w1E13 = Vector2.Dot(w1, e13);
341  float w3E13 = Vector2.Dot(w3, e13);
342  float d131 = w3E13;
343  float d132 = -w1E13;
344 
345  // Edge23
346  // [1 1 ][a2] = [1]
347  // [w2.e23 w3.e23][a3] = [0]
348  // a1 = 0
349  Vector2 e23 = w3 - w2;
350  float w2E23 = Vector2.Dot(w2, e23);
351  float w3E23 = Vector2.Dot(w3, e23);
352  float d231 = w3E23;
353  float d232 = -w2E23;
354 
355  // Triangle123
356  float n123 = MathUtils.Cross(e12, e13);
357 
358  float d1231 = n123 * MathUtils.Cross(w2, w3);
359  float d1232 = n123 * MathUtils.Cross(w3, w1);
360  float d1233 = n123 * MathUtils.Cross(w1, w2);
361 
362  // w1 region
363  if (d122 <= 0.0f && d132 <= 0.0f)
364  {
365  V.Value0.A = 1.0f;
366  Count = 1;
367  return;
368  }
369 
370  // e12
371  if (d121 > 0.0f && d122 > 0.0f && d1233 <= 0.0f)
372  {
373  float invD12 = 1.0f / (d121 + d122);
374  V.Value0.A = d121 * invD12;
375  V.Value1.A = d122 * invD12;
376  Count = 2;
377  return;
378  }
379 
380  // e13
381  if (d131 > 0.0f && d132 > 0.0f && d1232 <= 0.0f)
382  {
383  float invD13 = 1.0f / (d131 + d132);
384  V.Value0.A = d131 * invD13;
385  V.Value2.A = d132 * invD13;
386  Count = 2;
387  V.Value1 = V.Value2;
388  return;
389  }
390 
391  // w2 region
392  if (d121 <= 0.0f && d232 <= 0.0f)
393  {
394  V.Value1.A = 1.0f;
395  Count = 1;
396  V.Value0 = V.Value1;
397  return;
398  }
399 
400  // w3 region
401  if (d131 <= 0.0f && d231 <= 0.0f)
402  {
403  V.Value2.A = 1.0f;
404  Count = 1;
405  V.Value0 = V.Value2;
406  return;
407  }
408 
409  // e23
410  if (d231 > 0.0f && d232 > 0.0f && d1231 <= 0.0f)
411  {
412  float invD23 = 1.0f / (d231 + d232);
413  V.Value1.A = d231 * invD23;
414  V.Value2.A = d232 * invD23;
415  Count = 2;
416  V.Value0 = V.Value2;
417  return;
418  }
419 
420  // Must be in triangle123
421  float invD123 = 1.0f / (d1231 + d1232 + d1233);
422  V.Value0.A = d1231 * invD123;
423  V.Value1.A = d1232 * invD123;
424  V.Value2.A = d1233 * invD123;
425  Count = 3;
426  }
