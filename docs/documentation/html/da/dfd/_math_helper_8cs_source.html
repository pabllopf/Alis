88  public static float CatmullRom(float value1, float value2, float value3, float value4, float amount)
89  {
90  // Using formula from http://www.mvps.org/directx/articles/catmull/
91  // Internally using doubles not to lose precission
92  double amountSquared = amount * amount;
93  double amountCubed = amountSquared * amount;
94  return (float) (0.5 * (2.0 * value2 +
95  (value3 - value1) * amount +
96  (2.0 * value1 - 5.0 * value2 + 4.0 * value3 - value4) * amountSquared +
97  (3.0 * value2 - value1 - 3.0 * value3 + value4) * amountCubed));
98  }

142  public static float Hermite(float value1, float tangent1, float value2, float tangent2, float amount)
143  {
144  // All transformed to double not to lose precission
145  // Otherwise, for high numbers of param:amount the result is NaN instead of Infinity
146  double v1 = value1, v2 = value2, t1 = tangent1, t2 = tangent2, s = amount, result;
147  double sCubed = s * s * s;
148  double sSquared = s * s;
149 
150  if (amount == 0f)
151  {
152  result = value1;
153  }
154  else if (amount == 1f)
155  {
156  result = value2;
157  }
158  else
159  {
160  result = (2 * v1 - 2 * v2 + t2 + t1) * sCubed +
161  (3 * v2 - 3 * v1 - 2 * t1 - t2) * sSquared +
162  t1 * s +
163  v1;
164  }
165 
166  return (float) result;
167  }
