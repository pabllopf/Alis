54 
55  if (path.Closed)

74  public static void ConvertPathToPolygon(Path path, Body body, float density, int subdivisions)
75  {
76  if (!path.Closed)
77  {
78  throw new Exception("The path must be closed to convert to a polygon.");
79  }
80 
81  List<Vector2F> verts = path.GetVertices(subdivisions);
82 
83  List<Vertices> decomposedVerts =
84  Triangulate.ConvexPartition(new Vertices(verts), TriangulationAlgorithm.Bayazit);
85 
86  foreach (Vertices item in decomposedVerts)
87  {
88  body.AddFixture(new PolygonShape(item, density));
89  }
90  }

100  public static List<Body> EvenlyDistributeShapesAlongPath(World world, Path path, IEnumerable<Shape> shapes,

162  public static List<RevoluteJoint> AttachBodiesWithRevoluteJoint(World world, List<Body> bodies,
163  Vector2F localAnchorA, Vector2F localAnchorB, bool connectFirstAndLast, bool collideConnected)
164  {
165  List<RevoluteJoint> joints = new List<RevoluteJoint>(bodies.Count + 1);
166

168  {
169  RevoluteJoint joint = new RevoluteJoint(bodies[i], bodies[i - 1], localAnchorA, localAnchorB)
170  {
171  CollideConnected = collideConnected
172  };
173  world.AddJoint(joint);
174  joints.Add(joint);
175  }
176 
177  if (connectFirstAndLast)
178  {
179  RevoluteJoint lastjoint =
180  new RevoluteJoint(bodies[0], bodies[bodies.Count - 1], localAnchorA, localAnchorB)
181  {
182  CollideConnected = collideConnected
183  };
184  world.AddJoint(lastjoint);
185  joints.Add(lastjoint);
186  }
187 
188  return joints;
