63  public Dictionary<Body, Vector2> Activate(Vector2 pos, float radius, float force,
64  float maxForce = float.MaxValue)
65  {
66  HashSet<Body> affectedBodies = new HashSet<Body>();
67 
68  Aabb aabb;
69  aabb.LowerBound = pos - new Vector2(radius);
70  aabb.UpperBound = pos + new Vector2(radius);
71 
72  World.TestPointAllFixtures.ForEach(fixture =>
73  {
74  if (Vector2.Distance(fixture.Body.Position, pos) <= radius)
75  {
76  if (!affectedBodies.Contains(fixture.Body))
77  {
78  affectedBodies.Add(fixture.Body);
79  }
80  }
81  });
82 
83  return ApplyImpulse(pos, radius, force, maxForce, affectedBodies);
84  }

95  private Dictionary<Body, Vector2> ApplyImpulse(Vector2 pos, float radius, float force, float maxForce,
96  HashSet<Body> overlappingBodies)
97  {
98  Dictionary<Body, Vector2> forces = new Dictionary<Body, Vector2>(overlappingBodies.Count);
99 
100  foreach (Body overlappingBody in overlappingBodies)
101  {
102  if (IsActiveOn(overlappingBody))
103  {
104  float distance = Vector2.Distance(pos, overlappingBody.Position);
105  float forcePercent = GetPercent(distance, radius);
106 
107  Vector2 forceVector = pos - overlappingBody.Position;
108  forceVector *=
109  1f / (float) Math.Sqrt(forceVector.X * forceVector.X + forceVector.Y * forceVector.Y);
110  forceVector *= MathHelper.Min(force * forcePercent, maxForce);
111  forceVector *= -1;
112 
113  overlappingBody.ApplyLinearImpulse(forceVector);
114  forces.Add(overlappingBody, forceVector);
115  }
116  }
117 
118  return forces;
119  }

127  private float GetPercent(float distance, float radius)
128  {
129  //(1-(distance/radius))^power-1
130  float percent = (float) Math.Pow(1 - (distance - radius) / radius, Power) - 1;
131 
132  if (float.IsNaN(percent))
133  {
134  return 0f;
135  }
136 
137  return MathHelper.Clamp(percent, 0f, 1f);
138  }
