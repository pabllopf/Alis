470  [MethodImpl(MethodImplOptions.AggressiveInlining)]
471  public static Vector3F Transform(Vector3F value, Quaternion rotation)
472  {
473  float x2 = rotation.X + rotation.X;
474  float y2 = rotation.Y + rotation.Y;
475  float z2 = rotation.Z + rotation.Z;
476 
477  float wx2 = rotation.W * x2;
478  float wy2 = rotation.W * y2;
479  float wz2 = rotation.W * z2;
480  float xx2 = rotation.X * x2;
481  float xy2 = rotation.X * y2;
482  float xz2 = rotation.X * z2;
483  float yy2 = rotation.Y * y2;
484  float yz2 = rotation.Y * z2;
485  float zz2 = rotation.Z * z2;
486 
487  return new Vector3F(
488  value.X * (1.0f - yy2 - zz2) + value.Y * (xy2 - wz2) + value.Z * (xz2 + wy2),
489  value.X * (xy2 + wz2) + value.Y * (1.0f - xx2 - zz2) + value.Z * (yz2 - wx2),
490  value.X * (xz2 - wy2) + value.Y * (yz2 + wx2) + value.Z * (1.0f - xx2 - yy2)
491  );
492  }
