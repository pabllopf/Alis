416  [MethodImpl(MethodImplOptions.AggressiveInlining)]
417  public static Vector3F Transform(Vector3F value, Quaternion rotation)
418  {
419  float x2 = rotation.X + rotation.X;
420  float y2 = rotation.Y + rotation.Y;
421  float z2 = rotation.Z + rotation.Z;
422 
423  float wx2 = rotation.W * x2;
424  float wy2 = rotation.W * y2;
425  float wz2 = rotation.W * z2;
426  float xx2 = rotation.X * x2;
427  float xy2 = rotation.X * y2;
428  float xz2 = rotation.X * z2;
429  float yy2 = rotation.Y * y2;
430  float yz2 = rotation.Y * z2;
431  float zz2 = rotation.Z * z2;
432 
433  return new Vector3F(
434  value.X * (1.0f - yy2 - zz2) + value.Y * (xy2 - wz2) + value.Z * (xz2 + wy2),
435  value.X * (xy2 + wz2) + value.Y * (1.0f - xx2 - zz2) + value.Z * (yz2 - wx2),
436  value.X * (xz2 - wy2) + value.Y * (yz2 + wx2) + value.Z * (1.0f - xx2 - yy2)
437  );
438  }
