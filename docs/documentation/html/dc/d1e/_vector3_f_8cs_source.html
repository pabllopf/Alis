469  [MethodImpl(MethodImplOptions.AggressiveInlining)]
470  public static Vector3F Transform(Vector3F value, Quaternion rotation)
471  {
472  float x2 = rotation.X + rotation.X;
473  float y2 = rotation.Y + rotation.Y;
474  float z2 = rotation.Z + rotation.Z;
475 
476  float wx2 = rotation.W * x2;
477  float wy2 = rotation.W * y2;
478  float wz2 = rotation.W * z2;
479  float xx2 = rotation.X * x2;
480  float xy2 = rotation.X * y2;
481  float xz2 = rotation.X * z2;
482  float yy2 = rotation.Y * y2;
483  float yz2 = rotation.Y * z2;
484  float zz2 = rotation.Z * z2;
485 
486  return new Vector3F(
487  value.X * (1.0f - yy2 - zz2) + value.Y * (xy2 - wz2) + value.Z * (xz2 + wy2),
488  value.X * (xy2 + wz2) + value.Y * (1.0f - xx2 - zz2) + value.Z * (yz2 - wx2),
489  value.X * (xz2 - wy2) + value.Y * (yz2 + wx2) + value.Z * (1.0f - xx2 - yy2)
490  );
491  }
