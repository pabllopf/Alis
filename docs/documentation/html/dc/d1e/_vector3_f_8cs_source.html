401  [MethodImpl(MethodImplOptions.AggressiveInlining)]
402  public static Vector3F Transform(Vector3F value, Quaternion rotation)
403  {
404  float x2 = rotation.X + rotation.X;
405  float y2 = rotation.Y + rotation.Y;
406  float z2 = rotation.Z + rotation.Z;
407 
408  float wx2 = rotation.W * x2;
409  float wy2 = rotation.W * y2;
410  float wz2 = rotation.W * z2;
411  float xx2 = rotation.X * x2;
412  float xy2 = rotation.X * y2;
413  float xz2 = rotation.X * z2;
414  float yy2 = rotation.Y * y2;
415  float yz2 = rotation.Y * z2;
416  float zz2 = rotation.Z * z2;
417 
418  return new Vector3F(
419  value.X * (1.0f - yy2 - zz2) + value.Y * (xy2 - wz2) + value.Z * (xz2 + wy2),
420  value.X * (xy2 + wz2) + value.Y * (1.0f - xx2 - zz2) + value.Z * (yz2 - wx2),
421  value.X * (xz2 - wy2) + value.Y * (yz2 + wx2) + value.Z * (1.0f - xx2 - yy2)
422  );
423  }
