208  private void Solve()
209  {
210  // Clear all the island flags.
211  Bodies.ForEach(i => i.ClearFlags());
212 
213  // DisableIslandFlag for all joints.
214  Joints.ForEach(i => i.DisableIslandFlag());
215 
216  // Clear all flags of the contacts.
217  ContactManager.ClearFlags();
218 
219  // Island solving.
220  island.Solve(TimeStepGlobal, Gravity, true, ContactManager, Bodies);
221 
222  // Synchronize fixtures, check for out of range bodies.
223  Bodies.ForEach(i => i.CheckOutRange());
224 
225  // Look for new contacts.
226  ContactManager.FindNewContacts();
227  }

252  private void SolveToiEvents()
253  {
254  while (true)
255  {
256  // Reset minAlpha to 1.0f
257  float minAlpha = 1.0f;
258 
259  // Find the first TOI contact.
260  Contact minContact = ContactManager.GetTheMinContact(minAlpha);
261 
262  minAlpha = ContactManager.CalculateMinAlpha();
263 
264  if (minContact == null || IsMinAlphaGreaterThanEpsilon(minAlpha))
265  {
266  // No more TOI events. Done!
267  return;
268  }
269 
270  // Advance the bodies to the TOI.
271  Body[] bodies = BodyHelper.AdvanceBody(ContactManager, island, minContact, minAlpha);
272 
273  // Solve the TOI island.
274  SolveToiIsland(minAlpha, bodies[0].IslandIndex, bodies[1].IslandIndex);
275 
276  // Reset island flags and synchronize broad-phase proxies.
277  SynchronizeIslandBodies();
278 
279  // Commit fixture proxy movements to the broad-phase so that new contacts are created.
280  // Also, some contacts can be destroyed.
281  FindNewContacts();
282  }
283  }
284
