487  public void Step(float dt, int velocityIterations = 8, int positionIterations = 3)
488  {
489  //Velcro: We support disabling the world
490  if (!Enabled)
491  {
492  return;
493  }
494 
495  //Velcro: We reuse the timers to avoid generating garbage
496  Stopwatch stepTimer = timerPool.GetFromPool(true);
497 
498  {
499  //Velcro: We support add/removal of objects live in the engine.
500  Stopwatch timer = timerPool.GetFromPool(true);
501  ProcessChanges();
502  profile.AddRemoveTime = timer.ElapsedTicks;
503  timerPool.ReturnToPool(timer);
504  }
505 
506  // If new fixtures were added, we need to find the new contacts.
507  if (NewContacts)
508  {
509  //Velcro: We measure how much time is spent on finding new contacts
510  Stopwatch timer = timerPool.GetFromPool(true);
511  ContactManager.FindNewContacts();
512  NewContacts = false;
513  profile.NewContactsTime = timer.ElapsedTicks;
514  timerPool.ReturnToPool(timer);
515  }
516 
517  IsLocked = true;
518 
519  TimeStep step = new TimeStep();
520  step.DeltaTime = dt;
521  step.VelocityIterations = velocityIterations;
522  step.PositionIterations = positionIterations;
523  step.WarmStarting = WarmStartingEnabled;
524  if (dt > 0.0f)
525  {
526  step.InvertedDeltaTime = 1.0f / dt;
527  }
528  else
529  {
530  step.InvertedDeltaTime = 0.0f;
531  }
532 
533  step.DeltaTimeRatio = invDt0 * dt;
534 
535  {
536  //Velcro: We have the concept of controllers. We update them here
537  Stopwatch timer = timerPool.GetFromPool(true);
538  for (int i = 0; i < ControllerList.Count; i++)
539  {
540  ControllerList[i].Update(dt);
541  }
542 
543  profile.ControllersUpdateTime = timer.ElapsedTicks;
544  timerPool.ReturnToPool(timer);
545  }
546 
547  // Update contacts. This is where some contacts are destroyed.
548  {
549  Stopwatch timer = timerPool.GetFromPool(true);
550  ContactManager.Collide();
551  profile.Collide = timer.ElapsedTicks;
552  timerPool.ReturnToPool(timer);
553  }
554 
555  // Integrate velocities, solve velocity constraints, and integrate positions.
556  if (stepComplete && step.DeltaTime > 0.0f)
557  {
558  Stopwatch timer = timerPool.GetFromPool(true);
559  Solve(ref step);
560  profile.Solve = timer.ElapsedTicks;
561  timerPool.ReturnToPool(timer);
562  }
563 
564  // Handle TOI events.
565  if (ContinuousPhysicsEnabled && step.DeltaTime > 0.0f)
566  {
567  Stopwatch timer = timerPool.GetFromPool(true);
568  SolveToi(ref step);
569  profile.SolveToi = timer.ElapsedTicks;
570  timerPool.ReturnToPool(timer);
571  }
572 
573  if (step.DeltaTime > 0.0f)
574  {
575  invDt0 = step.InvertedDeltaTime;
576  }
577 
578  if (Settings.AutoClearForces)
579  {
580  ClearForces();
581  }
582 
583  {
584  //Velcro: We support breakable bodies. We update them here.
585  Stopwatch timer = timerPool.GetFromPool(true);
586 
587  for (int i = 0; i < BreakableBodyList.Count; i++)
588  {
589  BreakableBodyList[i].Update();
590  }
591 
592  profile.BreakableBodies = timer.ElapsedTicks;
593  timerPool.ReturnToPool(timer);
594  }
595 
596  IsLocked = false;
597 
598  profile.Step = stepTimer.ElapsedTicks;
599  timerPool.ReturnToPool(stepTimer);
600  }

857  private void Solve(ref TimeStep step)
858  {
859  profile.SolveInit = 0;
860  profile.SolveVelocity = 0;
861  profile.SolvePosition = 0;
862 
863  // Size the island for the worst case.
864  Island.Reset(BodyList.Count,
865  ContactManager.ContactCounter,
866  JointList.Count,
867  ContactManager);
868 
869  // Clear all the island flags.
870  foreach (Body b in BodyList)
871  {
872  b.Flags &= ~BodyFlags.IslandFlag;
873  }
874 
875  for (Contact c = ContactManager.ContactList; c != null; c = c.Next)
876  {
877  c.Flags &= ~ContactFlags.IslandFlag;
878  }
879 
880  foreach (Joint j in JointList)
881  {
882  j.IslandFlag = false;
883  }
884 
885  // Build and simulate all awake islands.
886  int stackSize = BodyList.Count;
887  if (stackSize > stack.Length)
888  {
889  stack = new Body[Math.Max(stack.Length * 2, stackSize)];
890  }
891 
892  for (int index = BodyList.Count - 1; index >= 0; index--)
893  {
894  Body seed = BodyList[index];
895  if ((seed.Flags & BodyFlags.IslandFlag) == BodyFlags.IslandFlag)
896  {
897  continue;
898  }
899 
900  if (!seed.Awake || !seed.Enabled)
901  {
902  continue;
903  }
904 
905  // The seed can be dynamic or kinematic.
906  if (seed.BodyType == BodyType.Static)
907  {
908  continue;
909  }
910 
911  // Reset island and stack.
912  Island.Clear();
913  int stackCount = 0;
914  stack[stackCount++] = seed;
915 
916  seed.Flags |= BodyFlags.IslandFlag;
917 
918  // Perform a depth first search (DFS) on the constraint graph.
919  while (stackCount > 0)
920  {
921  // Grab the next body off the stack and add it to the island.
922  Body b = stack[--stackCount];
923  Debug.Assert(b.Enabled);
924  Island.Add(b);
925 
926  // To keep islands as small as possible, we don't
927  // propagate islands across static bodies.
928  if (b.BodyType == BodyType.Static)
929  {
930  continue;
931  }
932 
933  // Make sure the body is awake (without resetting sleep timer).
934  b.Flags |= BodyFlags.AwakeFlag;
935 
936  // Search all contacts connected to this body.
937  for (ContactEdge ce = b.ContactList; ce != null; ce = ce.Next)
938  {
939  Contact contact = ce.Contact;
940 
941  // Has this contact already been added to an island?
942  if (contact.IslandFlag)
943  {
944  continue;
945  }
946 
947  // Is this contact solid and touching?
948  if (!contact.Enabled || !contact.IsTouching)
949  {
950  continue;
951  }
952 
953  // Skip sensors.
954  bool sensorA = contact.FixtureA.IsSensor;
955  bool sensorB = contact.FixtureB.IsSensor;
956  if (sensorA || sensorB)
957  {
958  continue;
959  }
960 
961  Island.Add(contact);
962  contact.Flags |= ContactFlags.IslandFlag;
963 
964  Body other = ce.Other;
965 
966  // Was the other body already added to this island?
967  if (other.IsIsland)
968  {
969  continue;
970  }
971 
972  Debug.Assert(stackCount < stackSize);
973  stack[stackCount++] = other;
974  other.Flags |= BodyFlags.IslandFlag;
975  }
976 
977  // Search all joints connect to this body.
978  for (JointEdge je = b.JointList; je != null; je = je.Next)
979  {
980  if (je.Joint.IslandFlag)
981  {
982  continue;
983  }
984 
985  Body other = je.Other;
986 
987  // WIP David
988  //Enter here when it's a non-fixed joint. Non-fixed joints have a other body.
989  if (other != null)
990  {
991  // Don't simulate joints connected to inactive bodies.
992  if (!other.Enabled)
993  {
994  continue;
995  }
996 
997  Island.Add(je.Joint);
998  je.Joint.IslandFlag = true;
999 
1000  if (other.IsIsland)
1001  {
1002  continue;
1003  }
1004 
1005  Debug.Assert(stackCount < stackSize);
1006  stack[stackCount++] = other;
1007 
1008  other.Flags |= BodyFlags.IslandFlag;
1009  }
1010  else
1011  {
1012  Island.Add(je.Joint);
1013  je.Joint.IslandFlag = true;
1014  }
1015  }
1016  }
1017 
1018  Profile profile = new Profile();
1019  Island.Solve(ref profile, ref step, ref gravity, SleepingAllowed);
1020  this.profile.SolveInit += profile.SolveInit;
1021  this.profile.SolveVelocity += profile.SolveVelocity;
1022  this.profile.SolvePosition += profile.SolvePosition;
1023 
1024  // Post solve cleanup.
1025  for (int i = 0; i < Island.BodyCount; ++i)
1026  {
1027  // Allow static bodies to participate in other islands.
1028  Body b = Island.Bodies[i];
1029  if (b.BodyType == BodyType.Static)
1030  {
1031  b.Flags &= ~BodyFlags.IslandFlag;
1032  }
1033  }
1034  }
1035 
1036  {
1037  Stopwatch timer = timerPool.GetFromPool(true);
1038 
1039  // Synchronize fixtures, check for out of range bodies.
1040  foreach (Body b in BodyList)
1041  {
1042  // If a body was not in an island then it did not move.
1043  if ((b.Flags & BodyFlags.IslandFlag) == 0)
1044  {
1045  continue;
1046  }
1047 
1048  if (b.BodyType == BodyType.Static)
1049  {
1050  continue;
1051  }
1052 
1053  // Update fixtures (for broad-phase).
1054  b.SynchronizeFixtures();
1055  }
1056 
1057  // Look for new contacts.
1058  ContactManager.FindNewContacts();
1059  profile.Broadphase = timer.ElapsedTicks;
1060  timerPool.ReturnToPool(timer);
1061  }
1062  }

1068  private void SolveToi(ref TimeStep step)
1069  {
1070  Island.Reset(2 * Settings.MaxToiContacts, Settings.MaxToiContacts, 0, ContactManager);
1071
