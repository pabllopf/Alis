170  public void Step(float dt, int velocityIterations = 8, int positionIterations = 3)
171  {
172  // We need to find the new contacts
173  ContactManager.FindNewContacts();
174 
175  // Define the time step
176  TimeStep.DeltaTime = dt;
177  TimeStep.VelocityIterations = velocityIterations;
178  TimeStep.PositionIterations = positionIterations;
179  TimeStep.InvertedDeltaTime = dt > 0.0f ? 1.0f / dt : 0.0f;
180  TimeStep.DeltaTimeRatio = TimeStep.InvertedDeltaTimeZero * dt;
181 
182  // Update the custom controllers
183  Controllers.ForEach(i => i.Update(dt));
184 
185  // Update contacts. This is where some contacts are destroyed.
186  ContactManager.Collide();
187 
188  // Integrate velocities, solve velocity constraints, and integrate positions.
189  Solve();
190 
191  // Handle TOI events.
192  SolveToi();
193 
194  // Time step invert time
195  TimeStep.InvertedDeltaTimeZero = TimeStep.DeltaTime > 0.0f ? TimeStep.InvertedDeltaTime : TimeStep.InvertedDeltaTimeZero;
196 
197  // Clear all the forces
198  ClearForces();
199 
200  // Update breakable bodies
201  BreakableBodies.ForEach(i => i.Update());
202  }

207  private void Solve()
208  {
209  // Size the island for the worst case.
210  Island.Reset(Bodies.Count,
211  ContactManager.ContactCounter,
212  Joints.Count,
213  ContactManager);
214 
215  // Clear all the island flags.
216  Bodies.ForEach(i => i.Flags &= ~BodyFlags.IslandFlag);
217 
218  for (Contact c = ContactManager.ContactList; c != null; c = c.Next)
219  {
220  c.Flags &= ~ContactFlags.IslandFlag;
221  }
222 
223  Joints.ForEach(i => i.IslandFlag = false);
224 
225  // Build and simulate all awake islands.
226  int stackSize = Bodies.Count;
227  if (stackSize > BodiesStack.Length)
228  {
229  BodiesStack = new Body[Math.Max(BodiesStack.Length * 2, stackSize)];
230  }
231 
232  for (int index = Bodies.Count - 1; index >= 0; index--)
233  {
234  Body seed = Bodies[index];
235  if ((seed.Flags & BodyFlags.IslandFlag) == BodyFlags.IslandFlag)
236  {
237  continue;
238  }
239 
240  if (!seed.Awake || !seed.Enabled)
241  {
242  continue;
243  }
244 
245  // The seed can be dynamic or kinematic.
246  if (seed.BodyType == BodyType.Static)
247  {
248  continue;
249  }
250 
251  // Reset island and stack.
252  Island.Clear();
253  int stackCount = 0;
254  BodiesStack[stackCount++] = seed;
255 
256  seed.Flags |= BodyFlags.IslandFlag;
257 
258  // Perform a depth first search (DFS) on the constraint graph.
259  while (stackCount > 0)
260  {
261  // Grab the next body off the stack and add it to the island.
262  Body b = BodiesStack[--stackCount];
263  //Debug.Assert(b.Enabled);
264  Island.Add(b);
265 
266  // To keep islands as small as possible, we don't
267  // propagate islands across static bodies.
268  if (b.BodyType == BodyType.Static)
269  {
270  continue;
271  }
272 
273  // Make sure the body is awake (without resetting sleep timer).
274  b.Flags |= BodyFlags.AwakeFlag;
275 
276  // Search all contacts connected to this body.
277  for (ContactEdge ce = b.ContactList; ce != null; ce = ce.Next)
278  {
279  Contact contact = ce.Contact;
280 
281  // Has this contact already been added to an island?
282  if (contact.IslandFlag)
283  {
284  continue;
285  }
286 
287  // Is this contact solid and touching?
288  if (!contact.Enabled || !contact.IsTouching)
289  {
290  continue;
291  }
292 
293  // Skip sensors.
294  bool sensorA = contact.FixtureA.IsSensor;
295  bool sensorB = contact.FixtureB.IsSensor;
296  if (sensorA || sensorB)
297  {
298  continue;
299  }
300 
301  Island.Add(contact);
302  contact.Flags |= ContactFlags.IslandFlag;
303 
304  Body other = ce.Other;
305 
306  // Was the other body already added to this island?
307  if (other.IsIsland)
308  {
309  continue;
310  }
311 
312  //Debug.Assert(stackCount < stackSize);
313  BodiesStack[stackCount++] = other;
314  other.Flags |= BodyFlags.IslandFlag;
315  }
316 
317  // Search all joints connect to this body.
318  for (JointEdge je = b.JointList; je != null; je = je.Next)
319  {
320  if (je.Joint.IslandFlag)
321  {
322  continue;
323  }
324 
325  Body other = je.Other;
326 
327  // WIP David
328  //Enter here when it's a non-fixed joint. Non-fixed joints have a other body.
329  if (other != null)
330  {
331  // Don't simulate joints connected to inactive bodies.
332  if (!other.Enabled)
333  {
334  continue;
335  }
336 
337  Island.Add(je.Joint);
338  je.Joint.IslandFlag = true;
339 
340  if (other.IsIsland)
341  {
342  continue;
343  }
344 
345  //Debug.Assert(stackCount < stackSize);
346  BodiesStack[stackCount++] = other;
347 
348  other.Flags |= BodyFlags.IslandFlag;
349  }
350  else
351  {
352  Island.Add(je.Joint);
353  je.Joint.IslandFlag = true;
354  }
355  }
356  }
357 
358  Island.Solve(TimeStep, Gravity, true);
359 
360  // Post solve cleanup.
361  for (int i = 0; i < Island.BodyCount; ++i)
362  {
363  // Allow static bodies to participate in other islands.
364  Body b = Island.Bodies[i];
365  if (b.BodyType == BodyType.Static)
366  {
367  b.Flags &= ~BodyFlags.IslandFlag;
368  }
369  }
370  }
371 
372  {
373  // Synchronize fixtures, check for out of range bodies.
374  for (int index = 0; index < Bodies.Count; index++)
375  {
376  Body b = Bodies[index];
377  // If a body was not in an island then it did not move.
378  if ((b.Flags & BodyFlags.IslandFlag) == 0)
379  {
380  continue;
381  }
382 
383  if (b.BodyType == BodyType.Static)
384  {
385  continue;
386  }
387 
388  // Update fixtures (for broad-phase).
389  b.SynchronizeFixtures();
390  }
391 
392  // Look for new contacts.
393  ContactManager.FindNewContacts();
394  }
395  }

400  private void SolveToi()
401  {
402  Island.Reset(2 * Settings.ToiContacts, Settings.ToiContacts, 0, ContactManager);
403 
404  Bodies.ForEach(i =>
405  {
406  i.Flags &= ~BodyFlags.IslandFlag;
407  i.Sweep.Alpha0 = 0.0f;
408  });
409 
410  for (Contact c = ContactManager.ContactList; c != null; c = c.Next)
411  {
412  // Invalidate TOI
413  c.Flags &= ~(ContactFlags.ToiFlag | ContactFlags.IslandFlag);
414  c.ToiCount = 0;
415  c.Toi = 1.0f;
416  }
417 
418  // Find TOI events and solve them.
419  for (;;)
420  {
421  // Find the first TOI.
422  Contact minContact = null;
423  float minAlpha = 1.0f;
424 
425  for (Contact c = ContactManager.ContactList; c != null; c = c.Next)
426  {
427  // Is this contact disabled?
428  if (!c.Enabled)
429  {
430  continue;
431  }
432 
433  // Prevent excessive sub-stepping.
434  if (c.ToiCount > Settings.SubSteps)
435  {
436  continue;
437  }
438 
439  float alpha;
440  if (c.ToiFlag)
441  {
442  // This contact has a valid cached TOI.
443  alpha = c.Toi;
444  }
445  else
446  {
447  Fixture fA = c.FixtureA;
448  Fixture fB = c.FixtureB;
449 
450  // Is there a sensor?
451  if (fA.IsSensorPrivate || fB.IsSensorPrivate)
452  {
453  continue;
454  }
455 
456  Body bA = fA.Body;
457  Body bB = fB.Body;
458 
459  BodyType typeA = bA.BodyType;
460  BodyType typeB = bB.BodyType;
461  //Debug.Assert(typeA == BodyType.Dynamic || typeB == BodyType.Dynamic);
462 
463  bool activeA = bA.Awake && (typeA != BodyType.Static);
464  bool activeB = bB.Awake && (typeB != BodyType.Static);
465 
466  // Is at least one body active (awake and dynamic or kinematic)?
467  if (!activeA && !activeB)
468  {
469  continue;
470  }
471 
472  bool collideA = (bA.IsBullet || typeA != BodyType.Dynamic) &&
473  ((fA.IgnoreCcdWith & fB.CollisionCategories) == 0) && !bA.IgnoreCcd;
474  bool collideB = (bB.IsBullet || typeB != BodyType.Dynamic) &&
475  ((fB.IgnoreCcdWith & fA.CollisionCategories) == 0) && !bB.IgnoreCcd;
476 
477  // Are these two non-bullet dynamic bodies?
478  if (!collideA && !collideB)
479  {
480  continue;
481  }
482 
483  // Compute the TOI for this contact.
484  // Put the sweeps onto the same time interval.
485  float alpha0 = bA.Sweep.Alpha0;
486 
487  if (bA.Sweep.Alpha0 < bB.Sweep.Alpha0)
488  {
489  alpha0 = bB.Sweep.Alpha0;
490  bA.Sweep.Advance(alpha0);
491  }
492  else if (bB.Sweep.Alpha0 < bA.Sweep.Alpha0)
493  {
494  alpha0 = bA.Sweep.Alpha0;
495  bB.Sweep.Advance(alpha0);
496  }
497 
498  //Debug.Assert(alpha0 < 1.0f);
499 
500  // Compute the time of impact in interval [0, minTOI]
501  ToiInput input = new ToiInput
502  {
503  ProxyA = new DistanceProxy(fA.Shape, c.ChildIndexA),
504  ProxyB = new DistanceProxy(fB.Shape, c.ChildIndexB),
505  SweepA = bA.Sweep,
506  SweepB = bB.Sweep,
507  Max = 1.0f
508  };
509 
510  TimeOfImpact.CalculateTimeOfImpact(ref input, out ToiOutput output);
511 
512  // Beta is the fraction of the remaining portion of the .
513  float beta = output.T;
514  if (output.State == ToiOutputState.Touching)
515  {
516  alpha = Math.Min(alpha0 + (1.0f - alpha0) * beta, 1.0f);
517  }
518  else
519  {
520  alpha = 1.0f;
521  }
522 
523  c.Toi = alpha;
524  c.Flags &= ~ContactFlags.ToiFlag;
525  }
526 
527  if (alpha < minAlpha)
528  {
529  // This is the minimum TOI found so far.
530  minContact = c;
531  minAlpha = alpha;
532  }
533  }
534 
535  if (minContact == null || 1.0f - 10.0f * Constant.Epsilon < minAlpha)
536  {
537  // No more TOI events. Done!
538  //StepComplete = true;
539  break;
540  }
541 
542  // Advance the bodies to the TOI.
543  Fixture fA1 = minContact.FixtureA;
544  Fixture fB1 = minContact.FixtureB;
545  Body bA0 = fA1.Body;
546  Body bB0 = fB1.Body;
547 
548  Sweep backup1 = bA0.Sweep;
549  Sweep backup2 = bB0.Sweep;
550 
551  bA0.Advance(minAlpha);
552  bB0.Advance(minAlpha);
553 
554  // The TOI contact likely has some new contact points.
555  minContact.Update(ContactManager);
556  minContact.Flags &= ~ContactFlags.ToiFlag;
557  ++minContact.ToiCount;
558 
559  // Is the contact solid?
560  if (!minContact.Enabled || !minContact.IsTouching)
561  {
562  // Restore the sweeps.
563  minContact.Flags &= ~ContactFlags.EnabledFlag;
564  bA0.Sweep = backup1;
565  bB0.Sweep = backup2;
566  bA0.SynchronizeTransform();
567  bB0.SynchronizeTransform();
568  continue;
569  }
570 
571  bA0.Awake = true;
572  bB0.Awake = true;
573 
574  // Build the island
575  Island.Clear();
576  Island.Add(bA0);
577  Island.Add(bB0);
578  Island.Add(minContact);
579 
580  bA0.Flags |= BodyFlags.IslandFlag;
581  bB0.Flags |= BodyFlags.IslandFlag;
582  minContact.Flags &= ~ContactFlags.IslandFlag;
583 
584  // Get contacts on bodyA and bodyB.
585  Body[] bodies = {bA0, bB0};
586  for (int i = 0; i < 2; ++i)
587  {
588  Body body = bodies[i];
589  if (body.BodyType == BodyType.Dynamic)
590  {
591  for (ContactEdge ce = body.ContactList; ce != null; ce = ce.Next)
592  {
593  Contact contact = ce.Contact;
594 
595  if (Island.BodyCount == Island.BodyCapacity)
596  {
597  break;
598  }
599 
600  if (Island.ContactCount == Island.ContactCapacity)
601  {
602  break;
603  }
604 
605  // Has this contact already been added to the island?
606  if (contact.IslandFlag)
607  {
608  continue;
609  }
610 
611  // Only add static, kinematic, or bullet bodies.
612  Body other = ce.Other;
613  if ((other.BodyType == BodyType.Dynamic) &&
614  !body.IsBullet && !other.IsBullet)
615  {
616  continue;
617  }
618 
619  // Skip sensors.
620  bool sensorA = contact.FixtureA.IsSensorPrivate;
621  bool sensorB = contact.FixtureB.IsSensorPrivate;
622  if (sensorA || sensorB)
623  {
624  continue;
625  }
626 
627  // Tentatively advance the body to the TOI.
628  Sweep backup = other.Sweep;
629  if (!other.IsIsland)
630  {
631  other.Advance(minAlpha);
632  }
633 
634  // Update the contact points
635  contact.Update(ContactManager);
636 
637  // Was the contact disabled by the user?
638  if (!contact.Enabled)
639  {
640  other.Sweep = backup;
641  other.SynchronizeTransform();
642  continue;
643  }
644 
645  // Are there contact points?
646  if (!contact.IsTouching)
647  {
648  other.Sweep = backup;
649  other.SynchronizeTransform();
650  continue;
651  }
652 
653  // Add the contact to the island
654  minContact.Flags |= ContactFlags.IslandFlag;
655  Island.Add(contact);
656 
657  // Has the other body already been added to the island?
658  if (other.IsIsland)
659  {
660  continue;
661  }
662 
663  // Add the other body to the island.
664  other.Flags |= BodyFlags.IslandFlag;
665 
666  if (other.BodyType != BodyType.Static)
667  {
668  other.Awake = true;
669  }
670 
671  Island.Add(other);
672  }
673  }
674  }
675 
676  TimeStep subStep = new TimeStep();
677  subStep.DeltaTime = (1.0f - minAlpha) * TimeStep.DeltaTime;
678  subStep.InvertedDeltaTime = 1.0f / subStep.DeltaTime;
679  subStep.DeltaTimeRatio = 1.0f;
680  subStep.PositionIterations = 20;
681  subStep.VelocityIterations = TimeStep.VelocityIterations;
682  subStep.WarmStarting = false;
683  Island.SolveToi(ref subStep, bA0.IslandIndex, bB0.IslandIndex);
684 
685  // Reset island flags and synchronize broad-phase proxies.
686  for (int i = 0; i < Island.BodyCount; ++i)
687  {
688  Body body = Island.Bodies[i];
689  body.Flags &= ~BodyFlags.IslandFlag;
690 
691  if (body.BodyType != BodyType.Dynamic)
692  {
693  continue;
694  }
695 
696  body.SynchronizeFixtures();
697 
698  // Invalidate all contact TOIs on this displaced body.
699  for (ContactEdge ce = body.ContactList; ce != null; ce = ce.Next)
700  {
701  ce.Contact.Flags &= ~(ContactFlags.ToiFlag | ContactFlags.IslandFlag);
702  }
703  }
704 
705  // Commit fixture proxy movements to the broad-phase so that new contacts are created.
706  // Also, some contacts can be destroyed.
707  ContactManager.FindNewContacts();
708  }
709  }
