99  CpuAccessFlags cpuFlags = CpuAccessFlags.None;
100  ResourceUsage resourceUsage = ResourceUsage.Default;
101  BindFlags bindFlags = BindFlags.None;
102  ResourceOptionFlags optionFlags = ResourceOptionFlags.None;
103 
104  if ((description.Usage & TextureUsage.RenderTarget) == TextureUsage.RenderTarget)
105  {
106  bindFlags |= BindFlags.RenderTarget;
107  }
108  if ((description.Usage & TextureUsage.DepthStencil) == TextureUsage.DepthStencil)
109  {
110  bindFlags |= BindFlags.DepthStencil;
111  }
112  if ((description.Usage & TextureUsage.Sampled) == TextureUsage.Sampled)
113  {
114  bindFlags |= BindFlags.ShaderResource;
115  }
116  if ((description.Usage & TextureUsage.Storage) == TextureUsage.Storage)
117  {
118  bindFlags |= BindFlags.UnorderedAccess;
119  }
120  if ((description.Usage & TextureUsage.Staging) == TextureUsage.Staging)
121  {
122  cpuFlags = CpuAccessFlags.Read | CpuAccessFlags.Write;
123  resourceUsage = ResourceUsage.Staging;
124  }
125 
126  if ((description.Usage & TextureUsage.GenerateMipmaps) != 0)
127  {
128  bindFlags |= BindFlags.RenderTarget | BindFlags.ShaderResource;
129  optionFlags |= ResourceOptionFlags.GenerateMips;
130  }
131 
132  int arraySize = (int)description.ArrayLayers;
133  if ((description.Usage & TextureUsage.Cubemap) == TextureUsage.Cubemap)
134  {
135  optionFlags |= ResourceOptionFlags.TextureCube;
136  arraySize *= 6;
137  }
138 
139  int roundedWidth = (int)description.Width;
140  int roundedHeight = (int)description.Height;
141  if (FormatHelpers.IsCompressedFormat(description.Format))
142  {
143  roundedWidth = ((roundedWidth + 3) / 4) * 4;
144  roundedHeight = ((roundedHeight + 3) / 4) * 4;
145  }
146 
147  if (Type == TextureType.Texture1D)
