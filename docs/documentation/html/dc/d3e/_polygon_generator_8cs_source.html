57  for (int i = 0; i < vertexCount; i++)

63  radius += scale / 2 * (0.5 - Rng.NextDouble());
64  }
65  else if (i % 50 == 0)
66  {
67  radius += scale / 5 * (0.5 - Rng.NextDouble());
68  }

72  }

74  radius = radius > scale / 2 ? scale / 2 : radius;
75  radius = radius < scale / 10 ? scale / 10 : radius;
76  } while (radius < scale / 10 || radius > scale / 2);
77

79  radius * Math.Sin(Constant.TwoPi * i / vertexCount));
