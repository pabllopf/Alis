186  public Vector2F GetPosition(float time)
187  {
188  Vector2F temp;
189 
190  if (ControlPoints.Count < 2)
191  {
192  throw new Exception("You need at least 2 control points to calculate a position.");
193  }
194 
195  if (Closed)
196  {
197  Add(ControlPoints[0]);
198 
199  deltaT = 1f / (ControlPoints.Count - 1);
200 
201  int p = (int) (time / deltaT);
202 
203  // use a circular indexing system
204  int p0 = p - 1;
205  if (p0 < 0)
206  {
207  p0 += ControlPoints.Count - 1;
208  }
209  else if (p0 >= ControlPoints.Count - 1)
210  {
211  p0 -= ControlPoints.Count - 1;
212  }
213 
214  int p1 = p;
215  if (p1 < 0)
216  {
217  p1 += ControlPoints.Count - 1;
218  }
219  else if (p1 >= ControlPoints.Count - 1)
220  {
221  p1 -= ControlPoints.Count - 1;
222  }
223 
224  int p2 = p + 1;
225  if (p2 < 0)
226  {
227  p2 += ControlPoints.Count - 1;
228  }
229  else if (p2 >= ControlPoints.Count - 1)
230  {
231  p2 -= ControlPoints.Count - 1;
232  }
233 
234  int p3 = p + 2;
235  if (p3 < 0)
236  {
237  p3 += ControlPoints.Count - 1;
238  }
239  else if (p3 >= ControlPoints.Count - 1)
240  {
241  p3 -= ControlPoints.Count - 1;
242  }
243 
244  // relative time
245  float lt = (time - deltaT * p) / deltaT;
246 
247  temp = CatmullRom(ControlPoints[p0], ControlPoints[p1], ControlPoints[p2], ControlPoints[p3],
248  lt);
249 
250  RemoveAt(ControlPoints.Count - 1);
251  }
252  else
253  {
254  int p = (int) (time / deltaT);
255 
256  //
257  int p0 = p - 1;
258  if (p0 < 0)
259  {
260  p0 = 0;
261  }
262  else if (p0 >= ControlPoints.Count - 1)
263  {
264  p0 = ControlPoints.Count - 1;
265  }
266 
267  int p1 = p;
268  if (p1 < 0)
269  {
270  p1 = 0;
271  }
272  else if (p1 >= ControlPoints.Count - 1)
273  {
274  p1 = ControlPoints.Count - 1;
275  }
276 
277  int p2 = p + 1;
278  if (p2 < 0)
279  {
280  p2 = 0;
281  }
282  else if (p2 >= ControlPoints.Count - 1)
283  {
284  p2 = ControlPoints.Count - 1;
285  }
286 
287  int p3 = p + 2;
288  if (p3 < 0)
289  {
290  p3 = 0;
291  }
292  else if (p3 >= ControlPoints.Count - 1)
293  {
294  p3 = ControlPoints.Count - 1;
295  }

403  public List<Vector3F> SubdivideEvenly(int divisions)
404  {
405  List<Vector3F> verts = new List<Vector3F>();
406 
407  float length = GetLength();
408 
409  float deltaLength = length / divisions + 0.001f;
410  float t = 0.000f;
411 
412  // we always start at the first control point
413  Vector2F start = ControlPoints[0];
414  Vector2F end = GetPosition(t);
415 
416  // increment t until we are at half the distance
417  while (deltaLength * 0.5f >= Vector2F.Distance(start, end))
418  {
419  end = GetPosition(t);
420  t += 0.0001f;
421 
422  if (t >= 1f)
423  {
424  break;
425  }
426  }
427 
428  start = end;
429 
430  // for each box
431  for (int i = 1; i < divisions; i++)
432  {
433  Vector2F normal = GetPositionNormal(t);
434  float angle = (float) Math.Atan2(normal.Y, normal.X);
435 
436  Vector3F addVector = new Vector3F(new Vector2F(end.X, end.Y), angle);
437  verts.Add(addVector);
438 
439  // until we reach the correct distance down the curve
440  while (deltaLength >= Vector2F.Distance(start, end))
441  {
442  end = GetPosition(t);
443  t += 0.00001f;
444 
445  if (t >= 1f)
446  {
447  break;
448  }
449  }
450 
451  if (t >= 1f)
452  {
453  break;
454  }
455 
456  start = end;
457  }
458 
459  return verts;
460  }
