33  internal static VkSurfaceKHR CreateSurface(VkGraphicsDevice gd, VkInstance instance, SwapchainSource swapchainSource)
34  {
35  // TODO a null GD is passed from VkSurfaceSource.CreateSurface for compatibility
36  // when VkSurfaceInfo is removed we do not have to handle gd == null anymore
37  var doCheck = gd != null;
38 
39  if (doCheck && !gd.HasSurfaceExtension(CommonStrings.VK_KHR_SURFACE_EXTENSION_NAME))
40  throw new VeldridException($"The required instance extension was not available: {CommonStrings.VK_KHR_SURFACE_EXTENSION_NAME}");
41 
42  switch (swapchainSource)
43  {
44  case XlibSwapchainSource xlibSource:
45  if (doCheck && !gd.HasSurfaceExtension(CommonStrings.VK_KHR_XLIB_SURFACE_EXTENSION_NAME))
46  {
47  throw new VeldridException($"The required instance extension was not available: {CommonStrings.VK_KHR_XLIB_SURFACE_EXTENSION_NAME}");
48  }
49  return CreateXlib(instance, xlibSource);
50  case WaylandSwapchainSource waylandSource:
51  if (doCheck && !gd.HasSurfaceExtension(CommonStrings.VK_KHR_WAYLAND_SURFACE_EXTENSION_NAME))
52  {
53  throw new VeldridException($"The required instance extension was not available: {CommonStrings.VK_KHR_WAYLAND_SURFACE_EXTENSION_NAME}");
54  }
55  return CreateWayland(instance, waylandSource);
56  case Win32SwapchainSource win32Source:
57  if (doCheck && !gd.HasSurfaceExtension(CommonStrings.VK_KHR_WIN32_SURFACE_EXTENSION_NAME))
58  {
59  throw new VeldridException($"The required instance extension was not available: {CommonStrings.VK_KHR_WIN32_SURFACE_EXTENSION_NAME}");
60  }
61  return CreateWin32(instance, win32Source);
62  case AndroidSurfaceSwapchainSource androidSource:
63  if (doCheck && !gd.HasSurfaceExtension(CommonStrings.VK_KHR_ANDROID_SURFACE_EXTENSION_NAME))
64  {
65  throw new VeldridException($"The required instance extension was not available: {CommonStrings.VK_KHR_ANDROID_SURFACE_EXTENSION_NAME}");
66  }
67  return CreateAndroidSurface(instance, androidSource);
68  case NSWindowSwapchainSource nsWindowSource:
69  if (doCheck)
70  {
71  bool hasMetalExtension = gd.HasSurfaceExtension(CommonStrings.VK_EXT_METAL_SURFACE_EXTENSION_NAME);
72  if (hasMetalExtension || gd.HasSurfaceExtension(CommonStrings.VK_MVK_MACOS_SURFACE_EXTENSION_NAME))
73  {
74  return CreateNSWindowSurface(gd, instance, nsWindowSource, hasMetalExtension);
75  }
76  else
77  {
78  throw new VeldridException($"Neither macOS surface extension was available: " +
79  $"{CommonStrings.VK_MVK_MACOS_SURFACE_EXTENSION_NAME}, {CommonStrings.VK_EXT_METAL_SURFACE_EXTENSION_NAME}");
80  }
81  }
82 
83  return CreateNSWindowSurface(gd, instance, nsWindowSource, false);
84  case NSViewSwapchainSource nsViewSource:
85  if (doCheck)
86  {
87  bool hasMetalExtension = gd.HasSurfaceExtension(CommonStrings.VK_EXT_METAL_SURFACE_EXTENSION_NAME);
88  if (hasMetalExtension || gd.HasSurfaceExtension(CommonStrings.VK_MVK_MACOS_SURFACE_EXTENSION_NAME))
89  {
90  return CreateNSViewSurface(gd, instance, nsViewSource, hasMetalExtension);
91  }
92  else
93  {
94  throw new VeldridException($"Neither macOS surface extension was available: " +
95  $"{CommonStrings.VK_MVK_MACOS_SURFACE_EXTENSION_NAME}, {CommonStrings.VK_EXT_METAL_SURFACE_EXTENSION_NAME}");
96  }
97  }
98 
99  return CreateNSViewSurface(gd, instance, nsViewSource, false);
100  case UIViewSwapchainSource uiViewSource:
101  if (doCheck)
102  {
103  bool hasMetalExtension = gd.HasSurfaceExtension(CommonStrings.VK_EXT_METAL_SURFACE_EXTENSION_NAME);
104  if (hasMetalExtension || gd.HasSurfaceExtension(CommonStrings.VK_MVK_IOS_SURFACE_EXTENSION_NAME))
105  {
106  return CreateUIViewSurface(gd, instance, uiViewSource, hasMetalExtension);
107  }
108  else
109  {
110  throw new VeldridException($"Neither macOS surface extension was available: " +
111  $"{CommonStrings.VK_MVK_MACOS_SURFACE_EXTENSION_NAME}, {CommonStrings.VK_MVK_IOS_SURFACE_EXTENSION_NAME}");
112  }
113  }
114 
115  return CreateUIViewSurface(gd, instance, uiViewSource, false);
116  default:
117  throw new VeldridException($"The provided SwapchainSource cannot be used to create a Vulkan surface.");
118  }
119  }

208  private static unsafe VkSurfaceKHR CreateNSViewSurface(VkGraphicsDevice gd, VkInstance instance, NSViewSwapchainSource nsViewSource, bool hasExtMetalSurface)
209  {
210  NSView contentView = new NSView(nsViewSource.NSView);
211 
212  if (!CAMetalLayer.TryCast(contentView.layer, out var metalLayer))
213  {
214  metalLayer = CAMetalLayer.New();
215  contentView.wantsLayer = true;
216  contentView.layer = metalLayer.NativePtr;
217  }
218 
219  if (hasExtMetalSurface)
220  {
221  VkMetalSurfaceCreateInfoEXT surfaceCI = new VkMetalSurfaceCreateInfoEXT();
222  surfaceCI.sType = VkMetalSurfaceCreateInfoEXT.VK_STRUCTURE_TYPE_METAL_SURFACE_CREATE_INFO_EXT;
223  surfaceCI.pLayer = metalLayer.NativePtr.ToPointer();
224  VkSurfaceKHR surface;
225  VkResult result = gd.CreateMetalSurfaceEXT(instance, &surfaceCI, null, &surface);
226  CheckResult(result);
227  return surface;
228  }
229  else
230  {
231  VkMacOSSurfaceCreateInfoMVK surfaceCI = VkMacOSSurfaceCreateInfoMVK.New();
232  surfaceCI.pView = contentView.NativePtr.ToPointer();
233  VkResult result = vkCreateMacOSSurfaceMVK(instance, ref surfaceCI, null, out VkSurfaceKHR surface);
234  CheckResult(result);
235  return surface;
236  }
237  }

247  private static VkSurfaceKHR CreateUIViewSurface(VkGraphicsDevice gd, VkInstance instance, UIViewSwapchainSource uiViewSource, bool hasExtMetalSurface)

250

259  if (hasExtMetalSurface)

262  surfaceCI.sType = VkMetalSurfaceCreateInfoEXT.VK_STRUCTURE_TYPE_METAL_SURFACE_CREATE_INFO_EXT;
263  surfaceCI.pLayer = metalLayer.NativePtr.ToPointer();
264  VkSurfaceKHR surface;
265  VkResult result = gd.CreateMetalSurfaceEXT(instance, &surfaceCI, null, &surface);
266  CheckResult(result);
267  return surface;
268  }
269  else
270  {
271  VkIOSSurfaceCreateInfoMVK surfaceCI = VkIOSSurfaceCreateInfoMVK.New();
272  surfaceCI.pView = uiView.NativePtr.ToPointer();
273  VkResult result = vkCreateIOSSurfaceMVK(instance, ref surfaceCI, null, out VkSurfaceKHR surface);
