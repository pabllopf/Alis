290  {
291  byte* nativeFilename;
292  int filenameByteCount = 0;
293  if (filename != null)
294  {
295  filenameByteCount = Encoding.UTF8.GetByteCount(filename);
296  if (filenameByteCount > Util.StackAllocationSizeLimit)
297  {
298  nativeFilename = Util.Allocate(filenameByteCount + 1);
299  }
300  else
301  {
302  byte* nativeFilenameStackBytes = stackalloc byte[filenameByteCount + 1];
303  nativeFilename = nativeFilenameStackBytes;
304  }
305 
306  int nativeFilenameOffset = Util.GetUtf8(filename, nativeFilename, filenameByteCount);
307  nativeFilename[nativeFilenameOffset] = 0;
308  }
309  else
310  {
311  nativeFilename = null;
312  }
313 
314  ImFontConfig* fontCfg = null;
315  ushort* glyphRanges = null;
316  ImFont* ret = ImGuiNative.ImFontAtlas_AddFontFromFileTTF(NativePtr, nativeFilename, sizePixels, fontCfg, glyphRanges);
317  if (filenameByteCount > Util.StackAllocationSizeLimit)
318  {
319  Util.Free(nativeFilename);
320  }
321 
322  return new ImFontPtr(ret);
323  }
