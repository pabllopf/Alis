281  public ImFontPtr AddFontFromFileTtf(string filename, float sizePixels)
282  {
283  byte* nativeFilename;
284  int filenameByteCount = 0;
285  if (filename != null)
286  {
287  filenameByteCount = Encoding.UTF8.GetByteCount(filename);
288  if (filenameByteCount > Util.StackAllocationSizeLimit)
289  {
290  nativeFilename = Util.Allocate(filenameByteCount + 1);
291  }
292  else
293  {
294  byte* nativeFilenameStackBytes = stackalloc byte[filenameByteCount + 1];
295  nativeFilename = nativeFilenameStackBytes;
296  }
297 
298  int nativeFilenameOffset = Util.GetUtf8(filename, nativeFilename, filenameByteCount);
299  nativeFilename[nativeFilenameOffset] = 0;
300  }
301  else
302  {
303  nativeFilename = null;
304  }
305 
306  ImFontConfig* fontCfg = null;
307  ushort* glyphRanges = null;
308  ImFont* ret = ImGuiNative.ImFontAtlas_AddFontFromFileTTF(NativePtr, nativeFilename, sizePixels, fontCfg, glyphRanges);
309  if (filenameByteCount > Util.StackAllocationSizeLimit)
310  {
311  Util.Free(nativeFilename);
312  }
313 
314  return new ImFontPtr(ret);
315  }
