282  public ImFontPtr AddFontFromFileTtf(string filename, float sizePixels)
283  {
284  byte* nativeFilename;
285  int filenameByteCount = 0;
286  if (filename != null)
287  {
288  filenameByteCount = Encoding.UTF8.GetByteCount(filename);
289  if (filenameByteCount > Util.StackAllocationSizeLimit)
290  {
291  nativeFilename = Util.Allocate(filenameByteCount + 1);
292  }
293  else
294  {
295  byte* nativeFilenameStackBytes = stackalloc byte[filenameByteCount + 1];
296  nativeFilename = nativeFilenameStackBytes;
297  }
298 
299  int nativeFilenameOffset = Util.GetUtf8(filename, nativeFilename, filenameByteCount);
300  nativeFilename[nativeFilenameOffset] = 0;
301  }
302  else
303  {
304  nativeFilename = null;
305  }
306 
307  ImFontConfig* fontCfg = null;
308  ushort* glyphRanges = null;
309  ImFont* ret = ImGuiNative.ImFontAtlas_AddFontFromFileTTF(NativePtr, nativeFilename, sizePixels, fontCfg, glyphRanges);
310  if (filenameByteCount > Util.StackAllocationSizeLimit)
311  {
312  Util.Free(nativeFilename);
313  }
314 
315  return new ImFontPtr(ret);
316  }
