292  {
293  byte* nativeFilename;
294  int filenameByteCount = 0;
295  if (filename != null)
296  {
297  filenameByteCount = Encoding.UTF8.GetByteCount(filename);
298  if (filenameByteCount > Util.StackAllocationSizeLimit)
299  {
300  nativeFilename = Util.Allocate(filenameByteCount + 1);
301  }
302  else
303  {
304  byte* nativeFilenameStackBytes = stackalloc byte[filenameByteCount + 1];
305  nativeFilename = nativeFilenameStackBytes;
306  }
307 
308  int nativeFilenameOffset = Util.GetUtf8(filename, nativeFilename, filenameByteCount);
309  nativeFilename[nativeFilenameOffset] = 0;
310  }
311  else
312  {
313  nativeFilename = null;
314  }
315 
316  ImFontConfig* fontCfg = null;
317  ushort* glyphRanges = null;
318  ImFont* ret = ImGuiNative.ImFontAtlas_AddFontFromFileTTF(NativePtr, nativeFilename, sizePixels, fontCfg, glyphRanges);
319  if (filenameByteCount > Util.StackAllocationSizeLimit)
320  {
321  Util.Free(nativeFilename);
322  }
323 
324  return new ImFontPtr(ret);
325  }
