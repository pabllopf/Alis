65  public static void Write(WebSocketOpCode opCode, ArraySegment<byte> fromPayload, MemoryStream toStream,
66  bool isLastFrame, bool isClient)
67  {
68  MemoryStream memoryStream = toStream;
69  byte finBitSetAsByte = isLastFrame ? (byte) 0x80 : (byte) 0x00;
70  byte byte1 = (byte) (finBitSetAsByte | (byte) opCode);
71  memoryStream.WriteByte(byte1);
72 
73  // NB, set the mask flag if we are constructing a client frame
74  byte maskBitSetAsByte = isClient ? (byte) 0x80 : (byte) 0x00;
75 
76  // depending on the size of the length we want to write it as a byte, ushort or ulong
77  if (fromPayload.Count < 126)
78  {
79  byte byte2 = (byte) (maskBitSetAsByte | (byte) fromPayload.Count);
80  memoryStream.WriteByte(byte2);
81  }
82  else if (fromPayload.Count <= ushort.MaxValue)
83  {
84  byte byte2 = (byte) (maskBitSetAsByte | 126);
85  memoryStream.WriteByte(byte2);
86  BinaryReaderWriter.WriteUShort((ushort) fromPayload.Count, memoryStream, false);
87  }
88  else
89  {
90  byte byte2 = (byte) (maskBitSetAsByte | 127);
91  memoryStream.WriteByte(byte2);
92  BinaryReaderWriter.WriteULong((ulong) fromPayload.Count, memoryStream, false);
93  }
94 
95  // if we are creating a client frame then we MUST mack the payload as per the spec
96  if (isClient)
97  {
98  byte[] maskKey = new byte[WebSocketFrameCommon.MaskKeyLength];
99  Random.NextBytes(maskKey);
100  memoryStream.Write(maskKey, 0, maskKey.Length);
101 
102  // mask the payload
103  ArraySegment<byte> maskKeyArraySegment = new ArraySegment<byte>(maskKey, 0, maskKey.Length);
104  WebSocketFrameCommon.ToggleMask(maskKeyArraySegment, fromPayload);
105  }
106 
107  memoryStream.Write(fromPayload.Array, fromPayload.Offset, fromPayload.Count);
108  }
109  }
110 }
