176  public void Solve(TimeStep step, Vector2F gravity, bool allowSleep)
177  {
178  float h = step.DeltaTime;
179 
180  // Integrate velocities and apply damping. Initialize the body state.
181  for (int i = 0; i < BodyCount; ++i)
182  {
183  Body b = Bodies[i];
184 
185  Vector2F c = b.Sweep.C;
186  float a = b.Sweep.A;
187  Vector2F v = b.LinearVelocity;
188  float w = b.AngularVelocity;
189 
190  // Store positions for continuous collision.
191  b.Sweep.C0 = b.Sweep.C;
192  b.Sweep.A0 = b.Sweep.A;
193 
194  if (b.BodyType == BodyType.Dynamic)
195  {
196  // Integrate velocities.
197  v += h * b.InvMass * (b.GravityScale * b.Mass * gravity + b.Force);
198  w += h * b.InvI * b.Torque;
199 
200  // Apply damping.
201  // ODE: dv/dt + c * v = 0
202  // Solution: v(t) = v0 * exp(-c * t)
203  // Time step: v(t + dt) = v0 * exp(-c * (t + dt)) = v0 * exp(-c * t) * exp(-c * dt) = v * exp(-c * dt)
204  // v2 = exp(-c * dt) * v1
205  // Taylor expansion:
206  // v2 = (1.0f - c * dt) * v1
207  v *= MathUtils.Clamp(1.0f - h * b.LinearDamping, 0.0f, 1.0f);
208  w *= MathUtils.Clamp(1.0f - h * b.AngularDamping, 0.0f, 1.0f);
209  }
210 
211  positions[i].C = c;
212  positions[i].A = a;
213  velocities[i].V = v;
214  velocities[i].W = w;
215  }
216 
217  timer.Restart();
218 
219  // Solver data
220  SolverData solverData = new SolverData
221  {
222  Step = step,
223  Positions = positions,
224  Velocities = velocities
225  };
226 
227  //Velcro: We reduce the amount of garbage by reusing the contactsolver and only resetting the state
228  contactSolver.Reset(step, ContactCount, contacts, positions, velocities);
229  contactSolver.InitializeVelocityConstraints();
230 
231  if (step.WarmStarting)
232  {
233  contactSolver.WarmStart();
234  }
235 
236  for (int i = 0; i < jointCount; ++i)
237  {
238  if (joints[i].Enabled)
239  {
240  joints[i].InitVelocityConstraints(ref solverData);
241  }
242  }
243 
244  //profile.SolveInit = timer.ElapsedTicks;
245 
246  // Solve velocity constraints.
247  timer.Restart();
248  for (int i = 0; i < step.VelocityIterations; ++i)
249  {
250  for (int j = 0; j < jointCount; ++j)
251  {
252  Joint joint = joints[j];
253 
254  if (!joint.Enabled)
255  {
256  continue;
257  }
258 
259  joint.SolveVelocityConstraints(ref solverData);
260  joint.Validate(step.InvertedDeltaTime);
261  }
262 
263  contactSolver.SolveVelocityConstraints();
264  }
265 
266  // Store impulses for warm starting.
267  contactSolver.StoreImpulses();
268  //profile.SolveVelocity = timer.ElapsedTicks;
269 
270  // Integrate positions
271  for (int i = 0; i < BodyCount; ++i)
272  {
273  Vector2F c = positions[i].C;
274  float a = positions[i].A;
275  Vector2F v = velocities[i].V;
276  float w = velocities[i].W;
277 
278  // Check for large velocities
279  Vector2F translation = h * v;
280  if (Vector2F.Dot(translation, translation) > Settings.Translation * Settings.Translation)
281  {
282  float ratio = Settings.Translation / translation.Length();
283  v *= ratio;
284  }
285 
286  float rotation = h * w;
287  if (rotation * rotation > Settings.Rotation * Settings.Rotation)
288  {
289  float ratio = Settings.Rotation / Math.Abs(rotation);
290  w *= ratio;
291  }
292 
293  // Integrate
294  c += h * v;
295  a += h * w;
296 
297  positions[i].C = c;
298  positions[i].A = a;
299  velocities[i].V = v;
300  velocities[i].W = w;
301  }
302 
303  // Solve position constraints
304  timer.Restart();
305  bool positionSolved = false;
306  for (int i = 0; i < step.PositionIterations; ++i)
307  {
308  bool contactsOkay = contactSolver.SolvePositionConstraints();
309 
310  bool jointsOkay = true;
311  for (int j = 0; j < jointCount; ++j)
312  {
313  Joint joint = joints[j];
314 
315  //Velcro: We support disabling joints
316  if (!joint.Enabled)
317  {
318  continue;
319  }
320 
321  bool jointOkay = joint.SolvePositionConstraints(ref solverData);
322  jointsOkay = jointsOkay && jointOkay;
323  }
324 
325  if (contactsOkay && jointsOkay)
326  {
327  // Exit early if the position errors are small.
328  positionSolved = true;
329  break;
330  }
331  }
332 
333  // Copy state buffers back to the bodies
334  for (int i = 0; i < BodyCount; ++i)
335  {
336  Body body = Bodies[i];
337  body.Sweep.C = positions[i].C;
338  body.Sweep.A = positions[i].A;
339  body.LinearVelocity = velocities[i].V;
340  body.AngularVelocity = velocities[i].W;
341  body.SynchronizeTransform();
342  }
343 
344  //profile.SolvePosition = timer.ElapsedTicks;
345 
346  Report(contactSolver.VelocityConstraints);
347 
348  if (allowSleep)
349  {
350  float minSleepTime = float.MaxValue;
351 
352  for (int i = 0; i < BodyCount; ++i)
353  {
354  Body b = Bodies[i];
355 
356  if (b.BodyType == BodyType.Static)
357  {
358  continue;
359  }
360 
361  if (!b.SleepingAllowed || b.AngularVelocity * b.AngularVelocity > angTolSqr ||
362  Vector2F.Dot(b.LinearVelocity, b.LinearVelocity) > linTolSqr)
363  {
364  b.SleepTime = 0.0f;
365  minSleepTime = 0.0f;
366  }
367  else
368  {
369  b.SleepTime += h;
370  minSleepTime = Math.Min(minSleepTime, b.SleepTime);
371  }
372  }
373 
374  if ((minSleepTime >= Settings.TimeToSleep) && positionSolved)
375  {
376  for (int i = 0; i < BodyCount; ++i)
377  {
378  Body b = Bodies[i];
379  b.Awake = false;
380  }
381  }
382  }
383  }

391  internal void SolveToi(ref TimeStep subStep, int toiIndexA, int toiIndexB)
392  {
393  Debug.Assert(toiIndexA < BodyCount);
394  Debug.Assert(toiIndexB < BodyCount);
395 
396  // Initialize the body state.
397  for (int i = 0; i < BodyCount; ++i)
398  {
399  Body b = Bodies[i];
400  positions[i].C = b.Sweep.C;
401  positions[i].A = b.Sweep.A;
402  velocities[i].V = b.LinearVelocity;
403  velocities[i].W = b.AngularVelocity;
404  }
405 
406  //Velcro: We reset the contact solver instead of craeting a new one to reduce garbage
407  contactSolver.Reset(subStep, ContactCount, contacts, positions, velocities);
408 
409  // Solve position constraints.
410  for (int i = 0; i < subStep.PositionIterations; ++i)
411  {
412  bool contactsOkay = contactSolver.SolveToiPositionConstraints(toiIndexA, toiIndexB);
413  if (contactsOkay)
414  {
415  break;
416  }
417  }
418 
419  // Leap of faith to new safe state.
420  Bodies[toiIndexA].Sweep.C0 = positions[toiIndexA].C;
421  Bodies[toiIndexA].Sweep.A0 = positions[toiIndexA].A;
422  Bodies[toiIndexB].Sweep.C0 = positions[toiIndexB].C;
423  Bodies[toiIndexB].Sweep.A0 = positions[toiIndexB].A;
424 
425  // No warm starting is needed for TOI events because warm
426  // starting impulses were applied in the discrete solver.
427  contactSolver.InitializeVelocityConstraints();
428 
429  // Solve velocity constraints.
430  for (int i = 0; i < subStep.VelocityIterations; ++i)
431  {
432  contactSolver.SolveVelocityConstraints();
433  }
434 
435  // Don't store the TOI contact forces for warm starting
436  // because they can be quite large.
437 
438  float h = subStep.DeltaTime;
439 
440  // Integrate positions.
441  for (int i = 0; i < BodyCount; ++i)
442  {
443  Vector2F c = positions[i].C;
444  float a = positions[i].A;
445  Vector2F v = velocities[i].V;
446  float w = velocities[i].W;
447 
448  // Check for large velocities
449  Vector2F translation = h * v;
450  if (Vector2F.Dot(translation, translation) > Settings.Translation * Settings.Translation)
451  {
452  float ratio = Settings.Translation / translation.Length();
453  v *= ratio;
454  }
455 
456  float rotation = h * w;
457  if (rotation * rotation > Settings.Rotation * Settings.Rotation)
458  {
459  float ratio = Settings.Rotation / Math.Abs(rotation);
460  w *= ratio;
461  }
462 
463  // Integrate
464  c += h * v;
465  a += h * w;
466 
467  positions[i].C = c;
468  positions[i].A = a;
469  velocities[i].V = v;
470  velocities[i].W = w;
471 
472  // Sync bodies
473  Body body = Bodies[i];
474  body.Sweep.C = c;
475  body.Sweep.A = a;
476  body.LinearVelocity = v;
477  body.AngularVelocity = w;
478  body.SynchronizeTransform();
479  }
480 
481  Report(contactSolver.VelocityConstraints);
482  }

519  private void Report(ContactVelocityConstraint[] constraints)
520  {
521  if (contactManager == null)
522  {
523  return;
524  }
525 
526  for (int i = 0; i < ContactCount; ++i)
527  {
528  Contact c = contacts[i];
529 
530  //Velcro feature: added after collision
531  c.FixtureA.AfterCollision?.Invoke(c.FixtureA, c.FixtureB, c, constraints[i]);
532  c.FixtureB.AfterCollision?.Invoke(c.FixtureB, c.FixtureA, c, constraints[i]);
533 
534  //Velcro optimization: We don't store the impulses and send it to the delegate. We just send the whole contact.
535  contactManager.PostSolve?.Invoke(c, constraints[i]);
536  }
537  }
