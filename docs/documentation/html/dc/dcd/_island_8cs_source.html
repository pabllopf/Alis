109  public void Solve(TimeStep step, Vector2F gravity, bool allowSleep, ContactManager contactManager, List<Body> bodiesOfWorld)
110  {
111  for (int index = bodiesOfWorld.Count - 1; index >= 0; index--)
112  {
113  Body body = bodiesOfWorld[index];
114  if ((body.Flags & BodyFlags.IslandFlag) == BodyFlags.IslandFlag)
115  {
116  continue;
117  }
118 
119  if (!body.Awake || !body.Enabled)
120  {
121  continue;
122  }
123 
124  // The seed can be dynamic or kinematic.
125  if (body.BodyType == BodyType.Static)
126  {
127  continue;
128  }
129 
130  // Reset island and stack.
131  Clear();
132 
133  body.Flags |= BodyFlags.IslandFlag;
134 
135  //Debug.Assert(b.Enabled);
136  Add(body);
137 
138  // To keep islands as small as possible, we don't
139  // propagate islands across static bodies.
140  if (body.BodyType == BodyType.Static)
141  {
142  continue;
143  }
144 
145  // Make sure the body is awake (without resetting sleep timer).
146  body.Flags |= BodyFlags.AwakeFlag;
147 
148  // Search all contacts connected to this body.
149  for (ContactEdge ce = body.ContactList; ce != null; ce = ce.Next)
150  {
151  Contact contact = ce.Contact;
152 
153  // Has this contact already been added to an island?
154  if (contact.IslandFlag)
155  {
156  continue;
157  }
158 
159  // Is this contact solid and touching?
160  if (!contact.Enabled || !contact.IsTouching)
161  {
162  continue;
163  }
164 
165  // Skip sensors.
166  bool sensorA = contact.FixtureA.IsSensor;
167  bool sensorB = contact.FixtureB.IsSensor;
168  if (sensorA || sensorB)
169  {
170  continue;
171  }
172 
173  Add(contact);
174  contact.Flags |= ContactFlags.IslandFlag;
175 
176  Body other = ce.Other;
177 
178  // Was the other body already added to this island?
179  if (other.IsIsland)
180  {
181  continue;
182  }
183 
184  //Debug.Assert(stackCount < stackSize);
185  other.Flags |= BodyFlags.IslandFlag;
186  }
187 
188  // Search all joints connect to this body.
189  for (JointEdge je = body.JointList; je != null; je = je.Next)
190  {
191  if (je.Joint.IslandFlag)
192  {
193  continue;
194  }
195 
196  Body other = je.Other;
197 
198  // WIP David
199  //Enter here when it's a non-fixed joint. Non-fixed joints have a other body.
200  if (other != null)
201  {
202  // Don't simulate joints connected to inactive bodies.
203  if (!other.Enabled)
204  {
205  continue;
206  }
207 
208  Add(je.Joint);
209  je.Joint.IslandFlag = true;
210 
211  if (other.IsIsland)
212  {
213  continue;
214  }
215 
216  other.Flags |= BodyFlags.IslandFlag;
217  }
218  else
219  {
220  Add(je.Joint);
221  je.Joint.IslandFlag = true;
222  }
223  }
224 
225 
226  float h = step.DeltaTime;
227 
228  // Integrate velocities and apply damping. Initialize the body state.
229  foreach (Body b in Bodies)
230  {
231  Vector2F c = b.Sweep.C;
232  float a = b.Sweep.A;
233  Vector2F v = b.LinearVelocity;
234  float w = b.AngularVelocity;
235 
236  // Store positions for continuous collision.
237  b.Sweep.C0 = b.Sweep.C;
238  b.Sweep.A0 = b.Sweep.A;
239 
240  if (b.BodyType == BodyType.Dynamic)
241  {
242  // Integrate velocities.
243  v += h * b.InvMass * (b.GravityScale * b.Mass * gravity + b.Force);
244  w += h * b.InvI * b.Torque;
245 
246  // Apply damping.
247  // ODE: dv/dt + c * v = 0
248  // Solution: v(t) = v0 * exp(-c * t)
249  // Time step: v(t + dt) = v0 * exp(-c * (t + dt)) = v0 * exp(-c * t) * exp(-c * dt) = v * exp(-c * dt)
250  // v2 = exp(-c * dt) * v1
251  // Taylor expansion:
252  // v2 = (1.0f - c * dt) * v1
253  v *= MathUtils.Clamp(1.0f - h * b.LinearDamping, 0.0f, 1.0f);
254  w *= MathUtils.Clamp(1.0f - h * b.AngularDamping, 0.0f, 1.0f);
255  }
256 
257  if (positions.Count <= Bodies.IndexOf(b))
258  {
259  positions.Add(new Position(c, a));
260  }
261  else
262  {
263  positions[Bodies.IndexOf(b)].C = c;
264  positions[Bodies.IndexOf(b)].A = a;
265  }
266 
267  if (velocities.Count <= Bodies.IndexOf(b))
268  {
269  velocities.Insert(Bodies.IndexOf(b), new Velocity(v, w));
270  }
271  else
272  {
273  velocities[Bodies.IndexOf(b)].V = v;
274  velocities[Bodies.IndexOf(b)].W = w;
275  }
276  }
277 
278  // Solver data
279  SolverData solverData = new SolverData
280  {
281  Step = step,
282  Positions = positions,
283  Velocities = velocities
284  };
285 
286  //Velcro: We reduce the amount of garbage by reusing the contact solver and only resetting the state
287  contactSolver.Reset(step, contacts.Count, contacts, positions, velocities);
288  contactSolver.InitializeVelocityConstraints();
289 
290  if (step.WarmStarting)
291  {
292  contactSolver.WarmStart();
293  }
294 
295  for (int i = 0; i < joints.Count; ++i)
296  {
297  if (joints[i].Enabled)
298  {
299  joints[i].InitVelocityConstraints(ref solverData);
300  }
301  }
302 
303  //profile.SolveInit = timer.ElapsedTicks;
304 
305  // Solve velocity constraints.
306  for (int i = 0; i < step.VelocityIterations; ++i)
307  {
308  for (int j = 0; j < joints.Count; ++j)
309  {
310  Joint joint = joints[j];
311 
312  if (!joint.Enabled)
313  {
314  continue;
315  }
316 
317  joint.SolveVelocityConstraints(ref solverData);
318  joint.Validate(step.InvertedDeltaTime);
319  }
320 
321  contactSolver.SolveVelocityConstraints();
322  }
323 
324  // Store impulses for warm starting.
325  contactSolver.StoreImpulses();
326  //profile.SolveVelocity = timer.ElapsedTicks;
327 
328  // Integrate positions
329  for (int i = 0; i < positions.Count; ++i)
330  {
331  Vector2F c = positions[i].C;
332  float a = positions[i].A;
333  Vector2F v = velocities[i].V;
334  float w = velocities[i].W;
335 
336  // Check for large velocities
337  Vector2F translation = h * v;
338  if (Vector2F.Dot(translation, translation) > Settings.Translation * Settings.Translation)
339  {
340  float ratio = Settings.Translation / translation.Length();
341  v *= ratio;
342  }
343 
344  float rotation = h * w;
345  if (rotation * rotation > Settings.Rotation * Settings.Rotation)
346  {
347  float ratio = Settings.Rotation / Math.Abs(rotation);
348  w *= ratio;
349  }
350 
351  // Integrate
352  c += h * v;
353  a += h * w;
354 
355  positions[i].C = c;
356  positions[i].A = a;
357  velocities[i].V = v;
358  velocities[i].W = w;
359  }
360 
361  // Solve position constraints
362  bool positionSolved = false;
363  for (int i = 0; i < step.PositionIterations; ++i)
364  {
365  bool contactsOkay = contactSolver.SolvePositionConstraints();
366 
367  bool jointsOkay = true;
368  for (int j = 0; j < joints.Count; ++j)
369  {
370  Joint joint = joints[j];
371 
372  //Velcro: We support disabling joints
373  if (!joint.Enabled)
374  {
375  continue;
376  }
377 
378  bool jointOkay = joint.SolvePositionConstraints(ref solverData);
379  jointsOkay = jointsOkay && jointOkay;
380  }
381 
382  if (contactsOkay && jointsOkay)
383  {
384  // Exit early if the position errors are small.
385  positionSolved = true;
386  break;
387  }
388  }
389 
390  // Copy state buffers back to the bodies
391  for (int i = 0; i < Bodies.Count; ++i)
392  {
393  ;
394  Bodies[i].Sweep.C = positions[i].C;
395  Bodies[i].Sweep.A = positions[i].A;
396  Bodies[i].LinearVelocity = velocities[i].V;
397  Bodies[i].AngularVelocity = velocities[i].W;
398  Bodies[i].SynchronizeTransform();
399  }
400 
401  //profile.SolvePosition = timer.ElapsedTicks;
402 
403  Report(contactSolver.VelocityConstraints, contactManager);
404 
405  if (allowSleep)
406  {
407  float minSleepTime = float.MaxValue;
408 
409  for (int i = 0; i < Bodies.Count; ++i)
410  {
411  Body b = Bodies[i];
412 
413  if (b.BodyType == BodyType.Static)
414  {
415  continue;
416  }
417 
418  if (!b.SleepingAllowed || b.AngularVelocity * b.AngularVelocity > AngTolSqr ||
419  Vector2F.Dot(b.LinearVelocity, b.LinearVelocity) > LinTolSqr)
420  {
421  b.SleepTime = 0.0f;
422  minSleepTime = 0.0f;
423  }
424  else
425  {
426  b.SleepTime += h;
427  minSleepTime = Math.Min(minSleepTime, b.SleepTime);
428  }
429  }
430 
431  if ((minSleepTime >= Settings.TimeToSleep) && positionSolved)
432  {
433  for (int i = 0; i < Bodies.Count; ++i)
434  {
435  Body b = Bodies[i];
436  b.Awake = false;
437  }
438  }
439  }
440  }
441 
442  // Post solve cleanup.
443  for (int i = 0; i < Bodies.Count; ++i)
444  {
445  // Allow static bodies to participate in other islands.
446  if (Bodies[i].BodyType == BodyType.Static)
447  {
448  Bodies[i].Flags &= ~BodyFlags.IslandFlag;
449  }
450  }
451  }

462  internal void SolveToi(float minAlpha, TimeStep subStep, int toiIndexA, int toiIndexB, ContactManager contactManager)
463  {
464  TimeStepSolveToi.DeltaTime = (1.0f - minAlpha) * subStep.DeltaTime;
465  TimeStepSolveToi.InvertedDeltaTime = 1.0f / ((1.0f - minAlpha) * subStep.DeltaTime);
466  TimeStepSolveToi.DeltaTimeRatio = 1.0f;
467  TimeStepSolveToi.PositionIterations = 20;
468  TimeStepSolveToi.VelocityIterations = subStep.VelocityIterations;
469  TimeStepSolveToi.WarmStarting = false;
470 
471 
472  Debug.Assert(toiIndexA < Bodies.Count);
473  Debug.Assert(toiIndexB < Bodies.Count);
474 
475  // Initialize the body state.
476  for (int i = 0; i < Bodies.Count; ++i)
477  {
478  Body b = Bodies[i];
479 
480  if (positions.Count <= Bodies.IndexOf(b))
481  {
482  positions.Add(new Position(b.Sweep.C, b.Sweep.A));
483  }
484  else
485  {
486  positions[Bodies.IndexOf(b)].C = b.Sweep.C;
487  positions[Bodies.IndexOf(b)].A = b.Sweep.A;
488  }
489 
490  if (velocities.Count <= Bodies.IndexOf(b))
491  {
492  velocities.Insert(Bodies.IndexOf(b), new Velocity(b.LinearVelocity, b.AngularVelocity));
493  }
494  else
495  {
496  velocities[Bodies.IndexOf(b)].V = b.LinearVelocity;
497  velocities[Bodies.IndexOf(b)].W = b.AngularVelocity;
498  }
499  }
500 
501  //Velcro: We reset the contact solver instead of craeting a new one to reduce garbage
502  contactSolver.Reset(subStep, contacts.Count, contacts, positions, velocities);
503 
504  // Solve position constraints.
505  for (int i = 0; i < subStep.PositionIterations; ++i)
506  {
507  bool contactsOkay = contactSolver.SolveToiPositionConstraints(toiIndexA, toiIndexB);
508  if (contactsOkay)
509  {
510  break;
511  }
512  }
513 
514  // Leap of faith to new safe state.
515  Bodies[toiIndexA].Sweep.C0 = positions[toiIndexA].C;
516  Bodies[toiIndexA].Sweep.A0 = positions[toiIndexA].A;
517  Bodies[toiIndexB].Sweep.C0 = positions[toiIndexB].C;
518  Bodies[toiIndexB].Sweep.A0 = positions[toiIndexB].A;
519 
520  // No warm starting is needed for TOI events because warm
521  // starting impulses were applied in the discrete solver.
522  contactSolver.InitializeVelocityConstraints();
523 
524  // Solve velocity constraints.
525  for (int i = 0; i < subStep.VelocityIterations; ++i)
526  {
527  contactSolver.SolveVelocityConstraints();
528  }
529 
530  // Don't store the TOI contact forces for warm starting
531  // because they can be quite large.
532 
533  float h = subStep.DeltaTime;
534 
535  // Integrate positions.
536  for (int i = 0; i < Bodies.Count; ++i)
537  {
538  Vector2F c = positions[i].C;
539  float a = positions[i].A;
540  Vector2F v = velocities[i].V;
541  float w = velocities[i].W;
542 
543  // Check for large velocities
544  Vector2F translation = h * v;
545  if (Vector2F.Dot(translation, translation) > Settings.Translation * Settings.Translation)
546  {
547  float ratio = Settings.Translation / translation.Length();
548  v *= ratio;
549  }
550 
551  float rotation = h * w;
552  if (rotation * rotation > Settings.Rotation * Settings.Rotation)
553  {
554  float ratio = Settings.Rotation / Math.Abs(rotation);
555  w *= ratio;
556  }
557 
558  // Integrate
559  c += h * v;
560  a += h * w;
561 
562  positions[i].C = c;
563  positions[i].A = a;
564  velocities[i].V = v;
565  velocities[i].W = w;
566 
567  // Sync bodies
568  Body body = Bodies[i];
569  body.Sweep.C = c;
570  body.Sweep.A = a;
571  body.LinearVelocity = v;
572  body.AngularVelocity = w;
573  body.SynchronizeTransform();
574  }
575 
576  Report(contactSolver.VelocityConstraints, contactManager);
577  }

607  private void Report(List<ContactVelocityConstraint> constraints, ContactManager contactManager)
608  {
609  for (int i = 0; i < contacts.Count; ++i)
610  {
611  Contact c = contacts[i];
612 
613  //Velcro feature: added after collision
614  c.FixtureA.AfterCollision?.Invoke(c.FixtureA, c.FixtureB, c, constraints[i]);
615  c.FixtureB.AfterCollision?.Invoke(c.FixtureB, c.FixtureA, c, constraints[i]);
616 
617  //Velcro optimization: We don't store the impulses and send it to the delegate. We just send the whole contact.
618  contactManager.PostSolve?.Invoke(c, constraints[i]);
619  }
