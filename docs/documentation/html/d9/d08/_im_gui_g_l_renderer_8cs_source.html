190  private void SetupRenderState(ImDrawDataPtr drawData, int fbWidth, int fbHeight)
191  {
192  glEnable(EnableCap.Blend);
193  glBlendEquation(BlendEquationMode.FuncAdd);
194  glBlendFunc(BlendingFactorSrc.SrcAlpha, BlendingFactorDest.OneMinusSrcAlpha);
195  glDisable(EnableCap.CullFace);
196  glDisable(EnableCap.DepthTest);
197  glEnable(EnableCap.ScissorTest);
198 
199  glUseProgram(_shader.ProgramID);
200 
201  float left = drawData.DisplayPos.X;
202  float right = drawData.DisplayPos.X + drawData.DisplaySize.X;
203  float top = drawData.DisplayPos.Y;
204  float bottom = drawData.DisplayPos.Y + drawData.DisplaySize.Y;
205 
206  _shader["Texture"].SetValue(0);
207  _shader["ProjMtx"].SetValue(Matrix4x4.CreateOrthographicOffCenter(left, right, bottom, top, -1, 1));
208  glBindSampler(0, 0);
209 
210  glBindVertexArray(_vertexArrayObject);
211 
212  // Bind vertex/index buffers and setup attributes for ImDrawVert
213  glBindBuffer(BufferTarget.ArrayBuffer, _vboHandle);
214  glBindBuffer(BufferTarget.ElementArrayBuffer, _elementsHandle);
215 
216  EnableVertexAttribArray(_shader["Position"].Location);
217  EnableVertexAttribArray(_shader["UV"].Location);
218  EnableVertexAttribArray(_shader["Color"].Location);
219 
220  int drawVertSize = Marshal.SizeOf<ImDrawVert>();
221  VertexAttribPointer(_shader["Position"].Location, 2, VertexAttribPointerType.Float, false, drawVertSize, Marshal.OffsetOf<ImDrawVert>("pos"));
222  VertexAttribPointer(_shader["UV"].Location, 2, VertexAttribPointerType.Float, false, drawVertSize, Marshal.OffsetOf<ImDrawVert>("uv"));
223  VertexAttribPointer(_shader["Color"].Location, 4, VertexAttribPointerType.UnsignedByte, true, drawVertSize, Marshal.OffsetOf<ImDrawVert>("col"));
224  }

229  private void RenderDrawData()
230  {
231  ImDrawDataPtr drawData = ImGui.GetDrawData();
232 
233  // Avoid rendering when minimized, scale coordinates for retina displays (screen coordinates != framebuffer coordinates)
234  int fbWidth = (int) (drawData.DisplaySize.X * drawData.FramebufferScale.X);
235  int fbHeight = (int) (drawData.DisplaySize.Y * drawData.FramebufferScale.Y);
236  if (fbWidth <= 0 || fbHeight <= 0)
237  {
238  return;
239  }
240 
241  SetupRenderState(drawData, fbWidth, fbHeight);
242 
243  Vector2 clipOffset = drawData.DisplayPos;
244  Vector2 clipScale = drawData.FramebufferScale;
245 
246  drawData.ScaleClipRects(clipScale);
247 
248  IntPtr lastTexId = ImGui.GetIO().Fonts.TexID;
249  glBindTexture(TextureTarget.Texture2D, (uint) lastTexId);
250 
251  int drawVertSize = Marshal.SizeOf<ImDrawVert>();
252  int drawIdxSize = sizeof(ushort);
253 
254  for (int n = 0; n < drawData.CmdListsCount; n++)
255  {
256  ImDrawListPtr cmdList = drawData.CmdListsRange[n];
257 
258  // Upload vertex/index buffers
259  glBufferData(BufferTarget.ArrayBuffer, (IntPtr) (cmdList.VtxBuffer.Size * drawVertSize), cmdList.VtxBuffer.Data, BufferUsageHint.StreamDraw);
260  glBufferData(BufferTarget.ElementArrayBuffer, (IntPtr) (cmdList.IdxBuffer.Size * drawIdxSize), cmdList.IdxBuffer.Data, BufferUsageHint.StreamDraw);
261 
262  for (int cmd_i = 0; cmd_i < cmdList.CmdBuffer.Size; cmd_i++)
263  {
264  ImDrawCmdPtr pcmd = cmdList.CmdBuffer[cmd_i];
265  if (pcmd.UserCallback != IntPtr.Zero)
266  {
267  Console.WriteLine("UserCallback not implemented");
268  }
269  else
270  {
271  // Project scissor/clipping rectangles into framebuffer space
272  Vector4 clip_rect = pcmd.ClipRect;
273 
274  clip_rect.X = pcmd.ClipRect.X - clipOffset.X;
275  clip_rect.Y = pcmd.ClipRect.Y - clipOffset.Y;
276  clip_rect.Z = pcmd.ClipRect.Z - clipOffset.X;
277  clip_rect.W = pcmd.ClipRect.W - clipOffset.Y;
278 
279  glScissor((int) clip_rect.X, (int) (fbHeight - clip_rect.W), (int) (clip_rect.Z - clip_rect.X), (int) (clip_rect.W - clip_rect.Y));
280 
281  // Bind texture, Draw
282  if (pcmd.TextureId != IntPtr.Zero)
283  {
284  if (pcmd.TextureId != lastTexId)
285  {
286  lastTexId = pcmd.TextureId;
287  glBindTexture(TextureTarget.Texture2D, (uint) pcmd.TextureId);
288  }
289  }
290 
291  glDrawElementsBaseVertex(BeginMode.Triangles, (int) pcmd.ElemCount, drawIdxSize == 2 ? DrawElementsType.UnsignedShort : DrawElementsType.UnsignedInt, (IntPtr) (pcmd.IdxOffset * drawIdxSize), (int) pcmd.VtxOffset);
292  }
293  }
294  }
295  }
