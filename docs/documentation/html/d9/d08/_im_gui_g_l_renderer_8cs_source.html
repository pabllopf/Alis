193  private void SetupRenderState(ImDrawDataPtr drawData, int fbWidth, int fbHeight)
194  {
195  GlEnable(EnableCap.Blend);
196  GlBlendEquation(BlendEquationMode.FuncAdd);
197  GlBlendFunc(BlendingFactorSrc.SrcAlpha, BlendingFactorDest.OneMinusSrcAlpha);
198  GlDisable(EnableCap.CullFace);
199  GlDisable(EnableCap.DepthTest);
200  GlEnable(EnableCap.ScissorTest);
201 
202  GlUseProgram(_shader.ProgramId);
203 
204  float left = drawData.DisplayPos.X;
205  float right = drawData.DisplayPos.X + drawData.DisplaySize.X;
206  float top = drawData.DisplayPos.Y;
207  float bottom = drawData.DisplayPos.Y + drawData.DisplaySize.Y;
208 
209  _shader["Texture"].SetValue(0);
210  _shader["ProjMtx"].SetValue(Matrix4X4F.CreateOrthographicOffCenter(left, right, bottom, top, -1, 1));
211  GlBindSampler(0, 0);
212 
213  GlBindVertexArray(_vertexArrayObject);
214 
215  // Bind vertex/index buffers and setup attributes for ImDrawVert
216  GlBindBuffer(BufferTarget.ArrayBuffer, _vboHandle);
217  GlBindBuffer(BufferTarget.ElementArrayBuffer, _elementsHandle);
218 
219  EnableVertexAttribArray(_shader["Position"].Location);
220  EnableVertexAttribArray(_shader["UV"].Location);
221  EnableVertexAttribArray(_shader["Color"].Location);
222 
223  int drawVertSize = Marshal.SizeOf<ImDrawVert>();
224  VertexAttribPointer(_shader["Position"].Location, 2, VertexAttribPointerType.Float, false, drawVertSize, Marshal.OffsetOf<ImDrawVert>("Pos"));
225  VertexAttribPointer(_shader["UV"].Location, 2, VertexAttribPointerType.Float, false, drawVertSize, Marshal.OffsetOf<ImDrawVert>("Uv"));
226  VertexAttribPointer(_shader["Color"].Location, 4, VertexAttribPointerType.UnsignedByte, true, drawVertSize, Marshal.OffsetOf<ImDrawVert>("Col"));
227  }

232  private void RenderDrawData()
233  {
234  ImDrawDataPtr drawData = ImGui.GetDrawData();
235 
236  // Avoid rendering when minimized, scale coordinates for retina displays (screen coordinates != framebuffer coordinates)
237  int fbWidth = (int) (drawData.DisplaySize.X * drawData.FramebufferScale.X);
238  int fbHeight = (int) (drawData.DisplaySize.Y * drawData.FramebufferScale.Y);
239  if (fbWidth <= 0 || fbHeight <= 0)
240  {
241  return;
242  }
243 
244  SetupRenderState(drawData, fbWidth, fbHeight);
245 
246  Vector2F clipOffset = drawData.DisplayPos;
247  Vector2F clipScale = drawData.FramebufferScale;
248 
249  drawData.ScaleClipRects(clipScale);
250 
251  IntPtr lastTexId = ImGui.GetIo().Fonts.TexId;
252  GlBindTexture(TextureTarget.Texture2D, (uint) lastTexId);
253 
254  int drawVertSize = Marshal.SizeOf<ImDrawVert>();
255  int drawIdxSize = sizeof(ushort);
256 
257  for (int n = 0; n < drawData.CmdListsCount; n++)
258  {
259  ImDrawListPtr cmdList = drawData.CmdListsRange[n];
260 
261  // Upload vertex/index buffers
262  GlBufferData(BufferTarget.ArrayBuffer, (IntPtr) (cmdList.VtxBuffer.Size * drawVertSize), cmdList.VtxBuffer.Data, BufferUsageHint.StreamDraw);
263  GlBufferData(BufferTarget.ElementArrayBuffer, (IntPtr) (cmdList.IdxBuffer.Size * drawIdxSize), cmdList.IdxBuffer.Data, BufferUsageHint.StreamDraw);
264 
265  for (int cmdI = 0; cmdI < cmdList.CmdBuffer.Size; cmdI++)
266  {
267  ImDrawCmdPtr pcmd = cmdList.CmdBuffer[cmdI];
268  if (pcmd.UserCallback != IntPtr.Zero)
269  {
270  Console.WriteLine("UserCallback not implemented");
271  }
272  else
273  {
274  // Project scissor/clipping rectangles into framebuffer space
275  Vector4F clipRect = pcmd.ClipRect;
276 
277  clipRect.X = pcmd.ClipRect.X - clipOffset.X;
278  clipRect.Y = pcmd.ClipRect.Y - clipOffset.Y;
279  clipRect.Z = pcmd.ClipRect.Z - clipOffset.X;
280  clipRect.W = pcmd.ClipRect.W - clipOffset.Y;
281 
282  GlScissor((int) clipRect.X, (int) (fbHeight - clipRect.W), (int) (clipRect.Z - clipRect.X), (int) (clipRect.W - clipRect.Y));
283 
284  // Bind texture, Draw
285  if (pcmd.TextureId != IntPtr.Zero)
286  {
287  if (pcmd.TextureId != lastTexId)
288  {
289  lastTexId = pcmd.TextureId;
290  GlBindTexture(TextureTarget.Texture2D, (uint) pcmd.TextureId);
291  }
292  }
293 
294  GlDrawElementsBaseVertex(BeginMode.Triangles, (int) pcmd.ElemCount, drawIdxSize == 2 ? DrawElementsType.UnsignedShort : DrawElementsType.UnsignedInt, (IntPtr) (pcmd.IdxOffset * drawIdxSize), (int) pcmd.VtxOffset);
295  }
296  }
297  }
298  }
