59  switch (pc.Type)

63  Vector2 pointA = MathUtils.Mul(ref xfA, pc.LocalPoint);
64  Vector2 pointB = MathUtils.Mul(ref xfB, pc.LocalPoints[0]);
65  normal = pointB - pointA;
66 
67  //Velcro: Fix to handle zero normalization

71  }
72 
73  point = 0.5f * (pointA + pointB);

75  }
76  break;
77 
78  case ManifoldType.FaceA:
79  {

82

84  separation = Vector2.Dot(clipPoint - planePoint, normal) - pc.RadiusA - pc.RadiusB;
85  point = clipPoint;
86  }
87  break;
88 
89  case ManifoldType.FaceB:

93

95  separation = Vector2.Dot(clipPoint - planePoint, normal) - pc.RadiusA - pc.RadiusB;
96  point = clipPoint;
97
