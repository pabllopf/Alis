53  public static void Initialize(ContactPositionConstraint pc, ref Transform xfA, ref Transform xfB, int index,
54  out Vector2F normal, out Vector2F point, out float separation)
55  {
56  Debug.Assert(pc.PointCount > 0);
57 
58  switch (pc.Type)
59  {
60  case ManifoldType.Circles:
61  {
62  Vector2F pointA = MathUtils.Mul(ref xfA, pc.LocalPoint);
63  Vector2F pointB = MathUtils.Mul(ref xfB, pc.LocalPoints[0]);
64  normal = pointB - pointA;
65 
66  //Velcro: Fix to handle zero normalization
67  if (normal != Vector2F.Zero)
68  {
69  normal = Vector2F.Normalize(normal);
70  }
71 
72  point = 0.5f * (pointA + pointB);
73  separation = Vector2F.Dot(pointB - pointA, normal) - pc.RadiusA - pc.RadiusB;
74  }
75  break;
76 
77  case ManifoldType.FaceA:
78  {
79  normal = MathUtils.Mul(xfA.Rotation, pc.LocalNormal);
80  Vector2F planePoint = MathUtils.Mul(ref xfA, pc.LocalPoint);
81 
82  Vector2F clipPoint = MathUtils.Mul(ref xfB, pc.LocalPoints[index]);
83  separation = Vector2F.Dot(clipPoint - planePoint, normal) - pc.RadiusA - pc.RadiusB;
84  point = clipPoint;
85  }
86  break;
87 
88  case ManifoldType.FaceB:
89  {
90  normal = MathUtils.Mul(xfB.Rotation, pc.LocalNormal);
91  Vector2F planePoint = MathUtils.Mul(ref xfB, pc.LocalPoint);
92 
93  Vector2F clipPoint = MathUtils.Mul(ref xfA, pc.LocalPoints[index]);
94  separation = Vector2F.Dot(clipPoint - planePoint, normal) - pc.RadiusA - pc.RadiusB;
95  point = clipPoint;
96 
97  // Ensure normal points from A to B
98  normal = -normal;
99  }
100  break;
101  default:

103  point = Vector2F.Zero;
104  separation = 0;
