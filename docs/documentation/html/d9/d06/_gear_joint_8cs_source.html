452  internal override void InitVelocityConstraints(ref SolverData data)
453  {
454  indexA = BodyA.IslandIndex;
455  indexB = BodyB.IslandIndex;
456  indexC = bodyC.IslandIndex;
457  indexD = bodyD.IslandIndex;
458  lcA = BodyA.Sweep.LocalCenter;
459  lcB = BodyB.Sweep.LocalCenter;
460  lcC = bodyC.Sweep.LocalCenter;
461  lcD = bodyD.Sweep.LocalCenter;
462  mA = BodyA.InvMass;
463  mB = BodyB.InvMass;
464  mC = bodyC.InvMass;
465  mD = bodyD.InvMass;
466  iA = BodyA.InvI;
467  iB = BodyB.InvI;
468  iC = bodyC.InvI;
469  iD = bodyD.InvI;
470 
471  float aA = data.Positions[indexA].A;
472  Vector2 vA = data.Velocities[indexA].V;
473  float wA = data.Velocities[indexA].W;
474 
475  float aB = data.Positions[indexB].A;
476  Vector2 vB = data.Velocities[indexB].V;
477  float wB = data.Velocities[indexB].W;
478 
479  float aC = data.Positions[indexC].A;
480  Vector2 vC = data.Velocities[indexC].V;
481  float wC = data.Velocities[indexC].W;
482 
483  float aD = data.Positions[indexD].A;
484  Vector2 vD = data.Velocities[indexD].V;
485  float wD = data.Velocities[indexD].W;
486 
487  Rotation qA = new Rotation(aA), qB = new Rotation(aB), qC = new Rotation(aC), qD = new Rotation(aD);
488 
489  mass = 0.0f;
490 
491  if (typeA == JointType.Revolute)
492  {
493  jvAc = Vector2.Zero;
494  jwA = 1.0f;
495  jwC = 1.0f;
496  mass += iA + iC;
497  }
498  else
499  {
500  Vector2 u = MathUtils.Mul(qC, localAxisC);
501  Vector2 rC = MathUtils.Mul(qC, localAnchorC - lcC);
502  Vector2 rA = MathUtils.Mul(qA, localAnchorA - lcA);
503  jvAc = u;
504  jwC = MathUtils.Cross(rC, u);
505  jwA = MathUtils.Cross(rA, u);
506  mass += mC + mA + iC * jwC * jwC + iA * jwA * jwA;
507  }
508 
509  if (typeB == JointType.Revolute)
510  {
511  jvBd = Vector2.Zero;
512  jwB = ratio;
513  jwD = ratio;
514  mass += ratio * ratio * (iB + iD);
515  }
516  else
517  {
518  Vector2 u = MathUtils.Mul(qD, localAxisD);
519  Vector2 rD = MathUtils.Mul(qD, localAnchorD - lcD);
520  Vector2 rB = MathUtils.Mul(qB, localAnchorB - lcB);
521  jvBd = ratio * u;
522  jwD = ratio * MathUtils.Cross(rD, u);
523  jwB = ratio * MathUtils.Cross(rB, u);
524  mass += ratio * ratio * (mD + mB) + iD * jwD * jwD + iB * jwB * jwB;
525  }
526 
527  // Compute effective mass.
528  mass = mass > 0.0f ? 1.0f / mass : 0.0f;
529 
530  if (data.Step.WarmStarting)
531  {
532  vA += mA * impulse * jvAc;
533  wA += iA * impulse * jwA;
534  vB += mB * impulse * jvBd;
535  wB += iB * impulse * jwB;
536  vC -= mC * impulse * jvAc;
537  wC -= iC * impulse * jwC;
538  vD -= mD * impulse * jvBd;
539  wD -= iD * impulse * jwD;
540  }
541  else
542  {
543  impulse = 0.0f;
544  }
545 
546  data.Velocities[indexA].V = vA;
547  data.Velocities[indexA].W = wA;
548  data.Velocities[indexB].V = vB;
549  data.Velocities[indexB].W = wB;
550  data.Velocities[indexC].V = vC;
551  data.Velocities[indexC].W = wC;
552  data.Velocities[indexD].V = vD;
553  data.Velocities[indexD].W = wD;

560  internal override void SolveVelocityConstraints(ref SolverData data)
561  {
562  Vector2 vA = data.Velocities[indexA].V;
563  float wA = data.Velocities[indexA].W;
564  Vector2 vB = data.Velocities[indexB].V;
565  float wB = data.Velocities[indexB].W;
566  Vector2 vC = data.Velocities[indexC].V;
567  float wC = data.Velocities[indexC].W;
568  Vector2 vD = data.Velocities[indexD].V;
569  float wD = data.Velocities[indexD].W;
570 
571  float dot = Vector2.Dot(jvAc, vA - vC) + Vector2.Dot(jvBd, vB - vD);
572  dot += jwA * wA - jwC * wC + (jwB * wB - jwD * wD);
573 
574  float impulseLocal = -mass * dot;
575  impulse += impulseLocal;
576 
577  vA += mA * impulseLocal * jvAc;
578  wA += iA * impulseLocal * jwA;
579  vB += mB * impulseLocal * jvBd;
580  wB += iB * impulseLocal * jwB;
581  vC -= mC * impulseLocal * jvAc;
582  wC -= iC * impulseLocal * jwC;
583  vD -= mD * impulseLocal * jvBd;
584  wD -= iD * impulseLocal * jwD;
585 
586  data.Velocities[indexA].V = vA;
587  data.Velocities[indexA].W = wA;
588  data.Velocities[indexB].V = vB;
589  data.Velocities[indexB].W = wB;
590  data.Velocities[indexC].V = vC;
591  data.Velocities[indexC].W = wC;
592  data.Velocities[indexD].V = vD;
593  data.Velocities[indexD].W = wD;
594  }

601  internal override bool SolvePositionConstraints(ref SolverData data)
602  {
603  Vector2 cA = data.Positions[indexA].C;
604  float aA = data.Positions[indexA].A;
605  Vector2 cB = data.Positions[indexB].C;
606  float aB = data.Positions[indexB].A;
607  Vector2 cC = data.Positions[indexC].C;
608  float aC = data.Positions[indexC].A;
609  Vector2 cD = data.Positions[indexD].C;
610  float aD = data.Positions[indexD].A;
611 
612  Rotation qA = new Rotation(aA), qB = new Rotation(aB), qC = new Rotation(aC), qD = new Rotation(aD);
613 
614  const float linearError = 0.0f;
615 
616  float coordinateA, coordinateB;
617 
618  Vector2 jvAcLocal, jvBdLocal;
619  float jwALocal, jwBLocal, jwCLocal, jwDLocal;
620  float massLocal = 0.0f;
621 
622  if (typeA == JointType.Revolute)
623  {
624  jvAcLocal = Vector2.Zero;
625  jwALocal = 1.0f;
626  jwCLocal = 1.0f;
627  massLocal += iA + iC;
628 
629  coordinateA = aA - aC - referenceAngleA;
630  }
631  else
632  {
633  Vector2 u = MathUtils.Mul(qC, localAxisC);
634  Vector2 rC = MathUtils.Mul(qC, localAnchorC - lcC);
635  Vector2 rA = MathUtils.Mul(qA, localAnchorA - lcA);
636  jvAcLocal = u;
637  jwCLocal = MathUtils.Cross(rC, u);
638  jwALocal = MathUtils.Cross(rA, u);
639  massLocal += mC + mA + iC * jwCLocal * jwCLocal + iA * jwALocal * jwALocal;
640 
641  Vector2 pC = localAnchorC - lcC;
642  Vector2 pA = MathUtils.MulT(qC, rA + (cA - cC));
643  coordinateA = Vector2.Dot(pA - pC, localAxisC);
644  }
645 
646  if (typeB == JointType.Revolute)
647  {
648  jvBdLocal = Vector2.Zero;
649  jwBLocal = ratio;
650  jwDLocal = ratio;
651  massLocal += ratio * ratio * (iB + iD);
652 
653  coordinateB = aB - aD - referenceAngleB;
654  }
655  else
656  {
657  Vector2 u = MathUtils.Mul(qD, localAxisD);
658  Vector2 rD = MathUtils.Mul(qD, localAnchorD - lcD);
659  Vector2 rB = MathUtils.Mul(qB, localAnchorB - lcB);
660  jvBdLocal = ratio * u;
661  jwDLocal = ratio * MathUtils.Cross(rD, u);
662  jwBLocal = ratio * MathUtils.Cross(rB, u);
663  massLocal += ratio * ratio * (mD + mB) + iD * jwDLocal * jwDLocal + iB * jwBLocal * jwBLocal;
664 
665  Vector2 pD = localAnchorD - lcD;
666  Vector2 pB = MathUtils.MulT(qD, rB + (cB - cD));
667  coordinateB = Vector2.Dot(pB - pD, localAxisD);
668  }
669 
670  float c = coordinateA + ratio * coordinateB - constant;
671 
672  float impulseLocal = 0.0f;
673  if (massLocal > 0.0f)
674  {
675  impulseLocal = -c / massLocal;
676  }
677 
678  cA += mA * impulseLocal * jvAcLocal;
679  aA += iA * impulseLocal * jwALocal;
680  cB += mB * impulseLocal * jvBdLocal;
681  aB += iB * impulseLocal * jwBLocal;
682  cC -= mC * impulseLocal * jvAcLocal;
683  aC -= iC * impulseLocal * jwCLocal;
684  cD -= mD * impulseLocal * jvBdLocal;
685  aD -= iD * impulseLocal * jwDLocal;
686 
687  data.Positions[indexA].C = cA;
688  data.Positions[indexA].A = aA;
689  data.Positions[indexB].C = cB;
690  data.Positions[indexB].A = aB;
691  data.Positions[indexC].C = cC;
692  data.Positions[indexC].A = aC;
693  data.Positions[indexD].C = cD;
694  data.Positions[indexD].A = aD;
695 
696  return linearError < Settings.LinearSlop;
697  }
