42  // Gear Joint:
43  // C0 = (coordinate1 + ratio * coordinate2)_initial
44  // C = (coordinate1 + ratio * coordinate2) - C0 = 0
45  // J = [J1 ratio * J2]
46  // K = J * invM * JT
47  // = J1 * invM1 * J1T + ratio * ratio * J2 * invM2 * J2T
48  //
49  // Revolute:
50  // coordinate = rotation
51  // Cdot = angularVelocity
52  // J = [0 0 1]
53  // K = J * invM * JT = invI
54  //
55  // Prismatic:
56  // coordinate = dot(p - pg, ug)
57  // Cdot = dot(v + cross(w, r), ug)
58  // J = [ug cross(r, ug)]
59  // K = J * invM * JT = invMass + invI * cross(r, ug)^2

450  internal override void InitVelocityConstraints(ref SolverData data)
451  {
452  indexA = BodyA.IslandIndex;
453  indexB = BodyB.IslandIndex;
454  indexC = bodyC.IslandIndex;
455  indexD = bodyD.IslandIndex;
456  lcA = BodyA.Sweep.LocalCenter;
457  lcB = BodyB.Sweep.LocalCenter;
458  lcC = bodyC.Sweep.LocalCenter;
459  lcD = bodyD.Sweep.LocalCenter;
460  mA = BodyA.InvMass;
461  mB = BodyB.InvMass;
462  mC = bodyC.InvMass;
463  mD = bodyD.InvMass;
464  iA = BodyA.InvI;
465  iB = BodyB.InvI;
466  iC = bodyC.InvI;
467  iD = bodyD.InvI;
468 
469  float aA = data.Positions[indexA].A;
470  Vector2 vA = data.Velocities[indexA].V;
471  float wA = data.Velocities[indexA].W;
472 
473  float aB = data.Positions[indexB].A;
474  Vector2 vB = data.Velocities[indexB].V;
475  float wB = data.Velocities[indexB].W;
476 
477  float aC = data.Positions[indexC].A;
478  Vector2 vC = data.Velocities[indexC].V;
479  float wC = data.Velocities[indexC].W;
480 
481  float aD = data.Positions[indexD].A;
482  Vector2 vD = data.Velocities[indexD].V;
483  float wD = data.Velocities[indexD].W;
484 
485  Rot qA = new Rot(aA), qB = new Rot(aB), qC = new Rot(aC), qD = new Rot(aD);
486 
487  mass = 0.0f;
488 
489  if (typeA == JointType.Revolute)
490  {
491  jvAc = Vector2.Zero;
492  jwA = 1.0f;
493  jwC = 1.0f;
494  mass += iA + iC;
495  }
496  else
497  {
498  Vector2 u = MathUtils.Mul(qC, localAxisC);
499  Vector2 rC = MathUtils.Mul(qC, localAnchorC - lcC);
500  Vector2 rA = MathUtils.Mul(qA, localAnchorA - lcA);
501  jvAc = u;
502  jwC = MathUtils.Cross(rC, u);
503  jwA = MathUtils.Cross(rA, u);
504  mass += mC + mA + iC * jwC * jwC + iA * jwA * jwA;
505  }
506 
507  if (typeB == JointType.Revolute)
508  {
509  jvBd = Vector2.Zero;
510  jwB = ratio;
511  jwD = ratio;
512  mass += ratio * ratio * (iB + iD);
513  }
514  else
515  {
516  Vector2 u = MathUtils.Mul(qD, localAxisD);
517  Vector2 rD = MathUtils.Mul(qD, localAnchorD - lcD);
518  Vector2 rB = MathUtils.Mul(qB, localAnchorB - lcB);
519  jvBd = ratio * u;
520  jwD = ratio * MathUtils.Cross(rD, u);
521  jwB = ratio * MathUtils.Cross(rB, u);
522  mass += ratio * ratio * (mD + mB) + iD * jwD * jwD + iB * jwB * jwB;
523  }
524 
525  // Compute effective mass.
526  mass = mass > 0.0f ? 1.0f / mass : 0.0f;
527 
528  if (data.Step.WarmStarting)
529  {
530  vA += mA * impulse * jvAc;
531  wA += iA * impulse * jwA;
532  vB += mB * impulse * jvBd;
533  wB += iB * impulse * jwB;
534  vC -= mC * impulse * jvAc;
535  wC -= iC * impulse * jwC;
536  vD -= mD * impulse * jvBd;
537  wD -= iD * impulse * jwD;
538  }
539  else
540  {
541  impulse = 0.0f;
542  }
543 
544  data.Velocities[indexA].V = vA;
545  data.Velocities[indexA].W = wA;
546  data.Velocities[indexB].V = vB;
547  data.Velocities[indexB].W = wB;
548  data.Velocities[indexC].V = vC;
549  data.Velocities[indexC].W = wC;
550  data.Velocities[indexD].V = vD;
551  data.Velocities[indexD].W = wD;
552  }

558  internal override void SolveVelocityConstraints(ref SolverData data)
559  {
560  Vector2 vA = data.Velocities[indexA].V;
561  float wA = data.Velocities[indexA].W;
562  Vector2 vB = data.Velocities[indexB].V;
563  float wB = data.Velocities[indexB].W;
564  Vector2 vC = data.Velocities[indexC].V;
565  float wC = data.Velocities[indexC].W;
566  Vector2 vD = data.Velocities[indexD].V;
567  float wD = data.Velocities[indexD].W;
568 
569  float cdot = Vector2.Dot(jvAc, vA - vC) + Vector2.Dot(jvBd, vB - vD);
570  cdot += jwA * wA - jwC * wC + (jwB * wB - jwD * wD);
571 
572  float impulse = -mass * cdot;
573  this.impulse += impulse;
574 
575  vA += mA * impulse * jvAc;
576  wA += iA * impulse * jwA;
577  vB += mB * impulse * jvBd;
578  wB += iB * impulse * jwB;
579  vC -= mC * impulse * jvAc;
580  wC -= iC * impulse * jwC;
581  vD -= mD * impulse * jvBd;
582  wD -= iD * impulse * jwD;
583 
584  data.Velocities[indexA].V = vA;
585  data.Velocities[indexA].W = wA;
586  data.Velocities[indexB].V = vB;
587  data.Velocities[indexB].W = wB;
588  data.Velocities[indexC].V = vC;
589  data.Velocities[indexC].W = wC;
590  data.Velocities[indexD].V = vD;
591  data.Velocities[indexD].W = wD;
592  }

599  internal override bool SolvePositionConstraints(ref SolverData data)
600  {
601  Vector2 cA = data.Positions[indexA].C;
602  float aA = data.Positions[indexA].A;
603  Vector2 cB = data.Positions[indexB].C;
604  float aB = data.Positions[indexB].A;
605  Vector2 cC = data.Positions[indexC].C;
606  float aC = data.Positions[indexC].A;
607  Vector2 cD = data.Positions[indexD].C;
608  float aD = data.Positions[indexD].A;
609 
610  Rot qA = new Rot(aA), qB = new Rot(aB), qC = new Rot(aC), qD = new Rot(aD);
611 
612  const float linearError = 0.0f;
613 
614  float coordinateA, coordinateB;
615 
616  Vector2 jvAc, jvBd;
617  float jwA, jwB, jwC, jwD;
618  float mass = 0.0f;
619 
620  if (typeA == JointType.Revolute)
621  {
622  jvAc = Vector2.Zero;
623  jwA = 1.0f;
624  jwC = 1.0f;
625  mass += iA + iC;
626 
627  coordinateA = aA - aC - referenceAngleA;
628  }
629  else
630  {
631  Vector2 u = MathUtils.Mul(qC, localAxisC);
632  Vector2 rC = MathUtils.Mul(qC, localAnchorC - lcC);
633  Vector2 rA = MathUtils.Mul(qA, localAnchorA - lcA);
634  jvAc = u;
635  jwC = MathUtils.Cross(rC, u);
636  jwA = MathUtils.Cross(rA, u);
637  mass += mC + mA + iC * jwC * jwC + iA * jwA * jwA;
638 
639  Vector2 pC = localAnchorC - lcC;
640  Vector2 pA = MathUtils.MulT(qC, rA + (cA - cC));
641  coordinateA = Vector2.Dot(pA - pC, localAxisC);
642  }
643 
644  if (typeB == JointType.Revolute)
645  {
646  jvBd = Vector2.Zero;
647  jwB = ratio;
648  jwD = ratio;
649  mass += ratio * ratio * (iB + iD);
650 
651  coordinateB = aB - aD - referenceAngleB;
652  }
653  else
654  {
655  Vector2 u = MathUtils.Mul(qD, localAxisD);
656  Vector2 rD = MathUtils.Mul(qD, localAnchorD - lcD);
657  Vector2 rB = MathUtils.Mul(qB, localAnchorB - lcB);
658  jvBd = ratio * u;
659  jwD = ratio * MathUtils.Cross(rD, u);
660  jwB = ratio * MathUtils.Cross(rB, u);
661  mass += ratio * ratio * (mD + mB) + iD * jwD * jwD + iB * jwB * jwB;
662 
663  Vector2 pD = localAnchorD - lcD;
664  Vector2 pB = MathUtils.MulT(qD, rB + (cB - cD));
665  coordinateB = Vector2.Dot(pB - pD, localAxisD);
666  }
667 
668  float c = coordinateA + ratio * coordinateB - constant;
669 
670  float impulse = 0.0f;
671  if (mass > 0.0f)
672  {
673  impulse = -c / mass;
674  }
675 
676  cA += mA * impulse * jvAc;
677  aA += iA * impulse * jwA;
678  cB += mB * impulse * jvBd;
679  aB += iB * impulse * jwB;
680  cC -= mC * impulse * jvAc;
681  aC -= iC * impulse * jwC;
682  cD -= mD * impulse * jvBd;
683  aD -= iD * impulse * jwD;
684 
685  data.Positions[indexA].C = cA;
686  data.Positions[indexA].A = aA;
687  data.Positions[indexB].C = cB;
688  data.Positions[indexB].A = aB;
689  data.Positions[indexC].C = cC;
690  data.Positions[indexC].A = aC;
691  data.Positions[indexD].C = cD;
692  data.Positions[indexD].A = aD;
693 
694  // TODO_ERIN not implemented
695  return linearError < Settings.LinearSlop;
696  }
697  }
