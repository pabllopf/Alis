454  internal override void InitVelocityConstraints(ref SolverData data)
455  {
456  indexA = BodyA.IslandIndex;
457  indexB = BodyB.IslandIndex;
458  indexC = bodyC.IslandIndex;
459  indexD = bodyD.IslandIndex;
460  lcA = BodyA.Sweep.LocalCenter;
461  lcB = BodyB.Sweep.LocalCenter;
462  lcC = bodyC.Sweep.LocalCenter;
463  lcD = bodyD.Sweep.LocalCenter;
464  mA = BodyA.InvMass;
465  mB = BodyB.InvMass;
466  mC = bodyC.InvMass;
467  mD = bodyD.InvMass;
468  iA = BodyA.InvI;
469  iB = BodyB.InvI;
470  iC = bodyC.InvI;
471  iD = bodyD.InvI;
472 
473  float aA = data.Positions[indexA].A;
474  Vector2F vA = data.Velocities[indexA].V;
475  float wA = data.Velocities[indexA].W;
476 
477  float aB = data.Positions[indexB].A;
478  Vector2F vB = data.Velocities[indexB].V;
479  float wB = data.Velocities[indexB].W;
480 
481  float aC = data.Positions[indexC].A;
482  Vector2F vC = data.Velocities[indexC].V;
483  float wC = data.Velocities[indexC].W;
484 
485  float aD = data.Positions[indexD].A;
486  Vector2F vD = data.Velocities[indexD].V;
487  float wD = data.Velocities[indexD].W;
488 
489  Rotation qA = new Rotation(aA), qB = new Rotation(aB), qC = new Rotation(aC), qD = new Rotation(aD);
490 
491  mass = 0.0f;
492 
493  if (typeA == JointType.Revolute)
494  {
495  jvAc = Vector2F.Zero;
496  jwA = 1.0f;
497  jwC = 1.0f;
498  mass += iA + iC;
499  }
500  else
501  {
502  Vector2F u = MathUtils.Mul(qC, localAxisC);
503  Vector2F rC = MathUtils.Mul(qC, localAnchorC - lcC);
504  Vector2F rA = MathUtils.Mul(qA, localAnchorA - lcA);
505  jvAc = u;
506  jwC = MathUtils.Cross(rC, u);
507  jwA = MathUtils.Cross(rA, u);
508  mass += mC + mA + iC * jwC * jwC + iA * jwA * jwA;
509  }
510 
511  if (typeB == JointType.Revolute)
512  {
513  jvBd = Vector2F.Zero;
514  jwB = ratio;
515  jwD = ratio;
516  mass += ratio * ratio * (iB + iD);
517  }
518  else
519  {
520  Vector2F u = MathUtils.Mul(qD, localAxisD);
521  Vector2F rD = MathUtils.Mul(qD, localAnchorD - lcD);
522  Vector2F rB = MathUtils.Mul(qB, localAnchorB - lcB);
523  jvBd = ratio * u;
524  jwD = ratio * MathUtils.Cross(rD, u);
525  jwB = ratio * MathUtils.Cross(rB, u);
526  mass += ratio * ratio * (mD + mB) + iD * jwD * jwD + iB * jwB * jwB;
527  }
528 
529  // Compute effective mass.
530  mass = mass > 0.0f ? 1.0f / mass : 0.0f;
531 
532  if (data.Step.WarmStarting)
533  {
534  vA += mA * impulse * jvAc;
535  wA += iA * impulse * jwA;
536  vB += mB * impulse * jvBd;
537  wB += iB * impulse * jwB;
538  vC -= mC * impulse * jvAc;
539  wC -= iC * impulse * jwC;
540  vD -= mD * impulse * jvBd;
541  wD -= iD * impulse * jwD;
542  }
543  else
544  {
545  impulse = 0.0f;
546  }
547 
548  data.Velocities[indexA].V = vA;
549  data.Velocities[indexA].W = wA;
550  data.Velocities[indexB].V = vB;
551  data.Velocities[indexB].W = wB;
552  data.Velocities[indexC].V = vC;
553  data.Velocities[indexC].W = wC;
554  data.Velocities[indexD].V = vD;
555  data.Velocities[indexD].W = wD;
556  }

562  internal override void SolveVelocityConstraints(ref SolverData data)
563  {
564  Vector2F vA = data.Velocities[indexA].V;
565  float wA = data.Velocities[indexA].W;
566  Vector2F vB = data.Velocities[indexB].V;
567  float wB = data.Velocities[indexB].W;
568  Vector2F vC = data.Velocities[indexC].V;
569  float wC = data.Velocities[indexC].W;
570  Vector2F vD = data.Velocities[indexD].V;
571  float wD = data.Velocities[indexD].W;
572 
573  float dot = Vector2F.Dot(jvAc, vA - vC) + Vector2F.Dot(jvBd, vB - vD);
574  dot += jwA * wA - jwC * wC + (jwB * wB - jwD * wD);
575 
576  float impulseLocal = -mass * dot;
577  this.impulse += impulseLocal;
578 
579  vA += mA * impulseLocal * jvAc;
580  wA += iA * impulseLocal * jwA;
581  vB += mB * impulseLocal * jvBd;
582  wB += iB * impulseLocal * jwB;
583  vC -= mC * impulseLocal * jvAc;
584  wC -= iC * impulseLocal * jwC;
585  vD -= mD * impulseLocal * jvBd;
586  wD -= iD * impulseLocal * jwD;
587 
588  data.Velocities[indexA].V = vA;
589  data.Velocities[indexA].W = wA;
590  data.Velocities[indexB].V = vB;
591  data.Velocities[indexB].W = wB;
592  data.Velocities[indexC].V = vC;
593  data.Velocities[indexC].W = wC;
594  data.Velocities[indexD].V = vD;
595  data.Velocities[indexD].W = wD;
596  }

603  internal override bool SolvePositionConstraints(ref SolverData data)
604  {
605  Vector2F cA = data.Positions[indexA].C;
606  float aA = data.Positions[indexA].A;
607  Vector2F cB = data.Positions[indexB].C;
608  float aB = data.Positions[indexB].A;
609  Vector2F cC = data.Positions[indexC].C;
610  float aC = data.Positions[indexC].A;
611  Vector2F cD = data.Positions[indexD].C;
612  float aD = data.Positions[indexD].A;
613 
614  Rotation qA = new Rotation(aA), qB = new Rotation(aB), qC = new Rotation(aC), qD = new Rotation(aD);
615 
616  const float linearError = 0.0f;
617 
618  float coordinateA, coordinateB;
619 
620  Vector2F jvAcLocal, jvBdLocal;
621  float jwALocal, jwBLocal, jwCLocal, jwDLocal;
622  float massLocal = 0.0f;
623 
624  if (typeA == JointType.Revolute)
625  {
626  jvAcLocal = Vector2F.Zero;
627  jwALocal = 1.0f;
628  jwCLocal = 1.0f;
629  massLocal += iA + iC;
630 
631  coordinateA = aA - aC - referenceAngleA;
632  }
633  else
634  {
635  Vector2F u = MathUtils.Mul(qC, localAxisC);
636  Vector2F rC = MathUtils.Mul(qC, localAnchorC - lcC);
637  Vector2F rA = MathUtils.Mul(qA, localAnchorA - lcA);
638  jvAcLocal = u;
639  jwCLocal = MathUtils.Cross(rC, u);
640  jwALocal = MathUtils.Cross(rA, u);
641  massLocal += mC + mA + iC * jwCLocal * jwCLocal + iA * jwALocal * jwALocal;
642 
643  Vector2F pC = localAnchorC - lcC;
644  Vector2F pA = MathUtils.MulT(qC, rA + (cA - cC));
645  coordinateA = Vector2F.Dot(pA - pC, localAxisC);
646  }
647 
648  if (typeB == JointType.Revolute)
649  {
650  jvBdLocal = Vector2F.Zero;
651  jwBLocal = ratio;
652  jwDLocal = ratio;
653  massLocal += ratio * ratio * (iB + iD);
654 
655  coordinateB = aB - aD - referenceAngleB;
656  }
657  else
658  {
659  Vector2F u = MathUtils.Mul(qD, localAxisD);
660  Vector2F rD = MathUtils.Mul(qD, localAnchorD - lcD);
661  Vector2F rB = MathUtils.Mul(qB, localAnchorB - lcB);
662  jvBdLocal = ratio * u;
663  jwDLocal = ratio * MathUtils.Cross(rD, u);
664  jwBLocal = ratio * MathUtils.Cross(rB, u);
665  massLocal += ratio * ratio * (mD + mB) + iD * jwDLocal * jwDLocal + iB * jwBLocal * jwBLocal;
666 
667  Vector2F pD = localAnchorD - lcD;
668  Vector2F pB = MathUtils.MulT(qD, rB + (cB - cD));
669  coordinateB = Vector2F.Dot(pB - pD, localAxisD);
670  }
671 
672  float c = coordinateA + ratio * coordinateB - constant;
673 
674  float impulseLocal = 0.0f;
675  if (massLocal > 0.0f)
676  {
677  impulseLocal = -c / massLocal;
678  }
679 
680  cA += mA * impulseLocal * jvAcLocal;
681  aA += iA * impulseLocal * jwALocal;
682  cB += mB * impulseLocal * jvBdLocal;
683  aB += iB * impulseLocal * jwBLocal;
684  cC -= mC * impulseLocal * jvAcLocal;
685  aC -= iC * impulseLocal * jwCLocal;
686  cD -= mD * impulseLocal * jvBdLocal;
687  aD -= iD * impulseLocal * jwDLocal;
688 
689  data.Positions[indexA].C = cA;
690  data.Positions[indexA].A = aA;
691  data.Positions[indexB].C = cB;
692  data.Positions[indexB].A = aB;
693  data.Positions[indexC].C = cC;
694  data.Positions[indexC].A = aC;
695  data.Positions[indexD].C = cD;
696  data.Positions[indexD].A = aD;
697 
698  return linearError < Settings.LinearSlop;
699  }
