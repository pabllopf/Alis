39  // Gear Joint:
40  // C0 = (coordinate1 + ratio * coordinate2)_initial
41  // C = (coordinate1 + ratio * coordinate2) - C0 = 0
42  // J = [J1 ratio * J2]
43  // K = J * invM * JT
44  // = J1 * invM1 * J1T + ratio * ratio * J2 * invM2 * J2T
45  //
46  // Revolute:
47  // coordinate = rotation
48  // Cdot = angularVelocity
49  // J = [0 0 1]
50  // K = J * invM * JT = invI
51  //
52  // Prismatic:
53  // coordinate = dot(p - pg, ug)
54  // Cdot = dot(v + cross(w, r), ug)
55  // J = [ug cross(r, ug)]
56  // K = J * invM * JT = invMass + invI * cross(r, ug)^2

462  internal override void InitVelocityConstraints(ref SolverData data)
463  {
464  indexA = BodyA.IslandIndex;
465  indexB = BodyB.IslandIndex;
466  indexC = bodyC.IslandIndex;
467  indexD = bodyD.IslandIndex;
468  lcA = BodyA.Sweep.LocalCenter;
469  lcB = BodyB.Sweep.LocalCenter;
470  lcC = bodyC.Sweep.LocalCenter;
471  lcD = bodyD.Sweep.LocalCenter;
472  mA = BodyA.InvMass;
473  mB = BodyB.InvMass;
474  mC = bodyC.InvMass;
475  mD = bodyD.InvMass;
476  iA = BodyA.InvI;
477  iB = BodyB.InvI;
478  iC = bodyC.InvI;
479  iD = bodyD.InvI;
480 
481  float aA = data.Positions[indexA].A;
482  Vector2F vA = data.Velocities[indexA].V;
483  float wA = data.Velocities[indexA].W;
484 
485  float aB = data.Positions[indexB].A;
486  Vector2F vB = data.Velocities[indexB].V;
487  float wB = data.Velocities[indexB].W;
488 
489  float aC = data.Positions[indexC].A;
490  Vector2F vC = data.Velocities[indexC].V;
491  float wC = data.Velocities[indexC].W;
492 
493  float aD = data.Positions[indexD].A;
494  Vector2F vD = data.Velocities[indexD].V;
495  float wD = data.Velocities[indexD].W;
496 
497  Rotation qA = new Rotation(aA), qB = new Rotation(aB), qC = new Rotation(aC), qD = new Rotation(aD);
498 
499  mass = 0.0f;
500 
501  if (typeA == JointType.Revolute)
502  {
503  jvAc = Vector2F.Zero;
504  jwA = 1.0f;
505  jwC = 1.0f;
506  mass += iA + iC;
507  }
508  else
509  {
510  Vector2F u = MathUtils.Mul(qC, localAxisC);
511  Vector2F rC = MathUtils.Mul(qC, localAnchorC - lcC);
512  Vector2F rA = MathUtils.Mul(qA, localAnchorA - lcA);
513  jvAc = u;
514  jwC = MathUtils.Cross(rC, u);
515  jwA = MathUtils.Cross(rA, u);
516  mass += mC + mA + iC * jwC * jwC + iA * jwA * jwA;
517  }
518 
519  if (typeB == JointType.Revolute)
520  {
521  jvBd = Vector2F.Zero;
522  jwB = ratio;
523  jwD = ratio;
524  mass += ratio * ratio * (iB + iD);
525  }
526  else
527  {
528  Vector2F u = MathUtils.Mul(qD, localAxisD);
529  Vector2F rD = MathUtils.Mul(qD, localAnchorD - lcD);
530  Vector2F rB = MathUtils.Mul(qB, localAnchorB - lcB);
531  jvBd = ratio * u;
532  jwD = ratio * MathUtils.Cross(rD, u);
533  jwB = ratio * MathUtils.Cross(rB, u);
534  mass += ratio * ratio * (mD + mB) + iD * jwD * jwD + iB * jwB * jwB;
535  }
536 
537  // Compute effective mass.
538  mass = mass > 0.0f ? 1.0f / mass : 0.0f;
539 
540  if (data.Step.WarmStarting)
541  {
542  vA += mA * impulse * jvAc;
543  wA += iA * impulse * jwA;
544  vB += mB * impulse * jvBd;
545  wB += iB * impulse * jwB;
546  vC -= mC * impulse * jvAc;
547  wC -= iC * impulse * jwC;
548  vD -= mD * impulse * jvBd;
549  wD -= iD * impulse * jwD;
550  }
551  else
552  {
553  impulse = 0.0f;
554  }
555 
556  data.Velocities[indexA].V = vA;
557  data.Velocities[indexA].W = wA;
558  data.Velocities[indexB].V = vB;
559  data.Velocities[indexB].W = wB;
560  data.Velocities[indexC].V = vC;
561  data.Velocities[indexC].W = wC;
562  data.Velocities[indexD].V = vD;
563  data.Velocities[indexD].W = wD;
564  }

570  internal override void SolveVelocityConstraints(ref SolverData data)
571  {
572  Vector2F vA = data.Velocities[indexA].V;
573  float wA = data.Velocities[indexA].W;
574  Vector2F vB = data.Velocities[indexB].V;
575  float wB = data.Velocities[indexB].W;
576  Vector2F vC = data.Velocities[indexC].V;
577  float wC = data.Velocities[indexC].W;
578  Vector2F vD = data.Velocities[indexD].V;
579  float wD = data.Velocities[indexD].W;
580 
581  float cdot = Vector2F.Dot(jvAc, vA - vC) + Vector2F.Dot(jvBd, vB - vD);
582  cdot += jwA * wA - jwC * wC + (jwB * wB - jwD * wD);
583 
584  float impulse = -mass * cdot;
585  this.impulse += impulse;
586 
587  vA += mA * impulse * jvAc;
588  wA += iA * impulse * jwA;
589  vB += mB * impulse * jvBd;
590  wB += iB * impulse * jwB;
591  vC -= mC * impulse * jvAc;
592  wC -= iC * impulse * jwC;
593  vD -= mD * impulse * jvBd;
594  wD -= iD * impulse * jwD;
595 
596  data.Velocities[indexA].V = vA;
597  data.Velocities[indexA].W = wA;
598  data.Velocities[indexB].V = vB;
599  data.Velocities[indexB].W = wB;
600  data.Velocities[indexC].V = vC;
601  data.Velocities[indexC].W = wC;
602  data.Velocities[indexD].V = vD;
603  data.Velocities[indexD].W = wD;
604  }

611  internal override bool SolvePositionConstraints(ref SolverData data)
612  {
613  Vector2F cA = data.Positions[indexA].C;
614  float aA = data.Positions[indexA].A;
615  Vector2F cB = data.Positions[indexB].C;
616  float aB = data.Positions[indexB].A;
617  Vector2F cC = data.Positions[indexC].C;
618  float aC = data.Positions[indexC].A;
619  Vector2F cD = data.Positions[indexD].C;
620  float aD = data.Positions[indexD].A;
621 
622  Rotation qA = new Rotation(aA), qB = new Rotation(aB), qC = new Rotation(aC), qD = new Rotation(aD);
623 
624  const float linearError = 0.0f;
625 
626  float coordinateA, coordinateB;
627 
628  Vector2F jvAc, jvBd;
629  float jwA, jwB, jwC, jwD;
630  float mass = 0.0f;
631 
632  if (typeA == JointType.Revolute)
633  {
634  jvAc = Vector2F.Zero;
635  jwA = 1.0f;
636  jwC = 1.0f;
637  mass += iA + iC;
638 
639  coordinateA = aA - aC - referenceAngleA;
640  }
641  else
642  {
643  Vector2F u = MathUtils.Mul(qC, localAxisC);
644  Vector2F rC = MathUtils.Mul(qC, localAnchorC - lcC);
645  Vector2F rA = MathUtils.Mul(qA, localAnchorA - lcA);
646  jvAc = u;
647  jwC = MathUtils.Cross(rC, u);
648  jwA = MathUtils.Cross(rA, u);
649  mass += mC + mA + iC * jwC * jwC + iA * jwA * jwA;
650 
651  Vector2F pC = localAnchorC - lcC;
652  Vector2F pA = MathUtils.MulT(qC, rA + (cA - cC));
653  coordinateA = Vector2F.Dot(pA - pC, localAxisC);
654  }
655 
656  if (typeB == JointType.Revolute)
657  {
658  jvBd = Vector2F.Zero;
659  jwB = ratio;
660  jwD = ratio;
661  mass += ratio * ratio * (iB + iD);
662 
663  coordinateB = aB - aD - referenceAngleB;
664  }
665  else
666  {
667  Vector2F u = MathUtils.Mul(qD, localAxisD);
668  Vector2F rD = MathUtils.Mul(qD, localAnchorD - lcD);
669  Vector2F rB = MathUtils.Mul(qB, localAnchorB - lcB);
670  jvBd = ratio * u;
671  jwD = ratio * MathUtils.Cross(rD, u);
672  jwB = ratio * MathUtils.Cross(rB, u);
673  mass += ratio * ratio * (mD + mB) + iD * jwD * jwD + iB * jwB * jwB;
674 
675  Vector2F pD = localAnchorD - lcD;
676  Vector2F pB = MathUtils.MulT(qD, rB + (cB - cD));
677  coordinateB = Vector2F.Dot(pB - pD, localAxisD);
678  }
679 
680  float c = coordinateA + ratio * coordinateB - constant;
681 
682  float impulse = 0.0f;
683  if (mass > 0.0f)
684  {
685  impulse = -c / mass;
686  }
687 
688  cA += mA * impulse * jvAc;
689  aA += iA * impulse * jwA;
690  cB += mB * impulse * jvBd;
691  aB += iB * impulse * jwB;
692  cC -= mC * impulse * jvAc;
693  aC -= iC * impulse * jwC;
694  cD -= mD * impulse * jvBd;
695  aD -= iD * impulse * jwD;
696 
697  data.Positions[indexA].C = cA;
698  data.Positions[indexA].A = aA;
699  data.Positions[indexB].C = cB;
700  data.Positions[indexB].A = aB;
701  data.Positions[indexC].C = cC;
702  data.Positions[indexC].A = aC;
703  data.Positions[indexD].C = cD;
704  data.Positions[indexD].A = aD;
705 
706  // TODO_ERIN not implemented
707  return linearError < Settings.LinearSlop;
708  }
