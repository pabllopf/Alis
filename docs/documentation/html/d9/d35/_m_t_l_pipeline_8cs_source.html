109  public MTLPipeline(ref GraphicsPipelineDescription description, MTLGraphicsDevice gd)
110  : base(ref description)
111  {
112  PrimitiveType = MTLFormats.VdToMTLPrimitiveTopology(description.PrimitiveTopology);
113  ResourceLayouts = new MTLResourceLayout[description.ResourceLayouts.Length];
114  NonVertexBufferCount = 0;
115  for (int i = 0; i < ResourceLayouts.Length; i++)
116  {
117  ResourceLayouts[i] = Util.AssertSubtype<ResourceLayout, MTLResourceLayout>(description.ResourceLayouts[i]);
118  NonVertexBufferCount += ResourceLayouts[i].BufferCount;
119  }
120  ResourceBindingModel = description.ResourceBindingModel ?? gd.ResourceBindingModel;
121 
122  CullMode = MTLFormats.VdToMTLCullMode(description.RasterizerState.CullMode);
123  FrontFace = MTLFormats.VdVoMTLFrontFace(description.RasterizerState.FrontFace);
124  FillMode = MTLFormats.VdToMTLFillMode(description.RasterizerState.FillMode);
125  ScissorTestEnabled = description.RasterizerState.ScissorTestEnabled;
126 
127  MTLRenderPipelineDescriptor mtlDesc = MTLRenderPipelineDescriptor.New();
128  foreach (Shader shader in description.ShaderSet.Shaders)
129  {
130  MTLShader mtlShader = Util.AssertSubtype<Shader, MTLShader>(shader);
131  MTLFunction specializedFunction;
132 
133  if (mtlShader.HasFunctionConstants)
134  {
135  // Need to create specialized MTLFunction.
136  MTLFunctionConstantValues constantValues = CreateConstantValues(description.ShaderSet.Specializations);
137  specializedFunction = mtlShader.Library.newFunctionWithNameConstantValues(mtlShader.EntryPoint, constantValues);
138  AddSpecializedFunction(specializedFunction);
139  ObjectiveCRuntime.release(constantValues.NativePtr);
140 
141  Debug.Assert(specializedFunction.NativePtr != IntPtr.Zero, "Failed to create specialized MTLFunction");
142  }
143  else
144  {
145  specializedFunction = mtlShader.Function;
146  }
147 
148  if (shader.Stage == ShaderStages.Vertex)
149  {
150  mtlDesc.vertexFunction = specializedFunction;
151  }
152  else if (shader.Stage == ShaderStages.Fragment)
153  {
154  mtlDesc.fragmentFunction = specializedFunction;
155  }
156  }
157 
158  // Vertex layouts
159  VertexLayoutDescription[] vdVertexLayouts = description.ShaderSet.VertexLayouts;
160  MTLVertexDescriptor vertexDescriptor = mtlDesc.vertexDescriptor;
161 
162  for (uint i = 0; i < vdVertexLayouts.Length; i++)
163  {
164  uint layoutIndex = ResourceBindingModel == ResourceBindingModel.Improved
165  ? NonVertexBufferCount + i
166  : i;
167  MTLVertexBufferLayoutDescriptor mtlLayout = vertexDescriptor.layouts[layoutIndex];
168  mtlLayout.stride = (UIntPtr)vdVertexLayouts[i].Stride;
169  uint stepRate = vdVertexLayouts[i].InstanceStepRate;
170  mtlLayout.stepFunction = stepRate == 0 ? MTLVertexStepFunction.PerVertex : MTLVertexStepFunction.PerInstance;
171  mtlLayout.stepRate = (UIntPtr)Math.Max(1, stepRate);
172  }
173 
174  uint element = 0;
175  for (uint i = 0; i < vdVertexLayouts.Length; i++)
176  {
177  uint offset = 0;
178  VertexLayoutDescription vdDesc = vdVertexLayouts[i];
179  for (uint j = 0; j < vdDesc.Elements.Length; j++)
180  {
181  VertexElementDescription elementDesc = vdDesc.Elements[j];
182  MTLVertexAttributeDescriptor mtlAttribute = vertexDescriptor.attributes[element];
183  mtlAttribute.bufferIndex = (UIntPtr)(ResourceBindingModel == ResourceBindingModel.Improved
184  ? NonVertexBufferCount + i
185  : i);
186  mtlAttribute.format = MTLFormats.VdToMTLVertexFormat(elementDesc.Format);
187  mtlAttribute.offset = elementDesc.Offset != 0 ? (UIntPtr)elementDesc.Offset : (UIntPtr)offset;
188  offset += FormatSizeHelpers.GetSizeInBytes(elementDesc.Format);
189  element += 1;
190  }
191  }
192 
193  VertexBufferCount = (uint)vdVertexLayouts.Length;
194 
195  // Outputs
196  OutputDescription outputs = description.Outputs;
197  BlendStateDescription blendStateDesc = description.BlendState;
198  BlendColor = blendStateDesc.BlendFactor;
199 
200  if (outputs.SampleCount != TextureSampleCount.Count1)
201  {
202  mtlDesc.sampleCount = (UIntPtr)FormatHelpers.GetSampleCountUInt32(outputs.SampleCount);
203  }
204 
205  if (outputs.DepthAttachment != null)
206  {
207  PixelFormat depthFormat = outputs.DepthAttachment.Value.Format;
208  MTLPixelFormat mtlDepthFormat = MTLFormats.VdToMTLPixelFormat(depthFormat, true);
209  mtlDesc.depthAttachmentPixelFormat = mtlDepthFormat;
210  if ((FormatHelpers.IsStencilFormat(depthFormat)))
211  {
212  HasStencil = true;
213  mtlDesc.stencilAttachmentPixelFormat = mtlDepthFormat;
214  }
215  }
216  for (uint i = 0; i < outputs.ColorAttachments.Length; i++)
217  {
218  BlendAttachmentDescription attachmentBlendDesc = blendStateDesc.AttachmentStates[i];
219  MTLRenderPipelineColorAttachmentDescriptor colorDesc = mtlDesc.colorAttachments[i];
220  colorDesc.pixelFormat = MTLFormats.VdToMTLPixelFormat(outputs.ColorAttachments[i].Format, false);
221  colorDesc.blendingEnabled = attachmentBlendDesc.BlendEnabled;
222  colorDesc.writeMask = MTLFormats.VdToMTLColorWriteMask(attachmentBlendDesc.ColorWriteMask.GetOrDefault());
223  colorDesc.alphaBlendOperation = MTLFormats.VdToMTLBlendOp(attachmentBlendDesc.AlphaFunction);
224  colorDesc.sourceAlphaBlendFactor = MTLFormats.VdToMTLBlendFactor(attachmentBlendDesc.SourceAlphaFactor);
225  colorDesc.destinationAlphaBlendFactor = MTLFormats.VdToMTLBlendFactor(attachmentBlendDesc.DestinationAlphaFactor);
226 
227  colorDesc.rgbBlendOperation = MTLFormats.VdToMTLBlendOp(attachmentBlendDesc.ColorFunction);
228  colorDesc.sourceRGBBlendFactor = MTLFormats.VdToMTLBlendFactor(attachmentBlendDesc.SourceColorFactor);
229  colorDesc.destinationRGBBlendFactor = MTLFormats.VdToMTLBlendFactor(attachmentBlendDesc.DestinationColorFactor);
230  }
231 
232  mtlDesc.alphaToCoverageEnabled = blendStateDesc.AlphaToCoverageEnabled;
233 
234  RenderPipelineState = gd.Device.newRenderPipelineStateWithDescriptor(mtlDesc);
235  ObjectiveCRuntime.release(mtlDesc.NativePtr);
236 
237  if (outputs.DepthAttachment != null)
238  {
239  MTLDepthStencilDescriptor depthDescriptor = MTLUtil.AllocInit<MTLDepthStencilDescriptor>(
240  nameof(MTLDepthStencilDescriptor));
241  depthDescriptor.depthCompareFunction = MTLFormats.VdToMTLCompareFunction(
242  description.DepthStencilState.DepthComparison);
243  depthDescriptor.depthWriteEnabled = description.DepthStencilState.DepthWriteEnabled;
244 
245  bool stencilEnabled = description.DepthStencilState.StencilTestEnabled;
246  if (stencilEnabled)
247  {
248  StencilReference = description.DepthStencilState.StencilReference;
249 
250  StencilBehaviorDescription vdFrontDesc = description.DepthStencilState.StencilFront;
251  MTLStencilDescriptor front = MTLUtil.AllocInit<MTLStencilDescriptor>(nameof(MTLStencilDescriptor));
252  front.readMask = stencilEnabled ? description.DepthStencilState.StencilReadMask : 0u;
253  front.writeMask = stencilEnabled ? description.DepthStencilState.StencilWriteMask : 0u;
254  front.depthFailureOperation = MTLFormats.VdToMTLStencilOperation(vdFrontDesc.DepthFail);
255  front.stencilFailureOperation = MTLFormats.VdToMTLStencilOperation(vdFrontDesc.Fail);
256  front.depthStencilPassOperation = MTLFormats.VdToMTLStencilOperation(vdFrontDesc.Pass);
257  front.stencilCompareFunction = MTLFormats.VdToMTLCompareFunction(vdFrontDesc.Comparison);
258  depthDescriptor.frontFaceStencil = front;
259 
260  StencilBehaviorDescription vdBackDesc = description.DepthStencilState.StencilBack;
261  MTLStencilDescriptor back = MTLUtil.AllocInit<MTLStencilDescriptor>(nameof(MTLStencilDescriptor));
262  back.readMask = stencilEnabled ? description.DepthStencilState.StencilReadMask : 0u;
263  back.writeMask = stencilEnabled ? description.DepthStencilState.StencilWriteMask : 0u;
264  back.depthFailureOperation = MTLFormats.VdToMTLStencilOperation(vdBackDesc.DepthFail);
265  back.stencilFailureOperation = MTLFormats.VdToMTLStencilOperation(vdBackDesc.Fail);
266  back.depthStencilPassOperation = MTLFormats.VdToMTLStencilOperation(vdBackDesc.Pass);
267  back.stencilCompareFunction = MTLFormats.VdToMTLCompareFunction(vdBackDesc.Comparison);
268  depthDescriptor.backFaceStencil = back;
269 
270  ObjectiveCRuntime.release(front.NativePtr);
271  ObjectiveCRuntime.release(back.NativePtr);
272  }
273 
274  DepthStencilState = gd.Device.newDepthStencilStateWithDescriptor(depthDescriptor);
275  ObjectiveCRuntime.release(depthDescriptor.NativePtr);
276  }
277 
278  DepthClipMode = description.DepthStencilState.DepthTestEnabled ? MTLDepthClipMode.Clip : MTLDepthClipMode.Clamp;
279  }

286  public MTLPipeline(ref ComputePipelineDescription description, MTLGraphicsDevice gd)
287  : base(ref description)
288  {
289  IsComputePipeline = true;
290  ResourceLayouts = new MTLResourceLayout[description.ResourceLayouts.Length];
291  for (int i = 0; i < ResourceLayouts.Length; i++)
292  {
293  ResourceLayouts[i] = Util.AssertSubtype<ResourceLayout, MTLResourceLayout>(description.ResourceLayouts[i]);
294  }
295 
296  ThreadsPerThreadgroup = new MTLSize(
297  description.ThreadGroupSizeX,
298  description.ThreadGroupSizeY,
299  description.ThreadGroupSizeZ);
300 
301  MTLComputePipelineDescriptor mtlDesc = MTLUtil.AllocInit<MTLComputePipelineDescriptor>(
302  nameof(MTLComputePipelineDescriptor));
303  MTLShader mtlShader = Util.AssertSubtype<Shader, MTLShader>(description.ComputeShader);
304  MTLFunction specializedFunction;
305  if (mtlShader.HasFunctionConstants)
306  {
307  // Need to create specialized MTLFunction.
308  MTLFunctionConstantValues constantValues = CreateConstantValues(description.Specializations);
309  specializedFunction = mtlShader.Library.newFunctionWithNameConstantValues(mtlShader.EntryPoint, constantValues);
310  AddSpecializedFunction(specializedFunction);
311  ObjectiveCRuntime.release(constantValues.NativePtr);
312 
313  Debug.Assert(specializedFunction.NativePtr != IntPtr.Zero, "Failed to create specialized MTLFunction");
314  }
315  else
316  {
317  specializedFunction = mtlShader.Function;
318  }
319 
320  mtlDesc.computeFunction = specializedFunction;
321  MTLPipelineBufferDescriptorArray buffers = mtlDesc.buffers;
322  uint bufferIndex = 0;
323  foreach (MTLResourceLayout layout in ResourceLayouts)
324  {
325  foreach (ResourceLayoutElementDescription rle in layout.Description.Elements)
326  {
327  ResourceKind kind = rle.Kind;
328  if (kind == ResourceKind.UniformBuffer
329  || kind == ResourceKind.StructuredBufferReadOnly)
330  {
331  MTLPipelineBufferDescriptor bufferDesc = buffers[bufferIndex];
332  bufferDesc.mutability = MTLMutability.Immutable;
333  bufferIndex += 1;
334  }
335  else if (kind == ResourceKind.StructuredBufferReadWrite)
336  {
337  MTLPipelineBufferDescriptor bufferDesc = buffers[bufferIndex];
338  bufferDesc.mutability = MTLMutability.Mutable;
339  bufferIndex += 1;
340  }
341  }
342  }
343 
344  ComputePipelineState = gd.Device.newComputePipelineStateWithDescriptor(mtlDesc);
345 
346  ObjectiveCRuntime.release(mtlDesc.NativePtr);
347  }

399  {

401  }
402  _specializedFunctions.Clear();
403  }
404 
405  _disposed = true;
406  }
