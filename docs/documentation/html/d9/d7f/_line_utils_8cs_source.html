49  public static float DistanceBetweenPointAndLineSegment(ref Vector2 point, ref Vector2 start, ref Vector2 end)
50  {
51  if (start == end)
52  {
53  return Vector2.Distance(point, start);
54  }
55 
56  Vector2 v = Vector2.Subtract(end, start);
57  Vector2 w = Vector2.Subtract(point, start);
58 
59  float c1 = Vector2.Dot(w, v);
60  if (c1 <= 0)
61  {
62  return Vector2.Distance(point, start);
63  }
64 
65  float c2 = Vector2.Dot(v, v);
66  if (c2 <= c1)
67  {
68  return Vector2.Distance(point, end);
69  }
70 
71  float b = c1 / c2;
72  Vector2 pointOnLine = Vector2.Add(start, Vector2.Multiply(v, b));
73  return Vector2.Distance(point, pointOnLine);
74  }

81  public static bool LineIntersect2(ref Vector2 a0, ref Vector2 a1, ref Vector2 b0, ref Vector2 b1,
82  out Vector2 intersectionPoint)
83  {
84  intersectionPoint = Vector2.Zero;
85 
86  if (a0 == b0 || a0 == b1 || a1 == b0 || a1 == b1)
87  {
88  return false;
89  }
90 
91  float x1 = a0.X;
92  float y1 = a0.Y;
93  float x2 = a1.X;
94  float y2 = a1.Y;
95  float x3 = b0.X;
96  float y3 = b0.Y;
97  float x4 = b1.X;
98  float y4 = b1.Y;
99 
100  //AABB early exit
101  if (Math.Max(x1, x2) < Math.Min(x3, x4) || Math.Max(x3, x4) < Math.Min(x1, x2))
102  {
103  return false;
104  }
105 
106  if (Math.Max(y1, y2) < Math.Min(y3, y4) || Math.Max(y3, y4) < Math.Min(y1, y2))
107  {
108  return false;
109  }
110 
111  float ua = (x4 - x3) * (y1 - y3) - (y4 - y3) * (x1 - x3);
112  float ub = (x2 - x1) * (y1 - y3) - (y2 - y1) * (x1 - x3);
113  float denom = (y4 - y3) * (x2 - x1) - (x4 - x3) * (y2 - y1);
114  if (Math.Abs(denom) < MathConstants.Epsilon)
115  {
116  //Lines are too close to parallel to call
117  return false;
118  }
119 
120  ua /= denom;
121  ub /= denom;
122 
123  if (0 < ua && ua < 1 && 0 < ub && ub < 1)
124  {
125  intersectionPoint.X = x1 + ua * (x2 - x1);
126  intersectionPoint.Y = y1 + ua * (y2 - y1);
127  return true;
128  }
129 
130  return false;
131  }

142  public static Vector2 LineIntersect(Vector2 p1, Vector2 p2, Vector2 q1, Vector2 q2)
143  {
144  Vector2 i = Vector2.Zero;
145  float a1 = p2.Y - p1.Y;
146  float b1 = p1.X - p2.X;
147  float c1 = a1 * p1.X + b1 * p1.Y;
148  float a2 = q2.Y - q1.Y;
149  float b2 = q1.X - q2.X;
150  float c2 = a2 * q1.X + b2 * q1.Y;
151  float det = a1 * b2 - a2 * b1;
152 
153  if (!MathUtils.FloatEquals(det, 0))
154  {
155  // lines are not parallel
156  i.X = (b2 * c1 - b1 * c2) / det;
157  i.Y = (a1 * c2 - a2 * c1) / det;
158  }
159 
160  return i;
161  }

182  public static bool LineIntersect(ref Vector2 point1, ref Vector2 point2, ref Vector2 point3, ref Vector2 point4,
183  bool firstIsSegment, bool secondIsSegment, out Vector2 point)
184  {
185  point = new Vector2();
186 
187  // these are reused later.
188  // each lettered sub-calculation is used twice, except
189  // for b and d, which are used 3 times
190  float a = point4.Y - point3.Y;
191  float b = point2.X - point1.X;
192  float c = point4.X - point3.X;
193  float d = point2.Y - point1.Y;
194 
195  // denominator to solution of linear system
196  float denom = a * b - c * d;
197 
198  // if denominator is 0, then lines are parallel
199  if (!(denom >= -MathConstants.Epsilon && denom <= MathConstants.Epsilon))
200  {
201  float e = point1.Y - point3.Y;
202  float f = point1.X - point3.X;
203  float oneOverDenom = 1.0f / denom;
204 
205  // numerator of first equation
206  float ua = c * e - a * f;
207  ua *= oneOverDenom;
208 
209  // check if intersection point of the two lines is on line segment 1
210  if (!firstIsSegment || ua >= 0.0f && ua <= 1.0f)
211  {
212  // numerator of second equation
213  float ub = b * e - d * f;
214  ub *= oneOverDenom;
215 
216  // check if intersection point of the two lines is on line segment 2
217  // means the line segments intersect, since we know it is on
218  // segment 1 as well.
219  if (!secondIsSegment || ub >= 0.0f && ub <= 1.0f)
220  {
221  // check if they are coincident (no collision in this case)
222  if (ua != 0f || ub != 0f)
223  {
224  //There is an intersection
225  point.X = point1.X + ua * b;
226  point.Y = point1.Y + ua * d;
227  return true;
228  }
229  }
230  }
231  }
232 
233  return false;
234  }
