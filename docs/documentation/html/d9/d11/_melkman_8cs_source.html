52  public static Vertices GetConvexHull(Vertices vertices)
53  {
54  if (vertices.Count <= 3)
55  {
56  return vertices;
57  }
58 
59  //We'll never need a queue larger than the current number of Vertices +1
60  //Create double-ended queue
61  Vector2F[] deque = new Vector2F[vertices.Count + 1];
62  int qf = 3, qb = 0; //Queue front index, queue back index
63 
64  //Start by placing first 3 vertices in convex CCW order
65  int startIndex = 3;
66  float k = MathUtils.Area(vertices[0], vertices[1], vertices[2]);
67  if (k == 0)
68  {
69  //Vertices are collinear.
70  deque[0] = vertices[0];
71  deque[1] = vertices[2]; //We can skip vertex 1 because it should be between 0 and 2
72  deque[2] = vertices[0];
73  qf = 2;
74 
75  //Go until the end of the collinear sequence of vertices
76  for (startIndex = 3; startIndex < vertices.Count; startIndex++)
77  {
78  Vector2F tmp = vertices[startIndex];
79  if (MathUtils.Area(ref deque[0], ref deque[1], ref tmp) == 0) //This point is also collinear
80  {
81  deque[1] = vertices[startIndex];
82  }
83  else
84  {
85  break;
86  }
87  }
88  }
89  else
90  {
91  deque[0] = deque[3] = vertices[2];
92  if (k > 0)
93  {
94  //Is Left. Set deque = {2, 0, 1, 2}
95  deque[1] = vertices[0];
96  deque[2] = vertices[1];
97  }
98  else
99  {
100  //Is Right. Set deque = {2, 1, 0, 2}
101  deque[1] = vertices[1];
102  deque[2] = vertices[0];
103  }
104  }
105 
106  int qfm1 = qf == 0 ? deque.Length - 1 : qf - 1;
107  int qbm1 = qb == deque.Length - 1 ? 0 : qb + 1;
108 
109  //Add vertices one at a time and adjust convex hull as needed
110  for (int i = startIndex; i < vertices.Count; i++)
111  {
112  Vector2F nextPt = vertices[i];
113 
114  //Ignore if it is already within the convex hull we have constructed
115  if ((MathUtils.Area(ref deque[qfm1], ref deque[qf], ref nextPt) > 0) &&
116  (MathUtils.Area(ref deque[qb], ref deque[qbm1], ref nextPt) > 0))
117  {
118  continue;
119  }
120 
121  //Pop front until convex
122  while (!(MathUtils.Area(ref deque[qfm1], ref deque[qf], ref nextPt) > 0))
123  {
124  //Pop the front element from the queue
125  qf = qfm1; //qf--;
126  qfm1 = qf == 0 ? deque.Length - 1 : qf - 1; //qfm1 = qf - 1;
127  }
128 
129  //Add vertex to the front of the queue
130  qf = qf == deque.Length - 1 ? 0 : qf + 1; //qf++;
131  qfm1 = qf == 0 ? deque.Length - 1 : qf - 1; //qfm1 = qf - 1;
132  deque[qf] = nextPt;
133 
134  //Pop back until convex
135  while (!(MathUtils.Area(ref deque[qb], ref deque[qbm1], ref nextPt) > 0))
136  {
137  //Pop the back element from the queue
138  qb = qbm1; //qb++;
139  qbm1 = qb == deque.Length - 1 ? 0 : qb + 1; //qbm1 = qb + 1;
140  }
141 
142  //Add vertex to the back of the queue
143  qb = qb == 0 ? deque.Length - 1 : qb - 1; //qb--;
144  qbm1 = qb == deque.Length - 1 ? 0 : qb + 1; //qbm1 = qb + 1;
145  deque[qb] = nextPt;
146  }
147 
148  //Create the convex hull from what is left in the deque
149  if (qb < qf)
150  {
151  Vertices convexHull = new Vertices(qf);
152 
153  for (int i = qb; i < qf; i++)
154  {
155  convexHull.Add(deque[i]);
156  }
157 
158  return convexHull;
159  }
160  else
161  {
162  Vertices convexHull = new Vertices(qf + deque.Length);
163 
164  for (int i = 0; i < qf; i++)
165  {
166  convexHull.Add(deque[i]);
167  }
168 
169  for (int i = qb; i < deque.Length; i++)
170  {
171  convexHull.Add(deque[i]);
172  }
173 
174  return convexHull;
175  }
176  }
177  }
178 }
