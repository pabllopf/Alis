47  public static Vertices GetConvexHull(Vertices vertices)
48  {
49  if (vertices.Count <= 3)
50  {
51  return vertices;
52  }
53 
54  // Find the right most point on the hull
55  int i0 = 0;
56  float x0 = vertices[0].X;
57  for (int i = 1; i < vertices.Count; ++i)
58  {
59  float x = vertices[i].X;
60  if (x > x0 || x == x0 && vertices[i].Y < vertices[i0].Y)
61  {
62  i0 = i;
63  x0 = x;
64  }
65  }
66 
67  int[] hull = new int[vertices.Count];
68  int m = 0;
69  int ih = i0;
70 
71  for (;;)
72  {
73  hull[m] = ih;
74 
75  int ie = 0;
76  for (int j = 1; j < vertices.Count; ++j)
77  {
78  if (ie == ih)
79  {
80  ie = j;
81  continue;
82  }
83 
84  Vector2 r = vertices[ie] - vertices[hull[m]];
85  Vector2 v = vertices[j] - vertices[hull[m]];
86  float c = MathUtils.Cross(ref r, ref v);
87  if (c < 0.0f)
88  {
89  ie = j;
90  }
91 
92  // Collinearity check
93  if (c == 0.0f && v.LengthSquared() > r.LengthSquared())
94  {
95  ie = j;
96  }
97  }
98 
99  ++m;
100  ih = ie;
101 
102  if (ie == i0)
103  {
104  break;
105  }
106  }
107 
108  Vertices result = new Vertices(m);
109 
110  // Copy vertices.
111  for (int i = 0; i < m; ++i)
112  {
113  result.Add(vertices[hull[i]]);
114  }
115 
116  return result;
117  }
118  }
