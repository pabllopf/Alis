389  public static Vector2F Transform(Vector2F value, Quaternion rotation)
390  {
391  float x2 = rotation.X + rotation.X;
392  float y2 = rotation.Y + rotation.Y;
393  float z2 = rotation.Z + rotation.Z;
394 
395  float wz2 = rotation.W * z2;
396  float xx2 = rotation.X * x2;
397  float xy2 = rotation.X * y2;
398  float yy2 = rotation.Y * y2;
399  float zz2 = rotation.Z * z2;
400 
401  return new Vector2F(
402  value.X * (1.0f - yy2 - zz2) + value.Y * (xy2 - wz2),
403  value.X * (xy2 + wz2) + value.Y * (1.0f - xx2 - zz2)
404  );
405  }
