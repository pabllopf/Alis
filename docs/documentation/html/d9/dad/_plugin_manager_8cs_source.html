76  public void LoadPlugins(string pluginsDirectory)
77  {
78  string platformFolder = GetPlatformFolder();

80 
81  string platformPluginsDirectory = Path.Combine(pluginsDirectory, platformFolder);

99  private string GetPlatformFolder()
100  {
101  if (RuntimeInformation.IsOSPlatform(OSPlatform.Windows))
102  {
103  return "Windows";
104  }
105 
106  if (RuntimeInformation.IsOSPlatform(OSPlatform.OSX))
107  {
108  return IsRunningOniOS() ? "ios" : "osx";
109  }
110 
111  if (RuntimeInformation.IsOSPlatform(OSPlatform.Linux))
112  {
113  return IsRunningOnAndroid() ? "android" : "linux";
114  }
115 
116  throw new NotSupportedException("Unsupported platform. Plugins will not be loaded.");
117  }

145  private void LoadPluginFromFile(string pluginFile)
146  {
147  try
148  {
149  Assembly assembly = Assembly.LoadFrom(pluginFile);
150  loadedAssemblies.Add(assembly);
151 
152  Type[] types = assembly.GetTypes();
153 
154  foreach (Type type in types)
155  {
156  if (typeof(IPlugin).IsAssignableFrom(type))
157  {
158  IPlugin plugin = (IPlugin) Activator.CreateInstance(type);
159  loadedPlugins.Add(plugin);
160  }
161  }
162  }
163  catch (Exception ex)
164  {
165  Console.WriteLine($"Error loading plugin from file '{pluginFile}': {ex.Message}");
166  }
167  }

175  private static bool IsPluginFile(string filePath)
176  {
177  string extension = Path.GetExtension(filePath);
178  return extension == ".dll" || extension == ".so" || extension == ".dylib";
179  }
180 
184  private void UnloadPlugins()
185  {

188  plugin.Dispose();
189  }
190 
191  loadedPlugins.Clear();
192  loadedAssemblies.Clear();
193  }

200  private static bool IsRunningOniOS() => RuntimeInformation.IsOSPlatform(OSPlatform.OSX) && (RuntimeInformation.OSDescription.Contains("iPhone") || RuntimeInformation.OSDescription.Contains("iPad"));
201 
202 
207  private static bool IsRunningOnAndroid() => RuntimeInformation.IsOSPlatform(OSPlatform.Linux) && RuntimeInformation.OSDescription.Contains("Android");
