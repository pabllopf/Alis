95  public PingPongManager(Guid guid, WebSocket webSocket, TimeSpan keepAliveInterval,

144  private async Task PingForever()
145  {
146  Events.Log.PingPongManagerStarted(_guid, (int) _keepAliveInterval.TotalSeconds);
147 
148  try
149  {
150  while (!_cancellationToken.IsCancellationRequested)
151  {
152  await Task.Delay(_keepAliveInterval, _cancellationToken);
153 
154  if (_webSocket.State != WebSocketState.Open)
155  {
156  break;
157  }
158 
159  if (_pingSentTicks != 0)
160  {
161  Events.Log.KeepAliveIntervalExpired(_guid, (int) _keepAliveInterval.TotalSeconds);
162  await _webSocket.CloseAsync(WebSocketCloseStatus.NormalClosure,
163  $"No Pong message received in response to a Ping after KeepAliveInterval {_keepAliveInterval}",
164  _cancellationToken);
165  break;
166  }
167 
168  if (!_cancellationToken.IsCancellationRequested)
169  {
170  _pingSentTicks = _stopwatch.Elapsed.Ticks;
171  ArraySegment<byte> buffer = new ArraySegment<byte>(BitConverter.GetBytes(_pingSentTicks));
172  await SendPing(buffer, _cancellationToken);
173  }
174  }
175  }
176  catch (OperationCanceledException)
177  {
178  // normal, do nothing
179  }
180 
181  Events.Log.PingPongManagerEnded(_guid);
182  }
183 
189  private void WebSocketImpl_Pong(object sender, PongEventArgs e)
190  {
191  _pingSentTicks = 0;
192  OnPong(e);
193  }
