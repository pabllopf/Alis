132  VkPhysicalDeviceMemoryProperties memProperties,

140  Vulkan.VkBuffer dedicatedBuffer)
141  {
142  if (dedicated)
143  {
144  if (dedicatedImage != VkImage.Null && _getImageMemoryRequirements2 != null)
145  {
146  VkImageMemoryRequirementsInfo2KHR requirementsInfo = VkImageMemoryRequirementsInfo2KHR.New();
147  requirementsInfo.image = dedicatedImage;
148  VkMemoryRequirements2KHR requirements = VkMemoryRequirements2KHR.New();
149  _getImageMemoryRequirements2(_device, &requirementsInfo, &requirements);
150  size = requirements.memoryRequirements.size;
151  }
152  else if(dedicatedBuffer != Vulkan.VkBuffer.Null && _getBufferMemoryRequirements2 != null)
153  {

157  _getBufferMemoryRequirements2(_device, &requirementsInfo, &requirements);

159  }
160  }
161  else
162  {
163  // Round up to the nearest multiple of bufferImageGranularity.
164  size = ((size / _bufferImageGranularity) + 1) * _bufferImageGranularity;
165  }
166  _totalAllocatedBytes += size;
167 
168  lock (_lock)
169  {
170  if (!TryFindMemoryType(memProperties, memoryTypeBits, flags, out var memoryTypeIndex))
171  {
172  throw new VeldridException("No suitable memory type.");
173  }
174 
175  ulong minDedicatedAllocationSize = persistentMapped
176  ? MinDedicatedAllocationSizeDynamic
177  : MinDedicatedAllocationSizeNonDynamic;
178

182  allocateInfo.allocationSize = size;

195  if (allocationResult != VkResult.Success)

198  }
199

207  }
208  }

211  }

457  public bool Allocate(ulong size, ulong alignment, out VkMemoryBlock block)
458  {
459  checked
460  {
461  for (int i = 0; i < _freeBlocks.Count; i++)
462  {
463  VkMemoryBlock freeBlock = _freeBlocks[i];
464  ulong alignedBlockSize = freeBlock.Size;
465  if (freeBlock.Offset % alignment != 0)
466  {
467  ulong alignmentCorrection = (alignment - freeBlock.Offset % alignment);
468  if (alignedBlockSize <= alignmentCorrection)
469  {
470  continue;
471  }
472  alignedBlockSize -= alignmentCorrection;
473  }
474 
475  if (alignedBlockSize >= size) // Valid match -- split it and return.
476  {
477  _freeBlocks.RemoveAt(i);
478 
479  freeBlock.Size = alignedBlockSize;
480  if ((freeBlock.Offset % alignment) != 0)
481  {
482  freeBlock.Offset += alignment - (freeBlock.Offset % alignment);
483  }
484 
485  block = freeBlock;
486 
487  if (alignedBlockSize != size)
488  {
489  VkMemoryBlock splitBlock = new VkMemoryBlock(
490  freeBlock.DeviceMemory,
491  freeBlock.Offset + size,
492  freeBlock.Size - size,
493  _memoryTypeIndex,
494  freeBlock.BaseMappedPointer,
495  false);
496  _freeBlocks.Insert(i, splitBlock);
497  block = freeBlock;
498  block.Size = size;
499  }
500 
501 #if DEBUG
502  CheckAllocatedBlock(block);
503 #endif
504  _totalAllocatedBytes += alignedBlockSize;
505  return true;
506  }
507  }
508 
509  block = default(VkMemoryBlock);
510  return false;
511  }
512  }

543  private void MergeContiguousBlocks()
544  {
545  int contiguousLength = 1;
546  for (int i = 0; i < _freeBlocks.Count - 1; i++)
547  {
548  ulong blockStart = _freeBlocks[i].Offset;
549  while (i + contiguousLength < _freeBlocks.Count
550  && _freeBlocks[i + contiguousLength - 1].End == _freeBlocks[i + contiguousLength].Offset)
551  {
552  contiguousLength += 1;
553  }
554 
555  if (contiguousLength > 1)
556  {
557  ulong blockEnd = _freeBlocks[i + contiguousLength - 1].End;
558  _freeBlocks.RemoveRange(i, contiguousLength);
559  VkMemoryBlock mergedBlock = new VkMemoryBlock(
560  Memory,
561  blockStart,
562  blockEnd - blockStart,
563  _memoryTypeIndex,
564  _mappedPtr,
565  false);
566  _freeBlocks.Insert(i, mergedBlock);
567  contiguousLength = 0;
568  }
569  }

585  private bool BlocksOverlap(VkMemoryBlock first, VkMemoryBlock second)
586  {
587  ulong firstStart = first.Offset;
588  ulong firstEnd = first.Offset + first.Size;
589  ulong secondStart = second.Offset;
590  ulong secondEnd = second.Offset + second.Size;
591

595  || firstStart <= secondStart && firstEnd >= secondEnd);
