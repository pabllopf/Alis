134  private void PostSolve(Contact contact, ContactVelocityConstraint impulse)
135  {
136  if (!Broken)
137  {
138  if (Parts.Contains(contact.FixtureA) || Parts.Contains(contact.FixtureB))
139  {
140  float maxImpulse = 0.0f;
141  int count = contact.Manifold.PointCount;
142 
143  for (int i = 0; i < count; ++i)
144  {
145  maxImpulse = Math.Max(maxImpulse, impulse.Points[i].NormalImpulse);
146  }
147 
148  if (maxImpulse > Strength)
149  {
150  // Flag the body for breaking.
151  @break = true;
152  }
153  }
154  }
155  }

160  public void Update()
161  {
162  if (@break)
163  {
164  Decompose();
165  Broken = true;
166  @break = false;
167  }
168 
169  // Cache velocities to improve movement on breakage.
170  if (!Broken)
171  {
172  //Enlarge the cache if needed
173  if (Parts.Count > angularVelocitiesCache.Length)
174  {
175  velocitiesCache = new Vector2[Parts.Count];
176  angularVelocitiesCache = new float[Parts.Count];
177  }
178 
179  //Cache the linear and angular velocities.
180  for (int i = 0; i < Parts.Count; i++)
181  {
182  velocitiesCache[i] = Parts[i].Body.LinearVelocity;
183  angularVelocitiesCache[i] = Parts[i].Body.AngularVelocity;
184  }
185  }
186  }
187 
191  private void Decompose()
192  {
193  //Unsubsribe from the PostSolve delegate
194  world.ContactManager.PostSolve -= PostSolve;
195 
196  for (int i = 0; i < Parts.Count; i++)
197  {
198  Fixture oldFixture = Parts[i];
199 
200  Shape shape = oldFixture.Shape.Clone();
201  object userData = oldFixture.UserData;
202 
203  MainBody.RemoveFixture(oldFixture);
204 
205  Body body = BodyFactory.CreateBody(world, MainBody.Position, MainBody.Rotation, BodyType.Dynamic,
206  MainBody.UserData);
207 
208  Fixture newFixture = body.AddFixture(shape);
209  newFixture.UserData = userData;
210  Parts[i] = newFixture;
211 
212  body.AngularVelocity = angularVelocitiesCache[i];
213  body.LinearVelocity = velocitiesCache[i];
214  }
215 
216  world.RemoveBody(MainBody);
217  world.RemoveBreakableBody(this);
218  }

226  }
