93  public void Reset(TimeStep stepParam, int countParam, List<Contact> contactList, List<Position> positionList, List<Velocity> velocitiesList)
94  {
95  step = stepParam;
96  count = countParam;
97  positions = positionList;
98  velocities = velocitiesList;
99  contacts = contactList;
100 
101  // grow the array
102  if (VelocityConstraints == null || VelocityConstraints.Count < countParam)
103  {
104  VelocityConstraints = new List<ContactVelocityConstraint>(countParam * 2);
105  positionConstraints = new List<ContactPositionConstraint>(countParam * 2);
106 
107  for (int i = 0; i < VelocityConstraints.Count; i++)
108  {
109  VelocityConstraints[i] = new ContactVelocityConstraint();
110  }
111 
112  for (int i = 0; i < positionConstraints.Count; i++)
113  {
114  positionConstraints[i] = new ContactPositionConstraint();
115  }
116  }
117 
118  // Initialize position independent portions of the constraints.
119  for (int i = 0; i < count; ++i)
120  {
121  Contact contact = contactList[i];
122 
123  Fixture fixtureA = contact.FixtureA;
124  Fixture fixtureB = contact.FixtureB;
125  Shape shapeA = fixtureA.Shape;
126  Shape shapeB = fixtureB.Shape;
127  float radiusA = shapeA.RadiusPrivate;
128  float radiusB = shapeB.RadiusPrivate;
129  Body bodyA = fixtureA.Body;
130  Body bodyB = fixtureB.Body;
131  Manifold manifold = contact.Manifold;
132 
133  int pointCount = manifold.PointCount;
134  Debug.Assert(pointCount > 0);
135 
136  if (VelocityConstraints.Count <= i)
137  {
138  VelocityConstraints.Add(new ContactVelocityConstraint());
139  }
140 
141  ContactVelocityConstraint vc = VelocityConstraints[i];
142  vc.Friction = contact.Friction;
143  vc.Restitution = contact.Restitution;
144  vc.Threshold = contact.RestitutionThreshold;
145  vc.TangentSpeed = contact.TangentSpeed;
146  vc.IndexA = bodyA.IslandIndex;
147  vc.IndexB = bodyB.IslandIndex;
148  vc.InvMassA = bodyA.InvMass;
149  vc.InvMassB = bodyB.InvMass;
150  vc.InvIa = bodyA.InvI;
151  vc.InvIb = bodyB.InvI;
152  vc.ContactIndex = i;
153  vc.PointCount = pointCount;
154  vc.K.SetZero();

156 
157  if (positionConstraints.Count <= i)
158  {
159  positionConstraints.Add(new ContactPositionConstraint());
160  }
161 
162  ContactPositionConstraint pc = positionConstraints[i];
163  pc.IndexA = bodyA.IslandIndex;
164  pc.IndexB = bodyB.IslandIndex;
165  pc.InvMassA = bodyA.InvMass;
166  pc.InvMassB = bodyB.InvMass;

168  pc.LocalCenterB = bodyB.Sweep.LocalCenter;
169  pc.InvIa = bodyA.InvI;

172  pc.LocalPoint = manifold.LocalPoint;
173  pc.PointCount = pointCount;
174  pc.RadiusA = radiusA;
175  pc.RadiusB = radiusB;
176  pc.Type = manifold.Type;
177 
178  for (int j = 0; j < pointCount; ++j)
179  {
180  ManifoldPoint cp = manifold.Points[j];
181  VelocityConstraintPoint vcp = vc.Points[j];
182 
183  if (stepParam.WarmStarting)
184  {
185  vcp.NormalImpulse = step.DeltaTimeRatio * cp.NormalImpulse;

188  else

190  vcp.NormalImpulse = 0.0f;
191  vcp.TangentImpulse = 0.0f;
192  }

195  vcp.Rb = Vector2F.Zero;
196  vcp.NormalMass = 0.0f;
197  vcp.TangentMass = 0.0f;
198  vcp.VelocityBias = 0.0f;

206  public void InitializeVelocityConstraints()
207  {
208  for (int i = 0; i < count; ++i)
209  {
210  ContactVelocityConstraint vc = VelocityConstraints[i];
211  ContactPositionConstraint pc = positionConstraints[i];
212 
213  float radiusA = pc.RadiusA;
214  float radiusB = pc.RadiusB;
215  Manifold manifold = contacts[vc.ContactIndex].Manifold;
216 
217  int indexA = vc.IndexA;
218  int indexB = vc.IndexB;
219 
220  float mA = vc.InvMassA;
221  float mB = vc.InvMassB;
222  float iA = vc.InvIa;
223  float iB = vc.InvIb;
224  Vector2F localCenterA = pc.LocalCenterA;
225  Vector2F localCenterB = pc.LocalCenterB;
226 
227  Vector2F cA = positions[indexA].C;
228  float aA = positions[indexA].A;
229  Vector2F vA = velocities[indexA].V;
230  float wA = velocities[indexA].W;
231 
232  Vector2F cB = positions[indexB].C;
233  float aB = positions[indexB].A;
234  Vector2F vB = velocities[indexB].V;
235  float wB = velocities[indexB].W;
236 
237  Debug.Assert(manifold.PointCount > 0);
238 
239  Transform xfA = new Transform();
240  Transform xfB = new Transform();
241  xfA.Rotation.Set(aA);
242  xfB.Rotation.Set(aB);
243  xfA.Position = cA - MathUtils.Mul(xfA.Rotation, localCenterA);
244  xfB.Position = cB - MathUtils.Mul(xfB.Rotation, localCenterB);
245 
246  WorldManifold.Initialize(ref manifold, ref xfA, radiusA, ref xfB, radiusB, out Vector2F normal,
247  out FixedArray2<Vector2F> points, out _);
248 
249  vc.Normal = normal;
250 
251  int pointCount = vc.PointCount;
252  for (int j = 0; j < pointCount; ++j)
253  {
254  VelocityConstraintPoint vcp = vc.Points[j];
255 
256  vcp.Ra = points[j] - cA;
257  vcp.Rb = points[j] - cB;
258 
259  float rnA = MathUtils.Cross(vcp.Ra, vc.Normal);
260  float rnB = MathUtils.Cross(vcp.Rb, vc.Normal);
261 
262  float kNormal = mA + mB + iA * rnA * rnA + iB * rnB * rnB;
263 
264  vcp.NormalMass = kNormal > 0.0f ? 1.0f / kNormal : 0.0f;
265 
266  Vector2F tangent = MathUtils.Cross(vc.Normal, 1.0f);
267 
268  float rtA = MathUtils.Cross(vcp.Ra, tangent);
269  float rtB = MathUtils.Cross(vcp.Rb, tangent);
270 
271  float kTangent = mA + mB + iA * rtA * rtA + iB * rtB * rtB;
272 
273  vcp.TangentMass = kTangent > 0.0f ? 1.0f / kTangent : 0.0f;
274 
275  // Setup a velocity bias for restitution.
276  vcp.VelocityBias = 0.0f;
277  float vRel = MathUtils.Dot(vc.Normal,
278  vB + MathUtils.Cross(wB, vcp.Rb) - vA - MathUtils.Cross(wA, vcp.Ra));
279  if (vRel < -vc.Threshold)
280  {
281  vcp.VelocityBias = -vc.Restitution * vRel;
282  }
283  }
284 
285  // If we have two points, then prepare the block solver.
286  if ((vc.PointCount == 2) && Settings.BlockSolve)
287  {
288  VelocityConstraintPoint vcp1 = vc.Points[0];
289  VelocityConstraintPoint vcp2 = vc.Points[1];
290 
291  float rn1A = MathUtils.Cross(vcp1.Ra, vc.Normal);
292  float rn1B = MathUtils.Cross(vcp1.Rb, vc.Normal);
293  float rn2A = MathUtils.Cross(vcp2.Ra, vc.Normal);
294  float rn2B = MathUtils.Cross(vcp2.Rb, vc.Normal);
295 
296  float k11 = mA + mB + iA * rn1A * rn1A + iB * rn1B * rn1B;
297  float k22 = mA + mB + iA * rn2A * rn2A + iB * rn2B * rn2B;
298  float k12 = mA + mB + iA * rn1A * rn2A + iB * rn1B * rn2B;
299 
300  // Ensure a reasonable condition number.
301  const float kMaxConditionNumber = 1000.0f;
302  if (k11 * k11 < kMaxConditionNumber * (k11 * k22 - k12 * k12))
303  {
304  // K is safe to invert.
305  vc.K.Ex = new Vector2F(k11, k12);
306  vc.K.Ey = new Vector2F(k12, k22);
307  vc.NormalMass = vc.K.Inverse;
308  }
309  else
310  {
311  vc.PointCount = 1;
312  }
313  }
314  }
315  }

320  public void WarmStart()
321  {
322  // Warm start.
323  for (int i = 0; i < count; ++i)
324  {
325  ContactVelocityConstraint vc = VelocityConstraints[i];
326 
327  int indexA = vc.IndexA;
328  int indexB = vc.IndexB;
329  float mA = vc.InvMassA;
330  float iA = vc.InvIa;
331  float mB = vc.InvMassB;
332  float iB = vc.InvIb;
333  int pointCount = vc.PointCount;
334 
335  Vector2F vA = velocities[indexA].V;
336  float wA = velocities[indexA].W;
337  Vector2F vB = velocities[indexB].V;
338  float wB = velocities[indexB].W;
339 
340  Vector2F normal = vc.Normal;
341  Vector2F tangent = MathUtils.Cross(normal, 1.0f);
342 
343  for (int j = 0; j < pointCount; ++j)
344  {
345  VelocityConstraintPoint vcp = vc.Points[j];
346  Vector2F p = vcp.NormalImpulse * normal + vcp.TangentImpulse * tangent;
347  wA -= iA * MathUtils.Cross(vcp.Ra, p);
348  vA -= mA * p;
349  wB += iB * MathUtils.Cross(vcp.Rb, p);
350  vB += mB * p;
351  }
352 
353  velocities[indexA].V = vA;
354  velocities[indexA].W = wA;
355  velocities[indexB].V = vB;
356  velocities[indexB].W = wB;
357  }

363  public void SolveVelocityConstraints()
364  {
365  for (int i = 0; i < count; ++i)
366  {
367  ContactVelocityConstraint vc = VelocityConstraints[i];
368 
369  int indexA = vc.IndexA;
370  int indexB = vc.IndexB;
371  float mA = vc.InvMassA;
372  float iA = vc.InvIa;
373  float mB = vc.InvMassB;
374  float iB = vc.InvIb;
375  int pointCount = vc.PointCount;
376 
377  Vector2F vA = velocities[indexA].V;
378  float wA = velocities[indexA].W;
379  Vector2F vB = velocities[indexB].V;
380  float wB = velocities[indexB].W;
381 
382  Vector2F normal = vc.Normal;
383  Vector2F tangent = MathUtils.Cross(normal, 1.0f);
384  float friction = vc.Friction;
385 
386  Debug.Assert(pointCount == 1 || pointCount == 2);
387 
388  // Solve tangent constraints first because non-penetration is more important
389  // than friction.
390  for (int j = 0; j < pointCount; ++j)
391  {
392  VelocityConstraintPoint vcp = vc.Points[j];
393 
394  // Relative velocity at contact
395  Vector2F dv = vB + MathUtils.Cross(wB, vcp.Rb) - vA - MathUtils.Cross(wA, vcp.Ra);
396 
397  // Compute tangent force
398  float vt = Vector2F.Dot(dv, tangent) - vc.TangentSpeed;
399  float lambda = vcp.TangentMass * -vt;
400 
401  // b2Clamp the accumulated force
402  float maxFriction = friction * vcp.NormalImpulse;
403  float newImpulse = MathUtils.Clamp(vcp.TangentImpulse + lambda, -maxFriction, maxFriction);
404  lambda = newImpulse - vcp.TangentImpulse;
405  vcp.TangentImpulse = newImpulse;
406 
407  // Apply contact impulse
408  Vector2F p = lambda * tangent;
409 
410  vA -= mA * p;
411  wA -= iA * MathUtils.Cross(vcp.Ra, p);
412 
413  vB += mB * p;
414  wB += iB * MathUtils.Cross(vcp.Rb, p);
415  }
416 
417  // Solve normal constraints
418  if (pointCount == 1 || !Settings.BlockSolve)
419  {
420  for (int j = 0; j < pointCount; ++j)
421  {
422  VelocityConstraintPoint vcp = vc.Points[j];
423 
424  // Relative velocity at contact
425  Vector2F dv = vB + MathUtils.Cross(wB, vcp.Rb) - vA - MathUtils.Cross(wA, vcp.Ra);
426 
427  // Compute normal impulse
428  float vn = Vector2F.Dot(dv, normal);
429  float lambda = -vcp.NormalMass * (vn - vcp.VelocityBias);
430 
431  // b2Clamp the accumulated impulse
432  float newImpulse = Math.Max(vcp.NormalImpulse + lambda, 0.0f);
433  lambda = newImpulse - vcp.NormalImpulse;
434  vcp.NormalImpulse = newImpulse;
435 
436  // Apply contact impulse
437  Vector2F p = lambda * normal;
438  vA -= mA * p;
439  wA -= iA * MathUtils.Cross(vcp.Ra, p);
440 
441  vB += mB * p;
442  wB += iB * MathUtils.Cross(vcp.Rb, p);
443  }
444  }
445  else
446  {
447  // Block solver developed in collaboration with Dirk Gregoris (back in 01/07 on Box2D_Lite).
448  // Build the mini LCP for this contact patch
449  //
450  // vn = A * x + b, vn >= 0, x >= 0 and vn_i * x_i = 0 with i = 1..2
451  //
452  // A = J * W * JT and J = ( -n, -r1 x n, n, r2 x n )
453  // b = vn0 - velocityBias
454  //
455  // The system is solved using the "Total enumeration method" (s. Murty). The complementary constraint vn_i * x_i
456  // implies that we must have in any solution either vn_i = 0 or x_i = 0. So for the 2D contact problem the cases
457  // vn1 = 0 and vn2 = 0, x1 = 0 and x2 = 0, x1 = 0 and vn2 = 0, x2 = 0 and vn1 = 0 need to be tested. The first valid
458  // solution that satisfies the problem is chosen.
459  //
460  // In order to account of the accumulated impulse 'a' (because of the iterative nature of the solver which only requires
461  // that the accumulated impulse is clamped and not the incremental impulse) we change the impulse variable (x_i).
462  //
463  // Substitute:
464  //
465  // x = a + d
466  //
467  // a := old total impulse
468  // x := new total impulse
469  // d := incremental impulse
470  //
471  // For the current iteration we extend the formula for the incremental impulse
472  // to compute the new total impulse:
473  //
474  // vn = A * d + b
475  // = A * (x - a) + b
476  // = A * x + b - A * a
477  // = A * x + b'
478  // b' = b - A * a;
479 
480  VelocityConstraintPoint cp1 = vc.Points[0];
481  VelocityConstraintPoint cp2 = vc.Points[1];
482 
483  Vector2F a = new Vector2F(cp1.NormalImpulse, cp2.NormalImpulse);
484  Debug.Assert((a.X >= 0.0f) && (a.Y >= 0.0f));
485 
486  // Relative velocity at contact
487  Vector2F dv1 = vB + MathUtils.Cross(wB, cp1.Rb) - vA - MathUtils.Cross(wA, cp1.Ra);
488  Vector2F dv2 = vB + MathUtils.Cross(wB, cp2.Rb) - vA - MathUtils.Cross(wA, cp2.Ra);
489 
490  // Compute normal velocity
491  float vn1 = Vector2F.Dot(dv1, normal);
492  float vn2 = Vector2F.Dot(dv2, normal);
493 
494  Vector2F b = new Vector2F(
495  vn1 - cp1.VelocityBias,
496  vn2 - cp2.VelocityBias
497  );
498 
499  // Compute b'
500  b -= MathUtils.Mul(ref vc.K, a);
501 
502  //
503  // Case 1: vn = 0
504  //
505  // 0 = A * x + b'
506  //
507  // Solve for x:
508  //
509  // x = - inv(A) * b'
510  //
511  Vector2F x = -MathUtils.Mul(ref vc.NormalMass, b);
512 
513  if ((x.X >= 0.0f) && (x.Y >= 0.0f))
514  {
515  cp1.NormalImpulse = x.X;
516  cp2.NormalImpulse = x.Y;
517 
518  break;
519  }
520 
521  x = new Vector2F(-cp1.NormalMass * b.X, 0.0f);
522  vn2 = vc.K.Ex.Y * x.X + b.Y;
523 
524  if ((x.X >= 0.0f) && (vn2 >= 0.0f))
525  {
526  cp1.NormalImpulse = x.X;
527  cp2.NormalImpulse = x.Y;
528 
529  break;
530  }
531 
532  x = new Vector2F(0.0f, -cp2.NormalMass * b.Y);
533  vn1 = vc.K.Ey.X * x.Y + b.X;
534 
535  if ((x.Y >= 0.0f) && (vn1 >= 0.0f))
536  {
537  cp1.NormalImpulse = x.X;
538  cp2.NormalImpulse = x.Y;
539 
540  break;
541  }
542 
543  x = Vector2F.Zero;
544  vn1 = b.X;
545  vn2 = b.Y;
546 
547  if ((vn1 >= 0.0f) && (vn2 >= 0.0f))
548  {
549  Vector2F d = x - a;
550 
551  Vector2F p1 = d.X * normal;
552  Vector2F p2 = d.Y * normal;
553  vA -= mA * (p1 + p2);
554  wA -= iA * (MathUtils.Cross(cp1.Ra, p1) + MathUtils.Cross(cp2.Ra, p2));
555 
556  vB += mB * (p1 + p2);
557  wB += iB * (MathUtils.Cross(cp1.Rb, p1) + MathUtils.Cross(cp2.Rb, p2));
558 
559  cp1.NormalImpulse = x.X;
560  cp2.NormalImpulse = x.Y;
561  }
562  }
563 
564  velocities[indexA].V = vA;
565  velocities[indexA].W = wA;
566  velocities[indexB].V = vB;
567  velocities[indexB].W = wB;
568  }
569  }

597  public bool SolvePositionConstraints()
598  {
599  float minSeparation = 0.0f;
600 
601  for (int i = 0; i < count; ++i)
602  {
603  ContactPositionConstraint pc = positionConstraints[i];
604 
605  int indexA = pc.IndexA;
606  int indexB = pc.IndexB;
607  Vector2F localCenterA = pc.LocalCenterA;
608  float mA = pc.InvMassA;
609  float iA = pc.InvIa;
610  Vector2F localCenterB = pc.LocalCenterB;
611  float mB = pc.InvMassB;
612  float iB = pc.InvIb;
613  int pointCount = pc.PointCount;
614 
615  Vector2F cA = positions[indexA].C;
616  float aA = positions[indexA].A;
617 
618  Vector2F cB = positions[indexB].C;
619  float aB = positions[indexB].A;
620 
621  for (int j = 0; j < pointCount; ++j)
622  {
623  Transform xfA = new Transform();
624  Transform xfB = new Transform();
625  xfA.Rotation.Set(aA);
626  xfB.Rotation.Set(aB);
627  xfA.Position = cA - MathUtils.Mul(xfA.Rotation, localCenterA);
628  xfB.Position = cB - MathUtils.Mul(xfB.Rotation, localCenterB);
629 
630  PositionSolverManifold.Initialize(pc, ref xfA, ref xfB, j, out Vector2F normal, out Vector2F point,
631  out float separation);
632 
633  Vector2F rA = point - cA;
634  Vector2F rB = point - cB;
635 
636  minSeparation = Math.Min(minSeparation, separation);
637 
638  float c = MathUtils.Clamp(Settings.Baumgarte * (separation + Settings.LinearSlop),
639  -Settings.LinearCorrection, 0.0f);
640 
641  float rnA = MathUtils.Cross(rA, normal);
642  float rnB = MathUtils.Cross(rB, normal);
643  float k = mA + mB + iA * rnA * rnA + iB * rnB * rnB;
644 
645  // Compute normal impulse
646  float impulse = k > 0.0f ? -c / k : 0.0f;
647 
648  Vector2F p = impulse * normal;
649 
650  cA -= mA * p;
651  aA -= iA * MathUtils.Cross(rA, p);
652 
653  cB += mB * p;
654  aB += iB * MathUtils.Cross(rB, p);
655  }
656 
657  positions[indexA].C = cA;
658  positions[indexA].A = aA;
659 
660  positions[indexB].C = cB;
661  positions[indexB].A = aB;
662  }
663 
664  // We can't expect min separation >= -b2_linearSlop because we don't
665  // push the separation above -b2_linearSlop.
666  return minSeparation >= -3.0f * Settings.LinearSlop;
667  }

669  // Sequential position solver for position constraints.
676  public bool SolveToiPositionConstraints(int toiIndexA, int toiIndexB)
677  {
678  float minSeparation = 0.0f;
679 
680  for (int i = 0; i < count; ++i)
681  {
682  ContactPositionConstraint pc = positionConstraints[i];
683 
684  int indexA = pc.IndexA;
685  int indexB = pc.IndexB;
686  Vector2F localCenterA = pc.LocalCenterA;
687  Vector2F localCenterB = pc.LocalCenterB;
688  int pointCount = pc.PointCount;
689 
690  float mA = 0.0f;
691  float iA = 0.0f;
692  if (indexA == toiIndexA || indexA == toiIndexB)
693  {
694  mA = pc.InvMassA;
695  iA = pc.InvIa;
696  }
697 
698  float mB = 0.0f;
699  float iB = 0.0f;
700  if (indexB == toiIndexA || indexB == toiIndexB)
701  {
702  mB = pc.InvMassB;
703  iB = pc.InvIb;
704  }
705 
706  Vector2F cA = positions[indexA].C;
707  float aA = positions[indexA].A;
708 
709  Vector2F cB = positions[indexB].C;
710  float aB = positions[indexB].A;
711 
712  // Solve normal constraints
713  for (int j = 0; j < pointCount; ++j)
714  {
715  Transform xfA = new Transform();
716  Transform xfB = new Transform();
717  xfA.Rotation.Set(aA);
718  xfB.Rotation.Set(aB);
719  xfA.Position = cA - MathUtils.Mul(xfA.Rotation, localCenterA);
720  xfB.Position = cB - MathUtils.Mul(xfB.Rotation, localCenterB);
721 
722  PositionSolverManifold.Initialize(pc, ref xfA, ref xfB, j, out Vector2F normal, out Vector2F point,
723  out float separation);
724 
725  Vector2F rA = point - cA;
726  Vector2F rB = point - cB;
727 
728  // Track max constraint error.
729  minSeparation = Math.Min(minSeparation, separation);
730 
731  // Prevent large corrections and allow slop.
732  float c = MathUtils.Clamp(Settings.ToiBaumgarte * (separation + Settings.LinearSlop),
733  -Settings.LinearCorrection, 0.0f);
734 
735  // Compute the effective mass.
736  float rnA = MathUtils.Cross(rA, normal);
737  float rnB = MathUtils.Cross(rB, normal);
738  float k = mA + mB + iA * rnA * rnA + iB * rnB * rnB;
739 
740  // Compute normal impulse
741  float impulse = k > 0.0f ? -c / k : 0.0f;
742 
743  Vector2F p = impulse * normal;
744 
745  cA -= mA * p;
746  aA -= iA * MathUtils.Cross(rA, p);
747 
748  cB += mB * p;
749  aB += iB * MathUtils.Cross(rB, p);
750  }
751 
752  positions[indexA].C = cA;
753  positions[indexA].A = aA;
754 
755  positions[indexB].C = cB;
756  positions[indexB].A = aB;
757  }
758 
759  // We can't expect min separation >= -b2_linearSlop because we don't
760  // push the separation above -b2_linearSlop.
761  return minSeparation >= -1.5f * Settings.LinearSlop;
762  }
763  }
764 }
