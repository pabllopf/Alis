115  public void Reset(TimeStep step, int count, Contact[] contacts, Position[] positions, Velocity[] velocities)
116  {
117  this.step = step;
118  this.count = count;
119  this.positions = positions;
120  this.velocities = velocities;
121  this.contacts = contacts;
122 
123  // grow the array
124  if (VelocityConstraints == null || VelocityConstraints.Length < count)
125  {
126  VelocityConstraints = new ContactVelocityConstraint[count * 2];
127  positionConstraints = new ContactPositionConstraint[count * 2];
128 
129  for (int i = 0; i < VelocityConstraints.Length; i++)
130  {
131  VelocityConstraints[i] = new ContactVelocityConstraint();
132  }
133 
134  for (int i = 0; i < positionConstraints.Length; i++)
135  {
136  positionConstraints[i] = new ContactPositionConstraint();
137  }
138  }
139 
140  // Initialize position independent portions of the constraints.
141  for (int i = 0; i < this.count; ++i)
142  {
143  Contact contact = contacts[i];
144 
145  Fixture fixtureA = contact.FixtureA;
146  Fixture fixtureB = contact.FixtureB;
147  Shape shapeA = fixtureA.Shape;
148  Shape shapeB = fixtureB.Shape;
149  float radiusA = shapeA.RadiusPrivate;
150  float radiusB = shapeB.RadiusPrivate;
151  Body bodyA = fixtureA.Body;
152  Body bodyB = fixtureB.Body;
153  Manifold manifold = contact.Manifold;
154 
155  int pointCount = manifold.PointCount;
156  Debug.Assert(pointCount > 0);
157 
158  ContactVelocityConstraint vc = VelocityConstraints[i];
159  vc.Friction = contact.Friction;
160  vc.Restitution = contact.Restitution;
161  vc.Threshold = contact.RestitutionThreshold;
162  vc.TangentSpeed = contact.TangentSpeed;
163  vc.IndexA = bodyA.IslandIndex;
164  vc.IndexB = bodyB.IslandIndex;
165  vc.InvMassA = bodyA.InvMass;
166  vc.InvMassB = bodyB.InvMass;
167  vc.InvIa = bodyA.InvI;
168  vc.InvIb = bodyB.InvI;
169  vc.ContactIndex = i;
170  vc.PointCount = pointCount;
171  vc.K.SetZero();
172  vc.NormalMass.SetZero();
173 
174  ContactPositionConstraint pc = positionConstraints[i];
175  pc.IndexA = bodyA.IslandIndex;
176  pc.IndexB = bodyB.IslandIndex;
177  pc.InvMassA = bodyA.InvMass;
178  pc.InvMassB = bodyB.InvMass;
179  pc.LocalCenterA = bodyA.Sweep.LocalCenter;
180  pc.LocalCenterB = bodyB.Sweep.LocalCenter;
181  pc.InvIa = bodyA.InvI;
182  pc.InvIb = bodyB.InvI;
183  pc.LocalNormal = manifold.LocalNormal;
184  pc.LocalPoint = manifold.LocalPoint;
185  pc.PointCount = pointCount;
186  pc.RadiusA = radiusA;
187  pc.RadiusB = radiusB;
188  pc.Type = manifold.Type;
189 
190  for (int j = 0; j < pointCount; ++j)
191  {
192  ManifoldPoint cp = manifold.Points[j];
193  VelocityConstraintPoint vcp = vc.Points[j];
194 
195  if (step.WarmStarting)
196  {
197  vcp.NormalImpulse = this.step.DeltaTimeRatio * cp.NormalImpulse;
198  vcp.TangentImpulse = this.step.DeltaTimeRatio * cp.TangentImpulse;
199  }
200  else
201  {
202  vcp.NormalImpulse = 0.0f;
203  vcp.TangentImpulse = 0.0f;
204  }
205 
206  vcp.RA = Vector2.Zero;
207  vcp.RB = Vector2.Zero;
208  vcp.NormalMass = 0.0f;
209  vcp.TangentMass = 0.0f;
210  vcp.VelocityBias = 0.0f;
211 
212  pc.LocalPoints[j] = cp.LocalPoint;
213  }
214  }
215  }

218  public void InitializeVelocityConstraints()
219  {
220  for (int i = 0; i < count; ++i)
221  {
222  ContactVelocityConstraint vc = VelocityConstraints[i];
223  ContactPositionConstraint pc = positionConstraints[i];
224 
225  float radiusA = pc.RadiusA;
226  float radiusB = pc.RadiusB;
227  Manifold manifold = contacts[vc.ContactIndex].Manifold;
228 
229  int indexA = vc.IndexA;
230  int indexB = vc.IndexB;
231 
232  float mA = vc.InvMassA;
233  float mB = vc.InvMassB;
234  float iA = vc.InvIa;
235  float iB = vc.InvIb;
236  Vector2 localCenterA = pc.LocalCenterA;
237  Vector2 localCenterB = pc.LocalCenterB;
238 
239  Vector2 cA = positions[indexA].C;
240  float aA = positions[indexA].A;
241  Vector2 vA = velocities[indexA].V;
242  float wA = velocities[indexA].W;
243 
244  Vector2 cB = positions[indexB].C;
245  float aB = positions[indexB].A;
246  Vector2 vB = velocities[indexB].V;
247  float wB = velocities[indexB].W;
248 
249  Debug.Assert(manifold.PointCount > 0);
250 
251  Transform xfA = new Transform();
252  Transform xfB = new Transform();
253  xfA.Q.Set(aA);
254  xfB.Q.Set(aB);
255  xfA.P = cA - MathUtils.Mul(xfA.Q, localCenterA);
256  xfB.P = cB - MathUtils.Mul(xfB.Q, localCenterB);
257 
258  WorldManifold.Initialize(ref manifold, ref xfA, radiusA, ref xfB, radiusB, out Vector2 normal,
259  out FixedArray2<Vector2> points, out _);
260 
261  vc.Normal = normal;
262 
263  int pointCount = vc.PointCount;
264  for (int j = 0; j < pointCount; ++j)
265  {
266  VelocityConstraintPoint vcp = vc.Points[j];
267 
268  vcp.RA = points[j] - cA;
269  vcp.RB = points[j] - cB;
270 
271  float rnA = MathUtils.Cross(vcp.RA, vc.Normal);
272  float rnB = MathUtils.Cross(vcp.RB, vc.Normal);
273 
274  float kNormal = mA + mB + iA * rnA * rnA + iB * rnB * rnB;
275 
276  vcp.NormalMass = kNormal > 0.0f ? 1.0f / kNormal : 0.0f;
277 
278  Vector2 tangent = MathUtils.Cross(vc.Normal, 1.0f);
279 
280  float rtA = MathUtils.Cross(vcp.RA, tangent);
281  float rtB = MathUtils.Cross(vcp.RB, tangent);
282 
283  float kTangent = mA + mB + iA * rtA * rtA + iB * rtB * rtB;
284 
285  vcp.TangentMass = kTangent > 0.0f ? 1.0f / kTangent : 0.0f;
286 
287  // Setup a velocity bias for restitution.
288  vcp.VelocityBias = 0.0f;
289  float vRel = MathUtils.Dot(vc.Normal,
290  vB + MathUtils.Cross(wB, vcp.RB) - vA - MathUtils.Cross(wA, vcp.RA));
291  if (vRel < -vc.Threshold)
292  {
293  vcp.VelocityBias = -vc.Restitution * vRel;
294  }
295  }
296 
297  // If we have two points, then prepare the block solver.
298  if (vc.PointCount == 2 && Settings.BlockSolve)
299  {
300  VelocityConstraintPoint vcp1 = vc.Points[0];
301  VelocityConstraintPoint vcp2 = vc.Points[1];
302 
303  float rn1A = MathUtils.Cross(vcp1.RA, vc.Normal);
304  float rn1B = MathUtils.Cross(vcp1.RB, vc.Normal);
305  float rn2A = MathUtils.Cross(vcp2.RA, vc.Normal);
306  float rn2B = MathUtils.Cross(vcp2.RB, vc.Normal);
307 
308  float k11 = mA + mB + iA * rn1A * rn1A + iB * rn1B * rn1B;
309  float k22 = mA + mB + iA * rn2A * rn2A + iB * rn2B * rn2B;
310  float k12 = mA + mB + iA * rn1A * rn2A + iB * rn1B * rn2B;
311 
312  // Ensure a reasonable condition number.
313  const float kMaxConditionNumber = 1000.0f;
314  if (k11 * k11 < kMaxConditionNumber * (k11 * k22 - k12 * k12))
315  {
316  // K is safe to invert.
317  vc.K.Ex = new Vector2(k11, k12);
318  vc.K.Ey = new Vector2(k12, k22);
319  vc.NormalMass = vc.K.Inverse;
320  }
321  else
322  {
323  // The constraints are redundant, just use one.
324  // TODO_ERIN use deepest?
325  vc.PointCount = 1;
326  }
327  }
328  }
329  }

334  public void WarmStart()
335  {
336  // Warm start.
337  for (int i = 0; i < count; ++i)
338  {
339  ContactVelocityConstraint vc = VelocityConstraints[i];
340 
341  int indexA = vc.IndexA;
342  int indexB = vc.IndexB;
343  float mA = vc.InvMassA;
344  float iA = vc.InvIa;
345  float mB = vc.InvMassB;
346  float iB = vc.InvIb;
347  int pointCount = vc.PointCount;
348 
349  Vector2 vA = velocities[indexA].V;
350  float wA = velocities[indexA].W;
351  Vector2 vB = velocities[indexB].V;
352  float wB = velocities[indexB].W;
353 
354  Vector2 normal = vc.Normal;
355  Vector2 tangent = MathUtils.Cross(normal, 1.0f);
356 
357  for (int j = 0; j < pointCount; ++j)
358  {
359  VelocityConstraintPoint vcp = vc.Points[j];
360  Vector2 p = vcp.NormalImpulse * normal + vcp.TangentImpulse * tangent;
361  wA -= iA * MathUtils.Cross(vcp.RA, p);
362  vA -= mA * p;
363  wB += iB * MathUtils.Cross(vcp.RB, p);
364  vB += mB * p;
365  }
366 
367  velocities[indexA].V = vA;
368  velocities[indexA].W = wA;
369  velocities[indexB].V = vB;
370  velocities[indexB].W = wB;
371  }
372  }

377  public void SolveVelocityConstraints()
378  {
379  for (int i = 0; i < count; ++i)
380  {
381  ContactVelocityConstraint vc = VelocityConstraints[i];
382 
383  int indexA = vc.IndexA;
384  int indexB = vc.IndexB;
385  float mA = vc.InvMassA;
386  float iA = vc.InvIa;
387  float mB = vc.InvMassB;
388  float iB = vc.InvIb;
389  int pointCount = vc.PointCount;
390 
391  Vector2 vA = velocities[indexA].V;
392  float wA = velocities[indexA].W;
393  Vector2 vB = velocities[indexB].V;
394  float wB = velocities[indexB].W;
395 
396  Vector2 normal = vc.Normal;
397  Vector2 tangent = MathUtils.Cross(normal, 1.0f);
398  float friction = vc.Friction;
399 
400  Debug.Assert(pointCount == 1 || pointCount == 2);
401 
402  // Solve tangent constraints first because non-penetration is more important
403  // than friction.
404  for (int j = 0; j < pointCount; ++j)
405  {
406  VelocityConstraintPoint vcp = vc.Points[j];
407 
408  // Relative velocity at contact
409  Vector2 dv = vB + MathUtils.Cross(wB, vcp.RB) - vA - MathUtils.Cross(wA, vcp.RA);
410 
411  // Compute tangent force
412  float vt = Vector2.Dot(dv, tangent) - vc.TangentSpeed;
413  float lambda = vcp.TangentMass * -vt;
414 
415  // b2Clamp the accumulated force
416  float maxFriction = friction * vcp.NormalImpulse;
417  float newImpulse = MathUtils.Clamp(vcp.TangentImpulse + lambda, -maxFriction, maxFriction);
418  lambda = newImpulse - vcp.TangentImpulse;
419  vcp.TangentImpulse = newImpulse;
420 
421  // Apply contact impulse
422  Vector2 p = lambda * tangent;
423 
424  vA -= mA * p;
425  wA -= iA * MathUtils.Cross(vcp.RA, p);
426 
427  vB += mB * p;
428  wB += iB * MathUtils.Cross(vcp.RB, p);
429  }
430 
431  // Solve normal constraints
432  if (pointCount == 1 || !Settings.BlockSolve)
433  {
434  for (int j = 0; j < pointCount; ++j)
435  {
436  VelocityConstraintPoint vcp = vc.Points[j];
437 
438  // Relative velocity at contact
439  Vector2 dv = vB + MathUtils.Cross(wB, vcp.RB) - vA - MathUtils.Cross(wA, vcp.RA);
440 
441  // Compute normal impulse
442  float vn = Vector2.Dot(dv, normal);
443  float lambda = -vcp.NormalMass * (vn - vcp.VelocityBias);
444 
445  // b2Clamp the accumulated impulse
446  float newImpulse = Math.Max(vcp.NormalImpulse + lambda, 0.0f);
447  lambda = newImpulse - vcp.NormalImpulse;
448  vcp.NormalImpulse = newImpulse;
449 
450  // Apply contact impulse
451  Vector2 p = lambda * normal;
452  vA -= mA * p;
453  wA -= iA * MathUtils.Cross(vcp.RA, p);
454 
455  vB += mB * p;
456  wB += iB * MathUtils.Cross(vcp.RB, p);
457  }
458  }
459  else
460  {
461  // Block solver developed in collaboration with Dirk Gregorius (back in 01/07 on Box2D_Lite).
462  // Build the mini LCP for this contact patch
463  //
464  // vn = A * x + b, vn >= 0, x >= 0 and vn_i * x_i = 0 with i = 1..2
465  //
466  // A = J * W * JT and J = ( -n, -r1 x n, n, r2 x n )
467  // b = vn0 - velocityBias
468  //
469  // The system is solved using the "Total enumeration method" (s. Murty). The complementary constraint vn_i * x_i
470  // implies that we must have in any solution either vn_i = 0 or x_i = 0. So for the 2D contact problem the cases
471  // vn1 = 0 and vn2 = 0, x1 = 0 and x2 = 0, x1 = 0 and vn2 = 0, x2 = 0 and vn1 = 0 need to be tested. The first valid
472  // solution that satisfies the problem is chosen.
473  //
474  // In order to account of the accumulated impulse 'a' (because of the iterative nature of the solver which only requires
475  // that the accumulated impulse is clamped and not the incremental impulse) we change the impulse variable (x_i).
476  //
477  // Substitute:
478  //
479  // x = a + d
480  //
481  // a := old total impulse
482  // x := new total impulse
483  // d := incremental impulse
484  //
485  // For the current iteration we extend the formula for the incremental impulse
486  // to compute the new total impulse:
487  //
488  // vn = A * d + b
489  // = A * (x - a) + b
490  // = A * x + b - A * a
491  // = A * x + b'
492  // b' = b - A * a;
493 
494  VelocityConstraintPoint cp1 = vc.Points[0];
495  VelocityConstraintPoint cp2 = vc.Points[1];
496 
497  Vector2 a = new Vector2(cp1.NormalImpulse, cp2.NormalImpulse);
498  Debug.Assert(a.X >= 0.0f && a.Y >= 0.0f);
499 
500  // Relative velocity at contact
501  Vector2 dv1 = vB + MathUtils.Cross(wB, cp1.RB) - vA - MathUtils.Cross(wA, cp1.RA);
502  Vector2 dv2 = vB + MathUtils.Cross(wB, cp2.RB) - vA - MathUtils.Cross(wA, cp2.RA);
503 
504  // Compute normal velocity
505  float vn1 = Vector2.Dot(dv1, normal);
506  float vn2 = Vector2.Dot(dv2, normal);
507 
508  Vector2 b = Vector2.Zero;
509  b.X = vn1 - cp1.VelocityBias;
510  b.Y = vn2 - cp2.VelocityBias;
511 
512  // Compute b'
513  b -= MathUtils.Mul(ref vc.K, a);
514 
515  for (;;)
516  {
517  //
518  // Case 1: vn = 0
519  //
520  // 0 = A * x + b'
521  //
522  // Solve for x:
523  //
524  // x = - inv(A) * b'
525  //
526  Vector2 x = -MathUtils.Mul(ref vc.NormalMass, b);
527 
528  if (x.X >= 0.0f && x.Y >= 0.0f)
529  {
530  // Get the incremental impulse
531  Vector2 d = x - a;
532 
533  // Apply incremental impulse
534  Vector2 p1 = d.X * normal;
535  Vector2 p2 = d.Y * normal;
536  vA -= mA * (p1 + p2);
537  wA -= iA * (MathUtils.Cross(cp1.RA, p1) + MathUtils.Cross(cp2.RA, p2));
538 
539  vB += mB * (p1 + p2);
540  wB += iB * (MathUtils.Cross(cp1.RB, p1) + MathUtils.Cross(cp2.RB, p2));
541 
542  // Accumulate
543  cp1.NormalImpulse = x.X;
544  cp2.NormalImpulse = x.Y;
545 
546 #if B2_DEBUG_SOLVER
547  // Postconditions
548  dv1 = vB + MathUtils.Cross(wB, cp1.rB) - vA - MathUtils.Cross(wA, cp1.rA);
549  dv2 = vB + MathUtils.Cross(wB, cp2.rB) - vA - MathUtils.Cross(wA, cp2.rA);
550 
551  // Compute normal velocity
552  vn1 = Vector2.Dot(dv1, normal);
553  vn2 = Vector2.Dot(dv2, normal);
554 
555  Debug.Assert(Math.Abs(vn1 - cp1.VelocityBias) < k_errorTol);
556  Debug.Assert(Math.Abs(vn2 - cp2.VelocityBias) < k_errorTol);
557 #endif
558  break;
559  }
560 
561  //
562  // Case 2: vn1 = 0 and x2 = 0
563  //
564  // 0 = a11 * x1 + a12 * 0 + b1'
565  // vn2 = a21 * x1 + a22 * 0 + b2'
566  //
567  x.X = -cp1.NormalMass * b.X;
568  x.Y = 0.0f;
569  vn1 = 0.0f;
570  vn2 = vc.K.Ex.Y * x.X + b.Y;
571 
572  if (x.X >= 0.0f && vn2 >= 0.0f)
573  {
574  // Get the incremental impulse
575  Vector2 d = x - a;
576 
577  // Apply incremental impulse
578  Vector2 p1 = d.X * normal;
579  Vector2 p2 = d.Y * normal;
580  vA -= mA * (p1 + p2);
581  wA -= iA * (MathUtils.Cross(cp1.RA, p1) + MathUtils.Cross(cp2.RA, p2));
582 
583  vB += mB * (p1 + p2);
584  wB += iB * (MathUtils.Cross(cp1.RB, p1) + MathUtils.Cross(cp2.RB, p2));
585 
586  // Accumulate
587  cp1.NormalImpulse = x.X;
588  cp2.NormalImpulse = x.Y;
589 
590 #if B2_DEBUG_SOLVER
591  // Postconditions
592  dv1 = vB + MathUtils.Cross(wB, cp1.rB) - vA - MathUtils.Cross(wA, cp1.rA);
593 
594  // Compute normal velocity
595  vn1 = Vector2.Dot(dv1, normal);
596 
597  Debug.Assert(Math.Abs(vn1 - cp1.VelocityBias) < k_errorTol);
598 #endif
599  break;
600  }
601 
602  //
603  // Case 3: vn2 = 0 and x1 = 0
604  //
605  // vn1 = a11 * 0 + a12 * x2 + b1'
606  // 0 = a21 * 0 + a22 * x2 + b2'
607  //
608  x.X = 0.0f;
609  x.Y = -cp2.NormalMass * b.Y;
610  vn1 = vc.K.Ey.X * x.Y + b.X;
611  vn2 = 0.0f;
612 
613  if (x.Y >= 0.0f && vn1 >= 0.0f)
614  {
615  // Resubstitute for the incremental impulse
616  Vector2 d = x - a;
617 
618  // Apply incremental impulse
619  Vector2 p1 = d.X * normal;
620  Vector2 p2 = d.Y * normal;
621  vA -= mA * (p1 + p2);
622  wA -= iA * (MathUtils.Cross(cp1.RA, p1) + MathUtils.Cross(cp2.RA, p2));
623 
624  vB += mB * (p1 + p2);
625  wB += iB * (MathUtils.Cross(cp1.RB, p1) + MathUtils.Cross(cp2.RB, p2));
626 
627  // Accumulate
628  cp1.NormalImpulse = x.X;
629  cp2.NormalImpulse = x.Y;
630 
631 #if B2_DEBUG_SOLVER
632  // Postconditions
633  dv2 = vB + MathUtils.Cross(wB, cp2.rB) - vA - MathUtils.Cross(wA, cp2.rA);
634 
635  // Compute normal velocity
636  vn2 = Vector2.Dot(dv2, normal);
637 
638  Debug.Assert(Math.Abs(vn2 - cp2.VelocityBias) < k_errorTol);
639 #endif
640  break;
641  }
642 
643  //
644  // Case 4: x1 = 0 and x2 = 0
645  //
646  // vn1 = b1
647  // vn2 = b2;
648  x.X = 0.0f;
649  x.Y = 0.0f;
650  vn1 = b.X;
651  vn2 = b.Y;
652 
653  if (vn1 >= 0.0f && vn2 >= 0.0f)
654  {
655  // Resubstitute for the incremental impulse
656  Vector2 d = x - a;
657 
658  // Apply incremental impulse
659  Vector2 p1 = d.X * normal;
660  Vector2 p2 = d.Y * normal;
661  vA -= mA * (p1 + p2);
662  wA -= iA * (MathUtils.Cross(cp1.RA, p1) + MathUtils.Cross(cp2.RA, p2));
663 
664  vB += mB * (p1 + p2);
665  wB += iB * (MathUtils.Cross(cp1.RB, p1) + MathUtils.Cross(cp2.RB, p2));
666 
667  // Accumulate
668  cp1.NormalImpulse = x.X;
669  cp2.NormalImpulse = x.Y;
670  }
671 
672  // No solution, give up. This is hit sometimes, but it doesn't seem to matter.
673  break;
674  }
675  }
676 
677  velocities[indexA].V = vA;
678  velocities[indexA].W = wA;
679  velocities[indexB].V = vB;
680  velocities[indexB].W = wB;
681  }

710  public bool SolvePositionConstraints()
711  {
712  float minSeparation = 0.0f;
713 
714  for (int i = 0; i < count; ++i)
715  {
716  ContactPositionConstraint pc = positionConstraints[i];
717 
718  int indexA = pc.IndexA;
719  int indexB = pc.IndexB;
720  Vector2 localCenterA = pc.LocalCenterA;
721  float mA = pc.InvMassA;
722  float iA = pc.InvIa;
723  Vector2 localCenterB = pc.LocalCenterB;
724  float mB = pc.InvMassB;
725  float iB = pc.InvIb;
726  int pointCount = pc.PointCount;
727 
728  Vector2 cA = positions[indexA].C;
729  float aA = positions[indexA].A;
730 
731  Vector2 cB = positions[indexB].C;
732  float aB = positions[indexB].A;
733 
734  // Solve normal constraints
735  for (int j = 0; j < pointCount; ++j)
736  {
737  Transform xfA = new Transform();
738  Transform xfB = new Transform();
739  xfA.Q.Set(aA);
740  xfB.Q.Set(aB);
741  xfA.P = cA - MathUtils.Mul(xfA.Q, localCenterA);
742  xfB.P = cB - MathUtils.Mul(xfB.Q, localCenterB);
743 
744  PositionSolverManifold.Initialize(pc, ref xfA, ref xfB, j, out Vector2 normal, out Vector2 point,
745  out float separation);
746 
747  Vector2 rA = point - cA;
748  Vector2 rB = point - cB;
749 
750  // Track max constraint error.
751  minSeparation = Math.Min(minSeparation, separation);
752 
753  // Prevent large corrections and allow slop.
754  float c = MathUtils.Clamp(Settings.Baumgarte * (separation + Settings.LinearSlop),
755  -Settings.MaxLinearCorrection, 0.0f);
756 
757  // Compute the effective mass.
758  float rnA = MathUtils.Cross(rA, normal);
759  float rnB = MathUtils.Cross(rB, normal);
760  float k = mA + mB + iA * rnA * rnA + iB * rnB * rnB;
761 
762  // Compute normal impulse
763  float impulse = k > 0.0f ? -c / k : 0.0f;
764 
765  Vector2 p = impulse * normal;
766 
767  cA -= mA * p;
768  aA -= iA * MathUtils.Cross(rA, p);
769 
770  cB += mB * p;
771  aB += iB * MathUtils.Cross(rB, p);
772  }
773 
774  positions[indexA].C = cA;
775  positions[indexA].A = aA;
776 
777  positions[indexB].C = cB;
778  positions[indexB].A = aB;
779  }
780 
781  // We can't expect minSpeparation >= -b2_linearSlop because we don't
782  // push the separation above -b2_linearSlop.
783  return minSeparation >= -3.0f * Settings.LinearSlop;
784  }
