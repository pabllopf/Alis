67  if (decayMultiplier != 0)

69  Vector2 forceVector;
70

73  forceVector = body.Position - Position;
74  }
75  else

80

84  }
85  }
86 
87  // Calculate random Variation

89  {
90  float strengthVariation = (float) Randomize.NextDouble() * MathHelper.Clamp(Variation, 0, 1);
91  forceVector = Vector2.Normalize(forceVector);
92  body.ApplyForce(forceVector * strength * decayMultiplier * strengthVariation);
93  }
94  else
95  {
