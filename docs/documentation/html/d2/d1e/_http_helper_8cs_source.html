71  public static string ComputeSocketAcceptString(string secWebSocketKey)
72  {
73  // this is a guid as per the web socket spec
74  const string webSocketGuid = "258EAFA5-E914-47DA-95CA-C5AB0DC85B11";
75  string concatenated = secWebSocketKey + webSocketGuid;

90  public static async Task<string> ReadHttpHeaderAsync(Stream stream, CancellationToken token)
91  {
92  int length = 1024 * 16; // 16KB buffer more than enough for http header
93  byte[] buffer = new byte[length];
94  int offset = 0;
95  int bytesRead = 0;
96 
97  do
98  {
99  if (offset >= length)
100  {
101  throw new EntityTooLargeException("Http header message too large to fit in buffer (16KB)");
102  }
103 
104  bytesRead = await stream.ReadAsync(buffer, offset, length - offset, token);
105  offset += bytesRead;
106  string header = Encoding.UTF8.GetString(buffer, 0, offset);
107 
108  // as per http specification, all headers should end this this
109  if (header.Contains("\r\n\r\n"))
110  {
111  return header;
112  }
113  } while (bytesRead > 0);
114 
115  return string.Empty;
116  }

164  public static IList<string> GetSubProtocols(string httpHeader)
165  {
166  Regex regex = new Regex(@"Sec-WebSocket-Protocol:(?<protocols>.+)", RegexOptions.IgnoreCase);
167  Match match = regex.Match(httpHeader);
168 
169  if (match.Success)
170  {
171  const int maxLen = 2048;
172  if (match.Length > maxLen)
173  {
174  throw new EntityTooLargeException(
175  $"Sec-WebSocket-Protocol exceeded the maximum of length of {maxLen}");
176  }
177 
178  // extract a csv list of sub protocols (in order of highest preference first)
179  string csv = match.Groups["protocols"].Value.Trim();
180  return csv.Split(new[] {','}, StringSplitOptions.RemoveEmptyEntries)
181  .Select(x => x.Trim())
182  .ToList();
183  }
184 
185  return new List<string>();
186  }
