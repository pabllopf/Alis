127  public float AreaRatio
128  {
129  get
130  {
131  if (this.root == NullNode)
132  {
133  return 0.0f;
134  }
135 
136  TreeNode<T> root = nodes[this.root];
137  float rootArea = root.Aabb.Perimeter;
138 
139  float totalArea = 0.0f;
140  for (int i = 0; i < nodeCapacity; ++i)
141  {
142  TreeNode<T> node = nodes[i];
143  if (node.Height < 0)
144  {
145  // Free node in pool
146  continue;
147  }
148 
149  totalArea += node.Aabb.Perimeter;
150  }
151 
152  return totalArea / rootArea;
153  }

233  public bool MoveProxy(int proxyId, ref Aabb aabb, Vector2 displacement)
234  {
235  Debug.Assert(0 <= proxyId && proxyId < nodeCapacity);
236 
237  Debug.Assert(nodes[proxyId].IsLeaf());
238 
239  // Extend AABB
240  Aabb fatAabb = new Aabb();
241  Vector2 r = new Vector2(Settings.AabbExtension, Settings.AabbExtension);
242  fatAabb.LowerBound = aabb.LowerBound - r;
243  fatAabb.UpperBound = aabb.UpperBound + r;
244 
245  // Predict AABB movement
246  Vector2 d = Settings.AabbMultiplier * displacement;
247 
248  if (d.X < 0.0f)
249  {
250  fatAabb.LowerBound.X += d.X;
251  }
252  else
253  {
254  fatAabb.UpperBound.X += d.X;
255  }
256 
257  if (d.Y < 0.0f)
258  {
259  fatAabb.LowerBound.Y += d.Y;
260  }
261  else
262  {
263  fatAabb.UpperBound.Y += d.Y;
264  }
265 
266  Aabb treeAabb = nodes[proxyId].Aabb;
267  if (treeAabb.Contains(ref aabb))
268  {
269  // The tree AABB still contains the object, but it might be too large.
270  // Perhaps the object was moving fast but has since gone to sleep.
271  // The huge AABB is larger than the new fat AABB.
272  Aabb hugeAabb = new Aabb
273  {
274  LowerBound = fatAabb.LowerBound - 4.0f * r,
275  UpperBound = fatAabb.UpperBound + 4.0f * r
276  };
277 
278  if (hugeAabb.Contains(ref treeAabb))
279  {
280  // The tree AABB contains the object AABB and the tree AABB is
281  // not too large. No tree update needed.
282  return false;
283  }
284 
285  // Otherwise the tree AABB is huge and needs to be shrunk
286  }
287 
288  RemoveLeaf(proxyId);
289 
290  nodes[proxyId].Aabb = fatAabb;
291 
292  InsertLeaf(proxyId);
293 
294  nodes[proxyId].Moved = true;
295 
296  return true;
297  }

346  public void Query(Func<int, bool> callback, ref Aabb aabb)
347  {
348  queryStack.Clear();
349  queryStack.Push(root);
350 
351  while (queryStack.Count > 0)
352  {
353  int nodeId = queryStack.Pop();
354  if (nodeId == NullNode)
355  {
356  continue;
357  }
358 
359  TreeNode<T> node = nodes[nodeId];
360 
361  if (Aabb.TestOverlap(ref node.Aabb, ref aabb))
362  {
363  if (node.IsLeaf())
364  {
365  bool proceed = callback(nodeId);
366  if (!proceed)
367  {
368  return;
369  }
370  }
371  else
372  {
373  queryStack.Push(node.Child1);
374  queryStack.Push(node.Child2);
375  }
376  }
377  }

388  public void RayCast(Func<RayCastInput, int, float> callback, ref RayCastInput input)
389  {
390  Vector2 p1 = input.Point1;
391  Vector2 p2 = input.Point2;
392  Vector2 r = p2 - p1;
393  Debug.Assert(r.LengthSquared() > 0.0f);
394  r = Vector2.Normalize(r);
395 
396  // v is perpendicular to the segment.
397  Vector2 absV = MathUtils.Abs(new Vector2(-r.Y, r.X)); //Velcro: Inlined the 'v' variable
398 
399  // Separating axis for segment (Gino, p80).
400  // |dot(v, p1 - c)| > dot(|v|, h)
401 
402  float maxFraction = input.MaxFraction;
403 
404  // Build a bounding box for the segment.
405  Aabb segmentAabb = new Aabb();
406  {
407  Vector2 t = p1 + maxFraction * (p2 - p1);
408  segmentAabb.LowerBound = Vector2.Min(p1, t);
409  segmentAabb.UpperBound = Vector2.Max(p1, t);
410  }
411 
412  raycastStack.Clear();
413  raycastStack.Push(root);
414 
415  while (raycastStack.Count > 0)
416  {
417  int nodeId = raycastStack.Pop();
418  if (nodeId == NullNode)
419  {
420  continue;
421  }
422 
423  TreeNode<T> node = nodes[nodeId];
424 
425  if (!Aabb.TestOverlap(ref node.Aabb, ref segmentAabb))
426  {
427  continue;
428  }
429 
430  // Separating axis for segment (Gino, p80).
431  // |dot(v, p1 - c)| > dot(|v|, h)
432  Vector2 c = node.Aabb.Center;
433  Vector2 h = node.Aabb.Extents;
434  float separation = Math.Abs(Vector2.Dot(new Vector2(-r.Y, r.X), p1 - c)) - Vector2.Dot(absV, h);
435  if (separation > 0.0f)
436  {
437  continue;
438  }
439 
440  if (node.IsLeaf())
441  {
442  RayCastInput subInput;
443  subInput.Point1 = input.Point1;
444  subInput.Point2 = input.Point2;
445  subInput.MaxFraction = maxFraction;
446 
447  float value = callback(subInput, nodeId);
448 
449  if (value == 0.0f)
450  {
451  // the client has terminated the raycast.
452  return;
453  }
454 
455  if (value > 0.0f)
456  {
457  // Update segment bounding box.
458  maxFraction = value;
459  Vector2 t = p1 + maxFraction * (p2 - p1);
460  segmentAabb.LowerBound = Vector2.Min(p1, t);
461  segmentAabb.UpperBound = Vector2.Max(p1, t);
462  }
463  }
464  else
465  {
466  raycastStack.Push(node.Child1);
467  raycastStack.Push(node.Child2);
468  }
469  }
470  }

476  private int AllocateNode()
477  {
478  // Expand the node pool as needed.
479  if (freeList == NullNode)
480  {
481  Debug.Assert(nodeCount == nodeCapacity);
482 
483  // The free list is empty. Rebuild a bigger pool.
484  TreeNode<T>[] oldNodes = nodes;
485  nodeCapacity *= 2;
486  nodes = new TreeNode<T>[nodeCapacity];
487  Array.Copy(oldNodes, nodes, nodeCount);
488 
489  // Build a linked list for the free list. The parent
490  // pointer becomes the "next" pointer.
491  for (int i = nodeCount; i < nodeCapacity - 1; ++i)
492  {
493  nodes[i] = new TreeNode<T>
494  {
495  ParentOrNext = i + 1,
496  Height = -1
497  };
498  }
499 
500  nodes[nodeCapacity - 1] = new TreeNode<T>
501  {
502  ParentOrNext = NullNode,
503  Height = -1
504  };
505  freeList = nodeCount;
506  }
507 
508  // Peel a node off the free list.
509  int nodeId = freeList;
510  freeList = nodes[nodeId].ParentOrNext;
511  nodes[nodeId].ParentOrNext = NullNode;
512  nodes[nodeId].Child1 = NullNode;
513  nodes[nodeId].Child2 = NullNode;
514  nodes[nodeId].Height = 0;
515  nodes[nodeId].UserData = default(T);
516  nodes[nodeId].Moved = false;
517  ++nodeCount;
518  return nodeId;
519  }

525  private void FreeNode(int nodeId)
526  {
527  Debug.Assert(0 <= nodeId && nodeId < nodeCapacity);
528  Debug.Assert(0 < nodeCount);
529  nodes[nodeId].ParentOrNext = freeList;
530  nodes[nodeId].Height = -1;
531  freeList = nodeId;
532  --nodeCount;

539  private void InsertLeaf(int leaf)
540  {
541  if (root == NullNode)
542  {
543  root = leaf;
544  nodes[root].ParentOrNext = NullNode;
545  return;
546  }
547 
548  // Find the best sibling for this node
549  Aabb leafAabb = nodes[leaf].Aabb;
550  int index = root;
551  while (!nodes[index].IsLeaf())
552  {
553  int child1 = nodes[index].Child1;
554  int child2 = nodes[index].Child2;
555 
556  float area = nodes[index].Aabb.Perimeter;
557 
558  Aabb combinedAabb = new Aabb();
559  combinedAabb.Combine(ref nodes[index].Aabb, ref leafAabb);
560  float combinedArea = combinedAabb.Perimeter;
561 
562  // Cost of creating a new parent for this node and the new leaf
563  float cost = 2.0f * combinedArea;
564 
565  // Minimum cost of pushing the leaf further down the tree
566  float inheritanceCost = 2.0f * (combinedArea - area);
567 
568  // Cost of descending into child1
569  float cost1;
570  if (nodes[child1].IsLeaf())
571  {
572  Aabb aabb = new Aabb();
573  aabb.Combine(ref leafAabb, ref nodes[child1].Aabb);
574  cost1 = aabb.Perimeter + inheritanceCost;
575  }
576  else
577  {
578  Aabb aabb = new Aabb();
579  aabb.Combine(ref leafAabb, ref nodes[child1].Aabb);
580  float oldArea = nodes[child1].Aabb.Perimeter;
581  float newArea = aabb.Perimeter;
582  cost1 = newArea - oldArea + inheritanceCost;
583  }
584 
585  // Cost of descending into child2
586  float cost2;
587  if (nodes[child2].IsLeaf())
588  {
589  Aabb aabb = new Aabb();
590  aabb.Combine(ref leafAabb, ref nodes[child2].Aabb);
591  cost2 = aabb.Perimeter + inheritanceCost;
592  }
593  else
594  {
595  Aabb aabb = new Aabb();
596  aabb.Combine(ref leafAabb, ref nodes[child2].Aabb);
597  float oldArea = nodes[child2].Aabb.Perimeter;
598  float newArea = aabb.Perimeter;
599  cost2 = newArea - oldArea + inheritanceCost;
600  }
601 
602  // Descend according to the minimum cost.
603  if (cost < cost1 && cost1 < cost2)
604  {
605  break;
606  }
607 
608  // Descend
609  if (cost1 < cost2)
610  {
611  index = child1;
612  }
613  else
614  {
615  index = child2;
616  }
617  }
618 
619  int sibling = index;
620 
621  // Create a new parent.
622  int oldParent = nodes[sibling].ParentOrNext;
623  int newParent = AllocateNode();
624  nodes[newParent].ParentOrNext = oldParent;
625  nodes[newParent].UserData = default(T);
626  nodes[newParent].Aabb.Combine(ref leafAabb, ref nodes[sibling].Aabb);
627  nodes[newParent].Height = nodes[sibling].Height + 1;
628 
629  if (oldParent != NullNode)
630  {
631  // The sibling was not the root.
632  if (nodes[oldParent].Child1 == sibling)
633  {
634  nodes[oldParent].Child1 = newParent;
635  }
636  else
637  {
638  nodes[oldParent].Child2 = newParent;
639  }
640 
641  nodes[newParent].Child1 = sibling;
642  nodes[newParent].Child2 = leaf;
643  nodes[sibling].ParentOrNext = newParent;
644  nodes[leaf].ParentOrNext = newParent;
645  }
646  else
647  {
648  // The sibling was the root.
649  nodes[newParent].Child1 = sibling;
650  nodes[newParent].Child2 = leaf;
651  nodes[sibling].ParentOrNext = newParent;
652  nodes[leaf].ParentOrNext = newParent;
653  root = newParent;
654  }
655 
656  // Walk back up the tree fixing heights and AABBs
657  index = nodes[leaf].ParentOrNext;
658  while (index != NullNode)
659  {
660  index = Balance(index);
661 
662  int child1 = nodes[index].Child1;
663  int child2 = nodes[index].Child2;
664 
665  Debug.Assert(child1 != NullNode);
666  Debug.Assert(child2 != NullNode);
667 
668  nodes[index].Height = 1 + Math.Max(nodes[child1].Height, nodes[child2].Height);
669  nodes[index].Aabb.Combine(ref nodes[child1].Aabb, ref nodes[child2].Aabb);
670 
671  index = nodes[index].ParentOrNext;
672  }
673 
674  //Validate();
675  }

681  private void RemoveLeaf(int leaf)
682  {
683  if (leaf == root)
684  {
685  root = NullNode;
686  return;
687  }
688 
689  int parent = nodes[leaf].ParentOrNext;
690  int grandParent = nodes[parent].ParentOrNext;
691  int sibling;
692  if (nodes[parent].Child1 == leaf)
693  {
694  sibling = nodes[parent].Child2;
695  }
696  else
697  {
698  sibling = nodes[parent].Child1;
699  }
700 
701  if (grandParent != NullNode)
702  {
703  // Destroy parent and connect sibling to grandParent.
704  if (nodes[grandParent].Child1 == parent)
705  {
706  nodes[grandParent].Child1 = sibling;
707  }
708  else
709  {
710  nodes[grandParent].Child2 = sibling;
711  }
712 
713  nodes[sibling].ParentOrNext = grandParent;
714  FreeNode(parent);
715 
716  // Adjust ancestor bounds.
717  int index = grandParent;
718  while (index != NullNode)
719  {
720  index = Balance(index);
721 
722  int child1 = nodes[index].Child1;
723  int child2 = nodes[index].Child2;
724 
725  nodes[index].Aabb.Combine(ref nodes[child1].Aabb, ref nodes[child2].Aabb);
726  nodes[index].Height = 1 + Math.Max(nodes[child1].Height, nodes[child2].Height);
727 
728  index = nodes[index].ParentOrNext;
729  }
730  }
731  else
732  {
733  root = sibling;
734  nodes[sibling].ParentOrNext = NullNode;
735  FreeNode(parent);
736  }
737 
738  //Validate();
739  }

744  private int Balance(int iA)
745  {
746  Debug.Assert(iA != NullNode);
747 
748  TreeNode<T> a = nodes[iA];
749  if (a.IsLeaf() || a.Height < 2)
750  {
751  return iA;
752  }
753 
754  int iB = a.Child1;
755  int iC = a.Child2;
756  Debug.Assert(0 <= iB && iB < nodeCapacity);
757  Debug.Assert(0 <= iC && iC < nodeCapacity);
758 
759  TreeNode<T> b = nodes[iB];
760  TreeNode<T> c = nodes[iC];
761 
762  int balance = c.Height - b.Height;
763 
764  // Rotate C up
765  if (balance > 1)
766  {
767  int iF = c.Child1;
768  int iG = c.Child2;
769  TreeNode<T> f = nodes[iF];
770  TreeNode<T> g = nodes[iG];
771  Debug.Assert(0 <= iF && iF < nodeCapacity);
772  Debug.Assert(0 <= iG && iG < nodeCapacity);
773 
774  // Swap A and C
775  c.Child1 = iA;
776  c.ParentOrNext = a.ParentOrNext;
777  a.ParentOrNext = iC;
778 
779  // A's old parent should point to C
780  if (c.ParentOrNext != NullNode)
781  {
782  if (nodes[c.ParentOrNext].Child1 == iA)
783  {
784  nodes[c.ParentOrNext].Child1 = iC;
785  }
786  else
787  {
788  Debug.Assert(nodes[c.ParentOrNext].Child2 == iA);
789  nodes[c.ParentOrNext].Child2 = iC;
790  }
791  }
792  else
793  {
794  root = iC;
795  }
796 
797  // Rotate
798  if (f.Height > g.Height)
799  {
800  c.Child2 = iF;
801  a.Child2 = iG;
802  g.ParentOrNext = iA;
803  a.Aabb.Combine(ref b.Aabb, ref g.Aabb);
804  c.Aabb.Combine(ref a.Aabb, ref f.Aabb);
805 
806  a.Height = 1 + Math.Max(b.Height, g.Height);
807  c.Height = 1 + Math.Max(a.Height, f.Height);
808  }
809  else
810  {
811  c.Child2 = iG;
812  a.Child2 = iF;
813  f.ParentOrNext = iA;
814  a.Aabb.Combine(ref b.Aabb, ref f.Aabb);
815  c.Aabb.Combine(ref a.Aabb, ref g.Aabb);
816 
817  a.Height = 1 + Math.Max(b.Height, f.Height);
818  c.Height = 1 + Math.Max(a.Height, g.Height);
819  }
820 
821  return iC;
822  }
823 
824  // Rotate B up
825  if (balance < -1)
826  {
827  int iD = b.Child1;
828  int iE = b.Child2;
829  TreeNode<T> d = nodes[iD];
830  TreeNode<T> e = nodes[iE];
831  Debug.Assert(0 <= iD && iD < nodeCapacity);
832  Debug.Assert(0 <= iE && iE < nodeCapacity);
833 
834  // Swap A and B
835  b.Child1 = iA;
836  b.ParentOrNext = a.ParentOrNext;
837  a.ParentOrNext = iB;
838 
839  // A's old parent should point to B
840  if (b.ParentOrNext != NullNode)
841  {
842  if (nodes[b.ParentOrNext].Child1 == iA)
843  {
844  nodes[b.ParentOrNext].Child1 = iB;
845  }
846  else
847  {
848  Debug.Assert(nodes[b.ParentOrNext].Child2 == iA);
849  nodes[b.ParentOrNext].Child2 = iB;
850  }
851  }
852  else
853  {
854  root = iB;
855  }
856 
857  // Rotate
858  if (d.Height > e.Height)
859  {
860  b.Child2 = iD;
861  a.Child1 = iE;
862  e.ParentOrNext = iA;
863  a.Aabb.Combine(ref c.Aabb, ref e.Aabb);
864  b.Aabb.Combine(ref a.Aabb, ref d.Aabb);
865 
866  a.Height = 1 + Math.Max(c.Height, e.Height);
867  b.Height = 1 + Math.Max(a.Height, d.Height);
868  }
869  else
870  {
871  b.Child2 = iE;
872  a.Child1 = iD;
873  d.ParentOrNext = iA;
874  a.Aabb.Combine(ref c.Aabb, ref d.Aabb);
875  b.Aabb.Combine(ref a.Aabb, ref e.Aabb);
876 
877  a.Height = 1 + Math.Max(c.Height, d.Height);
878  b.Height = 1 + Math.Max(a.Height, e.Height);
879  }
880 
881  return iB;
882  }
883 
884  return iA;
885  }

891  {
892  Debug.Assert(0 <= nodeId && nodeId < nodeCapacity);

894 
895  if (node.IsLeaf())
