351  public void RayCast(Func<RayCastInput, int, float> callback, ref RayCastInput input)
352  {
353  Vector2F p1 = input.Point1;
354  Vector2F p2 = input.Point2;
355  Vector2F r = p2 - p1;
356  r = Vector2F.Normalize(r);
357 
358  Vector2F absV = MathUtils.Abs(new Vector2F(-r.Y, r.X));
359 
360  float maxFraction = input.Fraction;
361 
362  Aabb segmentAabb = new Aabb();
363  {
364  Vector2F t = p1 + maxFraction * (p2 - p1);
365  segmentAabb.LowerBound = Vector2F.Min(p1, t);
366  segmentAabb.UpperBound = Vector2F.Max(p1, t);
367  }
368 
369  raycastStack.Clear();
370  raycastStack.Push(root);
371 
372  while (raycastStack.Count > 0)
373  {
374  int nodeId = raycastStack.Pop();
375  if (nodeId == NullNode)
376  {
377  continue;
378  }
379 
380  TreeNode<T> node = nodes[nodeId];
381 
382  if (!Aabb.TestOverlap(ref node.Aabb, ref segmentAabb))
383  {
384  continue;
385  }
386 
387  Vector2F c = node.Aabb.Center;
388  Vector2F h = node.Aabb.Extents;
389  float separation = Math.Abs(Vector2F.Dot(new Vector2F(-r.Y, r.X), p1 - c)) - Vector2F.Dot(absV, h);
390  if (separation > 0.0f)
391  {
392  continue;
393  }
394 
395  if (node.IsLeaf())
396  {
397  RayCastInput subInput;
398  subInput.Point1 = input.Point1;
399  subInput.Point2 = input.Point2;
400  subInput.Fraction = maxFraction;
401 
402  float value = callback(subInput, nodeId);
403 
404  if (value == 0.0f)
405  {
406  return;
407  }
408 
409  if (value > 0.0f)
410  {
411  maxFraction = value;
412  Vector2F t = p1 + maxFraction * (p2 - p1);
413  segmentAabb.LowerBound = Vector2F.Min(p1, t);
414  segmentAabb.UpperBound = Vector2F.Max(p1, t);
415  }
416  }
417  else
418  {
419  raycastStack.Push(node.Child1);
420  raycastStack.Push(node.Child2);
421  }
422  }
423  }

483  private void InsertLeaf(int leaf)
484  {
485  if (root == NullNode)
486  {
487  root = leaf;
488  nodes[root].ParentOrNext = NullNode;
489  return;
490  }
491 
492  Aabb leafAabb = nodes[leaf].Aabb;
493  int index = root;
494  while (!nodes[index].IsLeaf())
495  {
496  int child1 = nodes[index].Child1;
497  int child2 = nodes[index].Child2;
498 
499  float area = nodes[index].Aabb.Perimeter;
500 
501  Aabb combinedAabb = new Aabb();
502  combinedAabb.Combine(ref nodes[index].Aabb, ref leafAabb);
503  float combinedArea = combinedAabb.Perimeter;
504 
505  float cost = 2.0f * combinedArea;
506 
507  float inheritanceCost = 2.0f * (combinedArea - area);
508 
509  float cost1;
510  if (nodes[child1].IsLeaf())
511  {
512  Aabb aabb = new Aabb();
513  aabb.Combine(ref leafAabb, ref nodes[child1].Aabb);
514  cost1 = aabb.Perimeter + inheritanceCost;
515  }
516  else
517  {
518  Aabb aabb = new Aabb();
519  aabb.Combine(ref leafAabb, ref nodes[child1].Aabb);
520  float oldArea = nodes[child1].Aabb.Perimeter;
521  float newArea = aabb.Perimeter;
522  cost1 = newArea - oldArea + inheritanceCost;
523  }
524 
525  float cost2;
526  if (nodes[child2].IsLeaf())
527  {
528  Aabb aabb = new Aabb();
529  aabb.Combine(ref leafAabb, ref nodes[child2].Aabb);
530  cost2 = aabb.Perimeter + inheritanceCost;
531  }
532  else
533  {
534  Aabb aabb = new Aabb();
535  aabb.Combine(ref leafAabb, ref nodes[child2].Aabb);
536  float oldArea = nodes[child2].Aabb.Perimeter;
537  float newArea = aabb.Perimeter;
538  cost2 = newArea - oldArea + inheritanceCost;
539  }
540 
541  if ((cost < cost1) && (cost1 < cost2))
542  {
543  break;
544  }
545 
546  index = cost1 < cost2 ? child1 : child2;
547  }
548 
549  int sibling = index;
550 
551  int oldParent = nodes[sibling].ParentOrNext;
552  int newParent = AllocateNode();
553  nodes[newParent].ParentOrNext = oldParent;
554  nodes[newParent].UserData = default(T);
555  nodes[newParent].Aabb.Combine(ref leafAabb, ref nodes[sibling].Aabb);
556  nodes[newParent].Height = nodes[sibling].Height + 1;
557 
558  if (oldParent != NullNode)
559  {
560  if (nodes[oldParent].Child1 == sibling)
561  {
562  nodes[oldParent].Child1 = newParent;
563  }
564  else
565  {
566  nodes[oldParent].Child2 = newParent;
567  }
568 
569  nodes[newParent].Child1 = sibling;
570  nodes[newParent].Child2 = leaf;
571  nodes[sibling].ParentOrNext = newParent;
572  nodes[leaf].ParentOrNext = newParent;
573  }
574  else
575  {
576  nodes[newParent].Child1 = sibling;
577  nodes[newParent].Child2 = leaf;
578  nodes[sibling].ParentOrNext = newParent;
579  nodes[leaf].ParentOrNext = newParent;
580  root = newParent;
581  }
582 
583  index = nodes[leaf].ParentOrNext;
584  while (index != NullNode)
585  {
586  index = BalanceTo(index);
587 
588  int child1 = nodes[index].Child1;
589  int child2 = nodes[index].Child2;
590 
591  nodes[index].Height = 1 + Math.Max(nodes[child1].Height, nodes[child2].Height);
592  nodes[index].Aabb.Combine(ref nodes[child1].Aabb, ref nodes[child2].Aabb);
593 
594  index = nodes[index].ParentOrNext;
595  }
596  }

602  private void RemoveLeaf(int leaf)
603  {
604  if (leaf == root)
605  {
606  root = NullNode;
607  return;
608  }
609 
610  int parent = nodes[leaf].ParentOrNext;
611  int grandParent = nodes[parent].ParentOrNext;
612  int sibling = nodes[parent].Child1 == leaf ? nodes[parent].Child2 : nodes[parent].Child1;
613 
614  if (grandParent != NullNode)
615  {
616  if (nodes[grandParent].Child1 == parent)
617  {
618  nodes[grandParent].Child1 = sibling;
619  }
620  else
621  {
622  nodes[grandParent].Child2 = sibling;
623  }
624 
625  nodes[sibling].ParentOrNext = grandParent;
626  FreeNode(parent);
627 
628  int index = grandParent;
629  while (index != NullNode)
630  {
631  index = BalanceTo(index);
632 
633  int child1 = nodes[index].Child1;
634  int child2 = nodes[index].Child2;
635 
636  nodes[index].Aabb.Combine(ref nodes[child1].Aabb, ref nodes[child2].Aabb);
637  nodes[index].Height = 1 + Math.Max(nodes[child1].Height, nodes[child2].Height);
638 
639  index = nodes[index].ParentOrNext;
640  }
641  }
642  else
643  {
644  root = sibling;
645  nodes[sibling].ParentOrNext = NullNode;
646  FreeNode(parent);
647  }

653  private int BalanceTo(int iA)
654  {
655  TreeNode<T> a = nodes[iA];
656  if (a.IsLeaf() || a.Height < 2)
657  {
658  return iA;
659  }
660 
661  int iB = a.Child1;
662  int iC = a.Child2;
663 
664  TreeNode<T> b = nodes[iB];
665  TreeNode<T> c = nodes[iC];
666 
667  int balance = c.Height - b.Height;
668 
669  if (balance > 1)
670  {
671  int iF = c.Child1;
672  int iG = c.Child2;
673  TreeNode<T> f = nodes[iF];
674  TreeNode<T> g = nodes[iG];
675 
676  c.Child1 = iA;
677  c.ParentOrNext = a.ParentOrNext;
678  a.ParentOrNext = iC;
679 
680 
681  if (c.ParentOrNext != NullNode)
682  {
683  if (nodes[c.ParentOrNext].Child1 == iA)
684  {
685  nodes[c.ParentOrNext].Child1 = iC;
686  }
687  else
688  {
689  nodes[c.ParentOrNext].Child2 = iC;
690  }
691  }
692  else
693  {
694  root = iC;
695  }
696 
697 
698  if (f.Height > g.Height)
699  {
700  c.Child2 = iF;
701  a.Child2 = iG;
702  g.ParentOrNext = iA;
703  a.Aabb.Combine(ref b.Aabb, ref g.Aabb);
704  c.Aabb.Combine(ref a.Aabb, ref f.Aabb);
705 
706  a.Height = 1 + Math.Max(b.Height, g.Height);
707  c.Height = 1 + Math.Max(a.Height, f.Height);
708  }
709  else
710  {
711  c.Child2 = iG;
712  a.Child2 = iF;
713  f.ParentOrNext = iA;
714  a.Aabb.Combine(ref b.Aabb, ref f.Aabb);
715  c.Aabb.Combine(ref a.Aabb, ref g.Aabb);
716 
717  a.Height = 1 + Math.Max(b.Height, f.Height);
718  c.Height = 1 + Math.Max(a.Height, g.Height);
719  }
720 
721  return iC;
722  }
723 
724 
725  if (balance < -1)
726  {
727  int iD = b.Child1;
728  int iE = b.Child2;
729  TreeNode<T> d = nodes[iD];
730  TreeNode<T> e = nodes[iE];
731 
732 
733  b.Child1 = iA;
734  b.ParentOrNext = a.ParentOrNext;
735  a.ParentOrNext = iB;
736 
737 
738  if (b.ParentOrNext != NullNode)
739  {
740  if (nodes[b.ParentOrNext].Child1 == iA)
741  {
742  nodes[b.ParentOrNext].Child1 = iB;
743  }
744  else
745  {
746  nodes[b.ParentOrNext].Child2 = iB;
747  }
748  }
749  else
750  {
751  root = iB;
752  }
753 
754 
755  if (d.Height > e.Height)
756  {
757  b.Child2 = iD;
758  a.Child1 = iE;
759  e.ParentOrNext = iA;
760  a.Aabb.Combine(ref c.Aabb, ref e.Aabb);
761  b.Aabb.Combine(ref a.Aabb, ref d.Aabb);
762 
763  a.Height = 1 + Math.Max(c.Height, e.Height);
764  b.Height = 1 + Math.Max(a.Height, d.Height);
765  }
766  else
767  {
768  b.Child2 = iE;
769  a.Child1 = iD;
770  d.ParentOrNext = iA;
771  a.Aabb.Combine(ref c.Aabb, ref d.Aabb);
772  b.Aabb.Combine(ref a.Aabb, ref e.Aabb);
773 
774  a.Height = 1 + Math.Max(c.Height, d.Height);
775  b.Height = 1 + Math.Max(a.Height, e.Height);
776  }
777 
778  return iB;
779  }
780 
781  return iA;
782  }

876  public void RebuildBottomUp()
877  {
878  int[] ints = new int[nodeCount];
879  int count = 0;
880 
881 
882  for (int i = 0; i < nodeCapacity; ++i)
883  {
884  if (nodes[i].Height < 0)
885  {
886  continue;
887  }
888 
889  if (nodes[i].IsLeaf())
890  {
891  nodes[i].ParentOrNext = NullNode;
892  ints[count] = i;
893  ++count;
894  }
895  else
896  {
897  FreeNode(i);
898  }
899  }
900 
901  while (count > 1)
902  {
903  float minCost = float.MaxValue;
904  int iMin = -1, jMin = -1;
905  for (int i = 0; i < count; ++i)
906  {
907  Aabb aabBi = nodes[ints[i]].Aabb;
908 
909  for (int j = i + 1; j < count; ++j)
910  {
911  Aabb aabBj = nodes[ints[j]].Aabb;
912  Aabb b = new Aabb();
913  b.Combine(ref aabBi, ref aabBj);
914  float cost = b.Perimeter;
915  if (cost < minCost)
916  {
917  iMin = i;
918  jMin = j;
919  minCost = cost;
920  }
921  }
922  }
923 
924  int index1 = ints[iMin];
925  int index2 = ints[jMin];
926  TreeNode<T> child1 = nodes[index1];
927  TreeNode<T> child2 = nodes[index2];
928 
929  int parentIndex = AllocateNode();
930  TreeNode<T> parent = nodes[parentIndex];
931  parent.Child1 = index1;
