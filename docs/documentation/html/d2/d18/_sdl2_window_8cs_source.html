485  private void WindowOwnerRoutine(object state)
486  {
487  WindowParams wp = (WindowParams)state;
488  _window = wp.Create();
489  WindowID = SDL_GetWindowID(_window);
490  Sdl2WindowRegistry.RegisterWindow(this);
491  PostWindowCreated(wp.WindowFlags);
492  wp.ResetEvent.Set();
493 
494  double previousPollTimeMs = 0;
495  Stopwatch sw = new Stopwatch();
496  sw.Start();
497 
498  while (_exists)
499  {
500  if (_shouldClose && CloseCore())
501  {
502  return;
503  }
504 
505  double currentTick = sw.ElapsedTicks;
506  double currentTimeMs = sw.ElapsedTicks * (1000.0 / Stopwatch.Frequency);
507  if (LimitPollRate && currentTimeMs - previousPollTimeMs < PollIntervalInMs)
508  {
509  Thread.Sleep(0);
510  }
511  else
512  {
513  previousPollTimeMs = currentTimeMs;
514  ProcessEvents(null);
515  }

605  private unsafe void HandleEvent(SDL_Event* ev)
606  {
607  switch (ev->type)
608  {
609  case SDL_EventType.Quit:
610  Close();
611  break;
612  case SDL_EventType.Terminating:
613  Close();
614  break;
615  case SDL_EventType.WindowEvent:
616  SDL_WindowEvent windowEvent = Unsafe.Read<SDL_WindowEvent>(ev);
617  HandleWindowEvent(windowEvent);
618  break;
619  case SDL_EventType.KeyDown:
620  case SDL_EventType.KeyUp:
621  SDL_KeyboardEvent keyboardEvent = Unsafe.Read<SDL_KeyboardEvent>(ev);
622  HandleKeyboardEvent(keyboardEvent);
623  break;
624  case SDL_EventType.TextEditing:
625  break;
626  case SDL_EventType.TextInput:
627  SDL_TextInputEvent textInputEvent = Unsafe.Read<SDL_TextInputEvent>(ev);
628  HandleTextInputEvent(textInputEvent);
629  break;
630  case SDL_EventType.KeyMapChanged:
631  break;
632  case SDL_EventType.MouseMotion:
633  SDL_MouseMotionEvent mouseMotionEvent = Unsafe.Read<SDL_MouseMotionEvent>(ev);
634  HandleMouseMotionEvent(mouseMotionEvent);
635  break;
636  case SDL_EventType.MouseButtonDown:
637  case SDL_EventType.MouseButtonUp:
638  SDL_MouseButtonEvent mouseButtonEvent = Unsafe.Read<SDL_MouseButtonEvent>(ev);
639  HandleMouseButtonEvent(mouseButtonEvent);
640  break;
641  case SDL_EventType.MouseWheel:
642  SDL_MouseWheelEvent mouseWheelEvent = Unsafe.Read<SDL_MouseWheelEvent>(ev);
643  HandleMouseWheelEvent(mouseWheelEvent);
644  break;
645  case SDL_EventType.DropFile:
646  case SDL_EventType.DropBegin:
647  case SDL_EventType.DropTest:
648  SDL_DropEvent dropEvent = Unsafe.Read<SDL_DropEvent>(ev);
649  HandleDropEvent(dropEvent);
650  break;
651  default:
652  // Ignore
653  break;
654  }
655  }

673  private void HandleTextInputEvent(SDL_TextInputEvent textInputEvent)
674  {
675  uint byteCount = 0;
676  // Loop until the null terminator is found or the max size is reached.
677  while (byteCount < SDL_TextInputEvent.MaxTextSize && textInputEvent.text[byteCount++] != 0)
678  { }
679 
680  if (byteCount > 1)
681  {
682  // We don't want the null terminator.
683  byteCount -= 1;
684  int charCount = Encoding.UTF8.GetCharCount(textInputEvent.text, (int)byteCount);
685  char* charsPtr = stackalloc char[charCount];
686  Encoding.UTF8.GetChars(textInputEvent.text, (int)byteCount, charsPtr, charCount);
687  for (int i = 0; i < charCount; i++)
688  {
689  _privateSnapshot.KeyCharPressesList.Add(charsPtr[i]);
690  }
691  }
692  }
