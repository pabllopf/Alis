486  private void WindowOwnerRoutine(object state)
487  {
488  WindowParams wp = (WindowParams)state;
489  _window = wp.Create();
490  WindowID = SDL_GetWindowID(_window);
491  Sdl2WindowRegistry.RegisterWindow(this);
492  PostWindowCreated(wp.WindowFlags);
493  wp.ResetEvent.Set();
494 
495  double previousPollTimeMs = 0;
496  Stopwatch sw = new Stopwatch();
497  sw.Start();
498 
499  while (_exists)
500  {
501  if (_shouldClose && CloseCore())
502  {
503  return;
504  }
505 
506  double currentTick = sw.ElapsedTicks;
507  double currentTimeMs = sw.ElapsedTicks * (1000.0 / Stopwatch.Frequency);
508  if (LimitPollRate && currentTimeMs - previousPollTimeMs < PollIntervalInMs)
509  {
510  Thread.Sleep(0);
511  }
512  else
513  {
514  previousPollTimeMs = currentTimeMs;
515  ProcessEvents(null);
516  }

606  private unsafe void HandleEvent(SDL_Event* ev)
607  {
608  switch (ev->type)
609  {
610  case SDL_EventType.Quit:
611  Close();
612  break;
613  case SDL_EventType.Terminating:
614  Close();
615  break;
616  case SDL_EventType.WindowEvent:
617  SDL_WindowEvent windowEvent = Unsafe.Read<SDL_WindowEvent>(ev);
618  HandleWindowEvent(windowEvent);
619  break;
620  case SDL_EventType.KeyDown:
621  case SDL_EventType.KeyUp:
622  SDL_KeyboardEvent keyboardEvent = Unsafe.Read<SDL_KeyboardEvent>(ev);
623  HandleKeyboardEvent(keyboardEvent);
624  break;
625  case SDL_EventType.TextEditing:
626  break;
627  case SDL_EventType.TextInput:
628  SDL_TextInputEvent textInputEvent = Unsafe.Read<SDL_TextInputEvent>(ev);
629  HandleTextInputEvent(textInputEvent);
630  break;
631  case SDL_EventType.KeyMapChanged:
632  break;
633  case SDL_EventType.MouseMotion:
634  SDL_MouseMotionEvent mouseMotionEvent = Unsafe.Read<SDL_MouseMotionEvent>(ev);
635  HandleMouseMotionEvent(mouseMotionEvent);
636  break;
637  case SDL_EventType.MouseButtonDown:
638  case SDL_EventType.MouseButtonUp:
639  SDL_MouseButtonEvent mouseButtonEvent = Unsafe.Read<SDL_MouseButtonEvent>(ev);
640  HandleMouseButtonEvent(mouseButtonEvent);
641  break;
642  case SDL_EventType.MouseWheel:
643  SDL_MouseWheelEvent mouseWheelEvent = Unsafe.Read<SDL_MouseWheelEvent>(ev);
644  HandleMouseWheelEvent(mouseWheelEvent);
645  break;
646  case SDL_EventType.DropFile:
647  case SDL_EventType.DropBegin:
648  case SDL_EventType.DropTest:
649  SDL_DropEvent dropEvent = Unsafe.Read<SDL_DropEvent>(ev);
650  HandleDropEvent(dropEvent);
651  break;
652  default:
653  // Ignore
654  break;
655  }
656  }

674  private void HandleTextInputEvent(SDL_TextInputEvent textInputEvent)
675  {
676  uint byteCount = 0;
677  // Loop until the null terminator is found or the max size is reached.
678  while (byteCount < SDL_TextInputEvent.MaxTextSize && textInputEvent.text[byteCount++] != 0)
679  { }
680 
681  if (byteCount > 1)
682  {
683  // We don't want the null terminator.
684  byteCount -= 1;
685  int charCount = Encoding.UTF8.GetCharCount(textInputEvent.text, (int)byteCount);
686  char* charsPtr = stackalloc char[charCount];
687  Encoding.UTF8.GetChars(textInputEvent.text, (int)byteCount, charsPtr, charCount);
688  for (int i = 0; i < charCount; i++)
689  {
690  _privateSnapshot.KeyCharPressesList.Add(charsPtr[i]);
691  }
692  }
693  }
