72  public static void ComputeDistance(ref DistanceInput input, out DistanceOutput output, out SimplexCache cache)
73  {
74  cache = new SimplexCache();
75 
76  ++GjkCalls;
77 
78  // Initialize the simplex.
79  Simplex simplex = new Simplex();
80  simplex.ReadCache(ref cache, ref input.ProxyA, ref input.TransformA, ref input.ProxyB,
81  ref input.TransformB);
82 
83  // These store the vertices of the last simplex so that we
84  // can check for duplicates and prevent cycling.
85  FixedArray3<int> saveA = new FixedArray3<int>();
86  FixedArray3<int> saveB = new FixedArray3<int>();
87 
88  // Main iteration loop.
89  int iter = 0;
90 
91  //Velcro: Moved the max iterations to settings
92  while (iter < Settings.GjkIterations)
93  {
94  // Copy simplex so we can identify duplicates.
95  int saveCount = simplex.Count;
96  for (int i = 0; i < saveCount; ++i)
97  {
98  saveA[i] = simplex.V[i].IndexA;
99  saveB[i] = simplex.V[i].IndexB;
100  }
101 
102  switch (simplex.Count)
103  {
104  case 1:
105  break;
106  case 2:
107  simplex.Solve2();
108  break;
109  case 3:
110  simplex.Solve3();
111  break;
112  default:
113  Debug.Assert(false);
114  break;
115  }
116 
117  // If we have 3 points, then the origin is in the corresponding triangle.
118  if (simplex.Count == 3)
119  {
120  break;
121  }
122 
123  // Get search direction.
124  Vector2F d = simplex.GetSearchDirection();
125 
126  // Ensure the search direction is numerically fit.
127  if (d.LengthSquared() < Constant.Epsilon * Constant.Epsilon)
128  {
129  // The origin is probably contained by a line segment
130  // or triangle. Thus the shapes are overlapped.
131 
132  // We can't return zero here even though there may be overlap.
133  // In case the simplex is a point, segment, or triangle it is difficult
134  // to determine if the origin is contained in the CSO or very close to it.
135  break;
136  }
137 
138  // Compute a tentative new simplex vertex using support points.
139  SimplexVertex vertex = simplex.V[simplex.Count];
140  vertex.IndexA = input.ProxyA.GetSupport(MathUtils.MulT(input.TransformA.Rotation, -d));
141  vertex.Wa = MathUtils.Mul(ref input.TransformA, input.ProxyA.Vertices[vertex.IndexA]);
142 
143  vertex.IndexB = input.ProxyB.GetSupport(MathUtils.MulT(input.TransformB.Rotation, d));
144  vertex.Wb = MathUtils.Mul(ref input.TransformB, input.ProxyB.Vertices[vertex.IndexB]);
145  vertex.W = vertex.Wb - vertex.Wa;
146  simplex.V[simplex.Count] = vertex;
147 
148  // Iteration count is equated to the number of support point calls.
149  ++iter;
150 
151  ++GjkIters;
152 
153  // Check for duplicate support points. This is the main termination criteria.
154  bool duplicate = false;
155  for (int i = 0; i < saveCount; ++i)
156  {
157  if ((vertex.IndexA == saveA[i]) && (vertex.IndexB == saveB[i]))
158  {
159  duplicate = true;
160  break;
161  }
162  }
163 
164  // If we found a duplicate support point we must exit to avoid cycling.
165  if (duplicate)
166  {
167  break;
168  }
169 
170  // New vertex is OK and needed.
171  ++simplex.Count;
172  }
173 
174  GjkMaxIters = Math.Max(GjkMaxIters, iter);
175 
176  // Prepare output.
177  simplex.GetWitnessPoints(out output.PointA, out output.PointB);
178  output.Distance = (output.PointA - output.PointB).Length();
179  output.Iterations = iter;
180 
181  // Cache the simplex.
182  simplex.WriteCache(ref cache);
183 
184  // Apply radii if requested.
185  if (input.UseRadii)
186  {
187  float rA = input.ProxyA.Radius;
188  float rB = input.ProxyB.Radius;
189 
190  if ((output.Distance > rA + rB) && (output.Distance > Constant.Epsilon))
191  {
192  // Shapes are still no overlapped.
193  // Move the witness points to the outer surface.
194  output.Distance -= rA + rB;
195  Vector2F normal = output.PointB - output.PointA;
196  normal = Vector2F.Normalize(normal);
197  output.PointA += rA * normal;
198  output.PointB -= rB * normal;
199  }
200  else
201  {
202  // Shapes are overlapped when radii are considered.
203  // Move the witness points to the middle.
204  Vector2F p = 0.5f * (output.PointA + output.PointB);
205  output.PointA = p;
206  output.PointB = p;
207  output.Distance = 0.0f;
208  }
209  }
210  }

221  public static bool ShapeCast(ref ShapeCastInput input, out ShapeCastOutput output)
222  {
223  output = new ShapeCastOutput
224  {
225  Iterations = 0,
226  Lambda = 1.0f,
227  Normal = Vector2F.Zero,
228  Point = Vector2F.Zero
229  };
230 
231  DistanceProxy proxyA = input.ProxyA;
232  DistanceProxy proxyB = input.ProxyB;
233 
234  float radiusA = MathUtils.Max(proxyA.Radius, Settings.PolygonRadius);
235  float radiusB = MathUtils.Max(proxyB.Radius, Settings.PolygonRadius);
236  float radius = radiusA + radiusB;
237 
238  Transform xfA = input.TransformA;
239  Transform xfB = input.TransformB;
240 
241  Vector2F r = input.TranslationB;
242  Vector2F n = new Vector2F(0.0f, 0.0f);
243  float lambda = 0.0f;
244 
245  // Initial simplex
246  Simplex simplex = new Simplex
247  {
248  Count = 0
249  };
250 
251  // Get simplex vertices as an array.
252  //SimplexVertex vertices = simplex.V.Value0; //Velcro: we don't need this as we have an indexer instead
253 
254  // Get support point in -r direction
255  int indexA = proxyA.GetSupport(MathUtils.MulT(xfA.Rotation, -r));
256  Vector2F wA = MathUtils.Mul(ref xfA, proxyA.GetVertex(indexA));
257  int indexB = proxyB.GetSupport(MathUtils.MulT(xfB.Rotation, r));
258  Vector2F wB = MathUtils.Mul(ref xfB, proxyB.GetVertex(indexB));
259  Vector2F v = wA - wB;
260 
261  // Sigma is the target distance between polygons
262  float sigma = MathUtils.Max(Settings.PolygonRadius, radius - Settings.PolygonRadius);
263  float tolerance = 0.5f * Settings.LinearSlop;
264 
265  // Main iteration loop.
266  int iter = 0;
267 
268  //Velcro: We have moved the max iterations into settings
269  while ((iter < Settings.GjkIterations) && (v.Length() - sigma > tolerance))
270  {
271  Debug.Assert(simplex.Count < 3);
272 
273  output.Iterations += 1;
274 
275  // Support in direction -v (A - B)
276  indexA = proxyA.GetSupport(MathUtils.MulT(xfA.Rotation, -v));
277  wA = MathUtils.Mul(ref xfA, proxyA.GetVertex(indexA));
278  indexB = proxyB.GetSupport(MathUtils.MulT(xfB.Rotation, v));
279  wB = MathUtils.Mul(ref xfB, proxyB.GetVertex(indexB));
280  Vector2F p = wA - wB;
281 
282  // -v is a normal at p
283  v = Vector2F.Normalize(v);
284 
285  // Intersect ray with plane
286  float vp = MathUtils.Dot(ref v, ref p);
287  float vr = MathUtils.Dot(ref v, ref r);
288  if (vp - sigma > lambda * vr)
289  {
290  if (vr <= 0.0f)
291  {
292  return false;
293  }
294 
295  lambda = (vp - sigma) / vr;
296  if (lambda > 1.0f)
297  {
298  return false;
299  }
300 
301  n = -v;
302  simplex.Count = 0;
303  }
304 
305  // Reverse simplex since it works with B - A.
306  // Shift by lambda * r because we want the closest point to the current clip point.
307  // Note that the support point p is not shifted because we want the plane equation
308  // to be formed in unshifted space.
309  SimplexVertex vertex = simplex.V[simplex.Count];
310  vertex.IndexA = indexB;
311  vertex.Wa = wB + lambda * r;
312  vertex.IndexB = indexA;
313  vertex.Wb = wA;
314  vertex.W = vertex.Wb - vertex.Wa;
315  vertex.A = 1.0f;
316  simplex.V[simplex.Count] = vertex; //Velcro: we have to copy the value back
317  simplex.Count += 1;
318 
319  switch (simplex.Count)
320  {
321  case 1:
322  break;
323 
324  case 2:
325  simplex.Solve2();
326  break;
327 
328  case 3:
329  simplex.Solve3();
330  break;
331 
332  default:
333  Debug.Assert(false);
334  break;
335  }
336 
337  // If we have 3 points, then the origin is in the corresponding triangle.
338  if (simplex.Count == 3)
339  {
340  // Overlap
341  return false;
342  }
343 
344  // Get search direction.
345  v = simplex.GetClosestPoint();
346 
347  // Iteration count is equated to the number of support point calls.
348  ++iter;
349  }
350 
351  if (iter == 0)
352  {
353  // Initial overlap
354  return false;
355  }
356 
357  // Prepare output.
358  simplex.GetWitnessPoints(out _, out Vector2F pointB);
359 
360  if (v.LengthSquared() > 0.0f)
361  {
362  n = -v;
363  n = Vector2F.Normalize(n);
364  }
365 
366  output.Point = pointB + radiusA * n;
367  output.Normal = n;
368  output.Lambda = lambda;
369  output.Iterations = iter;
370  return true;
371  }
372  }
