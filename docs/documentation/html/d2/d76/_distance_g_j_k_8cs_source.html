72  public static void ComputeDistance(ref DistanceInput input, out DistanceOutput output, out SimplexCache cache)
73  {
74  cache = new SimplexCache();
75 
76  ++GjkCalls;
77 
78  // Initialize the simplex.
79  Simplex simplex = new Simplex();
80  simplex.ReadCache(ref cache, ref input.ProxyA, ref input.TransformA, ref input.ProxyB,
81  ref input.TransformB);
82 
83  // These store the vertices of the last simplex so that we
84  // can check for duplicates and prevent cycling.
85  FixedArray3<int> saveA = new FixedArray3<int>();
86  FixedArray3<int> saveB = new FixedArray3<int>();
87 
88  // Main iteration loop.
89  int iter = 0;
90 
91  //Velcro: Moved the max iterations to settings
92  while (iter < Settings.GjkIterations)
93  {
94  // Copy simplex so we can identify duplicates.
95  int saveCount = simplex.Count;
96  for (int i = 0; i < saveCount; ++i)
97  {
98  saveA[i] = simplex.V[i].IndexA;
99  saveB[i] = simplex.V[i].IndexB;
100  }
101 
102  switch (simplex.Count)
103  {
104  case 1:
105  break;
106  case 2:
107  simplex.Solve2();
108  break;
109  case 3:
110  simplex.Solve3();
111  break;
112  default:
113  Debug.Assert(false);
114  break;
115  }
116 
117  // If we have 3 points, then the origin is in the corresponding triangle.
118  if (simplex.Count == 3)
119  {
120  break;
121  }
122 
123  // Get search direction.
124  Vector2 d = simplex.GetSearchDirection();
125 
126  // Ensure the search direction is numerically fit.
127  if (d.LengthSquared() < Constant.Epsilon * Constant.Epsilon)
128  {
129  // The origin is probably contained by a line segment
130  // or triangle. Thus the shapes are overlapped.
131 
132  // We can't return zero here even though there may be overlap.
133  // In case the simplex is a point, segment, or triangle it is difficult
134  // to determine if the origin is contained in the CSO or very close to it.
135  break;
136  }
137 
138  // Compute a tentative new simplex vertex using support points.
139  SimplexVertex vertex = simplex.V[simplex.Count];
140  vertex.IndexA = input.ProxyA.GetSupport(MathUtils.MulT(input.TransformA.Rotation, -d));
141  vertex.Wa = MathUtils.Mul(ref input.TransformA, input.ProxyA.Vertices[vertex.IndexA]);
142 
143  vertex.IndexB = input.ProxyB.GetSupport(MathUtils.MulT(input.TransformB.Rotation, d));
144  vertex.Wb = MathUtils.Mul(ref input.TransformB, input.ProxyB.Vertices[vertex.IndexB]);
145  vertex.W = vertex.Wb - vertex.Wa;
146  simplex.V[simplex.Count] = vertex;
147 
148  // Iteration count is equated to the number of support point calls.
149  ++iter;
150 
151  ++GjkIter;
152 
153  // Check for duplicate support points. This is the main termination criteria.
154  bool duplicate = false;
155  for (int i = 0; i < saveCount; ++i)
156  {
157  if ((vertex.IndexA == saveA[i]) && (vertex.IndexB == saveB[i]))
158  {
159  duplicate = true;
160  break;
161  }
162  }
163 
164  // If we found a duplicate support point we must exit to avoid cycling.
165  if (duplicate)
166  {
167  break;
168  }
169 
170  // New vertex is OK and needed.
171  ++simplex.Count;
172  }
173 
174  _gjkMaxIter = Math.Max(_gjkMaxIter, iter);
175 
176  // Prepare output.
177  simplex.GetWitnessPoints(out output.PointA, out output.PointB);
178  output.Distance = (output.PointA - output.PointB).Length();
179  output.Iterations = iter;
180 
181  // Cache the simplex.
182  simplex.WriteCache(ref cache);
183 
184  // Apply radii if requested.
185  if (input.UseRadii)
186  {
187  float rA = input.ProxyA.Radius;
188  float rB = input.ProxyB.Radius;
189 
190  if ((output.Distance > rA + rB) && (output.Distance > Constant.Epsilon))
191  {
192  // Shapes are still no overlapped.
193  // Move the witness points to the outer surface.
194  output.Distance -= rA + rB;
195  Vector2 normal = output.PointB - output.PointA;
196  normal = Vector2.Normalize(normal);
197  output.PointA += rA * normal;
198  output.PointB -= rB * normal;
199  }
200  else
201  {
202  // Shapes are overlapped when radii are considered.
203  // Move the witness points to the middle.
204  Vector2 p = 0.5f * (output.PointA + output.PointB);
205  output.PointA = p;
206  output.PointB = p;
207  output.Distance = 0.0f;
208  }
209  }
210  }

217  public static bool ShapeCast(ref ShapeCastInput input, out ShapeCastOutput output)
218  {
219  InitializeOutput(out output);
220 
221  DistanceProxy proxyA = input.ProxyA;
222  DistanceProxy proxyB = input.ProxyB;
223 
224  float radiusA = CalculateRadius(proxyA);
225  float radiusB = CalculateRadius(proxyB);
226  float radius = radiusA + radiusB;
227 
228  Transform xfA = input.TransformA;
229  Transform xfB = input.TransformB;
230 
231  Vector2 r = input.TranslationB;
232  Vector2 n = Vector2.Zero;
233  float lambda = 0.0f;
234 
235  Simplex simplex = InitializeSimplex();
236 
237  int iter = 0;
238 
239  while (IterateUntilConverged(ref proxyA, ref proxyB, ref xfA, ref xfB, ref r, ref n, ref lambda, ref simplex, radius))
240  {
241  // Iteration count is equated to the number of support point calls.
242  ++iter;
243  }
244 
245  if (iter == 0)
246  {
247  // Initial overlap
248  return false;
249  }
250 
251  CalculateOutput(ref output, ref simplex, ref r, ref lambda, ref n, radiusA);
252  return true;
253  }

305  private static bool IterateUntilConverged(ref DistanceProxy proxyA, ref DistanceProxy proxyB,
306  ref Transform xfA, ref Transform xfB, ref Vector2 r, ref Vector2 n, ref float lambda, ref Simplex simplex, float radius)
307  {
308  Vector2 v = ComputeV(ref proxyA, ref proxyB, ref xfA, ref xfB, ref r, ref n, ref lambda, ref simplex, radius);
309 
310  if (IsConverged(v, ref lambda, radius))
311  {
312  return false;
313  }

315  UpdateSimplex(ref proxyA, ref proxyB, ref xfA, ref xfB, ref r, ref v, ref n, ref lambda, ref simplex, radius);
316  return true;

332  private static Vector2 ComputeV(ref DistanceProxy proxyA, ref DistanceProxy proxyB,
333  ref Transform xfA, ref Transform xfB, ref Vector2 r, ref Vector2 n, ref float lambda, ref Simplex simplex, float radius)
334  {
335  Vector2 v = ComputeSupport(ref proxyA, ref proxyB, ref xfA, ref xfB, ref r);
336 
337  if (IsNewDirectionNeeded(ref n, ref r, ref lambda, radius))
338  {
339  n = -v;
340  n = Vector2.Normalize(n);
341  simplex.Count = 0;
342  }
343 
344  return v;

356  private static Vector2 ComputeSupport(ref DistanceProxy proxyA, ref DistanceProxy proxyB, ref Transform xfA, ref Transform xfB,
357  ref Vector2 r)
358  {
359  int indexA = proxyA.GetSupport(MathUtils.MulT(xfA.Rotation, -r));
360  Vector2 wA = MathUtils.Mul(ref xfA, proxyA.GetVertex(indexA));
361  int indexB = proxyB.GetSupport(MathUtils.MulT(xfB.Rotation, r));
362  Vector2 wB = MathUtils.Mul(ref xfB, proxyB.GetVertex(indexB));
363  Vector2 v = wA - wB;
364 
365  return v;
366  }

375  private static bool IsConverged(Vector2 v, ref float lambda, float radius)
376  {
377  float sigma = MathUtils.Max(Settings.PolygonRadius, 2 * radius - Settings.PolygonRadius);
378  float tolerance = 0.5f * Settings.LinearSlop;
379 
380  return v.Length() - sigma <= tolerance || lambda > 1.0f;
381  }

391  private static bool IsNewDirectionNeeded(ref Vector2 n, ref Vector2 r, ref float lambda, float radius)
392  {
393  float vp = MathUtils.Dot(ref n, ref r);
394  return vp - (2 * radius - Settings.PolygonRadius) > lambda * MathUtils.Dot(ref n, ref r);
395  }

410  private static void UpdateSimplex(ref DistanceProxy proxyA, ref DistanceProxy proxyB,
411  ref Transform xfA, ref Transform xfB, ref Vector2 r, ref Vector2 v, ref Vector2 n, ref float lambda, ref Simplex simplex, float radius)
412  {
413  int indexA = proxyA.GetSupport(MathUtils.MulT(xfA.Rotation, -v));
414  Vector2 wA = MathUtils.Mul(ref xfA, proxyA.GetVertex(indexA));
415  int indexB = proxyB.GetSupport(MathUtils.MulT(xfB.Rotation, v));
416  Vector2 wB = MathUtils.Mul(ref xfB, proxyB.GetVertex(indexB));
417  Vector2 p = wA - wB;
418 
419  v = Vector2.Normalize(v);
420 
421  float vp = MathUtils.Dot(ref v, ref p);
422  float vr = MathUtils.Dot(ref v, ref r);
423 
424  if (vp - (2 * radius - Settings.PolygonRadius) > lambda * vr)
425  {
426  if (vr <= 0.0f)
427  {
428  return;
429  }
430 
431  lambda = (vp - (2 * radius - Settings.PolygonRadius)) / vr;
432 
433  if (lambda > 1.0f)
434  {
435  return;
436  }
437 
438  n = -v;
439  simplex.Count = 0;
440  }
441 
442  SimplexVertex vertex = simplex.V[simplex.Count];
443  vertex.IndexA = indexB;
444  vertex.Wa = wB + lambda * r;
445  vertex.IndexB = indexA;
446  vertex.Wb = wA;
447  vertex.W = vertex.Wb - vertex.Wa;
448  vertex.A = 1.0f;
449  simplex.V[simplex.Count] = vertex;
450  simplex.Count += 1;
451 
452  switch (simplex.Count)
453  {
454  case 1:
455  break;
456 
457  case 2:
458  simplex.Solve2();
459  break;
460 
461  case 3:
462  simplex.Solve3();
463  break;
464 
465  default:
466  Debug.Assert(false);
467  break;
468  }
469  }

481  {
482  simplex.GetWitnessPoints(out _, out Vector2 pointB);
483

485  {

488  }
489

491  output.Normal = n;
492  output.Lambda = lambda;
493  output.Iterations = simplex.Count;
494  }
