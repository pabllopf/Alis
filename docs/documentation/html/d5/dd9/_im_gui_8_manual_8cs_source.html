186  public static bool InputText(
187  string label,
188  ref string input,
189  uint maxLength,
190  ImGuiInputTextFlags flag,
191  ImGuiInputTextCallback callback,
192  IntPtr userData)
193  {
194  // Convert label and input to ANSI strings
195  IntPtr labelPtr = Marshal.StringToHGlobalAnsi(label);
196  IntPtr inputPtr = Marshal.StringToHGlobalAnsi(input);
197 
198  // Convert ANSI strings to UTF-8 bytes
199  byte* utf8LabelBytes = (byte*) labelPtr.ToPointer();
200  byte* utf8InputBytes = (byte*) inputPtr.ToPointer();
201 
202  // Create buffers for modified input
203  int inputBufSize = Math.Max((int) maxLength + 1, Encoding.UTF8.GetByteCount(input) + 1);
204  byte* modifiedUtf8InputBytes = stackalloc byte[inputBufSize];
205  byte* originalUtf8InputBytes = stackalloc byte[inputBufSize];
206 
207  // Copy input bytes to the modified input buffer
208  Unsafe.CopyBlock(modifiedUtf8InputBytes, utf8InputBytes, (uint) inputBufSize);
209 
210  // Call the ImGuiNative method
211  byte result = ImGuiNative.igInputText(
212  utf8LabelBytes,
213  modifiedUtf8InputBytes,
214  (uint) inputBufSize,
215  flag,
216  callback,
217  userData.ToPointer());
218 
219  // Check if the input was modified and update the input variable accordingly
220  if (!Util.AreStringsEqual(originalUtf8InputBytes, inputBufSize, modifiedUtf8InputBytes))
221  {
222  input = Encoding.UTF8.GetString(modifiedUtf8InputBytes, inputBufSize);
223  }
224 
225  // Free the memory allocated by Marshal.StringToHGlobalAnsi
226  Marshal.FreeHGlobal(labelPtr);
227  Marshal.FreeHGlobal(inputPtr);
228 
229  return result != 0;
230  }

456  private static byte* GetUtf8Bytes(string text, uint maxLength)
457  {
458  int byteCount = Encoding.UTF8.GetByteCount(text);
459  int inputBufSize = Math.Max((int) maxLength + 1, byteCount + 1);
460  byte[] utf8BytesArray = new byte[inputBufSize];

462  fixed (byte* utf8Bytes = utf8BytesArray)

646  private static Vector2 CalcTextSizeImpl(
647  string text,
648  int start = 0,
649  int? length = null,
650  bool hideTextAfterDoubleHash = false,
651  float wrapWidth = -1.0f)
652  {
653  Vector2 ret;
654  byte* nativeTextStart = null;
655  byte* nativeTextEnd = null;
656  int textByteCount = 0;
657  if (text != null)
658  {
659  int textToCopyLen = length.HasValue ? length.Value : text.Length;
660  textByteCount = Util.CalcSizeInUtf8(text, start, textToCopyLen);
661  if (textByteCount > Util.StackAllocationSizeLimit)
662  {
663  nativeTextStart = Util.Allocate(textByteCount + 1);
664  }
665  else
666  {
667  byte* nativeTextStackBytes = stackalloc byte[textByteCount + 1];
668  nativeTextStart = nativeTextStackBytes;
669  }
670 
671  int nativeTextOffset = Util.GetUtf8(text, start, textToCopyLen, nativeTextStart, textByteCount);
672  nativeTextStart[nativeTextOffset] = 0;
673  nativeTextEnd = nativeTextStart + nativeTextOffset;
674  }
675 
676  ImGuiNative.igCalcTextSize(&ret, nativeTextStart, nativeTextEnd, *(byte*) &hideTextAfterDoubleHash, wrapWidth);
677  if (textByteCount > Util.StackAllocationSizeLimit)
678  {
679  Util.Free(nativeTextStart);
680  }
681 
682  return ret;
683  }
