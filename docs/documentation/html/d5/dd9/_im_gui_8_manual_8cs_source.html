187  public static bool InputText(
188  string label,
189  ref string input,
190  uint maxLength,
191  ImGuiInputTextFlags flag,
192  ImGuiInputTextCallback callback,
193  IntPtr userData)
194  {
195  // Convert label and input to ANSI strings
196  IntPtr labelPtr = Marshal.StringToHGlobalAnsi(label);
197  IntPtr inputPtr = Marshal.StringToHGlobalAnsi(input);
198 
199  // Convert ANSI strings to UTF-8 bytes
200  byte* utf8LabelBytes = (byte*) labelPtr.ToPointer();
201  byte* utf8InputBytes = (byte*) inputPtr.ToPointer();
202 
203  // Create buffers for modified input
204  int inputBufSize = Math.Max((int) maxLength + 1, Encoding.UTF8.GetByteCount(input) + 1);
205  byte* modifiedUtf8InputBytes = stackalloc byte[inputBufSize];
206  byte* originalUtf8InputBytes = stackalloc byte[inputBufSize];
207 
208  // Copy input bytes to the modified input buffer
209  Unsafe.CopyBlock(modifiedUtf8InputBytes, utf8InputBytes, (uint) inputBufSize);
210 
211  // Call the ImGuiNative method
212  byte result = ImGuiNative.igInputText(
213  utf8LabelBytes,
214  modifiedUtf8InputBytes,
215  (uint) inputBufSize,
216  flag,
217  callback,
218  userData.ToPointer());
219 
220  // Check if the input was modified and update the input variable accordingly
221  if (!Util.AreStringsEqual(originalUtf8InputBytes, inputBufSize, modifiedUtf8InputBytes))
222  {
223  input = Encoding.UTF8.GetString(modifiedUtf8InputBytes, inputBufSize);
224  }
225 
226  // Free the memory allocated by Marshal.StringToHGlobalAnsi
227  Marshal.FreeHGlobal(labelPtr);
228  Marshal.FreeHGlobal(inputPtr);
229 
230  return result != 0;
231  }

647  private static Vector2 CalcTextSizeImpl(
648  string text,
649  int start = 0,
650  int? length = null,
651  bool hideTextAfterDoubleHash = false,
652  float wrapWidth = -1.0f)
653  {
654  Vector2 ret;
655  byte* nativeTextStart = null;
656  byte* nativeTextEnd = null;
657  int textByteCount = 0;
658  if (text != null)
659  {
660  int textToCopyLen = length.HasValue ? length.Value : text.Length;
661  textByteCount = Util.CalcSizeInUtf8(text, start, textToCopyLen);
662  if (textByteCount > Util.StackAllocationSizeLimit)
663  {
664  nativeTextStart = Util.Allocate(textByteCount + 1);
665  }
666  else
667  {
668  byte* nativeTextStackBytes = stackalloc byte[textByteCount + 1];
669  nativeTextStart = nativeTextStackBytes;
670  }
671 
672  int nativeTextOffset = Util.GetUtf8(text, start, textToCopyLen, nativeTextStart, textByteCount);
673  nativeTextStart[nativeTextOffset] = 0;
674  nativeTextEnd = nativeTextStart + nativeTextOffset;
675  }
676 
677  ImGuiNative.igCalcTextSize(&ret, nativeTextStart, nativeTextEnd, *(byte*) &hideTextAfterDoubleHash, wrapWidth);
678  if (textByteCount > Util.StackAllocationSizeLimit)
679  {
680  Util.Free(nativeTextStart);
681  }
682 
683  return ret;
684  }
