185  public static bool InputText(
186  string label,
187  ref string input,
188  uint maxLength,
189  ImGuiInputTextFlags flags,
190  ImGuiInputTextCallback callback,
191  IntPtr userData)
192  {
193  // Convert label and input to ANSI strings
194  IntPtr labelPtr = Marshal.StringToHGlobalAnsi(label);
195  IntPtr inputPtr = Marshal.StringToHGlobalAnsi(input);
196 
197  // Convert ANSI strings to UTF-8 bytes
198  byte* utf8LabelBytes = (byte*) labelPtr.ToPointer();
199  byte* utf8InputBytes = (byte*) inputPtr.ToPointer();
200 
201  // Create buffers for modified input
202  int inputBufSize = Math.Max((int) maxLength + 1, Encoding.UTF8.GetByteCount(input) + 1);
203  byte* modifiedUtf8InputBytes = stackalloc byte[inputBufSize];
204  byte* originalUtf8InputBytes = stackalloc byte[inputBufSize];
205 
206  // Copy input bytes to the modified input buffer
207  Unsafe.CopyBlock(modifiedUtf8InputBytes, utf8InputBytes, (uint) inputBufSize);
208 
209  // Call the ImGuiNative method
210  byte result = ImGuiNative.igInputText(
211  utf8LabelBytes,
212  modifiedUtf8InputBytes,
213  (uint) inputBufSize,
214  flags,
215  callback,
216  userData.ToPointer());
217 
218  // Check if the input was modified and update the input variable accordingly
219  if (!Util.AreStringsEqual(originalUtf8InputBytes, inputBufSize, modifiedUtf8InputBytes))
220  {
221  input = Encoding.UTF8.GetString(modifiedUtf8InputBytes, inputBufSize);
222  }
223 
224  // Free the memory allocated by Marshal.StringToHGlobalAnsi
225  Marshal.FreeHGlobal(labelPtr);
226  Marshal.FreeHGlobal(inputPtr);
227 
228  return result != 0;
229  }

676  private static Vector2 CalcTextSizeImpl(
677  string text,
678  int start = 0,
679  int? length = null,
680  bool hideTextAfterDoubleHash = false,
681  float wrapWidth = -1.0f)
682  {
683  Vector2 ret;
684  byte* nativeTextStart = null;
685  byte* nativeTextEnd = null;
686  int textByteCount = 0;
687  if (text != null)
688  {
689  int textToCopyLen = length.HasValue ? length.Value : text.Length;
690  textByteCount = Util.CalcSizeInUtf8(text, start, textToCopyLen);
691  if (textByteCount > Util.StackAllocationSizeLimit)
692  {
693  nativeTextStart = Util.Allocate(textByteCount + 1);
694  }
695  else
696  {
697  byte* nativeTextStackBytes = stackalloc byte[textByteCount + 1];
698  nativeTextStart = nativeTextStackBytes;
699  }
700 
701  int nativeTextOffset = Util.GetUtf8(text, start, textToCopyLen, nativeTextStart, textByteCount);
702  nativeTextStart[nativeTextOffset] = 0;
703  nativeTextEnd = nativeTextStart + nativeTextOffset;
704  }
705 
706  ImGuiNative.igCalcTextSize(&ret, nativeTextStart, nativeTextEnd, *(byte*) &hideTextAfterDoubleHash, wrapWidth);
707  if (textByteCount > Util.StackAllocationSizeLimit)
708  {
709  Util.Free(nativeTextStart);
710  }
711 
712  return ret;
713  }
