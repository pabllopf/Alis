186  public static bool InputText(
187  string label,
188  ref string input,
189  uint maxLength,
190  ImGuiInputTextFlags flags,
191  ImGuiInputTextCallback callback,
192  IntPtr userData)
193  {
194  // Convert label and input to ANSI strings
195  IntPtr labelPtr = Marshal.StringToHGlobalAnsi(label);
196  IntPtr inputPtr = Marshal.StringToHGlobalAnsi(input);
197 
198  // Convert ANSI strings to UTF-8 bytes
199  byte* utf8LabelBytes = (byte*)labelPtr.ToPointer();
200  byte* utf8InputBytes = (byte*)inputPtr.ToPointer();
201 
202  // Create buffers for modified input
203  int inputBufSize = Math.Max((int)maxLength + 1, Encoding.UTF8.GetByteCount(input) + 1);
204  byte* modifiedUtf8InputBytes = stackalloc byte[inputBufSize];
205  byte* originalUtf8InputBytes = stackalloc byte[inputBufSize];
206 
207  // Copy input bytes to the modified input buffer
208  Unsafe.CopyBlock(modifiedUtf8InputBytes, utf8InputBytes, (uint)inputBufSize);
209 
210  // Call the ImGuiNative method
211  byte result = ImGuiNative.igInputText(
212  utf8LabelBytes,
213  modifiedUtf8InputBytes,
214  (uint)inputBufSize,
215  flags,
216  callback,
217  userData.ToPointer());
218 
219  // Check if the input was modified and update the input variable accordingly
220  if (!Util.AreStringsEqual(originalUtf8InputBytes, inputBufSize, modifiedUtf8InputBytes))
221  {
222  input = Encoding.UTF8.GetString(modifiedUtf8InputBytes, inputBufSize);
223  }
224 
225  // Free the memory allocated by Marshal.StringToHGlobalAnsi
226  Marshal.FreeHGlobal(labelPtr);
227  Marshal.FreeHGlobal(inputPtr);
228 
229  return result != 0;
230  }

350  {
351  if (array1 == null && array2 == null)
352  {
353  return true;
354  }
355

362  {

367  }

688  private static Vector2 CalcTextSizeImpl(
689  string text,
690  int start = 0,
691  int? length = null,
692  bool hideTextAfterDoubleHash = false,
693  float wrapWidth = -1.0f)
694  {
695  Vector2 ret;
696  byte* nativeTextStart = null;
697  byte* nativeTextEnd = null;
698  int textByteCount = 0;
699  if (text != null)
700  {
701  int textToCopyLen = length.HasValue ? length.Value : text.Length;
702  textByteCount = Util.CalcSizeInUtf8(text, start, textToCopyLen);
703  if (textByteCount > Util.StackAllocationSizeLimit)
704  {
705  nativeTextStart = Util.Allocate(textByteCount + 1);
706  }
707  else
708  {
709  byte* nativeTextStackBytes = stackalloc byte[textByteCount + 1];
710  nativeTextStart = nativeTextStackBytes;
711  }
712 
713  int nativeTextOffset = Util.GetUtf8(text, start, textToCopyLen, nativeTextStart, textByteCount);
714  nativeTextStart[nativeTextOffset] = 0;
715  nativeTextEnd = nativeTextStart + nativeTextOffset;
716  }
717 
718  ImGuiNative.igCalcTextSize(&ret, nativeTextStart, nativeTextEnd, *(byte*) &hideTextAfterDoubleHash, wrapWidth);
719  if (textByteCount > Util.StackAllocationSizeLimit)
720  {
721  Util.Free(nativeTextStart);
722  }
723 
724  return ret;
725  }
