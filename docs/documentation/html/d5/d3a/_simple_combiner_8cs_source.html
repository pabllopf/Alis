48  public static List<Vertices> PolygonizeTriangles(List<Vertices> triangles, int maxPolys = int.MaxValue,
49  float tolerance = 0.001f)
50  {
51  if (triangles.Count <= 0)
52  {
53  return triangles;
54  }
55 
56  List<Vertices> polys = new List<Vertices>();
57 
58  bool[] covered = new bool[triangles.Count];
59  for (int i = 0; i < triangles.Count; ++i)
60  {
61  covered[i] = false;
62 
63  //Check here for degenerate triangles
64  Vertices triangle = triangles[i];
65  Vector2F a = triangle[0];
66  Vector2F b = triangle[1];
67  Vector2F c = triangle[2];
68 
69  if (((a.X == b.X) && (a.Y == b.Y)) || ((b.X == c.X) && (b.Y == c.Y)) || ((a.X == c.X) && (a.Y == c.Y)))
70  {
71  covered[i] = true;
72  }
73  }
74 
75  int polyIndex = 0;
76 
77  bool notDone = true;
78  while (notDone)
79  {
80  int currTri = -1;
81  for (int i = 0; i < triangles.Count; ++i)
82  {
83  if (covered[i])
84  {
85  continue;
86  }
87 
88  currTri = i;
89  break;
90  }
91 
92  if (currTri == -1)
93  {
94  notDone = false;
95  }
96  else
97  {
98  Vertices poly = new Vertices(3);
99 
100  for (int i = 0; i < 3; i++)
101  {
102  poly.Add(triangles[currTri][i]);
103  }
104 
105  covered[currTri] = true;
106  int index = 0;
107  for (int i = 0; i < 2 * triangles.Count; ++i, ++index)
108  {
109  while (index >= triangles.Count)
110  {
111  index -= triangles.Count;
112  }
113 
114  if (covered[index])
115  {
116  continue;
117  }
118 
119  Vertices newP = AddTriangle(triangles[index], poly);
120  if (newP == null)
121  {
122  continue; // is this right
123  }
124 
125  if (newP.Count > Settings.PolygonVertices)
126  {
127  continue;
128  }
129 
130  if (newP.IsConvex())
131  {
132  //Or should it be IsUsable? Maybe re-write IsConvex to apply the angle threshold from Box2d
133  poly = new Vertices(newP);
134  covered[index] = true;
135  }
136  }
137 
138  //We have a maximum of polygons that we need to keep under.
139  if (polyIndex < maxPolys)
140  {
141  SimplifyTools.MergeParallelEdges(poly, tolerance);
142 
143  //If identical points are present, a triangle gets
144  //borked by the MergeParallelEdges function, hence
145  //the vertex number check
146  if (poly.Count >= 3)
147  {
148  polys.Add(new Vertices(poly));
149  }
150  else
151  {
152  Debug.WriteLine("Skipping corrupt poly.");
153  }
154  }
155 
156  if (poly.Count >= 3)
157  {
158  polyIndex++; //Must be outside (polyIndex < polysLength) test
159  }
160  }
161  }
162 
163  //Remove empty vertice collections
164  for (int i = polys.Count - 1; i >= 0; i--)
165  {
166  if (polys[i].Count == 0)
167  {
168  polys.RemoveAt(i);
169  }
170  }
171 
172  return polys;
173  }

181  private static Vertices AddTriangle(Vertices t, Vertices vertices)
182  {
183  // First, find vertices that connect
184  int firstP = -1;
185  int firstT = -1;
186  int secondP = -1;
187  int secondT = -1;
188  for (int i = 0; i < vertices.Count; i++)
189  {
190  if ((t[0].X == vertices[i].X) && (t[0].Y == vertices[i].Y))
191  {
192  if (firstP == -1)
193  {
194  firstP = i;
195  firstT = 0;
196  }
197  else
198  {
199  secondP = i;
200  secondT = 0;
201  }
202  }
203  else if ((t[1].X == vertices[i].X) && (t[1].Y == vertices[i].Y))
204  {
205  if (firstP == -1)
206  {
207  firstP = i;
208  firstT = 1;
209  }
210  else
211  {
212  secondP = i;
213  secondT = 1;
214  }
215  }
216  else if ((t[2].X == vertices[i].X) && (t[2].Y == vertices[i].Y))
217  {
218  if (firstP == -1)
219  {
220  firstP = i;
221  firstT = 2;
222  }
223  else
224  {
225  secondP = i;
226  secondT = 2;
227  }
228  }
229  }
230 
231  // Fix ordering if first should be last vertex of poly
232  if ((firstP == 0) && (secondP == vertices.Count - 1))
233  {
234  firstP = vertices.Count - 1;
235  secondP = 0;
236  }
237 
238  // Didn't find it
239  if (secondP == -1)
240  {
241  return null;
242  }
243 
244  // Find tip index on triangle
245  int tipT = 0;
246  if (tipT == firstT || tipT == secondT)
247  {
248  tipT = 1;
249  }
250 
251  if (tipT == firstT || tipT == secondT)
252  {
253  tipT = 2;
254  }
255 
256  Vertices result = new Vertices(vertices.Count + 1);
257  for (int i = 0; i < vertices.Count; i++)
258  {
259  result.Add(vertices[i]);
260 
261  if (i == firstP)
262  {
263  result.Add(t[tipT]);
264  }
265  }
266 
267  return result;
268  }
269  }
