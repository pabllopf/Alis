96  private void AddPair(ref FixtureProxy proxyA, ref FixtureProxy proxyB)
97  {
98  Fixture fixtureA = proxyA.Fixture;
99  Fixture fixtureB = proxyB.Fixture;
100 
101  int indexA = proxyA.ChildIndex;
102  int indexB = proxyB.ChildIndex;
103 
104  Body bodyA = fixtureA.Body;
105  Body bodyB = fixtureB.Body;
106 
107  // Are the fixtures on the same body?
108  if (bodyA == bodyB)
109  {
110  return;
111  }
112 
113  // TODO_ERIN use a hash table to remove a potential bottleneck when both
114  // bodies have a lot of contacts.
115  // Does a contact already exist?
116  ContactEdge edge = bodyB.ContactList;
117  while (edge != null)
118  {
119  if (edge.Other == bodyA)
120  {
121  Fixture fA = edge.Contact.FixtureA;
122  Fixture fB = edge.Contact.FixtureB;
123  int iA = edge.Contact.ChildIndexA;
124  int iB = edge.Contact.ChildIndexB;
125 
126  if (fA == fixtureA && fB == fixtureB && iA == indexA && iB == indexB)
127  {
128  // A contact already exists.
129  return;
130  }
131 
132  if (fA == fixtureB && fB == fixtureA && iA == indexB && iB == indexA)
133  {
134  // A contact already exists.
135  return;
136  }
137  }
138 
139  edge = edge.Next;
140  }
141 
142  // Does a joint override collision? Is at least one body dynamic?
143  if (!bodyB.ShouldCollide(bodyA))
144  {
145  return;
146  }
147 
148  //Check default filter
149  if (!ShouldCollide(fixtureA, fixtureB))
150  {
151  return;
152  }
153 
154  // Check user filtering.
155  if (ContactFilter != null && !ContactFilter(fixtureA, fixtureB))
156  {
157  return;
158  }
159 
160  //Velcro: BeforeCollision delegate
161  if (fixtureA.BeforeCollision != null && !fixtureA.BeforeCollision(fixtureA, fixtureB))
162  {
163  return;
164  }
165 
166  if (fixtureB.BeforeCollision != null && !fixtureB.BeforeCollision(fixtureB, fixtureA))
167  {
168  return;
169  }
170 
171  // Call the factory.
172  Contact c = Contact.Create(fixtureA, indexA, fixtureB, indexB);
173  if (c == null)
174  {
175  return;
176  }
177 
178  // Contact creation may swap fixtures.
179  fixtureA = c.FixtureA;
180  fixtureB = c.FixtureB;
181  indexA = c.ChildIndexA;
182  indexB = c.ChildIndexB;
183  bodyA = fixtureA.Body;
184  bodyB = fixtureB.Body;
185 
186  // Insert into the world.
187  c.Previous = null;
188  c.Next = ContactList;
189  if (ContactList != null)
190  {
191  ContactList.Previous = c;
192  }
193 
194  ContactList = c;
195 
196  // Connect to island graph.
197 
198  // Connect to body A
199  c.NodeA.Contact = c;
200  c.NodeA.Other = bodyB;
201 
202  c.NodeA.Prev = null;
203  c.NodeA.Next = bodyA.ContactList;
204  if (bodyA.ContactList != null)
205  {
206  bodyA.ContactList.Prev = c.NodeA;
207  }
208 
209  bodyA.ContactList = c.NodeA;
210 
211  // Connect to body B
212  c.NodeB.Contact = c;
213  c.NodeB.Other = bodyA;
214 
215  c.NodeB.Prev = null;
216  c.NodeB.Next = bodyB.ContactList;
217  if (bodyB.ContactList != null)
218  {
219  bodyB.ContactList.Prev = c.NodeB;
220  }
221 
222  bodyB.ContactList = c.NodeB;
223  ++ContactCounter;
224  }

238  internal void Remove(Contact c)
239  {
240  if (c.FixtureA == null || c.FixtureB == null)
241  {
242  return;
243  }
244 
245  Fixture fixtureA = c.FixtureA;
246  Fixture fixtureB = c.FixtureB;
247 
248  //Velcro: When contacts are removed, we invoke OnSeparation
249  if (c.IsTouching)
250  {
251  //Report the separation to both participants:
252  fixtureA.OnSeparation?.Invoke(fixtureA, fixtureB, c);
253 
254  //Reverse the order of the reported fixtures. The first fixture is always the one that the user subscribed to.
255  fixtureB.OnSeparation?.Invoke(fixtureB, fixtureA, c);
256 
257  //The generic handler
258  EndContact?.Invoke(c);
259  }
260 
261  Body bodyA = fixtureA.Bodyprivate;
262  Body bodyB = fixtureB.Bodyprivate;
263 
264  // Remove from the world.
265  if (c.Previous != null)
266  {
267  c.Previous.Next = c.Next;
268  }
269 
270  if (c.Next != null)
271  {
272  c.Next.Previous = c.Previous;
273  }
274 
275  if (c == ContactList)
276  {
277  ContactList = c.Next;
278  }
279 
280  // Remove from body 1
281  if (c.NodeA.Prev != null)
282  {
283  c.NodeA.Prev.Next = c.NodeA.Next;
284  }
285 
286  if (c.NodeA.Next != null)
287  {
288  c.NodeA.Next.Prev = c.NodeA.Prev;
289  }
290 
291  if (c.NodeA == bodyA.ContactList)
292  {
293  bodyA.ContactList = c.NodeA.Next;
294  }
295 
296  // Remove from body 2
297  if (c.NodeB.Prev != null)
298  {
299  c.NodeB.Prev.Next = c.NodeB.Next;
300  }
301 
302  if (c.NodeB.Next != null)
303  {
304  c.NodeB.Next.Prev = c.NodeB.Prev;
305  }
306 
307  if (c.NodeB == bodyB.ContactList)
308  {
309  bodyB.ContactList = c.NodeB.Next;
310  }
311 
312  // Call the factory.
313  c.Destroy();
314  --ContactCounter;
315  }

321  internal void Collide()
322  {
323  // Update awake contacts.
324 
325  Contact c = ContactList;
326 
327  while (c != null)
328  {
329  Fixture fixtureA = c.FixtureA;
330  Fixture fixtureB = c.FixtureB;
331  int indexA = c.ChildIndexA;
332  int indexB = c.ChildIndexB;
333  Body bodyA = fixtureA.Bodyprivate;
334  Body bodyB = fixtureB.Bodyprivate;
335 
336  //Velcro: Do no try to collide disabled bodies
337  if (!bodyA.Enabled || !bodyB.Enabled)
338  {
339  c = c.Next;
340  continue;
341  }
342 
343  // Is this contact flagged for filtering?
344  if (c.FilterFlag)
345  {
346  // Should these bodies collide?
347  if (!bodyB.ShouldCollide(bodyA))
348  {
349  Contact cNuke = c;
350  c = cNuke.Next;
351  Remove(cNuke);
352  continue;
353  }
354 
355  // Check default filtering
356  if (!ShouldCollide(fixtureA, fixtureB))
357  {
358  Contact cNuke = c;
359  c = cNuke.Next;
360  Remove(cNuke);
361  continue;
362  }
363 
364  // Check user filtering.
365  if (ContactFilter != null && !ContactFilter(fixtureA, fixtureB))
366  {
367  Contact cNuke = c;
368  c = cNuke.Next;
369  Remove(cNuke);
370  continue;
371  }
372 
373  // Clear the filtering flag.
374  c.Flags &= ~ContactFlags.FilterFlag;
375  }
376 
377  bool activeA = bodyA.Awake && bodyA.BodyType != BodyType.Static;
378  bool activeB = bodyB.Awake && bodyB.BodyType != BodyType.Static;
379 
380  // At least one body must be awake and it must be dynamic or kinematic.
381  if (!activeA && !activeB)
382  {
383  c = c.Next;
384  continue;
385  }
386 
387  int proxyIdA = fixtureA.Proxies[indexA].ProxyId;
388  int proxyIdB = fixtureB.Proxies[indexB].ProxyId;
389  bool overlap = BroadPhase.TestOverlap(proxyIdA, proxyIdB);
390 
391  // Here we destroy contacts that cease to overlap in the broad-phase.
392  if (!overlap)
393  {
394  Contact cNuke = c;
395  c = cNuke.Next;
396  Remove(cNuke);
397  continue;
398  }
399 
400  // The contact persists.
401  c.Update(this);
402  c = c.Next;
403  }
404  }

412  private static bool ShouldCollide(Fixture fixtureA, Fixture fixtureB)
413  {
414  if (fixtureA.CollisionGroup == fixtureB.CollisionGroup &&
415  fixtureA.CollisionGroup != 0)
416  {
417  return fixtureA.CollisionGroup > 0;
418  }
419 
420  bool collide = (fixtureA.CollidesWith & fixtureB.CollisionCategories) != 0 &&
421  (fixtureA.CollisionCategories & fixtureB.CollidesWith) != 0;
422 
423  return collide;
424  }
425  }
