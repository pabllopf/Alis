53  public static List<Vertices> ConvexPartition(Vertices vertices)
54  {
55  Debug.Assert(vertices.Count > 3);
56 
57  Polygon.Polygon poly = new Polygon.Polygon();
58 
59  foreach (Vector2 vertex in vertices)
60  {
61  poly.Points.Add(new TriangulationPoint(vertex.X, vertex.Y));
62  }
63 
64  if (vertices.Holes != null)
65  {
66  foreach (Vertices holeVertices in vertices.Holes)
67  {
68  Polygon.Polygon hole = new Polygon.Polygon();
69 
70  foreach (Vector2 vertex in holeVertices)
71  {
72  hole.Points.Add(new TriangulationPoint(vertex.X, vertex.Y));
73  }
74 
75  poly.AddHole(hole);
76  }
77  }
78 
79  DtSweepContext tcx = new DtSweepContext();
80  tcx.PrepareTriangulation(poly);
81  DtSweep.Triangulate(tcx);
82 
83  List<Vertices> results = new List<Vertices>();
84 
85  foreach (DelaunayTriangle triangle in poly.Triangles)
86  {
87  Vertices v = new Vertices();
88  foreach (TriangulationPoint p in triangle.Points)
89  {
90  v.Add(new Vector2((float) p.X, (float) p.Y));
91  }
92 
93  results.Add(v);
94  }
95 
96  return results;
97  }
98  }
