149  native_type = Util.Allocate(type_byteCount + 1);
150  }
151  else

153  byte* native_type_stackBytes = stackalloc byte[type_byteCount + 1];
154  native_type = native_type_stackBytes;
155  }
156 
157  int native_type_offset = Util.GetUtf8(type, native_type, type_byteCount);
158  native_type[native_type_offset] = 0;
159  }
160  else

163  }
164 
165  byte ret = ImGuiNative.ImGuiPayload_IsDataType(NativePtr, native_type);
