477  List<Vertices> detectedPolygons = new List<Vertices>();
478 
479  Vector2? holeEntrance = null;
480  Vector2? polygonEntrance = null;
481 
482  List<Vector2> blackList = new List<Vector2>();
483 
484  bool searchOn;
485  do
486  {
487  Vertices polygon;
488  if (detectedPolygons.Count == 0)
489  {
490  // First pass / single polygon
491  polygon = new Vertices(CreateSimplePolygon(Vector2.Zero, Vector2.Zero));
492 
493  if (polygon.Count > 2)
494  {
495  polygonEntrance = GetTopMostVertex(polygon);
496  }
497  }
498  else if (polygonEntrance.HasValue)
499  {
500  // Multi pass / multiple polygons
501  polygon = new Vertices(CreateSimplePolygon(polygonEntrance.Value,
502  new Vector2(polygonEntrance.Value.X - 1f, polygonEntrance.Value.Y)));
503  }
504  else
505  {
506  break;
507  }
508 
509  searchOn = false;
510 
511  if (polygon.Count > 2)
512  {
513  if (holeDetection)
514  {
515  do
516  {
517  holeEntrance = SearchHoleEntrance(polygon, holeEntrance);
518 
519  if (holeEntrance.HasValue)
520  {
521  if (!blackList.Contains(holeEntrance.Value))
522  {
523  blackList.Add(holeEntrance.Value);
524  Vertices holePolygon = CreateSimplePolygon(holeEntrance.Value,
525  new Vector2(holeEntrance.Value.X + 1, holeEntrance.Value.Y));
526 
527  if (holePolygon != null && holePolygon.Count > 2)
528  {
529  switch (polygonDetectionType)
530  {
531  case VerticesDetectionType.Integrated:
532 
533  // Add first hole polygon vertex to close the hole polygon.
534  holePolygon.Add(holePolygon[0]);
535 
536  if (SplitPolygonEdge(polygon, holeEntrance.Value, out _,
537  out int vertex2Index))
538  {
539  polygon.InsertRange(vertex2Index, holePolygon);
540  }
541 
542  break;
543 
544  case VerticesDetectionType.Separated:
545  if (polygon.Holes == null)
546  {
547  polygon.Holes = new List<Vertices>();
548  }
549 
550  polygon.Holes.Add(holePolygon);
551  break;
552  }
553  }
554  }
555  else
556  {
557  break;
558  }
559  }
560  else
561  {
562  break;
563  }
564  } while (true);
565  }
566 
567  detectedPolygons.Add(polygon);
568  }
569 
570  if (multipartDetection || polygon.Count <= 2)
571  {
572  if (SearchNextHullEntrance(detectedPolygons, polygonEntrance.Value, out polygonEntrance))
573  {
574  searchOn = true;
575  }
576  }
577  } while (searchOn);
578 
579  if (detectedPolygons.Count == 0)
580  {
581  throw new Exception("Couldn't detect any vertices.");
582  }
583 
584  // Post processing.
585  if (PolygonDetectionType ==
586  VerticesDetectionType.Separated) // Only when VerticesDetectionType.Separated? -> Recheck.
587  {
588  ApplyTriangulationCompatibleWinding(ref detectedPolygons);
589  }
590 
591  if (transform != Matrix4x4.Identity)
592  {
593  ApplyTransform(ref detectedPolygons);
594  }
595 
596  return detectedPolygons;
597  }

638  private Vector2? SearchHoleEntrance(Vertices polygon, Vector2? lastHoleEntrance)
639  {
640  if (polygon == null)
641  {
642  throw new ArgumentNullException("'polygon' can't be null.");
643  }
644 
645  if (polygon.Count < 3)
646  {
647  throw new ArgumentException("'polygon.MainPolygon.Count' can't be less then 3.");
648  }
649 
650  List<float> xCoords;
651  Vector2? entrance;
652 
653  int startY;
654 
655  int lastSolid = 0;
656  bool foundSolid;
657  bool foundTransparent;
658 
659  // Set start y coordinate.
660  if (lastHoleEntrance.HasValue)
661  {
662  // We need the y coordinate only.
663  startY = (int) lastHoleEntrance.Value.Y;
664  }
665  else
666  {
667  // Start from the top of the polygon if last entrance == null.
668  startY = (int) GetTopMostCoord(polygon);
669  }
670 
671  // Set the end y coordinate.
672  int endY = (int) GetBottomMostCoord(polygon);
673 
674  if (startY > 0 && startY < height && endY > 0 && endY < height)
675  {
676  // go from top to bottom of the polygon
677  for (int y = startY; y <= endY; y++)
678  {
679  // get x-coord of every polygon edge which crosses y
680  xCoords = SearchCrossingEdges(polygon, y);
681 
682  // We need an even number of crossing edges.
683  // It's always a pair of start and end edge: nothing | polygon | hole | polygon | nothing ...
684  // If it's not then don't bother, it's probably a peak ...
685  // ...which should be filtered out by SearchCrossingEdges() anyway.
686  if (xCoords.Count > 1 && xCoords.Count % 2 == 0)
687  {
688  // Ok, this is short, but probably a little bit confusing.
689  // This part searches from left to right between the edges inside the polygon.
690  // The problem: We are using the polygon data to search in the texture data.
691  // That's simply not accurate, but necessary because of performance.
692  for (int i = 0; i < xCoords.Count; i += 2)
693  {
694  foundSolid = false;
695  foundTransparent = false;
696 
697  // We search between the edges inside the polygon.
698  for (int x = (int) xCoords[i]; x <= (int) xCoords[i + 1]; x++)
699  {
700  // First pass: IsSolid might return false.
701  // In that case the polygon edge doesn't lie on the texture's solid pixel, because of the hull tolerance.
702  // If the edge lies before the first solid pixel then we need to skip our transparent pixel finds.
703 
704  // The algorithm starts to search for a relevant transparent pixel (which indicates a possible hole)
705  // after it has found a solid pixel.
706 
707  // After we've found a solid and a transparent pixel (a hole's left edge)
708  // we search for a solid pixel again (a hole's right edge).
709  // When found the distance of that coordinate has to be greater then the hull tolerance.
710 
711  if (IsSolid(ref x, ref y))
712  {
713  if (!foundTransparent)
714  {
715  foundSolid = true;
716  lastSolid = x;
717  }
718 
719  if (foundSolid && foundTransparent)
720  {
721  entrance = new Vector2(lastSolid, y);
722 
723  if (DistanceToHullAcceptable(polygon, entrance.Value, true))
724  {
725  return entrance;
726  }
727 
728  entrance = null;
729  break;
730  }
731  }
732  else
733  {
734  if (foundSolid)
735  {
736  foundTransparent = true;
737  }
738  }
739  }
740  }
741  }
742  else
743  {
744  if (xCoords.Count % 2 == 0)
745  {
746  Debug.WriteLine("SearchCrossingEdges() % 2 != 0");
747  }
748  }
749  }
750  }
751 
752  return null;
753  }

769  }
770 
771  if (polygon.Count < 3)
772  {
773  throw new ArgumentException("'polygon.MainPolygon.Count' can't be less then 3.");
774  }
775 
776  // Check the distance to main polygon.
777  if (DistanceToHullAcceptable(polygon, point, higherDetail))
778  {

780  {
781  for (int i = 0; i < polygon.Holes.Count; i++)
782  {

808  private bool DistanceToHullAcceptable(Vertices polygon, Vector2 point, bool higherDetail)
809  {
810  if (polygon == null)
811  {
812  throw new ArgumentNullException(nameof(polygon), "'polygon' can't be null.");
813  }
814 
815  if (polygon.Count < 3)
816  {
817  throw new ArgumentException("'polygon.Count' can't be less then 3.");
818  }
819 
820  Vector2 edgeVertex2 = polygon[polygon.Count - 1];
821  Vector2 edgeVertex1;
822 
823  if (higherDetail)
824  {

826  {

831  {
832  return false;
833  }
834

836  }

839  }

842  {

847  {
848  return false;

988  private List<float> SearchCrossingEdges(Vertices polygon, int y)
989  {
990  // sick-o-note:
991  // Used to search the x coordinates of edges in the polygon for a specific y coordinate.
992  // (Usualy comming from the texture data, that's why it's an int and not a float.)
993 
994  List<float> edges = new List<float>();
995 
996  // current edge
997  Vector2 slope;
998  Vector2 vertex1; // i
999  Vector2 vertex2; // i - 1
1000 
1001  // next edge
1002  Vector2 nextSlope;
1003  Vector2 nextVertex; // i + 1
1004 
1005  bool addFind;
1006 
1007  if (polygon.Count > 2)
1008  {
1009  // There is a gap between the last and the first vertex in the vertex list.
1010  // We will bridge that by setting the last vertex (vertex2) to the last
1011  // vertex in the list.
1012  vertex2 = polygon[polygon.Count - 1];
1013 
1014  // We are moving along the polygon edges.
1015  for (int i = 0; i < polygon.Count; i++)
1016  {
1017  vertex1 = polygon[i];
1018 
1019  // Approx. check if the edge crosses our y coord.
1020  if (vertex1.Y >= y && vertex2.Y <= y ||
1021  vertex1.Y <= y && vertex2.Y >= y)
1022  {
1023  // Ignore edges that are parallel to y.
1024  if (vertex1.Y != vertex2.Y)
1025  {
1026  addFind = true;
1027  slope = vertex2 - vertex1;
1028 
1029  // Special treatment for edges that end at the y coord.

1165  private Vertices CreateSimplePolygon(Vector2 entrance, Vector2 last)
1166  {
1167  bool entranceFound = false;
1168  bool endOfHull = false;
1169 
1170  Vertices polygon = new Vertices(32);
1171  Vertices hullArea = new Vertices(32);
1172  Vertices endOfHullArea = new Vertices(32);
1173 
1174  Vector2 current = Vector2.Zero;
1175 
1176  // Get the entrance point.
1177  if (entrance == Vector2.Zero || !InBounds(ref entrance))
1178  {
1179  entranceFound = SearchHullEntrance(out entrance);
1180 
1181  if (entranceFound)
1182  {
1183  current = new Vector2(entrance.X - 1f, entrance.Y);
1184  }
1185  }
1186  else
1187  {
1188  if (IsSolid(ref entrance))
1189  {
1190  if (IsNearPixel(ref entrance, ref last))
1191  {
1192  current = last;
1193  entranceFound = true;
1194  }
1195  else
1196  {
1197  if (SearchNearPixels(false, ref entrance, out Vector2 temp))
1198  {
1199  current = temp;
1200  entranceFound = true;
1201  }
1202  else
1203  {
1204  entranceFound = false;
1205  }
1206  }
1207  }
1208  }
1209 
1210  if (entranceFound)
1211  {
1212  polygon.Add(entrance);
1213  hullArea.Add(entrance);
1214 
1215  Vector2 next = entrance;
1216 
1217  do
1218  {
1219  // Search in the pre vision list for an outstanding point.
1220  if (SearchForOutstandingVertex(hullArea, out Vector2 outstanding))
1221  {
1222  if (endOfHull)
1223  {
1224  // We have found the next pixel, but is it on the last bit of the hull?
1225  if (endOfHullArea.Contains(outstanding))
1226  {
1227  // Indeed.
1228  polygon.Add(outstanding);
1229  }
1230 
1231  // That's enough, quit.
1232  break;
1233  }
1234 
1235  // Add it and remove all vertices that don't matter anymore
1236  // (all the vertices before the outstanding).
1237  polygon.Add(outstanding);
1238  hullArea.RemoveRange(0, hullArea.IndexOf(outstanding));
1239  }
1240 
1241  // Last point gets current and current gets next. Our little spider is moving forward on the hull ;).
1242  last = current;
1243  current = next;
1244 
1245  // Get the next point on hull.
1246  if (GetNextHullPoint(ref last, ref current, out next))
1247  {
1248  // Add the vertex to a hull pre-vision list.
1249  hullArea.Add(next);
1250  }
1251  else
1252  {
1253  // Quit
1254  break;
1255  }
1256 
1257  if (next == entrance && !endOfHull)
1258  {
1259  // It's the last bit of the hull, search on and exit at next found vertex.
1260  endOfHull = true;
1261  endOfHullArea.AddRange(hullArea);
1262 
1263  // We don't want the last vertex to be the same as the first one, because it causes the triangulation code to crash.
1264  if (endOfHullArea.Contains(entrance))
1265  {
1266  endOfHullArea.Remove(entrance);
1267  }
1268  }
1269  } while (true);
1270  }
1271 
1272  return polygon;
1273  }
