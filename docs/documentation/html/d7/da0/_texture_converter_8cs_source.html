384  private List<Vertices> DetectVertices()
385  {
386  if (data == null)
387  {
388  throw new Exception(
389  "'_data' can't be null. You have to use SetTextureData(uint[] data, int width) before calling this method.");
390  }
391 
392  if (data.Length < 4)
393  {
394  throw new Exception(
395  "'_data' length can't be less then 4. Your texture must be at least 2 x 2 pixels in size. " +
396  "You have to use SetTextureData(uint[] data, int width) before calling this method.");
397  }
398 
399  if (width < 2)
400  {
401  throw new Exception(
402  "'_width' can't be less then 2. Your texture must be at least 2 x 2 pixels in size. " +
403  "You have to use SetTextureData(uint[] data, int width) before calling this method.");
404  }
405 
406  if (data.Length % width != 0)
407  {
408  throw new Exception(
409  "'_width' has an invalid value. You have to use SetTextureData(uint[] data, int width) before calling this method.");
410  }

412  List<Vertices> detectedPolygons = new List<Vertices>();
413 
414  Vector2F? holeEntrance = null;
415  Vector2F? polygonEntrance = null;
416 
417  List<Vector2F> blackList = new List<Vector2F>();
418 
419  bool searchOn;
420  do
421  {
422  Vertices polygon;
423  if (detectedPolygons.Count == 0)
424  {
425  // First pass / single polygon
426  polygon = new Vertices(CreateSimplePolygon(Vector2F.Zero, Vector2F.Zero));
427 
428  if (polygon.Count > 2)
429  {
430  polygonEntrance = GetTopMostVertex(polygon);
431  }
432  }
433  else if (polygonEntrance.HasValue)
434  {
435  // Multi pass / multiple polygons
436  polygon = new Vertices(CreateSimplePolygon(polygonEntrance.Value,
437  new Vector2F(polygonEntrance.Value.X - 1f, polygonEntrance.Value.Y)));
438  }
439  else
440  {
441  break;
442  }
443 
444  searchOn = false;
445 
446  if (polygon.Count > 2)
447  {
448  if (holeDetection)
449  {
450  do
451  {
452  holeEntrance = SearchHoleEntrance(polygon, holeEntrance);
453 
454  if (holeEntrance.HasValue)
455  {
456  if (!blackList.Contains(holeEntrance.Value))
457  {
458  blackList.Add(holeEntrance.Value);
459  Vertices holePolygon = CreateSimplePolygon(holeEntrance.Value,
460  new Vector2F(holeEntrance.Value.X + 1, holeEntrance.Value.Y));
461 
462  if ((holePolygon != null) && (holePolygon.Count > 2))
463  {
464  switch (polygonDetectionType)
465  {
466  case VerticesDetectionType.Integrated:
467 
468  // Add first hole polygon vertex to close the hole polygon.
469  holePolygon.Add(holePolygon[0]);
470 
471  if (SplitPolygonEdge(polygon, holeEntrance.Value,
472  out int vertex2Index))
473  {
474  polygon.InsertRange(vertex2Index, holePolygon);
475  }
476 
477  break;
478 
479  case VerticesDetectionType.Separated:
480  polygon.Holes ??= new List<Vertices>();
481 
482  polygon.Holes.Add(holePolygon);
483  break;
484  }
485  }
486  }
487  else
488  {
489  break;
490  }
491  }
492  else
493  {
494  break;
495  }
496  } while (true);
497  }
498 
499  detectedPolygons.Add(polygon);
500  }
501 
502  if (multipartDetection || polygon.Count <= 2)
503  {
504  if ((polygonEntrance != null) && SearchNextHullEntrance(detectedPolygons, polygonEntrance.Value, out polygonEntrance))
505  {
506  searchOn = true;
507  }
508  }
509  } while (searchOn);
510 
511  if (detectedPolygons.Count == 0)
512  {
513  throw new Exception("Couldn't detect any vertices.");
514  }
515 
516  // Post processing.
517  if (PolygonDetectionType ==
518  VerticesDetectionType.Separated) // Only when VerticesDetectionType.Separated? -> Recheck.
519  {
520  ApplyTriangulationCompatibleWinding(ref detectedPolygons);
521  }
522 
523  if (transform != Matrix4X4F.Identity)
524  {
525  ApplyTransform(ref detectedPolygons);
526  }
527 
528  return detectedPolygons;
529  }

570  private Vector2F? SearchHoleEntrance(Vertices polygon, Vector2F? lastHoleEntrance)
571  {
572  if (polygon == null)
573  {
574  throw new ArgumentNullException($"'polygon' can't be null.");
575  }
576 
577  if (polygon.Count < 3)
578  {
579  throw new ArgumentException("'polygon.MainPolygon.Count' can't be less then 3.");
580  }
581 
582  int startY;
583 
584  int lastSolid = 0;
585 
586  // Set start y coordinate.
587  if (lastHoleEntrance.HasValue)
588  {
589  // We need the y coordinate only.
590  startY = (int) lastHoleEntrance.Value.Y;
591  }
592  else
593  {
594  // Start from the top of the polygon if last entrance == null.
595  startY = (int) GetTopMostCoord(polygon);
596  }
597 
598  // Set the end y coordinate.
599  int endY = (int) GetBottomMostCoord(polygon);
600 
601  if ((startY > 0) && (startY < height) && (endY > 0) && (endY < height))
602  {
603  // go from top to bottom of the polygon
604  for (int y = startY; y <= endY; y++)
605  {
606  // get x-coord of every polygon edge which crosses y
607  List<float> xCoords = SearchCrossingEdges(polygon, y);
608 
609  // We need an even number of crossing edges.
610  // It's always a pair of start and end edge: nothing | polygon | hole | polygon | nothing ...
611  // If it's not then don't bother, it's probably a peak ...
612  // ...which should be filtered out by SearchCrossingEdges() anyway.
613  if ((xCoords.Count > 1) && (xCoords.Count % 2 == 0))
614  {
615  // Ok, this is short, but probably a little bit confusing.
616  // This part searches from left to right between the edges inside the polygon.
617  // The problem: We are using the polygon data to search in the texture data.
618  // That's simply not accurate, but necessary because of performance.
619  for (int i = 0; i < xCoords.Count; i += 2)
620  {
621  bool foundSolid = false;
622  bool foundTransparent = false;
623 
624  // We search between the edges inside the polygon.
625  for (int x = (int) xCoords[i]; x <= (int) xCoords[i + 1]; x++)
626  {
627  // First pass: IsSolid might return false.
628  // In that case the polygon edge doesn't lie on the texture's solid pixel, because of the hull tolerance.
629  // If the edge lies before the first solid pixel then we need to skip our transparent pixel finds.
630 
631  // The algorithm starts to search for a relevant transparent pixel (which indicates a possible hole)
632  // after it has found a solid pixel.
633 
634  // After we've found a solid and a transparent pixel (a hole's left edge)
635  // we search for a solid pixel again (a hole's right edge).
636  // When found the distance of that coordinate has to be greater then the hull tolerance.
637 
638  if (IsSolid(ref x, ref y))
639  {
640  if (!foundTransparent)
641  {
642  foundSolid = true;
643  lastSolid = x;
644  }
645 
646  if (foundTransparent)
647  {
648  Vector2F? entrance = new Vector2F(lastSolid, y);
649 
650  if (DistanceToHullAcceptable(polygon, entrance.Value, true))
651  {
652  return entrance;
653  }
654 
655  break;
656  }
657  }
658  else
659  {
660  if (foundSolid)
661  {
662  foundTransparent = true;
663  }
664  }
665  }
666  }
667  }
668  else
669  {
670  if (xCoords.Count % 2 == 0)
671  {
672  Debug.WriteLine("SearchCrossingEdges() % 2 != 0");
673  }
674  }
675  }
676  }
677 
678  return null;
679  }

696

698  {
699  throw new ArgumentException("'polygon.MainPolygon.Count' can't be less then 3.");
700  }
701 
702  // Check the distance to main polygon.
703  if (DistanceToHullAcceptable(polygon, point, higherDetail))
704  {

706  {
707  for (int i = 0; i < polygon.Holes.Count; i++)

735  {

737  {
738  throw new ArgumentNullException(nameof(polygon), "'polygon' can't be null.");
739  }
740 
741  if (polygon.Count < 3)
742  {
743  throw new ArgumentException("'polygon.Count' can't be less then 3.");
744  }
745 
746  Vector2F edgeVertex2 = polygon[^1];
747  Vector2F edgeVertex1;
748 
749  if (higherDetail)
750  {

752  {

757  {
758  return false;
759  }
760 
761  edgeVertex2 = polygon[i];
762  }
763 
764  return true;
765  }
766 
767  for (int i = 0; i < polygon.Count; i++)
768  {
769  edgeVertex1 = polygon[i];

773  {

914  private List<float> SearchCrossingEdges(Vertices polygon, int y)
915  {
916  List<float> edges = new List<float>();
917 
918  if (polygon.Count > 2)
919  {
920  // There is a gap between the last and the first vertex in the vertex list.
921  // We will bridge that by setting the last vertex (vertex2) to the last
922  // vertex in the list.
923  Vector2F vertex2 = polygon[^1]; // i - 1
924 
925  // We are moving along the polygon edges.
926  for (int i = 0; i < polygon.Count; i++)
927  {
928  Vector2F vertex1 = polygon[i]; // i
929 
930  // Approx. check if the edge crosses our y coord.
931  if (((vertex1.Y >= y) && (vertex2.Y <= y)) ||
932  ((vertex1.Y <= y) && (vertex2.Y >= y)))
933  {
934  // Ignore edges that are parallel to y.
935  if (Math.Abs(vertex1.Y - vertex2.Y) > 0.0001f)
936  {
937  bool addFind = true;
938  Vector2F slope = vertex2 - vertex1;
939 
940  // Special treatment for edges that end at the y coord.
941  if (Math.Abs(vertex1.Y - y) < 0.0001f)
942  {
943  // Create preview of the next edge.
944  Vector2F nextVertex = polygon[(i + 1) % polygon.Count]; // i + 1
945  Vector2F nextSlope = vertex1 - nextVertex;
946 
947  // Ignore peaks.
948  // If two edges are aligned like this: /\ and the y coordinate lies on the top,
949  // then we get the same x coord twice and we don't need that.
950  if (slope.Y > 0)
951  {
952  addFind = nextSlope.Y <= 0;
953  }
954  else
955  {
956  addFind = nextSlope.Y >= 0;
957  }
958  }
959 
960  if (addFind)
961  {
962  edges.Add((y - vertex1.Y) / slope.Y * slope.X +
963  vertex1.X); // Calculate and add the x coord.
964  }
965  }
966  }
967 
968  // vertex1 becomes vertex2 :).
969  vertex2 = vertex1;
970  }
971  }
972 
973  edges.Sort();
974  return edges;
975  }

984  private bool SplitPolygonEdge(Vertices polygon, Vector2F coordInsideThePolygon, out int vertex2Index)
985  {
986  int nearestEdgeVertex1Index = 0;
987  int nearestEdgeVertex2Index = 0;
988  bool edgeFound = false;
989 
990  float shortestDistance = float.MaxValue;
991 
992  bool edgeCoordFound = false;
993  Vector2F foundEdgeCoord = Vector2F.Zero;
994 
995  List<float> xCoords = SearchCrossingEdges(polygon, (int) coordInsideThePolygon.Y);
996

1071  {

1073  bool endOfHull = false;

1078 
1079  Vector2F current = Vector2F.Zero;
1080

1085 
1086  if (entranceFound)

1089  }
1090  }
1091  else

1094  {

1097  current = last;
1098  entranceFound = true;
1099  }
1100  else

1103  {
1104  current = temp;
1105  entranceFound = true;
1106  }
1107  }
1108  }
1109  }
1110 
1111  if (entranceFound)
1112  {
1113  polygon.Add(entrance);
1114  hullArea.Add(entrance);
1115

1117

1120  // Search in the pre vision list for an outstanding point.
1121  if (SearchForOutstandingVertex(hullArea, out Vector2F outstanding))
1122  {
1123  if (endOfHull)
1124  {
1125  // We have found the next pixel, but is it on the last bit of the hull?
1126  if (endOfHullArea.Contains(outstanding))
1127  {
1128  // Indeed.
1129  polygon.Add(outstanding);
1130  }
1131 
1132  // That's enough, quit.
1133  break;
1134  }
1135 
1136  // Add it and remove all vertices that don't matter anymore
1137  // (all the vertices before the outstanding).
1138  polygon.Add(outstanding);
1139  hullArea.RemoveRange(0, hullArea.IndexOf(outstanding));
1140  }
1141 
1142  // Last point gets current and current gets next. Our little spider is moving forward on the hull ;).
1143  last = current;
1144  current = next;
1145 
1146  // Get the next point on hull.
1147  if (GetNextHullPoint(ref last, ref current, out next))

1149  // Add the vertex to a hull pre-vision list.
1150  hullArea.Add(next);
1151  }
1152  else
1153  {
1154  // Quit
1155  break;
1156  }
1157 
1158  if ((next == entrance) && !endOfHull)
1159  {
1160  // It's the last bit of the hull, search on and exit at next found vertex.
1161  endOfHull = true;
1162  endOfHullArea.AddRange(hullArea);
1163 
1164  // We don't want the last vertex to be the same as the first one, because it causes the triangulation code to crash.
1165  if (endOfHullArea.Contains(entrance))
1166  {
1167  endOfHullArea.Remove(entrance);
1168  }
1169  }
1170  } while (true);
1171  }
1172
