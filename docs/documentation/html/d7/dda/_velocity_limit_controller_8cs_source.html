142  public override void Update(float dt)
143  {
144  foreach (Body body in bodies)
145  {
146  if (!IsActiveOn(body))
147  {
148  continue;
149  }
150 
151  if (LimitLinearVelocity)
152  {
153  //Translation
154  // Check for large velocities.
155  float translationX = dt * body.LinearVelocity.X;
156  float translationY = dt * body.LinearVelocity.Y;
157  float result = translationX * translationX + translationY * translationY;
158 
159  if (result > dt * maxLinearSqared)
160  {
161  float sq = (float) Math.Sqrt(result);
162 
163  float ratio = maxLinearVelocity / sq;
164 
165  Vector2 temp = body.LinearVelocity;
166  temp = new Vector2(
167  temp.X * ratio,
168  temp.Y * ratio
169  );
170  body.LinearVelocity = temp;
171  }
172  }
173 
174  if (LimitAngularVelocity)
175  {
176  //Rotation
177  float rotation = dt * body.AngularVelocity;
178  if (rotation * rotation > maxAngularSqared)
179  {
180  float ratio = maxAngularVelocity / Math.Abs(rotation);
181  body.AngularVelocity *= ratio;
182  }
183  }
184  }
