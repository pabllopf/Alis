545  private void CheckSubmittedFences()
546  {
547  lock (_submittedFencesLock)
548  {
549  for (int i = 0; i < _submittedFences.Count; i++)
550  {
551  FenceSubmissionInfo fsi = _submittedFences[i];
552  if (vkGetFenceStatus(_device, fsi.Fence) == VkResult.Success)
553  {
554  CompleteFenceSubmission(fsi);
555  _submittedFences.RemoveAt(i);
556  i -= 1;
557  }
558  else
559  {
560  break; // Submissions are in order; later submissions cannot complete if this one hasn't.
561  }
562  }
563  }

570  private void CompleteFenceSubmission(FenceSubmissionInfo fsi)
571  {
572  Vulkan.VkFence fence = fsi.Fence;
573  VkCommandBuffer completedCB = fsi.CommandBuffer;
574  fsi.CommandList?.CommandBufferCompleted(completedCB);
575  VkResult resetResult = vkResetFences(_device, 1, ref fence);
576  CheckResult(resetResult);
577  ReturnSubmissionFence(fence);
578  lock (_stagingResourcesLock)
579  {
580  if (_submittedStagingTextures.TryGetValue(completedCB, out VkTexture stagingTex))
581  {
582  _submittedStagingTextures.Remove(completedCB);
583  _availableStagingTextures.Add(stagingTex);
584  }
585  if (_submittedStagingBuffers.TryGetValue(completedCB, out VkBuffer stagingBuffer))
586  {
587  _submittedStagingBuffers.Remove(completedCB);
588  if (stagingBuffer.SizeInBytes <= MaxStagingBufferSize)
589  {
590  _availableStagingBuffers.Add(stagingBuffer);
591  }
592  else
593  {
594  stagingBuffer.Dispose();
595  }
596  }
597  if (_submittedSharedCommandPools.TryGetValue(completedCB, out SharedCommandPool sharedPool))
598  {
599  _submittedSharedCommandPools.Remove(completedCB);
600  lock (_graphicsCommandPoolLock)
601  {
602  if (sharedPool.IsCached)
603  {
604  _sharedGraphicsCommandPools.Push(sharedPool);
605  }
606  else
607  {
608  sharedPool.Destroy();
609  }
610  }
611  }
612  }
