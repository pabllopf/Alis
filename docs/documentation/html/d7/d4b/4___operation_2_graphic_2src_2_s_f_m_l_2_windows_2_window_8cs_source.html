466  private void CallEventHandler(Event e)
467  {
468  switch (e.Type)
469  {
470  case EventType.Closed:
471  if (Closed != null)
472  {
473  Closed(this, EventArgs.Empty);
474  }
475 
476  break;
477 
478  case EventType.GainedFocus:
479  if (GainedFocus != null)
480  {
481  GainedFocus(this, EventArgs.Empty);
482  }
483 
484  break;
485 
486  case EventType.JoystickButtonPressed:
487  if (JoystickButtonPressed != null)
488  {
489  JoystickButtonPressed(this, new JoystickButtonEventArgs(e.JoystickButton));
490  }
491 
492  break;
493 
494  case EventType.JoystickButtonReleased:
495  if (JoystickButtonReleased != null)
496  {
497  JoystickButtonReleased(this, new JoystickButtonEventArgs(e.JoystickButton));
498  }
499 
500  break;
501 
502  case EventType.JoystickMoved:
503  if (JoystickMoved != null)
504  {
505  JoystickMoved(this, new JoystickMoveEventArgs(e.JoystickMove));
506  }
507 
508  break;
509 
510  case EventType.JoystickConnected:
511  if (JoystickConnected != null)
512  {
513  JoystickConnected(this, new JoystickConnectEventArgs(e.JoystickConnect));
514  }
515 
516  break;
517 
518  case EventType.JoystickDisconnected:
519  if (JoystickDisconnected != null)
520  {
521  JoystickDisconnected(this, new JoystickConnectEventArgs(e.JoystickConnect));
522  }
523 
524  break;
525 
526  case EventType.KeyPressed:
527  if (KeyPressed != null)
528  {
529  KeyPressed(this, new KeyEventArgs(e.Key));
530  }
531 
532  break;
533 
534  case EventType.KeyReleased:
535  if (KeyReleased != null)
536  {
537  KeyReleased(this, new KeyEventArgs(e.Key));
538  }
539 
540  break;
541 
542  case EventType.LostFocus:
543  if (LostFocus != null)
544  {
545  LostFocus(this, EventArgs.Empty);
546  }
547 
548  break;
549 
550  case EventType.MouseButtonPressed:
551  if (MouseButtonPressed != null)
552  {
553  MouseButtonPressed(this, new MouseButtonEventArgs(e.MouseButton));
554  }
555 
556  break;
557 
558  case EventType.MouseButtonReleased:
559  if (MouseButtonReleased != null)
560  {
561  MouseButtonReleased(this, new MouseButtonEventArgs(e.MouseButton));
562  }
563 
564  break;
565 
566  case EventType.MouseEntered:
567  if (MouseEntered != null)
568  {
569  MouseEntered(this, EventArgs.Empty);
570  }
571 
572  break;
573 
574  case EventType.MouseLeft:
575  if (MouseLeft != null)
576  {
577  MouseLeft(this, EventArgs.Empty);
578  }
579 
580  break;
581 
582  case EventType.MouseMoved:
583  if (MouseMoved != null)
584  {
585  MouseMoved(this, new MouseMoveEventArgs(e.MouseMove));
586  }
587 
588  break;
589 
590  case EventType.MouseWheelScrolled:
591  if (MouseWheelScrolled != null)
592  {
593  MouseWheelScrolled(this, new MouseWheelScrollEventArgs(e.MouseWheelScroll));
594  }
595 
596  break;
597 
598  case EventType.Resized:
599  if (Resized != null)

601  Resized(this, new SizeEventArgs(e.Size));
602  }
603 
604  break;
605 
606  case EventType.TextEntered:

619
