477  private void CallEventHandler(Event e)
478  {
479  switch (e.Type)
480  {
481  case EventType.Closed:
482  if (Closed != null)
483  {
484  Closed(this, EventArgs.Empty);
485  }
486 
487  break;
488 
489  case EventType.GainedFocus:
490  if (GainedFocus != null)
491  {
492  GainedFocus(this, EventArgs.Empty);
493  }
494 
495  break;
496 
497  case EventType.JoystickButtonPressed:
498  if (JoystickButtonPressed != null)
499  {
500  JoystickButtonPressed(this, new JoystickButtonEventArgs(e.JoystickButton));
501  }
502 
503  break;
504 
505  case EventType.JoystickButtonReleased:
506  if (JoystickButtonReleased != null)
507  {
508  JoystickButtonReleased(this, new JoystickButtonEventArgs(e.JoystickButton));
509  }
510 
511  break;
512 
513  case EventType.JoystickMoved:
514  if (JoystickMoved != null)
515  {
516  JoystickMoved(this, new JoystickMoveEventArgs(e.JoystickMove));
517  }
518 
519  break;
520 
521  case EventType.JoystickConnected:
522  if (JoystickConnected != null)
523  {
524  JoystickConnected(this, new JoystickConnectEventArgs(e.JoystickConnect));
525  }
526 
527  break;
528 
529  case EventType.JoystickDisconnected:
530  if (JoystickDisconnected != null)
531  {
532  JoystickDisconnected(this, new JoystickConnectEventArgs(e.JoystickConnect));
533  }
534 
535  break;
536 
537  case EventType.KeyPressed:
538  if (KeyPressed != null)
539  {
540  KeyPressed(this, new KeyEventArgs(e.Key));
541  }
542 
543  break;
544 
545  case EventType.KeyReleased:
546  if (KeyReleased != null)
547  {
548  KeyReleased(this, new KeyEventArgs(e.Key));
549  }
550 
551  break;
552 
553  case EventType.LostFocus:
554  if (LostFocus != null)
555  {
556  LostFocus(this, EventArgs.Empty);
557  }
558 
559  break;
560 
561  case EventType.MouseButtonPressed:
562  if (MouseButtonPressed != null)
563  {
564  MouseButtonPressed(this, new MouseButtonEventArgs(e.MouseButton));
565  }
566 
567  break;
568 
569  case EventType.MouseButtonReleased:
570  if (MouseButtonReleased != null)
571  {
572  MouseButtonReleased(this, new MouseButtonEventArgs(e.MouseButton));
573  }
574 
575  break;
576 
577  case EventType.MouseEntered:
578  if (MouseEntered != null)
579  {
580  MouseEntered(this, EventArgs.Empty);
581  }
582 
583  break;
584 
585  case EventType.MouseLeft:
586  if (MouseLeft != null)
587  {
588  MouseLeft(this, EventArgs.Empty);
589  }
590 
591  break;
592 
593  case EventType.MouseMoved:
594  if (MouseMoved != null)
595  {
596  MouseMoved(this, new MouseMoveEventArgs(e.MouseMove));
597  }
598 
599  break;
600 
601  case EventType.MouseWheelScrolled:
602  if (MouseWheelScrolled != null)
603  {
604  MouseWheelScrolled(this, new MouseWheelScrollEventArgs(e.MouseWheelScroll));
605  }
606 
607  break;
608 
609  case EventType.Resized:
610  if (Resized != null)
611  {
612  Resized(this, new SizeEventArgs(e.Size));
613  }
614 
615  break;
616 
617  case EventType.TextEntered:
618  if (TextEntered != null)
619  {
620  TextEntered(this, new TextEventArgs(e.Text));
621  }
622 
623  break;
624

627  {
628  TouchBegan(this, new TouchEventArgs(e.Touch));
629  }
630 
631  break;

633  case EventType.TouchMoved:

636  TouchMoved(this, new TouchEventArgs(e.Touch));

644  TouchEnded(this, new TouchEventArgs(e.Touch));
645  }
646 
647  break;

649  case EventType.SensorChanged:

651  {
652  SensorChanged(this, new SensorEventArgs(e.Sensor));
653  }
654 
655  break;
