78  private static List<Vertices> TriangulatePolygon(Vertices vertices, float tolerance)
79  {
80  //Velcro note: Check is needed as invalid triangles can be returned in recursive calls.
81  if (vertices.Count < 3)
82  {
83  return new List<Vertices>();
84  }
85 
86  List<Vertices> results = new List<Vertices>();
87 
88  //Recurse and split on pinch points
89  Vertices pin = new Vertices(vertices);
90  if (ResolvePinchPoint(pin, out Vertices pA, out Vertices pB, tolerance))
91  {
92  List<Vertices> mergeA = TriangulatePolygon(pA, tolerance);
93  List<Vertices> mergeB = TriangulatePolygon(pB, tolerance);
94 
95  if (mergeA.Count == 0 || mergeB.Count == 0)
96  {
97  throw new Exception("Can't triangulate your polygon.");
98  }
99 
100  for (int i = 0; i < mergeA.Count; ++i)
101  {
102  results.Add(new Vertices(mergeA[i]));
103  }
104 
105  for (int i = 0; i < mergeB.Count; ++i)
106  {
107  results.Add(new Vertices(mergeB[i]));
108  }
109 
110  return results;
111  }
112 
113  Vertices[] buffer = new Vertices[vertices.Count - 2];
114  int bufferSize = 0;
115  float[] xrem = new float[vertices.Count];
116  float[] yrem = new float[vertices.Count];
117  for (int i = 0; i < vertices.Count; ++i)
118  {
119  xrem[i] = vertices[i].X;
120  yrem[i] = vertices[i].Y;
121  }
122 
123  int vNum = vertices.Count;
124 
125  while (vNum > 3)
126  {
127  // Find an ear
128  int earIndex = -1;
129  float earMaxMinCross = -10.0f;
130  for (int i = 0; i < vNum; ++i)
131  {
132  if (IsEar(i, xrem, yrem, vNum))
133  {
134  int lower = Remainder(i - 1, vNum);
135  int upper = Remainder(i + 1, vNum);
136  Vector2 d1 = new Vector2(xrem[upper] - xrem[i], yrem[upper] - yrem[i]);
137  Vector2 d2 = new Vector2(xrem[i] - xrem[lower], yrem[i] - yrem[lower]);
138  Vector2 d3 = new Vector2(xrem[lower] - xrem[upper], yrem[lower] - yrem[upper]);
139 
140  d1 = Vector2.Normalize(d1);
141  d2 = Vector2.Normalize(d2);
142  d3 = Vector2.Normalize(d3);
143  MathUtils.Cross(ref d1, ref d2, out float cross12);
144  cross12 = Math.Abs(cross12);
145 
146  MathUtils.Cross(ref d2, ref d3, out float cross23);
147  cross23 = Math.Abs(cross23);
148 
149  MathUtils.Cross(ref d3, ref d1, out float cross31);
150  cross31 = Math.Abs(cross31);
151 
152  //Find the maximum minimum angle
153  float minCross = Math.Min(cross12, Math.Min(cross23, cross31));
154  if (minCross > earMaxMinCross)
155  {
156  earIndex = i;
157  earMaxMinCross = minCross;
158  }
159  }
160  }
161 
162  // If we still haven't found an ear, we're screwed.
163  // Note: sometimes this is happening because the
164  // remaining points are collinear. Really these
165  // should just be thrown out without halting triangulation.
166  if (earIndex == -1)
167  {
168  for (int i = 0; i < bufferSize; i++)
169  {
170  results.Add(buffer[i]);
171  }
172 
173  return results;
174  }
175 
176  // Clip off the ear:
177  // - remove the ear tip from the list
178 
179  --vNum;
180  float[] newx = new float[vNum];
181  float[] newy = new float[vNum];
182  int currDest = 0;
183  for (int i = 0; i < vNum; ++i)
184  {
185  if (currDest == earIndex)
186  {
187  ++currDest;
188  }
189 
190  newx[i] = xrem[currDest];
191  newy[i] = yrem[currDest];
192  ++currDest;
193  }
194 
195  // - add the clipped triangle to the triangle list
196  int under = earIndex == 0 ? vNum : earIndex - 1;
197  int over = earIndex == vNum ? 0 : earIndex + 1;
198  Triangle toAdd = new Triangle(xrem[earIndex], yrem[earIndex], xrem[over], yrem[over], xrem[under],
199  yrem[under]);
200  buffer[bufferSize] = toAdd;
201  ++bufferSize;
202 
203  // - replace the old list with the new one
204  xrem = newx;
205  yrem = newy;
206  }
207 
208  Triangle tooAdd = new Triangle(xrem[1], yrem[1], xrem[2], yrem[2], xrem[0], yrem[0]);
209  buffer[bufferSize] = tooAdd;
210  ++bufferSize;
211 
212  for (int i = 0; i < bufferSize; i++)
213  {
214  results.Add(new Vertices(buffer[i]));
215  }
216 
217  return results;
218  }

230  private static bool ResolvePinchPoint(Vertices pin, out Vertices poutA, out Vertices poutB, float tolerance)
231  {
232  poutA = new Vertices();
233  poutB = new Vertices();
234 
235  if (pin.Count < 3)
236  {
237  return false;
238  }
239 
240  bool hasPinchPoint = false;
241  int pinchIndexA = -1;
242  int pinchIndexB = -1;
243  for (int i = 0; i < pin.Count; ++i)
244  {
245  for (int j = i + 1; j < pin.Count; ++j)
246  {
247  //Don't worry about pinch points where the points
248  //are actually just dupe neighbors
249  if ((Math.Abs(pin[i].X - pin[j].X) < tolerance) && (Math.Abs(pin[i].Y - pin[j].Y) < tolerance) &&
250  (j != i + 1))
251  {
252  pinchIndexA = i;
253  pinchIndexB = j;
254  hasPinchPoint = true;
255  break;
256  }
257  }
258 
259  if (hasPinchPoint)
260  {
261  break;
262  }
263  }
264 
265  if (hasPinchPoint)
266  {
267  int sizeA = pinchIndexB - pinchIndexA;
268  if (sizeA == pin.Count)
269  {
270  return false; //has dupe points at wraparound, not a problem here
271  }
272 
273  for (int i = 0; i < sizeA; ++i)
274  {
275  int ind = Remainder(pinchIndexA + i, pin.Count); // is this right
276  poutA.Add(pin[ind]);
277  }
278 
279  int sizeB = pin.Count - sizeA;
280  for (int i = 0; i < sizeB; ++i)
281  {
282  int ind = Remainder(pinchIndexB + i, pin.Count); // is this right
283  poutB.Add(pin[ind]);
284  }
285  }
286 
287  return hasPinchPoint;
288  }

294  private static int Remainder(int x, int modulus)
295  {
296  int rem = x % modulus;
297  while (rem < 0)
298  {
299  rem += modulus;
300  }
301 
302  return rem;

312  private static bool IsEar(int i, float[] xv, float[] yv, int xvLength)
313  {
314  float dx0, dy0, dx1, dy1;
315  if (i >= xvLength || i < 0 || xvLength < 3)
316  {
317  return false;
318  }
319 
320  int upper = i + 1;
321  int lower = i - 1;
322  if (i == 0)
323  {
324  dx0 = xv[0] - xv[xvLength - 1];
325  dy0 = yv[0] - yv[xvLength - 1];
326  dx1 = xv[1] - xv[0];
327  dy1 = yv[1] - yv[0];
328  lower = xvLength - 1;
329  }
330  else if (i == xvLength - 1)
331  {
332  dx0 = xv[i] - xv[i - 1];
333  dy0 = yv[i] - yv[i - 1];
334  dx1 = xv[0] - xv[i];
335  dy1 = yv[0] - yv[i];
336  upper = 0;
337  }
338  else
339  {
340  dx0 = xv[i] - xv[i - 1];
341  dy0 = yv[i] - yv[i - 1];
342  dx1 = xv[i + 1] - xv[i];
343  dy1 = yv[i + 1] - yv[i];
344  }
345 
346  float cross = dx0 * dy1 - dx1 * dy0;
347 
348  if (cross > 0)
349  {
350  return false;
351  }
352 
353  Triangle myTri = new Triangle(xv[i], yv[i], xv[upper], yv[upper], xv[lower], yv[lower]);
354 
355  for (int j = 0; j < xvLength; ++j)
356  {
357  if (j == i || j == lower || j == upper)
358  {
359  continue;
360  }
361 
362  if (myTri.IsInside(xv[j], yv[j]))
363  {
364  return false;
365  }
366  }
367 
368  return true;
369  }
370  }
