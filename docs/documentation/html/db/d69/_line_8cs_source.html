50  public static float DistanceBetweenPointAndLineSegment(ref Vector2F point, ref Vector2F start, ref Vector2F end)
51  {
52  if (start == end)
53  {
54  return Vector2F.Distance(point, start);
55  }
56 
57  Vector2F v = Vector2F.Subtract(end, start);
58  Vector2F w = Vector2F.Subtract(point, start);
59 
60  float c1 = Vector2F.Dot(w, v);
61  if (c1 <= 0)
62  {
63  return Vector2F.Distance(point, start);
64  }
65 
66  float c2 = Vector2F.Dot(v, v);
67  if (c2 <= c1)
68  {
69  return Vector2F.Distance(point, end);
70  }
71 
72  float b = c1 / c2;
73  Vector2F pointOnLine = Vector2F.Add(start, Vector2F.Multiply(v, b));
74  return Vector2F.Distance(point, pointOnLine);
75  }

82  public static bool LineIntersect2(ref Vector2F a0, ref Vector2F a1, ref Vector2F b0, ref Vector2F b1,
83  out Vector2F intersectionPoint)
84  {
85  intersectionPoint = Vector2F.Zero;
86 
87  if (a0 == b0 || a0 == b1 || a1 == b0 || a1 == b1)
88  {
89  return false;
90  }
91 
92  float x1 = a0.X;
93  float y1 = a0.Y;
94  float x2 = a1.X;
95  float y2 = a1.Y;
96  float x3 = b0.X;
97  float y3 = b0.Y;
98  float x4 = b1.X;
99  float y4 = b1.Y;
100 
101  //AABB early exit
102  if (Math.Max(x1, x2) < Math.Min(x3, x4) || Math.Max(x3, x4) < Math.Min(x1, x2))
103  {
104  return false;
105  }
106 
107  if (Math.Max(y1, y2) < Math.Min(y3, y4) || Math.Max(y3, y4) < Math.Min(y1, y2))
108  {
109  return false;
110  }
111 
112  float ua = (x4 - x3) * (y1 - y3) - (y4 - y3) * (x1 - x3);
113  float ub = (x2 - x1) * (y1 - y3) - (y2 - y1) * (x1 - x3);
114  float denom = (y4 - y3) * (x2 - x1) - (x4 - x3) * (y2 - y1);
115  if (Math.Abs(denom) < Constant.Epsilon)
116  {
117  //Lines are too close to parallel to call
118  return false;
119  }
120 
121  ua /= denom;
122  ub /= denom;
123 
124  if ((0 < ua) && (ua < 1) && (0 < ub) && (ub < 1))
125  {
126  intersectionPoint.X = x1 + ua * (x2 - x1);
127  intersectionPoint.Y = y1 + ua * (y2 - y1);
128  return true;
129  }
130 
131  return false;
132  }

143  public static Vector2F LineIntersect(Vector2F p1, Vector2F p2, Vector2F q1, Vector2F q2)
144  {
145  Vector2F i = Vector2F.Zero;
146  float a1 = p2.Y - p1.Y;
147  float b1 = p1.X - p2.X;
148  float c1 = a1 * p1.X + b1 * p1.Y;
149  float a2 = q2.Y - q1.Y;
150  float b2 = q1.X - q2.X;
151  float c2 = a2 * q1.X + b2 * q1.Y;
152  float det = a1 * b2 - a2 * b1;
153 
154  if (!MathUtils.FloatEquals(det, 0))
155  {
156  // lines are not parallel
157  i.X = (b2 * c1 - b1 * c2) / det;
158  i.Y = (a1 * c2 - a2 * c1) / det;
159  }
160 
161  return i;
162  }

183  public static bool LineIntersect(ref Vector2F point1, ref Vector2F point2, ref Vector2F point3, ref Vector2F point4,
184  bool firstIsSegment, bool secondIsSegment, out Vector2F point)
185  {
186  point = new Vector2F();
187 
188  // these are reused later.
189  // each lettered sub-calculation is used twice, except
190  // for b and d, which are used 3 times
191  float a = point4.Y - point3.Y;
192  float b = point2.X - point1.X;
193  float c = point4.X - point3.X;
194  float d = point2.Y - point1.Y;
195 
196  // denominator to solution of linear system
197  float denom = a * b - c * d;
198 
199  // if denominator is 0, then lines are parallel
200  if (!((denom >= -Constant.Epsilon) && (denom <= Constant.Epsilon)))
201  {
202  float e = point1.Y - point3.Y;
203  float f = point1.X - point3.X;
204  float oneOverDenom = 1.0f / denom;
205 
206  // numerator of first equation
207  float ua = c * e - a * f;
208  ua *= oneOverDenom;
209 
210  // check if intersection point of the two lines is on line segment 1
211  if (!firstIsSegment || ((ua >= 0.0f) && (ua <= 1.0f)))
212  {
213  // numerator of second equation
214  float ub = b * e - d * f;
215  ub *= oneOverDenom;
216 
217  // check if intersection point of the two lines is on line segment 2
218  // means the line segments intersect, since we know it is on
219  // segment 1 as well.
220  if (!secondIsSegment || ((ub >= 0.0f) && (ub <= 1.0f)))
221  {
222  // check if they are coincident (no collision in this case)
223  if (ua != 0f || ub != 0f)
224  {
225  //There is an intersection
226  point.X = point1.X + ua * b;
227  point.Y = point1.Y + ua * d;
228  return true;
229  }
230  }
231  }
232  }
233 
234  return false;
235  }
