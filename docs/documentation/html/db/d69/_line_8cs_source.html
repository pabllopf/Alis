50  public static float DistanceBetweenPointAndLineSegment(ref Vector2 point, ref Vector2 start, ref Vector2 end)
51  {
52  if (start == end)
53  {
54  return Vector2.Distance(point, start);
55  }
56 
57  Vector2 v = Vector2.Subtract(end, start);
58  Vector2 w = Vector2.Subtract(point, start);
59 
60  float c1 = Vector2.Dot(w, v);
61  if (c1 <= 0)
62  {
63  return Vector2.Distance(point, start);
64  }
65 
66  float c2 = Vector2.Dot(v, v);
67  if (c2 <= c1)
68  {
69  return Vector2.Distance(point, end);
70  }
71 
72  float b = c1 / c2;
73  Vector2 pointOnLine = Vector2.Add(start, Vector2.Multiply(v, b));
74  return Vector2.Distance(point, pointOnLine);
75  }

77  // From Eric Jordan's convex decomposition library
82  public static bool LineIntersect2(ref Vector2 a0, ref Vector2 a1, ref Vector2 b0, ref Vector2 b1,
83  out Vector2 intersectionPoint)
84  {
85  intersectionPoint = Vector2.Zero;
86 
87  if (a0 == b0 || a0 == b1 || a1 == b0 || a1 == b1)
88  {
89  return false;
90  }
91 
92  float x1 = a0.X;
93  float y1 = a0.Y;
94  float x2 = a1.X;
95  float y2 = a1.Y;
96  float x3 = b0.X;
97  float y3 = b0.Y;
98  float x4 = b1.X;
99  float y4 = b1.Y;
100 
101  // AABB early exit
102  if (Math.Max(x1, x2) < Math.Min(x3, x4) || Math.Max(x3, x4) < Math.Min(x1, x2))
103  {
104  return false;
105  }
106 
107  if (Math.Max(y1, y2) < Math.Min(y3, y4) || Math.Max(y3, y4) < Math.Min(y1, y2))
108  {
109  return false;
110  }
111 
112  float denom = (y4 - y3) * (x2 - x1) - (x4 - x3) * (y2 - y1);
113  if (Math.Abs(denom) < Constant.Epsilon)
114  {
115  // Lines are too close to parallel to call
116  return false;
117  }
118 
119  float ua = (x4 - x3) * (y1 - y3) - (y4 - y3) * (x1 - x3);
120  float ub = (x2 - x1) * (y1 - y3) - (y2 - y1) * (x1 - x3);
121 
122  ua /= denom;
123  ub /= denom;
124 
125  if (ua >= 0 && ua <= 1 && ub >= 0 && ub <= 1)
126  {
127  intersectionPoint = new Vector2(
128  x1 + ua * (x2 - x1),
129  y1 + ua * (y2 - y1)
130  );
131  return true;
132  }
133 
134  return false;
135  }

147  public static Vector2 LineIntersect(Vector2 p1, Vector2 p2, Vector2 q1, Vector2 q2)
148  {
149  Vector2 i = Vector2.Zero;
150  float a1 = p2.Y - p1.Y;
151  float b1 = p1.X - p2.X;
152  float c1 = a1 * p1.X + b1 * p1.Y;
153  float a2 = q2.Y - q1.Y;
154  float b2 = q1.X - q2.X;
155  float c2 = a2 * q1.X + b2 * q1.Y;
156  float det = a1 * b2 - a2 * b1;
157 
158  if (!MathUtils.FloatEquals(det, 0))
159  {
160  // lines are not parallel
161  i = new Vector2(
162  (b2 * c1 - b1 * c2) / det,
163  (a1 * c2 - a2 * c1) / det
164  );
165  }
166 
167  return i;
168  }

189  public static bool LineIntersect(ref Vector2 point1, ref Vector2 point2, ref Vector2 point3, ref Vector2 point4,
190  bool firstIsSegment, bool secondIsSegment, out Vector2 point)
191  {
192  point = new Vector2();
193 
194  // these are reused later.
195  // each lettered sub-calculation is used twice, except
196  // for b and d, which are used 3 times
197  float a = point4.Y - point3.Y;
198  float b = point2.X - point1.X;
199  float c = point4.X - point3.X;
200  float d = point2.Y - point1.Y;
201 
202  // denominator to solution of linear system
203  float denom = a * b - c * d;
204 
205  // if denominator is 0, then lines are parallel
206  if (!((denom >= -Constant.Epsilon) && (denom <= Constant.Epsilon)))
207  {
208  float e = point1.Y - point3.Y;
209  float f = point1.X - point3.X;
210  float oneOverDenom = 1.0f / denom;
211 
212  // numerator of first equation
213  float ua = c * e - a * f;
214  ua *= oneOverDenom;
215 
216  // check if intersection point of the two lines is on line segment 1
217  if (!firstIsSegment || ((ua >= 0.0f) && (ua <= 1.0f)))
218  {
219  // numerator of second equation
220  float ub = b * e - d * f;
221  ub *= oneOverDenom;
222 
223  // check if intersection point of the two lines is on line segment 2
224  // means the line segments intersect, since we know it is on
225  // segment 1 as well.
226  if (!secondIsSegment || ((ub >= 0.0f) && (ub <= 1.0f)))
227  {
228  // check if they are coincident (no collision in this case)
229  if (ua != 0f || ub != 0f)
230  {
231  //There is an intersection
232  point = new Vector2(
233  point1.X + ua * b,
234  point1.Y + ua * d
235  );
236 
237  return true;
238  }
239  }
240  }
241  }
242 
243  return false;
244  }
