206  public void NewFrame()
207  {
208  ImGui.NewFrame();
209  ImGuiIoPtr io = ImGui.GetIo();
210 
211  // Setup display size (every frame to accommodate for window resizing)
212  SDL_GetWindowSize(_window, out int w, out int h);
213  SDL_GL_GetDrawableSize(_window, out int displayW, out int displayH);
214  io.DisplaySize = new Vector2F(w, h);
215  if ((w > 0) && (h > 0))
216  {
217  io.DisplayFramebufferScale = new Vector2F((float) displayW / w, (float) displayH / h);
218  }
219 
220  // Setup time step (we don't use SDL_GetTicks() because it is using millisecond resolution)
221  ulong frequency = SDL_GetPerformanceFrequency();
222  ulong currentTime = SDL_GetPerformanceCounter();
223  io.DeltaTime = _time > 0 ? (float) ((double) (currentTime - _time) / frequency) : 1.0f / 60.0f;
224  if (io.DeltaTime <= 0)
225  {
226  io.DeltaTime = 0.016f;
227  }
228 
229  _time = currentTime;
230 
231  UpdateMousePosAndButtons();
232  }

238  public void ProcessEvent(SdlEvent evt)
239  {
240  ImGuiIoPtr io = ImGui.GetIo();
241  switch (evt.type)
242  {
243  case SdlEventType.SdlMousewheel:
244  {
245  if (evt.wheel.x > 0)
246  {
247  io.MouseWheelH += 1;
248  }
249 
250  if (evt.wheel.x < 0)
251  {
252  io.MouseWheelH -= 1;
253  }
254 
255  if (evt.wheel.y > 0)
256  {
257  io.MouseWheel += 1;
258  }
259 
260  if (evt.wheel.y < 0)
261  {
262  io.MouseWheel -= 1;
263  }
264 
265  return;
266  }
267  case SdlEventType.SdlMouseButtonDown:
268  {
269  if (evt.button.button == SdlButtonLeft)
270  {
271  _mousePressed[0] = true;
272  }
273 
274  if (evt.button.button == SdlButtonRight)
275  {
276  _mousePressed[1] = true;
277  }
278 
279  if (evt.button.button == SdlButtonMiddle)
280  {
281  _mousePressed[2] = true;
282  }
283 
284  return;
285  }
286  case SdlEventType.SdlTextInput:
287  {
288  string str = Encoding.UTF8.GetString(evt.text.Text);
289  io.AddInputCharactersUtf8(str);
290  return;
291  }
292  case SdlEventType.SdlKeydown:
293  case SdlEventType.SdlKeyup:
294  {
295  SdlScancode key = evt.key.keysym.scancode;
296  io.KeysDown[(int) key] = evt.type == SdlEventType.SdlKeydown;
297  Console.WriteLine("io.KeysDown[" + key + "] = " + evt.type + io.KeysDown[(int) key]);
298  io.KeyShift = (SDL_GetModState() & SdlKeymod.KmodShift) != 0;
299  io.KeyCtrl = (SDL_GetModState() & SdlKeymod.KmodCtrl) != 0;
300  io.KeyAlt = (SDL_GetModState() & SdlKeymod.KmodAlt) != 0;
301  io.KeySuper = (SDL_GetModState() & SdlKeymod.KmodGui) != 0;
302  break;
303  }
304  }
305  }

310  private void UpdateMousePosAndButtons()
311  {
312  ImGuiIoPtr io = ImGui.GetIo();
313 
314  // Set OS mouse position if requested (rarely used, only when ImGuiConfigFlags_NavEnableSetMousePos is enabled by user)
315  if (io.WantSetMousePos)
316  {
317  SDL_WarpMouseInWindow(_window, (int) io.MousePos.X, (int) io.MousePos.Y);
318  }
319  else
320  {
321  io.MousePos = new Vector2F(float.MinValue, float.MinValue);
322  }
323 
324  uint mouseButtons = SDL_GetMouseState(out int mx, out int my);
325  io.MouseDown[0] =
326  _mousePressed[0] ||
327  (mouseButtons & SDL_BUTTON(SdlButtonLeft)) !=
328  0; // If a mouse press event came, always pass it as "mouse held this frame", so we don't miss click-release events that are shorter than 1 frame.
329  io.MouseDown[1] = _mousePressed[1] || (mouseButtons & SDL_BUTTON(SdlButtonRight)) != 0;
330  io.MouseDown[2] = _mousePressed[2] || (mouseButtons & SDL_BUTTON(SdlButtonMiddle)) != 0;
331  _mousePressed[0] = _mousePressed[1] = _mousePressed[2] = false;
332 
333  IntPtr focusedWindow = SDL_GetKeyboardFocus();
334  if (_window == focusedWindow)
335  {
336  // SDL_GetMouseState() gives mouse position seemingly based on the last window entered/focused(?)
337  // The creation of a new windows at runtime and SDL_CaptureMouse both seems to severely mess up with that, so we retrieve that position globally.
338  SDL_GetWindowPosition(focusedWindow, out int wx, out int wy);
339  SDL_GetGlobalMouseState(out mx, out my);
340  mx -= wx;
341  my -= wy;
342  io.MousePos = new Vector2F(mx, my);
343  }
344 
345  // SDL_CaptureMouse() let the OS know e.g. that our imgui drag outside the SDL window boundaries shouldn't e.g. trigger the OS window resize cursor.
346  bool anyMouseButtonDown = ImGui.IsAnyMouseDown();
347  SDL_CaptureMouse(anyMouseButtonDown ? SdlBool.SdlTrue : SdlBool.SdlFalse);
348  }

406  private void SetupRenderState(ImDrawDataPtr drawData, int fbWidth, int fbHeight)
407  {
408  GlEnable(EnableCap.Blend);
409  GlBlendEquation(BlendEquationMode.FuncAdd);
410  GlBlendFunc(BlendingFactorSrc.SrcAlpha, BlendingFactorDest.OneMinusSrcAlpha);
411  GlDisable(EnableCap.CullFace);
412  GlDisable(EnableCap.DepthTest);
413  GlEnable(EnableCap.ScissorTest);
414 
415  GlUseProgram(_shader.ProgramId);
416 
417  float left = drawData.DisplayPos.X;
418  float right = drawData.DisplayPos.X + drawData.DisplaySize.X;
419  float top = drawData.DisplayPos.Y;
420  float bottom = drawData.DisplayPos.Y + drawData.DisplaySize.Y;
421 
422  _shader["Texture"].SetValue(0);
423  _shader["ProjMtx"].SetValue(Matrix4X4F.CreateOrthographicOffCenter(left, right, bottom, top, -1, 1));
424  GlBindSampler(0, 0);
425 
426  GlBindVertexArray(_vertexArrayObject);
427 
428  // Bind vertex/index buffers and setup attributes for ImDrawVert
429  GlBindBuffer(BufferTarget.ArrayBuffer, _vboHandle);
430  GlBindBuffer(BufferTarget.ElementArrayBuffer, _elementsHandle);
431 
432  EnableVertexAttribArray(_shader["Position"].Location);
433  EnableVertexAttribArray(_shader["UV"].Location);
434  EnableVertexAttribArray(_shader["Color"].Location);
435 
436  int drawVertSize = Marshal.SizeOf<ImDrawVert>();
437  VertexAttribPointer(_shader["Position"].Location, 2, VertexAttribPointerType.Float, false, drawVertSize, Marshal.OffsetOf<ImDrawVert>("Pos"));
438  VertexAttribPointer(_shader["UV"].Location, 2, VertexAttribPointerType.Float, false, drawVertSize, Marshal.OffsetOf<ImDrawVert>("Uv"));
439  VertexAttribPointer(_shader["Color"].Location, 4, VertexAttribPointerType.UnsignedByte, true, drawVertSize, Marshal.OffsetOf<ImDrawVert>("Col"));
440  }

445  private void RenderDrawData()
446  {
447  ImDrawDataPtr drawData = ImGui.GetDrawData();
448 
449  // Avoid rendering when minimized, scale coordinates for retina displays (screen coordinates != framebuffer coordinates)
450  int fbWidth = (int) (drawData.DisplaySize.X * drawData.FramebufferScale.X);
451  int fbHeight = (int) (drawData.DisplaySize.Y * drawData.FramebufferScale.Y);
452  if (fbWidth <= 0 || fbHeight <= 0)
453  {
454  return;
455  }
456 
457  SetupRenderState(drawData, fbWidth, fbHeight);
458 
459  Vector2F clipOffset = drawData.DisplayPos;
460  Vector2F clipScale = drawData.FramebufferScale;
461 
462  drawData.ScaleClipRects(clipScale);
463 
464  IntPtr lastTexId = ImGui.GetIo().Fonts.TexId;
465  GlBindTexture(TextureTarget.Texture2D, (uint) lastTexId);
466 
467  int drawVertSize = Marshal.SizeOf<ImDrawVert>();
468  int drawIdxSize = sizeof(ushort);
469 
470  for (int n = 0; n < drawData.CmdListsCount; n++)
471  {
472  ImDrawListPtr cmdList = drawData.CmdListsRange[n];
473 
474  // Upload vertex/index buffers
475  GlBufferData(BufferTarget.ArrayBuffer, (IntPtr) (cmdList.VtxBuffer.Size * drawVertSize), cmdList.VtxBuffer.Data, BufferUsageHint.StreamDraw);
476  GlBufferData(BufferTarget.ElementArrayBuffer, (IntPtr) (cmdList.IdxBuffer.Size * drawIdxSize), cmdList.IdxBuffer.Data, BufferUsageHint.StreamDraw);
477 
478  for (int cmdI = 0; cmdI < cmdList.CmdBuffer.Size; cmdI++)
479  {
480  ImDrawCmdPtr pcmd = cmdList.CmdBuffer[cmdI];
481  if (pcmd.UserCallback != IntPtr.Zero)
482  {
483  Console.WriteLine("UserCallback not implemented");
484  }
485  else
486  {
487  // Project scissor/clipping rectangles into framebuffer space
488  Vector4F clipRect = pcmd.ClipRect;
489 
490  clipRect.X = pcmd.ClipRect.X - clipOffset.X;
491  clipRect.Y = pcmd.ClipRect.Y - clipOffset.Y;
492  clipRect.Z = pcmd.ClipRect.Z - clipOffset.X;
493  clipRect.W = pcmd.ClipRect.W - clipOffset.Y;
494 
495  GlScissor((int) clipRect.X, (int) (fbHeight - clipRect.W), (int) (clipRect.Z - clipRect.X), (int) (clipRect.W - clipRect.Y));
496 
497  // Bind texture, Draw
498  if (pcmd.TextureId != IntPtr.Zero)
499  {
500  if (pcmd.TextureId != lastTexId)
501  {
502  lastTexId = pcmd.TextureId;
503  GlBindTexture(TextureTarget.Texture2D, (uint) pcmd.TextureId);
504  }
505  }
506 
507  GlDrawElementsBaseVertex(BeginMode.Triangles, (int) pcmd.ElemCount, drawIdxSize == 2 ? DrawElementsType.UnsignedShort : DrawElementsType.UnsignedInt, (IntPtr) (pcmd.IdxOffset * drawIdxSize), (int) pcmd.VtxOffset);
508  }
509  }
510  }
511  }
