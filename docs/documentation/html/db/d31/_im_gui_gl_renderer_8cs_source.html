216  public void NewFrame()
217  {
218  ImGui.NewFrame();
219  ImGuiIoPtr io = ImGui.GetIo();
220 
221  // Setup display size (every frame to accommodate for window resizing)
222  GetWindowSize(_window, out int w, out int h);
223  GlGetDrawableSize(_window, out int displayW, out int displayH);
224  io.DisplaySize = new Vector2F(w, h);
225  if ((w > 0) && (h > 0))
226  {
227  io.DisplayFramebufferScale = new Vector2F((float) displayW / w, (float) displayH / h);
228  }
229 
230  // Setup time step (we don't use SDL_GetTicks() because it is using millisecond resolution)
231  ulong frequency = GetPerformanceFrequency();
232  ulong currentTime = GetPerformanceCounter();
233  io.DeltaTime = _time > 0 ? (float) ((double) (currentTime - _time) / frequency) : 1.0f / 60.0f;
234  if (io.DeltaTime <= 0)
235  {
236  io.DeltaTime = 0.016f;
237  }
238 
239  _time = currentTime;
240 
241  UpdateMousePosAndButtons();
242  }

248  public void ProcessEvent(SdlEvent evt)
249  {
250  ImGuiIoPtr io = ImGui.GetIo();
251  switch (evt.type)
252  {
253  case SdlEventType.SdlMousewheel:
254  {
255  if (evt.wheel.x > 0)
256  {
257  io.MouseWheelH += 1;
258  }
259 
260  if (evt.wheel.x < 0)
261  {
262  io.MouseWheelH -= 1;
263  }
264 
265  if (evt.wheel.y > 0)
266  {
267  io.MouseWheel += 1;
268  }
269 
270  if (evt.wheel.y < 0)
271  {
272  io.MouseWheel -= 1;
273  }
274 
275  return;
276  }
277  case SdlEventType.SdlMouseButtonDown:
278  {
279  if (evt.button.button == ButtonLeft)
280  {
281  _mousePressed[0] = true;
282  }
283 
284  if (evt.button.button == ButtonRight)
285  {
286  _mousePressed[1] = true;
287  }
288 
289  if (evt.button.button == ButtonMiddle)
290  {
291  _mousePressed[2] = true;
292  }
293 
294  return;
295  }
296  case SdlEventType.SdlTextInput:
297  {
298  string str = Encoding.UTF8.GetString(evt.text.Text);
299  io.AddInputCharactersUtf8(str);
300  return;
301  }
302  case SdlEventType.SdlKeydown:
303  case SdlEventType.SdlKeyup:
304  {
305  SdlScancode key = evt.key.keysym.scancode;
306  io.KeysDown[(int) key] = evt.type == SdlEventType.SdlKeydown;
307  Console.WriteLine("io.KeysDown[" + key + "] = " + evt.type + io.KeysDown[(int) key]);
308  io.KeyShift = (GetModState() & SdlKeymod.KmodShift) != 0;
309  io.KeyCtrl = (GetModState() & SdlKeymod.KmodCtrl) != 0;
310  io.KeyAlt = (GetModState() & SdlKeymod.KmodAlt) != 0;
311  io.KeySuper = (GetModState() & SdlKeymod.KmodGui) != 0;
312  break;
313  }
314  }
315  }

320  private void UpdateMousePosAndButtons()
321  {
322  ImGuiIoPtr io = ImGui.GetIo();
323 
324  // Set OS mouse position if requested (rarely used, only when ImGuiConfigFlags_NavEnableSetMousePos is enabled by user)
325  if (io.WantSetMousePos)
326  {
327  WarpMouseInWindow(_window, (int) io.MousePos.X, (int) io.MousePos.Y);
328  }
329  else
330  {
331  io.MousePos = new Vector2F(float.MinValue, float.MinValue);
332  }
333 
334  uint mouseButtons = GetMouseStateOutXAndY(out int mx, out int my);
335  io.MouseDown[0] =
336  _mousePressed[0] ||
337  (mouseButtons & Button(ButtonLeft)) !=
338  0; // If a mouse press event came, always pass it as "mouse held this frame", so we don't miss click-release events that are shorter than 1 frame.
339  io.MouseDown[1] = _mousePressed[1] || (mouseButtons & Button(ButtonRight)) != 0;
340  io.MouseDown[2] = _mousePressed[2] || (mouseButtons & Button(ButtonMiddle)) != 0;
341  _mousePressed[0] = _mousePressed[1] = _mousePressed[2] = false;
342 
343  IntPtr focusedWindow = GetKeyboardFocus();
344  if (_window == focusedWindow)
345  {
346  // SDL_GetMouseState() gives mouse position seemingly based on the last window entered/focused(?)
347  // The creation of a new windows at runtime and SDL_CaptureMouse both seems to severely mess up with that, so we retrieve that position globally.
348  GetWindowPosition(focusedWindow, out int wx, out int wy);
349  GetGlobalMouseStateOutXAndOutY(out mx, out my);
350  mx -= wx;
351  my -= wy;
352  io.MousePos = new Vector2F(mx, my);
353  }
354 
355  // SDL_CaptureMouse() let the OS know e.g. that our imgui drag outside the SDL window boundaries shouldn't e.g. trigger the OS window resize cursor.
356  bool anyMouseButtonDown = ImGui.IsAnyMouseDown();
357  CaptureMouse(anyMouseButtonDown ? SdlBool.SdlTrue : SdlBool.SdlFalse);
358  }

416  private void SetupRenderState(ImDrawDataPtr drawData, int fbWidth, int fbHeight)
417  {
418  GlEnable(EnableCap.Blend);
419  GlBlendEquation(BlendEquationMode.FuncAdd);
420  GlBlendFunc(BlendingFactorSrc.SrcAlpha, BlendingFactorDest.OneMinusSrcAlpha);
421  GlDisable(EnableCap.CullFace);
422  GlDisable(EnableCap.DepthTest);
423  GlEnable(EnableCap.ScissorTest);
424 
425  GlUseProgram(_shader.ProgramId);
426 
427  float left = drawData.DisplayPos.X;
428  float right = drawData.DisplayPos.X + drawData.DisplaySize.X;
429  float top = drawData.DisplayPos.Y;
430  float bottom = drawData.DisplayPos.Y + drawData.DisplaySize.Y;
431 
432  _shader["Texture"].SetValue(0);
433  _shader["ProjMtx"].SetValue(Matrix4X4F.CreateOrthographicOffCenter(left, right, bottom, top, -1, 1));
434  GlBindSampler(0, 0);
435 
436  GlBindVertexArray(_vertexArrayObject);
437 
438  // Bind vertex/index buffers and setup attributes for ImDrawVert
439  GlBindBuffer(BufferTarget.ArrayBuffer, _vboHandle);
440  GlBindBuffer(BufferTarget.ElementArrayBuffer, _elementsHandle);
441 
442  EnableVertexAttribArray(_shader["Position"].Location);
443  EnableVertexAttribArray(_shader["UV"].Location);
444  EnableVertexAttribArray(_shader["Color"].Location);
445 
446  int drawVertSize = Marshal.SizeOf<ImDrawVert>();
447  VertexAttribPointer(_shader["Position"].Location, 2, VertexAttribPointerType.Float, false, drawVertSize, Marshal.OffsetOf<ImDrawVert>("Pos"));
448  VertexAttribPointer(_shader["UV"].Location, 2, VertexAttribPointerType.Float, false, drawVertSize, Marshal.OffsetOf<ImDrawVert>("Uv"));
449  VertexAttribPointer(_shader["Color"].Location, 4, VertexAttribPointerType.UnsignedByte, true, drawVertSize, Marshal.OffsetOf<ImDrawVert>("Col"));
450  }

455  private void RenderDrawData()
456  {
457  ImDrawDataPtr drawData = ImGui.GetDrawData();
458 
459  // Avoid rendering when minimized, scale coordinates for retina displays (screen coordinates != framebuffer coordinates)
460  int fbWidth = (int) (drawData.DisplaySize.X * drawData.FramebufferScale.X);
461  int fbHeight = (int) (drawData.DisplaySize.Y * drawData.FramebufferScale.Y);
462  if (fbWidth <= 0 || fbHeight <= 0)
463  {
464  return;
465  }
466 
467  SetupRenderState(drawData, fbWidth, fbHeight);
468 
469  Vector2F clipOffset = drawData.DisplayPos;
470  Vector2F clipScale = drawData.FramebufferScale;
471 
472  drawData.ScaleClipRects(clipScale);
473 
474  IntPtr lastTexId = ImGui.GetIo().Fonts.TexId;
475  Gl.GlBindTexture(TextureTarget.Texture2D, (uint) lastTexId);
476 
477  int drawVertSize = Marshal.SizeOf<ImDrawVert>();
478  int drawIdxSize = sizeof(ushort);
479 
480  for (int n = 0; n < drawData.CmdListsCount; n++)
481  {
482  ImDrawListPtr cmdList = drawData.CmdListsRange[n];
483 
484  // Upload vertex/index buffers
485  GlBufferData(BufferTarget.ArrayBuffer, (IntPtr) (cmdList.VtxBuffer.Size * drawVertSize), cmdList.VtxBuffer.Data, BufferUsageHint.StreamDraw);
486  GlBufferData(BufferTarget.ElementArrayBuffer, (IntPtr) (cmdList.IdxBuffer.Size * drawIdxSize), cmdList.IdxBuffer.Data, BufferUsageHint.StreamDraw);
487 
488  for (int cmdI = 0; cmdI < cmdList.CmdBuffer.Size; cmdI++)
489  {
490  ImDrawCmdPtr pcmd = cmdList.CmdBuffer[cmdI];
491  if (pcmd.UserCallback != IntPtr.Zero)
492  {
493  Console.WriteLine("UserCallback not implemented");
494  }
495  else
496  {
497  // Project scissor/clipping rectangles into framebuffer space
498  Vector4F clipRect = pcmd.ClipRect;
499 
500  clipRect.X = pcmd.ClipRect.X - clipOffset.X;
501  clipRect.Y = pcmd.ClipRect.Y - clipOffset.Y;
502  clipRect.Z = pcmd.ClipRect.Z - clipOffset.X;
503  clipRect.W = pcmd.ClipRect.W - clipOffset.Y;
504 
505  GlScissor((int) clipRect.X, (int) (fbHeight - clipRect.W), (int) (clipRect.Z - clipRect.X), (int) (clipRect.W - clipRect.Y));
506 
507  // Bind texture, Draw
508  if (pcmd.TextureId != IntPtr.Zero)
509  {
510  if (pcmd.TextureId != lastTexId)
511  {
512  lastTexId = pcmd.TextureId;
513  Gl.GlBindTexture(TextureTarget.Texture2D, (uint) pcmd.TextureId);
514  }
515  }
516 
517  GlDrawElementsBaseVertex(BeginMode.Triangles, (int) pcmd.ElemCount, drawIdxSize == 2 ? DrawElementsType.UnsignedShort : DrawElementsType.UnsignedInt, (IntPtr) (pcmd.IdxOffset * drawIdxSize), (int) pcmd.VtxOffset);
518  }
519  }
520  }
521  }
522  }
