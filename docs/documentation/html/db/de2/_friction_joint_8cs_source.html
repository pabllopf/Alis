262  internal override void InitVelocityConstraints(ref SolverData data)
263  {
264  indexA = BodyA.IslandIndex;
265  indexB = BodyB.IslandIndex;
266  localCenterA = BodyA.Sweep.LocalCenter;
267  localCenterB = BodyB.Sweep.LocalCenter;
268  invMassA = BodyA.InvMass;
269  invMassB = BodyB.InvMass;
270  invIa = BodyA.InvI;
271  invIb = BodyB.InvI;
272 
273  float aA = data.Positions[indexA].A;
274  Vector2F vA = data.Velocities[indexA].V;
275  float wA = data.Velocities[indexA].W;
276 
277  float aB = data.Positions[indexB].A;
278  Vector2F vB = data.Velocities[indexB].V;
279  float wB = data.Velocities[indexB].W;
280 
281  Rotation qA = new Rotation(aA), qB = new Rotation(aB);
282 
283  // Compute the effective mass matrix.
284  rA = MathUtils.Mul(qA, localAnchorA - localCenterA);
285  rB = MathUtils.Mul(qB, localAnchorB - localCenterB);
286 
287  // J = [-I -r1_skew I r2_skew]
288  // [ 0 -1 0 1]
289  // r_skew = [-ry; rx]
290 
291  // Matlab
292  // K = [ mA+r1y^2*iA+mB+r2y^2*iB, -r1y*iA*r1x-r2y*iB*r2x, -r1y*iA-r2y*iB]
293  // [ -r1y*iA*r1x-r2y*iB*r2x, mA+r1x^2*iA+mB+r2x^2*iB, r1x*iA+r2x*iB]
294  // [ -r1y*iA-r2y*iB, r1x*iA+r2x*iB, iA+iB]
295 
296  float mA = invMassA, mB = invMassB;
297  float iA = invIa, iB = invIb;
298 
299  Matrix2X2F k = new Matrix2X2F();
300  k.Ex.X = mA + mB + iA * rA.Y * rA.Y + iB * rB.Y * rB.Y;
301  k.Ex.Y = -iA * rA.X * rA.Y - iB * rB.X * rB.Y;
302  k.Ey.X = k.Ex.Y;
303  k.Ey.Y = mA + mB + iA * rA.X * rA.X + iB * rB.X * rB.X;
304 
305  linearMass = k.Inverse;
306 
307  angularMass = iA + iB;
308  if (angularMass > 0.0f)
309  {
310  angularMass = 1.0f / angularMass;
311  }
312 
313  if (data.Step.WarmStarting)
314  {
315  // Scale impulses to support a variable time step.
316  linearImpulse *= data.Step.DeltaTimeRatio;
317  angularImpulse *= data.Step.DeltaTimeRatio;
318 
319  Vector2F p = new Vector2F(linearImpulse.X, linearImpulse.Y);
320  vA -= mA * p;
321  wA -= iA * (MathUtils.Cross(rA, p) + angularImpulse);
322  vB += mB * p;
323  wB += iB * (MathUtils.Cross(rB, p) + angularImpulse);
324  }
325  else
326  {
327  linearImpulse = Vector2F.Zero;
328  angularImpulse = 0.0f;
329  }
330 
331  data.Velocities[indexA].V = vA;
332  data.Velocities[indexA].W = wA;
333  data.Velocities[indexB].V = vB;
334  data.Velocities[indexB].W = wB;
335  }

341  internal override void SolveVelocityConstraints(ref SolverData data)
342  {
343  Vector2F vA = data.Velocities[indexA].V;
344  float wA = data.Velocities[indexA].W;
345  Vector2F vB = data.Velocities[indexB].V;
346  float wB = data.Velocities[indexB].W;
347 
348  float mA = invMassA, mB = invMassB;
349  float iA = invIa, iB = invIb;
350 
351  float h = data.Step.DeltaTime;
352 
353  // Solve angular friction
354  {
355  float cdot = wB - wA;
356  float impulse = -angularMass * cdot;
357 
358  float oldImpulse = angularImpulse;
359  float maxImpulse = h * Torque;
360  angularImpulse = MathUtils.Clamp(angularImpulse + impulse, -maxImpulse, maxImpulse);
361  impulse = angularImpulse - oldImpulse;
362 
363  wA -= iA * impulse;
364  wB += iB * impulse;
365  }
366 
367  // Solve linear friction
368  {
369  Vector2F cdot = vB + MathUtils.Cross(wB, rB) - vA - MathUtils.Cross(wA, rA);
370 
371  Vector2F impulse = -MathUtils.Mul(ref linearMass, cdot);
372  Vector2F oldImpulse = linearImpulse;
373  linearImpulse += impulse;
374 
375  float maxImpulse = h * Force;
376 
377  if (linearImpulse.LengthSquared() > maxImpulse * maxImpulse)
378  {
379  linearImpulse = Vector2F.Normalize(linearImpulse);
380  linearImpulse *= maxImpulse;
381  }
382 
383  impulse = linearImpulse - oldImpulse;
384 
385  vA -= mA * impulse;
386  wA -= iA * MathUtils.Cross(rA, impulse);
387 
388  vB += mB * impulse;
389  wB += iB * MathUtils.Cross(rB, impulse);
390  }
391 
392  data.Velocities[indexA].V = vA;
393  data.Velocities[indexA].W = wA;
394  data.Velocities[indexB].V = vB;
395  data.Velocities[indexB].W = wB;
396  }
