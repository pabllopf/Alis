248  internal override void InitVelocityConstraints(ref SolverData data)
249  {
250  indexA = BodyA.IslandIndex;
251  indexB = BodyB.IslandIndex;
252  localCenterA = BodyA.Sweep.LocalCenter;
253  localCenterB = BodyB.Sweep.LocalCenter;
254  invMassA = BodyA.InvMass;
255  invMassB = BodyB.InvMass;
256  invIa = BodyA.InvI;
257  invIb = BodyB.InvI;
258 
259  float aA = data.Positions[indexA].A;
260  Vector2 vA = data.Velocities[indexA].V;
261  float wA = data.Velocities[indexA].W;
262 
263  float aB = data.Positions[indexB].A;
264  Vector2 vB = data.Velocities[indexB].V;
265  float wB = data.Velocities[indexB].W;
266 
267  Rot qA = new Rot(aA), qB = new Rot(aB);
268 
269  // Compute the effective mass matrix.
270  rA = MathUtils.Mul(qA, localAnchorA - localCenterA);
271  rB = MathUtils.Mul(qB, localAnchorB - localCenterB);
272 
273  // J = [-I -r1_skew I r2_skew]
274  // [ 0 -1 0 1]
275  // r_skew = [-ry; rx]
276 
277  // Matlab
278  // K = [ mA+r1y^2*iA+mB+r2y^2*iB, -r1y*iA*r1x-r2y*iB*r2x, -r1y*iA-r2y*iB]
279  // [ -r1y*iA*r1x-r2y*iB*r2x, mA+r1x^2*iA+mB+r2x^2*iB, r1x*iA+r2x*iB]
280  // [ -r1y*iA-r2y*iB, r1x*iA+r2x*iB, iA+iB]
281 
282  float mA = invMassA, mB = invMassB;
283  float iA = invIa, iB = invIb;
284 
285  Mat22 k = new Mat22();
286  k.Ex.X = mA + mB + iA * rA.Y * rA.Y + iB * rB.Y * rB.Y;
287  k.Ex.Y = -iA * rA.X * rA.Y - iB * rB.X * rB.Y;
288  k.Ey.X = k.Ex.Y;
289  k.Ey.Y = mA + mB + iA * rA.X * rA.X + iB * rB.X * rB.X;
290 
291  linearMass = k.Inverse;
292 
293  angularMass = iA + iB;
294  if (angularMass > 0.0f)
295  {
296  angularMass = 1.0f / angularMass;
297  }
298 
299  if (data.Step.WarmStarting)
300  {
301  // Scale impulses to support a variable time step.
302  linearImpulse *= data.Step.DeltaTimeRatio;
303  angularImpulse *= data.Step.DeltaTimeRatio;
304 
305  Vector2 p = new Vector2(linearImpulse.X, linearImpulse.Y);
306  vA -= mA * p;
307  wA -= iA * (MathUtils.Cross(rA, p) + angularImpulse);
308  vB += mB * p;
309  wB += iB * (MathUtils.Cross(rB, p) + angularImpulse);
310  }
311  else
312  {
313  linearImpulse = Vector2.Zero;
314  angularImpulse = 0.0f;
315  }
316 
317  data.Velocities[indexA].V = vA;
318  data.Velocities[indexA].W = wA;
319  data.Velocities[indexB].V = vB;
320  data.Velocities[indexB].W = wB;
321  }

327  internal override void SolveVelocityConstraints(ref SolverData data)
328  {
329  Vector2 vA = data.Velocities[indexA].V;
330  float wA = data.Velocities[indexA].W;
331  Vector2 vB = data.Velocities[indexB].V;
332  float wB = data.Velocities[indexB].W;
333 
334  float mA = invMassA, mB = invMassB;
335  float iA = invIa, iB = invIb;
336 
337  float h = data.Step.DeltaTime;
338 
339  // Solve angular friction
340  {
341  float cdot = wB - wA;
342  float impulse = -angularMass * cdot;
343 
344  float oldImpulse = angularImpulse;
345  float maxImpulse = h * MaxTorque;
346  angularImpulse = MathUtils.Clamp(angularImpulse + impulse, -maxImpulse, maxImpulse);
347  impulse = angularImpulse - oldImpulse;
348 
349  wA -= iA * impulse;
350  wB += iB * impulse;
351  }
352 
353  // Solve linear friction
354  {
355  Vector2 cdot = vB + MathUtils.Cross(wB, rB) - vA - MathUtils.Cross(wA, rA);
356 
357  Vector2 impulse = -MathUtils.Mul(ref linearMass, cdot);
358  Vector2 oldImpulse = linearImpulse;
359  linearImpulse += impulse;
360 
361  float maxImpulse = h * MaxForce;
362 
363  if (linearImpulse.LengthSquared() > maxImpulse * maxImpulse)
364  {
365  linearImpulse = Vector2.Normalize(linearImpulse);
366  linearImpulse *= maxImpulse;
367  }
368 
369  impulse = linearImpulse - oldImpulse;
370 
371  vA -= mA * impulse;
372  wA -= iA * MathUtils.Cross(rA, impulse);
373 
374  vB += mB * impulse;
375  wB += iB * MathUtils.Cross(rB, impulse);
376  }
377 
378  data.Velocities[indexA].V = vA;
379  data.Velocities[indexA].W = wA;
380  data.Velocities[indexB].V = vB;
381  data.Velocities[indexB].W = wB;
382  }
