214  internal override void InitVelocityConstraints(ref SolverData data)
215  {
216  indexA = BodyA.IslandIndex;
217  indexB = BodyB.IslandIndex;
218  Vector2 localCenterA = BodyA.Sweep.LocalCenter;
219  Vector2 localCenterB = BodyB.Sweep.LocalCenter;
220  invMassA = BodyA.InvMass;
221  invMassB = BodyB.InvMass;
222  invIa = BodyA.InvI;
223  invIb = BodyB.InvI;
224 
225  float aA = data.Positions[indexA].A;
226  Vector2 vA = data.Velocities[indexA].V;
227  float wA = data.Velocities[indexA].W;
228 
229  float aB = data.Positions[indexB].A;
230  Vector2 vB = data.Velocities[indexB].V;
231  float wB = data.Velocities[indexB].W;
232 
233  Rotation qA = new Rotation(aA), qB = new Rotation(aB);
234 
235  // Compute the effective mass matrix.
236  rA = MathUtils.Mul(qA, LocalAnchorA - localCenterA);
237  rB = MathUtils.Mul(qB, LocalAnchorB - localCenterB);
238 
239  // J = [-I -r1_skew I r2_skew]
240  // [ 0 -1 0 1]
241  // r_skew = [-ry; rx]
242 
243  // Matlab
244  // K = [ mA+r1y^2*iA+mB+r2y^2*iB, -r1y*iA*r1x-r2y*iB*r2x, -r1y*iA-r2y*iB]
245  // [ -r1y*iA*r1x-r2y*iB*r2x, mA+r1x^2*iA+mB+r2x^2*iB, r1x*iA+r2x*iB]
246  // [ -r1y*iA-r2y*iB, r1x*iA+r2x*iB, iA+iB]
247 
248  float mA = invMassA, mB = invMassB;
249  float iA = invIa, iB = invIb;
250 
251  Matrix2X2F k = new Matrix2X2F(
252  mA + mB + iA * rA.Y * rA.Y + iB * rB.Y * rB.Y,
253  -iA * rA.X * rA.Y - iB * rB.X * rB.Y,
254  -iA * rA.X * rA.Y - iB * rB.X * rB.Y,
255  mA + mB + iA * rA.X * rA.X + iB * rB.X * rB.X
256  );
257 
258  linearMass = k.Inverse;
259 
260  angularMass = iA + iB;
261  if (angularMass > 0.0f)
262  {
263  angularMass = 1.0f / angularMass;
264  }
265 
266  if (data.Step.WarmStarting)
267  {
268  // Scale impulses to support a variable time step.
269  linearImpulse *= data.Step.DeltaTimeRatio;
270  angularImpulse *= data.Step.DeltaTimeRatio;
271 
272  Vector2 p = new Vector2(linearImpulse.X, linearImpulse.Y);
273  vA -= mA * p;
274  wA -= iA * (MathUtils.Cross(rA, p) + angularImpulse);
275  vB += mB * p;
276  wB += iB * (MathUtils.Cross(rB, p) + angularImpulse);
277  }
278  else
279  {
280  linearImpulse = Vector2.Zero;
281  angularImpulse = 0.0f;
282  }
283 
284  data.Velocities[indexA].V = vA;
285  data.Velocities[indexA].W = wA;
286  data.Velocities[indexB].V = vB;
287  data.Velocities[indexB].W = wB;
288  }

323  private void SolveAngularFriction(ref float wA, ref float wB, float iA, float iB, float h)
324  {
325  float cDot = wB - wA;
326  float impulse = -angularMass * cDot;
327 
328  float oldImpulse = angularImpulse;
329  float maxImpulse = h * Torque;
330  angularImpulse = MathUtils.Clamp(angularImpulse + impulse, -maxImpulse, maxImpulse);
331  impulse = angularImpulse - oldImpulse;
332 
333  wA -= iA * impulse;
334  wB += iB * impulse;
335  }
336 
349  private void SolveLinearFriction(ref Vector2 vA, ref float wA, ref Vector2 vB, ref float wB, float mA, float mB, float iA, float iB, float h)
350  {
351  Vector2 cDot = vB + MathUtils.Cross(wB, rB) - vA - MathUtils.Cross(wA, rA);
352 
353  Vector2 impulse = -MathUtils.Mul(ref linearMass, cDot);
354  Vector2 oldImpulse = linearImpulse;
355  linearImpulse += impulse;
356 
357  float maxImpulse = h * Force;
358 
359  if (linearImpulse.LengthSquared() > maxImpulse * maxImpulse)
360  {
361  linearImpulse = Vector2.Normalize(linearImpulse);
362  linearImpulse *= maxImpulse;
363  }
364 
365  impulse = linearImpulse - oldImpulse;
366 
367  vA -= mA * impulse;
368  wA -= iA * MathUtils.Cross(rA, impulse);
369 
370  vB += mB * impulse;
371  wB += iB * MathUtils.Cross(rB, impulse);
372  }
