1623  {

1628  case WorkItemType.ExecuteList:
1629  {

1631  try
1632  {
1633  list.ExecuteAll(_gd._commandExecutor);
1634  }
1635  finally
1636  {
1637  if (!_gd.CheckCommandListDisposal(list.Parent))
1638  {

1640  }
1641  }
1642  }
1643  break;
1644  case WorkItemType.Map:
1645  {
1646  MappableResource resourceToMap = (MappableResource)workItem.Object0;

1660  ExecuteUnmapResource(resourceToMap, resultPtr->Subresource, mre);

1663  break;
