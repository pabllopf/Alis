288  internal static uint GetRegionSize(uint width, uint height, uint depth, PixelFormat format)
289  {
290  uint blockSizeInBytes;
291  if (IsCompressedFormat(format))
292  {
293  Debug.Assert((width % 4 == 0 || width < 4) && (height % 4 == 0 || height < 4));
294  blockSizeInBytes = GetBlockSizeInBytes(format);
295  width /= 4;
296  height /= 4;
297  }
298  else
299  {
300  blockSizeInBytes = FormatSizeHelpers.GetSizeInBytes(format);
301  }
302 
303  return width * height * depth * blockSizeInBytes;
