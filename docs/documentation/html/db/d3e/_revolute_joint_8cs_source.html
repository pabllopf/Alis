425  internal override void InitVelocityConstraints(ref SolverData data)
426  {
427  indexA = BodyA.IslandIndex;
428  indexB = BodyB.IslandIndex;
429  localCenterA = BodyA.Sweep.LocalCenter;
430  localCenterB = BodyB.Sweep.LocalCenter;
431  invMassA = BodyA.InvMass;
432  invMassB = BodyB.InvMass;
433  invIa = BodyA.InvI;
434  invIb = BodyB.InvI;
435 
436  float aA = data.Positions[indexA].A;
437  Vector2 vA = data.Velocities[indexA].V;
438  float wA = data.Velocities[indexA].W;
439 
440  float aB = data.Positions[indexB].A;
441  Vector2 vB = data.Velocities[indexB].V;
442  float wB = data.Velocities[indexB].W;
443 
444  Rot qA = new Rot(aA), qB = new Rot(aB);
445 
446  rA = MathUtils.Mul(qA, LocalAnchorA - localCenterA);
447  rB = MathUtils.Mul(qB, LocalAnchorB - localCenterB);
448 
449  // J = [-I -r1_skew I r2_skew]
450  // r_skew = [-ry; rx]
451 
452  // Matlab
453  // K = [ mA+r1y^2*iA+mB+r2y^2*iB, -r1y*iA*r1x-r2y*iB*r2x]
454  // [ -r1y*iA*r1x-r2y*iB*r2x, mA+r1x^2*iA+mB+r2x^2*iB]
455 
456  float mA = invMassA, mB = invMassB;
457  float iA = invIa, iB = invIb;
458 
459  k.Ex.X = mA + mB + rA.Y * rA.Y * iA + rB.Y * rB.Y * iB;
460  k.Ey.X = -rA.Y * rA.X * iA - rB.Y * rB.X * iB;
461  k.Ex.Y = k.Ey.X;
462  k.Ey.Y = mA + mB + rA.X * rA.X * iA + rB.X * rB.X * iB;
463 
464  axialMass = iA + iB;
465  bool fixedRotation;
466  if (axialMass > 0.0f)
467  {
468  axialMass = 1.0f / axialMass;
469  fixedRotation = false;
470  }
471  else
472  {
473  fixedRotation = true;
474  }
475 
476  angle = aB - aA - ReferenceAngle;
477  if (enableLimit == false || fixedRotation)
478  {
479  lowerImpulse = 0.0f;
480  upperImpulse = 0.0f;
481  }
482 
483  if (enableMotor == false || fixedRotation)
484  {
485  motorImpulse = 0.0f;
486  }
487 
488  if (data.Step.WarmStarting)
489  {
490  // Scale impulses to support a variable time step.
491  impulse *= data.Step.DeltaTimeRatio;
492  motorImpulse *= data.Step.DeltaTimeRatio;
493  lowerImpulse *= data.Step.DeltaTimeRatio;
494  upperImpulse *= data.Step.DeltaTimeRatio;
495 
496  float axialImpulse = motorImpulse + lowerImpulse - upperImpulse;
497  Vector2 p = new Vector2(impulse.X, impulse.Y);
498 
499  vA -= mA * p;
500  wA -= iA * (MathUtils.Cross(rA, p) + axialImpulse);
501 
502  vB += mB * p;
503  wB += iB * (MathUtils.Cross(rB, p) + axialImpulse);
504  }
505  else
506  {
507  impulse = Vector2.Zero;
508  motorImpulse = 0.0f;
509  lowerImpulse = 0.0f;
510  upperImpulse = 0.0f;
511  }
512 
513  data.Velocities[indexA].V = vA;
514  data.Velocities[indexA].W = wA;
515  data.Velocities[indexB].V = vB;
516  data.Velocities[indexB].W = wB;
517  }

523  internal override void SolveVelocityConstraints(ref SolverData data)
524  {
525  Vector2 vA = data.Velocities[indexA].V;
526  float wA = data.Velocities[indexA].W;
527  Vector2 vB = data.Velocities[indexB].V;
528  float wB = data.Velocities[indexB].W;
529 
530  float mA = invMassA, mB = invMassB;
531  float iA = invIa, iB = invIb;
532 
533  bool fixedRotation = iA + iB == 0.0f;
534 
535  // Solve motor constraint.
536  if (enableMotor && fixedRotation == false)
537  {
538  float cdot = wB - wA - motorSpeed;
539  float impulse = -axialMass * cdot;
540  float oldImpulse = motorImpulse;
541  float maxImpulse = data.Step.DeltaTime * maxMotorTorque;
542  motorImpulse = MathUtils.Clamp(motorImpulse + impulse, -maxImpulse, maxImpulse);
543  impulse = motorImpulse - oldImpulse;
544 
545  wA -= iA * impulse;
546  wB += iB * impulse;
547  }
548 
549  if (enableLimit && fixedRotation == false)
550  {
551  // Lower limit
552  {
553  float c = angle - lowerAngle;
554  float cdot = wB - wA;
555  float impulse = -axialMass * (cdot + MathUtils.Max(c, 0.0f) * data.Step.InvertedDeltaTime);
556  float oldImpulse = lowerImpulse;
557  lowerImpulse = MathUtils.Max(lowerImpulse + impulse, 0.0f);
558  impulse = lowerImpulse - oldImpulse;
559 
560  wA -= iA * impulse;
561  wB += iB * impulse;
562  }
563 
564  // Upper limit
565  // Note: signs are flipped to keep C positive when the constraint is satisfied.
566  // This also keeps the impulse positive when the limit is active.
567  {
568  float c = upperAngle - angle;
569  float cdot = wA - wB;
570  float impulse = -axialMass * (cdot + MathUtils.Max(c, 0.0f) * data.Step.InvertedDeltaTime);
571  float oldImpulse = upperImpulse;
572  upperImpulse = MathUtils.Max(upperImpulse + impulse, 0.0f);
573  impulse = upperImpulse - oldImpulse;
574 
575  wA += iA * impulse;
576  wB -= iB * impulse;
577  }
578  }
579 
580  // Solve point-to-point constraint
581  {
582  Vector2 cdot = vB + MathUtils.Cross(wB, rB) - vA - MathUtils.Cross(wA, rA);
583  Vector2 impulse = k.Solve(-cdot);
584 
585  this.impulse.X += impulse.X;
586  this.impulse.Y += impulse.Y;
587 
588  vA -= mA * impulse;
589  wA -= iA * MathUtils.Cross(rA, impulse);
590 
591  vB += mB * impulse;
592  wB += iB * MathUtils.Cross(rB, impulse);
593  }
594 
595  data.Velocities[indexA].V = vA;
596  data.Velocities[indexA].W = wA;
597  data.Velocities[indexB].V = vB;
598  data.Velocities[indexB].W = wB;
599  }

606  internal override bool SolvePositionConstraints(ref SolverData data)
607  {
608  Vector2 cA = data.Positions[indexA].C;
609  float aA = data.Positions[indexA].A;
610  Vector2 cB = data.Positions[indexB].C;
611  float aB = data.Positions[indexB].A;
612 
613  Rot qA = new Rot(aA), qB = new Rot(aB);
614 
615  float angularError = 0.0f;
616  float positionError = 0.0f;
617 
618  bool fixedRotation = invIa + invIb == 0.0f;
619 
620  // Solve angular limit constraint
621  if (enableLimit && fixedRotation == false)
622  {
623  float angle = aB - aA - ReferenceAngle;
624  float c = 0.0f;
625 
626  if (MathUtils.Abs(upperAngle - lowerAngle) < 2.0f * Settings.AngularSlop)
627  {
628  // Prevent large angular corrections
629  c = MathUtils.Clamp(angle - lowerAngle, -Settings.MaxAngularCorrection,
630  Settings.MaxAngularCorrection);
631  }
632  else if (angle <= lowerAngle)
633  {
634  // Prevent large angular corrections and allow some slop.
635  c = MathUtils.Clamp(angle - lowerAngle + Settings.AngularSlop, -Settings.MaxAngularCorrection,
636  0.0f);
637  }
638  else if (angle >= upperAngle)
639  {
640  // Prevent large angular corrections and allow some slop.
641  c = MathUtils.Clamp(angle - upperAngle - Settings.AngularSlop, 0.0f,
642  Settings.MaxAngularCorrection);
643  }
644 
645  float limitImpulse = -axialMass * c;
646  aA -= invIa * limitImpulse;
647  aB += invIb * limitImpulse;
648  angularError = MathUtils.Abs(c);
649  }
650 
651  // Solve point-to-point constraint.
652  {
653  qA.Set(aA);
654  qB.Set(aB);
655  Vector2 rA = MathUtils.Mul(qA, LocalAnchorA - localCenterA);
656  Vector2 rB = MathUtils.Mul(qB, LocalAnchorB - localCenterB);
657 
658  Vector2 c = cB + rB - cA - rA;
659  positionError = c.Length();
660 
661  float mA = invMassA, mB = invMassB;
662  float iA = invIa, iB = invIb;
663 
664  Mat22 k;
665  k.Ex.X = mA + mB + iA * rA.Y * rA.Y + iB * rB.Y * rB.Y;
666  k.Ex.Y = -iA * rA.X * rA.Y - iB * rB.X * rB.Y;
667  k.Ey.X = k.Ex.Y;
668  k.Ey.Y = mA + mB + iA * rA.X * rA.X + iB * rB.X * rB.X;
669 
670  Vector2 impulse = -k.Solve(c);
671 
672  cA -= mA * impulse;
673  aA -= iA * MathUtils.Cross(rA, impulse);
674 
675  cB += mB * impulse;
676  aB += iB * MathUtils.Cross(rB, impulse);
677  }
678 
679  data.Positions[indexA].C = cA;
680  data.Positions[indexA].A = aA;
681  data.Positions[indexB].C = cB;
682  data.Positions[indexB].A = aB;
683 
684  return positionError <= Settings.LinearSlop && angularError <= Settings.AngularSlop;
685  }
686  }
