460  internal override void InitVelocityConstraints(ref SolverData data)
461  {
462  indexA = BodyA.IslandIndex;
463  indexB = BodyB.IslandIndex;
464  localCenterA = BodyA.Sweep.LocalCenter;
465  localCenterB = BodyB.Sweep.LocalCenter;
466  invMassA = BodyA.InvMass;
467  invMassB = BodyB.InvMass;
468  invIa = BodyA.InvI;
469  invIb = BodyB.InvI;
470 
471  float aA = data.Positions[indexA].A;
472  Vector2F vA = data.Velocities[indexA].V;
473  float wA = data.Velocities[indexA].W;
474 
475  float aB = data.Positions[indexB].A;
476  Vector2F vB = data.Velocities[indexB].V;
477  float wB = data.Velocities[indexB].W;
478 
479  Rotation qA = new Rotation(aA), qB = new Rotation(aB);
480 
481  rA = MathUtils.Mul(qA, LocalAnchorA - localCenterA);
482  rB = MathUtils.Mul(qB, LocalAnchorB - localCenterB);
483 
484  // J = [-I -r1_skew I r2_skew]
485  // r_skew = [-ry; rx]
486 
487  // Matlab
488  // K = [ mA+r1y^2*iA+mB+r2y^2*iB, -r1y*iA*r1x-r2y*iB*r2x]
489  // [ -r1y*iA*r1x-r2y*iB*r2x, mA+r1x^2*iA+mB+r2x^2*iB]
490 
491  float mA = invMassA, mB = invMassB;
492  float iA = invIa, iB = invIb;
493 
494  k = new Matrix2X2F(
495  mA + mB + rA.Y * rA.Y * iA + rB.Y * rB.Y * iB,
496  -rA.Y * rA.X * iA - rB.Y * rB.X * iB,
497  -rA.Y * rA.X * iA - rB.Y * rB.X * iB,
498  mA + mB + rA.X * rA.X * iA + rB.X * rB.X * iB
499  );
500 
501  axialMass = iA + iB;
502  bool fixedRotation;
503  if (axialMass > 0.0f)
504  {
505  axialMass = 1.0f / axialMass;
506  fixedRotation = false;
507  }
508  else
509  {
510  fixedRotation = true;
511  }
512 
513  angle = aB - aA - ReferenceAngle;
514  if (enableLimit == false || fixedRotation)
515  {
516  lowerImpulse = 0.0f;
517  upperImpulse = 0.0f;
518  }
519 
520  if (enableMotor == false || fixedRotation)
521  {
522  motorImpulse = 0.0f;
523  }
524 
525  if (data.Step.WarmStarting)
526  {
527  // Scale impulses to support a variable time step.
528  impulse *= data.Step.DeltaTimeRatio;
529  motorImpulse *= data.Step.DeltaTimeRatio;
530  lowerImpulse *= data.Step.DeltaTimeRatio;
531  upperImpulse *= data.Step.DeltaTimeRatio;
532 
533  float axialImpulse = motorImpulse + lowerImpulse - upperImpulse;
534  Vector2F p = new Vector2F(impulse.X, impulse.Y);
535 
536  vA -= mA * p;
537  wA -= iA * (MathUtils.Cross(rA, p) + axialImpulse);
538 
539  vB += mB * p;
540  wB += iB * (MathUtils.Cross(rB, p) + axialImpulse);
541  }
542  else
543  {
544  impulse = Vector2F.Zero;
545  motorImpulse = 0.0f;
546  lowerImpulse = 0.0f;
547  upperImpulse = 0.0f;
548  }
549 
550  data.Velocities[indexA].V = vA;
551  data.Velocities[indexA].W = wA;
552  data.Velocities[indexB].V = vB;
553  data.Velocities[indexB].W = wB;
554  }

560  internal override void SolveVelocityConstraints(ref SolverData data)
561  {
562  Vector2F vA = data.Velocities[indexA].V;
563  float wA = data.Velocities[indexA].W;
564  Vector2F vB = data.Velocities[indexB].V;
565  float wB = data.Velocities[indexB].W;
566 
567  float mA = invMassA, mB = invMassB;
568  float iA = invIa, iB = invIb;
569 
570  bool fixedRotation = iA + iB == 0.0f;
571 
572  // Solve motor constraint.
573  if (enableMotor && (fixedRotation == false))
574  {
575  float cdot = wB - wA - motorSpeed;
576  float impulse = -axialMass * cdot;
577  float oldImpulse = motorImpulse;
578  float maxImpulse = data.Step.DeltaTime * maxMotorTorque;
579  motorImpulse = MathUtils.Clamp(motorImpulse + impulse, -maxImpulse, maxImpulse);
580  impulse = motorImpulse - oldImpulse;
581 
582  wA -= iA * impulse;
583  wB += iB * impulse;
584  }
585 
586  if (enableLimit && (fixedRotation == false))
587  {
588  // Lower limit
589  {
590  float c = angle - lowerAngle;
591  float cdot = wB - wA;
592  float impulse = -axialMass * (cdot + MathUtils.Max(c, 0.0f) * data.Step.InvertedDeltaTime);
593  float oldImpulse = lowerImpulse;
594  lowerImpulse = MathUtils.Max(lowerImpulse + impulse, 0.0f);
595  impulse = lowerImpulse - oldImpulse;
596 
597  wA -= iA * impulse;
598  wB += iB * impulse;
599  }
600 
601  // Upper limit
602  // Note: signs are flipped to keep C positive when the constraint is satisfied.
603  // This also keeps the impulse positive when the limit is active.
604  {
605  float c = upperAngle - angle;
606  float cdot = wA - wB;
607  float impulse = -axialMass * (cdot + MathUtils.Max(c, 0.0f) * data.Step.InvertedDeltaTime);
608  float oldImpulse = upperImpulse;
609  upperImpulse = MathUtils.Max(upperImpulse + impulse, 0.0f);
610  impulse = upperImpulse - oldImpulse;
611 
612  wA += iA * impulse;
613  wB -= iB * impulse;
614  }
615  }
616 
617  // Solve point-to-point constraint
618  {
619  Vector2F cdot = vB + MathUtils.Cross(wB, rB) - vA - MathUtils.Cross(wA, rA);
620  Vector2F impulse = k.Solve(-cdot);
621 
622  this.impulse = new Vector2F(impulse.X, impulse.Y);
623 
624  vA -= mA * impulse;
625  wA -= iA * MathUtils.Cross(rA, impulse);
626 
627  vB += mB * impulse;
628  wB += iB * MathUtils.Cross(rB, impulse);
629  }
630 
631  data.Velocities[indexA].V = vA;
632  data.Velocities[indexA].W = wA;
633  data.Velocities[indexB].V = vB;
634  data.Velocities[indexB].W = wB;
635  }

642  internal override bool SolvePositionConstraints(ref SolverData data)
643  {
644  Vector2F cA = data.Positions[indexA].C;
645  float aA = data.Positions[indexA].A;
646  Vector2F cB = data.Positions[indexB].C;
647  float aB = data.Positions[indexB].A;
648 
649  Rotation qA = new Rotation(aA), qB = new Rotation(aB);
650 
651  float angularError = 0.0f;
652  float positionError = 0.0f;
653 
654  bool fixedRotation = invIa + invIb == 0.0f;
655 
656  // Solve angular limit constraint
657  if (enableLimit && (fixedRotation == false))
658  {
659  float angle = aB - aA - ReferenceAngle;
660  float c = 0.0f;
661 
662  if (MathUtils.Abs(upperAngle - lowerAngle) < 2.0f * Settings.AngularSlop)
663  {
664  // Prevent large angular corrections
665  c = MathUtils.Clamp(angle - lowerAngle, -Settings.AngularCorrection,
666  Settings.AngularCorrection);
667  }
668  else if (angle <= lowerAngle)
669  {
670  // Prevent large angular corrections and allow some slop.
671  c = MathUtils.Clamp(angle - lowerAngle + Settings.AngularSlop, -Settings.AngularCorrection,
672  0.0f);
673  }
674  else if (angle >= upperAngle)
675  {
676  // Prevent large angular corrections and allow some slop.
677  c = MathUtils.Clamp(angle - upperAngle - Settings.AngularSlop, 0.0f,
678  Settings.AngularCorrection);
679  }
680 
681  float limitImpulse = -axialMass * c;
682  aA -= invIa * limitImpulse;
683  aB += invIb * limitImpulse;
684  angularError = MathUtils.Abs(c);
685  }
686 
687  // Solve point-to-point constraint.
688  {
689  qA.Set(aA);
690  qB.Set(aB);
691  Vector2F rA = MathUtils.Mul(qA, LocalAnchorA - localCenterA);
692  Vector2F rB = MathUtils.Mul(qB, LocalAnchorB - localCenterB);
693 
694  Vector2F c = cB + rB - cA - rA;
695  positionError = c.Length();
696 
697  float mA = invMassA, mB = invMassB;
698  float iA = invIa, iB = invIb;
699 
700  Matrix2X2F k = new Matrix2X2F(
701  mA + mB + iA * rA.Y * rA.Y + iB * rB.Y * rB.Y,
702  -iA * rA.X * rA.Y - iB * rB.X * rB.Y,
703  -iA * rA.X * rA.Y - iB * rB.X * rB.Y,
704  mA + mB + iA * rA.X * rA.X + iB * rB.X * rB.X
705  );
706 
707  Vector2F impulse = -k.Solve(c);
708 
709  cA -= mA * impulse;
710  aA -= iA * MathUtils.Cross(rA, impulse);
711 
712  cB += mB * impulse;
713  aB += iB * MathUtils.Cross(rB, impulse);
714  }
715 
716  data.Positions[indexA].C = cA;
717  data.Positions[indexA].A = aA;
718  data.Positions[indexB].C = cB;
719  data.Positions[indexB].A = aB;
720 
721  return (positionError <= Settings.LinearSlop) && (angularError <= Settings.AngularSlop);
722  }
723  }
