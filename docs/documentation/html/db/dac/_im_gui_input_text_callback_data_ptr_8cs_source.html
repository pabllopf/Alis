190  {

192  int textByteCount = 0;
193  if (text != null)
194  {
195  textByteCount = Encoding.UTF8.GetByteCount(text);
196  if (textByteCount > Util.StackAllocationSizeLimit)
197  {
198  nativeText = Util.Allocate(textByteCount + 1);
199  }
200  else
201  {
202  byte* nativeTextStackBytes = stackalloc byte[textByteCount + 1];
203  nativeText = nativeTextStackBytes;
204  }
205 
206  int nativeTextOffset = Util.GetUtf8(text, nativeText, textByteCount);
207  nativeText[nativeTextOffset] = 0;
208  }
209  else
210  {
211  nativeText = null;
212  }
213 
214  byte* nativeTextEnd = null;
215  ImGuiNative.ImGuiInputTextCallbackData_InsertChars(NativePtr, pos, nativeText, nativeTextEnd);
