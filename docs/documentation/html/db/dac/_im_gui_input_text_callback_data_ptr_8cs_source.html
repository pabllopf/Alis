192  {

196  {

200  nativeText = Util.Allocate(textByteCount + 1);
201  }
202  else
203  {
204  byte* nativeTextStackBytes = stackalloc byte[textByteCount + 1];
205  nativeText = nativeTextStackBytes;
206  }
207 
208  int nativeTextOffset = Util.GetUtf8(text, nativeText, textByteCount);
209  nativeText[nativeTextOffset] = 0;
210  }
211  else

213  nativeText = null;
214  }
215 
216  byte* nativeTextEnd = null;
217  ImGuiNative.ImGuiInputTextCallbackData_InsertChars(NativePtr, pos, nativeText, nativeTextEnd);
