185  public void InsertChars(int pos, string text)
186  {
187  byte* nativeText;
188  int textByteCount = 0;
189  if (text != null)
190  {
191  textByteCount = Encoding.UTF8.GetByteCount(text);
192  if (textByteCount > Util.StackAllocationSizeLimit)
193  {
194  nativeText = Util.Allocate(textByteCount + 1);
195  }
196  else
197  {
198  byte* nativeTextStackBytes = stackalloc byte[textByteCount + 1];
199  nativeText = nativeTextStackBytes;
200  }
201 
202  int nativeTextOffset = Util.GetUtf8(text, nativeText, textByteCount);
203  nativeText[nativeTextOffset] = 0;
204  }
205  else
206  {
207  nativeText = null;
208  }
209 
210  byte* nativeTextEnd = null;
211  ImGuiNative.ImGuiInputTextCallbackData_InsertChars(NativePtr, pos, nativeText, nativeTextEnd);
212  if (textByteCount > Util.StackAllocationSizeLimit)
213  {
214  Util.Free(nativeText);
215  }
