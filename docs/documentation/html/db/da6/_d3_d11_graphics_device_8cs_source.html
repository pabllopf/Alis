437  if ((usage & TextureUsage.RenderTarget) != 0 && (fs & FormatSupport.RenderTarget) == 0
438  || (usage & TextureUsage.DepthStencil) != 0 && (fs & FormatSupport.DepthStencil) == 0
439  || (usage & TextureUsage.Sampled) != 0 && (fs & FormatSupport.ShaderSample) == 0
440  || (usage & TextureUsage.Cubemap) != 0 && (fs & FormatSupport.TextureCube) == 0

453  if (CheckFormatMultisample(dxgiFormat, 4)) { sampleCounts |= (1 << 2); }

476  protected override MappedResource MapCore(MappableResource resource, MapMode mode, uint subresource)
477  {
478  MappedResourceCacheKey key = new MappedResourceCacheKey(resource, subresource);
479  lock (_mappedResourceLock)
480  {
481  if (_mappedResources.TryGetValue(key, out MappedResourceInfo info))
482  {
483  if (info.Mode != mode)
484  {
485  throw new VeldridException("The given resource was already mapped with a different MapMode.");
486  }
487 
488  info.RefCount += 1;
489  _mappedResources[key] = info;
490  }
491  else
492  {
493  // No current mapping exists -- create one.
494 
495  if (resource is D3D11Buffer buffer)
496  {
497  lock (_immediateContextLock)
498  {
499  MappedSubresource msr = _immediateContext.Map(
500  buffer.Buffer,
501  0,
502  D3D11Formats.VdToD3D11MapMode((buffer.Usage & BufferUsage.Dynamic) == BufferUsage.Dynamic, mode),
503  Vortice.Direct3D11.MapFlags.None);
504 
505  info.MappedResource = new MappedResource(resource, mode, msr.DataPointer, buffer.SizeInBytes);
506  info.RefCount = 1;
507  info.Mode = mode;
508  _mappedResources.Add(key, info);
509  }
510  }
511  else
512  {
513  D3D11Texture texture = Util.AssertSubtype<MappableResource, D3D11Texture>(resource);
514  lock (_immediateContextLock)
515  {
516  Util.GetMipLevelAndArrayLayer(texture, subresource, out uint mipLevel, out uint arrayLayer);
517  _immediateContext.Map(
518  texture.DeviceTexture,
519  (int)mipLevel,
520  (int)arrayLayer,
521  D3D11Formats.VdToD3D11MapMode(false, mode),
522  Vortice.Direct3D11.MapFlags.None,
523  out int mipSize,
524  out MappedSubresource msr);
525 
526  info.MappedResource = new MappedResource(
527  resource,
528  mode,
529  msr.DataPointer,
530  texture.Height * (uint)msr.RowPitch,
531  subresource,
532  (uint)msr.RowPitch,
533  (uint)msr.DepthPitch);
534  info.RefCount = 1;
535  info.Mode = mode;
536  _mappedResources.Add(key, info);
537  }
538  }
539  }
540 
541  return info.MappedResource;
542  }

597  private protected unsafe override void UpdateBufferCore(DeviceBuffer buffer, uint bufferOffsetInBytes, IntPtr source, uint sizeInBytes)
598  {
599  D3D11Buffer d3dBuffer = Util.AssertSubtype<DeviceBuffer, D3D11Buffer>(buffer);
600  if (sizeInBytes == 0)
601  {
602  return;
603  }
604 
605  bool isDynamic = (buffer.Usage & BufferUsage.Dynamic) == BufferUsage.Dynamic;
606  bool isStaging = (buffer.Usage & BufferUsage.Staging) == BufferUsage.Staging;
607  bool isUniformBuffer = (buffer.Usage & BufferUsage.UniformBuffer) == BufferUsage.UniformBuffer;
608  bool updateFullBuffer = bufferOffsetInBytes == 0 && sizeInBytes == buffer.SizeInBytes;
609  bool useUpdateSubresource = (!isDynamic && !isStaging) && (!isUniformBuffer || updateFullBuffer);
610  bool useMap = (isDynamic && updateFullBuffer) || isStaging;
611 
612  if (useUpdateSubresource)
613  {
614  Box? subregion = new Box((int)bufferOffsetInBytes, 0, 0, (int)(sizeInBytes + bufferOffsetInBytes), 1, 1);
615 
616  if (isUniformBuffer)
617  {
618  subregion = null;
619  }
620 
621  lock (_immediateContextLock)
622  {
623  _immediateContext.UpdateSubresource(d3dBuffer.Buffer, 0, subregion, source, 0, 0);
624  }
625  }
626  else if (useMap)
627  {
628  MappedResource mr = MapCore(buffer, MapMode.Write, 0);
629  if (sizeInBytes < 1024)
630  {
631  Unsafe.CopyBlock((byte*)mr.Data + bufferOffsetInBytes, source.ToPointer(), sizeInBytes);
632  }
633  else
634  {
635  Buffer.MemoryCopy(
636  source.ToPointer(),
637  (byte*)mr.Data + bufferOffsetInBytes,
638  buffer.SizeInBytes,
639  sizeInBytes);
640  }
641  UnmapCore(buffer, 0);
642  }
643  else
644  {
645  D3D11Buffer staging = GetFreeStagingBuffer(sizeInBytes);
646  UpdateBuffer(staging, 0, source, sizeInBytes);
647  Box sourceRegion = new Box(0, 0, 0, (int)sizeInBytes, 1, 1);
648  lock (_immediateContextLock)
649  {
650  _immediateContext.CopySubresourceRegion(
651  d3dBuffer.Buffer, 0, (int)bufferOffsetInBytes, 0, 0,
652  staging.Buffer, 0,
653  sourceRegion);
654  }
655 
656  lock (_stagingResourcesLock)
657  {
658  _availableStagingBuffers.Add(staging);
659  }
660  }
661  }

782  public override bool WaitForFences(Fence[] fences, bool waitAll, ulong nanosecondTimeout)
783  {
784  int msTimeout;
785  if (nanosecondTimeout == ulong.MaxValue)
786  {
787  msTimeout = -1;
788  }
789  else
790  {
791  msTimeout = (int)Math.Min(nanosecondTimeout / 1_000_000, int.MaxValue);
792  }
793 
794  ManualResetEvent[] events = GetResetEventArray(fences.Length);
795  for (int i = 0; i < fences.Length; i++)
796  {
797  events[i] = Util.AssertSubtype<Fence, D3D11Fence>(fences[i]).ResetEvent;
798  }
799  bool result;
800  if (waitAll)
801  {
802  result = WaitHandle.WaitAll(events, msTimeout);
803  }
804  else
805  {
806  int index = WaitHandle.WaitAny(events, msTimeout);
807  result = index != WaitHandle.WaitTimeout;
808  }
809 
810  ReturnResetEventArray(events);
811 
812  return result;
813  }
