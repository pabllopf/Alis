64  public List<Trapezoid> FollowEdge(Edge edge)
65  {
66  List<Trapezoid> trapezoids = new List<Trapezoid>
67  {
68  Locate(edge)
69  };
70  int j = 0;
71 
72  while (edge.Q.X > trapezoids[j].RightPoint.X)
73  {
74  if (edge.IsAbove(trapezoids[j].RightPoint))
75  {
76  trapezoids.Add(trapezoids[j].UpperRight);
77  }
78  else
79  {
80  trapezoids.Add(trapezoids[j].LowerRight);
81  }
82 
83  j += 1;
84  }
85 
86  return trapezoids;
87  }
