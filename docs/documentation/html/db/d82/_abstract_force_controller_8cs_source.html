214  protected float GetDecayMultiplier(Body body)
215  {
216  float distance = (body.Position - Position).Length();
217  switch (DecayMode)
218  {
219  case DecayModes.None:
220  {
221  return 1.0f;
222  }
223  case DecayModes.Step:
224  {
225  if (distance < DecayEnd)
226  {
227  return 1.0f;
228  }
229 
230  return 0.0f;
231  }
232  case DecayModes.Linear:
233  {
234  if (distance < DecayStart)
235  {
236  return 1.0f;
237  }
238 
239  if (distance > DecayEnd)
240  {
241  return 0.0f;
242  }
243 
244  return DecayEnd - DecayStart / distance - DecayStart;
245  }
246  case DecayModes.InverseSquare:
247  {
248  if (distance < DecayStart)
249  {
250  return 1.0f;
251  }
252 
253  return 1.0f / ((distance - DecayStart) * (distance - DecayStart));
254  }
255  case DecayModes.Curve:
256  {
257  if (distance < DecayStart)
258  {
259  return 1.0f;
260  }
261 
262  return DecayCurve.Evaluate(distance - DecayStart);
263  }
264  default:
265  return 1.0f;
266  }
267  }

291  case TimingModes.Triggered:
292  {
293  if (Enabled && Triggered)
294  {
295  if (ImpulseTime < ImpulseLength)
296  {
297  ApplyForce(dt, Strength);
298  ImpulseTime += dt;
299  }
300  else
301  {
302  Triggered = false;
303  }
304  }
305 
306  break;
307  }
308  case TimingModes.Curve:
309  {
310  if (Enabled && Triggered)
311  {
312  if (ImpulseTime < ImpulseLength)
313  {
314  ApplyForce(dt, Strength * StrengthCurve.Evaluate(ImpulseTime));
315  ImpulseTime += dt;
316  }
317  else
318  {
319  Triggered = false;
320  }
321  }
322 
323  break;
324  }
325  }
