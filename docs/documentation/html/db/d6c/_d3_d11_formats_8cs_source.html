281  {

284  {

287  if ((usage & BufferUsage.IndexBuffer) == BufferUsage.IndexBuffer)
288  {
289  flags |= BindFlags.IndexBuffer;
290  }
291  if ((usage & BufferUsage.UniformBuffer) == BufferUsage.UniformBuffer)
292  {
293  flags |= BindFlags.ConstantBuffer;
294  }
295  if ((usage & BufferUsage.StructuredBufferReadOnly) == BufferUsage.StructuredBufferReadOnly
296  || (usage & BufferUsage.StructuredBufferReadWrite) == BufferUsage.StructuredBufferReadWrite)
297  {
298  flags |= BindFlags.ShaderResource;
299  }
300  if ((usage & BufferUsage.StructuredBufferReadWrite) == BufferUsage.StructuredBufferReadWrite)
301  {
302  flags |= BindFlags.UnorderedAccess;
303  }
304 
305  return flags;
306  }
