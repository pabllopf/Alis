299  private void EnlargeBufferIfRequired(int count)
300  {
301  // we cannot fit the data into the existing buffer, time for a new buffer
302  if (count > _buffer.Length - _ms.Position)
303  {
304  int position = (int) _ms.Position;
305 
306  // double the buffer size
307  long newSize = (long) _buffer.Length * 2;
308 
309  // make sure the new size is big enough
310  long requiredSize = (long) count + _buffer.Length - position;
311 
312  if (requiredSize > int.MaxValue)
313  {
314  throw new WebSocketBufferOverflowException(
315  $"Tried to create a buffer ({requiredSize:#,##0} bytes) that was larger than the max allowed size ({int.MaxValue:#,##0})");
316  }
317 
318  if (requiredSize > newSize)
319  {
320  // compute the power of two larger than requiredSize. so 40000 => 65536
321  long candidateSize = (long) Math.Pow(2, Math.Ceiling(Math.Log(requiredSize) / Math.Log(2)));
322  if (candidateSize > int.MaxValue)
323  {
324  newSize = requiredSize;
325  }
326  else
327  {
328  newSize = candidateSize;
329  }
330  }
331 
332  byte[] newBuffer = new byte[newSize];
333  Buffer.BlockCopy(_buffer, 0, newBuffer, 0, position);
334  _ms = new MemoryStream(newBuffer, 0, newBuffer.Length, true, true)
335  {
336  Position = position
337  };
338 
339  _buffer = newBuffer;
340  }
341  }
