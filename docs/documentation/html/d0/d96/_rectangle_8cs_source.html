119  public bool Intersects(Rectangle rect, out Rectangle overlap)
120  {
121  // Rectangles with negative dimensions are allowed, so we must handle them correctly
122 
123  // Compute the min and max of the first rectangle on both axes
124  float r1MinX = System.Math.Min(Left, Left + Width);
125  float r1MaxX = System.Math.Max(Left, Left + Width);
126  float r1MinY = System.Math.Min(Top, Top + Height);
127  float r1MaxY = System.Math.Max(Top, Top + Height);
128 
129  // Compute the min and max of the second rectangle on both axes
130  float r2MinX = System.Math.Min(rect.Left, rect.Left + rect.Width);
131  float r2MaxX = System.Math.Max(rect.Left, rect.Left + rect.Width);
132  float r2MinY = System.Math.Min(rect.Top, rect.Top + rect.Height);
133  float r2MaxY = System.Math.Max(rect.Top, rect.Top + rect.Height);
134 
135  // Compute the intersection boundaries
136  float interLeft = System.Math.Max(r1MinX, r2MinX);
137  float interTop = System.Math.Max(r1MinY, r2MinY);
138  float interRight = System.Math.Min(r1MaxX, r2MaxX);
139  float interBottom = System.Math.Min(r1MaxY, r2MaxY);
140 
141  // If the intersection is valid (positive non zero area), then there is an intersection
142  if ((interLeft < interRight) && (interTop < interBottom))
143  {
144  overlap.Left = interLeft;
145  overlap.Top = interTop;
146  overlap.Width = interRight - interLeft;
147  overlap.Height = interBottom - interTop;
148  return true;
149  }
150 
151  overlap.Left = 0;
152  overlap.Top = 0;
153  overlap.Width = 0;
154  overlap.Height = 0;
155  return false;
156  }
