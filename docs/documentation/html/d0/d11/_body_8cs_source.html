863  public void RemoveFixture(Fixture fixture)
864  {
865  Debug.Assert(!World.IsLocked);
866  if (World.IsLocked)
867  {
868  return;
869  }
870 
871  if (fixture == null)
872  {
873  return;
874  }
875 
876  Debug.Assert(fixture.Body == this);
877 
878  // Remove the fixture from this body's singly linked list.
879  Debug.Assert(FixtureList.Count > 0);
880 
881  // You tried to remove a fixture that not present in the fixturelist.
882  Debug.Assert(FixtureList.Contains(fixture));
883 
884  // Destroy any contacts associated with the fixture.
885  ContactEdge edge = ContactList;
886  while (edge != null)
887  {
888  Contact c = edge.Contact;
889  edge = edge.Next;
890 
891  Fixture fixtureA = c.FixtureA;
892  Fixture fixtureB = c.FixtureB;
893 
894  if (fixture == fixtureA || fixture == fixtureB)
895  {
896  // This destroys the contact and removes it from
897  // this body's contact list.
898  World.ContactManager.Remove(c);
899  }
900  }
901 
902  if ((Flags & BodyFlags.Enabled) == BodyFlags.Enabled)
903  {
904  IBroadPhase broadPhase = World.ContactManager.BroadPhase;
905  fixture.DestroyProxies(broadPhase);
906  }
907 
908  FixtureList.Remove(fixture);
909  fixture.Destroy();
910  fixture.Bodyprivate = null;
911 
912  ResetMassData();
913  }

1120  public void ResetMassData()
1121  {
1122  // Compute mass data from shapes. Each shape has its own density.
1123  mass = 0.0f;
1124  InvMass = 0.0f;
1125  inertia = 0.0f;
1126  InvI = 0.0f;
1127  Sweep.LocalCenter = Vector2.Zero;
1128 
1129  //Velcro: We have mass on static bodies to support attaching joints to them
1130  // Kinematic bodies have zero mass.
1131  if (Type == BodyType.Kinematic)
1132  {
1133  Sweep.C0 = Xf.P;
1134  Sweep.C = Xf.P;
1135  Sweep.A0 = Sweep.A;
1136  return;
1137  }
1138 
1139  Debug.Assert(Type == BodyType.Dynamic || Type == BodyType.Static);
1140 
1141  // Accumulate mass over all fixtures.
1142  Vector2 localCenter = Vector2.Zero;
1143  foreach (Fixture f in FixtureList)
1144  {
1145  if (f.Shape.DensityPrivate == 0.0f)
1146  {
1147  continue;
1148  }
1149 
1150  MassData massData = f.Shape.MassDataPrivate;
1151  mass += massData.Mass;
1152  localCenter += massData.Mass * massData.Centroid;
