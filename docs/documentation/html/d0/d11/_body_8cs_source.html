1069  public void ResetMassData()
1070  {
1071  // Compute mass data from shapes. Each shape has its own density.
1072  mass = 0.0f;
1073  InvMass = 0.0f;
1074  inertia = 0.0f;
1075  InvI = 0.0f;
1076  Sweep.LocalCenter = Vector2F.Zero;
1077 
1078  //Velcro: We have mass on static bodies to support attaching joints to them
1079  // Kinematic bodies have zero mass.
1080  if (Type == BodyType.Kinematic)
1081  {
1082  Sweep.C0 = Xf.Position;
1083  Sweep.C = Xf.Position;
1084  Sweep.A0 = Sweep.A;
1085  return;
1086  }
1087 
1088  //Debug.Assert(Type == BodyType.Dynamic || Type == BodyType.Static);
1089 
1090  // Accumulate mass over all fixtures.
1091  Vector2F localCenter = Vector2F.Zero;
1092  foreach (Fixture f in FixtureList)
1093  {
1094  if (f.Shape.DensityPrivate == 0.0f)
1095  {
1096  continue;
1097  }
1098 
1099  MassData massData = f.Shape.MassDataPrivate;
1100  mass += massData.Mass;
1101  localCenter += massData.Mass * massData.Centroid;
1102  inertia += massData.Inertia;
1103  }
1104 
1105  //Velcro: Static bodies only have mass, they don't have other properties. A little hacky tho...
1106  if (Type == BodyType.Static)
1107  {
1108  Sweep.C0 = Sweep.C = Xf.Position;
1109  return;
1110  }
1111 
1112  // Compute center of mass.
1113  if (mass > 0.0f)
1114  {
1115  InvMass = 1.0f / mass;
1116  localCenter *= InvMass;
1117  }
1118 
1119  if ((inertia > 0.0f) && ((Flags & BodyFlags.FixedRotationFlag) == 0))
1120  {
1121  // Center the inertia about the center of mass.
1122  inertia -= mass * Vector2F.Dot(localCenter, localCenter);
1123 
1124  //Debug.Assert(inertia > 0.0f);
1125  InvI = 1.0f / inertia;
1126  }
1127  else
