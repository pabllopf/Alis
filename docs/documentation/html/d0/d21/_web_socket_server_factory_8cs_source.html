125  private static void CheckWebSocketVersion(string httpHeader)
126  {
127  Regex webSocketVersionRegex = new Regex("Sec-WebSocket-Version: (.*)", RegexOptions.IgnoreCase, TimeSpan.FromMilliseconds(100));
128 
129  // check the version. Support version 13 and above
130  const int webSocketVersion = 13;
131  Match match = webSocketVersionRegex.Match(httpHeader);
132  if (match.Success)
133  {
134  int secWebSocketVersion = Convert.ToInt32(match.Groups[1].Value.Trim());
135  if (secWebSocketVersion < webSocketVersion)
136  {
137  throw new WebSocketVersionNotSupportedException(string.Format(
138  "WebSocket Version {0} not suported. Must be {1} or above", secWebSocketVersion,
139  webSocketVersion));
140  }
141  }
142  else
143  {
144  throw new WebSocketVersionNotSupportedException("Cannot find \"Sec-WebSocket-Version\" in http header");
145  }
146  }

157  private static async Task PerformHandshakeAsync(Guid guid, string httpHeader, string subProtocol, Stream stream,
158  CancellationToken token)
159  {
160  try
161  {
162  Regex webSocketKeyRegex = new Regex("Sec-WebSocket-Key: (.*)", RegexOptions.IgnoreCase, TimeSpan.FromMilliseconds(100));
163  CheckWebSocketVersion(httpHeader);
164 
165  Match match = webSocketKeyRegex.Match(httpHeader);
166  if (match.Success)
167  {
168  string secWebSocketKey = match.Groups[1].Value.Trim();
169  string setWebSocketAccept = HttpHelper.ComputeSocketAcceptString(secWebSocketKey);
170  string response = "HTTP/1.1 101 Switching Protocols\r\n"
171  + "Connection: Upgrade\r\n"
172  + "Upgrade: websocket\r\n"
173  + (subProtocol != null ? $"Sec-WebSocket-Protocol: {subProtocol}\r\n" : "")
174  + $"Sec-WebSocket-Accept: {setWebSocketAccept}";
175 
176  Events.Log.SendingHandshakeResponse(guid, response);
177  await HttpHelper.WriteHttpHeaderAsync(response, stream, token);
178  }
179  else
180  {
181  throw new SecWebSocketKeyMissingException("Unable to read \"Sec-WebSocket-Key\" from http header");
182  }
183  }
184  catch (WebSocketVersionNotSupportedException ex)
185  {
186  Events.Log.WebSocketVersionNotSupported(guid, ex.ToString());
187  string response = "HTTP/1.1 426 Upgrade Required\r\nSec-WebSocket-Version: 13" + ex.Message;
188  await HttpHelper.WriteHttpHeaderAsync(response, stream, token);
189  throw;
190  }
191  catch (Exception ex)
192  {
193  Events.Log.BadRequest(guid, ex.ToString());
194  await HttpHelper.WriteHttpHeaderAsync("HTTP/1.1 400 Bad Request", stream, token);
195  throw;
196  }
197  }
198  }
