283  internal override void InitVelocityConstraints(ref SolverData data)
284  {
285  indexA = BodyA.IslandIndex;
286  indexB = BodyB.IslandIndex;
287  localCenterA = BodyA.Sweep.LocalCenter;
288  localCenterB = BodyB.Sweep.LocalCenter;
289  invMassA = BodyA.InvMass;
290  invMassB = BodyB.InvMass;
291  invIa = BodyA.InvI;
292  invIb = BodyB.InvI;
293 
294  float aA = data.Positions[indexA].A;
295  Vector2 vA = data.Velocities[indexA].V;
296  float wA = data.Velocities[indexA].W;
297 
298  float aB = data.Positions[indexB].A;
299  Vector2 vB = data.Velocities[indexB].V;
300  float wB = data.Velocities[indexB].W;
301 
302  Rot qA = new Rot(aA), qB = new Rot(aB);
303 
304  rA = MathUtils.Mul(qA, localAnchorA - localCenterA);
305  rB = MathUtils.Mul(qB, localAnchorB - localCenterB);
306 
307  // J = [-I -r1_skew I r2_skew]
308  // [ 0 -1 0 1]
309  // r_skew = [-ry; rx]
310 
311  // Matlab
312  // K = [ mA+r1y^2*iA+mB+r2y^2*iB, -r1y*iA*r1x-r2y*iB*r2x, -r1y*iA-r2y*iB]
313  // [ -r1y*iA*r1x-r2y*iB*r2x, mA+r1x^2*iA+mB+r2x^2*iB, r1x*iA+r2x*iB]
314  // [ -r1y*iA-r2y*iB, r1x*iA+r2x*iB, iA+iB]
315 
316  float mA = invMassA, mB = invMassB;
317  float iA = invIa, iB = invIb;
318 
319  Mat33 kk;
320  kk.Ex.X = mA + mB + rA.Y * rA.Y * iA + rB.Y * rB.Y * iB;
321  kk.Ey.X = -rA.Y * rA.X * iA - rB.Y * rB.X * iB;
322  kk.Ez.X = -rA.Y * iA - rB.Y * iB;
323  kk.Ex.Y = kk.Ey.X;
324  kk.Ey.Y = mA + mB + rA.X * rA.X * iA + rB.X * rB.X * iB;
325  kk.Ez.Y = rA.X * iA + rB.X * iB;
326  kk.Ex.Z = kk.Ez.X;
327  kk.Ey.Z = kk.Ez.Y;
328  kk.Ez.Z = iA + iB;
329 
330  if (stiffness > 0.0f)
331  {
332  kk.GetInverse22(ref mass);
333 
334  float invM = iA + iB;
335 
336  float c = aB - aA - referenceAngle;
337 
338  // Damping coefficient
339  float d = damping;
340 
341  // Spring stiffness
342  float k = stiffness;
343 
344  // magic formulas
345  float h = data.Step.DeltaTime;
346  gamma = h * (d + h * k);
347  gamma = gamma != 0.0f ? 1.0f / gamma : 0.0f;
348  bias = c * h * k * gamma;
349 
350  invM += gamma;
351  mass.Ez.Z = invM != 0.0f ? 1.0f / invM : 0.0f;
352  }
353  else if (kk.Ez.Z == 0.0f)
354  {
355  kk.GetInverse22(ref mass);
356  gamma = 0.0f;
357  bias = 0.0f;
358  }
359  else
360  {
361  kk.GetSymInverse33(ref mass);
362  gamma = 0.0f;
363  bias = 0.0f;
364  }
365 
366  if (data.Step.WarmStarting)
367  {
368  // Scale impulses to support a variable time step.
369  impulse *= data.Step.DeltaTimeRatio;
370 
371  Vector2 p = new Vector2(impulse.X, impulse.Y);
372 
373  vA -= mA * p;
374  wA -= iA * (MathUtils.Cross(rA, p) + impulse.Z);
375 
376  vB += mB * p;
377  wB += iB * (MathUtils.Cross(rB, p) + impulse.Z);
378  }
379  else
380  {
381  impulse = Vector3.Zero;
382  }
383 
384  data.Velocities[indexA].V = vA;
385  data.Velocities[indexA].W = wA;
386  data.Velocities[indexB].V = vB;
387  data.Velocities[indexB].W = wB;
388  }

394  internal override void SolveVelocityConstraints(ref SolverData data)
395  {
396  Vector2 vA = data.Velocities[indexA].V;
397  float wA = data.Velocities[indexA].W;
398  Vector2 vB = data.Velocities[indexB].V;
399  float wB = data.Velocities[indexB].W;
400 
401  float mA = invMassA, mB = invMassB;
402  float iA = invIa, iB = invIb;
403 
404  if (stiffness > 0.0f)
405  {
406  float cdot2 = wB - wA;
407 
408  float impulse2 = -mass.Ez.Z * (cdot2 + bias + gamma * impulse.Z);
409  impulse.Z += impulse2;
410 
411  wA -= iA * impulse2;
412  wB += iB * impulse2;
413 
414  Vector2 cdot1 = vB + MathUtils.Cross(wB, rB) - vA - MathUtils.Cross(wA, rA);
415 
416  Vector2 impulse1 = -MathUtils.Mul22(mass, cdot1);
417  impulse.X += impulse1.X;
418  impulse.Y += impulse1.Y;
419 
420  Vector2 p = impulse1;
421 
422  vA -= mA * p;
423  wA -= iA * MathUtils.Cross(rA, p);
424 
425  vB += mB * p;
426  wB += iB * MathUtils.Cross(rB, p);
427  }
428  else
429  {
430  Vector2 cdot1 = vB + MathUtils.Cross(wB, rB) - vA - MathUtils.Cross(wA, rA);
431  float cdot2 = wB - wA;
432  Vector3 cdot = new Vector3(cdot1.X, cdot1.Y, cdot2);
433 
434  Vector3 impulse = -MathUtils.Mul(mass, cdot);
435  this.impulse += impulse;
436 
437  Vector2 p = new Vector2(impulse.X, impulse.Y);
438 
439  vA -= mA * p;
440  wA -= iA * (MathUtils.Cross(rA, p) + impulse.Z);
441 
442  vB += mB * p;
443  wB += iB * (MathUtils.Cross(rB, p) + impulse.Z);
444  }
445 
446  data.Velocities[indexA].V = vA;
447  data.Velocities[indexA].W = wA;
448  data.Velocities[indexB].V = vB;
449  data.Velocities[indexB].W = wB;
450  }

457  internal override bool SolvePositionConstraints(ref SolverData data)
458  {
459  Vector2 cA = data.Positions[indexA].C;
460  float aA = data.Positions[indexA].A;
461  Vector2 cB = data.Positions[indexB].C;
462  float aB = data.Positions[indexB].A;
463 
464  Rot qA = new Rot(aA), qB = new Rot(aB);
465 
466  float mA = invMassA, mB = invMassB;
467  float iA = invIa, iB = invIb;
468 
469  Vector2 rA = MathUtils.Mul(qA, LocalAnchorA - localCenterA);
470  Vector2 rB = MathUtils.Mul(qB, LocalAnchorB - localCenterB);
471 
472  float positionError, angularError;
473 
474  Mat33 k = new Mat33();
475  k.Ex.X = mA + mB + rA.Y * rA.Y * iA + rB.Y * rB.Y * iB;
476  k.Ey.X = -rA.Y * rA.X * iA - rB.Y * rB.X * iB;
477  k.Ez.X = -rA.Y * iA - rB.Y * iB;
478  k.Ex.Y = k.Ey.X;
479  k.Ey.Y = mA + mB + rA.X * rA.X * iA + rB.X * rB.X * iB;
480  k.Ez.Y = rA.X * iA + rB.X * iB;
481  k.Ex.Z = k.Ez.X;
482  k.Ey.Z = k.Ez.Y;
483  k.Ez.Z = iA + iB;
484 
485  if (stiffness > 0.0f)
486  {
487  Vector2 c1 = cB + rB - cA - rA;
488 
489  positionError = c1.Length();
490  angularError = 0.0f;
491 
492  Vector2 p = -k.Solve22(c1);
493 
494  cA -= mA * p;
495  aA -= iA * MathUtils.Cross(rA, p);
496 
497  cB += mB * p;
498  aB += iB * MathUtils.Cross(rB, p);
499  }
500  else
501  {
502  Vector2 c1 = cB + rB - cA - rA;
503  float c2 = aB - aA - referenceAngle;
504 
505  positionError = c1.Length();
506  angularError = Math.Abs(c2);
507 
508  Vector3 c = new Vector3(c1.X, c1.Y, c2);
509 
510  Vector3 impulse;
511  if (k.Ez.Z > 0.0f)
512  {
513  impulse = -k.Solve33(c);
514  }
515  else
516  {
517  Vector2 impulse2 = -k.Solve22(c1);
518  impulse = new Vector3(impulse2.X, impulse2.Y, 0.0f);
519  }
520 
521  Vector2 p = new Vector2(impulse.X, impulse.Y);
522 
523  cA -= mA * p;
524  aA -= iA * (MathUtils.Cross(rA, p) + impulse.Z);
525 
526  cB += mB * p;
527  aB += iB * (MathUtils.Cross(rB, p) + impulse.Z);
528  }
529 
530  data.Positions[indexA].C = cA;
531  data.Positions[indexA].A = aA;
532  data.Positions[indexB].C = cB;
533  data.Positions[indexB].A = aB;
534 
535  return positionError <= Settings.LinearSlop && angularError <= Settings.AngularSlop;
536  }
537  }
