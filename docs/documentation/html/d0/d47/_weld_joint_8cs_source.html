281  internal override void InitVelocityConstraints(ref SolverData data)
282  {
283  indexA = BodyA.IslandIndex;
284  indexB = BodyB.IslandIndex;
285  localCenterA = BodyA.Sweep.LocalCenter;
286  localCenterB = BodyB.Sweep.LocalCenter;
287  invMassA = BodyA.InvMass;
288  invMassB = BodyB.InvMass;
289  invIa = BodyA.InvI;
290  invIb = BodyB.InvI;
291 
292  float aA = data.Positions[indexA].A;
293  Vector2 vA = data.Velocities[indexA].V;
294  float wA = data.Velocities[indexA].W;
295 
296  float aB = data.Positions[indexB].A;
297  Vector2 vB = data.Velocities[indexB].V;
298  float wB = data.Velocities[indexB].W;
299 
300  Rot qA = new Rot(aA), qB = new Rot(aB);
301 
302  rA = MathUtils.Mul(qA, localAnchorA - localCenterA);
303  rB = MathUtils.Mul(qB, localAnchorB - localCenterB);
304 
305  // J = [-I -r1_skew I r2_skew]
306  // [ 0 -1 0 1]
307  // r_skew = [-ry; rx]
308 
309  // Matlab
310  // K = [ mA+r1y^2*iA+mB+r2y^2*iB, -r1y*iA*r1x-r2y*iB*r2x, -r1y*iA-r2y*iB]
311  // [ -r1y*iA*r1x-r2y*iB*r2x, mA+r1x^2*iA+mB+r2x^2*iB, r1x*iA+r2x*iB]
312  // [ -r1y*iA-r2y*iB, r1x*iA+r2x*iB, iA+iB]
313 
314  float mA = invMassA, mB = invMassB;
315  float iA = invIa, iB = invIb;
316 
317  Mat33 kk;
318  kk.Ex.X = mA + mB + rA.Y * rA.Y * iA + rB.Y * rB.Y * iB;
319  kk.Ey.X = -rA.Y * rA.X * iA - rB.Y * rB.X * iB;
320  kk.Ez.X = -rA.Y * iA - rB.Y * iB;
321  kk.Ex.Y = kk.Ey.X;
322  kk.Ey.Y = mA + mB + rA.X * rA.X * iA + rB.X * rB.X * iB;
323  kk.Ez.Y = rA.X * iA + rB.X * iB;
324  kk.Ex.Z = kk.Ez.X;
325  kk.Ey.Z = kk.Ez.Y;
326  kk.Ez.Z = iA + iB;
327 
328  if (stiffness > 0.0f)
329  {
330  kk.GetInverse22(ref mass);
331 
332  float invM = iA + iB;
333 
334  float c = aB - aA - referenceAngle;
335 
336  // Damping coefficient
337  float d = damping;
338 
339  // Spring stiffness
340  float k = stiffness;
341 
342  // magic formulas
343  float h = data.Step.DeltaTime;
344  gamma = h * (d + h * k);
345  gamma = gamma != 0.0f ? 1.0f / gamma : 0.0f;
346  bias = c * h * k * gamma;
347 
348  invM += gamma;
349  mass.Ez.Z = invM != 0.0f ? 1.0f / invM : 0.0f;
350  }
351  else if (kk.Ez.Z == 0.0f)
352  {
353  kk.GetInverse22(ref mass);
354  gamma = 0.0f;
355  bias = 0.0f;
356  }
357  else
358  {
359  kk.GetSymInverse33(ref mass);
360  gamma = 0.0f;
361  bias = 0.0f;
362  }
363 
364  if (data.Step.WarmStarting)
365  {
366  // Scale impulses to support a variable time step.
367  impulse *= data.Step.DeltaTimeRatio;
368 
369  Vector2 p = new Vector2(impulse.X, impulse.Y);
370 
371  vA -= mA * p;
372  wA -= iA * (MathUtils.Cross(rA, p) + impulse.Z);
373 
374  vB += mB * p;
375  wB += iB * (MathUtils.Cross(rB, p) + impulse.Z);
376  }
377  else
378  {
379  impulse = Vector3.Zero;
380  }
381 
382  data.Velocities[indexA].V = vA;
383  data.Velocities[indexA].W = wA;
384  data.Velocities[indexB].V = vB;
385  data.Velocities[indexB].W = wB;
386  }

392  internal override void SolveVelocityConstraints(ref SolverData data)
393  {
394  Vector2 vA = data.Velocities[indexA].V;
395  float wA = data.Velocities[indexA].W;
396  Vector2 vB = data.Velocities[indexB].V;
397  float wB = data.Velocities[indexB].W;
398 
399  float mA = invMassA, mB = invMassB;
400  float iA = invIa, iB = invIb;
401 
402  if (stiffness > 0.0f)
403  {
404  float cdot2 = wB - wA;
405 
406  float impulse2 = -mass.Ez.Z * (cdot2 + bias + gamma * impulse.Z);
407  impulse.Z += impulse2;
408 
409  wA -= iA * impulse2;
410  wB += iB * impulse2;
411 
412  Vector2 cdot1 = vB + MathUtils.Cross(wB, rB) - vA - MathUtils.Cross(wA, rA);
413 
414  Vector2 impulse1 = -MathUtils.Mul22(mass, cdot1);
415  impulse.X += impulse1.X;
416  impulse.Y += impulse1.Y;
417 
418  Vector2 p = impulse1;
419 
420  vA -= mA * p;
421  wA -= iA * MathUtils.Cross(rA, p);
422 
423  vB += mB * p;
424  wB += iB * MathUtils.Cross(rB, p);
425  }
426  else
427  {
428  Vector2 cdot1 = vB + MathUtils.Cross(wB, rB) - vA - MathUtils.Cross(wA, rA);
429  float cdot2 = wB - wA;
430  Vector3 cdot = new Vector3(cdot1.X, cdot1.Y, cdot2);
431 
432  Vector3 impulse = -MathUtils.Mul(mass, cdot);
433  this.impulse += impulse;
434 
435  Vector2 p = new Vector2(impulse.X, impulse.Y);
436 
437  vA -= mA * p;
438  wA -= iA * (MathUtils.Cross(rA, p) + impulse.Z);
439 
440  vB += mB * p;
441  wB += iB * (MathUtils.Cross(rB, p) + impulse.Z);
442  }
443 
444  data.Velocities[indexA].V = vA;
445  data.Velocities[indexA].W = wA;
446  data.Velocities[indexB].V = vB;
447  data.Velocities[indexB].W = wB;
448  }

455  internal override bool SolvePositionConstraints(ref SolverData data)
456  {
457  Vector2 cA = data.Positions[indexA].C;
458  float aA = data.Positions[indexA].A;
459  Vector2 cB = data.Positions[indexB].C;
460  float aB = data.Positions[indexB].A;
461 
462  Rot qA = new Rot(aA), qB = new Rot(aB);
463 
464  float mA = invMassA, mB = invMassB;
465  float iA = invIa, iB = invIb;
466 
467  Vector2 rA = MathUtils.Mul(qA, LocalAnchorA - localCenterA);
468  Vector2 rB = MathUtils.Mul(qB, LocalAnchorB - localCenterB);
469 
470  float positionError, angularError;
471 
472  Mat33 k = new Mat33();
473  k.Ex.X = mA + mB + rA.Y * rA.Y * iA + rB.Y * rB.Y * iB;
474  k.Ey.X = -rA.Y * rA.X * iA - rB.Y * rB.X * iB;
475  k.Ez.X = -rA.Y * iA - rB.Y * iB;
476  k.Ex.Y = k.Ey.X;
477  k.Ey.Y = mA + mB + rA.X * rA.X * iA + rB.X * rB.X * iB;
478  k.Ez.Y = rA.X * iA + rB.X * iB;
479  k.Ex.Z = k.Ez.X;
480  k.Ey.Z = k.Ez.Y;
481  k.Ez.Z = iA + iB;
482 
483  if (stiffness > 0.0f)
484  {
485  Vector2 c1 = cB + rB - cA - rA;
486 
487  positionError = c1.Length();
488  angularError = 0.0f;
489 
490  Vector2 p = -k.Solve22(c1);
491 
492  cA -= mA * p;
493  aA -= iA * MathUtils.Cross(rA, p);
494 
495  cB += mB * p;
496  aB += iB * MathUtils.Cross(rB, p);
497  }
498  else
499  {
500  Vector2 c1 = cB + rB - cA - rA;
501  float c2 = aB - aA - referenceAngle;
502 
503  positionError = c1.Length();
504  angularError = Math.Abs(c2);
505 
506  Vector3 c = new Vector3(c1.X, c1.Y, c2);
507 
508  Vector3 impulse;
509  if (k.Ez.Z > 0.0f)
510  {
511  impulse = -k.Solve33(c);
512  }
513  else
514  {
515  Vector2 impulse2 = -k.Solve22(c1);
516  impulse = new Vector3(impulse2.X, impulse2.Y, 0.0f);
517  }
518 
519  Vector2 p = new Vector2(impulse.X, impulse.Y);
520 
521  cA -= mA * p;
522  aA -= iA * (MathUtils.Cross(rA, p) + impulse.Z);
523 
524  cB += mB * p;
525  aB += iB * (MathUtils.Cross(rB, p) + impulse.Z);
526  }
527 
528  data.Positions[indexA].C = cA;
529  data.Positions[indexA].A = aA;
530  data.Positions[indexB].C = cB;
531  data.Positions[indexB].A = aB;
532 
533  return positionError <= Settings.LinearSlop && angularError <= Settings.AngularSlop;
534  }
535  }
