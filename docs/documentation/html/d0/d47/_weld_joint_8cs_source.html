298  internal override void InitVelocityConstraints(ref SolverData data)
299  {
300  indexA = BodyA.IslandIndex;
301  indexB = BodyB.IslandIndex;
302  localCenterA = BodyA.Sweep.LocalCenter;
303  localCenterB = BodyB.Sweep.LocalCenter;
304  invMassA = BodyA.InvMass;
305  invMassB = BodyB.InvMass;
306  invIa = BodyA.InvI;
307  invIb = BodyB.InvI;
308 
309  float aA = data.Positions[indexA].A;
310  Vector2F vA = data.Velocities[indexA].V;
311  float wA = data.Velocities[indexA].W;
312 
313  float aB = data.Positions[indexB].A;
314  Vector2F vB = data.Velocities[indexB].V;
315  float wB = data.Velocities[indexB].W;
316 
317  Rotation qA = new Rotation(aA), qB = new Rotation(aB);
318 
319  rA = MathUtils.Mul(qA, localAnchorA - localCenterA);
320  rB = MathUtils.Mul(qB, localAnchorB - localCenterB);
321 
322  // J = [-I -r1_skew I r2_skew]
323  // [ 0 -1 0 1]
324  // r_skew = [-ry; rx]
325 
326  // Matlab
327  // K = [ mA+r1y^2*iA+mB+r2y^2*iB, -r1y*iA*r1x-r2y*iB*r2x, -r1y*iA-r2y*iB]
328  // [ -r1y*iA*r1x-r2y*iB*r2x, mA+r1x^2*iA+mB+r2x^2*iB, r1x*iA+r2x*iB]
329  // [ -r1y*iA-r2y*iB, r1x*iA+r2x*iB, iA+iB]
330 
331  float mA = invMassA, mB = invMassB;
332  float iA = invIa, iB = invIb;
333 
334  Matrix3X3F kk = new Matrix3X3F(
335  mA + mB + rA.Y * rA.Y * iA + rB.Y * rB.Y * iB,
336  -rA.Y * rA.X * iA - rB.Y * rB.X * iB,
337  -rA.Y * iA - rB.Y * iB,
338  -rA.Y * rA.X * iA - rB.Y * rB.X * iB,
339  mA + mB + rA.X * rA.X * iA + rB.X * rB.X * iB,
340  rA.X * iA + rB.X * iB,
341  -rA.Y * iA - rB.Y * iB,
342  rA.X * iA + rB.X * iB,
343  iA + iB
344  );
345 
346  if (stiffness > 0.0f)
347  {
348  kk.GetInverse22(ref mass);
349 
350  float invM = iA + iB;
351 
352  float c = aB - aA - referenceAngle;
353 
354  // Damping coefficient
355  float d = damping;
356 
357  // Spring stiffness
358  float k = stiffness;
359 
360  // magic formulas
361  float h = data.Step.DeltaTime;
362  gamma = h * (d + h * k);
363  gamma = gamma != 0.0f ? 1.0f / gamma : 0.0f;
364  bias = c * h * k * gamma;
365 
366  invM += gamma;
367  mass.Ez = new Vector3F(
368  mass.Ez.X,
369  mass.Ez.Y,
370  invM != 0.0f ? 1.0f / invM : 0.0f
371  );
372  }
373  else if (kk.Ez.Z == 0.0f)
374  {
375  kk.GetInverse22(ref mass);
376  gamma = 0.0f;
377  bias = 0.0f;
378  }
379  else
380  {
381  kk.GetSymInverse33(ref mass);
382  gamma = 0.0f;
383  bias = 0.0f;
384  }
385 
386  if (data.Step.WarmStarting)
387  {
388  // Scale impulses to support a variable time step.
389  impulse *= data.Step.DeltaTimeRatio;
390 
391  Vector2F p = new Vector2F(impulse.X, impulse.Y);
392 
393  vA -= mA * p;
394  wA -= iA * (MathUtils.Cross(rA, p) + impulse.Z);
395 
396  vB += mB * p;
397  wB += iB * (MathUtils.Cross(rB, p) + impulse.Z);
398  }
399  else
400  {
401  impulse = Vector3F.Zero;
402  }
403 
404  data.Velocities[indexA].V = vA;
405  data.Velocities[indexA].W = wA;
406  data.Velocities[indexB].V = vB;
407  data.Velocities[indexB].W = wB;
408  }

414  internal override void SolveVelocityConstraints(ref SolverData data)
415  {
416  Vector2F vA = data.Velocities[indexA].V;
417  float wA = data.Velocities[indexA].W;
418  Vector2F vB = data.Velocities[indexB].V;
419  float wB = data.Velocities[indexB].W;
420 
421  float mA = invMassA, mB = invMassB;
422  float iA = invIa, iB = invIb;
423 
424  if (stiffness > 0.0f)
425  {
426  float cdot2 = wB - wA;
427 
428  float impulse2 = -mass.Ez.Z * (cdot2 + bias + gamma * impulse.Z);
429  impulse = new Vector3F(
430  impulse.X,
431  impulse.Y,
432  impulse2
433  );
434 
435  wA -= iA * impulse2;
436  wB += iB * impulse2;
437 
438  Vector2F cdot1 = vB + MathUtils.Cross(wB, rB) - vA - MathUtils.Cross(wA, rA);
439 
440  Vector2F impulse1 = -MathUtils.Mul22(mass, cdot1);
441  impulse = new Vector3F(
442  impulse1.X,
443  impulse1.Y,
444  impulse.Z
445  );
446 
447  Vector2F p = impulse1;
448 
449  vA -= mA * p;
450  wA -= iA * MathUtils.Cross(rA, p);
451 
452  vB += mB * p;
453  wB += iB * MathUtils.Cross(rB, p);
454  }
455  else
456  {
457  Vector2F cdot1 = vB + MathUtils.Cross(wB, rB) - vA - MathUtils.Cross(wA, rA);
458  float cdot2 = wB - wA;
459  Vector3F cdot = new Vector3F(cdot1.X, cdot1.Y, cdot2);
460 
461  Vector3F impulse = -MathUtils.Mul(mass, cdot);
462  this.impulse += impulse;
463 
464  Vector2F p = new Vector2F(impulse.X, impulse.Y);
465 
466  vA -= mA * p;
467  wA -= iA * (MathUtils.Cross(rA, p) + impulse.Z);
468 
469  vB += mB * p;
470  wB += iB * (MathUtils.Cross(rB, p) + impulse.Z);
471  }
472 
473  data.Velocities[indexA].V = vA;
474  data.Velocities[indexA].W = wA;
475  data.Velocities[indexB].V = vB;
476  data.Velocities[indexB].W = wB;
477  }

484  internal override bool SolvePositionConstraints(ref SolverData data)
485  {
486  Vector2F cA = data.Positions[indexA].C;
487  float aA = data.Positions[indexA].A;
488  Vector2F cB = data.Positions[indexB].C;
489  float aB = data.Positions[indexB].A;
490 
491  Rotation qA = new Rotation(aA), qB = new Rotation(aB);
492 
493  float mA = invMassA, mB = invMassB;
494  float iA = invIa, iB = invIb;
495 
496  Vector2F rA = MathUtils.Mul(qA, LocalAnchorA - localCenterA);
497  Vector2F rB = MathUtils.Mul(qB, LocalAnchorB - localCenterB);
498 
499  float positionError, angularError;
500 
501  Matrix3X3F k = new Matrix3X3F(
502  mA + mB + rA.Y * rA.Y * iA + rB.Y * rB.Y * iB,
503  -rA.Y * rA.X * iA - rB.Y * rB.X * iB,
504  -rA.Y * iA - rB.Y * iB,
505  -rA.Y * rA.X * iA - rB.Y * rB.X * iB,
506  mA + mB + rA.X * rA.X * iA + rB.X * rB.X * iB,
507  rA.X * iA + rB.X * iB,
508  -rA.Y * iA - rB.Y * iB,
509  rA.X * iA + rB.X * iB,
510  iA + iB
511  );
512 
513  if (stiffness > 0.0f)
514  {
515  Vector2F c1 = cB + rB - cA - rA;
516 
517  positionError = c1.Length();
518  angularError = 0.0f;
519 
520  Vector2F p = -k.Solve22(c1);
521 
522  cA -= mA * p;
523  aA -= iA * MathUtils.Cross(rA, p);
524 
525  cB += mB * p;
526  aB += iB * MathUtils.Cross(rB, p);
527  }
528  else
529  {
530  Vector2F c1 = cB + rB - cA - rA;
531  float c2 = aB - aA - referenceAngle;
532 
533  positionError = c1.Length();
534  angularError = Math.Abs(c2);
535 
536  Vector3F c = new Vector3F(c1.X, c1.Y, c2);
537 
538  Vector3F impulse;
539  if (k.Ez.Z > 0.0f)
540  {
541  impulse = -k.Solve33(c);
542  }
543  else
544  {
545  Vector2F impulse2 = -k.Solve22(c1);
546  impulse = new Vector3F(impulse2.X, impulse2.Y, 0.0f);
547  }
548 
549  Vector2F p = new Vector2F(impulse.X, impulse.Y);
550 
551  cA -= mA * p;
552  aA -= iA * (MathUtils.Cross(rA, p) + impulse.Z);
553 
554  cB += mB * p;
555  aB += iB * (MathUtils.Cross(rB, p) + impulse.Z);
556  }
557 
558  data.Positions[indexA].C = cA;
559  data.Positions[indexA].A = aA;
560  data.Positions[indexB].C = cB;
561  data.Positions[indexB].A = aB;
562 
563  return (positionError <= Settings.LinearSlop) && (angularError <= Settings.AngularSlop);
564  }
565  }
