49  internal static Body[] AdvanceBody(ContactManager contactManager, Island island, Contact minContact, float minAlpha)
50  {
51  // Advance the bodies to the TOI.
52  Fixture fA1 = minContact.FixtureA;
53  Fixture fB1 = minContact.FixtureB;
54  Body bA0 = fA1.Body;
55  Body bB0 = fB1.Body;
56 
57  Body[] bodies = {bA0, bB0};
58 
59  Sweep backup1 = bA0.Sweep;
60  Sweep backup2 = bB0.Sweep;
61 
62  bA0.Advance(minAlpha);
63  bB0.Advance(minAlpha);
64 
65  // The TOI contact likely has some new contact points.
66  minContact.Update(contactManager);
67  minContact.Flags &= ~ContactFlags.ToiFlag;
68  ++minContact.ToiCount;
69 
70  // Is the contact solid?
71  if (!minContact.Enabled || !minContact.IsTouching)
72  {
73  // Restore the sweeps.
74  minContact.Flags &= ~ContactFlags.EnabledFlag;
75  bA0.Sweep = backup1;
76  bB0.Sweep = backup2;
77  bA0.SynchronizeTransform();
78  bB0.SynchronizeTransform();
79  return bodies;
80  }
81 
82  bA0.Awake = true;
83  bB0.Awake = true;
84 
85  // Build the island
86  island.Clear();
87  island.Add(bA0);
88  island.Add(bB0);
89  island.Add(minContact);
90 
91  bA0.Flags |= BodyFlags.IslandFlag;
92  bB0.Flags |= BodyFlags.IslandFlag;
93  minContact.Flags &= ~ContactFlags.IslandFlag;
94 
95  // Get contacts on bodyA and bodyB;
96  for (int i = 0; i < 2; ++i)
97  {
98  Body body = bodies[i];
99  if (body.BodyType == BodyType.Dynamic)
100  {
101  for (ContactEdge ce = body.ContactList; ce != null; ce = ce.Next)
102  {
103  Contact contact = ce.Contact;
104 
105  // Has this contact already been added to the island?
106  if (contact.IslandFlag)
107  {
108  continue;
109  }
110 
111  // Only add static, kinematic, or bullet bodies.
112  Body other = ce.Other;
113  if ((other.BodyType == BodyType.Dynamic) &&
114  !body.IsBullet && !other.IsBullet)
115  {
116  continue;
117  }
118 
119  // Skip sensors.
120  bool sensorA = contact.FixtureA.IsSensorPrivate;
121  bool sensorB = contact.FixtureB.IsSensorPrivate;
122  if (sensorA || sensorB)
123  {
124  continue;
125  }
126 
127  // Tentatively advance the body to the TOI.
128  Sweep backup = other.Sweep;
129  if (!other.IsIsland)
130  {
131  other.Advance(minAlpha);
132  }
133 
134  // Update the contact points
135  contact.Update(contactManager);
136 
137  // Was the contact disabled by the user?
138  if (!contact.Enabled)
139  {
140  other.Sweep = backup;
141  other.SynchronizeTransform();
142  continue;
143  }
144 
145  // Are there contact points?
146  if (!contact.IsTouching)
147  {
148  other.Sweep = backup;
149  other.SynchronizeTransform();
150  continue;
151  }
152 
153  // Add the contact to the island
154  minContact.Flags |= ContactFlags.IslandFlag;
155  island.Add(contact);
156 
157  // Has the other body already been added to the island?
158  if (other.IsIsland)
159  {
160  continue;
161  }
162 
163  // Add the other body to the island.
164  other.Flags |= BodyFlags.IslandFlag;
165 
166  if (other.BodyType != BodyType.Static)
167  {
168  other.Awake = true;
169  }
170 
171  island.Add(other);
172  }
173  }
174  }
175 
176  return bodies;
177  }
178  }
