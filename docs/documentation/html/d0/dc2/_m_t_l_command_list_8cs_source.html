611  Texture source, uint srcX, uint srcY, uint srcZ, uint srcMipLevel, uint srcBaseArrayLayer,

613  uint width, uint height, uint depth, uint layerCount)

617  MTLTexture dstMTLTexture = Util.AssertSubtype<Texture, MTLTexture>(destination);
618 
619  bool srcIsStaging = (source.Usage & TextureUsage.Staging) != 0;
620  bool dstIsStaging = (destination.Usage & TextureUsage.Staging) != 0;
621  if (srcIsStaging && !dstIsStaging)

623  // Staging -> Normal
624  MetalBindings.MTLBuffer srcBuffer = srcMTLTexture.StagingBuffer;
625  MetalBindings.MTLTexture dstTexture = dstMTLTexture.DeviceTexture;
626 
627  Util.GetMipDimensions(srcMTLTexture, srcMipLevel, out uint mipWidth, out uint mipHeight, out uint mipDepth);
628  for (uint layer = 0; layer < layerCount; layer++)
629  {
630  uint blockSize = FormatHelpers.IsCompressedFormat(srcMTLTexture.Format) ? 4u : 1u;
631  uint compressedSrcX = srcX / blockSize;
632  uint compressedSrcY = srcY / blockSize;
633  uint blockSizeInBytes = blockSize == 1
634  ? FormatSizeHelpers.GetSizeInBytes(srcMTLTexture.Format)
635  : FormatHelpers.GetBlockSizeInBytes(srcMTLTexture.Format);
636 
637  ulong srcSubresourceBase = Util.ComputeSubresourceOffset(
638  srcMTLTexture,
639  srcMipLevel,
640  layer + srcBaseArrayLayer);
641  srcMTLTexture.GetSubresourceLayout(
642  srcMipLevel,
643  srcBaseArrayLayer + layer,
644  out uint srcRowPitch,
645  out uint srcDepthPitch);
646  ulong sourceOffset = srcSubresourceBase
647  + srcDepthPitch * srcZ
648  + srcRowPitch * compressedSrcY
649  + blockSizeInBytes * compressedSrcX;
650 
651  uint copyWidth = width > mipWidth && width <= blockSize
652  ? mipWidth
653  : width;
654 
655  uint copyHeight = height > mipHeight && height <= blockSize
656  ? mipHeight
657  : height;

660  if (dstMTLTexture.Type != TextureType.Texture3D)

665  srcBuffer,

668  (UIntPtr)srcDepthPitch,
669  sourceSize,
670  dstTexture,
671  (UIntPtr)(dstBaseArrayLayer + layer),
672  (UIntPtr)dstMipLevel,
673  new MTLOrigin(dstX, dstY, dstZ));
674  }
675  }
676  else if (srcIsStaging && dstIsStaging)
677  {
678  for (uint layer = 0; layer < layerCount; layer++)
679  {
680  // Staging -> Staging

686  srcMipLevel,
687  srcBaseArrayLayer + layer,

689  out uint srcDepthPitch);

692  dstMTLTexture,

699  out uint dstDepthPitch);

702  if (blockSize == 1)

707  for (uint yy = 0; yy < height; yy++)

739  for (uint row = 0; row < numRows; row++)
