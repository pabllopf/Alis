610  protected override void CopyTextureCore(
611  Texture source, uint srcX, uint srcY, uint srcZ, uint srcMipLevel, uint srcBaseArrayLayer,
612  Texture destination, uint dstX, uint dstY, uint dstZ, uint dstMipLevel, uint dstBaseArrayLayer,
613  uint width, uint height, uint depth, uint layerCount)
614  {
615  EnsureBlitEncoder();
616  MTLTexture srcMTLTexture = Util.AssertSubtype<Texture, MTLTexture>(source);
617  MTLTexture dstMTLTexture = Util.AssertSubtype<Texture, MTLTexture>(destination);
618 
619  bool srcIsStaging = (source.Usage & TextureUsage.Staging) != 0;
620  bool dstIsStaging = (destination.Usage & TextureUsage.Staging) != 0;
621  if (srcIsStaging && !dstIsStaging)
622  {
623  // Staging -> Normal
624  Metal.MTLBuffer srcBuffer = srcMTLTexture.StagingBuffer;
625  Metal.MTLTexture dstTexture = dstMTLTexture.DeviceTexture;
626 
627  Util.GetMipDimensions(srcMTLTexture, srcMipLevel, out uint mipWidth, out uint mipHeight, out uint mipDepth);
628  for (uint layer = 0; layer < layerCount; layer++)
629  {
630  uint blockSize = FormatHelpers.IsCompressedFormat(srcMTLTexture.Format) ? 4u : 1u;
631  uint compressedSrcX = srcX / blockSize;
632  uint compressedSrcY = srcY / blockSize;
633  uint blockSizeInBytes = blockSize == 1
634  ? FormatSizeHelpers.GetSizeInBytes(srcMTLTexture.Format)
635  : FormatHelpers.GetBlockSizeInBytes(srcMTLTexture.Format);
636 
637  ulong srcSubresourceBase = Util.ComputeSubresourceOffset(
638  srcMTLTexture,
639  srcMipLevel,
640  layer + srcBaseArrayLayer);
641  srcMTLTexture.GetSubresourceLayout(
642  srcMipLevel,
643  srcBaseArrayLayer + layer,
644  out uint srcRowPitch,
645  out uint srcDepthPitch);
646  ulong sourceOffset = srcSubresourceBase
647  + srcDepthPitch * srcZ
648  + srcRowPitch * compressedSrcY
649  + blockSizeInBytes * compressedSrcX;
650 
651  uint copyWidth = width > mipWidth && width <= blockSize
652  ? mipWidth
653  : width;
654 
655  uint copyHeight = height > mipHeight && height <= blockSize
656  ? mipHeight
657  : height;
658 
659  MTLSize sourceSize = new MTLSize(copyWidth, copyHeight, depth);
660  if (dstMTLTexture.Type != TextureType.Texture3D)
661  {
662  srcDepthPitch = 0;
663  }
664  _bce.copyFromBuffer(
665  srcBuffer,
666  (UIntPtr)sourceOffset,
667  (UIntPtr)srcRowPitch,
668  (UIntPtr)srcDepthPitch,
669  sourceSize,
670  dstTexture,
671  (UIntPtr)(dstBaseArrayLayer + layer),
672  (UIntPtr)dstMipLevel,
673  new MTLOrigin(dstX, dstY, dstZ));
674  }
675  }
676  else if (srcIsStaging && dstIsStaging)
677  {
678  for (uint layer = 0; layer < layerCount; layer++)
679  {
680  // Staging -> Staging
681  ulong srcSubresourceBase = Util.ComputeSubresourceOffset(
682  srcMTLTexture,
683  srcMipLevel,
684  layer + srcBaseArrayLayer);
685  srcMTLTexture.GetSubresourceLayout(
686  srcMipLevel,
687  srcBaseArrayLayer + layer,
688  out uint srcRowPitch,
689  out uint srcDepthPitch);
690 
691  ulong dstSubresourceBase = Util.ComputeSubresourceOffset(
692  dstMTLTexture,
693  dstMipLevel,
694  layer + dstBaseArrayLayer);
695  dstMTLTexture.GetSubresourceLayout(
696  dstMipLevel,
697  dstBaseArrayLayer + layer,
698  out uint dstRowPitch,
699  out uint dstDepthPitch);
700 
701  uint blockSize = FormatHelpers.IsCompressedFormat(dstMTLTexture.Format) ? 4u : 1u;
702  if (blockSize == 1)
703  {
704  uint pixelSize = FormatSizeHelpers.GetSizeInBytes(dstMTLTexture.Format);
705  uint copySize = width * pixelSize;
706  for (uint zz = 0; zz < depth; zz++)
707  for (uint yy = 0; yy < height; yy++)
708  {
709  ulong srcRowOffset = srcSubresourceBase
710  + srcDepthPitch * (zz + srcZ)
711  + srcRowPitch * (yy + srcY)
712  + pixelSize * srcX;
713  ulong dstRowOffset = dstSubresourceBase
714  + dstDepthPitch * (zz + dstZ)
715  + dstRowPitch * (yy + dstY)

723  }
724  }

726  {

728  uint paddedHeight = Math.Max(blockSize, height);
729  uint numRows = FormatHelpers.GetNumRows(paddedHeight, srcMTLTexture.Format);
730  uint rowPitch = FormatHelpers.GetRowPitch(paddedWidth, srcMTLTexture.Format);
731

733  uint compressedSrcY = srcY / 4;
734  uint compressedDstX = dstX / 4;
735  uint compressedDstY = dstY / 4;
736  uint blockSizeInBytes = FormatHelpers.GetBlockSizeInBytes(srcMTLTexture.Format);
737 
738  for (uint zz = 0; zz < depth; zz++)
739  for (uint row = 0; row < numRows; row++)
740  {

742  + srcDepthPitch * (zz + srcZ)
743  + srcRowPitch * (row + compressedSrcY)
744  + blockSizeInBytes * compressedSrcX;

746  + dstDepthPitch * (zz + dstZ)

774  uint bufferRowLength = Math.Max(mipWidth, blockSize);
775  uint bufferImageHeight = Math.Max(mipHeight, blockSize);

777  uint compressedDstY = dstY / blockSize;
778  uint blockSizeInBytes = blockSize == 1
779  ? FormatSizeHelpers.GetSizeInBytes(srcMTLTexture.Format)
780  : FormatHelpers.GetBlockSizeInBytes(srcMTLTexture.Format);
781  uint rowPitch = FormatHelpers.GetRowPitch(bufferRowLength, srcMTLTexture.Format);
782  uint depthPitch = FormatHelpers.GetDepthPitch(rowPitch, bufferImageHeight, srcMTLTexture.Format);
783 
784  ulong dstOffset = Util.ComputeSubresourceOffset(dstMTLTexture, dstMipLevel, dstBaseArrayLayer + layer)
785  + (dstZ * depthPitch)

791  (UIntPtr)(srcBaseArrayLayer + layer),
792  (UIntPtr)srcMipLevel,
793  srcOrigin,
794  srcSize,
795  dstMTLTexture.StagingBuffer,
796  (UIntPtr)dstOffset,
797  (UIntPtr)dstBytesPerRow,
798  (UIntPtr)dstBytesPerImage);
799  }

805  {
806  _bce.copyFromTexture(
807  srcMTLTexture.DeviceTexture,
808  (UIntPtr)(srcBaseArrayLayer + layer),
809  (UIntPtr)srcMipLevel,

811  new MTLSize(width, height, depth),
