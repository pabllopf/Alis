325  private bool GetData(ref Chunk dataChunk, IntPtr userData)
326  {
327  if (OnGetData(out myTempBuffer))
328  {
329  unsafe
330  {
331  fixed (short* samplesPtr = myTempBuffer)
332  {
333  dataChunk.samples = samplesPtr;
334  dataChunk.sampleCount = (uint) myTempBuffer.Length;
335  }
336  }
337 
338  return true;
