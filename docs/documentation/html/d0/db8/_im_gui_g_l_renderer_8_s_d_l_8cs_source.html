86  public void NewFrame()
87  {
88  ImGui.NewFrame();
89  ImGuiIoPtr io = ImGui.GetIo();
90 
91  // Setup display size (every frame to accommodate for window resizing)
92  SDL_GetWindowSize(_window, out int w, out int h);
93  SDL_GL_GetDrawableSize(_window, out int displayW, out int displayH);
94  io.DisplaySize = new Vector2(w, h);
95  if ((w > 0) && (h > 0))
96  {
97  io.DisplayFramebufferScale = new Vector2((float) displayW / w, (float) displayH / h);
98  }
99 
100  // Setup time step (we don't use SDL_GetTicks() because it is using millisecond resolution)
101  ulong frequency = SDL_GetPerformanceFrequency();
102  ulong currentTime = SDL_GetPerformanceCounter();
103  io.DeltaTime = _time > 0 ? (float) ((double) (currentTime - _time) / frequency) : 1.0f / 60.0f;
104  if (io.DeltaTime <= 0)
105  {
106  io.DeltaTime = 0.016f;
107  }
108 
109  _time = currentTime;
110 
111  UpdateMousePosAndButtons();
112  }

118  public unsafe void ProcessEvent(SdlEvent evt)
119  {
120  ImGuiIoPtr io = ImGui.GetIo();
121  switch (evt.type)
122  {
123  case SdlEventType.SdlMousewheel:
124  {
125  if (evt.wheel.x > 0)
126  {
127  io.MouseWheelH += 1;
128  }
129 
130  if (evt.wheel.x < 0)
131  {
132  io.MouseWheelH -= 1;
133  }
134 
135  if (evt.wheel.y > 0)
136  {
137  io.MouseWheel += 1;
138  }
139 
140  if (evt.wheel.y < 0)
141  {
142  io.MouseWheel -= 1;
143  }
144 
145  return;
146  }
147  case SdlEventType.SdlMousebuttondown:
148  {
149  if (evt.button.button == SdlButtonLeft)
150  {
151  _mousePressed[0] = true;
152  }
153 
154  if (evt.button.button == SdlButtonRight)
155  {
156  _mousePressed[1] = true;
157  }
158 
159  if (evt.button.button == SdlButtonMiddle)
160  {
161  _mousePressed[2] = true;
162  }
163 
164  return;
165  }
166  case SdlEventType.SdlTextinput:
167  {
168  string str = new string((sbyte*) evt.text.text);
169  io.AddInputCharactersUtf8(str);
170  return;
171  }
172  case SdlEventType.SdlKeydown:
173  case SdlEventType.SdlKeyup:
174  {
175  SdlScancode key = evt.key.keysym.scancode;
176  io.KeysDown[(int) key] = evt.type == SdlEventType.SdlKeydown;
177  io.KeyShift = (SDL_GetModState() & SdlKeymod.KmodShift) != 0;
178  io.KeyCtrl = (SDL_GetModState() & SdlKeymod.KmodCtrl) != 0;
179  io.KeyAlt = (SDL_GetModState() & SdlKeymod.KmodAlt) != 0;
180  io.KeySuper = (SDL_GetModState() & SdlKeymod.KmodGui) != 0;
181  break;
182  }
183  }
184  }

189  private void UpdateMousePosAndButtons()
190  {
191  ImGuiIoPtr io = ImGui.GetIo();
192 
193  // Set OS mouse position if requested (rarely used, only when ImGuiConfigFlags_NavEnableSetMousePos is enabled by user)
194  if (io.WantSetMousePos)
195  {
196  SDL_WarpMouseInWindow(_window, (int) io.MousePos.X, (int) io.MousePos.Y);
197  }
198  else
199  {
200  io.MousePos = new Vector2(float.MinValue, float.MinValue);
201  }
202 
203  uint mouseButtons = SDL_GetMouseState(out int mx, out int my);
204  io.MouseDown[0] =
205  _mousePressed[0] ||
206  (mouseButtons & SDL_BUTTON(SdlButtonLeft)) !=
207  0; // If a mouse press event came, always pass it as "mouse held this frame", so we don't miss click-release events that are shorter than 1 frame.
208  io.MouseDown[1] = _mousePressed[1] || (mouseButtons & SDL_BUTTON(SdlButtonRight)) != 0;
209  io.MouseDown[2] = _mousePressed[2] || (mouseButtons & SDL_BUTTON(SdlButtonMiddle)) != 0;
210  _mousePressed[0] = _mousePressed[1] = _mousePressed[2] = false;
211 
212  IntPtr focusedWindow = SDL_GetKeyboardFocus();
213  if (_window == focusedWindow)
214  {
215  // SDL_GetMouseState() gives mouse position seemingly based on the last window entered/focused(?)
216  // The creation of a new windows at runtime and SDL_CaptureMouse both seems to severely mess up with that, so we retrieve that position globally.
217  SDL_GetWindowPosition(focusedWindow, out int wx, out int wy);
218  SDL_GetGlobalMouseState(out mx, out my);
219  mx -= wx;
220  my -= wy;
221  io.MousePos = new Vector2(mx, my);
222  }
223 
224  // SDL_CaptureMouse() let the OS know e.g. that our imgui drag outside the SDL window boundaries shouldn't e.g. trigger the OS window resize cursor.
225  bool anyMouseButtonDown = ImGui.IsAnyMouseDown();
226  SDL_CaptureMouse(anyMouseButtonDown ? SdlBool.SdlTrue : SdlBool.SdlFalse);
227  }
