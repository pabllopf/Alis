88  public void NewFrame()
89  {
90  ImGui.NewFrame();
91  ImGuiIoPtr io = ImGui.GetIo();
92 
93  // Setup display size (every frame to accommodate for window resizing)
94  SDL_GetWindowSize(_window, out int w, out int h);
95  SDL_GL_GetDrawableSize(_window, out int displayW, out int displayH);
96  io.DisplaySize = new Vector2(w, h);
97  if ((w > 0) && (h > 0))
98  {
99  io.DisplayFramebufferScale = new Vector2((float) displayW / w, (float) displayH / h);
100  }
101 
102  // Setup time step (we don't use SDL_GetTicks() because it is using millisecond resolution)
103  ulong frequency = SDL_GetPerformanceFrequency();
104  ulong currentTime = SDL_GetPerformanceCounter();
105  io.DeltaTime = _time > 0 ? (float) ((double) (currentTime - _time) / frequency) : 1.0f / 60.0f;
106  if (io.DeltaTime <= 0)
107  {
108  io.DeltaTime = 0.016f;
109  }
110 
111  _time = currentTime;
112 
113  UpdateMousePosAndButtons();
114  }

120  public void ProcessEvent(SdlEvent evt)
121  {
122  ImGuiIoPtr io = ImGui.GetIo();
123  switch (evt.type)
124  {
125  case SdlEventType.SdlMousewheel:
126  {
127  if (evt.wheel.x > 0)
128  {
129  io.MouseWheelH += 1;
130  }
131 
132  if (evt.wheel.x < 0)
133  {
134  io.MouseWheelH -= 1;
135  }
136 
137  if (evt.wheel.y > 0)
138  {
139  io.MouseWheel += 1;
140  }
141 
142  if (evt.wheel.y < 0)
143  {
144  io.MouseWheel -= 1;
145  }
146 
147  return;
148  }
149  case SdlEventType.SdlMousebuttondown:
150  {
151  if (evt.button.button == SdlButtonLeft)
152  {
153  _mousePressed[0] = true;
154  }
155 
156  if (evt.button.button == SdlButtonRight)
157  {
158  _mousePressed[1] = true;
159  }
160 
161  if (evt.button.button == SdlButtonMiddle)
162  {
163  _mousePressed[2] = true;
164  }
165 
166  return;
167  }
168  case SdlEventType.SdlTextinput:
169  {
170  string str = Encoding.UTF8.GetString(evt.text.text);
171  io.AddInputCharactersUtf8(str);
172  return;
173  }
174  case SdlEventType.SdlKeydown:
175  case SdlEventType.SdlKeyup:
176  {
177  SdlScancode key = evt.key.keysym.scancode;
178  io.KeysDown[(int) key] = evt.type == SdlEventType.SdlKeydown;
179  Console.WriteLine("io.KeysDown[" + key + "] = " + evt.type + io.KeysDown[(int) key]);
180  io.KeyShift = (SDL_GetModState() & SdlKeymod.KmodShift) != 0;
181  io.KeyCtrl = (SDL_GetModState() & SdlKeymod.KmodCtrl) != 0;
182  io.KeyAlt = (SDL_GetModState() & SdlKeymod.KmodAlt) != 0;
183  io.KeySuper = (SDL_GetModState() & SdlKeymod.KmodGui) != 0;
184  break;
185  }
186  }
187  }

192  private void UpdateMousePosAndButtons()
193  {
194  ImGuiIoPtr io = ImGui.GetIo();
195 
196  // Set OS mouse position if requested (rarely used, only when ImGuiConfigFlags_NavEnableSetMousePos is enabled by user)
197  if (io.WantSetMousePos)
198  {
199  SDL_WarpMouseInWindow(_window, (int) io.MousePos.X, (int) io.MousePos.Y);
200  }
201  else
202  {
203  io.MousePos = new Vector2(float.MinValue, float.MinValue);
204  }
205 
206  uint mouseButtons = SDL_GetMouseState(out int mx, out int my);
207  io.MouseDown[0] =
208  _mousePressed[0] ||
209  (mouseButtons & SDL_BUTTON(SdlButtonLeft)) !=
210  0; // If a mouse press event came, always pass it as "mouse held this frame", so we don't miss click-release events that are shorter than 1 frame.
211  io.MouseDown[1] = _mousePressed[1] || (mouseButtons & SDL_BUTTON(SdlButtonRight)) != 0;
212  io.MouseDown[2] = _mousePressed[2] || (mouseButtons & SDL_BUTTON(SdlButtonMiddle)) != 0;
213  _mousePressed[0] = _mousePressed[1] = _mousePressed[2] = false;
214 
215  IntPtr focusedWindow = SDL_GetKeyboardFocus();
216  if (_window == focusedWindow)
217  {
218  // SDL_GetMouseState() gives mouse position seemingly based on the last window entered/focused(?)
219  // The creation of a new windows at runtime and SDL_CaptureMouse both seems to severely mess up with that, so we retrieve that position globally.
220  SDL_GetWindowPosition(focusedWindow, out int wx, out int wy);
221  SDL_GetGlobalMouseState(out mx, out my);
222  mx -= wx;
223  my -= wy;
224  io.MousePos = new Vector2(mx, my);
225  }
226 
227  // SDL_CaptureMouse() let the OS know e.g. that our imgui drag outside the SDL window boundaries shouldn't e.g. trigger the OS window resize cursor.
228  bool anyMouseButtonDown = ImGui.IsAnyMouseDown();
229  SDL_CaptureMouse(anyMouseButtonDown ? SdlBool.SdlTrue : SdlBool.SdlFalse);
230  }
