103  public Shader(Stream vertexShaderStream, Stream geometryShaderStream, Stream fragmentShaderStream) :
104  base(IntPtr.Zero)
105  {
106  // using these funky conditional operators because StreamAdaptor doesn't have some method for dealing with
107  // its constructor argument being null
108  using (StreamAdaptor vertexAdaptor =
109  vertexShaderStream != null ? new StreamAdaptor(vertexShaderStream) : null,
110  geometryAdaptor = geometryShaderStream != null ? new StreamAdaptor(geometryShaderStream) : null,
111  fragmentAdaptor = fragmentShaderStream != null ? new StreamAdaptor(fragmentShaderStream) : null)
112  {
113  CPointer = sfShader_createFromStream(vertexAdaptor != null ? vertexAdaptor.InputStreamPtr : IntPtr.Zero,
114  geometryAdaptor != null ? geometryAdaptor.InputStreamPtr : IntPtr.Zero,
115  fragmentAdaptor != null ? fragmentAdaptor.InputStreamPtr : IntPtr.Zero);
116  }
117 
118  if (CPointer == IntPtr.Zero)
119  {
120  throw new LoadingFailedException("shader");
121  }
122  }
