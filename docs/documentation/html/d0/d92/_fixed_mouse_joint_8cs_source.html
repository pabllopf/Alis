216  internal override void InitVelocityConstraints(ref SolverData data)
217  {
218  indexA = BodyA.IslandIndex;
219  localCenterA = BodyA.Sweep.LocalCenter;
220  invMassA = BodyA.InvMass;
221  invIa = BodyA.InvI;
222 
223  Vector2 cA = data.Positions[indexA].C;
224  float aA = data.Positions[indexA].A;
225  Vector2 vA = data.Velocities[indexA].V;
226  float wA = data.Velocities[indexA].W;
227 
228  Rotation qA = new Rotation(aA);
229 
230  float d = Damping;
231  float k = Stiffness;
232 
233  // magic formulas
234  // gamma has units of inverse mass.
235  // beta has units of inverse time.
236  float h = data.Step.DeltaTime;
237  gamma = h * (d + h * k);
238  if (gamma != 0.0f)
239  {
240  gamma = 1.0f / gamma;
241  }
242 
243  beta = h * k * gamma;
244 
245  // Compute the effective mass matrix.
246  rA = MathUtils.Mul(qA, LocalAnchorA - localCenterA);
247 
248  // K = [(1/m1 + 1/m2) * eye(2) - skew(r1) * invI1 * skew(r1) - skew(r2) * invI2 * skew(r2)]
249  // = [1/m1+1/m2 0 ] + invI1 * [r1.y*r1.y -r1.x*r1.y] + invI2 * [r1.y*r1.y -r1.x*r1.y]
250  // [ 0 1/m1+1/m2] [-r1.x*r1.y r1.x*r1.x] [-r1.x*r1.y r1.x*r1.x]
251  Matrix2X2F kk = new Matrix2X2F(
252  invMassA + invIa * rA.Y * rA.Y + gamma,
253  -invIa * rA.X * rA.Y,
254  -invIa * rA.X * rA.Y,
255  invMassA + invIa * rA.X * rA.X + gamma
256  );
257 
258  mass = kk.Inverse;
259 
260  c = cA + rA - targetB;
261  c *= beta;
262 
263  // Cheat with some damping
264  wA *= 0.98f;
265 
266  if (data.Step.WarmStarting)
267  {
268  impulse *= data.Step.DeltaTimeRatio;
269  vA += invMassA * impulse;
270  wA += invIa * MathUtils.Cross(rA, impulse);
271  }
272  else
273  {
274  impulse = Vector2.Zero;
275  }
276 
277  data.Velocities[indexA].V = vA;
278  data.Velocities[indexA].W = wA;
279  }
280 
285  internal override void SolveVelocityConstraints(ref SolverData data)
286  {
287  Vector2 vA = data.Velocities[indexA].V;
288  float wA = data.Velocities[indexA].W;
289 
290  // cDot = v + cross(w, r)
291  Vector2 cDot = vA + MathUtils.Cross(wA, rA);
292  Vector2 impulseLocal = MathUtils.Mul(ref mass, -(cDot + c + gamma * impulse));
293 
294  Vector2 oldImpulse = impulse;
295  impulse += impulseLocal;
296  float maxImpulse = data.Step.DeltaTime * Force;
297  if (impulse.LengthSquared() > maxImpulse * maxImpulse)
298  {
299  impulse *= maxImpulse / impulse.Length();
300  }
301 
302  impulseLocal = impulse - oldImpulse;
303 
304  vA += invMassA * impulseLocal;
305  wA += invIa * MathUtils.Cross(rA, impulseLocal);
306 
307  data.Velocities[indexA].V = vA;
308  data.Velocities[indexA].W = wA;
309  }
