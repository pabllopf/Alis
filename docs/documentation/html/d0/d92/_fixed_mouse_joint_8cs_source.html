238  internal override void InitVelocityConstraints(ref SolverData data)
239  {
240  indexA = BodyA.IslandIndex;
241  localCenterA = BodyA.Sweep.LocalCenter;
242  invMassA = BodyA.InvMass;
243  invIa = BodyA.InvI;
244 
245  Vector2 cA = data.Positions[indexA].C;
246  float aA = data.Positions[indexA].A;
247  Vector2 vA = data.Velocities[indexA].V;
248  float wA = data.Velocities[indexA].W;
249 
250  Rot qA = new Rot(aA);
251 
252  float d = damping;
253  float k = stiffness;
254 
255  // magic formulas
256  // gamma has units of inverse mass.
257  // beta has units of inverse time.
258  float h = data.Step.DeltaTime;
259  gamma = h * (d + h * k);
260  if (gamma != 0.0f)
261  {
262  gamma = 1.0f / gamma;
263  }
264 
265  beta = h * k * gamma;
266 
267  // Compute the effective mass matrix.
268  rA = MathUtils.Mul(qA, localAnchorA - localCenterA);
269 
270  // K = [(1/m1 + 1/m2) * eye(2) - skew(r1) * invI1 * skew(r1) - skew(r2) * invI2 * skew(r2)]
271  // = [1/m1+1/m2 0 ] + invI1 * [r1.y*r1.y -r1.x*r1.y] + invI2 * [r1.y*r1.y -r1.x*r1.y]
272  // [ 0 1/m1+1/m2] [-r1.x*r1.y r1.x*r1.x] [-r1.x*r1.y r1.x*r1.x]
273  Mat22 kk = new Mat22();
274  kk.Ex.X = invMassA + invIa * rA.Y * rA.Y + gamma;
275  kk.Ex.Y = -invIa * rA.X * rA.Y;
276  kk.Ey.X = kk.Ex.Y;
277  kk.Ey.Y = invMassA + invIa * rA.X * rA.X + gamma;
278 
279  mass = kk.Inverse;
280 
281  c = cA + rA - targetB;
282  c *= beta;
283 
284  // Cheat with some damping
285  wA *= 0.98f;
286 
287  if (data.Step.WarmStarting)
288  {
289  impulse *= data.Step.DeltaTimeRatio;
290  vA += invMassA * impulse;
291  wA += invIa * MathUtils.Cross(rA, impulse);
292  }
293  else
294  {
295  impulse = Vector2.Zero;
296  }
297 
298  data.Velocities[indexA].V = vA;
299  data.Velocities[indexA].W = wA;
300  }
301 
306  internal override void SolveVelocityConstraints(ref SolverData data)
307  {
308  Vector2 vA = data.Velocities[indexA].V;
309  float wA = data.Velocities[indexA].W;
310 
311  // Cdot = v + cross(w, r)
312  Vector2 cdot = vA + MathUtils.Cross(wA, rA);
313  Vector2 impulse = MathUtils.Mul(ref mass, -(cdot + c + gamma * this.impulse));
314 
315  Vector2 oldImpulse = this.impulse;
316  this.impulse += impulse;
317  float maxImpulse = data.Step.DeltaTime * maxForce;
318  if (this.impulse.LengthSquared() > maxImpulse * maxImpulse)
319  {
320  this.impulse *= maxImpulse / this.impulse.Length();
321  }
322 
323  impulse = this.impulse - oldImpulse;
324 
325  vA += invMassA * impulse;
326  wA += invIa * MathUtils.Cross(rA, impulse);
327 
328  data.Velocities[indexA].V = vA;
329  data.Velocities[indexA].W = wA;
330  }
