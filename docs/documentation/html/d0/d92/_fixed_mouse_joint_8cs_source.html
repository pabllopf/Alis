252  internal override void InitVelocityConstraints(ref SolverData data)
253  {
254  indexA = BodyA.IslandIndex;
255  localCenterA = BodyA.Sweep.LocalCenter;
256  invMassA = BodyA.InvMass;
257  invIa = BodyA.InvI;
258 
259  Vector2F cA = data.Positions[indexA].C;
260  float aA = data.Positions[indexA].A;
261  Vector2F vA = data.Velocities[indexA].V;
262  float wA = data.Velocities[indexA].W;
263 
264  Rotation qA = new Rotation(aA);
265 
266  float d = damping;
267  float k = stiffness;
268 
269  // magic formulas
270  // gamma has units of inverse mass.
271  // beta has units of inverse time.
272  float h = data.Step.DeltaTime;
273  gamma = h * (d + h * k);
274  if (gamma != 0.0f)
275  {
276  gamma = 1.0f / gamma;
277  }
278 
279  beta = h * k * gamma;
280 
281  // Compute the effective mass matrix.
282  rA = MathUtils.Mul(qA, localAnchorA - localCenterA);
283 
284  // K = [(1/m1 + 1/m2) * eye(2) - skew(r1) * invI1 * skew(r1) - skew(r2) * invI2 * skew(r2)]
285  // = [1/m1+1/m2 0 ] + invI1 * [r1.y*r1.y -r1.x*r1.y] + invI2 * [r1.y*r1.y -r1.x*r1.y]
286  // [ 0 1/m1+1/m2] [-r1.x*r1.y r1.x*r1.x] [-r1.x*r1.y r1.x*r1.x]
287  Matrix2X2F kk = new Matrix2X2F();
288  kk.Ex.X = invMassA + invIa * rA.Y * rA.Y + gamma;
289  kk.Ex.Y = -invIa * rA.X * rA.Y;
290  kk.Ey.X = kk.Ex.Y;
291  kk.Ey.Y = invMassA + invIa * rA.X * rA.X + gamma;
292 
293  mass = kk.Inverse;
294 
295  c = cA + rA - targetB;
296  c *= beta;
297 
298  // Cheat with some damping
299  wA *= 0.98f;
300 
301  if (data.Step.WarmStarting)
302  {
303  impulse *= data.Step.DeltaTimeRatio;
304  vA += invMassA * impulse;
305  wA += invIa * MathUtils.Cross(rA, impulse);
306  }
307  else
308  {
309  impulse = Vector2F.Zero;
310  }
311 
312  data.Velocities[indexA].V = vA;
313  data.Velocities[indexA].W = wA;
314  }
315 
320  internal override void SolveVelocityConstraints(ref SolverData data)
321  {
322  Vector2F vA = data.Velocities[indexA].V;
323  float wA = data.Velocities[indexA].W;
324 
325  // Cdot = v + cross(w, r)
326  Vector2F cdot = vA + MathUtils.Cross(wA, rA);
327  Vector2F impulse = MathUtils.Mul(ref mass, -(cdot + c + gamma * this.impulse));
328 
329  Vector2F oldImpulse = this.impulse;
330  this.impulse += impulse;
331  float maxImpulse = data.Step.DeltaTime * maxForce;
332  if (this.impulse.LengthSquared() > maxImpulse * maxImpulse)
333  {
334  this.impulse *= maxImpulse / this.impulse.Length();
335  }
336 
337  impulse = this.impulse - oldImpulse;
338 
339  vA += invMassA * impulse;
340  wA += invIa * MathUtils.Cross(rA, impulse);
341 
342  data.Velocities[indexA].V = vA;
343  data.Velocities[indexA].W = wA;
344  }
