691  public override bool WaitForFences(Fence[] fences, bool waitAll, ulong nanosecondTimeout)
692  {
693  int msTimeout;
694  if (nanosecondTimeout == ulong.MaxValue)
695  {
696  msTimeout = -1;
697  }
698  else
699  {
700  msTimeout = (int)Math.Min(nanosecondTimeout / 1_000_000, int.MaxValue);
701  }
702 
703  ManualResetEvent[] events = GetResetEventArray(fences.Length);
704  for (int i = 0; i < fences.Length; i++)
705  {
706  events[i] = Util.AssertSubtype<Fence, MTLFence>(fences[i]).ResetEvent;
707  }
708  bool result;
709  if (waitAll)
710  {
711  result = WaitHandle.WaitAll(events, msTimeout);
712  }
713  else
714  {
715  int index = WaitHandle.WaitAny(events, msTimeout);
716  result = index != WaitHandle.WaitTimeout;
717  }
718 
719  ReturnResetEventArray(events);
720 
721  return result;
722  }
