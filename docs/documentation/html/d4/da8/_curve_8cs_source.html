82  public float Evaluate(float position)
83  {
84  if (Keys.Count == 0)
85  {
86  return 0f;
87  }
88 
89  if (Keys.Count == 1)
90  {
91  return Keys[0].Value;
92  }
93 
94  CurveKey first = Keys[0];
95  CurveKey last = Keys[Keys.Count - 1];
96 
97  if (position < first.Position)
98  {
99  switch (PreLoop)
100  {
101  case CurveLoopType.Constant:
102  //constant
103  return first.Value;
104 
105  case CurveLoopType.Linear:
106  // linear y = a*x +b with a tangeant of last point
107  return first.Value - first.TangentIn * (first.Position - position);
108 
109  case CurveLoopType.Cycle:
110  //start -> end / start -> end
111  int cycle = GetNumberOfCycle(position);
112  float virtualPos = position - cycle * (last.Position - first.Position);
113  return GetCurvePosition(virtualPos);
114 
115  case CurveLoopType.CycleOffset:
116  //make the curve continue (with no step) so must up the curve each cycle of delta(value)
117  cycle = GetNumberOfCycle(position);
118  virtualPos = position - cycle * (last.Position - first.Position);
119  return GetCurvePosition(virtualPos) + cycle * (last.Value - first.Value);
120 
121  case CurveLoopType.Oscillate:
122  //go back on curve from end and target start
123  // start-> end / end -> start
124  cycle = GetNumberOfCycle(position);
125  if (0 == cycle % 2f) //if pair
126  {
127  virtualPos = position - cycle * (last.Position - first.Position);
128  }
129  else
130  {
131  virtualPos = last.Position - position + first.Position +
132  cycle * (last.Position - first.Position);
133  }
134 
135  return GetCurvePosition(virtualPos);
136  }
137  }
138  else if (position > last.Position)
139  {
140  int cycle;
141  switch (PostLoop)
142  {
143  case CurveLoopType.Constant:
144  //constant
145  return last.Value;
146 
147  case CurveLoopType.Linear:
148  // linear y = a*x +b with a tangeant of last point
149  return last.Value + first.TangentOut * (position - last.Position);
150 
151  case CurveLoopType.Cycle:
152  //start -> end / start -> end
153  cycle = GetNumberOfCycle(position);
154  float virtualPos = position - cycle * (last.Position - first.Position);
155  return GetCurvePosition(virtualPos);
156 
157  case CurveLoopType.CycleOffset:
158  //make the curve continue (with no step) so must up the curve each cycle of delta(value)
159  cycle = GetNumberOfCycle(position);
160  virtualPos = position - cycle * (last.Position - first.Position);
161  return GetCurvePosition(virtualPos) + cycle * (last.Value - first.Value);
162 
163  case CurveLoopType.Oscillate:
164  //go back on curve from end and target start
165  // start-> end / end -> start
166  cycle = GetNumberOfCycle(position);
167  virtualPos = position - cycle * (last.Position - first.Position);
168  if (0 == cycle % 2f) //if pair
169  {
170  virtualPos = position - cycle * (last.Position - first.Position);
171  }
172  else
173  {
174  virtualPos = last.Position - position + first.Position +
175  cycle * (last.Position - first.Position);
176  }
177 
178  return GetCurvePosition(virtualPos);
179  }
180  }
181 
182  //in curve
183  return GetCurvePosition(position);
184  }

216  public void ComputeTangent(int keyIndex, CurveTangent tangentInType, CurveTangent tangentOutType)
217  {
218  // See http://msdn.microsoft.com/en-us/library/microsoft.xna.framework.curvetangent.aspx
219 
220  CurveKey key = Keys[keyIndex];
221 
222  float p0, p, p1;
223  p0 = p = p1 = key.Position;
224 
225  float v0, v, v1;
226  v0 = v = v1 = key.Value;
227 
228  if (keyIndex > 0)
229  {
230  p0 = Keys[keyIndex - 1].Position;
231  v0 = Keys[keyIndex - 1].Value;
232  }
233 
234  if (keyIndex < Keys.Count - 1)
235  {
236  p1 = Keys[keyIndex + 1].Position;
237  v1 = Keys[keyIndex + 1].Value;
238  }
239 
240  switch (tangentInType)
241  {
242  case CurveTangent.Flat:
243  key.TangentIn = 0;
244  break;
245  case CurveTangent.Linear:
246  key.TangentIn = v - v0;
247  break;
248  case CurveTangent.Smooth:
249  float pn = p1 - p0;
250  if (Math.Abs(pn) < float.Epsilon)
251  {
252  key.TangentIn = 0;
253  }
254  else
255  {
256  key.TangentIn = (v1 - v0) * ((p - p0) / pn);
257  }
258 
259  break;
260  }
261 
262  switch (tangentOutType)
263  {
264  case CurveTangent.Flat:
265  key.TangentOut = 0;
266  break;
267  case CurveTangent.Linear:
268  key.TangentOut = v1 - v;
269  break;
270  case CurveTangent.Smooth:
271  float pn = p1 - p0;
272  if (Math.Abs(pn) < float.Epsilon)
273  {
274  key.TangentOut = 0;
275  }
276  else
277  {
278  key.TangentOut = (v1 - v0) * ((p1 - p) / pn);
279  }
280 
281  break;
282  }
283  }

290  private int GetNumberOfCycle(float position)
291  {
292  float cycle = (position - Keys[0].Position) / (Keys[Keys.Count - 1].Position - Keys[0].Position);
293  if (cycle < 0f)
294  {
295  cycle--;
296  }
297 
298  return (int) cycle;

306  private float GetCurvePosition(float position)
307  {
308  //only for position in curve
309  CurveKey prev = Keys[0];
310  CurveKey next;
311  for (int i = 1; i < Keys.Count; ++i)
312  {
313  next = Keys[i];
314  if (next.Position >= position)
315  {
316  if (prev.Continuity == CurveContinuity.Step)
317  {
318  if (position >= 1f)
319  {
320  return next.Value;
321  }
322 
323  return prev.Value;
324  }
325 
326  float t = (position - prev.Position) / (next.Position - prev.Position); //to have t in [0,1]
327  float ts = t * t;
328  float tss = ts * t;
329 
330  //After a lot of search on internet I have found all about spline function
331  // and bezier (phi'sss ancien) but finaly use hermite curve
332  //http://en.wikipedia.org/wiki/Cubic_Hermite_spline
333  //P(t) = (2*t^3 - 3t^2 + 1)*P0 + (t^3 - 2t^2 + t)m0 + (-2t^3 + 3t^2)P1 + (t^3-t^2)m1
334  //with P0.value = prev.value , m0 = prev.tangentOut, P1= next.value, m1 = next.TangentIn
335  return (2 * tss - 3 * ts + 1f) * prev.Value + (tss - 2 * ts + t) * prev.TangentOut +
336  (3 * ts - 2 * tss) * next.Value + (tss - ts) * next.TangentIn;
337  }
338 
339  prev = next;
340  }
341 
342  return 0f;
343  }
