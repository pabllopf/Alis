112  public bool Intersects(IntRect rect, out IntRect overlap)
113  {
114  // Rectangles with negative dimensions are allowed, so we must handle them correctly
115 
116  // Compute the min and max of the first rectangle on both axes
117  int r1MinX = System.Math.Min(Left, Left + Width);
118  int r1MaxX = System.Math.Max(Left, Left + Width);
119  int r1MinY = System.Math.Min(Top, Top + Height);
120  int r1MaxY = System.Math.Max(Top, Top + Height);
121 
122  // Compute the min and max of the second rectangle on both axes
123  int r2MinX = System.Math.Min(rect.Left, rect.Left + rect.Width);
124  int r2MaxX = System.Math.Max(rect.Left, rect.Left + rect.Width);
125  int r2MinY = System.Math.Min(rect.Top, rect.Top + rect.Height);
126  int r2MaxY = System.Math.Max(rect.Top, rect.Top + rect.Height);
127 
128  // Compute the intersection boundaries
129  int interLeft = System.Math.Max(r1MinX, r2MinX);
130  int interTop = System.Math.Max(r1MinY, r2MinY);
131  int interRight = System.Math.Min(r1MaxX, r2MaxX);
132  int interBottom = System.Math.Min(r1MaxY, r2MaxY);
133 
134  // If the intersection is valid (positive non zero area), then there is an intersection
135  if ((interLeft < interRight) && (interTop < interBottom))
136  {
137  overlap.Left = interLeft;
138  overlap.Top = interTop;
139  overlap.Width = interRight - interLeft;
140  overlap.Height = interBottom - interTop;
141  return true;
142  }
143 
144  overlap.Left = 0;
145  overlap.Top = 0;
146  overlap.Width = 0;
147  overlap.Height = 0;
148  return false;
149  }
