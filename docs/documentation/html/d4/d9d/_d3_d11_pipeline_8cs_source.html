100  {
101  if (stages[i].Stage == ShaderStages.Vertex)

103  D3D11Shader d3d11VertexShader = ((D3D11Shader)stages[i]);
104  VertexShader = (ID3D11VertexShader)d3d11VertexShader.DeviceShader;
105  vsBytecode = d3d11VertexShader.Bytecode;
106  }
107  if (stages[i].Stage == ShaderStages.Geometry)
108  {
109  GeometryShader = (ID3D11GeometryShader)((D3D11Shader)stages[i]).DeviceShader;
110  }

112  {
113  HullShader = (ID3D11HullShader)((D3D11Shader)stages[i]).DeviceShader;

115  if (stages[i].Stage == ShaderStages.TessellationEvaluation)

119  if (stages[i].Stage == ShaderStages.Fragment)
