111  public bool Intersects(FloatRect rect, out FloatRect overlap)
112  {
113  // Rectangles with negative dimensions are allowed, so we must handle them correctly
114 
115  // Compute the min and max of the first rectangle on both axes
116  float r1MinX = System.Math.Min(Left, Left + Width);
117  float r1MaxX = System.Math.Max(Left, Left + Width);
118  float r1MinY = System.Math.Min(Top, Top + Height);
119  float r1MaxY = System.Math.Max(Top, Top + Height);
120 
121  // Compute the min and max of the second rectangle on both axes
122  float r2MinX = System.Math.Min(rect.Left, rect.Left + rect.Width);
123  float r2MaxX = System.Math.Max(rect.Left, rect.Left + rect.Width);
124  float r2MinY = System.Math.Min(rect.Top, rect.Top + rect.Height);
125  float r2MaxY = System.Math.Max(rect.Top, rect.Top + rect.Height);
126 
127  // Compute the intersection boundaries
128  float interLeft = System.Math.Max(r1MinX, r2MinX);
129  float interTop = System.Math.Max(r1MinY, r2MinY);
130  float interRight = System.Math.Min(r1MaxX, r2MaxX);
131  float interBottom = System.Math.Min(r1MaxY, r2MaxY);
132 
133  // If the intersection is valid (positive non zero area), then there is an intersection
134  if ((interLeft < interRight) && (interTop < interBottom))
135  {
136  overlap.Left = interLeft;
137  overlap.Top = interTop;
138  overlap.Width = interRight - interLeft;
139  overlap.Height = interBottom - interTop;
140  return true;
141  }
142 
143  overlap.Left = 0;
144  overlap.Top = 0;
145  overlap.Width = 0;
146  overlap.Height = 0;
147  return false;
148  }
