204  public Transform Transform
205  {
206  get
207  {
208  if (myTransformNeedUpdate)
209  {
210  myTransformNeedUpdate = false;
211 
212  float angle = -myRotation * 3.141592654F / 180.0F;
213  float cosine = (float) Math.Cos(angle);
214  float sine = (float) Math.Sin(angle);
215  float sxc = myScale.X * cosine;
216  float syc = myScale.Y * cosine;
217  float sxs = myScale.X * sine;
218  float sys = myScale.Y * sine;
219  float tx = -myOrigin.X * sxc - myOrigin.Y * sys + myPosition.X;
220  float ty = myOrigin.X * sxs - myOrigin.Y * syc + myPosition.Y;
221 
222  myTransform = new Transform(sxc, sys, tx,
223  -sxs, syc, ty,
224  0.0F, 0.0F, 1.0F);
225  }
226

228  }
