205  public Transform Transform
206  {
207  get
208  {
209  if (myTransformNeedUpdate)
210  {
211  myTransformNeedUpdate = false;
212 
213  float angle = -myRotation * 3.141592654F / 180.0F;
214  float cosine = (float) Math.Cos(angle);
215  float sine = (float) Math.Sin(angle);
216  float sxc = myScale.X * cosine;
217  float syc = myScale.Y * cosine;
218  float sxs = myScale.X * sine;
219  float sys = myScale.Y * sine;
220  float tx = -myOrigin.X * sxc - myOrigin.Y * sys + myPosition.X;
221  float ty = myOrigin.X * sxs - myOrigin.Y * syc + myPosition.Y;
222 
223  myTransform = new Transform(sxc, sys, tx,
224  -sxs, syc, ty,
225  0.0F, 0.0F, 1.0F);
226  }
227 
228  return myTransform;
229  }
