610  public void AddInputCharactersUtf8(string str)
611  {
612  byte* nativeStr;
613  int strByteCount = 0;
614  if (str != null)
615  {
616  strByteCount = Encoding.UTF8.GetByteCount(str);
617  if (strByteCount > Util.StackAllocationSizeLimit)
618  {
619  nativeStr = Util.Allocate(strByteCount + 1);
620  }
621  else
622  {
623  byte* nativeStrStackBytes = stackalloc byte[strByteCount + 1];
624  nativeStr = nativeStrStackBytes;
625  }
626 
627  int nativeStrOffset = Util.GetUtf8(str, nativeStr, strByteCount);
628  nativeStr[nativeStrOffset] = 0;
629  }
630  else
631  {
632  nativeStr = null;
633  }
634 
635  ImGuiNative.ImGuiIO_AddInputCharactersUTF8(NativePtr, nativeStr);
636  if (strByteCount > Util.StackAllocationSizeLimit)
637  {
638  Util.Free(nativeStr);
639  }
640  }
