634  public void AddInputCharactersUTF8(string str)
635  {
636  byte* native_str;
637  int str_byteCount = 0;
638  if (str != null)
639  {
640  str_byteCount = Encoding.UTF8.GetByteCount(str);
641  if (str_byteCount > Util.StackAllocationSizeLimit)
642  {
643  native_str = Util.Allocate(str_byteCount + 1);
644  }
645  else
646  {
647  byte* native_str_stackBytes = stackalloc byte[str_byteCount + 1];
648  native_str = native_str_stackBytes;
649  }
650 
651  int native_str_offset = Util.GetUtf8(str, native_str, str_byteCount);
652  native_str[native_str_offset] = 0;
653  }
654  else
655  {
656  native_str = null;
657  }
658 
659  ImGuiNative.ImGuiIO_AddInputCharactersUTF8(NativePtr, native_str);
660  if (str_byteCount > Util.StackAllocationSizeLimit)
661  {
662  Util.Free(native_str);
663  }
664  }
