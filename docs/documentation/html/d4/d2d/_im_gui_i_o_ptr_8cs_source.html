634  public void AddInputCharactersUtf8(string str)
635  {
636  byte* nativeStr;
637  int strByteCount = 0;
638  if (str != null)
639  {
640  strByteCount = Encoding.UTF8.GetByteCount(str);
641  if (strByteCount > Util.StackAllocationSizeLimit)
642  {
643  nativeStr = Util.Allocate(strByteCount + 1);
644  }
645  else
646  {
647  byte* nativeStrStackBytes = stackalloc byte[strByteCount + 1];
648  nativeStr = nativeStrStackBytes;
649  }
650 
651  int nativeStrOffset = Util.GetUtf8(str, nativeStr, strByteCount);
652  nativeStr[nativeStrOffset] = 0;
653  }
654  else
655  {
656  nativeStr = null;
657  }
658 
659  ImGuiNative.ImGuiIO_AddInputCharactersUTF8(NativePtr, nativeStr);
660  if (strByteCount > Util.StackAllocationSizeLimit)
661  {
662  Util.Free(nativeStr);
663  }
664  }
