609  public void AddInputCharactersUtf8(string str)
610  {
611  byte* nativeStr;
612  int strByteCount = 0;
613  if (str != null)
614  {
615  strByteCount = Encoding.UTF8.GetByteCount(str);
616  if (strByteCount > Util.StackAllocationSizeLimit)
617  {
618  nativeStr = Util.Allocate(strByteCount + 1);
619  }
620  else
621  {
622  byte* nativeStrStackBytes = stackalloc byte[strByteCount + 1];
623  nativeStr = nativeStrStackBytes;
624  }
625 
626  int nativeStrOffset = Util.GetUtf8(str, nativeStr, strByteCount);
627  nativeStr[nativeStrOffset] = 0;
628  }
629  else
630  {
631  nativeStr = null;
632  }
633 
634  ImGuiNative.ImGuiIO_AddInputCharactersUTF8(NativePtr, nativeStr);
635  if (strByteCount > Util.StackAllocationSizeLimit)
636  {
637  Util.Free(nativeStr);
638  }
639  }
