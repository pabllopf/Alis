75  public static void GetPointStates(out FixedArray2<PointState> state1, out FixedArray2<PointState> state2,
76  ref Manifold manifold1, ref Manifold manifold2)
77  {
78  state1 = new FixedArray2<PointState>();
79  state2 = new FixedArray2<PointState>();
80 
81  for (int i = 0; i < Settings.ManifoldPoints; ++i)
82  {
83  state1[i] = PointState.Null;
84  state2[i] = PointState.Null;
85  }
86 
87  // Detect persists and removes.
88  for (int i = 0; i < manifold1.PointCount; ++i)
89  {
90  ContactId id = manifold1.Points[i].Id;
91 
92  state1[i] = PointState.Remove;
93 
94  for (int j = 0; j < manifold2.PointCount; ++j)
95  {
96  if (manifold2.Points[j].Id.Key == id.Key)
97  {
98  state1[i] = PointState.Persist;
99  break;
100  }
101  }
102  }
103 
104  // Detect persists and adds.
105  for (int i = 0; i < manifold2.PointCount; ++i)
106  {
107  ContactId id = manifold2.Points[i].Id;
108 
109  state2[i] = PointState.Add;
110 
111  for (int j = 0; j < manifold1.PointCount; ++j)
112  {
113  if (manifold1.Points[j].Id.Key == id.Key)
114  {
115  state2[i] = PointState.Persist;
116  break;
117  }
118  }
119  }
120  }

129  internal static int ClipSegmentToLine(out FixedArray2<ClipVertex> vOut, ref FixedArray2<ClipVertex> vIn,
130  Vector2F normal, float offset, int vertexIndexA)
131  {
132  vOut = new FixedArray2<ClipVertex>();
133 
134  // Start with no output points
135  int count = 0;
136 
137  // Calculate the distance of end points to the line
138  float distance0 = Vector2F.Dot(normal, vIn.Value0.V) - offset;
139  float distance1 = Vector2F.Dot(normal, vIn.Value1.V) - offset;
140 
141  // If the points are behind the plane
142  if (distance0 <= 0.0f)
143  {
144  vOut[count++] = vIn.Value0;
145  }
146 
147  if (distance1 <= 0.0f)
148  {
149  vOut[count++] = vIn.Value1;
150  }
151 
152  // If the points are on different sides of the plane
153  if (distance0 * distance1 < 0.0f)
154  {
155  // Find intersection point of edge and plane
156  float interp = distance0 / (distance0 - distance1);
157 
158  ClipVertex cv = vOut[count];
159  cv.V = vIn.Value0.V + interp * (vIn.Value1.V - vIn.Value0.V);
160 
161  // VertexA is hitting edgeB.
162  cv.Id.ContactFeature.IndexA = (byte) vertexIndexA;
163  cv.Id.ContactFeature.IndexB = vIn.Value0.Id.ContactFeature.IndexB;
164  cv.Id.ContactFeature.TypeA = ContactFeatureType.Vertex;
165  cv.Id.ContactFeature.TypeB = ContactFeatureType.Face;
166  vOut[count] = cv;
167 
168  ++count;
169  }
170 
171  return count;
172  }
173  }
