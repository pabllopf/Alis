210  public static unsafe SwapchainSource GetSwapchainSource(Sdl2Window window)
211  {
212  IntPtr sdlHandle = window.SdlWindowHandle;
213  SDL_SysWMinfo sysWmInfo;
214  Sdl2Native.SDL_GetVersion(&sysWmInfo.version);
215  Sdl2Native.SDL_GetWMWindowInfo(sdlHandle, &sysWmInfo);
216  switch (sysWmInfo.subsystem)
217  {
218  case SysWMType.Windows:
219  Win32WindowInfo w32Info = Unsafe.Read<Win32WindowInfo>(&sysWmInfo.info);
220  return SwapchainSource.CreateWin32(w32Info.Sdl2Window, w32Info.hinstance);
221  case SysWMType.X11:
222  X11WindowInfo x11Info = Unsafe.Read<X11WindowInfo>(&sysWmInfo.info);
223  return SwapchainSource.CreateXlib(
224  x11Info.display,
225  x11Info.Sdl2Window);
226  case SysWMType.Wayland:
227  WaylandWindowInfo wlInfo = Unsafe.Read<WaylandWindowInfo>(&sysWmInfo.info);
228  return SwapchainSource.CreateWayland(wlInfo.display, wlInfo.surface);
229  case SysWMType.Cocoa:
230  CocoaWindowInfo cocoaInfo = Unsafe.Read<CocoaWindowInfo>(&sysWmInfo.info);
231  IntPtr nsWindow = cocoaInfo.Window;
232  return SwapchainSource.CreateNSWindow(nsWindow);
233  default:
234  throw new PlatformNotSupportedException("Cannot create a SwapchainSource for " + sysWmInfo.subsystem + ".");
235  }
236  }

418  public static unsafe void SetSDLGLContextAttributes(GraphicsDeviceOptions options, GraphicsBackend backend)
419  {
420  if (backend != GraphicsBackend.OpenGL && backend != GraphicsBackend.OpenGLES)
421  {
422  throw new VeldridException(
423  $"{nameof(backend)} must be {nameof(GraphicsBackend.OpenGL)} or {nameof(GraphicsBackend.OpenGLES)}.");
424  }
425 
426  SDL_GLContextFlag contextFlags = options.Debug
427  ? SDL_GLContextFlag.Debug | SDL_GLContextFlag.ForwardCompatible
428  : SDL_GLContextFlag.ForwardCompatible;
429 
430  Sdl2Native.SDL_GL_SetAttribute(SDL_GLAttribute.ContextFlags, (int)contextFlags);
431 
432  (int major, int minor) = GetMaxGLVersion(backend == GraphicsBackend.OpenGLES);
433 
434  if (backend == GraphicsBackend.OpenGL)
435  {
436  Sdl2Native.SDL_GL_SetAttribute(SDL_GLAttribute.ContextProfileMask, (int)SDL_GLProfile.Core);
437  Sdl2Native.SDL_GL_SetAttribute(SDL_GLAttribute.ContextMajorVersion, major);
438  Sdl2Native.SDL_GL_SetAttribute(SDL_GLAttribute.ContextMinorVersion, minor);
439  }
440  else
441  {
442  Sdl2Native.SDL_GL_SetAttribute(SDL_GLAttribute.ContextProfileMask, (int)SDL_GLProfile.ES);
443  Sdl2Native.SDL_GL_SetAttribute(SDL_GLAttribute.ContextMajorVersion, major);
444  Sdl2Native.SDL_GL_SetAttribute(SDL_GLAttribute.ContextMinorVersion, minor);
445  }
446 
447  int depthBits = 0;
448  int stencilBits = 0;
449  if (options.SwapchainDepthFormat.HasValue)
450  {
451  switch (options.SwapchainDepthFormat)
452  {
453  case PixelFormat.R16_UNorm:
454  depthBits = 16;
455  break;
456  case PixelFormat.D24_UNorm_S8_UInt:
457  depthBits = 24;
458  stencilBits = 8;
459  break;
460  case PixelFormat.R32_Float:
461  depthBits = 32;
462  break;
463  case PixelFormat.D32_Float_S8_UInt:
464  depthBits = 32;
465  stencilBits = 8;
466  break;
467  default:
468  throw new VeldridException("Invalid depth format: " + options.SwapchainDepthFormat.Value);
469  }
470  }
471 
472  int result = Sdl2Native.SDL_GL_SetAttribute(SDL_GLAttribute.DepthSize, depthBits);
473  result = Sdl2Native.SDL_GL_SetAttribute(SDL_GLAttribute.StencilSize, stencilBits);
474 
475  if (options.SwapchainSrgbFormat)
476  {
477  Sdl2Native.SDL_GL_SetAttribute(SDL_GLAttribute.FramebufferSrgbCapable, 1);
478  }
479  else
480  {
481  Sdl2Native.SDL_GL_SetAttribute(SDL_GLAttribute.FramebufferSrgbCapable, 0);
482  }
483  }
