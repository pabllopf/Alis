211  public static unsafe SwapchainSource GetSwapchainSource(Sdl2Window window)
212  {
213  IntPtr sdlHandle = window.SdlWindowHandle;
214  SDL_SysWMinfo sysWmInfo;
215  Sdl2Native.SDL_GetVersion(&sysWmInfo.version);
216  Sdl2Native.SDL_GetWMWindowInfo(sdlHandle, &sysWmInfo);
217  switch (sysWmInfo.subsystem)
218  {
219  case SysWMType.Windows:
220  Win32WindowInfo w32Info = Unsafe.Read<Win32WindowInfo>(&sysWmInfo.info);
221  return SwapchainSource.CreateWin32(w32Info.Sdl2Window, w32Info.hinstance);
222  case SysWMType.X11:
223  X11WindowInfo x11Info = Unsafe.Read<X11WindowInfo>(&sysWmInfo.info);
224  return SwapchainSource.CreateXlib(
225  x11Info.display,
226  x11Info.Sdl2Window);
227  case SysWMType.Wayland:
228  WaylandWindowInfo wlInfo = Unsafe.Read<WaylandWindowInfo>(&sysWmInfo.info);
229  return SwapchainSource.CreateWayland(wlInfo.display, wlInfo.surface);
230  case SysWMType.Cocoa:
231  CocoaWindowInfo cocoaInfo = Unsafe.Read<CocoaWindowInfo>(&sysWmInfo.info);
232  IntPtr nsWindow = cocoaInfo.Window;
233  return SwapchainSource.CreateNSWindow(nsWindow);
234  default:
235  throw new PlatformNotSupportedException("Cannot create a SwapchainSource for " + sysWmInfo.subsystem + ".");
236  }
237  }

419  public static unsafe void SetSDLGLContextAttributes(GraphicsDeviceOptions options, GraphicsBackend backend)
420  {
421  if (backend != GraphicsBackend.OpenGL && backend != GraphicsBackend.OpenGLES)
422  {
423  throw new VeldridException(
424  $"{nameof(backend)} must be {nameof(GraphicsBackend.OpenGL)} or {nameof(GraphicsBackend.OpenGLES)}.");
425  }
426 
427  SDL_GLContextFlag contextFlags = options.Debug
428  ? SDL_GLContextFlag.Debug | SDL_GLContextFlag.ForwardCompatible
429  : SDL_GLContextFlag.ForwardCompatible;
430 
431  Sdl2Native.SDL_GL_SetAttribute(SDL_GLAttribute.ContextFlags, (int)contextFlags);
432 
433  (int major, int minor) = GetMaxGLVersion(backend == GraphicsBackend.OpenGLES);
434 
435  if (backend == GraphicsBackend.OpenGL)
436  {
437  Sdl2Native.SDL_GL_SetAttribute(SDL_GLAttribute.ContextProfileMask, (int)SDL_GLProfile.Core);
438  Sdl2Native.SDL_GL_SetAttribute(SDL_GLAttribute.ContextMajorVersion, major);
439  Sdl2Native.SDL_GL_SetAttribute(SDL_GLAttribute.ContextMinorVersion, minor);
440  }
441  else
442  {
443  Sdl2Native.SDL_GL_SetAttribute(SDL_GLAttribute.ContextProfileMask, (int)SDL_GLProfile.ES);
444  Sdl2Native.SDL_GL_SetAttribute(SDL_GLAttribute.ContextMajorVersion, major);
445  Sdl2Native.SDL_GL_SetAttribute(SDL_GLAttribute.ContextMinorVersion, minor);
446  }
447 
448  int depthBits = 0;
449  int stencilBits = 0;
450  if (options.SwapchainDepthFormat.HasValue)
451  {
452  switch (options.SwapchainDepthFormat)
453  {
454  case PixelFormat.R16_UNorm:
455  depthBits = 16;
456  break;
457  case PixelFormat.D24_UNorm_S8_UInt:
458  depthBits = 24;
459  stencilBits = 8;
460  break;
461  case PixelFormat.R32_Float:
462  depthBits = 32;
463  break;
464  case PixelFormat.D32_Float_S8_UInt:
465  depthBits = 32;
466  stencilBits = 8;
467  break;
468  default:
469  throw new VeldridException("Invalid depth format: " + options.SwapchainDepthFormat.Value);
470  }
471  }
472 
473  int result = Sdl2Native.SDL_GL_SetAttribute(SDL_GLAttribute.DepthSize, depthBits);
474  result = Sdl2Native.SDL_GL_SetAttribute(SDL_GLAttribute.StencilSize, stencilBits);
475 
476  if (options.SwapchainSrgbFormat)
477  {
478  Sdl2Native.SDL_GL_SetAttribute(SDL_GLAttribute.FramebufferSrgbCapable, 1);
479  }
480  else
481  {
482  Sdl2Native.SDL_GL_SetAttribute(SDL_GLAttribute.FramebufferSrgbCapable, 0);
483  }
484  }
