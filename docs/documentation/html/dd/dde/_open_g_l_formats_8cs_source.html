574  internal static void VdToGLTextureMinMagFilter(SamplerFilter filter, bool mip, out TextureMinFilter min, out TextureMagFilter mag)
575  {
576  switch (filter)
577  {
578  case SamplerFilter.Anisotropic:
579  case SamplerFilter.MinPoint_MagPoint_MipPoint:
580  min = mip ? TextureMinFilter.NearestMipmapNearest : TextureMinFilter.Nearest;
581  mag = TextureMagFilter.Nearest;
