43  // Pulley:
44  // length1 = norm(p1 - s1)
45  // length2 = norm(p2 - s2)
46  // C0 = (length1 + ratio * length2)_initial
47  // C = C0 - (length1 + ratio * length2)
48  // u1 = (p1 - s1) / norm(p1 - s1)
49  // u2 = (p2 - s2) / norm(p2 - s2)
50  // Cdot = -dot(u1, v1 + cross(w1, r1)) - ratio * dot(u2, v2 + cross(w2, r2))
51  // J = -[u1 cross(r1, u1) ratio * u2 ratio * cross(r2, u2)]
52  // K = J * invM * JT
53  // = invMass1 + invI1 * cross(r1, u1)^2 + ratio^2 * (invMass2 + invI2 * cross(r2, u2)^2)

208  public PulleyJoint(Body bodyA, Body bodyB, Vector2 anchorA, Vector2 anchorB, Vector2 worldAnchorA,
209  Vector2 worldAnchorB, float ratio, bool useWorldCoordinates = false)
210  : base(bodyA, bodyB, JointType.Pulley)
211  {
212  this.worldAnchorA = worldAnchorA;
213  this.worldAnchorB = worldAnchorB;
214 
215  if (useWorldCoordinates)
216  {
217  localAnchorA = bodyA.GetLocalPoint(anchorA);
218  localAnchorB = bodyB.GetLocalPoint(anchorB);
219 
220  Vector2 dA = anchorA - worldAnchorA;
221  lengthA = dA.Length();
222  Vector2 dB = anchorB - worldAnchorB;
223  lengthB = dB.Length();
224  }
225  else
226  {
227  localAnchorA = anchorA;
228  localAnchorB = anchorB;
229 
230  Vector2 dA = anchorA - bodyA.GetLocalPoint(worldAnchorA);
231  lengthA = dA.Length();
232  Vector2 dB = anchorB - bodyB.GetLocalPoint(worldAnchorB);
233  lengthB = dB.Length();
234  }
235 
236  Debug.Assert(ratio != 0.0f);
237  Debug.Assert(ratio > MathConstants.Epsilon);
238 
239  this.ratio = ratio;
240  constant = lengthA + ratio * lengthB;
241  impulse = 0.0f;

349  internal override void InitVelocityConstraints(ref SolverData data)
350  {
351  indexA = BodyA.IslandIndex;
352  indexB = BodyB.IslandIndex;
353  localCenterA = BodyA.Sweep.LocalCenter;
354  localCenterB = BodyB.Sweep.LocalCenter;
355  invMassA = BodyA.InvMass;
356  invMassB = BodyB.InvMass;
357  invIa = BodyA.InvI;
358  invIb = BodyB.InvI;
359 
360  Vector2 cA = data.Positions[indexA].C;
361  float aA = data.Positions[indexA].A;
362  Vector2 vA = data.Velocities[indexA].V;
363  float wA = data.Velocities[indexA].W;
364 
365  Vector2 cB = data.Positions[indexB].C;
366  float aB = data.Positions[indexB].A;
367  Vector2 vB = data.Velocities[indexB].V;
368  float wB = data.Velocities[indexB].W;
369 
370  Rot qA = new Rot(aA), qB = new Rot(aB);
371 
372  rA = MathUtils.Mul(qA, localAnchorA - localCenterA);
373  rB = MathUtils.Mul(qB, localAnchorB - localCenterB);
374 
375  // Get the pulley axes.
376  uA = cA + rA - worldAnchorA;
377  uB = cB + rB - worldAnchorB;
378 
379  float lengthA = uA.Length();
380  float lengthB = uB.Length();
381 
382  if (lengthA > 10.0f * Settings.LinearSlop)
383  {
384  uA *= 1.0f / lengthA;
385  }
386  else
387  {
388  uA = Vector2.Zero;
389  }
390 
391  if (lengthB > 10.0f * Settings.LinearSlop)
392  {
393  uB *= 1.0f / lengthB;
394  }
395  else
396  {
397  uB = Vector2.Zero;
398  }
399 
400  // Compute effective mass.
401  float ruA = MathUtils.Cross(rA, uA);
402  float ruB = MathUtils.Cross(rB, uB);
403 
404  float mA = invMassA + invIa * ruA * ruA;
405  float mB = invMassB + invIb * ruB * ruB;
406 
407  mass = mA + ratio * ratio * mB;
408 
409  if (mass > 0.0f)
410  {
411  mass = 1.0f / mass;
412  }
413 
414  if (data.Step.WarmStarting)
415  {
416  // Scale impulses to support variable time steps.
417  impulse *= data.Step.DeltaTimeRatio;
418 
419  // Warm starting.
420  Vector2 pa = -impulse * uA;
421  Vector2 pb = -ratio * impulse * uB;
422 
423  vA += invMassA * pa;
424  wA += invIa * MathUtils.Cross(rA, pa);
425  vB += invMassB * pb;
426  wB += invIb * MathUtils.Cross(rB, pb);
427  }
428  else
429  {
430  impulse = 0.0f;
431  }
432 
433  data.Velocities[indexA].V = vA;
434  data.Velocities[indexA].W = wA;
435  data.Velocities[indexB].V = vB;
436  data.Velocities[indexB].W = wB;
437  }

443  internal override void SolveVelocityConstraints(ref SolverData data)
444  {
445  Vector2 vA = data.Velocities[indexA].V;
446  float wA = data.Velocities[indexA].W;
447  Vector2 vB = data.Velocities[indexB].V;
448  float wB = data.Velocities[indexB].W;
449 
450  Vector2 vpA = vA + MathUtils.Cross(wA, rA);
451  Vector2 vpB = vB + MathUtils.Cross(wB, rB);
452 
453  float cdot = -Vector2.Dot(uA, vpA) - ratio * Vector2.Dot(uB, vpB);
454  float impulse = -mass * cdot;
455  this.impulse += impulse;
456 
457  Vector2 pa = -impulse * uA;
458  Vector2 pb = -ratio * impulse * uB;
459  vA += invMassA * pa;
460  wA += invIa * MathUtils.Cross(rA, pa);
461  vB += invMassB * pb;
462  wB += invIb * MathUtils.Cross(rB, pb);
463 
464  data.Velocities[indexA].V = vA;
465  data.Velocities[indexA].W = wA;
466  data.Velocities[indexB].V = vB;
467  data.Velocities[indexB].W = wB;
468  }

475  internal override bool SolvePositionConstraints(ref SolverData data)
476  {
477  Vector2 cA = data.Positions[indexA].C;
478  float aA = data.Positions[indexA].A;
479  Vector2 cB = data.Positions[indexB].C;
480  float aB = data.Positions[indexB].A;
481 
482  Rot qA = new Rot(aA), qB = new Rot(aB);
483 
484  Vector2 rA = MathUtils.Mul(qA, localAnchorA - localCenterA);
485  Vector2 rB = MathUtils.Mul(qB, localAnchorB - localCenterB);
486 
487  // Get the pulley axes.
488  Vector2 uA = cA + rA - worldAnchorA;
489  Vector2 uB = cB + rB - worldAnchorB;
490 
491  float lengthA = uA.Length();
492  float lengthB = uB.Length();
493 
494  if (lengthA > 10.0f * Settings.LinearSlop)
495  {
496  uA *= 1.0f / lengthA;
497  }
498  else
499  {
500  uA = Vector2.Zero;
501  }
502 
503  if (lengthB > 10.0f * Settings.LinearSlop)
504  {
505  uB *= 1.0f / lengthB;
506  }
507  else
508  {
509  uB = Vector2.Zero;
510  }
511 
512  // Compute effective mass.
513  float ruA = MathUtils.Cross(rA, uA);
514  float ruB = MathUtils.Cross(rB, uB);
515 
516  float mA = invMassA + invIa * ruA * ruA;
517  float mB = invMassB + invIb * ruB * ruB;
518 
519  float mass = mA + ratio * ratio * mB;
520 
521  if (mass > 0.0f)
522  {
523  mass = 1.0f / mass;
524  }
525 
526  float c = constant - lengthA - ratio * lengthB;
527  float linearError = Math.Abs(c);
528 
529  float impulse = -mass * c;
530 
531  Vector2 pa = -impulse * uA;
532  Vector2 pb = -ratio * impulse * uB;
533 
534  cA += invMassA * pa;
535  aA += invIa * MathUtils.Cross(rA, pa);
536  cB += invMassB * pb;
537  aB += invIb * MathUtils.Cross(rB, pb);
538 
539  data.Positions[indexA].C = cA;
540  data.Positions[indexA].A = aA;
541  data.Positions[indexB].C = cB;
542  data.Positions[indexB].A = aB;
543 
544  return linearError < Settings.LinearSlop;
545  }
546  }
