39  // Pulley:
40  // length1 = norm(p1 - s1)
41  // length2 = norm(p2 - s2)
42  // C0 = (length1 + ratio * length2)_initial
43  // C = C0 - (length1 + ratio * length2)
44  // u1 = (p1 - s1) / norm(p1 - s1)
45  // u2 = (p2 - s2) / norm(p2 - s2)
46  // Cdot = -dot(u1, v1 + cross(w1, r1)) - ratio * dot(u2, v2 + cross(w2, r2))
47  // J = -[u1 cross(r1, u1) ratio * u2 ratio * cross(r2, u2)]
48  // K = J * invM * JT
49  // = invMass1 + invI1 * cross(r1, u1)^2 + ratio^2 * (invMass2 + invI2 * cross(r2, u2)^2)

387  internal override void InitVelocityConstraints(ref SolverData data)
388  {
389  indexA = BodyA.IslandIndex;
390  indexB = BodyB.IslandIndex;
391  localCenterA = BodyA.Sweep.LocalCenter;
392  localCenterB = BodyB.Sweep.LocalCenter;
393  invMassA = BodyA.InvMass;
394  invMassB = BodyB.InvMass;
395  invIa = BodyA.InvI;
396  invIb = BodyB.InvI;
397 
398  Vector2F cA = data.Positions[indexA].C;
399  float aA = data.Positions[indexA].A;
400  Vector2F vA = data.Velocities[indexA].V;
401  float wA = data.Velocities[indexA].W;
402 
403  Vector2F cB = data.Positions[indexB].C;
404  float aB = data.Positions[indexB].A;
405  Vector2F vB = data.Velocities[indexB].V;
406  float wB = data.Velocities[indexB].W;
407 
408  Rotation qA = new Rotation(aA), qB = new Rotation(aB);
409 
410  rA = MathUtils.Mul(qA, localAnchorA - localCenterA);
411  rB = MathUtils.Mul(qB, localAnchorB - localCenterB);
412 
413  // Get the pulley axes.
414  uA = cA + rA - worldAnchorA;
415  uB = cB + rB - worldAnchorB;
416 
417  float lengthA = uA.Length();
418  float lengthB = uB.Length();
419 
420  if (lengthA > 10.0f * Settings.LinearSlop)
421  {
422  uA *= 1.0f / lengthA;
423  }
424  else
425  {
426  uA = Vector2F.Zero;
427  }
428 
429  if (lengthB > 10.0f * Settings.LinearSlop)
430  {
431  uB *= 1.0f / lengthB;
432  }
433  else
434  {
435  uB = Vector2F.Zero;
436  }
437 
438  // Compute effective mass.
439  float ruA = MathUtils.Cross(rA, uA);
440  float ruB = MathUtils.Cross(rB, uB);
441 
442  float mA = invMassA + invIa * ruA * ruA;
443  float mB = invMassB + invIb * ruB * ruB;
444 
445  mass = mA + ratio * ratio * mB;
446 
447  if (mass > 0.0f)
448  {
449  mass = 1.0f / mass;
450  }
451 
452  if (data.Step.WarmStarting)
453  {
454  // Scale impulses to support variable time steps.
455  impulse *= data.Step.DeltaTimeRatio;
456 
457  // Warm starting.
458  Vector2F pa = -impulse * uA;
459  Vector2F pb = -ratio * impulse * uB;
460 
461  vA += invMassA * pa;
462  wA += invIa * MathUtils.Cross(rA, pa);
463  vB += invMassB * pb;
464  wB += invIb * MathUtils.Cross(rB, pb);
465  }
466  else
467  {
468  impulse = 0.0f;
469  }
470 
471  data.Velocities[indexA].V = vA;
472  data.Velocities[indexA].W = wA;
473  data.Velocities[indexB].V = vB;
474  data.Velocities[indexB].W = wB;
475  }

481  internal override void SolveVelocityConstraints(ref SolverData data)
482  {
483  Vector2F vA = data.Velocities[indexA].V;
484  float wA = data.Velocities[indexA].W;
485  Vector2F vB = data.Velocities[indexB].V;
486  float wB = data.Velocities[indexB].W;
487 
488  Vector2F vpA = vA + MathUtils.Cross(wA, rA);
489  Vector2F vpB = vB + MathUtils.Cross(wB, rB);
490 
491  float cdot = -Vector2F.Dot(uA, vpA) - ratio * Vector2F.Dot(uB, vpB);
492  float impulse = -mass * cdot;
493  this.impulse += impulse;
494 
495  Vector2F pa = -impulse * uA;
496  Vector2F pb = -ratio * impulse * uB;
497  vA += invMassA * pa;
498  wA += invIa * MathUtils.Cross(rA, pa);
499  vB += invMassB * pb;
500  wB += invIb * MathUtils.Cross(rB, pb);
501 
502  data.Velocities[indexA].V = vA;
503  data.Velocities[indexA].W = wA;
504  data.Velocities[indexB].V = vB;
505  data.Velocities[indexB].W = wB;
506  }

513  internal override bool SolvePositionConstraints(ref SolverData data)
514  {
515  Vector2F cA = data.Positions[indexA].C;
516  float aA = data.Positions[indexA].A;
517  Vector2F cB = data.Positions[indexB].C;
518  float aB = data.Positions[indexB].A;
519 
520  Rotation qA = new Rotation(aA), qB = new Rotation(aB);
521 
522  Vector2F rA = MathUtils.Mul(qA, localAnchorA - localCenterA);
523  Vector2F rB = MathUtils.Mul(qB, localAnchorB - localCenterB);
524 
525  // Get the pulley axes.
526  Vector2F uA = cA + rA - worldAnchorA;
527  Vector2F uB = cB + rB - worldAnchorB;
528 
529  float lengthA = uA.Length();
530  float lengthB = uB.Length();
531 
532  if (lengthA > 10.0f * Settings.LinearSlop)
533  {
534  uA *= 1.0f / lengthA;
535  }
536  else
537  {
538  uA = Vector2F.Zero;
539  }
540 
541  if (lengthB > 10.0f * Settings.LinearSlop)
542  {
543  uB *= 1.0f / lengthB;
544  }
545  else
546  {
547  uB = Vector2F.Zero;
548  }
549 
550  // Compute effective mass.
551  float ruA = MathUtils.Cross(rA, uA);
552  float ruB = MathUtils.Cross(rB, uB);
553 
554  float mA = invMassA + invIa * ruA * ruA;
555  float mB = invMassB + invIb * ruB * ruB;
556 
557  float mass = mA + ratio * ratio * mB;
558 
559  if (mass > 0.0f)
560  {
561  mass = 1.0f / mass;
562  }
563 
564  float c = constant - lengthA - ratio * lengthB;
565  float linearError = Math.Abs(c);
566 
567  float impulse = -mass * c;
568 
569  Vector2F pa = -impulse * uA;
570  Vector2F pb = -ratio * impulse * uB;
571 
572  cA += invMassA * pa;
573  aA += invIa * MathUtils.Cross(rA, pa);
574  cB += invMassB * pb;
575  aB += invIb * MathUtils.Cross(rB, pb);
576 
577  data.Positions[indexA].C = cA;
578  data.Positions[indexA].A = aA;
579  data.Positions[indexB].C = cB;
580  data.Positions[indexB].A = aB;
581 
582  return linearError < Settings.LinearSlop;
583  }
584  }
