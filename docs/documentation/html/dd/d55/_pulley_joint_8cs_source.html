42  // Pulley:
43  // length1 = norm(p1 - s1)
44  // length2 = norm(p2 - s2)
45  // C0 = (length1 + ratio * length2)_initial
46  // C = C0 - (length1 + ratio * length2)
47  // u1 = (p1 - s1) / norm(p1 - s1)
48  // u2 = (p2 - s2) / norm(p2 - s2)
49  // Cdot = -dot(u1, v1 + cross(w1, r1)) - ratio * dot(u2, v2 + cross(w2, r2))
50  // J = -[u1 cross(r1, u1) ratio * u2 ratio * cross(r2, u2)]
51  // K = J * invM * JT
52  // = invMass1 + invI1 * cross(r1, u1)^2 + ratio^2 * (invMass2 + invI2 * cross(r2, u2)^2)

207  public PulleyJoint(Body bodyA, Body bodyB, Vector2 anchorA, Vector2 anchorB, Vector2 worldAnchorA,
208  Vector2 worldAnchorB, float ratio, bool useWorldCoordinates = false)
209  : base(bodyA, bodyB, JointType.Pulley)

230  lengthA = dA.Length();

232  lengthB = dB.Length();
233  }

236  Debug.Assert(ratio > MathConstants.Epsilon);

348  internal override void InitVelocityConstraints(ref SolverData data)
349  {
350  indexA = BodyA.IslandIndex;
351  indexB = BodyB.IslandIndex;
352  localCenterA = BodyA.Sweep.LocalCenter;
353  localCenterB = BodyB.Sweep.LocalCenter;
354  invMassA = BodyA.InvMass;
355  invMassB = BodyB.InvMass;
356  invIa = BodyA.InvI;
357  invIb = BodyB.InvI;
358 
359  Vector2 cA = data.Positions[indexA].C;
360  float aA = data.Positions[indexA].A;
361  Vector2 vA = data.Velocities[indexA].V;
362  float wA = data.Velocities[indexA].W;
363 
364  Vector2 cB = data.Positions[indexB].C;
365  float aB = data.Positions[indexB].A;
366  Vector2 vB = data.Velocities[indexB].V;
367  float wB = data.Velocities[indexB].W;
368 
369  Rot qA = new Rot(aA), qB = new Rot(aB);
370 
371  rA = MathUtils.Mul(qA, localAnchorA - localCenterA);
372  rB = MathUtils.Mul(qB, localAnchorB - localCenterB);
373 
374  // Get the pulley axes.
375  uA = cA + rA - worldAnchorA;
376  uB = cB + rB - worldAnchorB;
377 
378  float lengthA = uA.Length();
379  float lengthB = uB.Length();
380 
381  if (lengthA > 10.0f * Settings.LinearSlop)
382  {
383  uA *= 1.0f / lengthA;
384  }
385  else
386  {
387  uA = Vector2.Zero;
388  }
389 
390  if (lengthB > 10.0f * Settings.LinearSlop)
391  {
392  uB *= 1.0f / lengthB;
393  }
394  else
395  {
396  uB = Vector2.Zero;
397  }
398 
399  // Compute effective mass.
400  float ruA = MathUtils.Cross(rA, uA);
401  float ruB = MathUtils.Cross(rB, uB);
402 
403  float mA = invMassA + invIa * ruA * ruA;
404  float mB = invMassB + invIb * ruB * ruB;
405 
406  mass = mA + ratio * ratio * mB;
407 
408  if (mass > 0.0f)
409  {
410  mass = 1.0f / mass;
411  }
412 
413  if (data.Step.WarmStarting)
414  {
415  // Scale impulses to support variable time steps.
416  impulse *= data.Step.DeltaTimeRatio;
417 
418  // Warm starting.
419  Vector2 pa = -impulse * uA;
420  Vector2 pb = -ratio * impulse * uB;
421 
422  vA += invMassA * pa;
423  wA += invIa * MathUtils.Cross(rA, pa);
424  vB += invMassB * pb;
425  wB += invIb * MathUtils.Cross(rB, pb);
426  }
427  else
428  {
429  impulse = 0.0f;
430  }
431 
432  data.Velocities[indexA].V = vA;
433  data.Velocities[indexA].W = wA;
434  data.Velocities[indexB].V = vB;
435  data.Velocities[indexB].W = wB;
436  }

442  internal override void SolveVelocityConstraints(ref SolverData data)
443  {
444  Vector2 vA = data.Velocities[indexA].V;
445  float wA = data.Velocities[indexA].W;
446  Vector2 vB = data.Velocities[indexB].V;
447  float wB = data.Velocities[indexB].W;
448 
449  Vector2 vpA = vA + MathUtils.Cross(wA, rA);
450  Vector2 vpB = vB + MathUtils.Cross(wB, rB);
451 
452  float cdot = -Vector2.Dot(uA, vpA) - ratio * Vector2.Dot(uB, vpB);
453  float impulse = -mass * cdot;
454  this.impulse += impulse;
455 
456  Vector2 pa = -impulse * uA;
457  Vector2 pb = -ratio * impulse * uB;
458  vA += invMassA * pa;
459  wA += invIa * MathUtils.Cross(rA, pa);
460  vB += invMassB * pb;
461  wB += invIb * MathUtils.Cross(rB, pb);
462 
463  data.Velocities[indexA].V = vA;
464  data.Velocities[indexA].W = wA;
465  data.Velocities[indexB].V = vB;
466  data.Velocities[indexB].W = wB;
467  }

474  internal override bool SolvePositionConstraints(ref SolverData data)
475  {
476  Vector2 cA = data.Positions[indexA].C;
477  float aA = data.Positions[indexA].A;
478  Vector2 cB = data.Positions[indexB].C;
479  float aB = data.Positions[indexB].A;
480 
481  Rot qA = new Rot(aA), qB = new Rot(aB);
482 
483  Vector2 rA = MathUtils.Mul(qA, localAnchorA - localCenterA);
484  Vector2 rB = MathUtils.Mul(qB, localAnchorB - localCenterB);
485 
486  // Get the pulley axes.
487  Vector2 uA = cA + rA - worldAnchorA;
488  Vector2 uB = cB + rB - worldAnchorB;
489 
490  float lengthA = uA.Length();
491  float lengthB = uB.Length();
492 
493  if (lengthA > 10.0f * Settings.LinearSlop)
494  {
495  uA *= 1.0f / lengthA;
496  }
497  else
498  {
499  uA = Vector2.Zero;
500  }
501 
502  if (lengthB > 10.0f * Settings.LinearSlop)
503  {
504  uB *= 1.0f / lengthB;
505  }
506  else
507  {
508  uB = Vector2.Zero;
509  }
510 
511  // Compute effective mass.
512  float ruA = MathUtils.Cross(rA, uA);
513  float ruB = MathUtils.Cross(rB, uB);
514 
515  float mA = invMassA + invIa * ruA * ruA;
516  float mB = invMassB + invIb * ruB * ruB;
517 
518  float mass = mA + ratio * ratio * mB;
519 
520  if (mass > 0.0f)
521  {
522  mass = 1.0f / mass;
523  }
524 
525  float c = constant - lengthA - ratio * lengthB;
526  float linearError = Math.Abs(c);
527 
528  float impulse = -mass * c;
529 
530  Vector2 pa = -impulse * uA;
531  Vector2 pb = -ratio * impulse * uB;
532 
533  cA += invMassA * pa;
534  aA += invIa * MathUtils.Cross(rA, pa);
535  cB += invMassB * pb;
536  aB += invIb * MathUtils.Cross(rB, pb);
537 
538  data.Positions[indexA].C = cA;
539  data.Positions[indexA].A = aA;
540  data.Positions[indexB].C = cB;
541  data.Positions[indexB].A = aB;
542 
543  return linearError < Settings.LinearSlop;
544  }
545  }
