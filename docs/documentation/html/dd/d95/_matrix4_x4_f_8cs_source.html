454  public static Matrix4X4F CreateConstrainedBillboard(Vector3 objectPosition, Vector3 cameraPosition, Vector3 rotateAxis, Vector3 cameraForwardVector, Vector3 objectForwardVector)
455  {
456  // Treat the case when object and camera positions are too close.
457  Vector3 faceDir = objectPosition - cameraPosition;
458  float norm = faceDir.LengthSquared();
459 
460  if (norm < BillboardEpsilon)
461  {
462  faceDir = -cameraForwardVector;
463  }
464  else
465  {
466  faceDir = Vector3.Multiply(faceDir, 1.0f / MathF.Sqrt(norm));
467  }
468 
469  Vector3 yaxis = rotateAxis;
470  Vector3 xaxis;
471  Vector3 zaxis;
472 
473  // Treat the case when angle between faceDir and rotateAxis is too close to 0.
474  float dot = Vector3.Dot(rotateAxis, faceDir);
475 
476  if (MathF.Abs(dot) > BillboardMinAngle)
477  {
478  zaxis = objectForwardVector;
479 
480  // Make sure passed values are useful for compute.
481  dot = Vector3.Dot(rotateAxis, zaxis);
482 
483  if (MathF.Abs(dot) > BillboardMinAngle)
484  {
485  zaxis = MathF.Abs(rotateAxis.Z) > BillboardMinAngle ? new Vector3(1, 0, 0) : new Vector3(0, 0, -1);
486  }
487 
488  xaxis = Vector3.Normalize(Vector3.Cross(rotateAxis, zaxis));
489  zaxis = Vector3.Normalize(Vector3.Cross(xaxis, rotateAxis));
490  }
491  else
492  {
493  xaxis = Vector3.Normalize(Vector3.Cross(rotateAxis, faceDir));
494  zaxis = Vector3.Normalize(Vector3.Cross(xaxis, yaxis));
495  }
496 
497  Matrix4X4F result;
498 
499  result.M11 = xaxis.X;
500  result.M12 = xaxis.Y;
501  result.M13 = xaxis.Z;
502  result.M14 = 0.0f;
503 
504  result.M21 = yaxis.X;
505  result.M22 = yaxis.Y;
506  result.M23 = yaxis.Z;
507  result.M24 = 0.0f;
508 
509  result.M31 = zaxis.X;
510  result.M32 = zaxis.Y;
511  result.M33 = zaxis.Z;
512  result.M34 = 0.0f;
513 
514  result.M41 = objectPosition.X;
515  result.M42 = objectPosition.Y;
516  result.M43 = objectPosition.Z;
517  result.M44 = 1.0f;
518 
519  return result;

526  public static Matrix4X4F CreateFromAxisAngle(Vector3 axis, float angle)
527  {
528  // a: angle
529  // x, y, z: unit vector for axis.
530  //
531  // Rotation matrix M can compute by using below equation.
532  //
533  // T T
534  // M = uu + (cos a)( I-uu ) + (sin a)S
535  //
536  // Where:
537  //
538  // u = ( x, y, z )
539  //
540  // [ 0 -z y ]
541  // S = [ z 0 -x ]
542  // [ -y x 0 ]
543  //
544  // [ 1 0 0 ]
545  // I = [ 0 1 0 ]
546  // [ 0 0 1 ]
547  //
548  //
549  // [ xx+cosa*(1-xx) yx-cosa*yx-sina*z zx-cosa*xz+sina*y ]
550  // M = [ xy-cosa*yx+sina*z yy+cosa(1-yy) yz-cosa*yz-sina*x ]
551  // [ zx-cosa*zx-sina*y zy-cosa*zy+sina*x zz+cosa*(1-zz) ]
552  //
553  float x = axis.X, y = axis.Y, z = axis.Z;
554  float sa = MathF.Sin(angle), ca = MathF.Cos(angle);
555  float xx = x * x, yy = y * y, zz = z * z;
556  float xy = x * y, xz = x * z, yz = y * z;
557 
558  Matrix4X4F result = Identity;
559 
560  result.M11 = xx + ca * (1.0f - xx);
561  result.M12 = xy - ca * xy + sa * z;
562  result.M13 = xz - ca * xz - sa * y;
563 
564  result.M21 = xy - ca * xy - sa * z;
565  result.M22 = yy + ca * (1.0f - yy);
566  result.M23 = yz - ca * yz + sa * x;
567 
568  result.M31 = xz - ca * xz + sa * y;
569  result.M32 = yz - ca * yz - sa * x;
570  result.M33 = zz + ca * (1.0f - zz);
571 
572  return result;
573  }

578  public static Matrix4X4F CreateFromQuaternion(Quaternion quaternion)
579  {
580  Matrix4X4F result = Identity;
581 
582  float xx = quaternion.X * quaternion.X;
583  float yy = quaternion.Y * quaternion.Y;
584  float zz = quaternion.Z * quaternion.Z;
585 
586  float xy = quaternion.X * quaternion.Y;
587  float wz = quaternion.Z * quaternion.W;
588  float xz = quaternion.Z * quaternion.X;
589  float wy = quaternion.Y * quaternion.W;
590  float yz = quaternion.Y * quaternion.Z;
591  float wx = quaternion.X * quaternion.W;
592 
593  result.M11 = 1.0f - 2.0f * (yy + zz);
594  result.M12 = 2.0f * (xy + wz);
595  result.M13 = 2.0f * (xz - wy);
596 
597  result.M21 = 2.0f * (xy - wz);
598  result.M22 = 1.0f - 2.0f * (zz + xx);
599  result.M23 = 2.0f * (yz + wx);
600 
601  result.M31 = 2.0f * (xz + wy);
602  result.M32 = 2.0f * (yz - wx);
603  result.M33 = 1.0f - 2.0f * (yy + xx);
604 
605  return result;
606  }

707  public static Matrix4X4F CreatePerspective(float width, float height, float nearPlaneDistance, float farPlaneDistance)
708  {
709  if (nearPlaneDistance <= 0.0f)
710  {
711  throw new ArgumentOutOfRangeException(nameof(nearPlaneDistance));
712  }
713 
714  if (farPlaneDistance <= 0.0f)
715  {
716  throw new ArgumentOutOfRangeException(nameof(farPlaneDistance));
717  }
718 
719  if (nearPlaneDistance >= farPlaneDistance)
720  {
721  throw new ArgumentOutOfRangeException(nameof(nearPlaneDistance));
722  }
723 
724  Matrix4X4F result;
725 
726  result.M11 = 2.0f * nearPlaneDistance / width;
727  result.M12 = result.M13 = result.M14 = 0.0f;
728 
729  result.M22 = 2.0f * nearPlaneDistance / height;
730  result.M21 = result.M23 = result.M24 = 0.0f;
731

1364  public readonly float GetDeterminant()
1365  {
1366  // | a b c d | | f g h | | e g h | | e f h | | e f g |
1367  // | e f g h | = a | j k l | - b | i k l | + c | i j l | - d | i j k |
1368  // | i j k l | | n o p | | m o p | | m n p | | m n o |
1369  // | m n o p |
1370  //
1371  // | f g h |
1372  // a | j k l | = a ( f ( kp - lo ) - g ( jp - ln ) + h ( jo - kn ) )
1373  // | n o p |
1374  //
1375  // | e g h |
1376  // b | i k l | = b ( e ( kp - lo ) - g ( ip - lm ) + h ( io - km ) )
1377  // | m o p |
1378  //
1379  // | e f h |
1380  // c | i j l | = c ( e ( jp - ln ) - f ( ip - lm ) + h ( in - jm ) )
1381  // | m n p |
1382  //
1383  // | e f g |
1384  // d | i j k | = d ( e ( jo - kn ) - f ( io - km ) + g ( in - jm ) )
1385  // | m n o |
1386  //
1387  // Cost of operation
1388  // 17 adds and 28 muls.
1389  //
1390  // add: 6 + 8 + 3 = 17
1391  // mul: 12 + 16 = 28
1392 
1393  float a = M11, b = M12, c = M13, d = M14;
1394  float e = M21, f = M22, g = M23, h = M24;
1395  float i = M31, j = M32, k = M33, l = M34;
1396  float m = M41, n = M42, o = M43, p = M44;
1397 
1398  float kpLo = k * p - l * o;
1399  float jpLn = j * p - l * n;
1400  float joKn = j * o - k * n;
1401  float ipLm = i * p - l * m;
1402  float ioKm = i * o - k * m;
1403  float inJm = i * n - j * m;
1404 
1405  return a * (f * kpLo - g * jpLn + h * joKn) -
1406  b * (e * kpLo - g * ipLm + h * ioKm) +
1407  c * (e * jpLn - f * ipLm + h * inJm) -
1408  d * (e * joKn - f * ioKm + g * inJm);
1409  }
