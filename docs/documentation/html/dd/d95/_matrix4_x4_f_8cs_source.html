294  public static Matrix4X4F Transform(Matrix4X4F value, Quaternion rotation)
295  {
296  // Compute rotation matrix.
297  float x2 = rotation.X + rotation.X;
298  float y2 = rotation.Y + rotation.Y;
299  float z2 = rotation.Z + rotation.Z;
300 
301  float wx2 = rotation.W * x2;
302  float wy2 = rotation.W * y2;
303  float wz2 = rotation.W * z2;
304  float xx2 = rotation.X * x2;
305  float xy2 = rotation.X * y2;
306  float xz2 = rotation.X * z2;
307  float yy2 = rotation.Y * y2;
308  float yz2 = rotation.Y * z2;
309  float zz2 = rotation.Z * z2;
310 
311  float q11 = 1.0f - yy2 - zz2;
312  float q21 = xy2 - wz2;
313  float q31 = xz2 + wy2;
314 
315  float q12 = xy2 + wz2;
316  float q22 = 1.0f - xx2 - zz2;
317  float q32 = yz2 - wx2;
318 
319  float q13 = xz2 - wy2;
320  float q23 = yz2 + wx2;
321  float q33 = 1.0f - xx2 - yy2;
322 
323  Matrix4X4F result = new Matrix4X4F(
324  value.M11 * q11 + value.M12 * q21 + value.M13 * q31,
325  value.M11 * q12 + value.M12 * q22 + value.M13 * q32,
326  value.M11 * q13 + value.M12 * q23 + value.M13 * q33,
327  value.M14,

366  public readonly float GetDeterminant()
367  {
368  // | a b c d | | f g h | | e g h | | e f h | | e f g |
369  // | e f g h | = a | j k l | - b | i k l | + c | i j l | - d | i j k |
370  // | i j k l | | n o p | | m o p | | m n p | | m n o |
371  // | m n o p |
372  //
373  // | f g h |
374  // a | j k l | = a ( f ( kp - lo ) - g ( jp - ln ) + h ( jo - kn ) )
375  // | n o p |
376  //
377  // | e g h |
378  // b | i k l | = b ( e ( kp - lo ) - g ( ip - lm ) + h ( io - km ) )
379  // | m o p |
380  //
381  // | e f h |
382  // c | i j l | = c ( e ( jp - ln ) - f ( ip - lm ) + h ( in - jm ) )
383  // | m n p |
384  //
385  // | e f g |
386  // d | i j k | = d ( e ( jo - kn ) - f ( io - km ) + g ( in - jm ) )
387  // | m n o |
388  //
389  // Cost of operation
390  // 17 adds and 28 muls.
391  //
392  // add: 6 + 8 + 3 = 17
393  // mul: 12 + 16 = 28
394 
395  float a = M11, b = M12, c = M13, d = M14;
396  float e = M21, f = M22, g = M23, h = M24;
397  float i = M31, j = M32, k = M33, l = M34;
398  float m = M41, n = M42, o = M43, p = M44;
399 
400  float kpLo = k * p - l * o;
401  float jpLn = j * p - l * n;
402  float joKn = j * o - k * n;
403  float ipLm = i * p - l * m;
404  float ioKm = i * o - k * m;
405  float inJm = i * n - j * m;
406 
407  return a * (f * kpLo - g * jpLn + h * joKn) -
408  b * (e * kpLo - g * ipLm + h * ioKm) +
409  c * (e * jpLn - f * ipLm + h * inJm) -
410  d * (e * joKn - f * ioKm + g * inJm);
411  }
