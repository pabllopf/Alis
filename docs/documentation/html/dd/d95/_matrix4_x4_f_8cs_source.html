373  public static Matrix4X4F Transform(Matrix4X4F value, Quaternion rotation)
374  {
375  // Compute rotation matrix.
376  float x2 = rotation.X + rotation.X;
377  float y2 = rotation.Y + rotation.Y;
378  float z2 = rotation.Z + rotation.Z;
379 
380  float wx2 = rotation.W * x2;
381  float wy2 = rotation.W * y2;
382  float wz2 = rotation.W * z2;
383  float xx2 = rotation.X * x2;
384  float xy2 = rotation.X * y2;
385  float xz2 = rotation.X * z2;
386  float yy2 = rotation.Y * y2;
387  float yz2 = rotation.Y * z2;
388  float zz2 = rotation.Z * z2;
389 
390  float q11 = 1.0f - yy2 - zz2;
391  float q21 = xy2 - wz2;
392  float q31 = xz2 + wy2;
393 
394  float q12 = xy2 + wz2;
395  float q22 = 1.0f - xx2 - zz2;
396  float q32 = yz2 - wx2;
397 
398  float q13 = xz2 - wy2;
399  float q23 = yz2 + wx2;
400  float q33 = 1.0f - xx2 - yy2;
401 
402  Matrix4X4F result = new Matrix4X4F(
403  value.M11 * q11 + value.M12 * q21 + value.M13 * q31,
404  value.M11 * q12 + value.M12 * q22 + value.M13 * q32,
405  value.M11 * q13 + value.M12 * q23 + value.M13 * q33,
406  value.M14,
407  value.M21 * q11 + value.M22 * q21 + value.M23 * q31,
408  value.M21 * q12 + value.M22 * q22 + value.M23 * q32,
409  value.M21 * q13 + value.M22 * q23 + value.M23 * q33,
410  value.M24,
411  value.M31 * q11 + value.M32 * q21 + value.M33 * q31,
412  value.M31 * q12 + value.M32 * q22 + value.M33 * q32,
413  value.M31 * q13 + value.M32 * q23 + value.M33 * q33,
414  value.M34,
415  value.M41 * q11 + value.M42 * q21 + value.M43 * q31,
416  value.M41 * q12 + value.M42 * q22 + value.M43 * q32,
417  value.M41 * q13 + value.M42 * q23 + value.M43 * q33,
418  value.M44
419  );
420 
421  return result;
422  }

445  public float GetDeterminant()
446  {
447  // | a b c d | | f g h | | e g h | | e f h | | e f g |
448  // | e f g h | = a | j k l | - b | i k l | + c | i j l | - d | i j k |
449  // | i j k l | | n o p | | m o p | | m n p | | m n o |
450  // | m n o p |
451  //
452  // | f g h |
453  // a | j k l | = a ( f ( kp - lo ) - g ( jp - ln ) + h ( jo - kn ) )
454  // | n o p |
455  //
456  // | e g h |
457  // b | i k l | = b ( e ( kp - lo ) - g ( ip - lm ) + h ( io - km ) )
458  // | m o p |
459  //
460  // | e f h |
461  // c | i j l | = c ( e ( jp - ln ) - f ( ip - lm ) + h ( in - jm ) )
462  // | m n p |
463  //
464  // | e f g |
465  // d | i j k | = d ( e ( jo - kn ) - f ( io - km ) + g ( in - jm ) )
466  // | m n o |
467  //
468  // Cost of operation
469  // 17 adds and 28 muls.
470  //
471  // add: 6 + 8 + 3 = 17
472  // mul: 12 + 16 = 28
473 
474  float a = M11, b = M12, c = M13, d = M14;
475  float e = M21, f = M22, g = M23, h = M24;
476  float i = M31, j = M32, k = M33, l = M34;
477  float m = M41, n = M42, o = M43, p = M44;
478 
479  float kpLo = k * p - l * o;
480  float jpLn = j * p - l * n;
481  float joKn = j * o - k * n;
482  float ipLm = i * p - l * m;
483  float ioKm = i * o - k * m;
484  float inJm = i * n - j * m;
485 
486  return a * (f * kpLo - g * jpLn + h * joKn) -
487  b * (e * kpLo - g * ipLm + h * ioKm) +
488  c * (e * jpLn - f * ipLm + h * inJm) -
489  d * (e * joKn - f * ioKm + g * inJm);
490  }
