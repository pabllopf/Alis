707  public uint GetFramebuffer(uint mipLevel, uint arrayLayer)
708  {
709  Debug.Assert(!FormatHelpers.IsCompressedFormat(Format));
710  Debug.Assert(Created);
711 
712  uint subresource = CalculateSubresource(mipLevel, arrayLayer);
713  if (_framebuffers[subresource] == 0)
714  {
715  FramebufferTarget framebufferTarget = SampleCount == TextureSampleCount.Count1
716  ? FramebufferTarget.DrawFramebuffer
717  : FramebufferTarget.ReadFramebuffer;
718 
719  glGenFramebuffers(1, out _framebuffers[subresource]);
720  CheckLastError();
721 
722  glBindFramebuffer(framebufferTarget, _framebuffers[subresource]);
723  CheckLastError();
724 
725  _gd.TextureSamplerManager.SetTextureTransient(TextureTarget, Texture);
726 
727  if (TextureTarget == TextureTarget.Texture2D || TextureTarget == TextureTarget.Texture2DMultisample)
728  {
729  glFramebufferTexture2D(
730  framebufferTarget,
731  GLFramebufferAttachment.ColorAttachment0,
732  TextureTarget,
733  Texture,
734  (int)mipLevel);
735  CheckLastError();
736  }
737  else if (TextureTarget == TextureTarget.Texture2DArray
738  || TextureTarget == TextureTarget.Texture2DMultisampleArray
739  || TextureTarget == TextureTarget.Texture3D)
740  {
741  glFramebufferTextureLayer(
742  framebufferTarget,
743  GLFramebufferAttachment.ColorAttachment0,
744  Texture,
745  (int)mipLevel,
746  (int)arrayLayer);
747  CheckLastError();
748  }
749 
750  FramebufferErrorCode errorCode = glCheckFramebufferStatus(framebufferTarget);
751  if (errorCode != FramebufferErrorCode.FramebufferComplete)
752  {
753  throw new VeldridException("Failed to create texture copy FBO: " + errorCode);
754  }
755  }
756 
757  return _framebuffers[subresource];
758  }
