194  public override void Update()
195  {
196  while (Sdl.SDL_PollEvent(out sdlEvent) != 0)
197  {
198  if (Sdl.SDL_NumJoysticks() > 0)
199  {
200  for (int joystickId = 0; joystickId < joysticks.Count; joystickId++)
201  {
202  for (int index = 0; index < buttons.Count; index++)
203  {
204  if ((Sdl.SDL_JoystickGetButton(joysticks[joystickId], (int) buttons[index]) != 0)
205  && !buttonsTemp.ContainsKey($"{joysticks[joystickId]}|{buttons[index]}"))
206  {
207  buttonsTemp.Add($"{joysticks[joystickId]}|{buttons[index]}", buttons[index]);
208  foreach (GameObject currentSceneGameObject in SceneManager.CurrentSceneManager.CurrentScene.GameObjects)
209  {
210  currentSceneGameObject.Components.ForEach(i => i.OnPressButton(buttons[index], joystickId));
211  }
212  }
213 
214  if ((Sdl.SDL_JoystickGetButton(joysticks[joystickId], (int) buttons[index]) == 0)
215  && buttonsTemp.ContainsKey($"{joysticks[joystickId]}|{buttons[index]}"))
216  {
217  buttonsTemp.Remove($"{joysticks[joystickId]}|{buttons[index]}");
218  foreach (GameObject currentSceneGameObject in SceneManager.CurrentSceneManager.CurrentScene.GameObjects)
219  {
220  currentSceneGameObject.Components.ForEach(i => i.OnReleaseButton(buttons[index], joystickId));
221  }
222  }
223  }
224  }
225  }
226  }
227 
228  if (Sdl.SDL_NumJoysticks() > 0)
229  {
230  for (int joystickId = 0; joystickId < joysticks.Count; joystickId++)
231  {
232  for (int index = 0; index < buttons.Count; index++)
233  {
234  if ((Sdl.SDL_JoystickGetButton(joysticks[joystickId], (int) buttons[index]) != 0)
235  && buttonsTemp.ContainsKey($"{joysticks[joystickId]}|{buttons[index]}"))
236  {
237  foreach (GameObject currentSceneGameObject in SceneManager.CurrentSceneManager.CurrentScene.GameObjects)
238  {
239  currentSceneGameObject.Components.ForEach(i => i.OnPressDownButton(buttons[index], joystickId));
240  }
241  }
242  }
243  }
