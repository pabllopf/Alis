296  internal override void InitVelocityConstraints(ref SolverData data)
297  {
298  indexA = BodyA.IslandIndex;
299  indexB = BodyB.IslandIndex;
300  localCenterA = BodyA.Sweep.LocalCenter;
301  localCenterB = BodyB.Sweep.LocalCenter;
302  invMassA = BodyA.InvMass;
303  invMassB = BodyB.InvMass;
304  invIa = BodyA.InvI;
305  invIb = BodyB.InvI;
306 
307  Vector2 cA = data.Positions[indexA].C;
308  float aA = data.Positions[indexA].A;
309  Vector2 vA = data.Velocities[indexA].V;
310  float wA = data.Velocities[indexA].W;
311 
312  Vector2 cB = data.Positions[indexB].C;
313  float aB = data.Positions[indexB].A;
314  Vector2 vB = data.Velocities[indexB].V;
315  float wB = data.Velocities[indexB].W;
316 
317  Rot qA = new Rot(aA);
318  Rot qB = new Rot(aB);
319 
320  // Compute the effective mass matrix.
321  rA = MathUtils.Mul(qA, linearOffset - localCenterA);
322  rB = MathUtils.Mul(qB, -localCenterB);
323 
324  // J = [-I -r1_skew I r2_skew]
325  // [ 0 -1 0 1]
326  // r_skew = [-ry; rx]
327 
328  // Matlab
329  // K = [ mA+r1y^2*iA+mB+r2y^2*iB, -r1y*iA*r1x-r2y*iB*r2x, -r1y*iA-r2y*iB]
330  // [ -r1y*iA*r1x-r2y*iB*r2x, mA+r1x^2*iA+mB+r2x^2*iB, r1x*iA+r2x*iB]
331  // [ -r1y*iA-r2y*iB, r1x*iA+r2x*iB, iA+iB]
332 
333  float mA = invMassA, mB = invMassB;
334  float iA = invIa, iB = invIb;
335 
336  // Upper 2 by 2 of K for point to point
337  Mat22 k = new Mat22();
338  k.Ex.X = mA + mB + iA * rA.Y * rA.Y + iB * rB.Y * rB.Y;
339  k.Ex.Y = -iA * rA.X * rA.Y - iB * rB.X * rB.Y;
340  k.Ey.X = k.Ex.Y;
341  k.Ey.Y = mA + mB + iA * rA.X * rA.X + iB * rB.X * rB.X;
342 
343  linearMass = k.Inverse;
344 
345  angularMass = iA + iB;
346  if (angularMass > 0.0f)
347  {
348  angularMass = 1.0f / angularMass;
349  }
350 
351  linearError = cB + rB - cA - rA;
352  angularError = aB - aA - angularOffset;
353 
354  if (data.Step.WarmStarting)
355  {
356  // Scale impulses to support a variable time step.
357  linearImpulse *= data.Step.DeltaTimeRatio;
358  angularImpulse *= data.Step.DeltaTimeRatio;
359 
360  Vector2 p = new Vector2(linearImpulse.X, linearImpulse.Y);
361  vA -= mA * p;
362  wA -= iA * (MathUtils.Cross(rA, p) + angularImpulse);
363  vB += mB * p;
364  wB += iB * (MathUtils.Cross(rB, p) + angularImpulse);
365  }
366  else
367  {
368  linearImpulse = Vector2.Zero;
369  angularImpulse = 0.0f;
370  }
371 
372  data.Velocities[indexA].V = vA;
373  data.Velocities[indexA].W = wA;
374  data.Velocities[indexB].V = vB;
375  data.Velocities[indexB].W = wB;
376  }

382  internal override void SolveVelocityConstraints(ref SolverData data)
383  {
384  Vector2 vA = data.Velocities[indexA].V;
385  float wA = data.Velocities[indexA].W;
386  Vector2 vB = data.Velocities[indexB].V;
387  float wB = data.Velocities[indexB].W;
388 
389  float mA = invMassA, mB = invMassB;
390  float iA = invIa, iB = invIb;
391 
392  float h = data.Step.DeltaTime;
393  float invH = data.Step.InvertedDeltaTime;
394 
395  // Solve angular friction
396  {
397  float cdot = wB - wA + invH * correctionFactor * angularError;
398  float impulse = -angularMass * cdot;
399 
400  float oldImpulse = angularImpulse;
401  float maxImpulse = h * maxTorque;
402  angularImpulse = MathUtils.Clamp(angularImpulse + impulse, -maxImpulse, maxImpulse);
403  impulse = angularImpulse - oldImpulse;
404 
405  wA -= iA * impulse;
406  wB += iB * impulse;
407  }
408 
409  // Solve linear friction
410  {
411  Vector2 cdot = vB + MathUtils.Cross(wB, rB) - vA - MathUtils.Cross(wA, rA) +
412  invH * correctionFactor * linearError;
413 
414  Vector2 impulse = -MathUtils.Mul(ref linearMass, ref cdot);
415  Vector2 oldImpulse = linearImpulse;
416  linearImpulse += impulse;
417 
418  float maxImpulse = h * maxForce;
419 
420  if (linearImpulse.LengthSquared() > maxImpulse * maxImpulse)
421  {
422  linearImpulse = Vector2.Normalize(linearImpulse);
423  linearImpulse *= maxImpulse;
424  }
425 
426  impulse = linearImpulse - oldImpulse;
427 
428  vA -= mA * impulse;
429  wA -= iA * MathUtils.Cross(rA, impulse);
430 
431  vB += mB * impulse;
432  wB += iB * MathUtils.Cross(rB, impulse);
433  }
434 
435  data.Velocities[indexA].V = vA;
436  data.Velocities[indexA].W = wA;
437  data.Velocities[indexB].V = vB;
438  data.Velocities[indexB].W = wB;
439  }
