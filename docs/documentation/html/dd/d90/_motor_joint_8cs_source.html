311  internal override void InitVelocityConstraints(ref SolverData data)
312  {
313  indexA = BodyA.IslandIndex;
314  indexB = BodyB.IslandIndex;
315  localCenterA = BodyA.Sweep.LocalCenter;
316  localCenterB = BodyB.Sweep.LocalCenter;
317  invMassA = BodyA.InvMass;
318  invMassB = BodyB.InvMass;
319  invIa = BodyA.InvI;
320  invIb = BodyB.InvI;
321 
322  Vector2F cA = data.Positions[indexA].C;
323  float aA = data.Positions[indexA].A;
324  Vector2F vA = data.Velocities[indexA].V;
325  float wA = data.Velocities[indexA].W;
326 
327  Vector2F cB = data.Positions[indexB].C;
328  float aB = data.Positions[indexB].A;
329  Vector2F vB = data.Velocities[indexB].V;
330  float wB = data.Velocities[indexB].W;
331 
332  Rotation qA = new Rotation(aA);
333  Rotation qB = new Rotation(aB);
334 
335  // Compute the effective mass matrix.
336  rA = MathUtils.Mul(qA, linearOffset - localCenterA);
337  rB = MathUtils.Mul(qB, -localCenterB);
338 
339  // J = [-I -r1_skew I r2_skew]
340  // [ 0 -1 0 1]
341  // r_skew = [-ry; rx]
342 
343  // Matlab
344  // K = [ mA+r1y^2*iA+mB+r2y^2*iB, -r1y*iA*r1x-r2y*iB*r2x, -r1y*iA-r2y*iB]
345  // [ -r1y*iA*r1x-r2y*iB*r2x, mA+r1x^2*iA+mB+r2x^2*iB, r1x*iA+r2x*iB]
346  // [ -r1y*iA-r2y*iB, r1x*iA+r2x*iB, iA+iB]
347 
348  float mA = invMassA, mB = invMassB;
349  float iA = invIa, iB = invIb;
350 
351  // Upper 2 by 2 of K for point to point
352  Matrix2X2F k = new Matrix2X2F(
353  mA + mB + iA * rA.Y * rA.Y + iB * rB.Y * rB.Y,
354  -iA * rA.X * rA.Y - iB * rB.X * rB.Y,
355  -iA * rA.X * rA.Y - iB * rB.X * rB.Y,
356  mA + mB + iA * rA.X * rA.X + iB * rB.X * rB.X
357  );
358 
359  linearMass = k.Inverse;
360 
361  angularMass = iA + iB;
362  if (angularMass > 0.0f)
363  {
364  angularMass = 1.0f / angularMass;
365  }
366 
367  linearError = cB + rB - cA - rA;
368  angularError = aB - aA - angularOffset;
369 
370  if (data.Step.WarmStarting)
371  {
372  // Scale impulses to support a variable time step.
373  linearImpulse *= data.Step.DeltaTimeRatio;
374  angularImpulse *= data.Step.DeltaTimeRatio;
375 
376  Vector2F p = new Vector2F(linearImpulse.X, linearImpulse.Y);
377  vA -= mA * p;
378  wA -= iA * (MathUtils.Cross(rA, p) + angularImpulse);
379  vB += mB * p;
380  wB += iB * (MathUtils.Cross(rB, p) + angularImpulse);
381  }
382  else
383  {
384  linearImpulse = Vector2F.Zero;
385  angularImpulse = 0.0f;
386  }
387 
388  data.Velocities[indexA].V = vA;
389  data.Velocities[indexA].W = wA;
390  data.Velocities[indexB].V = vB;
391  data.Velocities[indexB].W = wB;
392  }

398  internal override void SolveVelocityConstraints(ref SolverData data)
399  {
400  Vector2F vA = data.Velocities[indexA].V;
401  float wA = data.Velocities[indexA].W;
402  Vector2F vB = data.Velocities[indexB].V;
403  float wB = data.Velocities[indexB].W;
404 
405  float mA = invMassA, mB = invMassB;
406  float iA = invIa, iB = invIb;
407 
408  float h = data.Step.DeltaTime;
409  float invH = data.Step.InvertedDeltaTime;
410 
411  // Solve angular friction
412  {
413  float cdot = wB - wA + invH * correctionFactor * angularError;
414  float impulse = -angularMass * cdot;
415 
416  float oldImpulse = angularImpulse;
417  float maxImpulse = h * maxTorque;
418  angularImpulse = MathUtils.Clamp(angularImpulse + impulse, -maxImpulse, maxImpulse);
419  impulse = angularImpulse - oldImpulse;
420 
421  wA -= iA * impulse;
422  wB += iB * impulse;
423  }
424 
425  // Solve linear friction
426  {
427  Vector2F cdot = vB + MathUtils.Cross(wB, rB) - vA - MathUtils.Cross(wA, rA) +
428  invH * correctionFactor * linearError;
429 
430  Vector2F impulse = -MathUtils.Mul(ref linearMass, ref cdot);
431  Vector2F oldImpulse = linearImpulse;
432  linearImpulse += impulse;
433 
434  float maxImpulse = h * maxForce;
435 
436  if (linearImpulse.LengthSquared() > maxImpulse * maxImpulse)
437  {
438  linearImpulse = Vector2F.Normalize(linearImpulse);
439  linearImpulse *= maxImpulse;
440  }
441 
442  impulse = linearImpulse - oldImpulse;
443 
444  vA -= mA * impulse;
445  wA -= iA * MathUtils.Cross(rA, impulse);
446 
447  vB += mB * impulse;
448  wB += iB * MathUtils.Cross(rB, impulse);
449  }
450 
451  data.Velocities[indexA].V = vA;
452  data.Velocities[indexA].W = wA;
453  data.Velocities[indexB].V = vB;
454  data.Velocities[indexB].W = wB;
455  }
