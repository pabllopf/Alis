295  internal override void InitVelocityConstraints(ref SolverData data)
296  {
297  indexA = BodyA.IslandIndex;
298  indexB = BodyB.IslandIndex;
299  localCenterA = BodyA.Sweep.LocalCenter;
300  localCenterB = BodyB.Sweep.LocalCenter;
301  invMassA = BodyA.InvMass;
302  invMassB = BodyB.InvMass;
303  invIa = BodyA.InvI;
304  invIb = BodyB.InvI;
305 
306  Vector2 cA = data.Positions[indexA].C;
307  float aA = data.Positions[indexA].A;
308  Vector2 vA = data.Velocities[indexA].V;
309  float wA = data.Velocities[indexA].W;
310 
311  Vector2 cB = data.Positions[indexB].C;
312  float aB = data.Positions[indexB].A;
313  Vector2 vB = data.Velocities[indexB].V;
314  float wB = data.Velocities[indexB].W;
315 
316  Rot qA = new Rot(aA);
317  Rot qB = new Rot(aB);
318 
319  // Compute the effective mass matrix.
320  rA = MathUtils.Mul(qA, linearOffset - localCenterA);
321  rB = MathUtils.Mul(qB, -localCenterB);
322 
323  // J = [-I -r1_skew I r2_skew]
324  // [ 0 -1 0 1]
325  // r_skew = [-ry; rx]
326 
327  // Matlab
328  // K = [ mA+r1y^2*iA+mB+r2y^2*iB, -r1y*iA*r1x-r2y*iB*r2x, -r1y*iA-r2y*iB]
329  // [ -r1y*iA*r1x-r2y*iB*r2x, mA+r1x^2*iA+mB+r2x^2*iB, r1x*iA+r2x*iB]
330  // [ -r1y*iA-r2y*iB, r1x*iA+r2x*iB, iA+iB]
331 
332  float mA = invMassA, mB = invMassB;
333  float iA = invIa, iB = invIb;
334 
335  // Upper 2 by 2 of K for point to point
336  Mat22 k = new Mat22();
337  k.Ex.X = mA + mB + iA * rA.Y * rA.Y + iB * rB.Y * rB.Y;
338  k.Ex.Y = -iA * rA.X * rA.Y - iB * rB.X * rB.Y;
339  k.Ey.X = k.Ex.Y;
340  k.Ey.Y = mA + mB + iA * rA.X * rA.X + iB * rB.X * rB.X;
341 
342  linearMass = k.Inverse;
343 
344  angularMass = iA + iB;
345  if (angularMass > 0.0f)
346  {
347  angularMass = 1.0f / angularMass;
348  }
349 
350  linearError = cB + rB - cA - rA;
351  angularError = aB - aA - angularOffset;
352 
353  if (data.Step.WarmStarting)
354  {
355  // Scale impulses to support a variable time step.
356  linearImpulse *= data.Step.DeltaTimeRatio;
357  angularImpulse *= data.Step.DeltaTimeRatio;
358 
359  Vector2 p = new Vector2(linearImpulse.X, linearImpulse.Y);
360  vA -= mA * p;
361  wA -= iA * (MathUtils.Cross(rA, p) + angularImpulse);
362  vB += mB * p;
363  wB += iB * (MathUtils.Cross(rB, p) + angularImpulse);
364  }
365  else
366  {
367  linearImpulse = Vector2.Zero;
368  angularImpulse = 0.0f;
369  }
370 
371  data.Velocities[indexA].V = vA;
372  data.Velocities[indexA].W = wA;
373  data.Velocities[indexB].V = vB;
374  data.Velocities[indexB].W = wB;
375  }

381  internal override void SolveVelocityConstraints(ref SolverData data)
382  {
383  Vector2 vA = data.Velocities[indexA].V;
384  float wA = data.Velocities[indexA].W;
385  Vector2 vB = data.Velocities[indexB].V;
386  float wB = data.Velocities[indexB].W;
387 
388  float mA = invMassA, mB = invMassB;
389  float iA = invIa, iB = invIb;
390 
391  float h = data.Step.DeltaTime;
392  float invH = data.Step.InvertedDeltaTime;
393 
394  // Solve angular friction
395  {
396  float cdot = wB - wA + invH * correctionFactor * angularError;
397  float impulse = -angularMass * cdot;
398 
399  float oldImpulse = angularImpulse;
400  float maxImpulse = h * maxTorque;
401  angularImpulse = MathUtils.Clamp(angularImpulse + impulse, -maxImpulse, maxImpulse);
402  impulse = angularImpulse - oldImpulse;
403 
404  wA -= iA * impulse;
405  wB += iB * impulse;
406  }
407 
408  // Solve linear friction
409  {
410  Vector2 cdot = vB + MathUtils.Cross(wB, rB) - vA - MathUtils.Cross(wA, rA) +
411  invH * correctionFactor * linearError;
412 
413  Vector2 impulse = -MathUtils.Mul(ref linearMass, ref cdot);
414  Vector2 oldImpulse = linearImpulse;
415  linearImpulse += impulse;
416 
417  float maxImpulse = h * maxForce;
418 
419  if (linearImpulse.LengthSquared() > maxImpulse * maxImpulse)
420  {
421  linearImpulse = Vector2.Normalize(linearImpulse);
422  linearImpulse *= maxImpulse;
423  }
424 
425  impulse = linearImpulse - oldImpulse;
426 
427  vA -= mA * impulse;
428  wA -= iA * MathUtils.Cross(rA, impulse);
429 
430  vB += mB * impulse;
431  wB += iB * MathUtils.Cross(rB, impulse);
432  }
433 
434  data.Velocities[indexA].V = vA;
435  data.Velocities[indexA].W = wA;
436  data.Velocities[indexB].V = vB;
437  data.Velocities[indexB].W = wB;
438  }
