66  private static List<Vertices> TriangulatePolygon(Vertices vertices, float tolerance)
67  {
68  if (vertices.Count < 3)
69  {
70  return new List<Vertices>();
71  }
72 
73  List<Vertices> results = new List<Vertices>();
74 
75  List<Vertices> pinchedPolygon = TriangulatePinchedPolygon(vertices, tolerance);
76  results.AddRange(pinchedPolygon ?? TriangulateRegularPolygon(vertices));
77 
78  return results;
79  }

81  // Helper method to triangulate a pinched polygon
89  private static List<Vertices> TriangulatePinchedPolygon(Vertices vertices, float tolerance)
90  {
91  Vertices pin = new Vertices(vertices);
92  if (ResolvePinchPoint(pin, out Vertices pA, out Vertices pB, tolerance))
93  {
94  List<Vertices> mergeA = TriangulatePolygon(pA, tolerance);
95  List<Vertices> mergeB = TriangulatePolygon(pB, tolerance);
96 
97  if (mergeA.Count == 0 || mergeB.Count == 0)
98  {
99  throw new TriangulateException("Can't triangulate your polygon.");
100  }
101 
102  List<Vertices> result = new List<Vertices>();
103  result.AddRange(mergeA);
104  result.AddRange(mergeB);
105  return result;
106  }
107 
108  return null;
109  }

111  // Helper method to triangulate a regular polygon
117  private static List<Vertices> TriangulateRegularPolygon(Vertices vertices)
118  {
119  List<Vertices> results = new List<Vertices>();
120 
121  float[] xRem = new float[vertices.Count];
122  float[] yRem = new float[vertices.Count];
123  for (int i = 0; i < vertices.Count; ++i)
124  {
125  xRem[i] = vertices[i].X;
126  yRem[i] = vertices[i].Y;
127  }
128 
129  int vNum = vertices.Count;
130 
131  while (vNum > 3)
132  {
133  int earIndex = FindEar(xRem, yRem, vNum);
134  if (earIndex == -1)
135  {
136  results.AddRange(GenerateTrianglesFromBuffer(xRem, yRem, vNum));
137  return results;
138  }
139 
140  vNum = ClipEar(earIndex, ref xRem, ref yRem, vNum, results);
141  }
142 
143  results.AddRange(GenerateTrianglesFromBuffer(xRem, yRem, vNum));
144  return results;
145  }

176  private static int ClipEar(int earIndex, ref float[] xRem, ref float[] yRem, int vNum, List<Vertices> results)
177  {
178  int under = earIndex == 0 ? vNum - 1 : earIndex - 1;
179  int over = earIndex == vNum - 1 ? 0 : earIndex + 1;
180 
181  // Add the clipped triangle to the results
182  results.Add(new Vertices(new Triangle(xRem[earIndex], yRem[earIndex], xRem[over], yRem[over], xRem[under], yRem[under])));
183 
184  // Remove the ear tip from the lists
185  float[] newX = new float[vNum - 1];
186  float[] newY = new float[vNum - 1];
187  int currDest = 0;
188  for (int i = 0; i < vNum - 1; ++i)
189  {
190  if (currDest == earIndex)
191  {
192  ++currDest;
193  }
194 
195  newX[i] = xRem[currDest];
196  newY[i] = yRem[currDest];
197  ++currDest;
198  }
199 
200  // Update arrays
201  xRem = newX;
202  yRem = newY;
203 
204  return vNum - 1;
205  }

263  private static bool FindPinchPoint(Vertices vertices, float tolerance, out int pinchIndexA, out int pinchIndexB)
264  {
265  pinchIndexA = -1;
266  pinchIndexB = -1;
267 
268  for (int i = 0; i < vertices.Count; ++i)
269  {
270  for (int j = i + 1; j < vertices.Count; ++j)
271  {
272  if (IsPinchPoint(vertices[i], vertices[j], tolerance))
273  {
274  pinchIndexA = i;
275  pinchIndexB = j;
276  return true;
277  }
278  }
279  }
280 
281  return false;
282  }

306  private static bool SplitVertices(Vertices pin, int pinchIndexA, int pinchIndexB, Vertices poutA, Vertices poutB)
307  {
308  int sizeA = pinchIndexB - pinchIndexA;
309 
310  if (sizeA == pin.Count)
311  {
312  return false; // Duplicate points at wraparound, not a problem here
313  }
314 
315  for (int i = 0; i < sizeA; ++i)
316  {
317  int ind = RemainderLocal(pinchIndexA + i, pin.Count);
318  poutA.Add(pin[ind]);
319  }
320 
321  int sizeB = pin.Count - sizeA;
322  for (int i = 0; i < sizeB; ++i)
323  {
324  int ind = RemainderLocal(pinchIndexB + i, pin.Count);
325  poutB.Add(pin[ind]);
326  }
327 
328  return true;
329  }

349  private static bool IsEar(int i, float[] xv, float[] yv, int xvLength)
350  {
351  float dx0, dy0, dx1, dy1;
352  if (i >= xvLength || i < 0 || xvLength < 3)
353  {
354  return false;
355  }
356 
357  int upper = i + 1;
358  int lower = i - 1;
359  if (i == 0)
360  {
361  dx0 = xv[0] - xv[xvLength - 1];
362  dy0 = yv[0] - yv[xvLength - 1];
363  dx1 = xv[1] - xv[0];
364  dy1 = yv[1] - yv[0];
365  lower = xvLength - 1;
366  }
367  else if (i == xvLength - 1)
368  {
369  dx0 = xv[i] - xv[i - 1];
370  dy0 = yv[i] - yv[i - 1];
371  dx1 = xv[0] - xv[i];
372  dy1 = yv[0] - yv[i];
373  upper = 0;
374  }
375  else
376  {
377  dx0 = xv[i] - xv[i - 1];
378  dy0 = yv[i] - yv[i - 1];
379  dx1 = xv[i + 1] - xv[i];
380  dy1 = yv[i + 1] - yv[i];
381  }
382 
383  float cross = dx0 * dy1 - dx1 * dy0;
384 
385  if (cross > 0)
386  {
387  return false;
388  }
389 
390  Triangle myTri = new Triangle(xv[i], yv[i], xv[upper], yv[upper], xv[lower], yv[lower]);
391 
392  for (int j = 0; j < xvLength; ++j)
393  {
394  if (j == i || j == lower || j == upper)
395  {
396  continue;
397  }
398 
399  if (myTri.IsInside(xv[j], yv[j]))
400  {
401  return false;
402  }
403  }
404 
405  return true;
406  }
407  }
