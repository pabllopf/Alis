180  public string DisplayedString
181  {
182  get
183  {
184  // Get a pointer to the source string (UTF-32)
185  IntPtr source = sfText_getUnicodeString(CPointer);
186 
187  // Find its length (find the terminating 0)
188  uint length = 0;
189  unsafe
190  {
191  for (uint* ptr = (uint*) source.ToPointer(); *ptr != 0; ++ptr)
192  {
193  length++;
194  }
195  }
196 
197  // Copy it to a byte array
198  byte[] sourceBytes = new byte[length * 4];
199  Marshal.Copy(source, sourceBytes, 0, sourceBytes.Length);
200 
201  // Convert it to a C# string
202  return Encoding.UTF32.GetString(sourceBytes);
203  }
204 
205  set
206  {
207  // Copy the string to a null-terminated UTF-32 byte array
208  byte[] utf32 = Encoding.UTF32.GetBytes(value + '\0');
209 
210  // Pass it to the C API
211  unsafe
212  {
213  fixed (byte* ptr = utf32)
214  {
215  sfText_setUnicodeString(CPointer, (IntPtr) ptr);
216  }
217  }
218  }
219  }
