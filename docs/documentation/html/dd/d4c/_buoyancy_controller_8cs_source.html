127  public override void Update(float dt)
128  {
129  uniqueBodies.Clear();
130 
131  World.TestPointAllFixtures.ForEach(fixture =>
132  {
133  if (!fixture.Body.IsStatic && fixture.Body.Awake)
134  {
135  if (!uniqueBodies.Contains(fixture.Body))
136  {
137  uniqueBodies.Add(fixture.Body);
138  }
139  }
140  });
141 
142  foreach (Body body in uniqueBodies)
143  {
144  Vector2 areac = Vector2.Zero;
145  Vector2 massc = Vector2.Zero;
146  float area = 0;
147  float mass = 0;
148 
149  for (int j = 0; j < body.FixtureList.Count; j++)
150  {
151  Fixture fixture = body.FixtureList[j];
152 
153  if (fixture.Shape.ShapeType != ShapeType.Polygon && fixture.Shape.ShapeType != ShapeType.Circle)
154  {
155  continue;
156  }
157 
158  Shape shape = fixture.Shape;
159 
160  float sarea = ComputeSubmergedArea(shape, ref normal, offset, ref body.Xf, out Vector2 sc);
161  area += sarea;
162  areac.X += sarea * sc.X;
163  areac.Y += sarea * sc.Y;
164 
165  mass += sarea * shape.DensityPrivate;
166  massc.X += sarea * sc.X * shape.DensityPrivate;
167  massc.Y += sarea * sc.Y * shape.DensityPrivate;
168  }
169 
170  areac.X /= area;
171  areac.Y /= area;
172  massc.X /= mass;
173  massc.Y /= mass;
174 
175  if (area < MathConstants.Epsilon)
176  {
177  continue;
178  }
179 
180  //Buoyancy
181  Vector2 buoyancyForce = -Density * area * gravity;
182  body.ApplyForce(buoyancyForce, massc);
183 
184  //Linear drag
185  Vector2 dragForce = body.GetLinearVelocityFromWorldPoint(areac) - Velocity;
186  dragForce *= -LinearDragCoefficient * area;
187  body.ApplyForce(dragForce, areac);
188 
189  //Angular drag
190  body.ApplyTorque(-body.Inertia / body.Mass * area * body.AngularVelocity * AngularDragCoefficient);
191  }
192  }

205  private float ComputeSubmergedArea(Shape shape, ref Vector2 normal, float offset, ref Transform xf,
206  out Vector2 sc)
207  {
208  switch (shape.ShapeType)
209  {
210  case ShapeType.Circle:
211  {
212  CircleShape circleShape = (CircleShape) shape;
213 
214  sc = Vector2.Zero;
215 
216  float radius2 = circleShape.RadiusPrivate * circleShape.RadiusPrivate;
217 
218  Vector2 p = MathUtils.Mul(ref xf, circleShape.Position);
219  float l = -(Vector2.Dot(normal, p) - offset);
220  if (l < -circleShape.RadiusPrivate + MathConstants.Epsilon)
221 
222  //Completely dry
223  {
224  return 0;
225  }
226 
227  if (l > circleShape.RadiusPrivate)
228  {
229  //Completely wet
230  sc = p;
231  return MathConstants.Pi * radius2;
232  }
233 
234  //Magic
235  float l2 = l * l;
236  float area = radius2 * (float) (Math.Asin(l / circleShape.RadiusPrivate) + MathConstants.Pi / 2 +
237  l * Math.Sqrt(radius2 - l2));
238  float com = -2.0f / 3.0f * (float) Math.Pow(radius2 - l2, 1.5f) / area;
239 
240  sc.X = p.X + normal.X * com;
241  sc.Y = p.Y + normal.Y * com;
242 
243  return area;
244  }
245  case ShapeType.Edge:
246  sc = Vector2.Zero;
247  return 0;
248  case ShapeType.Polygon:
249  {
250  sc = Vector2.Zero;
251 
252  PolygonShape polygonShape = (PolygonShape) shape;
253 
254  //Transform plane into shape co-ordinates
255  Vector2 normalL = MathUtils.MulT(xf.Q, normal);
256  float offsetL = offset - Vector2.Dot(normal, xf.P);
257 
258  float[] depths = new float[Settings.MaxPolygonVertices];
259  int diveCount = 0;
260  int intoIndex = -1;
261  int outoIndex = -1;
262 
263  bool lastSubmerged = false;
264  int i;
265  for (i = 0; i < polygonShape.VerticesPrivate.Count; i++)
266  {
267  depths[i] = Vector2.Dot(normalL, polygonShape.VerticesPrivate[i]) - offsetL;
268  bool isSubmerged = depths[i] < -MathConstants.Epsilon;
269  if (i > 0)
270  {
271  if (isSubmerged)
272  {
273  if (!lastSubmerged)
274  {
275  intoIndex = i - 1;
276  diveCount++;
277  }
278  }
279  else
280  {
281  if (lastSubmerged)
282  {
283  outoIndex = i - 1;
284  diveCount++;
285  }
286  }
287  }
288 
289  lastSubmerged = isSubmerged;
290  }
291 
292  switch (diveCount)
293  {
294  case 0:
295  if (lastSubmerged)
296  {
297  //Completely submerged
298  sc = MathUtils.Mul(ref xf, polygonShape.MassDataPrivate.Centroid);
299  return polygonShape.MassDataPrivate.Mass / Density;
300  }
301 
302  //Completely dry
303  return 0;
304  case 1:
305  if (intoIndex == -1)
306  {
307  intoIndex = polygonShape.VerticesPrivate.Count - 1;
308  }
309  else
310  {
311  outoIndex = polygonShape.VerticesPrivate.Count - 1;
312  }
313 
314  break;
315  }
316 
317  int intoIndex2 = (intoIndex + 1) % polygonShape.VerticesPrivate.Count;
318  int outoIndex2 = (outoIndex + 1) % polygonShape.VerticesPrivate.Count;
319 
320  float intoLambda = (0 - depths[intoIndex]) / (depths[intoIndex2] - depths[intoIndex]);
321  float outoLambda = (0 - depths[outoIndex]) / (depths[outoIndex2] - depths[outoIndex]);
322 
323  Vector2 intoVec = new Vector2(
324  polygonShape.VerticesPrivate[intoIndex].X * (1 - intoLambda) +
325  polygonShape.VerticesPrivate[intoIndex2].X * intoLambda,
326  polygonShape.VerticesPrivate[intoIndex].Y * (1 - intoLambda) +
327  polygonShape.VerticesPrivate[intoIndex2].Y * intoLambda);
328  Vector2 outoVec = new Vector2(
329  polygonShape.VerticesPrivate[outoIndex].X * (1 - outoLambda) +
330  polygonShape.VerticesPrivate[outoIndex2].X * outoLambda,
331  polygonShape.VerticesPrivate[outoIndex].Y * (1 - outoLambda) +
332  polygonShape.VerticesPrivate[outoIndex2].Y * outoLambda);
333 
334  //Initialize accumulator
335  float area = 0;
336  Vector2 center = new Vector2(0, 0);
337  Vector2 p2 = polygonShape.VerticesPrivate[intoIndex2];
338 
339  const float kInv3 = 1.0f / 3.0f;
340 
341  //An awkward loop from intoIndex2+1 to outIndex2
342  i = intoIndex2;
343  while (i != outoIndex2)
344  {
345  i = (i + 1) % polygonShape.VerticesPrivate.Count;
346  Vector2 p3;
347  if (i == outoIndex2)
348  {
349  p3 = outoVec;
350  }
351  else
352  {
353  p3 = polygonShape.VerticesPrivate[i];
354  }
355 
356  //Add the triangle formed by intoVec,p2,p3
357  {
358  Vector2 e1 = p2 - intoVec;
359  Vector2 e2 = p3 - intoVec;
360 
361  float d = MathUtils.Cross(e1, e2);
362 
363  float triangleArea = 0.5f * d;
364 
365  area += triangleArea;
366 
367  // Area weighted centroid
368  center += triangleArea * kInv3 * (intoVec + p2 + p3);
369  }
370 
371  p2 = p3;
372  }
373 
374  //Normalize and transform centroid
375  center *= 1.0f / area;
376 
377  sc = MathUtils.Mul(ref xf, center);
378 
379  return area;
380  }
381  case ShapeType.Chain:
382  sc = Vector2.Zero;
383  return 0;
384  case ShapeType.Unknown:
385  case ShapeType.TypeCount:
386  throw new NotSupportedException();
387  default:
388  throw new ArgumentOutOfRangeException();
389  }
390  }
391  }
