126  public override void Update(float dt)
127  {
128  uniqueBodies.Clear();
129 
130  World.TestPointAllFixtures.ForEach(fixture =>
131  {
132  if (!fixture.Body.IsStatic && fixture.Body.Awake)
133  {
134  if (!uniqueBodies.Contains(fixture.Body))
135  {
136  uniqueBodies.Add(fixture.Body);
137  }
138  }
139  });
140 
141  foreach (Body body in uniqueBodies)
142  {
143  Vector2 areac = Vector2.Zero;
144  Vector2 massc = Vector2.Zero;
145  float area = 0;
146  float mass = 0;
147 
148  for (int j = 0; j < body.FixtureList.Count; j++)
149  {
150  Fixture fixture = body.FixtureList[j];
151 
152  if (fixture.Shape.ShapeType != ShapeType.Polygon && fixture.Shape.ShapeType != ShapeType.Circle)
153  {
154  continue;
155  }
156 
157  Shape shape = fixture.Shape;
158 
159  float sarea = ComputeSubmergedArea(shape, ref normal, offset, ref body.Xf, out Vector2 sc);
160  area += sarea;
161  areac.X += sarea * sc.X;
162  areac.Y += sarea * sc.Y;
163 
164  mass += sarea * shape.DensityPrivate;
165  massc.X += sarea * sc.X * shape.DensityPrivate;
166  massc.Y += sarea * sc.Y * shape.DensityPrivate;
167  }
168 
169  areac.X /= area;
170  areac.Y /= area;
171  massc.X /= mass;
172  massc.Y /= mass;
173 
174  if (area < MathConstants.Epsilon)
175  {
176  continue;
177  }
178 
179  //Buoyancy
180  Vector2 buoyancyForce = -Density * area * gravity;
181  body.ApplyForce(buoyancyForce, massc);
182 
183  //Linear drag
184  Vector2 dragForce = body.GetLinearVelocityFromWorldPoint(areac) - Velocity;
185  dragForce *= -LinearDragCoefficient * area;
186  body.ApplyForce(dragForce, areac);
187 
188  //Angular drag
189  body.ApplyTorque(-body.Inertia / body.Mass * area * body.AngularVelocity * AngularDragCoefficient);
190  }
191  }

204  private float ComputeSubmergedArea(Shape shape, ref Vector2 normal, float offset, ref Transform xf,
205  out Vector2 sc)
206  {
207  switch (shape.ShapeType)
208  {
209  case ShapeType.Circle:
210  {
211  CircleShape circleShape = (CircleShape) shape;
212 
213  sc = Vector2.Zero;
214 
215  float radius2 = circleShape.RadiusPrivate * circleShape.RadiusPrivate;
216 
217  Vector2 p = MathUtils.Mul(ref xf, circleShape.Position);
218  float l = -(Vector2.Dot(normal, p) - offset);
219  if (l < -circleShape.RadiusPrivate + MathConstants.Epsilon)
220 
221  //Completely dry
222  {
223  return 0;
224  }
225 
226  if (l > circleShape.RadiusPrivate)
227  {
228  //Completely wet
229  sc = p;
230  return MathConstants.Pi * radius2;
231  }
232 
233  //Magic
234  float l2 = l * l;
235  float area = radius2 * (float) (Math.Asin(l / circleShape.RadiusPrivate) + MathConstants.Pi / 2 +
236  l * Math.Sqrt(radius2 - l2));
237  float com = -2.0f / 3.0f * (float) Math.Pow(radius2 - l2, 1.5f) / area;
238 
239  sc.X = p.X + normal.X * com;
240  sc.Y = p.Y + normal.Y * com;
241 
242  return area;
243  }
244  case ShapeType.Edge:
245  sc = Vector2.Zero;
246  return 0;
247  case ShapeType.Polygon:
248  {
249  sc = Vector2.Zero;
250 
251  PolygonShape polygonShape = (PolygonShape) shape;
252 
253  //Transform plane into shape co-ordinates
254  Vector2 normalL = MathUtils.MulT(xf.Q, normal);
255  float offsetL = offset - Vector2.Dot(normal, xf.P);
256 
257  float[] depths = new float[Settings.MaxPolygonVertices];
258  int diveCount = 0;
259  int intoIndex = -1;
260  int outoIndex = -1;
261 
262  bool lastSubmerged = false;
263  int i;
264  for (i = 0; i < polygonShape.VerticesPrivate.Count; i++)
265  {
266  depths[i] = Vector2.Dot(normalL, polygonShape.VerticesPrivate[i]) - offsetL;
267  bool isSubmerged = depths[i] < -MathConstants.Epsilon;
268  if (i > 0)
269  {
270  if (isSubmerged)
271  {
272  if (!lastSubmerged)
273  {
274  intoIndex = i - 1;
275  diveCount++;
276  }
277  }
278  else
279  {
280  if (lastSubmerged)
281  {
282  outoIndex = i - 1;
283  diveCount++;
284  }
285  }
286  }
287 
288  lastSubmerged = isSubmerged;
289  }
290 
291  switch (diveCount)
292  {
293  case 0:
294  if (lastSubmerged)
295  {
296  //Completely submerged
297  sc = MathUtils.Mul(ref xf, polygonShape.MassDataPrivate.Centroid);
298  return polygonShape.MassDataPrivate.Mass / Density;
299  }
300 
301  //Completely dry
302  return 0;
303  case 1:
304  if (intoIndex == -1)
305  {
306  intoIndex = polygonShape.VerticesPrivate.Count - 1;
307  }
308  else
309  {
310  outoIndex = polygonShape.VerticesPrivate.Count - 1;
311  }
312 
313  break;
314  }
315 
316  int intoIndex2 = (intoIndex + 1) % polygonShape.VerticesPrivate.Count;
317  int outoIndex2 = (outoIndex + 1) % polygonShape.VerticesPrivate.Count;
318 
319  float intoLambda = (0 - depths[intoIndex]) / (depths[intoIndex2] - depths[intoIndex]);
320  float outoLambda = (0 - depths[outoIndex]) / (depths[outoIndex2] - depths[outoIndex]);
321 
322  Vector2 intoVec = new Vector2(
323  polygonShape.VerticesPrivate[intoIndex].X * (1 - intoLambda) +
324  polygonShape.VerticesPrivate[intoIndex2].X * intoLambda,
325  polygonShape.VerticesPrivate[intoIndex].Y * (1 - intoLambda) +
326  polygonShape.VerticesPrivate[intoIndex2].Y * intoLambda);
327  Vector2 outoVec = new Vector2(
328  polygonShape.VerticesPrivate[outoIndex].X * (1 - outoLambda) +
329  polygonShape.VerticesPrivate[outoIndex2].X * outoLambda,
330  polygonShape.VerticesPrivate[outoIndex].Y * (1 - outoLambda) +
331  polygonShape.VerticesPrivate[outoIndex2].Y * outoLambda);
332 
333  //Initialize accumulator
334  float area = 0;
335  Vector2 center = new Vector2(0, 0);
336  Vector2 p2 = polygonShape.VerticesPrivate[intoIndex2];
337 
338  const float kInv3 = 1.0f / 3.0f;
339 
340  //An awkward loop from intoIndex2+1 to outIndex2
341  i = intoIndex2;
342  while (i != outoIndex2)
343  {
344  i = (i + 1) % polygonShape.VerticesPrivate.Count;
345  Vector2 p3;
346  if (i == outoIndex2)
347  {
348  p3 = outoVec;
349  }
350  else
351  {
352  p3 = polygonShape.VerticesPrivate[i];
353  }
354 
355  //Add the triangle formed by intoVec,p2,p3
356  {
357  Vector2 e1 = p2 - intoVec;
358  Vector2 e2 = p3 - intoVec;
359 
360  float d = MathUtils.Cross(e1, e2);
361 
362  float triangleArea = 0.5f * d;
363 
364  area += triangleArea;
365 
366  // Area weighted centroid
367  center += triangleArea * kInv3 * (intoVec + p2 + p3);
368  }
369 
370  p2 = p3;
371  }
372 
373  //Normalize and transform centroid
374  center *= 1.0f / area;
375 
376  sc = MathUtils.Mul(ref xf, center);
377 
378  return area;
379  }
380  case ShapeType.Chain:
381  sc = Vector2.Zero;
382  return 0;
383  case ShapeType.Unknown:
384  case ShapeType.TypeCount:
385  throw new NotSupportedException();
386  default:
387  throw new ArgumentOutOfRangeException();
388  }
389  }
390  }
