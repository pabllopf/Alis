69  private static List<Vertices> TriangulatePolygon(Vertices vertices)
70  {
71  List<Vertices> list = new List<Vertices>();
72  Vector2F lowerInt = new Vector2F();
73  Vector2F upperInt = new Vector2F(); // intersection points
74  int lowerIndex = 0, upperIndex = 0;
75  Vertices lowerPoly, upperPoly;
76 
77  for (int i = 0; i < vertices.Count; ++i)
78  {
79  if (Reflex(i, vertices))
80  {
81  float upperDist;
82  float lowerDist = upperDist = float.MaxValue;
83  for (int j = 0; j < vertices.Count; ++j)
84  {
85  // if line intersects with an edge
86  float d;
87  Vector2F p;
88  if (Left(At(i - 1, vertices), At(i, vertices), At(j, vertices)) &&
89  RightOn(At(i - 1, vertices), At(i, vertices), At(j - 1, vertices)))
90  {
91  // find the point of intersection
92  p = Line.LineIntersect(At(i - 1, vertices), At(i, vertices), At(j, vertices),
93  At(j - 1, vertices));
94 
95  if (Right(At(i + 1, vertices), At(i, vertices), p))
96  {
97  // make sure it's inside the poly
98  d = SquareDist(At(i, vertices), p);
99  if (d < lowerDist)
100  {
101  // keep only the closest intersection
102  lowerDist = d;
103  lowerInt = p;
104  lowerIndex = j;
105  }
106  }
107  }
108 
109  if (Left(At(i + 1, vertices), At(i, vertices), At(j + 1, vertices)) &&
110  RightOn(At(i + 1, vertices), At(i, vertices), At(j, vertices)))
111  {
112  p = Line.LineIntersect(At(i + 1, vertices), At(i, vertices), At(j, vertices),
113  At(j + 1, vertices));
114 
115  if (Left(At(i - 1, vertices), At(i, vertices), p))
116  {
117  d = SquareDist(At(i, vertices), p);
118  if (d < upperDist)
119  {
120  upperDist = d;
121  upperIndex = j;
122  upperInt = p;
123  }
124  }
125  }
126  }
127 
128  // if there are no vertices to connect to, choose a point in the middle
129  if (lowerIndex == (upperIndex + 1) % vertices.Count)
130  {
131  Vector2F p = (lowerInt + upperInt) / 2;
132 
133  lowerPoly = Copy(i, upperIndex, vertices);
134  lowerPoly.Add(p);
135  upperPoly = Copy(lowerIndex, i, vertices);
136  upperPoly.Add(p);
137  }
138  else
139  {
140  double highestScore = 0, bestIndex = lowerIndex;
141  while (upperIndex < lowerIndex)
142  {
143  upperIndex += vertices.Count;
144  }
145 
146  for (int j = lowerIndex; j <= upperIndex; ++j)
147  {
148  if (CanSee(i, j, vertices))
149  {
150  double score = 1 / (SquareDist(At(i, vertices), At(j, vertices)) + 1);
151  if (Reflex(j, vertices))
152  {
153  if (RightOn(At(j - 1, vertices), At(j, vertices), At(i, vertices)) &&
154  LeftOn(At(j + 1, vertices), At(j, vertices), At(i, vertices)))
155  {
156  score += 3;
157  }
158  else
159  {
160  score += 2;
161  }
162  }
163  else
164  {
165  score += 1;
166  }
167 
168  if (score > highestScore)
169  {
170  bestIndex = j;
171  highestScore = score;
172  }
173  }
174  }
175 
176  lowerPoly = Copy(i, (int) bestIndex, vertices);
177  upperPoly = Copy((int) bestIndex, i, vertices);
178  }
179 
180  list.AddRange(TriangulatePolygon(lowerPoly));
181  list.AddRange(TriangulatePolygon(upperPoly));
182  return list;
183  }
184  }
185 
186  // polygon is already convex
187  if (vertices.Count > Settings.PolygonVertices)
188  {
189  lowerPoly = Copy(0, vertices.Count / 2, vertices);
190  upperPoly = Copy(vertices.Count / 2, 0, vertices);
191  list.AddRange(TriangulatePolygon(lowerPoly));
192  list.AddRange(TriangulatePolygon(upperPoly));
193  }
194  else
195  {
196  list.Add(vertices);
197  }
198 
199  return list;
200  }

208  private static Vector2F At(int i, Vertices vertices)
209  {
210  int s = vertices.Count;
211  return vertices[i < 0 ? s - 1 - (-i - 1) % s : i % s];
212  }

221  private static Vertices Copy(int i, int j, Vertices vertices)
222  {
223  while (j < i)
224  {

226  }

245  private static bool CanSee(int i, int j, Vertices vertices)
246  {
247  if (Reflex(i, vertices))
248  {
249  if (LeftOn(At(i, vertices), At(i - 1, vertices), At(j, vertices)) &&
250  RightOn(At(i, vertices), At(i + 1, vertices), At(j, vertices)))
251  {
252  return false;
253  }
254  }
255  else
256  {
257  if (RightOn(At(i, vertices), At(i + 1, vertices), At(j, vertices)) ||
258  LeftOn(At(i, vertices), At(i - 1, vertices), At(j, vertices)))
259  {
260  return false;
261  }
262  }
263 
264  if (Reflex(j, vertices))
265  {
266  if (LeftOn(At(j, vertices), At(j - 1, vertices), At(i, vertices)) &&
267  RightOn(At(j, vertices), At(j + 1, vertices), At(i, vertices)))
268  {
269  return false;
270  }
271  }
272  else
273  {
274  if (RightOn(At(j, vertices), At(j + 1, vertices), At(i, vertices)) ||
275  LeftOn(At(j, vertices), At(j - 1, vertices), At(i, vertices)))
276  {
277  return false;
278  }
279  }
280 
281  for (int k = 0; k < vertices.Count; ++k)
282  {
283  if ((k + 1) % vertices.Count == i || k == i || (k + 1) % vertices.Count == j || k == j)
284  {
285  continue; // ignore incident edges
286  }
287 
288  if (Line.LineIntersect(At(i, vertices), At(j, vertices), At(k, vertices), At(k + 1, vertices),
289  out _))
290  {
291  return false;
292  }
293  }
294 
295  return true;
296  }

357  private static float SquareDist(Vector2F a, Vector2F b)
358  {
