153  public void CreateGLResources(SamplerDescription description, bool mipmapped, GraphicsBackend backend)
154  {

158  glSamplerParameteri(_sampler, SamplerParameterName.TextureWrapS, (int)OpenGLFormats.VdToGLTextureWrapMode(description.AddressModeU));
159  CheckLastError();

161  CheckLastError();
162  glSamplerParameteri(_sampler, SamplerParameterName.TextureWrapR, (int)OpenGLFormats.VdToGLTextureWrapMode(description.AddressModeW));
163  CheckLastError();
164 
165  if (description.AddressModeU == SamplerAddressMode.Border
166  || description.AddressModeV == SamplerAddressMode.Border
167  || description.AddressModeW == SamplerAddressMode.Border)
168  {
169  RgbaFloat borderColor = ToRgbaFloat(description.BorderColor);
170  glSamplerParameterfv(_sampler, SamplerParameterName.TextureBorderColor, (float*)&borderColor);
171  CheckLastError();
172  }
173 
174  glSamplerParameterf(_sampler, SamplerParameterName.TextureMinLod, description.MinimumLod);
175  CheckLastError();
176  glSamplerParameterf(_sampler, SamplerParameterName.TextureMaxLod, description.MaximumLod);
177  CheckLastError();
178  if (backend == GraphicsBackend.OpenGL && description.LodBias != 0)
179  {
180  glSamplerParameterf(_sampler, SamplerParameterName.TextureLodBias, description.LodBias);
181  CheckLastError();
182  }
183 
184  if (description.Filter == SamplerFilter.Anisotropic)
185  {
186  glSamplerParameterf(_sampler, SamplerParameterName.TextureMaxAnisotropyExt, description.MaximumAnisotropy);
187  CheckLastError();
188  glSamplerParameteri(_sampler, SamplerParameterName.TextureMinFilter, mipmapped ? (int)TextureMinFilter.LinearMipmapLinear : (int)TextureMinFilter.Linear);
189  CheckLastError();
190  glSamplerParameteri(_sampler, SamplerParameterName.TextureMagFilter, (int)TextureMagFilter.Linear);
191  CheckLastError();
192  }
193  else
194  {
195  OpenGLFormats.VdToGLTextureMinMagFilter(description.Filter, mipmapped, out TextureMinFilter min, out TextureMagFilter mag);
196  glSamplerParameteri(_sampler, SamplerParameterName.TextureMinFilter, (int)min);
197  CheckLastError();
198  glSamplerParameteri(_sampler, SamplerParameterName.TextureMagFilter, (int)mag);
199  CheckLastError();
200  }
201 
202  if (description.ComparisonKind != null)
203  {
204  glSamplerParameteri(_sampler, SamplerParameterName.TextureCompareMode, (int)TextureCompareMode.CompareRefToTexture);
205  CheckLastError();
206  glSamplerParameteri(_sampler, SamplerParameterName.TextureCompareFunc, (int)OpenGLFormats.VdToGLDepthFunction(description.ComparisonKind.Value));
207  CheckLastError();
208  }
