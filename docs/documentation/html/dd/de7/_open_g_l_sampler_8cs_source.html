154  public void CreateGLResources(SamplerDescription description, bool mipmapped, GraphicsBackend backend)
155  {
156  glGenSamplers(1, out _sampler);
157  CheckLastError();
158 
159  glSamplerParameteri(_sampler, SamplerParameterName.TextureWrapS, (int)OpenGLFormats.VdToGLTextureWrapMode(description.AddressModeU));
160  CheckLastError();
161  glSamplerParameteri(_sampler, SamplerParameterName.TextureWrapT, (int)OpenGLFormats.VdToGLTextureWrapMode(description.AddressModeV));
162  CheckLastError();
163  glSamplerParameteri(_sampler, SamplerParameterName.TextureWrapR, (int)OpenGLFormats.VdToGLTextureWrapMode(description.AddressModeW));
164  CheckLastError();
165 
166  if (description.AddressModeU == SamplerAddressMode.Border
167  || description.AddressModeV == SamplerAddressMode.Border
168  || description.AddressModeW == SamplerAddressMode.Border)
169  {
170  RgbaFloat borderColor = ToRgbaFloat(description.BorderColor);
171  glSamplerParameterfv(_sampler, SamplerParameterName.TextureBorderColor, (float*)&borderColor);
172  CheckLastError();
173  }
174 
175  glSamplerParameterf(_sampler, SamplerParameterName.TextureMinLod, description.MinimumLod);
176  CheckLastError();
177  glSamplerParameterf(_sampler, SamplerParameterName.TextureMaxLod, description.MaximumLod);
178  CheckLastError();
179  if (backend == GraphicsBackend.OpenGL && description.LodBias != 0)
180  {
181  glSamplerParameterf(_sampler, SamplerParameterName.TextureLodBias, description.LodBias);
182  CheckLastError();
183  }
184 
185  if (description.Filter == SamplerFilter.Anisotropic)
186  {
187  glSamplerParameterf(_sampler, SamplerParameterName.TextureMaxAnisotropyExt, description.MaximumAnisotropy);
188  CheckLastError();
189  glSamplerParameteri(_sampler, SamplerParameterName.TextureMinFilter, mipmapped ? (int)TextureMinFilter.LinearMipmapLinear : (int)TextureMinFilter.Linear);
190  CheckLastError();
191  glSamplerParameteri(_sampler, SamplerParameterName.TextureMagFilter, (int)TextureMagFilter.Linear);
192  CheckLastError();
193  }
194  else
195  {
196  OpenGLFormats.VdToGLTextureMinMagFilter(description.Filter, mipmapped, out TextureMinFilter min, out TextureMagFilter mag);
197  glSamplerParameteri(_sampler, SamplerParameterName.TextureMinFilter, (int)min);
198  CheckLastError();
199  glSamplerParameteri(_sampler, SamplerParameterName.TextureMagFilter, (int)mag);
200  CheckLastError();
201  }
202 
203  if (description.ComparisonKind != null)
204  {
205  glSamplerParameteri(_sampler, SamplerParameterName.TextureCompareMode, (int)TextureCompareMode.CompareRefToTexture);
206  CheckLastError();
207  glSamplerParameteri(_sampler, SamplerParameterName.TextureCompareFunc, (int)OpenGLFormats.VdToGLDepthFunction(description.ComparisonKind.Value));
208  CheckLastError();
209  }
210  }
