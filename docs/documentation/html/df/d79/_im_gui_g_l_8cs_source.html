64  public static (IntPtr, IntPtr) CreateWindowAndGlContext(string title, int width, int height, bool fullscreen = false, bool highDpi = false)
65  {
66  // initialize SDL and set a few defaults for the OpenGL context
67  Init(InitVideo);
68  GlSetAttributeByInt(SdlGlAttr.SdlGlContextFlags, (int) SdlGlContext.SdlGlContextForwardCompatibleFlag);
69  GlSetAttributeByProfile(SdlGlAttr.SdlGlContextProfileMask, SdlGlProfile.SdlGlContextProfileCore);
70  GlSetAttributeByInt(SdlGlAttr.SdlGlContextMajorVersion, 3);
71  GlSetAttributeByInt(SdlGlAttr.SdlGlContextMinorVersion, 2);
72 
73  GlSetAttributeByProfile(SdlGlAttr.SdlGlContextProfileMask, SdlGlProfile.SdlGlContextProfileCore);
74  GlSetAttributeByInt(SdlGlAttr.SdlGlDoubleBuffer, 1);
75  GlSetAttributeByInt(SdlGlAttr.SdlGlDepthSize, 24);
76  GlSetAttributeByInt(SdlGlAttr.SdlGlAlphaSize, 8);
77  GlSetAttributeByInt(SdlGlAttr.SdlGlStencilSize, 8);
78 
79  // create the window which should be able to have a valid OpenGL context and is resizable
80  SdlWindowFlags flags = SdlWindowFlags.SdlWindowOpengl | SdlWindowFlags.SdlWindowResizable;
81  if (fullscreen)
82  {
83  flags |= SdlWindowFlags.SdlWindowFullscreen;
84  }
85 
86  if (highDpi)
87  {
88  flags |= SdlWindowFlags.SdlWindowAllowHighdpi;
89  }
90 
91  IntPtr window = CreateWindow(title, WindowPosCentered, WindowPosCentered, width, height, flags);
92  IntPtr glContext = CreateGlContext(window);
93  return (window, glContext);
94  }

102  private static IntPtr CreateGlContext(IntPtr window)
103  {
104  IntPtr glContext = GlCreateContext(window);
105  if (glContext == IntPtr.Zero)
106  {
107  throw new Exception("CouldNotCreateContext");
108  }
109 
110  GlMakeCurrent(window, glContext);
111  GlSetSwapInterval(1);
112 
113  // initialize the screen to black as soon as possible
114  GlClearColor(0f, 0f, 0f, 1f);
115  GlClear(ClearBufferMask.ColorBufferBit);
116  GlSwapWindow(window);
117 
118  Console.WriteLine($"GL Version: {GlGetString(StringName.Version)}");
119  return glContext;
120  }

131  public static uint LoadTexture(IntPtr pixelData, int width, int height, PixelFormat format = PixelFormat.Rgba, PixelInternalFormat internalFormat = PixelInternalFormat.Rgba)
132  {
133  uint textureId = GenTexture();

138  GlTexParameteri(TextureTarget.Texture2D, TextureParameterName.TextureMinFilter, TextureParameter.Linear);
