103  public static (IntPtr, IntPtr) CreateWindowAndGlContext(string title, int width, int height, bool fullscreen = false, bool highDpi = false)
104  {
105  // initialize SDL and set a few defaults for the OpenGL context
106  SDL_Init(SdlInitVideo);
107  SDL_GL_SetAttribute(SdlGLattr.SdlGlContextFlags, (int) SdlGLcontext.SdlGlContextForwardCompatibleFlag);
108  SDL_GL_SetAttribute(SdlGLattr.SdlGlContextProfileMask, SdlGLprofile.SdlGlContextProfileCore);
109  SDL_GL_SetAttribute(SdlGLattr.SdlGlContextMajorVersion, 3);
110  SDL_GL_SetAttribute(SdlGLattr.SdlGlContextMinorVersion, 2);
111 
112  SDL_GL_SetAttribute(SdlGLattr.SdlGlContextProfileMask, SdlGLprofile.SdlGlContextProfileCore);
113  SDL_GL_SetAttribute(SdlGLattr.SdlGlDoublebuffer, 1);
114  SDL_GL_SetAttribute(SdlGLattr.SdlGlDepthSize, 24);
115  SDL_GL_SetAttribute(SdlGLattr.SdlGlAlphaSize, 8);
116  SDL_GL_SetAttribute(SdlGLattr.SdlGlStencilSize, 8);
117 
118  // create the window which should be able to have a valid OpenGL context and is resizable
119  SdlWindowFlags flags = SdlWindowFlags.SdlWindowOpengl | SdlWindowFlags.SdlWindowResizable;
120  if (fullscreen)
121  {
122  flags |= SdlWindowFlags.SdlWindowFullscreen;
123  }
124 
125  if (highDpi)
126  {
127  flags |= SdlWindowFlags.SdlWindowAllowHighdpi;
128  }
129 
130  IntPtr window = SDL_CreateWindow(title, SdlWindowposCentered, SdlWindowposCentered, width, height, flags);
131  IntPtr glContext = CreateGlContext(window);
132  return (window, glContext);
133  }

141  private static IntPtr CreateGlContext(IntPtr window)
142  {
143  IntPtr glContext = SDL_GL_CreateContext(window);
144  if (glContext == IntPtr.Zero)
145  {
146  throw new Exception("CouldNotCreateContext");
147  }
148 
149  SDL_GL_MakeCurrent(window, glContext);
150  SDL_GL_SetSwapInterval(1);
151 
152  // initialize the screen to black as soon as possible
153  glClearColor(0f, 0f, 0f, 1f);
154  glClear(ClearBufferMask.ColorBufferBit);
155  SDL_GL_SwapWindow(window);
156 
157  Console.WriteLine($"GL Version: {glGetString(StringName.Version)}");
158  return glContext;
159  }

170  public static uint LoadTexture(IntPtr pixelData, int width, int height, PixelFormat format = PixelFormat.Rgba, PixelInternalFormat internalFormat = PixelInternalFormat.Rgba)
171  {

177  glTexParameteri(TextureTarget.Texture2D, TextureParameterName.TextureMinFilter, TextureParameter.Linear);
