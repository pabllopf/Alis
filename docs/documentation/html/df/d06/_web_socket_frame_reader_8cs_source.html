95  public static async Task<WebSocketReadCursor> ReadAsync(Stream fromStream, ArraySegment<byte> intoBuffer,
96  CancellationToken cancellationToken)
97  {
98  // allocate a small buffer to read small chunks of data from the stream
99  ArraySegment<byte> smallBuffer = new ArraySegment<byte>(new byte[8]);
100 
101  await BinaryReaderWriter.ReadExactly(2, fromStream, smallBuffer, cancellationToken);
102  byte byte1 = smallBuffer.Array[0];
103  byte byte2 = smallBuffer.Array[1];
104 
105  // process first byte
106  byte finBitFlag = 0x80;
107  byte opCodeFlag = 0x0F;
108  bool isFinBitSet = (byte1 & finBitFlag) == finBitFlag;
109  WebSocketOpCode opCode = (WebSocketOpCode) (byte1 & opCodeFlag);
110 
111  // read and process second byte
112  byte maskFlag = 0x80;
113  bool isMaskBitSet = (byte2 & maskFlag) == maskFlag;
114  uint len = await ReadLength(byte2, smallBuffer, fromStream, cancellationToken);
115  int count = (int) len;
116  int minCount = CalculateNumBytesToRead(count, intoBuffer.Count);
117  ArraySegment<byte> maskKey = new ArraySegment<byte>();
118 
119  try
120  {
121  // use the masking key to decode the data if needed
122  if (isMaskBitSet)
123  {
124  maskKey = new ArraySegment<byte>(smallBuffer.Array, 0, WebSocketFrameCommon.MaskKeyLength);
125  await BinaryReaderWriter.ReadExactly(maskKey.Count, fromStream, maskKey, cancellationToken);
126  await BinaryReaderWriter.ReadExactly(minCount, fromStream, intoBuffer, cancellationToken);
127  ArraySegment<byte> payloadToMask =
128  new ArraySegment<byte>(intoBuffer.Array, intoBuffer.Offset, minCount);
129  WebSocketFrameCommon.ToggleMask(maskKey, payloadToMask);
130  }
131  else
132  {
133  await BinaryReaderWriter.ReadExactly(minCount, fromStream, intoBuffer, cancellationToken);
134  }
135  }
136  catch (InternalBufferOverflowException e)
137  {
138  throw new InternalBufferOverflowException(
139  $"Supplied buffer too small to read {0} bytes from {Enum.GetName(typeof(WebSocketOpCode), opCode)} frame",
140  e);
141  }
142 
143  WebSocketFrame frame;
144  if (opCode == WebSocketOpCode.ConnectionClose)
145  {
146  frame = DecodeCloseFrame(isFinBitSet, opCode, count, intoBuffer, maskKey);
147  }
148  else
149  {
150  // note that by this point the payload will be populated
151  frame = new WebSocketFrame(isFinBitSet, opCode, count, maskKey);
152  }
153 
154  return new WebSocketReadCursor(frame, minCount, count - minCount);
155  }

160  private static WebSocketFrame DecodeCloseFrame(bool isFinBitSet, WebSocketOpCode opCode, int count,
161  ArraySegment<byte> buffer, ArraySegment<byte> maskKey)
162  {
163  WebSocketCloseStatus closeStatus;
164  string closeStatusDescription;
165 
166  if (count >= 2)
167  {
168  Array.Reverse(buffer.Array, buffer.Offset, 2); // network byte order
169  int closeStatusCode = BitConverter.ToUInt16(buffer.Array, buffer.Offset);
170  if (Enum.IsDefined(typeof(WebSocketCloseStatus), closeStatusCode))
171  {
172  closeStatus = (WebSocketCloseStatus) closeStatusCode;
173  }
174  else
175  {
176  closeStatus = WebSocketCloseStatus.Empty;
177  }
178 
179  int offset = buffer.Offset + 2;
180  int descCount = count - 2;
181 
182  if (descCount > 0)
183  {
184  closeStatusDescription = Encoding.UTF8.GetString(buffer.Array, offset, descCount);
185  }
186  else
187  {
188  closeStatusDescription = null;
189  }
190  }
191  else
192  {
193  closeStatus = WebSocketCloseStatus.Empty;
194  closeStatusDescription = null;
195  }
196 
197  return new WebSocketFrame(isFinBitSet, opCode, count, closeStatus, closeStatusDescription, maskKey);
198  }

203  private static async Task<uint> ReadLength(byte byte2, ArraySegment<byte> smallBuffer, Stream fromStream,
204  CancellationToken cancellationToken)
205  {
206  byte payloadLenFlag = 0x7F;
207  uint len = (uint) (byte2 & payloadLenFlag);
208 
209  // read a short length or a long length depending on the value of len
210  if (len == 126)
211  {
212  len = await BinaryReaderWriter.ReadUShortExactly(fromStream, false, smallBuffer, cancellationToken);
213  }
214  else if (len == 127)
215  {
216  len = (uint) await BinaryReaderWriter.ReadULongExactly(fromStream, false, smallBuffer,
217  cancellationToken);
218  const uint
219  maxLen = 2147483648; // 2GB - not part of the spec but just a precaution. Send large volumes of data in smaller frames.
220 
221  // protect ourselves against bad data
222  if (len > maxLen || len < 0)
223  {
224  throw new ArgumentOutOfRangeException(
225  $"Payload length out of range. Min 0 max 2GB. Actual {len:#,##0} bytes.");
226  }
227  }
228 
229  return len;
230  }
