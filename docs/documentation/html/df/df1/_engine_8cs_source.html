75  public void Start()
76  {
77  // create a window, GL context and our ImGui renderer
78  // this is fast solution for create SDL_Window and SDL_Render
79  (_window, _glContext) = ImGuiGl.CreateWindowAndGlContext("SDL Window (OpenGL)", 800, 600);
80  _renderer = new ImGuiGlRenderer(_window, _glContext);
81 
82  while (!_quit)
83  {
84  while (Sdl.PollEvent(out SdlEvent e) != 0)
85  {
86  _renderer.ProcessEvent(e);
87  switch (e.type)
88  {
89  case SdlEventType.SdlQuit:
90  {
91  _quit = true;
92  break;
93  }
94  case SdlEventType.SdlKeydown:
95  {
96  switch (e.key.keysym.sym)
97  {
98  case SdlKeycode.SdlkEscape:
99  case SdlKeycode.SdlkQ:
100  _quit = true;
101  break;
102  }
103 
104  break;
105  }
106  }
107  }
108 
109  _renderer.ClearColor(0.05f, 0.05f, 0.05f, 1.00f);
110  _renderer.NewFrame();
111  ImGui.ShowDemoWindow();
112  _renderer.Render();
113  Sdl.GlSwapWindow(_window);
114  }
115 
116  Sdl.GlDeleteContext(_glContext);
117  Sdl.DestroyWindow(_window);
118  Sdl.Quit();
119  }
120  }
