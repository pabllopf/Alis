77  public void Start()
78  {
79  // create a window, GL context and our ImGui renderer
80  // this is fast solution for create SDL_Window and SDL_Render
81  (_window, _glContext) = ImGuiGl.CreateWindowAndGlContext("SDL Window (OpenGL)", 800, 600);
82  _renderer = new ImGuiGlRenderer(_window, _glContext);
83 
84  while (!_quit)
85  {
86  while (Sdl.PollEvent(out SdlEvent e) != 0)
87  {
88  _renderer.ProcessEvent(e);
89  switch (e.type)
90  {
91  case SdlEventType.SdlQuit:
92  {
93  _quit = true;
94  break;
95  }
96  case SdlEventType.SdlKeydown:
97  {
98  switch (e.key.keysym.sym)
99  {
100  case SdlKeycode.SdlkEscape:
101  case SdlKeycode.SdlkQ:
102  _quit = true;
103  break;
104  }
105 
106  break;
107  }
108  }
109  }
110 
111  _renderer.ClearColor(0.05f, 0.05f, 0.05f, 1.00f);
112  _renderer.NewFrame();
113 
114 
115  ImGui.ShowDemoWindow();
116 
117  // ImNodes.ShowDemoWindow();
118  ImGui.Begin("simple node editor");
119 
120  ImNodes.BeginNodeEditor();
121  ImNodes.BeginNode(1);
122 
123  ImNodes.BeginNodeTitleBar();
124  ImGui.TextUnformatted("simple node :)");
125  ImNodes.EndNodeTitleBar();
126 
127  ImNodes.BeginInputAttribute(2);
128  ImGui.Text("input");
129  ImNodes.EndInputAttribute();
130 
131  ImNodes.BeginOutputAttribute(3);
132  ImGui.Indent(40);
133  ImGui.Text("output");
134  ImNodes.EndOutputAttribute();
135 
136  ImNodes.EndNode();
137  ImNodes.EndNodeEditor();
138 
139  ImGui.End();
140 
141  ImPlot.ShowDemoWindow();
142 
143 
144  _renderer.Render();
145  Sdl.GlSwapWindow(_window);
146  }
147 
148  Sdl.GlDeleteContext(_glContext);
149  Sdl.DestroyWindow(_window);
150  Sdl.Quit();
151  }
152  }
