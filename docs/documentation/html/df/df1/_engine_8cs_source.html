571  public void ProcessEvent(SdlEvent evt)
572  {
573  ImGuiIoPtr io = ImGui.GetIo();
574  switch (evt.type)
575  {
576  case SdlEventType.SdlMousewheel:
577  {
578  if (evt.wheel.x > 0)
579  {
580  io.MouseWheelH += 1;
581  }
582 
583  if (evt.wheel.x < 0)
584  {
585  io.MouseWheelH -= 1;
586  }
587 
588  if (evt.wheel.y > 0)
589  {
590  io.MouseWheel += 1;
591  }
592 
593  if (evt.wheel.y < 0)
594  {
595  io.MouseWheel -= 1;
596  }
597 
598  return;
599  }
600  case SdlEventType.SdlMouseButtonDown:
601  {
602  if (evt.button.button == Sdl.ButtonLeft)
603  {
604  _mousePressed[0] = true;
605  }
606 
607  if (evt.button.button == Sdl.ButtonRight)
608  {
609  _mousePressed[1] = true;
610  }
611 
612  if (evt.button.button == Sdl.ButtonMiddle)
613  {
614  _mousePressed[2] = true;
615  }
616 
617  return;
618  }
619  case SdlEventType.SdlTextInput:
620  {
621  string str = Encoding.UTF8.GetString(evt.text.Text);
622  io.AddInputCharactersUtf8(str);
623  return;
624  }
625  case SdlEventType.SdlKeydown:
626  case SdlEventType.SdlKeyup:
627  {
628  SdlScancode key = evt.key.keysym.scancode;
629  io.KeysDown[(int) key] = evt.type == SdlEventType.SdlKeydown;
630  Console.WriteLine("io.KeysDown[" + key + "] = " + evt.type + io.KeysDown[(int) key]);
631  io.KeyShift = (Sdl.GetModState() & SdlKeymod.KmodShift) != 0;
632  io.KeyCtrl = (Sdl.GetModState() & SdlKeymod.KmodCtrl) != 0;
633  io.KeyAlt = (Sdl.GetModState() & SdlKeymod.KmodAlt) != 0;
634  io.KeySuper = (Sdl.GetModState() & SdlKeymod.KmodGui) != 0;
635  break;
636  }
637  }

643  private void UpdateMousePosAndButtons()
644  {
645  ImGuiIoPtr io = ImGui.GetIo();
646 
647  // Set OS mouse position if requested (rarely used, only when ImGuiConfigFlags_NavEnableSetMousePos is enabled by user)
648  if (io.WantSetMousePos)
649  {
650  Sdl.WarpMouseInWindow(_window, (int) io.MousePos.X, (int) io.MousePos.Y);
651  }
652  else
653  {
654  io.MousePos = new Vector2(float.MinValue, float.MinValue);
655  }
656 
657  uint mouseButtons = Sdl.GetMouseStateOutXAndY(out int mx, out int my);
658  io.MouseDown[0] =
659  _mousePressed[0] ||
660  (mouseButtons & Sdl.Button(Sdl.ButtonLeft)) !=
661  0; // If a mouse press event came, always pass it as "mouse held this frame", so we don't miss click-release events that are shorter than 1 frame.
662  io.MouseDown[1] = _mousePressed[1] || (mouseButtons & Sdl.Button(Sdl.ButtonRight)) != 0;
663  io.MouseDown[2] = _mousePressed[2] || (mouseButtons & Sdl.Button(Sdl.ButtonMiddle)) != 0;
664  _mousePressed[0] = _mousePressed[1] = _mousePressed[2] = false;
665 
666  IntPtr focusedWindow = Sdl.GetKeyboardFocus();
667  if (_window == focusedWindow)
668  {
669  // SDL_GetMouseState() gives mouse position seemingly based on the last window entered/focused(?)
670  // The creation of a new windows at runtime and SDL_CaptureMouse both seems to severely mess up with that, so we retrieve that position globally.
671  Sdl.GetWindowPosition(focusedWindow, out int wx, out int wy);
672  Sdl.GetGlobalMouseStateOutXAndOutY(out mx, out my);
673  mx -= wx;
674  my -= wy;
675  io.MousePos = new Vector2(mx, my);
676  }
677 
678  // SDL_CaptureMouse() let the OS know e.g. that our imgui drag outside the SDL window boundaries shouldn't e.g. trigger the OS window resize cursor.
679  bool anyMouseButtonDown = ImGui.IsAnyMouseDown();
680  Sdl.CaptureMouse(anyMouseButtonDown ? SdlBool.SdlTrue : SdlBool.SdlFalse);
681  }

689  private void SetupRenderState(ImDrawDataPtr drawData, int fbWidth, int fbHeight)
690  {
691  Gl.GlEnable(EnableCap.Blend);
692  Gl.GlBlendEquation(BlendEquationMode.FuncAdd);
693  Gl.GlBlendFunc(BlendingFactorSrc.SrcAlpha, BlendingFactorDest.OneMinusSrcAlpha);
694  Gl.GlDisable(EnableCap.CullFace);
695  Gl.GlDisable(EnableCap.DepthTest);
696  Gl.GlEnable(EnableCap.ScissorTest);
697 
698  Gl.GlUseProgram(_shader.ProgramId);
699 
700  float left = drawData.DisplayPos.X;
701  float right = drawData.DisplayPos.X + drawData.DisplaySize.X;
702  float top = drawData.DisplayPos.Y;
703  float bottom = drawData.DisplayPos.Y + drawData.DisplaySize.Y;
704 
705  _shader["Texture"].SetValue(0);
706  _shader["ProjMtx"].SetValue(Matrix4X4F.CreateOrthographicOffCenter(left, right, bottom, top, -1, 1));
707  Gl.GlBindSampler(0, 0);
708 
709  Gl.GlBindVertexArray(_vertexArrayObject);
710 
711  // Bind vertex/index buffers and setup attributes for ImDrawVert
712  Gl.GlBindBuffer(BufferTarget.ArrayBuffer, _vboHandle);
713  Gl.GlBindBuffer(BufferTarget.ElementArrayBuffer, _elementsHandle);
714 
715  Gl.EnableVertexAttribArray(_shader["Position"].Location);
716  Gl.EnableVertexAttribArray(_shader["UV"].Location);
717  Gl.EnableVertexAttribArray(_shader["Color"].Location);
718 
719  int drawVertSize = Marshal.SizeOf<ImDrawVert>();
720  Gl.VertexAttribPointer(_shader["Position"].Location, 2, VertexAttribPointerType.Float, false, drawVertSize, Marshal.OffsetOf<ImDrawVert>("Pos"));
721  Gl.VertexAttribPointer(_shader["UV"].Location, 2, VertexAttribPointerType.Float, false, drawVertSize, Marshal.OffsetOf<ImDrawVert>("Uv"));
