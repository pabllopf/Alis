528  public void ProcessEvent(SdlEvent evt)
529  {
530  ImGuiIOPtr io = ImGui.GetIO();
531  switch (evt.type)
532  {
533  case SdlEventType.SdlMousewheel:
534  {
535  if (evt.wheel.x > 0)
536  {
537  io.MouseWheelH += 1;
538  }
539 
540  if (evt.wheel.x < 0)
541  {
542  io.MouseWheelH -= 1;
543  }
544 
545  if (evt.wheel.y > 0)
546  {
547  io.MouseWheel += 1;
548  }
549 
550  if (evt.wheel.y < 0)
551  {
552  io.MouseWheel -= 1;
553  }
554 
555  return;
556  }
557  case SdlEventType.SdlMouseButtonDown:
558  {
559  if (evt.button.button == Sdl.ButtonLeft)
560  {
561  _mousePressed[0] = true;
562  }
563 
564  if (evt.button.button == Sdl.ButtonRight)
565  {
566  _mousePressed[1] = true;
567  }
568 
569  if (evt.button.button == Sdl.ButtonMiddle)
570  {
571  _mousePressed[2] = true;
572  }
573 
574  return;
575  }
576  case SdlEventType.SdlTextInput:
577  {
578  string str = Encoding.UTF8.GetString(evt.text.Text);
579  io.AddInputCharactersUTF8(str);
580  return;
581  }
582  case SdlEventType.SdlKeydown:
583  case SdlEventType.SdlKeyup:
584  {
585  SdlScancode key = evt.key.keysym.scancode;
586  io.KeysDown[(int) key] = evt.type == SdlEventType.SdlKeydown;
587  Console.WriteLine("io.KeysDown[" + key + "] = " + evt.type + io.KeysDown[(int) key]);
588  io.KeyShift = (Sdl.GetModState() & SdlKeymod.KmodShift) != 0;
589  io.KeyCtrl = (Sdl.GetModState() & SdlKeymod.KmodCtrl) != 0;
590  io.KeyAlt = (Sdl.GetModState() & SdlKeymod.KmodAlt) != 0;
591  io.KeySuper = (Sdl.GetModState() & SdlKeymod.KmodGui) != 0;
592  break;
593  }
594  }
595  }

600  private void UpdateMousePosAndButtons()
601  {
602  ImGuiIOPtr io = ImGui.GetIO();
603 
604  // Set OS mouse position if requested (rarely used, only when ImGuiConfigFlags_NavEnableSetMousePos is enabled by user)
605  if (io.WantSetMousePos)
606  {
607  Sdl.WarpMouseInWindow(_window, (int) io.MousePos.X, (int) io.MousePos.Y);
608  }
609  else
610  {
611  io.MousePos = new Vector2(float.MinValue, float.MinValue);
612  }
613 
614  uint mouseButtons = Sdl.GetMouseStateOutXAndY(out int mx, out int my);
615  io.MouseDown[0] =
616  _mousePressed[0] ||
617  (mouseButtons & Sdl.Button(Sdl.ButtonLeft)) !=
618  0; // If a mouse press event came, always pass it as "mouse held this frame", so we don't miss click-release events that are shorter than 1 frame.
619  io.MouseDown[1] = _mousePressed[1] || (mouseButtons & Sdl.Button(Sdl.ButtonRight)) != 0;
620  io.MouseDown[2] = _mousePressed[2] || (mouseButtons & Sdl.Button(Sdl.ButtonMiddle)) != 0;
621  _mousePressed[0] = _mousePressed[1] = _mousePressed[2] = false;
622 
623  IntPtr focusedWindow = Sdl.GetKeyboardFocus();
624  if (_window == focusedWindow)
625  {
626  // SDL_GetMouseState() gives mouse position seemingly based on the last window entered/focused(?)
627  // The creation of a new windows at runtime and SDL_CaptureMouse both seems to severely mess up with that, so we retrieve that position globally.
628  Sdl.GetWindowPosition(focusedWindow, out int wx, out int wy);
629  Sdl.GetGlobalMouseStateOutXAndOutY(out mx, out my);
630  mx -= wx;
631  my -= wy;
632  io.MousePos = new Vector2(mx, my);
633  }
634 
635  // SDL_CaptureMouse() let the OS know e.g. that our imgui drag outside the SDL window boundaries shouldn't e.g. trigger the OS window resize cursor.
636  bool anyMouseButtonDown = ImGui.IsAnyMouseDown();
637  Sdl.CaptureMouse(anyMouseButtonDown ? SdlBool.SdlTrue : SdlBool.SdlFalse);
638  }

647  {
648  Gl. GlEnable(EnableCap.Blend);
649  Gl. GlBlendEquation(BlendEquationMode.FuncAdd);
650  Gl. GlBlendFunc(BlendingFactorSrc.SrcAlpha, BlendingFactorDest.OneMinusSrcAlpha);
651  Gl. GlDisable(EnableCap.CullFace);
652  Gl. GlDisable(EnableCap.DepthTest);
653  Gl.GlEnable(EnableCap.ScissorTest);
654 
655  Gl.GlUseProgram(_shader.ProgramId);
656 
657  float left = drawData.DisplayPos.X;
658  float right = drawData.DisplayPos.X + drawData.DisplaySize.X;
659  float top = drawData.DisplayPos.Y;
660  float bottom = drawData.DisplayPos.Y + drawData.DisplaySize.Y;
661 
662  _shader["Texture"].SetValue(0);
663  _shader["ProjMtx"].SetValue(Matrix4X4F.CreateOrthographicOffCenter(left, right, bottom, top, -1, 1));
664  Gl.GlBindSampler(0, 0);
665 
666  Gl.GlBindVertexArray(_vertexArrayObject);
667 
668  // Bind vertex/index buffers and setup attributes for ImDrawVert
669  Gl.GlBindBuffer(BufferTarget.ArrayBuffer, _vboHandle);
670  Gl.GlBindBuffer(BufferTarget.ElementArrayBuffer, _elementsHandle);
671 
672  Gl.EnableVertexAttribArray(_shader["Position"].Location);
673  Gl.EnableVertexAttribArray(_shader["UV"].Location);
674  Gl.EnableVertexAttribArray(_shader["Color"].Location);
675 
676  int drawVertSize = Marshal.SizeOf<ImDrawVert>();
677  Gl.VertexAttribPointer(_shader["Position"].Location, 2, VertexAttribPointerType.Float, false, drawVertSize, Marshal.OffsetOf<ImDrawVert>("Pos"));
678  Gl.VertexAttribPointer(_shader["UV"].Location, 2, VertexAttribPointerType.Float, false, drawVertSize, Marshal.OffsetOf<ImDrawVert>("Uv"));
679  Gl.VertexAttribPointer(_shader["Color"].Location, 4, VertexAttribPointerType.UnsignedByte, true, drawVertSize, Marshal.OffsetOf<ImDrawVert>("Col"));
680  }

718  public static uint LoadTexture(IntPtr pixelData, int width, int height, PixelFormat format = PixelFormat.Rgba, PixelInternalFormat internalFormat = PixelInternalFormat.Rgba)
719  {
720  uint textureId = Gl.GenTexture();
721  Gl.GlPixelStorei(PixelStoreParameter.UnpackAlignment, 1);
722  Gl.GlBindTexture(TextureTarget.Texture2D, textureId);
723  Gl.GlTexImage2D(TextureTarget.Texture2D, 0, internalFormat, width, height, 0, format, PixelType.UnsignedByte, pixelData);
