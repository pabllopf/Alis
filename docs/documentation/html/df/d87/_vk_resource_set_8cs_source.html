79  public VkResourceSet(VkGraphicsDevice gd, ref ResourceSetDescription description)
80  : base(ref description)
81  {
82  _gd = gd;
83  RefCount = new ResourceRefCount(DisposeCore);
84  VkResourceLayout vkLayout = Util.AssertSubtype<ResourceLayout, VkResourceLayout>(description.Layout);
85 
86  VkDescriptorSetLayout dsl = vkLayout.DescriptorSetLayout;
87  _descriptorCounts = vkLayout.DescriptorResourceCounts;
88  _descriptorAllocationToken = _gd.DescriptorPoolManager.Allocate(_descriptorCounts, dsl);
89 
90  BindableResource[] boundResources = description.BoundResources;
91  uint descriptorWriteCount = (uint)boundResources.Length;
92  VkWriteDescriptorSet* descriptorWrites = stackalloc VkWriteDescriptorSet[(int)descriptorWriteCount];
93  VkDescriptorBufferInfo* bufferInfos = stackalloc VkDescriptorBufferInfo[(int)descriptorWriteCount];
94  VkDescriptorImageInfo* imageInfos = stackalloc VkDescriptorImageInfo[(int)descriptorWriteCount];
95 
96  for (int i = 0; i < descriptorWriteCount; i++)
97  {
98  VkDescriptorType type = vkLayout.DescriptorTypes[i];
99 
100  descriptorWrites[i].sType = VkStructureType.WriteDescriptorSet;
101  descriptorWrites[i].descriptorCount = 1;
102  descriptorWrites[i].descriptorType = type;
103  descriptorWrites[i].dstBinding = (uint)i;
104  descriptorWrites[i].dstSet = _descriptorAllocationToken.Set;
105 
106  if (type == VkDescriptorType.UniformBuffer || type == VkDescriptorType.UniformBufferDynamic
107  || type == VkDescriptorType.StorageBuffer || type == VkDescriptorType.StorageBufferDynamic)
108  {
109  DeviceBufferRange range = Util.GetBufferRange(boundResources[i], 0);
110  VkBuffer rangedVkBuffer = Util.AssertSubtype<DeviceBuffer, VkBuffer>(range.Buffer);
111  bufferInfos[i].buffer = rangedVkBuffer.DeviceBuffer;
112  bufferInfos[i].offset = range.Offset;
113  bufferInfos[i].range = range.SizeInBytes;
114  descriptorWrites[i].pBufferInfo = &bufferInfos[i];
115  _refCounts.Add(rangedVkBuffer.RefCount);
116  }
117  else if (type == VkDescriptorType.SampledImage)
118  {
119  TextureView texView = Util.GetTextureView(_gd, boundResources[i]);
120  VkTextureView vkTexView = Util.AssertSubtype<TextureView, VkTextureView>(texView);
121  imageInfos[i].imageView = vkTexView.ImageView;
122  imageInfos[i].imageLayout = VkImageLayout.ShaderReadOnlyOptimal;
123  descriptorWrites[i].pImageInfo = &imageInfos[i];
124  _sampledTextures.Add(Util.AssertSubtype<Texture, VkTexture>(texView.Target));
125  _refCounts.Add(vkTexView.RefCount);
126  }
127  else if (type == VkDescriptorType.StorageImage)
128  {
129  TextureView texView = Util.GetTextureView(_gd, boundResources[i]);
130  VkTextureView vkTexView = Util.AssertSubtype<TextureView, VkTextureView>(texView);
131  imageInfos[i].imageView = vkTexView.ImageView;
132  imageInfos[i].imageLayout = VkImageLayout.General;
133  descriptorWrites[i].pImageInfo = &imageInfos[i];
134  _storageImages.Add(Util.AssertSubtype<Texture, VkTexture>(texView.Target));
135  _refCounts.Add(vkTexView.RefCount);
136  }
137  else if (type == VkDescriptorType.Sampler)
138  {
139  VkSampler sampler = Util.AssertSubtype<BindableResource, VkSampler>(boundResources[i]);
140  imageInfos[i].sampler = sampler.DeviceSampler;
141  descriptorWrites[i].pImageInfo = &imageInfos[i];
142  _refCounts.Add(sampler.RefCount);
143  }
144  }
145 
146  vkUpdateDescriptorSets(_gd.Device, descriptorWriteCount, descriptorWrites, 0, null);
