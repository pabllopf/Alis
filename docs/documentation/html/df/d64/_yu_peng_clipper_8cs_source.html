97  private static List<Vertices> Execute(Vertices subject, Vertices clip, PolyClipType clipType,
98  out PolyClipError error)
99  {
100  Debug.Assert(subject.IsSimple() && clip.IsSimple(), "Non simple input!",
101  "Input polygons must be simple (cannot intersect themselves).");
102 
103  // Copy polygons
104 
105  // Calculate the intersection and touch points between
106  // subject and clip and add them to both
107  CalculateIntersections(subject, clip, out Vertices slicedSubject, out Vertices slicedClip);
108 
109  // Translate polygons into upper right quadrant
110  // as the algorithm depends on it
111  Vector2 lbSubject = subject.GetAabb().LowerBound;
112  Vector2 lbClip = clip.GetAabb().LowerBound;
113  Vector2 translate = Vector2.Min(lbSubject, lbClip);
114  translate = Vector2.One - translate;
115  if (translate != Vector2.Zero)
116  {
117  slicedSubject.Translate(ref translate);
118  slicedClip.Translate(ref translate);
119  }
120 
121  // Enforce counterclockwise contours
122  slicedSubject.ForceCounterClockWise();
123  slicedClip.ForceCounterClockWise();
124 
125  // Build simplical chains from the polygons and calculate the
126  // the corresponding coefficients
127  CalculateSimplicalChain(slicedSubject, out List<float> subjectCoeff, out List<Edge> subjectSimplices);
128  CalculateSimplicalChain(slicedClip, out List<float> clipCoeff, out List<Edge> clipSimplices);
129 
130  // Determine the characteristics function for all non-original edges
131  // in subject and clip simplical chain and combine the edges contributing
132  // to the result, depending on the clipType
133  CalculateResultChain(subjectCoeff, subjectSimplices, clipCoeff, clipSimplices, clipType,
134  out List<Edge> resultSimplices);
135 
136  // Convert result chain back to polygon(s)
137  error = BuildPolygonsFromChain(resultSimplices, out List<Vertices> result);
138 
139  // Reverse the polygon translation from the beginning
140  // and remove collinear points from output
141  translate *= -1f;
142  for (int i = 0; i < result.Count; ++i)
143  {
144  result[i].Translate(ref translate);
145  SimplifyTools.CollinearSimplify(result[i]);
146  }
147 
148  return result;
149  }

156  private static void CalculateIntersections(Vertices polygon1, Vertices polygon2, out Vertices slicedPoly1,
157  out Vertices slicedPoly2)
158  {
159  slicedPoly1 = new Vertices(polygon1);
160  slicedPoly2 = new Vertices(polygon2);
161 
162  // Iterate through polygon1's edges
163  for (int i = 0; i < polygon1.Count; i++)
164  {
165  // Get edge vertices
166  Vector2 a = polygon1[i];
167  Vector2 b = polygon1[polygon1.NextIndex(i)];
168 
169  // Get intersections between this edge and polygon2
170  for (int j = 0; j < polygon2.Count; j++)
171  {
172  Vector2 c = polygon2[j];
173  Vector2 d = polygon2[polygon2.NextIndex(j)];
174 
175  // Check if the edges intersect
176  if (LineUtils.LineIntersect(a, b, c, d, out Vector2 intersectionPoint))
177  {
178  // calculate alpha values for sorting multiple intersections points on a edge
179  float alpha = GetAlpha(a, b, intersectionPoint);
180 
181  // Insert intersection point into first polygon
182  if (alpha > 0f && alpha < 1f)
183  {
184  int index = slicedPoly1.IndexOf(a) + 1;
185  while (index < slicedPoly1.Count &&
186  GetAlpha(a, b, slicedPoly1[index]) <= alpha)
187  {
188  ++index;
189  }
190 
191  slicedPoly1.Insert(index, intersectionPoint);
192  }
193 
194  // Insert intersection point into second polygon
195  alpha = GetAlpha(c, d, intersectionPoint);
196  if (alpha > 0f && alpha < 1f)
197  {
198  int index = slicedPoly2.IndexOf(c) + 1;
199  while (index < slicedPoly2.Count &&
200  GetAlpha(c, d, slicedPoly2[index]) <= alpha)
201  {
202  ++index;
203  }
204 
205  slicedPoly2.Insert(index, intersectionPoint);
206  }
207  }
208  }
209  }
210 
211  // Check for very small edges
212  for (int i = 0; i < slicedPoly1.Count; ++i)
213  {
214  int iNext = slicedPoly1.NextIndex(i);
215 
216  //If they are closer than the distance remove vertex
217  if ((slicedPoly1[iNext] - slicedPoly1[i]).LengthSquared() <= ClipperEpsilonSquared)
218  {
219  slicedPoly1.RemoveAt(i);
220  --i;
221  }
222  }
223 
224  for (int i = 0; i < slicedPoly2.Count; ++i)
225  {
226  int iNext = slicedPoly2.NextIndex(i);
227 
228  //If they are closer than the distance remove vertex
229  if ((slicedPoly2[iNext] - slicedPoly2[i]).LengthSquared() <= ClipperEpsilonSquared)
230  {
231  slicedPoly2.RemoveAt(i);
232  --i;
233  }
234  }
235  }

255  private static void CalculateResultChain(List<float> poly1Coeff, List<Edge> poly1Simplicies,
256  List<float> poly2Coeff, List<Edge> poly2Simplicies, PolyClipType clipType, out List<Edge> resultSimplices)
257  {
258  resultSimplices = new List<Edge>();
259 
260  for (int i = 0; i < poly1Simplicies.Count; ++i)
261  {
262  float edgeCharacter = 0;
263  if (poly2Simplicies.Contains(poly1Simplicies[i]))
264  {
265  edgeCharacter = 1f;
266  }
267  else if (poly2Simplicies.Contains(-poly1Simplicies[i]) && clipType == PolyClipType.Union)
268  {
269  edgeCharacter = 1f;
270  }
271  else
272  {
273  for (int j = 0; j < poly2Simplicies.Count; ++j)
274  {
275  if (!poly2Simplicies.Contains(-poly1Simplicies[i]))
276  {
277  edgeCharacter += CalculateBeta(poly1Simplicies[i].GetCenter(),
278  poly2Simplicies[j], poly2Coeff[j]);
279  }
280  }
281  }
282 
283  if (clipType == PolyClipType.Intersect)
284  {
285  if (edgeCharacter == 1f)
286  {
287  resultSimplices.Add(poly1Simplicies[i]);
288  }
289  }
290  else
291  {
292  if (edgeCharacter == 0f)
293  {
294  resultSimplices.Add(poly1Simplicies[i]);
295  }
296  }
297  }
298 
299  for (int i = 0; i < poly2Simplicies.Count; ++i)
300  {
301  float edgeCharacter = 0f;
302  if (!resultSimplices.Contains(poly2Simplicies[i]) &&
303  !resultSimplices.Contains(-poly2Simplicies[i]))
304  {
305  if (poly1Simplicies.Contains(-poly2Simplicies[i]) && clipType == PolyClipType.Union)
306  {
307  edgeCharacter = 1f;
308  }
309  else
310  {
311  edgeCharacter = 0f;
312  for (int j = 0; j < poly1Simplicies.Count; ++j)
313  {
314  if (!poly1Simplicies.Contains(poly2Simplicies[i]) &&
315  !poly1Simplicies.Contains(-poly2Simplicies[i]))
316  {
317  edgeCharacter += CalculateBeta(poly2Simplicies[i].GetCenter(),
318  poly1Simplicies[j], poly1Coeff[j]);
319  }
320  }
321 
322  if (clipType == PolyClipType.Intersect || clipType == PolyClipType.Difference)
323  {
324  if (edgeCharacter == 1f)
325  {
326  resultSimplices.Add(-poly2Simplicies[i]);
327  }
328  }
329  else
330  {
331  if (edgeCharacter == 0f)
332  {
333  resultSimplices.Add(poly2Simplicies[i]);
334  }
335  }
336  }
337  }
338  }

343  private static PolyClipError BuildPolygonsFromChain(List<Edge> simplicies, out List<Vertices> result)
344  {
345  result = new List<Vertices>();
346  PolyClipError errVal = PolyClipError.None;
347 
348  while (simplicies.Count > 0)
349  {
350  Vertices output = new Vertices
351  {
352  simplicies[0].EdgeStart,
353  simplicies[0].EdgeEnd
354  };
355  simplicies.RemoveAt(0);
356  bool closed = false;
357  int index = 0;
358  int count = simplicies.Count; // Needed to catch infinite loops
359  while (!closed && simplicies.Count > 0)
360  {
361  if (VectorEqual(output[output.Count - 1], simplicies[index].EdgeStart))
362  {
363  if (VectorEqual(simplicies[index].EdgeEnd, output[0]))
364  {
365  closed = true;
366  }
367  else
368  {
369  output.Add(simplicies[index].EdgeEnd);
370  }
371 
372  simplicies.RemoveAt(index);
373  --index;
374  }
375  else if (VectorEqual(output[output.Count - 1], simplicies[index].EdgeEnd))
376  {
377  if (VectorEqual(simplicies[index].EdgeStart, output[0]))
378  {
379  closed = true;
380  }
381  else
382  {
383  output.Add(simplicies[index].EdgeStart);
384  }
385 
386  simplicies.RemoveAt(index);
387  --index;
388  }
389 
390  if (!closed)
391  {
392  if (++index == simplicies.Count)
393  {
394  if (count == simplicies.Count)
395  {
396  result = new List<Vertices>();
397  Debug.WriteLine("Undefined error while building result polygon(s).");
398  return PolyClipError.BrokenResult;
399  }
400 
401  index = 0;
402  count = simplicies.Count;
403  }
404  }
405  }
406 
407  if (output.Count < 3)
408  {
409  errVal = PolyClipError.DegeneratedOutput;
410  Debug.WriteLine("Degenerated output polygon produced (vertices < 3).");
411  }
412 
413  result.Add(output);
414  }
415 
416  return errVal;
417  }

421  private static float CalculateBeta(Vector2 point, Edge e, float coefficient)
422  {
423  float result = 0f;
424  if (PointInSimplex(point, e))
425  {
426  result = coefficient;
427  }
428 
429  if (PointOnLineSegment(Vector2.Zero, e.EdgeStart, point) ||
430  PointOnLineSegment(Vector2.Zero, e.EdgeEnd, point))
431  {
432  result = .5f * coefficient;
433  }
434 
435  return result;
436  }
