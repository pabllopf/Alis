605  private void ActivateResourceSet(uint slot, BoundResourceSetInfo brsi, bool graphics)
606  {
607  D3D11ResourceSet d3d11RS = Util.AssertSubtype<ResourceSet, D3D11ResourceSet>(brsi.Set);
608 
609  int cbBase = GetConstantBufferBase(slot, graphics);
610  int uaBase = GetUnorderedAccessBase(slot, graphics);
611  int textureBase = GetTextureBase(slot, graphics);
612  int samplerBase = GetSamplerBase(slot, graphics);
613 
614  D3D11ResourceLayout layout = d3d11RS.Layout;
615  BindableResource[] resources = d3d11RS.Resources;
616  uint dynamicOffsetIndex = 0;
617  for (int i = 0; i < resources.Length; i++)
618  {
619  BindableResource resource = resources[i];
620  uint bufferOffset = 0;
621  if (layout.IsDynamicBuffer(i))
622  {
623  bufferOffset = brsi.Offsets.Get(dynamicOffsetIndex);
624  dynamicOffsetIndex += 1;
625  }
626  D3D11ResourceLayout.ResourceBindingInfo rbi = layout.GetDeviceSlotIndex(i);
627  switch (rbi.Kind)
628  {
629  case ResourceKind.UniformBuffer:
630  {
631  D3D11BufferRange range = GetBufferRange(resource, bufferOffset);
632  BindUniformBuffer(range, cbBase + rbi.Slot, rbi.Stages);
633  break;
634  }
635  case ResourceKind.StructuredBufferReadOnly:
636  {
637  D3D11BufferRange range = GetBufferRange(resource, bufferOffset);
638  BindStorageBufferView(range, textureBase + rbi.Slot, rbi.Stages);
639  break;
640  }
641  case ResourceKind.StructuredBufferReadWrite:
642  {
643  D3D11BufferRange range = GetBufferRange(resource, bufferOffset);
644  ID3D11UnorderedAccessView uav = range.Buffer.GetUnorderedAccessView(range.Offset, range.Size);
645  BindUnorderedAccessView(null, range.Buffer, uav, uaBase + rbi.Slot, rbi.Stages, slot);
646  break;
647  }
648  case ResourceKind.TextureReadOnly:
649  TextureView texView = Util.GetTextureView(_gd, resource);
650  D3D11TextureView d3d11TexView = Util.AssertSubtype<TextureView, D3D11TextureView>(texView);
651  UnbindUAVTexture(d3d11TexView.Target);
652  BindTextureView(d3d11TexView, textureBase + rbi.Slot, rbi.Stages, slot);
653  break;
654  case ResourceKind.TextureReadWrite:
655  TextureView rwTexView = Util.GetTextureView(_gd, resource);
656  D3D11TextureView d3d11RWTexView = Util.AssertSubtype<TextureView, D3D11TextureView>(rwTexView);
657  UnbindSRVTexture(d3d11RWTexView.Target);
658  BindUnorderedAccessView(d3d11RWTexView.Target, null, d3d11RWTexView.UnorderedAccessView, uaBase + rbi.Slot, rbi.Stages, slot);
659  break;
660  case ResourceKind.Sampler:
661  D3D11Sampler sampler = Util.AssertSubtype<BindableResource, D3D11Sampler>(resource);
662  BindSampler(sampler, samplerBase + rbi.Slot, rbi.Stages);
663  break;
664  default: throw Illegal.Value<ResourceKind>();
665  }
666  }
667  }
