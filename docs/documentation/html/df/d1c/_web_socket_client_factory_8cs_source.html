141  private async Task<WebSocket> ConnectAsync(Guid guid, Stream responseStream, string secWebSocketKey,
142  TimeSpan keepAliveInterval, string secWebSocketExtensions, bool includeExceptionInCloseResponse,
143  CancellationToken token)
144  {
145  Events.Log.ReadingHttpResponse(guid);
146  string response = string.Empty;
147 
148  try
149  {
150  response = await HttpHelper.ReadHttpHeaderAsync(responseStream, token);
151  }
152  catch (Exception ex)
153  {
154  Events.Log.ReadHttpResponseError(guid, ex.ToString());
155  throw new WebSocketHandshakeFailedException("Handshake unexpected failure", ex);
156  }
157 
158  ThrowIfInvalidResponseCode(response);
159  ThrowIfInvalidAcceptString(guid, response, secWebSocketKey);
160  string subProtocol = GetSubProtocolFromHeader(response);
161  return new WebSocketImplementation(guid, _bufferFactory, responseStream, keepAliveInterval,
162  secWebSocketExtensions, includeExceptionInCloseResponse, true, subProtocol);
163  }

170  private string GetSubProtocolFromHeader(string response)
171  {
172  // make sure we escape the accept string which could contain special regex characters
173  string regexPattern = "Sec-WebSocket-Protocol: (.*)";
174  Regex regex = new Regex(regexPattern, RegexOptions.IgnoreCase);
175  Match match = regex.Match(response);
176  if (match.Success)
177  {
178  return match.Groups[1].Value.Trim();
179  }
180 
181  return null;
182  }

191  private void ThrowIfInvalidAcceptString(Guid guid, string response, string secWebSocketKey)
192  {
193  // make sure we escape the accept string which could contain special regex characters
194  string regexPattern = "Sec-WebSocket-Accept: (.*)";
195  Regex regex = new Regex(regexPattern, RegexOptions.IgnoreCase);
196  string actualAcceptString = regex.Match(response).Groups[1].Value.Trim();
197 
198  // check the accept string
199  string expectedAcceptString = HttpHelper.ComputeSocketAcceptString(secWebSocketKey);
200  if (expectedAcceptString != actualAcceptString)
201  {
202  string warning =
203  string.Format(
204  $"Handshake failed because the accept string from the server '{expectedAcceptString}' was not the expected string '{actualAcceptString}'");
205  Events.Log.HandshakeFailure(guid, warning);
206  throw new WebSocketHandshakeFailedException(warning);
207  }
208 
209  Events.Log.ClientHandshakeSuccess(guid);
210  }

218  private void ThrowIfInvalidResponseCode(string responseHeader)
219  {
220  string responseCode = HttpHelper.ReadHttpResponseCode(responseHeader);
221  if (responseCode == null)
222  {
223  throw new InvalidHttpResponseCodeException(null, null, responseHeader);
224  }
225 
226  if (!responseCode.StartsWith("101 ", StringComparison.InvariantCultureIgnoreCase))
227  {
228  string[] lines = responseHeader.Split(new[] {"\r\n"}, StringSplitOptions.None);
229 
230  for (int i = 0; i < lines.Length; i++)
231  {
232  // if there is more to the message than just the header
233  if (string.IsNullOrWhiteSpace(lines[i]))
234  {
235  StringBuilder builder = new StringBuilder();
236  for (int j = i + 1; j < lines.Length - 1; j++)
237  {
238  builder.AppendLine(lines[j]);
239  }
240 
241  string responseDetails = builder.ToString();
242  throw new InvalidHttpResponseCodeException(responseCode, responseDetails, responseHeader);
243  }
244  }
245  }

271  protected virtual async Task<Stream> GetStream(Guid loggingGuid, bool isSecure, bool noDelay, string host,
272  int port, CancellationToken cancellationToken)
273  {
274  TcpClient tcpClient = new TcpClient();
275  tcpClient.NoDelay = noDelay;
276  IPAddress ipAddress;
277  if (IPAddress.TryParse(host, out ipAddress))
278  {
279  Events.Log.ClientConnectingToIpAddress(loggingGuid, ipAddress.ToString(), port);
280  await tcpClient.ConnectAsync(ipAddress, port);
281  }
282  else
283  {
284  Events.Log.ClientConnectingToHost(loggingGuid, host, port);
285  await tcpClient.ConnectAsync(host, port);
286  }
287 
288  cancellationToken.ThrowIfCancellationRequested();
289  Stream stream = tcpClient.GetStream();
290 
291  if (isSecure)
292  {
293  SslStream sslStream = new SslStream(stream, false, ValidateServerCertificate, null);
294  Events.Log.AttemtingToSecureSslConnection(loggingGuid);
295 
296  // This will throw an AuthenticationException if the certificate is not valid
297  TlsAuthenticateAsClient(sslStream, host);
298  Events.Log.ConnectionSecured(loggingGuid);
299  return sslStream;
300  }
301 
302  Events.Log.ConnectionNotSecure(loggingGuid);
303  return stream;
304  }

354  private async Task<WebSocket> PerformHandshake(Guid guid, Uri uri, Stream stream,
355  WebSocketClientOptions options, CancellationToken token)
356  {
357  Random rand = new Random();
358  byte[] keyAsBytes = new byte[16];
359  rand.NextBytes(keyAsBytes);
360  string secWebSocketKey = Convert.ToBase64String(keyAsBytes);
361  string additionalHeaders = GetAdditionalHeaders(options.AdditionalHttpHeaders);
362  string handshakeHttpRequest = $"GET {uri.PathAndQuery} HTTP/1.1\r\n" +
363  $"Host: {uri.Host}:{uri.Port}\r\n" +
364  "Upgrade: websocket\r\n" +
365  "Connection: Upgrade\r\n" +
366  $"Sec-WebSocket-Key: {secWebSocketKey}\r\n" +
367  $"Origin: http://{uri.Host}:{uri.Port}\r\n" +
368  $"Sec-WebSocket-Protocol: {options.SecWebSocketProtocol}\r\n" +
369  additionalHeaders +
370  "Sec-WebSocket-Version: 13\r\n\r\n";
371 
372  byte[] httpRequest = Encoding.UTF8.GetBytes(handshakeHttpRequest);
373  stream.Write(httpRequest, 0, httpRequest.Length);
374  Events.Log.HandshakeSent(guid, handshakeHttpRequest);
375  return await ConnectAsync(stream, secWebSocketKey, options, token);
376  }
