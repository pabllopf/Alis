203  EnsureType<Vector4>();
204  GlUniform4F(Location, param[0], param[1], param[2], param[3]);
205  }
206  else if (param.Length == 3)
207  {
208  EnsureType<Vector3>();
209  GlUniform3F(Location, param[0], param[1], param[2]);
210  }
211  else if (param.Length == 2)
212  {
213  EnsureType<Vector2>();
214  GlUniform2F(Location, param[0], param[1]);
215  }
216  else if (param.Length == 1)
217  {
218  EnsureType<float>();
219  GlUniform1F(Location, param[0]);
220  }
221  else
222  {
223  throw new ArgumentException("param was an unexpected length.", nameof(param));
224  }
225  }
