50  public static T CreateInterceptor<T>(T instance) where T : class
51  {
52  Type type = typeof(T);
53  TypeBuilder typeBuilder = CreateDynamicTypeBuilder(type);
54 
55  foreach (MethodInfo method in type.GetMethods(BindingFlags.Instance | BindingFlags.Public))
56  {
57  PrePostMethodAttribute attribute = method.GetCustomAttribute<PrePostMethodAttribute>();
58  if (attribute != null)
59  {
60  MethodBuilder methodBuilder = typeBuilder.DefineMethod(
61  method.Name,
62  MethodAttributes.Public | MethodAttributes.Virtual,
63  CallingConventions.Standard,
64  method.ReturnType,
65  Type.EmptyTypes
66  );
67 
68  ILGenerator ilGenerator = methodBuilder.GetILGenerator();
69  GenerateMethodBody(ilGenerator, method, attribute);
70  typeBuilder.DefineMethodOverride(methodBuilder, method);
71  }
72  }
73 
74  Type proxyType = typeBuilder.CreateType();
75  ConstructorInfo constructor = proxyType.GetConstructor(new[] { type });
76  return (T)constructor.Invoke(new object[] { instance });
77  }

84  private static TypeBuilder CreateDynamicTypeBuilder(Type type)
85  {
86  string assemblyName = type.FullName + "ProxyAssembly";
87  AssemblyBuilder assemblyBuilder = AssemblyBuilder.DefineDynamicAssembly(
88  new AssemblyName(assemblyName),
89  AssemblyBuilderAccess.Run
90  );
91 
92  ModuleBuilder moduleBuilder = assemblyBuilder.DefineDynamicModule(assemblyName);
93  string typeName = type.Name + "Proxy";
94  TypeBuilder typeBuilder = moduleBuilder.DefineType(
95  typeName,
96  TypeAttributes.Public | TypeAttributes.Class,
97  type
98  );
99 
100  ConstructorBuilder constructorBuilder = typeBuilder.DefineConstructor(
101  MethodAttributes.Public,
102  CallingConventions.HasThis,
103  new[] { type }
104  );
105 
106  ILGenerator constructorIL = constructorBuilder.GetILGenerator();
107  constructorIL.Emit(OpCodes.Ldarg_0);
108  constructorIL.Emit(OpCodes.Call, type.GetConstructor(Type.EmptyTypes));
109  constructorIL.Emit(OpCodes.Ret);
110 
111  return typeBuilder;
112  }

120  private static void GenerateMethodBody(ILGenerator ilGenerator, MethodInfo method, PrePostMethodAttribute attribute)
121  {
122  MethodInfo onEntryMethod = typeof(PrePostMethodAttribute).GetMethod("OnEntry");
123  MethodInfo onExitMethod = typeof(PrePostMethodAttribute).GetMethod("OnExit");
124 
125  ilGenerator.Emit(OpCodes.Ldarg_0);
126  ilGenerator.Emit(OpCodes.Callvirt, onEntryMethod);
127 
128  ilGenerator.Emit(OpCodes.Ldarg_0);
129  ilGenerator.Emit(OpCodes.Call, method);
130 
131  ilGenerator.Emit(OpCodes.Ldarg_0);
132  ilGenerator.Emit(OpCodes.Callvirt, onExitMethod);
133 
134  ilGenerator.Emit(OpCodes.Ret);
135  }
