110  public override void Update(float dt)
111  {
112  Vector2 f = Vector2.Zero;
113 
114  foreach (Body worldBody in World.BodyList)
115  {
116  if (!IsActiveOn(worldBody))
117  {
118  continue;
119  }
120 
121  foreach (Body controllerBody in Bodies)
122  {
123  if (worldBody == controllerBody || worldBody.IsStatic && controllerBody.IsStatic ||
124  !controllerBody.Enabled)
125  {
126  continue;
127  }
128 
129  Vector2 d = controllerBody.Position - worldBody.Position;
130  float r2 = d.LengthSquared();
131 
132  if (r2 <= MathConstants.Epsilon || r2 > MaxRadius * MaxRadius || r2 < MinRadius * MinRadius)
133  {
134  continue;
135  }
136 
137  switch (GravityType)
138  {
139  case GravityType.DistanceSquared:
140  f = Strength / r2 * worldBody.Mass * controllerBody.Mass * d;
141  break;
142  case GravityType.Linear:
143  f = Strength / (float) Math.Sqrt(r2) * worldBody.Mass * controllerBody.Mass * d;
144  break;
145  }
146 
147  worldBody.ApplyForce(ref f);
148  }
149 
150  foreach (Vector2 point in Points)
151  {
152  Vector2 d = point - worldBody.Position;
153  float r2 = d.LengthSquared();
154 
155  if (r2 <= MathConstants.Epsilon || r2 > MaxRadius * MaxRadius || r2 < MinRadius * MinRadius)
156  {
157  continue;
158  }
159 
160  switch (GravityType)
161  {
162  case GravityType.DistanceSquared:
163  f = Strength / r2 * worldBody.Mass * d;
164  break;
165  case GravityType.Linear:
166  f = Strength / (float) Math.Sqrt(r2) * worldBody.Mass * d;
167  break;
168  }
169 
170  worldBody.ApplyForce(ref f);
171  }
172  }
