181  public string DisplayedString
182  {
183  get
184  {
185  // Get a pointer to the source string (UTF-32)
186  IntPtr source = sfText_getUnicodeString(CPointer);
187 
188  // Find its length (find the terminating 0)
189  uint length = 0;
190  unsafe
191  {
192  for (uint* ptr = (uint*) source.ToPointer(); *ptr != 0; ++ptr)
193  {
194  length++;
195  }
196  }
197 
198  // Copy it to a byte array
199  byte[] sourceBytes = new byte[length * 4];
200  Marshal.Copy(source, sourceBytes, 0, sourceBytes.Length);
201 
202  // Convert it to a C# string
203  return Encoding.UTF32.GetString(sourceBytes);
204  }
205 
206  set
207  {
208  // Copy the string to a null-terminated UTF-32 byte array
209  byte[] utf32 = Encoding.UTF32.GetBytes(value + '\0');
210 
211  // Pass it to the C API
212  unsafe
213  {
214  fixed (byte* ptr = utf32)
215  {
216  sfText_setUnicodeString(CPointer, (IntPtr) ptr);
217  }
218  }
219  }
220  }
