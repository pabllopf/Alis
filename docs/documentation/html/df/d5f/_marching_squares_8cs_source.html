91  public static List<Vertices> DetectSquares(Aabb domain, float cellWidth, float cellHeight, sbyte[,] f,
92  int lerpCount, bool combine)
93  {
94  CxFastList<GeomPoly> ret = new CxFastList<GeomPoly>();
95

97

102  int xn = (int) (domain.Extents.X * 2 / cellWidth);
103  bool xp = xn == domain.Extents.X * 2 / cellWidth;
104  int yn = (int) (domain.Extents.Y * 2 / cellHeight);
105  bool yp = yn == domain.Extents.Y * 2 / cellHeight;
106  if (!xp)
107  {
108  xn++;
109  }
110 
111  if (!yp)

113  yn++;
114  }

116  sbyte[,] fs = new sbyte[xn + 1, yn + 1];
117  GeomPolyVal[,] ps = new GeomPolyVal[xn + 1, yn + 1];
118 
119  //populate shared function lookups.
120  for (int x = 0; x < xn + 1; x++)
121  {
122  int x0;
123  if (x == xn)
124  {
125  x0 = (int) domain.UpperBound.X;
126  }
127  else
128  {
129  x0 = (int) (x * cellWidth + domain.LowerBound.X);
130  }
131 
132  for (int y = 0; y < yn + 1; y++)
133  {
134  int y0;
135  if (y == yn)
136  {
137  y0 = (int) domain.UpperBound.Y;
138  }
139  else
140  {
141  y0 = (int) (y * cellHeight + domain.LowerBound.Y);
142  }
143 
144  fs[x, y] = f[x0, y0];
145  }
146  }
147 
148  //generate sub-polys and combine to scan lines
149  for (int y = 0; y < yn; y++)
150  {
151  float y0 = y * cellHeight + domain.LowerBound.Y;
152  float y1;
153  if (y == yn - 1)
154  {
155  y1 = domain.UpperBound.Y;
156  }
157  else
158  {
159  y1 = y0 + cellHeight;
160  }
161 
162  GeomPoly pre = null;
163  for (int x = 0; x < xn; x++)
164  {
165  float x0 = x * cellWidth + domain.LowerBound.X;
166  float x1;
167  if (x == xn - 1)
168  {
169  x1 = domain.UpperBound.X;
170  }
171  else
172  {
173  x1 = x0 + cellWidth;
174  }
175 
176  gp = new GeomPoly();
177 
178  int key = MarchSquare(f, fs, ref gp, x, y, x0, y0, x1, y1, lerpCount);
179  if (gp.Length != 0)
180  {
181  if (combine && (pre != null) && ((key & 9) != 0))
182  {
183  CombLeft(ref pre, ref gp);
184  gp = pre;
185  }
186  else
187  {
188  ret.Add(gp);
189  }
190 
191  ps[x, y] = new GeomPolyVal(gp, key);
192  }
193  else
194  {
195  gp = null;
196  }
197 
198  pre = gp;
199  }
200  }
201 
202  if (!combine)
203  {
204  polyList = ret.GetListOfElements();
205 
206  foreach (GeomPoly poly in polyList)
207  {
208  verticesList.Add(new Vertices(poly.Points.GetListOfElements()));
209  }
210 
211  return verticesList;
212  }

214  //combine scan lines together
215  for (int y = 1; y < yn; y++)
216  {
217  int x = 0;
218  while (x < xn)
219  {
220  GeomPolyVal p = ps[x, y];
221 
222  //skip along scan line if no polygon exists at this point
223  if (p == null)
224  {
225  x++;
226  continue;
227  }
228 
229  //skip along if current polygon cannot be combined above.
230  if ((p.Key & 12) == 0)
231  {
232  x++;
233  continue;
234  }
235 
236  //skip along if no polygon exists above.
237  GeomPolyVal u = ps[x, y - 1];
238  if (u == null)
239  {
240  x++;
241  continue;
242  }
243 
244  //skip along if polygon above cannot be combined with.
245  if ((u.Key & 3) == 0)
246  {
247  x++;
248  continue;
249  }
250

252  float ay = y * cellHeight + domain.LowerBound.Y;
253 
254  CxFastList<Vector2F> bp = p.GeomP.Points;
255  CxFastList<Vector2F> ap = u.GeomP.Points;
256 
257  //skip if it's already been combined with above polygon
258  if (u.GeomP == p.GeomP)
259  {
260  x++;
261  continue;
262  }
263 
264  //combine above (but disallow the hole thingies
265  CxFastListNode<Vector2F> bi = bp.Begin();
266  while (Square(bi.GetElem().Y - ay) > Constant.Epsilon || bi.GetElem().X < ax)
267  {
268  bi = bi.GetNext();
269  }
270 
271  //NOTE: Unused
272  //Vector2 b0 = bi.elem();
273  Vector2F b1 = bi.GetNext().GetElem();
274  if (Square(b1.Y - ay) > Constant.Epsilon)
275  {
276  x++;
277  continue;
278  }
279 
280  bool brk = true;
281  CxFastListNode<Vector2F> ai = ap.Begin();
282  while (ai != ap.End())
283  {
284  if (VecDsq(ai.GetElem(), b1) < Constant.Epsilon)
285  {
286  brk = false;
287  break;
288  }
289 
290  ai = ai.GetNext();
291  }
292 
293  if (brk)
294  {
295  x++;
296  continue;
297  }
298 
299  CxFastListNode<Vector2F> bj = bi.GetNext().GetNext();
300  if (bj == bp.End())
301  {
302  bj = bp.Begin();
303  }
304 
305  while (bj != bi)
306  {
307  ai = ap.Insert(ai, bj.GetElem()); // .clone()
308  bj = bj.GetNext();
309  if (bj == bp.End())
310  {
311  bj = bp.Begin();
312  }
313 
314  u.GeomP.Length++;
315  }
316 
317  //u.p.simplify(float.Epsilon,float.Epsilon);
318  //
319  ax = x + 1;
320  while (ax < xn)
321  {
322  GeomPolyVal p2 = ps[(int) ax, y];
323  if (p2 == null || p2.GeomP != p.GeomP)
324  {
325  ax++;
326  continue;
327  }
328 
329  p2.GeomP = u.GeomP;
330  ax++;
331  }
332 
333  ax = x - 1;
334  while (ax >= 0)
335  {
336  GeomPolyVal p2 = ps[(int) ax, y];
337  if (p2 == null || p2.GeomP != p.GeomP)
338  {
339  ax--;
340  continue;
341  }
342 
343  p2.GeomP = u.GeomP;
344  ax--;
345  }
346 
347  ret.Remove(p.GeomP);
348  p.GeomP = u.GeomP;
349 
350  x = (int) ((bi.GetNext().GetElem().X - domain.LowerBound.X) / cellWidth) + 1;
351 
352  //x++; this was already commented out!
353  }
354  }
355 
356  polyList = ret.GetListOfElements();
357 
358  foreach (GeomPoly poly in polyList)
359  {
360  verticesList.Add(new Vertices(poly.Points.GetListOfElements()));
361  }
362 
363  return verticesList;

374  private static float Lerp(float x0, float x1, float v0, float v1)
375  {
376  float dv = v0 - v1;
377  float t;
378  if (dv * dv < Constant.Epsilon)
379  {
380  t = 0.5f;
381  }
382  else
383  {
384  t = v0 / dv;
385  }
386 
387  return x0 + t * (x1 - x0);
388  }

391  private static float Xlerp(float x0, float x1, float y, float v0, float v1, sbyte[,] f, int c)
392  {
393  float xm = Lerp(x0, x1, v0, v1);
394  if (c == 0)
395  {
396  return xm;
397  }
398 
399  sbyte vm = f[(int) xm, (int) y];
400 
401  if (v0 * vm < 0)
402  {
403  return Xlerp(x0, xm, y, v0, vm, f, c - 1);
404  }
405 
406  return Xlerp(xm, x1, y, vm, v1, f, c - 1);
407  }

410  private static float Ylerp(float y0, float y1, float x, float v0, float v1, sbyte[,] f, int c)

459  private static int MarchSquare(sbyte[,] f, sbyte[,] fs, ref GeomPoly poly, int ax, int ay, float x0, float y0,
460  float x1, float y1, int bin)
461  {
462  //key lookup
463  int key = 0;
464  sbyte v0 = fs[ax, ay];
465  if (v0 < 0)
466  {
467  key |= 8;
468  }
469 
470  sbyte v1 = fs[ax + 1, ay];
471  if (v1 < 0)
472  {
473  key |= 4;
474  }
475 
476  sbyte v2 = fs[ax + 1, ay + 1];
477  if (v2 < 0)
478  {
479  key |= 2;
480  }
481 
482  sbyte v3 = fs[ax, ay + 1];
483  if (v3 < 0)
484  {
485  key |= 1;
486  }
487 
488  int val = LookMarch[key];
489  if (val != 0)
490  {
491  CxFastListNode<Vector2F> pi = null;
492  for (int i = 0; i < 8; i++)
493  {
494  Vector2F p;
495  if ((val & (1 << i)) != 0)
496  {
497  if ((i == 7) && ((val & 1) == 0))
498  {
499  poly.Points.Add(p = new Vector2F(x0, Ylerp(y0, y1, x0, v0, v3, f, bin)));
500  }
501  else
502  {
503  if (i == 0)
504  {
505  p = new Vector2F(x0, y0);
506  }
507  else if (i == 2)
508  {
509  p = new Vector2F(x1, y0);
510  }
511  else if (i == 4)
512  {
513  p = new Vector2F(x1, y1);
514  }
515  else if (i == 6)
516  {
517  p = new Vector2F(x0, y1);
518  }
519 
520  else if (i == 1)
521  {
522  p = new Vector2F(Xlerp(x0, x1, y0, v0, v1, f, bin), y0);
523  }
524  else if (i == 5)
525  {
526  p = new Vector2F(Xlerp(x0, x1, y1, v3, v2, f, bin), y1);
527  }
528 
529  else if (i == 3)
530  {
531  p = new Vector2F(x1, Ylerp(y0, y1, x1, v1, v2, f, bin));
532  }
533  else
534  {
535  p = new Vector2F(x0, Ylerp(y0, y1, x0, v0, v3, f, bin));
536  }
537 
538  pi = poly.Points.Insert(pi, p);
539  }
540 
541  poly.Length++;
542  }
543  }
544 
545  //poly.simplify(float.Epsilon,float.Epsilon);
546  }
547 
548  return key;
549  }

555  private static void CombLeft(ref GeomPoly polya, ref GeomPoly polyb)
556  {
557  CxFastList<Vector2F> ap = polya.Points;
558  CxFastList<Vector2F> bp = polyb.Points;
559  CxFastListNode<Vector2F> ai = ap.Begin();
560  CxFastListNode<Vector2F> bi = bp.Begin();
561 
562  Vector2F b = bi.GetElem();
563  CxFastListNode<Vector2F> prea = null;
564  while (ai != ap.End())
565  {
566  Vector2F a = ai.GetElem();
567  if (VecDsq(a, b) < Constant.Epsilon)
568  {
569  //ignore shared vertex if parallel
570  if (prea != null)
571  {
572  Vector2F a0 = prea.GetElem();
573  b = bi.GetNext().GetElem();
574 
575  Vector2F u = a - a0;
576 
577  //vec_new(u); vec_sub(a.p.p, a0.p.p, u);
578  Vector2F v = b - a;
579 
580  //vec_new(v); vec_sub(b.p.p, a.p.p, v);
581  float dot = VecCross(u, v);
582  if (dot * dot < Constant.Epsilon)
583  {
584  ap.Erase(prea, ai);
585  polya.Length--;
586  ai = prea;
587  }
588  }
589 
590  //insert polyb into polya
591  bool fst = true;
592  CxFastListNode<Vector2F> preb = null;
593  while (!bp.Empty())
594  {
595  Vector2F bb = bp.Front();
596  bp.Pop();
597  if (!fst && !bp.Empty())
598  {
599  ai = ap.Insert(ai, bb);
600  polya.Length++;
601  preb = ai;
602  }
603 
604  fst = false;
605  }
606 
607  //ignore shared vertex if parallel
608  ai = ai.GetNext();
609  Vector2F a1 = ai.GetElem();
610  ai = ai.GetNext();
611  if (ai == ap.End())
612  {
613  ai = ap.Begin();
614  }
615 
616  Vector2F a2 = ai.GetElem();
617  Vector2F a00 = preb.GetElem();
618  Vector2F uu = a1 - a00;
619 
620  //vec_new(u); vec_sub(a1.p, a0.p, u);
621  Vector2F vv = a2 - a1;
622 
623  //vec_new(v); vec_sub(a2.p, a1.p, v);
624  float dot1 = VecCross(uu, vv);
625  if (dot1 * dot1 < Constant.Epsilon)
626  {
627  ap.Erase(preb, preb.GetNext());
628  polya.Length--;
629  }
630 
631  return;
632  }
633 
634  prea = ai;
635  ai = ai.GetNext();
636  }

640  internal class CxFastList<T>

658  public CxFastListNode<T> Add(T value)
659  {
660  CxFastListNode<T> newNode = new CxFastListNode<T>(value);
661  if (head == null)
662  {
663  newNode.Next = null;
664  head = newNode;
665  return newNode;
666  }
667 
668  newNode.Next = head;
669  head = newNode;
670 
671  return newNode;

675  public bool Remove(T value)
676  {
677  CxFastListNode<T> head = this.head;
678  CxFastListNode<T> prev = this.head;
679 
680  EqualityComparer<T> comparer = EqualityComparer<T>.Default;
681 
682  if (head != null)
683  {
684  if (value != null)
685  {
686  do
687  {
688  // if we are on the value to be removed
689  if (comparer.Equals(head.Elt, value))
690  {
691  // then we need to patch the list
692  // check to see if we are removing the _head
693  if (head == this.head)
694  {
695  this.head = head.Next;
696  return true;
697  }
698 
699  // were not at the head
700  prev.Next = head.Next;
701  return true;
702  }
703 
704  // cache the current as the previous for the next go around
705  prev = head;
706  head = head.Next;
707  } while (head != null);
708  }
709  }
710 
711  return false;
712  }
713 
722  public CxFastListNode<T> Pop() => Erase(null, head);
723 
725  public CxFastListNode<T> Insert(CxFastListNode<T> node, T value)
726  {
727  if (node == null)
728  {
729  return Add(value);
730  }
731 
732  CxFastListNode<T> newNode = new CxFastListNode<T>(value);
733  CxFastListNode<T> nextNode = node.Next;
734  newNode.Next = nextNode;
735  node.Next = newNode;
736 
737  return newNode;
738  }

744  public CxFastListNode<T> Erase(CxFastListNode<T> prev, CxFastListNode<T> node)
745  {
746  // cache the node after the node to be removed
747  CxFastListNode<T> nextNode = node.Next;
748  if (prev != null)
749  {
750  prev.Next = nextNode;
751  }
752  else if (head != null)
753  {
754  head = head.Next;
755  }
756  else
757  {
758  return null;
759  }
760 
761  return nextNode;
762  }

812  public CxFastListNode<T> Find(T value)
813  {
814  // start at head
815  CxFastListNode<T> head = this.head;
816  EqualityComparer<T> comparer = EqualityComparer<T>.Default;
817  if (head != null)
818  {
819  if (value != null)
820  {
821  do
822  {
823  if (comparer.Equals(head.Elt, value))
824  {
825  return head;
826  }
827 
828  head = head.Next;
829  } while (head != this.head);
830  }
831  else
832  {
833  do
834  {
835  if (head.Elt == null)
836  {
837  return head;
838  }
839 
840  head = head.Next;
841  } while (head != this.head);
842  }
843  }
844 
845  return null;
846  }
