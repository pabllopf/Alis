176  case PixelFormat.BC3_UNorm_SRgb:
177  case PixelFormat.BC4_UNorm:
178  case PixelFormat.BC4_SNorm:

180  case PixelFormat.BC5_SNorm:

191  || metalFeatures.IsSupported(MTLFeatureSet.iOS_GPUFamily2_v1)
192  || metalFeatures.IsSupported(MTLFeatureSet.iOS_GPUFamily3_v1)
193  || metalFeatures.IsSupported(MTLFeatureSet.iOS_GPUFamily4_v1);

197  || metalFeatures.IsSupported(MTLFeatureSet.macOS_GPUFamily1_v2)
