328  {
329 #if VALIDATE_USAGE

331  {

334  {
335  throw new VeldridException("Buffers must have the Staging or Dynamic usage flag to be mapped.");
336  }

338  {

340  }

342  {

344  $"{nameof(MapMode)}.{nameof(MapMode.Read)} and {nameof(MapMode)}.{nameof(MapMode.ReadWrite)} can only be used on buffers created with {nameof(BufferUsage)}.{nameof(BufferUsage.Staging)}.");
345  }

348  {

351  throw new VeldridException("Texture must have the Staging usage flag to be mapped.");
352  }

354  {
355  throw new VeldridException(

584  [Conditional("VALIDATE_USAGE")]
585  private static void ValidateUpdateTextureParameters(
586  Texture texture,
587  uint sizeInBytes,
588  uint x, uint y, uint z,
589  uint width, uint height, uint depth,
590  uint mipLevel, uint arrayLayer)
591  {
592  if (FormatHelpers.IsCompressedFormat(texture.Format))
593  {
594  if (x % 4 != 0 || y % 4 != 0 || height % 4 != 0 || width % 4 != 0)
595  {
596  Util.GetMipDimensions(texture, mipLevel, out uint mipWidth, out uint mipHeight, out _);
597  if (width != mipWidth && height != mipHeight)
598  {
599  throw new VeldridException($"Updates to block-compressed textures must use a region that is block-size aligned and sized.");
600  }
601  }
602  }
603  uint expectedSize = FormatHelpers.GetRegionSize(width, height, depth, texture.Format);
604  if (sizeInBytes < expectedSize)
605  {
606  throw new VeldridException(
607  $"The data size is less than expected for the given update region. At least {expectedSize} bytes must be provided, but only {sizeInBytes} were.");
608  }
609 
610  // Compressed textures don't necessarily need to have a Texture.Width and Texture.Height that are a multiple of 4.
611  // But the mipdata width and height *does* need to be a multiple of 4.
612  uint roundedTextureWidth, roundedTextureHeight;
613  if (FormatHelpers.IsCompressedFormat(texture.Format))
614  {
615  roundedTextureWidth = (texture.Width + 3) / 4 * 4;
616  roundedTextureHeight = (texture.Height + 3) / 4 * 4;
617  }
618  else
619  {
620  roundedTextureWidth = texture.Width;
621  roundedTextureHeight = texture.Height;
622  }
623 
624  if (x + width > roundedTextureWidth || y + height > roundedTextureHeight || z + depth > texture.Depth)
625  {
626  throw new VeldridException($"The given region does not fit into the Texture.");
627  }
628 
629  if (mipLevel >= texture.MipLevels)
630  {
631  throw new VeldridException(
632  $"{nameof(mipLevel)} ({mipLevel}) must be less than the Texture's mip level count ({texture.MipLevels}).");
633  }
634 
635  uint effectiveArrayLayers = texture.ArrayLayers;
636  if ((texture.Usage & TextureUsage.Cubemap) != 0)
637  {
638  effectiveArrayLayers *= 6;
639  }
640  if (arrayLayer >= effectiveArrayLayers)
641  {
642  throw new VeldridException(
643  $"{nameof(arrayLayer)} ({arrayLayer}) must be less than the Texture's effective array layer count ({effectiveArrayLayers}).");
644  }
645  }
