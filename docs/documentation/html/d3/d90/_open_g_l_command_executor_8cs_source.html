419  private void FlushVertexLayouts()
420  {
421  uint totalSlotsBound = 0;
422  VertexLayoutDescription[] layouts = _graphicsPipeline.VertexLayouts;
423  for (int i = 0; i < layouts.Length; i++)
424  {
425  VertexLayoutDescription input = layouts[i];
426  OpenGLBuffer vb = _vertexBuffers[i];
427  glBindBuffer(BufferTarget.ArrayBuffer, vb.Buffer);
428  uint offset = 0;
429  uint vbOffset = _vbOffsets[i];
430  for (uint slot = 0; slot < input.Elements.Length; slot++)
431  {
432  ref VertexElementDescription element = ref input.Elements[slot]; // Large structure -- use by reference.
433  uint actualSlot = totalSlotsBound + slot;
434  if (actualSlot >= _vertexAttributesBound)
435  {
436  glEnableVertexAttribArray(actualSlot);
437  }
438  VertexAttribPointerType type = OpenGLFormats.VdToGLVertexAttribPointerType(
439  element.Format,
440  out bool normalized,
441  out bool isInteger);
442 
443  uint actualOffset = element.Offset != 0 ? element.Offset : offset;
444  actualOffset += vbOffset;
445 
446  if (isInteger && !normalized)
447  {
448  glVertexAttribIPointer(
449  actualSlot,
450  FormatHelpers.GetElementCount(element.Format),
451  type,
452  (uint)_graphicsPipeline.VertexStrides[i],
453  (void*)actualOffset);
454  CheckLastError();
455  }
456  else
457  {
458  glVertexAttribPointer(
459  actualSlot,
460  FormatHelpers.GetElementCount(element.Format),
461  type,
462  normalized,
463  (uint)_graphicsPipeline.VertexStrides[i],
464  (void*)actualOffset);
465  CheckLastError();
466  }
467 
468  uint stepRate = input.InstanceStepRate;
469  if (_vertexAttribDivisors[actualSlot] != stepRate)
470  {
471  glVertexAttribDivisor(actualSlot, stepRate);
472  _vertexAttribDivisors[actualSlot] = stepRate;
473  }
474 
475  offset += FormatSizeHelpers.GetSizeInBytes(element.Format);
476  }
477 
478  totalSlotsBound += (uint)input.Elements.Length;
479  }
480 
481  for (uint extraSlot = totalSlotsBound; extraSlot < _vertexAttributesBound; extraSlot++)
482  {
483  glDisableVertexAttribArray(extraSlot);
484  }
485 
486  _vertexAttributesBound = totalSlotsBound;
487  }

651  private void ActivateGraphicsPipeline()
652  {
653  _graphicsPipelineActive = true;
654  _graphicsPipeline.EnsureResourcesCreated();
655 
656  Util.EnsureArrayMinimumSize(ref _graphicsResourceSets, (uint)_graphicsPipeline.ResourceLayouts.Length);
657  Util.EnsureArrayMinimumSize(ref _newGraphicsResourceSets, (uint)_graphicsPipeline.ResourceLayouts.Length);
658 
659  // Force ResourceSets to be re-bound.
660  for (int i = 0; i < _graphicsPipeline.ResourceLayouts.Length; i++)
661  {
662  _newGraphicsResourceSets[i] = true;
663  }
664 
665  // Blend State
666 
667  BlendStateDescription blendState = _graphicsPipeline.BlendState;
668  glBlendColor(blendState.BlendFactor.R, blendState.BlendFactor.G, blendState.BlendFactor.B, blendState.BlendFactor.A);
669  CheckLastError();
670 
671  if (blendState.AlphaToCoverageEnabled)
672  {
673  glEnable(EnableCap.SampleAlphaToCoverage);
674  CheckLastError();
675  }
676  else
677  {
678  glDisable(EnableCap.SampleAlphaToCoverage);
679  CheckLastError();
680  }
681 
682  if (_features.IndependentBlend)
683  {
684  for (uint i = 0; i < blendState.AttachmentStates.Length; i++)
685  {
686  BlendAttachmentDescription attachment = blendState.AttachmentStates[i];
687  ColorWriteMask colorMask = attachment.ColorWriteMask.GetOrDefault();
688 
689  glColorMaski(
690  i,
691  (colorMask & ColorWriteMask.Red) == ColorWriteMask.Red,
692  (colorMask & ColorWriteMask.Green) == ColorWriteMask.Green,
693  (colorMask & ColorWriteMask.Blue) == ColorWriteMask.Blue,
694  (colorMask & ColorWriteMask.Alpha) == ColorWriteMask.Alpha);
695  CheckLastError();
696 
697  if (!attachment.BlendEnabled)
698  {
699  glDisablei(EnableCap.Blend, i);
700  CheckLastError();
701  }
702  else
703  {
704  glEnablei(EnableCap.Blend, i);
705  CheckLastError();
706 
707  glBlendFuncSeparatei(
708  i,
709  OpenGLFormats.VdToGLBlendFactorSrc(attachment.SourceColorFactor),
710  OpenGLFormats.VdToGLBlendFactorDest(attachment.DestinationColorFactor),
711  OpenGLFormats.VdToGLBlendFactorSrc(attachment.SourceAlphaFactor),
712  OpenGLFormats.VdToGLBlendFactorDest(attachment.DestinationAlphaFactor));
713  CheckLastError();
714 
715  glBlendEquationSeparatei(
716  i,
717  OpenGLFormats.VdToGLBlendEquationMode(attachment.ColorFunction),
718  OpenGLFormats.VdToGLBlendEquationMode(attachment.AlphaFunction));
719  CheckLastError();
720  }
721  }
722  }
723  else if (blendState.AttachmentStates.Length > 0)
724  {
725  BlendAttachmentDescription attachment = blendState.AttachmentStates[0];
726  ColorWriteMask colorMask = attachment.ColorWriteMask.GetOrDefault();
727 
728  glColorMask(
729  (colorMask & ColorWriteMask.Red) == ColorWriteMask.Red,
730  (colorMask & ColorWriteMask.Green) == ColorWriteMask.Green,
731  (colorMask & ColorWriteMask.Blue) == ColorWriteMask.Blue,
732  (colorMask & ColorWriteMask.Alpha) == ColorWriteMask.Alpha);
733  CheckLastError();
734 
735  if (!attachment.BlendEnabled)
736  {
737  glDisable(EnableCap.Blend);
738  CheckLastError();
739  }
740  else
741  {
742  glEnable(EnableCap.Blend);
743  CheckLastError();
744 
745  glBlendFuncSeparate(
746  OpenGLFormats.VdToGLBlendFactorSrc(attachment.SourceColorFactor),
747  OpenGLFormats.VdToGLBlendFactorDest(attachment.DestinationColorFactor),
748  OpenGLFormats.VdToGLBlendFactorSrc(attachment.SourceAlphaFactor),
749  OpenGLFormats.VdToGLBlendFactorDest(attachment.DestinationAlphaFactor));
750  CheckLastError();
751 
752  glBlendEquationSeparate(
753  OpenGLFormats.VdToGLBlendEquationMode(attachment.ColorFunction),
754  OpenGLFormats.VdToGLBlendEquationMode(attachment.AlphaFunction));
755  CheckLastError();
756  }
757  }
758 
759  // Depth Stencil State
760 
761  DepthStencilStateDescription dss = _graphicsPipeline.DepthStencilState;
762  if (!dss.DepthTestEnabled)
763  {
764  glDisable(EnableCap.DepthTest);
765  CheckLastError();
766  }
767  else
768  {
769  glEnable(EnableCap.DepthTest);
770  CheckLastError();
771 
772  glDepthFunc(OpenGLFormats.VdToGLDepthFunction(dss.DepthComparison));
773  CheckLastError();
774  }
775 
776  glDepthMask(dss.DepthWriteEnabled);
777  CheckLastError();
778 
779  if (dss.StencilTestEnabled)
780  {
781  glEnable(EnableCap.StencilTest);
782  CheckLastError();
783 
784  glStencilFuncSeparate(
785  CullFaceMode.Front,
786  OpenGLFormats.VdToGLStencilFunction(dss.StencilFront.Comparison),
787  (int)dss.StencilReference,
788  dss.StencilReadMask);
789  CheckLastError();
790 
791  glStencilOpSeparate(
792  CullFaceMode.Front,
793  OpenGLFormats.VdToGLStencilOp(dss.StencilFront.Fail),
794  OpenGLFormats.VdToGLStencilOp(dss.StencilFront.DepthFail),
795  OpenGLFormats.VdToGLStencilOp(dss.StencilFront.Pass));
796  CheckLastError();
797 
798  glStencilFuncSeparate(
799  CullFaceMode.Back,
800  OpenGLFormats.VdToGLStencilFunction(dss.StencilBack.Comparison),
801  (int)dss.StencilReference,
802  dss.StencilReadMask);
803  CheckLastError();
804 
805  glStencilOpSeparate(
806  CullFaceMode.Back,
807  OpenGLFormats.VdToGLStencilOp(dss.StencilBack.Fail),
808  OpenGLFormats.VdToGLStencilOp(dss.StencilBack.DepthFail),
809  OpenGLFormats.VdToGLStencilOp(dss.StencilBack.Pass));
810  CheckLastError();
811 
812  glStencilMask(dss.StencilWriteMask);
813  CheckLastError();
814  }
815  else
816  {
817  glDisable(EnableCap.StencilTest);
818  CheckLastError();
819  }

821  // Rasterizer State
822 
823  RasterizerStateDescription rs = _graphicsPipeline.RasterizerState;
824  if (rs.CullMode == FaceCullMode.None)
825  {
826  glDisable(EnableCap.CullFace);
827  CheckLastError();
828  }
829  else
830  {
831  glEnable(EnableCap.CullFace);
832  CheckLastError();
833 
834  glCullFace(OpenGLFormats.VdToGLCullFaceMode(rs.CullMode));
835  CheckLastError();
836  }
837 
838  if (_backend == GraphicsBackend.OpenGL)
839  {
840  glPolygonMode(MaterialFace.FrontAndBack, OpenGLFormats.VdToGLPolygonMode(rs.FillMode));
841  CheckLastError();
842  }
843 
844  if (!rs.ScissorTestEnabled)
845  {
846  glDisable(EnableCap.ScissorTest);
847  CheckLastError();
848  }
849  else
850  {
851  glEnable(EnableCap.ScissorTest);
852  CheckLastError();
853  }
854 
855  if (_backend == GraphicsBackend.OpenGL)
856  {
857  if (!rs.DepthClipEnabled)
858  {
859  glEnable(EnableCap.DepthClamp);
860  CheckLastError();
861  }
862  else
863  {
864  glDisable(EnableCap.DepthClamp);
865  CheckLastError();
866  }
867  }
868 
869  glFrontFace(OpenGLFormats.VdToGLFrontFaceDirection(rs.FrontFace));
870  CheckLastError();
871 
872  // Primitive Topology
873  _primitiveType = OpenGLFormats.VdToGLPrimitiveType(_graphicsPipeline.PrimitiveTopology);
874 
875  // Shader Set
876  glUseProgram(_graphicsPipeline.Program);
877  CheckLastError();
878 
879  int vertexStridesCount = _graphicsPipeline.VertexStrides.Length;
880  Util.EnsureArrayMinimumSize(ref _vertexBuffers, (uint)vertexStridesCount);
881  Util.EnsureArrayMinimumSize(ref _vbOffsets, (uint)vertexStridesCount);
882 
883  uint totalVertexElements = 0;
884  for (int i = 0; i < _graphicsPipeline.VertexLayouts.Length; i++)
885  {
886  totalVertexElements += (uint)_graphicsPipeline.VertexLayouts[i].Elements.Length;
887  }
888  Util.EnsureArrayMinimumSize(ref _vertexAttribDivisors, totalVertexElements);
889  }
