418  private void FlushVertexLayouts()
419  {
420  uint totalSlotsBound = 0;
421  VertexLayoutDescription[] layouts = _graphicsPipeline.VertexLayouts;
422  for (int i = 0; i < layouts.Length; i++)
423  {
424  VertexLayoutDescription input = layouts[i];
425  OpenGLBuffer vb = _vertexBuffers[i];
426  glBindBuffer(BufferTarget.ArrayBuffer, vb.Buffer);
427  uint offset = 0;
428  uint vbOffset = _vbOffsets[i];
429  for (uint slot = 0; slot < input.Elements.Length; slot++)
430  {
431  ref VertexElementDescription element = ref input.Elements[slot]; // Large structure -- use by reference.
432  uint actualSlot = totalSlotsBound + slot;
433  if (actualSlot >= _vertexAttributesBound)
434  {
435  glEnableVertexAttribArray(actualSlot);
436  }
437  VertexAttribPointerType type = OpenGLFormats.VdToGLVertexAttribPointerType(
438  element.Format,
439  out bool normalized,
440  out bool isInteger);
441 
442  uint actualOffset = element.Offset != 0 ? element.Offset : offset;
443  actualOffset += vbOffset;
444 
445  if (isInteger && !normalized)
446  {
447  glVertexAttribIPointer(
448  actualSlot,
449  FormatHelpers.GetElementCount(element.Format),
450  type,
451  (uint)_graphicsPipeline.VertexStrides[i],
452  (void*)actualOffset);
453  CheckLastError();
454  }
455  else
456  {
457  glVertexAttribPointer(
458  actualSlot,
459  FormatHelpers.GetElementCount(element.Format),
460  type,
461  normalized,
462  (uint)_graphicsPipeline.VertexStrides[i],
463  (void*)actualOffset);
464  CheckLastError();
465  }
466 
467  uint stepRate = input.InstanceStepRate;
468  if (_vertexAttribDivisors[actualSlot] != stepRate)
469  {
470  glVertexAttribDivisor(actualSlot, stepRate);
471  _vertexAttribDivisors[actualSlot] = stepRate;
472  }
473 
474  offset += FormatSizeHelpers.GetSizeInBytes(element.Format);
475  }
476 
477  totalSlotsBound += (uint)input.Elements.Length;
478  }
479 
480  for (uint extraSlot = totalSlotsBound; extraSlot < _vertexAttributesBound; extraSlot++)
481  {
482  glDisableVertexAttribArray(extraSlot);
483  }
484 
485  _vertexAttributesBound = totalSlotsBound;
486  }

650  private void ActivateGraphicsPipeline()
651  {
652  _graphicsPipelineActive = true;
653  _graphicsPipeline.EnsureResourcesCreated();
654 
655  Util.EnsureArrayMinimumSize(ref _graphicsResourceSets, (uint)_graphicsPipeline.ResourceLayouts.Length);
656  Util.EnsureArrayMinimumSize(ref _newGraphicsResourceSets, (uint)_graphicsPipeline.ResourceLayouts.Length);
657 
658  // Force ResourceSets to be re-bound.
659  for (int i = 0; i < _graphicsPipeline.ResourceLayouts.Length; i++)
660  {
661  _newGraphicsResourceSets[i] = true;
662  }
663 
664  // Blend State
665 
666  BlendStateDescription blendState = _graphicsPipeline.BlendState;
667  glBlendColor(blendState.BlendFactor.R, blendState.BlendFactor.G, blendState.BlendFactor.B, blendState.BlendFactor.A);
668  CheckLastError();
669 
670  if (blendState.AlphaToCoverageEnabled)
671  {
672  glEnable(EnableCap.SampleAlphaToCoverage);
673  CheckLastError();
674  }
675  else
676  {
677  glDisable(EnableCap.SampleAlphaToCoverage);
678  CheckLastError();
679  }
680 
681  if (_features.IndependentBlend)
682  {
683  for (uint i = 0; i < blendState.AttachmentStates.Length; i++)
684  {
685  BlendAttachmentDescription attachment = blendState.AttachmentStates[i];
686  ColorWriteMask colorMask = attachment.ColorWriteMask.GetOrDefault();
687 
688  glColorMaski(
689  i,
690  (colorMask & ColorWriteMask.Red) == ColorWriteMask.Red,
691  (colorMask & ColorWriteMask.Green) == ColorWriteMask.Green,
692  (colorMask & ColorWriteMask.Blue) == ColorWriteMask.Blue,
693  (colorMask & ColorWriteMask.Alpha) == ColorWriteMask.Alpha);
694  CheckLastError();
695 
696  if (!attachment.BlendEnabled)
697  {
698  glDisablei(EnableCap.Blend, i);
699  CheckLastError();
700  }
701  else
702  {
703  glEnablei(EnableCap.Blend, i);
704  CheckLastError();
705 
706  glBlendFuncSeparatei(
707  i,
708  OpenGLFormats.VdToGLBlendFactorSrc(attachment.SourceColorFactor),
709  OpenGLFormats.VdToGLBlendFactorDest(attachment.DestinationColorFactor),
710  OpenGLFormats.VdToGLBlendFactorSrc(attachment.SourceAlphaFactor),
711  OpenGLFormats.VdToGLBlendFactorDest(attachment.DestinationAlphaFactor));
712  CheckLastError();
713 
714  glBlendEquationSeparatei(
715  i,
716  OpenGLFormats.VdToGLBlendEquationMode(attachment.ColorFunction),
717  OpenGLFormats.VdToGLBlendEquationMode(attachment.AlphaFunction));
718  CheckLastError();
719  }
720  }
721  }
722  else if (blendState.AttachmentStates.Length > 0)
723  {
724  BlendAttachmentDescription attachment = blendState.AttachmentStates[0];
725  ColorWriteMask colorMask = attachment.ColorWriteMask.GetOrDefault();
726 
727  glColorMask(
728  (colorMask & ColorWriteMask.Red) == ColorWriteMask.Red,
729  (colorMask & ColorWriteMask.Green) == ColorWriteMask.Green,
730  (colorMask & ColorWriteMask.Blue) == ColorWriteMask.Blue,
731  (colorMask & ColorWriteMask.Alpha) == ColorWriteMask.Alpha);
732  CheckLastError();
733 
734  if (!attachment.BlendEnabled)
735  {
736  glDisable(EnableCap.Blend);
737  CheckLastError();
738  }
739  else
740  {
741  glEnable(EnableCap.Blend);
742  CheckLastError();
743 
744  glBlendFuncSeparate(
745  OpenGLFormats.VdToGLBlendFactorSrc(attachment.SourceColorFactor),
746  OpenGLFormats.VdToGLBlendFactorDest(attachment.DestinationColorFactor),
747  OpenGLFormats.VdToGLBlendFactorSrc(attachment.SourceAlphaFactor),
748  OpenGLFormats.VdToGLBlendFactorDest(attachment.DestinationAlphaFactor));
749  CheckLastError();
750 
751  glBlendEquationSeparate(
752  OpenGLFormats.VdToGLBlendEquationMode(attachment.ColorFunction),
753  OpenGLFormats.VdToGLBlendEquationMode(attachment.AlphaFunction));
754  CheckLastError();
755  }
756  }
757 
758  // Depth Stencil State
759 
760  DepthStencilStateDescription dss = _graphicsPipeline.DepthStencilState;
761  if (!dss.DepthTestEnabled)
762  {
763  glDisable(EnableCap.DepthTest);
764  CheckLastError();
765  }
766  else
767  {
768  glEnable(EnableCap.DepthTest);
769  CheckLastError();
770 
771  glDepthFunc(OpenGLFormats.VdToGLDepthFunction(dss.DepthComparison));
772  CheckLastError();
773  }
774 
775  glDepthMask(dss.DepthWriteEnabled);
776  CheckLastError();
777 
778  if (dss.StencilTestEnabled)
779  {
780  glEnable(EnableCap.StencilTest);
781  CheckLastError();
782 
783  glStencilFuncSeparate(
784  CullFaceMode.Front,
785  OpenGLFormats.VdToGLStencilFunction(dss.StencilFront.Comparison),
786  (int)dss.StencilReference,
787  dss.StencilReadMask);
788  CheckLastError();
789 
790  glStencilOpSeparate(
791  CullFaceMode.Front,
792  OpenGLFormats.VdToGLStencilOp(dss.StencilFront.Fail),
793  OpenGLFormats.VdToGLStencilOp(dss.StencilFront.DepthFail),
794  OpenGLFormats.VdToGLStencilOp(dss.StencilFront.Pass));
795  CheckLastError();
796 
797  glStencilFuncSeparate(
798  CullFaceMode.Back,
799  OpenGLFormats.VdToGLStencilFunction(dss.StencilBack.Comparison),
800  (int)dss.StencilReference,
801  dss.StencilReadMask);
802  CheckLastError();
803 
804  glStencilOpSeparate(
805  CullFaceMode.Back,
806  OpenGLFormats.VdToGLStencilOp(dss.StencilBack.Fail),
807  OpenGLFormats.VdToGLStencilOp(dss.StencilBack.DepthFail),
808  OpenGLFormats.VdToGLStencilOp(dss.StencilBack.Pass));
809  CheckLastError();
810 
811  glStencilMask(dss.StencilWriteMask);
812  CheckLastError();
813  }
814  else
815  {
816  glDisable(EnableCap.StencilTest);
817  CheckLastError();
818  }
819 
820  // Rasterizer State
821 
822  RasterizerStateDescription rs = _graphicsPipeline.RasterizerState;
823  if (rs.CullMode == FaceCullMode.None)
824  {
825  glDisable(EnableCap.CullFace);
826  CheckLastError();
827  }
828  else
829  {
830  glEnable(EnableCap.CullFace);
831  CheckLastError();
832 
833  glCullFace(OpenGLFormats.VdToGLCullFaceMode(rs.CullMode));
834  CheckLastError();
835  }
836 
837  if (_backend == GraphicsBackend.OpenGL)
838  {
839  glPolygonMode(MaterialFace.FrontAndBack, OpenGLFormats.VdToGLPolygonMode(rs.FillMode));
840  CheckLastError();
841  }
842 
843  if (!rs.ScissorTestEnabled)
844  {
845  glDisable(EnableCap.ScissorTest);
846  CheckLastError();
847  }
848  else
849  {
850  glEnable(EnableCap.ScissorTest);
851  CheckLastError();
852  }
853 
854  if (_backend == GraphicsBackend.OpenGL)
855  {
856  if (!rs.DepthClipEnabled)
857  {
858  glEnable(EnableCap.DepthClamp);
859  CheckLastError();
860  }
861  else
862  {
863  glDisable(EnableCap.DepthClamp);
864  CheckLastError();
865  }
866  }
867 
868  glFrontFace(OpenGLFormats.VdToGLFrontFaceDirection(rs.FrontFace));
869  CheckLastError();
870 
871  // Primitive Topology
872  _primitiveType = OpenGLFormats.VdToGLPrimitiveType(_graphicsPipeline.PrimitiveTopology);
873 
874  // Shader Set
875  glUseProgram(_graphicsPipeline.Program);
876  CheckLastError();
877 
878  int vertexStridesCount = _graphicsPipeline.VertexStrides.Length;
879  Util.EnsureArrayMinimumSize(ref _vertexBuffers, (uint)vertexStridesCount);
880  Util.EnsureArrayMinimumSize(ref _vbOffsets, (uint)vertexStridesCount);
881 
882  uint totalVertexElements = 0;
883  for (int i = 0; i < _graphicsPipeline.VertexLayouts.Length; i++)
884  {
885  totalVertexElements += (uint)_graphicsPipeline.VertexLayouts[i].Elements.Length;
886  }
887  Util.EnsureArrayMinimumSize(ref _vertexAttribDivisors, totalVertexElements);
888  }
