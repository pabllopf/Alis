48  public static Vertices GetConvexHull(Vertices vertices)
49  {
50  if (vertices.Count <= 3)
51  {
52  return vertices;
53  }
54 
55  Vertices pointSet = new Vertices(vertices);
56 
57  //Sort by X-axis
58  pointSet.Sort(PointComparerPrivate);
59 
60  Vector2[] h = new Vector2[pointSet.Count];
61  Vertices res;
62 
63  int top = -1; // indices for bottom and top of the stack
64  int i; // array scan index
65 
66  // Get the indices of points with min x-coord and min|max y-coord
67  const int minmin = 0;
68  float xmin = pointSet[0].X;
69  for (i = 1; i < pointSet.Count; i++)
70  {
71  if (pointSet[i].X != xmin)
72  {
73  break;
74  }
75  }
76 
77  // degenerate case: all x-coords == xmin
78  int minmax = i - 1;
79  if (minmax == pointSet.Count - 1)
80  {
81  h[++top] = pointSet[minmin];
82 
83  if (pointSet[minmax].Y != pointSet[minmin].Y) // a nontrivial segment
84  {
85  h[++top] = pointSet[minmax];
86  }
87 
88  h[++top] = pointSet[minmin]; // add polygon endpoint
89 
90  res = new Vertices(top + 1);
91  for (int j = 0; j < top + 1; j++)
92  {
93  res.Add(h[j]);
94  }
95 
96  return res;
97  }
98 
99  top = -1;
100 
101  // Get the indices of points with max x-coord and min|max y-coord
102  int maxmax = pointSet.Count - 1;
103  float xmax = pointSet[pointSet.Count - 1].X;
104  for (i = pointSet.Count - 2; i >= 0; i--)
105  {
106  if (pointSet[i].X != xmax)
107  {
108  break;
109  }
110  }
111 
112  int maxmin = i + 1;
113 
114  // Compute the lower hull on the stack H
115  h[++top] = pointSet[minmin]; // push minmin point onto stack
116  i = minmax;
117  while (++i <= maxmin)
118  {
119  // the lower line joins P[minmin] with P[maxmin]
120  if (MathUtils.Area(pointSet[minmin], pointSet[maxmin], pointSet[i]) >= 0 && i < maxmin)
121  {
122  continue; // ignore P[i] above or on the lower line
123  }
124 
125  while (top > 0) // there are at least 2 points on the stack
126  {
127  // test if P[i] is left of the line at the stack top
128  if (MathUtils.Area(h[top - 1], h[top], pointSet[i]) > 0)
129  {
130  break; // P[i] is a new hull vertex
131  }
132 
133  top--; // pop top point off stack
134  }
135 
136  h[++top] = pointSet[i]; // push P[i] onto stack
137  }
138 
139  // GetNext, compute the upper hull on the stack H above the bottom hull
140  if (maxmax != maxmin) // if distinct xmax points
141  {
142  h[++top] = pointSet[maxmax]; // push maxmax point onto stack
143  }
144 
145  int bot = top;
146  i = maxmin;
147  while (--i >= minmax)
148  {
149  // the upper line joins P[maxmax] with P[minmax]
150  if (MathUtils.Area(pointSet[maxmax], pointSet[minmax], pointSet[i]) >= 0 && i > minmax)
151  {
152  continue; // ignore P[i] below or on the upper line
153  }
154 
155  while (top > bot) // at least 2 points on the upper stack
156  {
157  // test if P[i] is left of the line at the stack top
158  if (MathUtils.Area(h[top - 1], h[top], pointSet[i]) > 0)
159  {
160  break; // P[i] is a new hull vertex
161  }
162 
163  top--; // pop top point off stack
164  }
165 
166  h[++top] = pointSet[i]; // push P[i] onto stack
167  }
168 
169  if (minmax != minmin)
170  {
171  h[++top] = pointSet[minmin]; // push joining endpoint onto stack
172  }
173 
174  res = new Vertices(top + 1);
175 
176  for (int j = 0; j < top + 1; j++)
177  {
178  res.Add(h[j]);
179  }
180 
181  return res;
182  }
183  }
184 }
