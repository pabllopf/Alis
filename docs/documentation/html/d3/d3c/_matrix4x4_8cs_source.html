324  float y2 = rotation.Y + rotation.Y;

332  float xz2 = rotation.X * z2;
333  float yy2 = rotation.Y * y2;
334  float yz2 = rotation.Y * z2;
335  float zz2 = rotation.Z * z2;
336 
337  float q11 = 1.0f - yy2 - zz2;
338  float q21 = xy2 - wz2;
339  float q31 = xz2 + wy2;
340

342  float q22 = 1.0f - xx2 - zz2;

392  public float GetDeterminant()
393  {
394  // | a b c d | | f g h | | e g h | | e f h | | e f g |
395  // | e f g h | = a | j k l | - b | i k l | + c | i j l | - d | i j k |
396  // | i j k l | | n o p | | m o p | | m n p | | m n o |
397  // | m n o p |
398  //
399  // | f g h |
400  // a | j k l | = a ( f ( kp - lo ) - g ( jp - ln ) + h ( jo - kn ) )
401  // | n o p |
402  //
403  // | e g h |
404  // b | i k l | = b ( e ( kp - lo ) - g ( ip - lm ) + h ( io - km ) )
405  // | m o p |
406  //
407  // | e f h |
408  // c | i j l | = c ( e ( jp - ln ) - f ( ip - lm ) + h ( in - jm ) )
409  // | m n p |
410  //
411  // | e f g |
412  // d | i j k | = d ( e ( jo - kn ) - f ( io - km ) + g ( in - jm ) )
413  // | m n o |
414  //
415  // Cost of operation
416  // 17 adds and 28 muls.
417  //
418  // add: 6 + 8 + 3 = 17
419  // mul: 12 + 16 = 28
420 
421  float a = M11, b = M12, c = M13, d = M14;
422  float e = M21, f = M22, g = M23, h = M24;
423  float i = M31, j = M32, k = M33, l = M34;
424  float m = M41, n = M42, o = M43, p = M44;
425 
426  float kpLo = k * p - l * o;
427  float jpLn = j * p - l * n;
428  float joKn = j * o - k * n;
429  float ipLm = i * p - l * m;
430  float ioKm = i * o - k * m;
431  float inJm = i * n - j * m;
432 
433  return a * (f * kpLo - g * jpLn + h * joKn) -
434  b * (e * kpLo - g * ipLm + h * ioKm) +
435  c * (e * jpLn - f * ipLm + h * inJm) -
436  d * (e * joKn - f * ioKm + g * inJm);
437  }
