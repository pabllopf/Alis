47  /* Methodology:
48  * Force applied at a ray is inversely proportional to the square of distance from source
49  * AABB is used to query for shapes that may be affected
50  * For each RIGID BODY (not shape -- this is an optimization) that is matched, loop through its vertices to determine
51  * the extreme points -- if there is structure that contains outlining polygon, use that as an additional optimization
52  * Evenly cast a number of rays against the shape - number roughly proportional to the arc coverage
53  * - Something like every 3 degrees should do the trick although this can be altered depending on the distance (if really close don't need such a high density of rays)
54  * - There should be a minimum number of rays (3-5?) applied to each body so that small bodies far away are still accurately modeled
55  * - Be sure to have the forces of each ray be proportional to the average arc length covered by each.
56  * For each ray that actually intersects with the shape (non intersections indicate something blocking the path of explosion):
57  * - Apply the appropriate force dotted with the negative of the collision normal at the collision point
58  * - Optionally apply linear interpolation between aforementioned Normal force and the original explosion force in the direction of ray to simulate "surface friction" of sorts
59  */

99  : base(world, PhysicsLogicType.Explosion)

116  public Dictionary<Fixture, Vector2> Activate(Vector2 pos, float radius, float maxForce)
117  {
118  Aabb aabb;
119  aabb.LowerBound = pos + new Vector2(-radius, -radius);
120  aabb.UpperBound = pos + new Vector2(radius, radius);
121  Fixture[] shapes = new Fixture[MaxShapes];
122 
123  // More than 5 shapes in an explosion could be possible, but still strange.
124  Fixture[] containedShapes = new Fixture[5];
125  bool exit = false;
126 
127  int shapeCount = 0;
128  int containedShapeCount = 0;
129 
130  // Query the world for overlapping shapes.
131  World.TestPointAllFixtures.ForEach(fixture =>
132  {
133  if (fixture.TestPoint(ref pos))
134  {
135  if (IgnoreWhenInsideShape)
136  {
137  exit = true;
138  }
139  else
140  {
141  containedShapes[containedShapeCount++] = fixture;
142  }
143  }
144  else
145  {
146  shapes[shapeCount++] = fixture;
147  }
148  });
149 
150  if (exit)
151  {
152  return new Dictionary<Fixture, Vector2>();
153  }
154 
155  Dictionary<Fixture, Vector2> exploded = new Dictionary<Fixture, Vector2>(shapeCount + containedShapeCount);
156 
157  // Per shape max/min angles for now.
158  float[] vals = new float[shapeCount * 2];
159  int valIndex = 0;
160  for (int i = 0; i < shapeCount; ++i)
161  {
162  PolygonShape ps;
163  if (shapes[i].Shape is CircleShape cs)
164  {
165  // We create a "diamond" approximation of the circle
166  Vertices v = new Vertices();
167  Vector2 vec = Vector2.Zero + new Vector2(cs.RadiusPrivate, 0);
168  v.Add(vec);
169  vec = Vector2.Zero + new Vector2(0, cs.RadiusPrivate);
170  v.Add(vec);
171  vec = Vector2.Zero + new Vector2(-cs.RadiusPrivate, cs.RadiusPrivate);
172  v.Add(vec);
173  vec = Vector2.Zero + new Vector2(0, -cs.RadiusPrivate);
174  v.Add(vec);
175  ps = new PolygonShape(v, 0);
176  }
177  else
178  {
179  ps = shapes[i].Shape as PolygonShape;
180  }
181 
182  if (shapes[i].Body.BodyType == BodyType.Dynamic && ps != null)
183  {
184  Vector2 toCentroid = shapes[i].Body.GetWorldPoint(ps.MassDataPrivate.Centroid) - pos;
185  float angleToCentroid = (float) Math.Atan2(toCentroid.Y, toCentroid.X);
186  float min = float.MaxValue;
187  float max = float.MinValue;
188  float minAbsolute = 0.0f;
189  float maxAbsolute = 0.0f;
190 
191  for (int j = 0; j < ps.VerticesPrivate.Count; ++j)
192  {
193  Vector2 toVertex = shapes[i].Body.GetWorldPoint(ps.VerticesPrivate[j]) - pos;
194  float newAngle = (float) Math.Atan2(toVertex.Y, toVertex.X);
195  float diff = newAngle - angleToCentroid;
196 
197  diff = (diff - MathConstants.Pi) % (2 * MathConstants.Pi);
198 
199  // the minus pi is important. It means cutoff for going other direction is at 180 deg where it needs to be
200 
201  if (diff < 0.0f)
202  {
203  diff += 2 * MathConstants.Pi; // correction for not handling negs
204  }
205 
206  diff -= MathConstants.Pi;
207 
208  if (Math.Abs(diff) > MathConstants.Pi)
209  {
210  continue; // Something's wrong, point not in shape but exists angle diff > 180
211  }
212 
213  if (diff > max)
214  {
215  max = diff;
216  maxAbsolute = newAngle;
217  }
218 
219  if (diff < min)
220  {
221  min = diff;
222  minAbsolute = newAngle;
223  }
224  }
225 
226  vals[valIndex] = minAbsolute;
227  ++valIndex;
228  vals[valIndex] = maxAbsolute;
229  ++valIndex;
230  }
231  }
232 
233  Array.Sort(vals, 0, valIndex, rdc);
234  data.Clear();
235  bool rayMissed = true;
236 
237  for (int i = 0; i < valIndex; ++i)
238  {
239  Fixture fixture = null;
240  float midpt;
241 
242  int iplus = i == valIndex - 1 ? 0 : i + 1;
243  if (vals[i] == vals[iplus])
244  {
245  continue;
246  }
247 
248  if (i == valIndex - 1)
249  {
250  // the single edgecase
251  midpt = vals[0] + MathConstants.Pi * 2 + vals[i];
252  }
253  else
254  {
255  midpt = vals[i + 1] + vals[i];
256  }
257 
258  midpt /= 2;
259 
260  Vector2 p1 = pos;
261  Vector2 p2 = radius * new Vector2((float) Math.Cos(midpt), (float) Math.Sin(midpt)) + pos;
262 
263  // RaycastOne
264  bool hitClosest = false;
265  World.RayCast((f, p, n, fr) =>
266  {
267  Body body = f.Body;
268 
269  if (!IsActiveOn(body))
270  {
271  return 0;
272  }
273 
274  hitClosest = true;
275  fixture = f;
276  return fr;
277  }, p1, p2);
278 
279  //draws radius points
280  if (hitClosest && fixture.Body.BodyType == BodyType.Dynamic)
281  {
282  if (data.Any() && data.Last().Body == fixture.Body && !rayMissed)
283  {
284  int laPos = data.Count - 1;
285  ShapeData la = data[laPos];
286  la.Max = vals[iplus];
287  data[laPos] = la;
288  }
289  else
290  {
291  // make new
292  ShapeData d;
293  d.Body = fixture.Body;
294  d.Min = vals[i];
295  d.Max = vals[iplus];
296  data.Add(d);
297  }
298 
299  if (data.Count > 1
300  && i == valIndex - 1
301  && data.Last().Body == data.First().Body
302  && data.Last().Max == data.First().Min)
303  {
304  ShapeData fi = data[0];
305  fi.Min = data.Last().Min;
306  data.RemoveAt(data.Count - 1);
307  data[0] = fi;
308  while (data.First().Min >= data.First().Max)
309  {
310  fi.Min -= MathConstants.Pi * 2;
311  data[0] = fi;
312  }
313  }
314 
315  int lastPos = data.Count - 1;
316  ShapeData last = data[lastPos];
317  while (data.Count > 0
318  && data.Last().Min >= data.Last().Max) // just making sure min<max
319  {
320  last.Min = data.Last().Min - 2 * MathConstants.Pi;
321  data[lastPos] = last;
322  }
323 
324  rayMissed = false;
325  }
326  else
327  {
328  rayMissed = true; // raycast did not find a shape
329  }
330  }
331 
332  for (int i = 0; i < data.Count; ++i)
333  {
334  if (!IsActiveOn(data[i].Body))
335  {
336  continue;
337  }
338 
339  float arclen = data[i].Max - data[i].Min;
340 
341  float first = MathHelper.Min(MaxEdgeOffset, EdgeRatio * arclen);
342  int insertedRays = (int) Math.Ceiling((arclen - 2.0f * first - (MinRays - 1) * MaxAngle) / MaxAngle);
343 
344  if (insertedRays < 0)
345  {
346  insertedRays = 0;
347  }
348 
349  float offset = (arclen - first * 2.0f) / ((float) MinRays + insertedRays - 1);
350 
351  //Note: This loop can go into infinite as it operates on floats.
352  //Added FloatEquals with a large epsilon.
353  for (float j = data[i].Min + first;
354  j < data[i].Max || MathUtils.FloatEquals(j, data[i].Max, 0.0001f);
355  j += offset)
356  {
357  Vector2 p1 = pos;
358  Vector2 p2 = pos + radius * new Vector2((float) Math.Cos(j), (float) Math.Sin(j));
359  Vector2 hitpoint = Vector2.Zero;
360  float minlambda = float.MaxValue;
361 
362  List<Fixture> fl = data[i].Body.FixtureList;
363  for (int x = 0; x < fl.Count; x++)
364  {
365  Fixture f = fl[x];
366  RayCastInput ri;
367  ri.Point1 = p1;
368  ri.Point2 = p2;
369  ri.MaxFraction = 50f;
370 
371  if (f.RayCast(out RayCastOutput ro, ref ri, 0))
372  {
373  if (minlambda > ro.Fraction)
374  {
375  minlambda = ro.Fraction;
376  hitpoint = ro.Fraction * p2 + (1 - ro.Fraction) * p1;
377  }
378  }
379 
380  // the force that is to be applied for this particular ray.
381  // offset is angular coverage. lambda*length of segment is distance.
382  float impulse = arclen / (MinRays + insertedRays) * maxForce * 180.0f / MathConstants.Pi *
383  (1.0f - Math.Min(1.0f, minlambda));
384 
385  // We Apply the impulse!!!
386  Vector2 vectImp =
387  Vector2.Dot(impulse * new Vector2((float) Math.Cos(j), (float) Math.Sin(j)), -ro.Normal) *
388  new Vector2((float) Math.Cos(j), (float) Math.Sin(j));
389  data[i].Body.ApplyLinearImpulse(ref vectImp, ref hitpoint);
390 
391  // We gather the fixtures for returning them
392  if (exploded.ContainsKey(f))
393  {
394  exploded[f] += vectImp;
395  }
396  else
397  {
398  exploded.Add(f, vectImp);
399  }
400 
401  if (minlambda > 1.0f)
402  {
403  hitpoint = p2;
404  }
405  }
406  }
407  }
408 
409  // We check contained shapes
410  for (int i = 0; i < containedShapeCount; ++i)
411  {
412  Fixture fix = containedShapes[i];
413 
414  if (!IsActiveOn(fix.Body))
415  {
416  continue;
417  }
418 
419  float impulse = MinRays * maxForce * 180.0f / MathConstants.Pi;
420  Vector2 hitPoint;
421 
422  if (fix.Shape is CircleShape circShape)
423  {
424  hitPoint = fix.Body.GetWorldPoint(circShape.Position);
425  }
426  else
427  {
428  PolygonShape shape = fix.Shape as PolygonShape;
429  hitPoint = fix.Body.GetWorldPoint(shape.MassDataPrivate.Centroid);
430  }
431 
432  Vector2 vectImp = impulse * (hitPoint - pos);
433 
434  fix.Body.ApplyLinearImpulse(ref vectImp, ref hitPoint);
435 
436  if (!exploded.ContainsKey(fix))
437  {
438  exploded.Add(fix, vectImp);
439  }
440  }
441 
442  return exploded;
443  }
444  }
