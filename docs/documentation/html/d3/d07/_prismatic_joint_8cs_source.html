517  internal override void InitVelocityConstraints(ref SolverData data)
518  {
519  indexA = BodyA.IslandIndex;
520  indexB = BodyB.IslandIndex;
521  localCenterA = BodyA.Sweep.LocalCenter;
522  localCenterB = BodyB.Sweep.LocalCenter;
523  invMassA = BodyA.InvMass;
524  invMassB = BodyB.InvMass;
525  invIa = BodyA.InvI;
526  invIb = BodyB.InvI;
527 
528  Vector2F cA = data.Positions[indexA].C;
529  float aA = data.Positions[indexA].A;
530  Vector2F vA = data.Velocities[indexA].V;
531  float wA = data.Velocities[indexA].W;
532 
533  Vector2F cB = data.Positions[indexB].C;
534  float aB = data.Positions[indexB].A;
535  Vector2F vB = data.Velocities[indexB].V;
536  float wB = data.Velocities[indexB].W;
537 
538  Rotation qA = new Rotation(aA), qB = new Rotation(aB);
539 
540  Vector2F rA = MathUtils.Mul(qA, LocalAnchorA - localCenterA);
541  Vector2F rB = MathUtils.Mul(qB, LocalAnchorB - localCenterB);
542  Vector2F d = cB - cA + rB - rA;
543 
544  float mA = invMassA, mB = invMassB;
545  float iA = invIa, iB = invIb;
546 
547  {
548  axis = MathUtils.Mul(qA, LocalXAxisA);
549  a1 = MathUtils.Cross(d + rA, axis);
550  a2 = MathUtils.Cross(rB, axis);
551 
552  axialMass = mA + mB + iA * a1 * a1 + iB * a2 * a2;
553  if (axialMass > 0.0f)
554  {
555  axialMass = 1.0f / axialMass;
556  }
557  }
558 
559  {
560  perp = MathUtils.Mul(qA, localYAxisA);
561 
562  s1 = MathUtils.Cross(d + rA, perp);
563  s2 = MathUtils.Cross(rB, perp);
564 
565  float k11 = mA + mB + iA * s1 * s1 + iB * s2 * s2;
566  float k12 = iA * s1 + iB * s2;
567  float k22 = iA + iB;
568  if (k22 == 0.0f)
569  {
570  k22 = 1.0f;
571  }
572 
573  k.Ex = new Vector2F(k11, k12);
574  k.Ey = new Vector2F(k12, k22);
575  }
576 
577  if (enableLimit)
578  {
579  translation = Vector2F.Dot(axis, d);
580  }
581  else
582  {
583  lowerImpulse = 0.0f;
584  upperImpulse = 0.0f;
585  }
586 
587  if (!enableMotor)
588  {
589  motorImpulse = 0.0f;
590  }
591 
592  if (data.Step.WarmStarting)
593  {
594  impulse *= data.Step.DeltaTimeRatio;
595  motorImpulse *= data.Step.DeltaTimeRatio;
596  lowerImpulse *= data.Step.DeltaTimeRatio;
597  upperImpulse *= data.Step.DeltaTimeRatio;
598 
599  float axialImpulse = motorImpulse + lowerImpulse - upperImpulse;
600  Vector2F p = impulse.X * perp + axialImpulse * axis;
601  float la = impulse.X * s1 + impulse.Y + axialImpulse * a1;
602  float lb = impulse.X * s2 + impulse.Y + axialImpulse * a2;
603 
604  vA -= mA * p;
605  wA -= iA * la;
606 
607  vB += mB * p;
608  wB += iB * lb;
609  }
610  else
611  {
612  impulse = Vector2F.Zero;
613  motorImpulse = 0.0f;
614  lowerImpulse = 0.0f;
615  upperImpulse = 0.0f;
616  }
617 
618  data.Velocities[indexA].V = vA;
619  data.Velocities[indexA].W = wA;
620  data.Velocities[indexB].V = vB;
621  data.Velocities[indexB].W = wB;
622  }

628  internal override void SolveVelocityConstraints(ref SolverData data)
629  {
630  Vector2F vA = data.Velocities[indexA].V;
631  float wA = data.Velocities[indexA].W;
632  Vector2F vB = data.Velocities[indexB].V;
633  float wB = data.Velocities[indexB].W;
634 
635  float mA = invMassA, mB = invMassB;
636  float iA = invIa, iB = invIb;
637 
638  if (enableMotor)
639  {
640  float dot = Vector2F.Dot(axis, vB - vA) + a2 * wB - a1 * wA;
641  float impulseLocal = axialMass * (motorSpeed - dot);
642  float oldImpulse = motorImpulse;
643  float maxImpulse = data.Step.DeltaTime * maxMotorForce;
644  motorImpulse = MathUtils.Clamp(motorImpulse + impulseLocal, -maxImpulse, maxImpulse);
645  impulseLocal = motorImpulse - oldImpulse;
646 
647  Vector2F p = impulseLocal * axis;
648  float la = impulseLocal * a1;
649  float lb = impulseLocal * a2;
650 
651  vA -= mA * p;
652  wA -= iA * la;
653  vB += mB * p;
654  wB += iB * lb;
655  }
656 
657  if (enableLimit)
658  {
659  {
660  float c = translation - lowerTranslation;
661  float dot = MathUtils.Dot(axis, vB - vA) + a2 * wB - a1 * wA;
662  float impulseLocal = -axialMass * (dot + MathUtils.Max(c, 0.0f) * data.Step.InvertedDeltaTime);
663  float oldImpulse = lowerImpulse;
664  lowerImpulse = MathUtils.Max(lowerImpulse + impulseLocal, 0.0f);
665  impulseLocal = lowerImpulse - oldImpulse;
666 
667  Vector2F p = impulseLocal * axis;
668  float la = impulseLocal * a1;
669  float lb = impulseLocal * a2;
670 
671  vA -= mA * p;
672  wA -= iA * la;
673  vB += mB * p;
674  wB += iB * lb;
675  }
676 
677  {
678  float c = upperTranslation - translation;
679  float dot = MathUtils.Dot(axis, vA - vB) + a1 * wA - a2 * wB;
680  float impulseLocal = -axialMass * (dot + MathUtils.Max(c, 0.0f) * data.Step.InvertedDeltaTime);
681  float oldImpulse = upperImpulse;
682  upperImpulse = MathUtils.Max(upperImpulse + impulseLocal, 0.0f);
683  impulseLocal = upperImpulse - oldImpulse;
684 
685  Vector2F p = impulseLocal * axis;
686  float la = impulseLocal * a1;
687  float lb = impulseLocal * a2;
688 
689  vA += mA * p;
690  wA += iA * la;
691  vB -= mB * p;
692  wB -= iB * lb;
693  }
694  }
695 
696  {
697  Vector2F dot = new Vector2F(
698  MathUtils.Dot(perp, vB - vA) + s2 * wB - s1 * wA,
699  wB - wA
700  );
701 
702  Vector2F df = k.Solve(-dot);
703  impulse += df;
704 
705  Vector2F p = df.X * perp;
706  float la = df.X * s1 + df.Y;
707  float lb = df.X * s2 + df.Y;
708 
709  vA -= mA * p;
710  wA -= iA * la;
711 
712  vB += mB * p;
713  wB += iB * lb;
714  }
715 
716  data.Velocities[indexA].V = vA;
717  data.Velocities[indexA].W = wA;
718  data.Velocities[indexB].V = vB;
719  data.Velocities[indexB].W = wB;
720  }

735  internal override bool SolvePositionConstraints(ref SolverData data)
736  {
737  Vector2F cA = data.Positions[indexA].C;
738  float aA = data.Positions[indexA].A;
739  Vector2F cB = data.Positions[indexB].C;
740  float aB = data.Positions[indexB].A;
741 
742  Rotation qA = new Rotation(aA), qB = new Rotation(aB);
743 
744  float mA = invMassA, mB = invMassB;
745  float iA = invIa, iB = invIb;
746 
747  Vector2F rA = MathUtils.Mul(qA, LocalAnchorA - localCenterA);
748  Vector2F rB = MathUtils.Mul(qB, LocalAnchorB - localCenterB);
749  Vector2F d = cB + rB - cA - rA;
750 
751  Vector2F axisLocal = MathUtils.Mul(qA, LocalXAxisA);
752  float a1Local = MathUtils.Cross(d + rA, axisLocal);
753  float a2Local = MathUtils.Cross(rB, axisLocal);
754  Vector2F perpLocal = MathUtils.Mul(qA, localYAxisA);
755 
756  float s1Local = MathUtils.Cross(d + rA, perpLocal);
757  float s2Local = MathUtils.Cross(rB, perpLocal);
758 
759  Vector3F impulseLocal;
760  Vector2F c1 = new Vector2F(
761  Vector2F.Dot(perpLocal, d),
762  aB - aA - ReferenceAngle
763  );
764 
765  float linearError = MathUtils.Abs(c1.X);
766  float angularError = MathUtils.Abs(c1.Y);
767 
768  bool active = false;
769  float c2 = 0.0f;
770  if (enableLimit)
771  {
772  float translationLocal = MathUtils.Dot(axisLocal, d);
773  if (MathUtils.Abs(upperTranslation - lowerTranslation) < 2.0f * Settings.LinearSlop)
774  {
775  c2 = translationLocal;
776  linearError = MathUtils.Max(linearError, MathUtils.Abs(translationLocal));
777  active = true;
778  }
779  else if (translationLocal <= lowerTranslation)
780  {
781  c2 = MathUtils.Min(translationLocal - lowerTranslation, 0.0f);
782  linearError = MathUtils.Max(linearError, lowerTranslation - translationLocal);
783  active = true;
784  }
785  else if (translationLocal >= upperTranslation)
786  {
787  c2 = MathUtils.Max(translationLocal - upperTranslation, 0.0f);
788  linearError = MathUtils.Max(linearError, translationLocal - upperTranslation);
789  active = true;
790  }
791  }
792 
793  if (active)
794  {
795  float k11 = mA + mB + iA * s1Local * s1Local + iB * s2Local * s2Local;
796  float k12 = iA * s1Local + iB * s2Local;
797  float k13 = iA * s1Local * a1Local + iB * s2Local * a2Local;
798  float k22 = iA + iB;
799  if (k22 == 0.0f)
800  {
801  k22 = 1.0f;
802  }
803 
804  float k23 = iA * a1Local + iB * a2Local;
805  float k33 = mA + mB + iA * a1Local * a1Local + iB * a2Local * a2Local;
806 
807  Matrix3X3F matrix3X3F = new Matrix3X3F(
808  new Vector3F(k11, k12, k13),
809  new Vector3F(k12, k22, k23),
810  new Vector3F(k13, k23, k33)
811  )
812  {
813  Ex = new Vector3F(k11, k12, k13),
814  Ey = new Vector3F(k12, k22, k23),
815  Ez = new Vector3F(k13, k23, k33)
816  };
817 
818  Vector3F c = new Vector3F(
819  c1.X,
820  c1.Y,
821  c2
822  );
823 
824  impulseLocal = matrix3X3F.Solve33(-c);
825  }
826  else
827  {
828  float k11 = mA + mB + iA * s1Local * s1Local + iB * s2Local * s2Local;
829  float k12 = iA * s1Local + iB * s2Local;
830  float k22 = iA + iB;
831  if (k22 == 0.0f)
832  {
833  k22 = 1.0f;
834  }
835 
836  Matrix2X2F matrix2X2F = new Matrix2X2F
837  {
838  Ex = new Vector2F(k11, k12),
839  Ey = new Vector2F(k12, k22)
840  };
841 
842  Vector2F impulse1 = matrix2X2F.Solve(-c1);
843  impulseLocal = new Vector3F(
844  impulse1.X,
845  impulse1.Y,
846  0.0f
847  );
848  }
849 
850  Vector2F p = impulseLocal.X * perpLocal + impulseLocal.Z * axisLocal;
851  float la = impulseLocal.X * s1Local + impulseLocal.Y + impulseLocal.Z * a1Local;
852  float lb = impulseLocal.X * s2Local + impulseLocal.Y + impulseLocal.Z * a2Local;
853 
854  cA -= mA * p;
855  aA -= iA * la;
856  cB += mB * p;
857  aB += iB * lb;
858 
859  data.Positions[indexA].C = cA;
860  data.Positions[indexA].A = aA;
861  data.Positions[indexB].C = cB;
862  data.Positions[indexB].A = aB;
863 
864  return (linearError <= Settings.LinearSlop) && (angularError <= Settings.AngularSlop);
865  }
