41  // Linear constraint (point-to-line)
42  // d = p2 - p1 = x2 + r2 - x1 - r1
43  // C = dot(perp, d)
44  // Cdot = dot(d, cross(w1, perp)) + dot(perp, v2 + cross(w2, r2) - v1 - cross(w1, r1))
45  // = -dot(perp, v1) - dot(cross(d + r1, perp), w1) + dot(perp, v2) + dot(cross(r2, perp), v2)
46  // J = [-perp, -cross(d + r1, perp), perp, cross(r2,perp)]
47  //
48  // Angular constraint
49  // C = a2 - a1 + a_initial
50  // Cdot = w2 - w1
51  // J = [0 0 -1 0 0 1]
52  //
53  // K = J * invM * JT
54  //
55  // J = [-a -s1 a s2]
56  // [0 -1 0 1]
57  // a = perp
58  // s1 = cross(d + r1, a) = cross(p2 - x1, a)
59  // s2 = cross(r2, a) = cross(p2 - x2, a)
60 
61  // Motor/Limit linear constraint
62  // C = dot(ax1, d)
63  // Cdot = -dot(ax1, v1) - dot(cross(d + r1, ax1), w1) + dot(ax1, v2) + dot(cross(r2, ax1), v2)
64  // J = [-ax1 -cross(d+r1,ax1) ax1 cross(r2,ax1)]
65 
66  // Predictive limit is applied even when the limit is not active.
67  // Prevents a constraint speed that can lead to a constraint error in one time step.
68  // Want C2 = C1 + h * Cdot >= 0
69  // Or:
70  // Cdot + C1/h >= 0
71  // I do not apply a negative constraint error because that is handled in position correction.
72  // So:
73  // Cdot + max(C1, 0)/h >= 0
74 
75  // Block Solver
76  // We develop a block solver that includes the angular and linear constraints. This makes the limit stiffer.
77  //
78  // The Jacobian has 2 rows:
79  // J = [-uT -s1 uT s2] // linear
80  // [0 -1 0 1] // angular
81  //
82  // u = perp
83  // s1 = cross(d + r1, u), s2 = cross(r2, u)
84  // a1 = cross(d + r1, v), a2 = cross(r2, v)

523  internal override void InitVelocityConstraints(ref SolverData data)
524  {
525  indexA = BodyA.IslandIndex;
526  indexB = BodyB.IslandIndex;
527  localCenterA = BodyA.Sweep.LocalCenter;
528  localCenterB = BodyB.Sweep.LocalCenter;
529  invMassA = BodyA.InvMass;
530  invMassB = BodyB.InvMass;
531  invIa = BodyA.InvI;
532  invIb = BodyB.InvI;
533 
534  Vector2 cA = data.Positions[indexA].C;
535  float aA = data.Positions[indexA].A;
536  Vector2 vA = data.Velocities[indexA].V;
537  float wA = data.Velocities[indexA].W;
538 
539  Vector2 cB = data.Positions[indexB].C;
540  float aB = data.Positions[indexB].A;
541  Vector2 vB = data.Velocities[indexB].V;
542  float wB = data.Velocities[indexB].W;
543 
544  Rot qA = new Rot(aA), qB = new Rot(aB);
545 
546  // Compute the effective masses.
547  Vector2 rA = MathUtils.Mul(qA, LocalAnchorA - localCenterA);
548  Vector2 rB = MathUtils.Mul(qB, LocalAnchorB - localCenterB);
549  Vector2 d = cB - cA + rB - rA;
550 
551  float mA = invMassA, mB = invMassB;
552  float iA = invIa, iB = invIb;
553 
554  // Compute motor Jacobian and effective mass.
555  {
556  axis = MathUtils.Mul(qA, LocalXAxisA);
557  a1 = MathUtils.Cross(d + rA, axis);
558  a2 = MathUtils.Cross(rB, axis);
559 
560  axialMass = mA + mB + iA * a1 * a1 + iB * a2 * a2;
561  if (axialMass > 0.0f)
562  {
563  axialMass = 1.0f / axialMass;
564  }
565  }
566 
567  // Prismatic constraint.
568  {
569  perp = MathUtils.Mul(qA, localYAxisA);
570 
571  s1 = MathUtils.Cross(d + rA, perp);
572  s2 = MathUtils.Cross(rB, perp);
573 
574  float k11 = mA + mB + iA * s1 * s1 + iB * s2 * s2;
575  float k12 = iA * s1 + iB * s2;
576  float k22 = iA + iB;
577  if (k22 == 0.0f)
578  {
579  // For bodies with fixed rotation.
580  k22 = 1.0f;
581  }
582 
583  k.Ex = new Vector2(k11, k12);
584  k.Ey = new Vector2(k12, k22);
585  }
586 
587  if (enableLimit)
588  {
589  translation = Vector2.Dot(axis, d);
590  }
591  else
592  {
593  lowerImpulse = 0.0f;
594  upperImpulse = 0.0f;
595  }
596 
597  if (!enableMotor)
598  {
599  motorImpulse = 0.0f;
600  }
601 
602  if (data.Step.WarmStarting)
603  {
604  // Account for variable time step.
605  impulse *= data.Step.DeltaTimeRatio;
606  motorImpulse *= data.Step.DeltaTimeRatio;
607  lowerImpulse *= data.Step.DeltaTimeRatio;
608  upperImpulse *= data.Step.DeltaTimeRatio;
609 
610  float axialImpulse = motorImpulse + lowerImpulse - upperImpulse;
611  Vector2 p = impulse.X * perp + axialImpulse * axis;
612  float la = impulse.X * s1 + impulse.Y + axialImpulse * a1;
613  float lb = impulse.X * s2 + impulse.Y + axialImpulse * a2;
614 
615  vA -= mA * p;
616  wA -= iA * la;
617 
618  vB += mB * p;
619  wB += iB * lb;
620  }
621  else
622  {
623  impulse = Vector2.Zero;
624  motorImpulse = 0.0f;
625  lowerImpulse = 0.0f;
626  upperImpulse = 0.0f;
627  }
628 
629  data.Velocities[indexA].V = vA;
630  data.Velocities[indexA].W = wA;
631  data.Velocities[indexB].V = vB;
632  data.Velocities[indexB].W = wB;
633  }

639  internal override void SolveVelocityConstraints(ref SolverData data)
640  {
641  Vector2 vA = data.Velocities[indexA].V;
642  float wA = data.Velocities[indexA].W;
643  Vector2 vB = data.Velocities[indexB].V;
644  float wB = data.Velocities[indexB].W;
645 
646  float mA = invMassA, mB = invMassB;
647  float iA = invIa, iB = invIb;
648 
649  // Solve linear motor constraint.
650  if (enableMotor)
651  {
652  float cdot = Vector2.Dot(axis, vB - vA) + a2 * wB - a1 * wA;
653  float impulse = axialMass * (motorSpeed - cdot);
654  float oldImpulse = motorImpulse;
655  float maxImpulse = data.Step.DeltaTime * maxMotorForce;
656  motorImpulse = MathUtils.Clamp(motorImpulse + impulse, -maxImpulse, maxImpulse);
657  impulse = motorImpulse - oldImpulse;
658 
659  Vector2 p = impulse * axis;
660  float la = impulse * a1;
661  float lb = impulse * a2;
662 
663  vA -= mA * p;
664  wA -= iA * la;
665  vB += mB * p;
666  wB += iB * lb;
667  }
668 
669  if (enableLimit)
670  {
671  // Lower limit
672  {
673  float c = translation - lowerTranslation;
674  float cdot = MathUtils.Dot(axis, vB - vA) + a2 * wB - a1 * wA;
675  float impulse = -axialMass * (cdot + MathUtils.Max(c, 0.0f) * data.Step.InvertedDeltaTime);
676  float oldImpulse = lowerImpulse;
677  lowerImpulse = MathUtils.Max(lowerImpulse + impulse, 0.0f);
678  impulse = lowerImpulse - oldImpulse;
679 
680  Vector2 p = impulse * axis;
681  float la = impulse * a1;
682  float lb = impulse * a2;
683 
684  vA -= mA * p;
685  wA -= iA * la;
686  vB += mB * p;
687  wB += iB * lb;
688  }
689 
690  // Upper limit
691  // Note: signs are flipped to keep C positive when the constraint is satisfied.
692  // This also keeps the impulse positive when the limit is active.
693  {
694  float c = upperTranslation - translation;
695  float cdot = MathUtils.Dot(axis, vA - vB) + a1 * wA - a2 * wB;
696  float impulse = -axialMass * (cdot + MathUtils.Max(c, 0.0f) * data.Step.InvertedDeltaTime);
697  float oldImpulse = upperImpulse;
698  upperImpulse = MathUtils.Max(upperImpulse + impulse, 0.0f);
699  impulse = upperImpulse - oldImpulse;
700 
701  Vector2 p = impulse * axis;
702  float la = impulse * a1;
703  float lb = impulse * a2;
704 
705  vA += mA * p;
706  wA += iA * la;
707  vB -= mB * p;
708  wB -= iB * lb;
709  }
710  }
711 
712  // Solve the prismatic constraint in block form.
713  {
714  Vector2 cdot;
715  cdot.X = MathUtils.Dot(perp, vB - vA) + s2 * wB - s1 * wA;
716  cdot.Y = wB - wA;
717 
718  Vector2 df = k.Solve(-cdot);
719  impulse += df;
720 
721  Vector2 p = df.X * perp;
722  float la = df.X * s1 + df.Y;
723  float lb = df.X * s2 + df.Y;
724 
725  vA -= mA * p;
726  wA -= iA * la;
727 
728  vB += mB * p;
729  wB += iB * lb;
730  }
731 
732  data.Velocities[indexA].V = vA;
733  data.Velocities[indexA].W = wA;
734  data.Velocities[indexB].V = vB;
735  data.Velocities[indexB].W = wB;
736  }

738  // A velocity based solver computes reaction forces(impulses) using the velocity constraint solver. Under this context,
739  // the position solver is not there to resolve forces. It is only there to cope with integration error.
740  //
741  // Therefore, the pseudo impulses in the position solver do not have any physical meaning. Thus it is okay if they suck.
742  //
743  // We could take the active state from the velocity solver. However, the joint might push past the limit when the velocity
744  // solver indicates the limit is inactive.
750  internal override bool SolvePositionConstraints(ref SolverData data)
751  {
752  Vector2 cA = data.Positions[indexA].C;
753  float aA = data.Positions[indexA].A;
754  Vector2 cB = data.Positions[indexB].C;
755  float aB = data.Positions[indexB].A;
756 
757  Rot qA = new Rot(aA), qB = new Rot(aB);
758 
759  float mA = invMassA, mB = invMassB;
760  float iA = invIa, iB = invIb;
761 
762  // Compute fresh Jacobians
763  Vector2 rA = MathUtils.Mul(qA, LocalAnchorA - localCenterA);
764  Vector2 rB = MathUtils.Mul(qB, LocalAnchorB - localCenterB);
765  Vector2 d = cB + rB - cA - rA;
766 
767  Vector2 axis = MathUtils.Mul(qA, LocalXAxisA);
768  float a1 = MathUtils.Cross(d + rA, axis);
769  float a2 = MathUtils.Cross(rB, axis);
770  Vector2 perp = MathUtils.Mul(qA, localYAxisA);
771 
772  float s1 = MathUtils.Cross(d + rA, perp);
773  float s2 = MathUtils.Cross(rB, perp);
774 
775  Vector3 impulse;
776  Vector2 c1;
777  c1.X = Vector2.Dot(perp, d);
778  c1.Y = aB - aA - ReferenceAngle;
779 
780  float linearError = MathUtils.Abs(c1.X);
781  float angularError = MathUtils.Abs(c1.Y);
782 
783  bool active = false;
784  float c2 = 0.0f;
785  if (enableLimit)
786  {
787  float translation = MathUtils.Dot(axis, d);
788  if (MathUtils.Abs(upperTranslation - lowerTranslation) < 2.0f * Settings.LinearSlop)
789  {
790  c2 = translation;
791  linearError = MathUtils.Max(linearError, MathUtils.Abs(translation));
792  active = true;
793  }
794  else if (translation <= lowerTranslation)
795  {
796  c2 = MathUtils.Min(translation - lowerTranslation, 0.0f);
797  linearError = MathUtils.Max(linearError, lowerTranslation - translation);
798  active = true;
799  }
800  else if (translation >= upperTranslation)
801  {
802  c2 = MathUtils.Max(translation - upperTranslation, 0.0f);
803  linearError = MathUtils.Max(linearError, translation - upperTranslation);
804  active = true;
805  }
806  }
807 
808  if (active)
809  {
810  float k11 = mA + mB + iA * s1 * s1 + iB * s2 * s2;
811  float k12 = iA * s1 + iB * s2;
812  float k13 = iA * s1 * a1 + iB * s2 * a2;
813  float k22 = iA + iB;
814  if (k22 == 0.0f)
815  {
816  // For fixed rotation
817  k22 = 1.0f;
818  }
819 
820  float k23 = iA * a1 + iB * a2;
821  float k33 = mA + mB + iA * a1 * a1 + iB * a2 * a2;
822 
823  Mat33 k;
824  k.Ex = new Vector3(k11, k12, k13);
825  k.Ey = new Vector3(k12, k22, k23);
826  k.Ez = new Vector3(k13, k23, k33);
827 
828  Vector3 c;
829  c.X = c1.X;
830  c.Y = c1.Y;
831  c.Z = c2;
832 
833  impulse = k.Solve33(-c);
834  }
835  else
836  {
837  float k11 = mA + mB + iA * s1 * s1 + iB * s2 * s2;
838  float k12 = iA * s1 + iB * s2;
839  float k22 = iA + iB;
840  if (k22 == 0.0f)
841  {
842  k22 = 1.0f;
843  }
844 
845  Mat22 k;
846  k.Ex = new Vector2(k11, k12);
847  k.Ey = new Vector2(k12, k22);
848 
849  Vector2 impulse1 = k.Solve(-c1);
850  impulse.X = impulse1.X;
851  impulse.Y = impulse1.Y;
852  impulse.Z = 0.0f;
853  }
854 
855  Vector2 p = impulse.X * perp + impulse.Z * axis;
856  float la = impulse.X * s1 + impulse.Y + impulse.Z * a1;
857  float lb = impulse.X * s2 + impulse.Y + impulse.Z * a2;
858 
859  cA -= mA * p;
860  aA -= iA * la;
861  cB += mB * p;
862  aB += iB * lb;
863 
864  data.Positions[indexA].C = cA;
865  data.Positions[indexA].A = aA;
866  data.Positions[indexB].C = cB;
867  data.Positions[indexB].A = aB;
868 
869  return linearError <= Settings.LinearSlop && angularError <= Settings.AngularSlop;
870  }
