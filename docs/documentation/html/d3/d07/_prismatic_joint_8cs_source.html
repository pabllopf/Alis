40  // Linear constraint (point-to-line)
41  // d = p2 - p1 = x2 + r2 - x1 - r1
42  // C = dot(perp, d)
43  // Cdot = dot(d, cross(w1, perp)) + dot(perp, v2 + cross(w2, r2) - v1 - cross(w1, r1))
44  // = -dot(perp, v1) - dot(cross(d + r1, perp), w1) + dot(perp, v2) + dot(cross(r2, perp), v2)
45  // J = [-perp, -cross(d + r1, perp), perp, cross(r2,perp)]
46  //
47  // Angular constraint
48  // C = a2 - a1 + a_initial
49  // Cdot = w2 - w1
50  // J = [0 0 -1 0 0 1]
51  //
52  // K = J * invM * JT
53  //
54  // J = [-a -s1 a s2]
55  // [0 -1 0 1]
56  // a = perp
57  // s1 = cross(d + r1, a) = cross(p2 - x1, a)
58  // s2 = cross(r2, a) = cross(p2 - x2, a)
59 
60  // Motor/Limit linear constraint
61  // C = dot(ax1, d)
62  // Cdot = -dot(ax1, v1) - dot(cross(d + r1, ax1), w1) + dot(ax1, v2) + dot(cross(r2, ax1), v2)
63  // J = [-ax1 -cross(d+r1,ax1) ax1 cross(r2,ax1)]
64 
65  // Predictive limit is applied even when the limit is not active.
66  // Prevents a constraint speed that can lead to a constraint error in one time step.
67  // Want C2 = C1 + h * Cdot >= 0
68  // Or:
69  // Cdot + C1/h >= 0
70  // I do not apply a negative constraint error because that is handled in position correction.
71  // So:
72  // Cdot + max(C1, 0)/h >= 0
73 
74  // Block Solver
75  // We develop a block solver that includes the angular and linear constraints. This makes the limit stiffer.
76  //
77  // The Jacobian has 2 rows:
78  // J = [-uT -s1 uT s2] // linear
79  // [0 -1 0 1] // angular
80  //
81  // u = perp
82  // s1 = cross(d + r1, u), s2 = cross(r2, u)
83  // a1 = cross(d + r1, v), a2 = cross(r2, v)

555  internal override void InitVelocityConstraints(ref SolverData data)
556  {
557  indexA = BodyA.IslandIndex;
558  indexB = BodyB.IslandIndex;
559  localCenterA = BodyA.Sweep.LocalCenter;
560  localCenterB = BodyB.Sweep.LocalCenter;
561  invMassA = BodyA.InvMass;
562  invMassB = BodyB.InvMass;
563  invIa = BodyA.InvI;
564  invIb = BodyB.InvI;
565 
566  Vector2F cA = data.Positions[indexA].C;
567  float aA = data.Positions[indexA].A;
568  Vector2F vA = data.Velocities[indexA].V;
569  float wA = data.Velocities[indexA].W;
570 
571  Vector2F cB = data.Positions[indexB].C;
572  float aB = data.Positions[indexB].A;
573  Vector2F vB = data.Velocities[indexB].V;
574  float wB = data.Velocities[indexB].W;
575 
576  Rotation qA = new Rotation(aA), qB = new Rotation(aB);
577 
578  // Compute the effective masses.
579  Vector2F rA = MathUtils.Mul(qA, LocalAnchorA - localCenterA);
580  Vector2F rB = MathUtils.Mul(qB, LocalAnchorB - localCenterB);
581  Vector2F d = cB - cA + rB - rA;
582 
583  float mA = invMassA, mB = invMassB;
584  float iA = invIa, iB = invIb;
585 
586  // Compute motor Jacobian and effective mass.
587  {
588  axis = MathUtils.Mul(qA, LocalXAxisA);
589  a1 = MathUtils.Cross(d + rA, axis);
590  a2 = MathUtils.Cross(rB, axis);
591 
592  axialMass = mA + mB + iA * a1 * a1 + iB * a2 * a2;
593  if (axialMass > 0.0f)
594  {
595  axialMass = 1.0f / axialMass;
596  }
597  }
598 
599  // Prismatic constraint.
600  {
601  perp = MathUtils.Mul(qA, localYAxisA);
602 
603  s1 = MathUtils.Cross(d + rA, perp);
604  s2 = MathUtils.Cross(rB, perp);
605 
606  float k11 = mA + mB + iA * s1 * s1 + iB * s2 * s2;
607  float k12 = iA * s1 + iB * s2;
608  float k22 = iA + iB;
609  if (k22 == 0.0f)
610  {
611  // For bodies with fixed rotation.
612  k22 = 1.0f;
613  }
614 
615  k.Ex = new Vector2F(k11, k12);
616  k.Ey = new Vector2F(k12, k22);
617  }
618 
619  if (enableLimit)
620  {
621  translation = Vector2F.Dot(axis, d);
622  }
623  else
624  {
625  lowerImpulse = 0.0f;
626  upperImpulse = 0.0f;
627  }
628 
629  if (!enableMotor)
630  {
631  motorImpulse = 0.0f;
632  }
633 
634  if (data.Step.WarmStarting)
635  {
636  // Account for variable time step.
637  impulse *= data.Step.DeltaTimeRatio;
638  motorImpulse *= data.Step.DeltaTimeRatio;
639  lowerImpulse *= data.Step.DeltaTimeRatio;
640  upperImpulse *= data.Step.DeltaTimeRatio;
641 
642  float axialImpulse = motorImpulse + lowerImpulse - upperImpulse;
643  Vector2F p = impulse.X * perp + axialImpulse * axis;
644  float la = impulse.X * s1 + impulse.Y + axialImpulse * a1;
645  float lb = impulse.X * s2 + impulse.Y + axialImpulse * a2;
646 
647  vA -= mA * p;
648  wA -= iA * la;
649 
650  vB += mB * p;
651  wB += iB * lb;
652  }
653  else
654  {
655  impulse = Vector2F.Zero;
656  motorImpulse = 0.0f;
657  lowerImpulse = 0.0f;
658  upperImpulse = 0.0f;
659  }
660 
661  data.Velocities[indexA].V = vA;
662  data.Velocities[indexA].W = wA;
663  data.Velocities[indexB].V = vB;
664  data.Velocities[indexB].W = wB;
665  }

671  internal override void SolveVelocityConstraints(ref SolverData data)
672  {
673  Vector2F vA = data.Velocities[indexA].V;
674  float wA = data.Velocities[indexA].W;
675  Vector2F vB = data.Velocities[indexB].V;
676  float wB = data.Velocities[indexB].W;
677 
678  float mA = invMassA, mB = invMassB;
679  float iA = invIa, iB = invIb;
680 
681  // Solve linear motor constraint.
682  if (enableMotor)
683  {
684  float cdot = Vector2F.Dot(axis, vB - vA) + a2 * wB - a1 * wA;
685  float impulse = axialMass * (motorSpeed - cdot);
686  float oldImpulse = motorImpulse;
687  float maxImpulse = data.Step.DeltaTime * maxMotorForce;
688  motorImpulse = MathUtils.Clamp(motorImpulse + impulse, -maxImpulse, maxImpulse);
689  impulse = motorImpulse - oldImpulse;
690 
691  Vector2F p = impulse * axis;
692  float la = impulse * a1;
693  float lb = impulse * a2;
694 
695  vA -= mA * p;
696  wA -= iA * la;
697  vB += mB * p;
698  wB += iB * lb;
699  }
700 
701  if (enableLimit)
702  {
703  // Lower limit
704  {
705  float c = translation - lowerTranslation;
706  float cdot = MathUtils.Dot(axis, vB - vA) + a2 * wB - a1 * wA;
707  float impulse = -axialMass * (cdot + MathUtils.Max(c, 0.0f) * data.Step.InvertedDeltaTime);
708  float oldImpulse = lowerImpulse;
709  lowerImpulse = MathUtils.Max(lowerImpulse + impulse, 0.0f);
710  impulse = lowerImpulse - oldImpulse;
711 
712  Vector2F p = impulse * axis;
713  float la = impulse * a1;
714  float lb = impulse * a2;
715 
716  vA -= mA * p;
717  wA -= iA * la;
718  vB += mB * p;
719  wB += iB * lb;
720  }
721 
722  // Upper limit
723  // Note: signs are flipped to keep C positive when the constraint is satisfied.
724  // This also keeps the impulse positive when the limit is active.
725  {
726  float c = upperTranslation - translation;
727  float cdot = MathUtils.Dot(axis, vA - vB) + a1 * wA - a2 * wB;
728  float impulse = -axialMass * (cdot + MathUtils.Max(c, 0.0f) * data.Step.InvertedDeltaTime);
729  float oldImpulse = upperImpulse;
730  upperImpulse = MathUtils.Max(upperImpulse + impulse, 0.0f);
731  impulse = upperImpulse - oldImpulse;
732 
733  Vector2F p = impulse * axis;
734  float la = impulse * a1;
735  float lb = impulse * a2;
736 
737  vA += mA * p;
738  wA += iA * la;
739  vB -= mB * p;
740  wB -= iB * lb;
741  }
742  }
743 
744  // Solve the prismatic constraint in block form.
745  {
746  Vector2F cdot;
747  cdot.X = MathUtils.Dot(perp, vB - vA) + s2 * wB - s1 * wA;
748  cdot.Y = wB - wA;
749 
750  Vector2F df = k.Solve(-cdot);
751  impulse += df;
752 
753  Vector2F p = df.X * perp;
754  float la = df.X * s1 + df.Y;
755  float lb = df.X * s2 + df.Y;
756 
757  vA -= mA * p;
758  wA -= iA * la;
759 
760  vB += mB * p;
761  wB += iB * lb;
762  }
763 
764  data.Velocities[indexA].V = vA;
765  data.Velocities[indexA].W = wA;
766  data.Velocities[indexB].V = vB;
767  data.Velocities[indexB].W = wB;
768  }

770  // A velocity based solver computes reaction forces(impulses) using the velocity constraint solver. Under this context,
771  // the position solver is not there to resolve forces. It is only there to cope with integration error.
772  //
773  // Therefore, the pseudo impulses in the position solver do not have any physical meaning. Thus it is okay if they suck.
774  //
775  // We could take the active state from the velocity solver. However, the joint might push past the limit when the velocity
776  // solver indicates the limit is inactive.
782  internal override bool SolvePositionConstraints(ref SolverData data)
783  {
784  Vector2F cA = data.Positions[indexA].C;
785  float aA = data.Positions[indexA].A;
786  Vector2F cB = data.Positions[indexB].C;
787  float aB = data.Positions[indexB].A;
788 
789  Rotation qA = new Rotation(aA), qB = new Rotation(aB);
790 
791  float mA = invMassA, mB = invMassB;
792  float iA = invIa, iB = invIb;
793 
794  // Compute fresh Jacobians
795  Vector2F rA = MathUtils.Mul(qA, LocalAnchorA - localCenterA);
796  Vector2F rB = MathUtils.Mul(qB, LocalAnchorB - localCenterB);
797  Vector2F d = cB + rB - cA - rA;
798 
799  Vector2F axis = MathUtils.Mul(qA, LocalXAxisA);
800  float a1 = MathUtils.Cross(d + rA, axis);
801  float a2 = MathUtils.Cross(rB, axis);
802  Vector2F perp = MathUtils.Mul(qA, localYAxisA);
803 
804  float s1 = MathUtils.Cross(d + rA, perp);
805  float s2 = MathUtils.Cross(rB, perp);
806 
807  Vector3F impulse;
808  Vector2F c1;
809  c1.X = Vector2F.Dot(perp, d);
810  c1.Y = aB - aA - ReferenceAngle;
811 
812  float linearError = MathUtils.Abs(c1.X);
813  float angularError = MathUtils.Abs(c1.Y);
814 
815  bool active = false;
816  float c2 = 0.0f;
817  if (enableLimit)
818  {
819  float translation = MathUtils.Dot(axis, d);
820  if (MathUtils.Abs(upperTranslation - lowerTranslation) < 2.0f * Settings.LinearSlop)
821  {
822  c2 = translation;
823  linearError = MathUtils.Max(linearError, MathUtils.Abs(translation));
824  active = true;
825  }
826  else if (translation <= lowerTranslation)
827  {
828  c2 = MathUtils.Min(translation - lowerTranslation, 0.0f);
829  linearError = MathUtils.Max(linearError, lowerTranslation - translation);
830  active = true;
831  }
832  else if (translation >= upperTranslation)
833  {
834  c2 = MathUtils.Max(translation - upperTranslation, 0.0f);
835  linearError = MathUtils.Max(linearError, translation - upperTranslation);
836  active = true;
837  }
838  }
839 
840  if (active)
841  {
842  float k11 = mA + mB + iA * s1 * s1 + iB * s2 * s2;
843  float k12 = iA * s1 + iB * s2;
844  float k13 = iA * s1 * a1 + iB * s2 * a2;
845  float k22 = iA + iB;
846  if (k22 == 0.0f)
847  {
848  // For fixed rotation
849  k22 = 1.0f;
850  }
851 
852  float k23 = iA * a1 + iB * a2;
853  float k33 = mA + mB + iA * a1 * a1 + iB * a2 * a2;
854 
855  Matrix3X3F k;
856  k.Ex = new Vector3F(k11, k12, k13);
857  k.Ey = new Vector3F(k12, k22, k23);
858  k.Ez = new Vector3F(k13, k23, k33);
859 
860  Vector3F c;
861  c.X = c1.X;
862  c.Y = c1.Y;
863  c.Z = c2;
864 
865  impulse = k.Solve33(-c);
866  }
867  else
868  {
869  float k11 = mA + mB + iA * s1 * s1 + iB * s2 * s2;
870  float k12 = iA * s1 + iB * s2;
871  float k22 = iA + iB;
872  if (k22 == 0.0f)
873  {
874  k22 = 1.0f;
875  }
876 
877  Matrix2X2F k;
878  k.Ex = new Vector2F(k11, k12);
879  k.Ey = new Vector2F(k12, k22);
880 
881  Vector2F impulse1 = k.Solve(-c1);
882  impulse.X = impulse1.X;
883  impulse.Y = impulse1.Y;
884  impulse.Z = 0.0f;
885  }
