43  // Linear constraint (point-to-line)
44  // d = p2 - p1 = x2 + r2 - x1 - r1
45  // C = dot(perp, d)
46  // Cdot = dot(d, cross(w1, perp)) + dot(perp, v2 + cross(w2, r2) - v1 - cross(w1, r1))
47  // = -dot(perp, v1) - dot(cross(d + r1, perp), w1) + dot(perp, v2) + dot(cross(r2, perp), v2)
48  // J = [-perp, -cross(d + r1, perp), perp, cross(r2,perp)]
49  //
50  // Angular constraint
51  // C = a2 - a1 + a_initial
52  // Cdot = w2 - w1
53  // J = [0 0 -1 0 0 1]
54  //
55  // K = J * invM * JT
56  //
57  // J = [-a -s1 a s2]
58  // [0 -1 0 1]
59  // a = perp
60  // s1 = cross(d + r1, a) = cross(p2 - x1, a)
61  // s2 = cross(r2, a) = cross(p2 - x2, a)
62 
63  // Motor/Limit linear constraint
64  // C = dot(ax1, d)
65  // Cdot = -dot(ax1, v1) - dot(cross(d + r1, ax1), w1) + dot(ax1, v2) + dot(cross(r2, ax1), v2)
66  // J = [-ax1 -cross(d+r1,ax1) ax1 cross(r2,ax1)]
67 
68  // Predictive limit is applied even when the limit is not active.
69  // Prevents a constraint speed that can lead to a constraint error in one time step.
70  // Want C2 = C1 + h * Cdot >= 0
71  // Or:
72  // Cdot + C1/h >= 0
73  // I do not apply a negative constraint error because that is handled in position correction.
74  // So:
75  // Cdot + max(C1, 0)/h >= 0
76 
77  // Block Solver
78  // We develop a block solver that includes the angular and linear constraints. This makes the limit stiffer.
79  //
80  // The Jacobian has 2 rows:
81  // J = [-uT -s1 uT s2] // linear
82  // [0 -1 0 1] // angular
83  //
84  // u = perp
85  // s1 = cross(d + r1, u), s2 = cross(r2, u)
86  // a1 = cross(d + r1, v), a2 = cross(r2, v)
87

525  internal override void InitVelocityConstraints(ref SolverData data)
526  {
527  indexA = BodyA.IslandIndex;
528  indexB = BodyB.IslandIndex;
529  localCenterA = BodyA.Sweep.LocalCenter;
530  localCenterB = BodyB.Sweep.LocalCenter;
531  invMassA = BodyA.InvMass;
532  invMassB = BodyB.InvMass;
533  invIa = BodyA.InvI;
534  invIb = BodyB.InvI;
535 
536  Vector2 cA = data.Positions[indexA].C;
537  float aA = data.Positions[indexA].A;
538  Vector2 vA = data.Velocities[indexA].V;
539  float wA = data.Velocities[indexA].W;
540 
541  Vector2 cB = data.Positions[indexB].C;
542  float aB = data.Positions[indexB].A;
543  Vector2 vB = data.Velocities[indexB].V;
544  float wB = data.Velocities[indexB].W;
545 
546  Rot qA = new Rot(aA), qB = new Rot(aB);
547 
548  // Compute the effective masses.
549  Vector2 rA = MathUtils.Mul(qA, LocalAnchorA - localCenterA);
550  Vector2 rB = MathUtils.Mul(qB, LocalAnchorB - localCenterB);
551  Vector2 d = cB - cA + rB - rA;
552 
553  float mA = invMassA, mB = invMassB;
554  float iA = invIa, iB = invIb;
555 
556  // Compute motor Jacobian and effective mass.
557  {
558  axis = MathUtils.Mul(qA, LocalXAxisA);
559  a1 = MathUtils.Cross(d + rA, axis);
560  a2 = MathUtils.Cross(rB, axis);
561 
562  axialMass = mA + mB + iA * a1 * a1 + iB * a2 * a2;
563  if (axialMass > 0.0f)
564  {
565  axialMass = 1.0f / axialMass;
566  }
567  }
568 
569  // Prismatic constraint.
570  {
571  perp = MathUtils.Mul(qA, localYAxisA);
572 
573  s1 = MathUtils.Cross(d + rA, perp);
574  s2 = MathUtils.Cross(rB, perp);
575 
576  float k11 = mA + mB + iA * s1 * s1 + iB * s2 * s2;
577  float k12 = iA * s1 + iB * s2;
578  float k22 = iA + iB;
579  if (k22 == 0.0f)
580  {
581  // For bodies with fixed rotation.
582  k22 = 1.0f;
583  }
584 
585  k.Ex = new Vector2(k11, k12);
586  k.Ey = new Vector2(k12, k22);
587  }
588 
589  if (enableLimit)
590  {
591  translation = Vector2.Dot(axis, d);
592  }
593  else
594  {
595  lowerImpulse = 0.0f;
596  upperImpulse = 0.0f;
597  }
598 
599  if (!enableMotor)
600  {
601  motorImpulse = 0.0f;
602  }
603 
604  if (data.Step.WarmStarting)
605  {
606  // Account for variable time step.
607  impulse *= data.Step.DeltaTimeRatio;
608  motorImpulse *= data.Step.DeltaTimeRatio;
609  lowerImpulse *= data.Step.DeltaTimeRatio;
610  upperImpulse *= data.Step.DeltaTimeRatio;
611 
612  float axialImpulse = motorImpulse + lowerImpulse - upperImpulse;
613  Vector2 p = impulse.X * perp + axialImpulse * axis;
614  float la = impulse.X * s1 + impulse.Y + axialImpulse * a1;
615  float lb = impulse.X * s2 + impulse.Y + axialImpulse * a2;
616 
617  vA -= mA * p;
618  wA -= iA * la;
619 
620  vB += mB * p;
621  wB += iB * lb;
622  }
623  else
624  {
625  impulse = Vector2.Zero;
626  motorImpulse = 0.0f;
627  lowerImpulse = 0.0f;
628  upperImpulse = 0.0f;
629  }
630 
631  data.Velocities[indexA].V = vA;
632  data.Velocities[indexA].W = wA;
633  data.Velocities[indexB].V = vB;
634  data.Velocities[indexB].W = wB;
635  }

641  internal override void SolveVelocityConstraints(ref SolverData data)
642  {
643  Vector2 vA = data.Velocities[indexA].V;
644  float wA = data.Velocities[indexA].W;
645  Vector2 vB = data.Velocities[indexB].V;
646  float wB = data.Velocities[indexB].W;
647 
648  float mA = invMassA, mB = invMassB;
649  float iA = invIa, iB = invIb;
650 
651  // Solve linear motor constraint.
652  if (enableMotor)
653  {
654  float cdot = Vector2.Dot(axis, vB - vA) + a2 * wB - a1 * wA;
655  float impulse = axialMass * (motorSpeed - cdot);
656  float oldImpulse = motorImpulse;
657  float maxImpulse = data.Step.DeltaTime * maxMotorForce;
658  motorImpulse = MathUtils.Clamp(motorImpulse + impulse, -maxImpulse, maxImpulse);
659  impulse = motorImpulse - oldImpulse;
660 
661  Vector2 p = impulse * axis;
662  float la = impulse * a1;
663  float lb = impulse * a2;
664 
665  vA -= mA * p;
666  wA -= iA * la;
667  vB += mB * p;
668  wB += iB * lb;
669  }
670 
671  if (enableLimit)
672  {
673  // Lower limit
674  {
675  float c = translation - lowerTranslation;
676  float cdot = MathUtils.Dot(axis, vB - vA) + a2 * wB - a1 * wA;
677  float impulse = -axialMass * (cdot + MathUtils.Max(c, 0.0f) * data.Step.InvertedDeltaTime);
678  float oldImpulse = lowerImpulse;
679  lowerImpulse = MathUtils.Max(lowerImpulse + impulse, 0.0f);
680  impulse = lowerImpulse - oldImpulse;
681 
682  Vector2 p = impulse * axis;
683  float la = impulse * a1;
684  float lb = impulse * a2;
685 
686  vA -= mA * p;
687  wA -= iA * la;
688  vB += mB * p;
689  wB += iB * lb;
690  }
691 
692  // Upper limit
693  // Note: signs are flipped to keep C positive when the constraint is satisfied.
694  // This also keeps the impulse positive when the limit is active.
695  {
696  float c = upperTranslation - translation;
697  float cdot = MathUtils.Dot(axis, vA - vB) + a1 * wA - a2 * wB;
698  float impulse = -axialMass * (cdot + MathUtils.Max(c, 0.0f) * data.Step.InvertedDeltaTime);
699  float oldImpulse = upperImpulse;
700  upperImpulse = MathUtils.Max(upperImpulse + impulse, 0.0f);
701  impulse = upperImpulse - oldImpulse;
702 
703  Vector2 p = impulse * axis;
704  float la = impulse * a1;
705  float lb = impulse * a2;
706 
707  vA += mA * p;
708  wA += iA * la;
709  vB -= mB * p;
710  wB -= iB * lb;
711  }
712  }
713 
714  // Solve the prismatic constraint in block form.
715  {
716  Vector2 cdot;
717  cdot.X = MathUtils.Dot(perp, vB - vA) + s2 * wB - s1 * wA;
718  cdot.Y = wB - wA;
719 
720  Vector2 df = k.Solve(-cdot);
721  impulse += df;
722 
723  Vector2 p = df.X * perp;
724  float la = df.X * s1 + df.Y;
725  float lb = df.X * s2 + df.Y;
726 
727  vA -= mA * p;
728  wA -= iA * la;
729 
730  vB += mB * p;
731  wB += iB * lb;
732  }
733 
734  data.Velocities[indexA].V = vA;
735  data.Velocities[indexA].W = wA;
736  data.Velocities[indexB].V = vB;
737  data.Velocities[indexB].W = wB;
738  }

740  // A velocity based solver computes reaction forces(impulses) using the velocity constraint solver. Under this context,
741  // the position solver is not there to resolve forces. It is only there to cope with integration error.
742  //
743  // Therefore, the pseudo impulses in the position solver do not have any physical meaning. Thus it is okay if they suck.
744  //
745  // We could take the active state from the velocity solver. However, the joint might push past the limit when the velocity
746  // solver indicates the limit is inactive.
752  internal override bool SolvePositionConstraints(ref SolverData data)
753  {
754  Vector2 cA = data.Positions[indexA].C;
755  float aA = data.Positions[indexA].A;
756  Vector2 cB = data.Positions[indexB].C;
757  float aB = data.Positions[indexB].A;
758 
759  Rot qA = new Rot(aA), qB = new Rot(aB);
760 
761  float mA = invMassA, mB = invMassB;
762  float iA = invIa, iB = invIb;
763 
764  // Compute fresh Jacobians
765  Vector2 rA = MathUtils.Mul(qA, LocalAnchorA - localCenterA);
766  Vector2 rB = MathUtils.Mul(qB, LocalAnchorB - localCenterB);
767  Vector2 d = cB + rB - cA - rA;
768 
769  Vector2 axis = MathUtils.Mul(qA, LocalXAxisA);
770  float a1 = MathUtils.Cross(d + rA, axis);
771  float a2 = MathUtils.Cross(rB, axis);
772  Vector2 perp = MathUtils.Mul(qA, localYAxisA);
773 
774  float s1 = MathUtils.Cross(d + rA, perp);
775  float s2 = MathUtils.Cross(rB, perp);
776 
777  Vector3 impulse;
778  Vector2 c1;
779  c1.X = Vector2.Dot(perp, d);
780  c1.Y = aB - aA - ReferenceAngle;
781 
782  float linearError = MathUtils.Abs(c1.X);
783  float angularError = MathUtils.Abs(c1.Y);
784 
785  bool active = false;
786  float c2 = 0.0f;
787  if (enableLimit)
788  {
789  float translation = MathUtils.Dot(axis, d);
790  if (MathUtils.Abs(upperTranslation - lowerTranslation) < 2.0f * Settings.LinearSlop)
791  {
792  c2 = translation;
793  linearError = MathUtils.Max(linearError, MathUtils.Abs(translation));
794  active = true;
795  }
796  else if (translation <= lowerTranslation)
797  {
798  c2 = MathUtils.Min(translation - lowerTranslation, 0.0f);
799  linearError = MathUtils.Max(linearError, lowerTranslation - translation);
800  active = true;
801  }
802  else if (translation >= upperTranslation)
803  {
804  c2 = MathUtils.Max(translation - upperTranslation, 0.0f);
805  linearError = MathUtils.Max(linearError, translation - upperTranslation);
806  active = true;
807  }
808  }
809 
810  if (active)
811  {
812  float k11 = mA + mB + iA * s1 * s1 + iB * s2 * s2;
813  float k12 = iA * s1 + iB * s2;
814  float k13 = iA * s1 * a1 + iB * s2 * a2;
815  float k22 = iA + iB;
816  if (k22 == 0.0f)
817  {
818  // For fixed rotation
819  k22 = 1.0f;
820  }
821 
822  float k23 = iA * a1 + iB * a2;
823  float k33 = mA + mB + iA * a1 * a1 + iB * a2 * a2;
824 
825  Mat33 k;
826  k.Ex = new Vector3(k11, k12, k13);
827  k.Ey = new Vector3(k12, k22, k23);
828  k.Ez = new Vector3(k13, k23, k33);
829 
830  Vector3 c;
831  c.X = c1.X;
832  c.Y = c1.Y;
833  c.Z = c2;
834 
835  impulse = k.Solve33(-c);
836  }
837  else
838  {
839  float k11 = mA + mB + iA * s1 * s1 + iB * s2 * s2;
840  float k12 = iA * s1 + iB * s2;
841  float k22 = iA + iB;
842  if (k22 == 0.0f)
843  {
844  k22 = 1.0f;
845  }
846 
847  Mat22 k;
848  k.Ex = new Vector2(k11, k12);
849  k.Ey = new Vector2(k12, k22);
850 
851  Vector2 impulse1 = k.Solve(-c1);
852  impulse.X = impulse1.X;
853  impulse.Y = impulse1.Y;
854  impulse.Z = 0.0f;
855  }
856 
857  Vector2 p = impulse.X * perp + impulse.Z * axis;
858  float la = impulse.X * s1 + impulse.Y + impulse.Z * a1;
859  float lb = impulse.X * s2 + impulse.Y + impulse.Z * a2;
860 
861  cA -= mA * p;
862  aA -= iA * la;
863  cB += mB * p;
864  aB += iB * lb;
865 
866  data.Positions[indexA].C = cA;
867  data.Positions[indexA].A = aA;
868  data.Positions[indexB].C = cB;
869  data.Positions[indexB].A = aB;
870 
871  return linearError <= Settings.LinearSlop && angularError <= Settings.AngularSlop;
872  }
