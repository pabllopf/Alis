509  internal override void InitVelocityConstraints(ref SolverData data)
510  {
511  indexA = BodyA.IslandIndex;
512  indexB = BodyB.IslandIndex;
513  localCenterA = BodyA.Sweep.LocalCenter;
514  localCenterB = BodyB.Sweep.LocalCenter;
515  invMassA = BodyA.InvMass;
516  invMassB = BodyB.InvMass;
517  invIa = BodyA.InvI;
518  invIb = BodyB.InvI;
519 
520  Vector2 cA = data.Positions[indexA].C;
521  float aA = data.Positions[indexA].A;
522  Vector2 vA = data.Velocities[indexA].V;
523  float wA = data.Velocities[indexA].W;
524 
525  Vector2 cB = data.Positions[indexB].C;
526  float aB = data.Positions[indexB].A;
527  Vector2 vB = data.Velocities[indexB].V;
528  float wB = data.Velocities[indexB].W;
529 
530  Rotation qA = new Rotation(aA), qB = new Rotation(aB);
531 
532  Vector2 rA = MathUtils.Mul(qA, LocalAnchorA - localCenterA);
533  Vector2 rB = MathUtils.Mul(qB, LocalAnchorB - localCenterB);
534  Vector2 d = cB - cA + rB - rA;
535 
536  float mA = invMassA, mB = invMassB;
537  float iA = invIa, iB = invIb;
538 
539  {
540  axis = MathUtils.Mul(qA, LocalXAxisA);
541  a1 = MathUtils.Cross(d + rA, axis);
542  a2 = MathUtils.Cross(rB, axis);
543 
544  axialMass = mA + mB + iA * a1 * a1 + iB * a2 * a2;
545  if (axialMass > 0.0f)
546  {
547  axialMass = 1.0f / axialMass;
548  }
549  }
550 
551  {
552  perp = MathUtils.Mul(qA, localYAxisA);
553 
554  s1 = MathUtils.Cross(d + rA, perp);
555  s2 = MathUtils.Cross(rB, perp);
556 
557  float k11 = mA + mB + iA * s1 * s1 + iB * s2 * s2;
558  float k12 = iA * s1 + iB * s2;
559  float k22 = iA + iB;
560  if (k22 == 0.0f)
561  {
562  k22 = 1.0f;
563  }
564 
565  k.Ex = new Vector2(k11, k12);
566  k.Ey = new Vector2(k12, k22);
567  }
568 
569  if (enableLimit)
570  {
571  translation = Vector2.Dot(axis, d);
572  }
573  else
574  {
575  lowerImpulse = 0.0f;
576  upperImpulse = 0.0f;
577  }
578 
579  if (!enableMotor)
580  {
581  motorImpulse = 0.0f;
582  }
583 
584  if (data.Step.WarmStarting)
585  {
586  impulse *= data.Step.DeltaTimeRatio;
587  motorImpulse *= data.Step.DeltaTimeRatio;
588  lowerImpulse *= data.Step.DeltaTimeRatio;
589  upperImpulse *= data.Step.DeltaTimeRatio;
590 
591  float axialImpulse = motorImpulse + lowerImpulse - upperImpulse;
592  Vector2 p = impulse.X * perp + axialImpulse * axis;
593  float la = impulse.X * s1 + impulse.Y + axialImpulse * a1;
594  float lb = impulse.X * s2 + impulse.Y + axialImpulse * a2;
595 
596  vA -= mA * p;
597  wA -= iA * la;
598 
599  vB += mB * p;
600  wB += iB * lb;
601  }
602  else
603  {
604  impulse = Vector2.Zero;
605  motorImpulse = 0.0f;
606  lowerImpulse = 0.0f;
607  upperImpulse = 0.0f;
608  }
609 
610  data.Velocities[indexA].V = vA;
611  data.Velocities[indexA].W = wA;
612  data.Velocities[indexB].V = vB;
613  data.Velocities[indexB].W = wB;
614  }

620  internal override void SolveVelocityConstraints(ref SolverData data)
621  {
622  Vector2 vA = data.Velocities[indexA].V;
623  float wA = data.Velocities[indexA].W;
624  Vector2 vB = data.Velocities[indexB].V;
625  float wB = data.Velocities[indexB].W;
626 
627  float mA = invMassA, mB = invMassB;
628  float iA = invIa, iB = invIb;
629 
630  if (enableMotor)
631  {
632  float dot = Vector2.Dot(axis, vB - vA) + a2 * wB - a1 * wA;
633  float impulseLocal = axialMass * (motorSpeed - dot);
634  float oldImpulse = motorImpulse;
635  float maxImpulse = data.Step.DeltaTime * maxMotorForce;
636  motorImpulse = MathUtils.Clamp(motorImpulse + impulseLocal, -maxImpulse, maxImpulse);
637  impulseLocal = motorImpulse - oldImpulse;
638 
639  Vector2 p = impulseLocal * axis;
640  float la = impulseLocal * a1;
641  float lb = impulseLocal * a2;
642 
643  vA -= mA * p;
644  wA -= iA * la;
645  vB += mB * p;
646  wB += iB * lb;
647  }
648 
649  if (enableLimit)
650  {
651  {
652  float c = translation - lowerTranslation;
653  float dot = MathUtils.Dot(axis, vB - vA) + a2 * wB - a1 * wA;
654  float impulseLocal = -axialMass * (dot + MathUtils.Max(c, 0.0f) * data.Step.InvertedDeltaTime);
655  float oldImpulse = lowerImpulse;
656  lowerImpulse = MathUtils.Max(lowerImpulse + impulseLocal, 0.0f);
657  impulseLocal = lowerImpulse - oldImpulse;
658 
659  Vector2 p = impulseLocal * axis;
660  float la = impulseLocal * a1;
661  float lb = impulseLocal * a2;
662 
663  vA -= mA * p;
664  wA -= iA * la;
665  vB += mB * p;
666  wB += iB * lb;
667  }
668 
669  {
670  float c = upperTranslation - translation;
671  float dot = MathUtils.Dot(axis, vA - vB) + a1 * wA - a2 * wB;
672  float impulseLocal = -axialMass * (dot + MathUtils.Max(c, 0.0f) * data.Step.InvertedDeltaTime);
673  float oldImpulse = upperImpulse;
674  upperImpulse = MathUtils.Max(upperImpulse + impulseLocal, 0.0f);
675  impulseLocal = upperImpulse - oldImpulse;
676 
677  Vector2 p = impulseLocal * axis;
678  float la = impulseLocal * a1;
679  float lb = impulseLocal * a2;
680 
681  vA += mA * p;
682  wA += iA * la;
683  vB -= mB * p;
684  wB -= iB * lb;
685  }
686  }
687 
688  {
689  Vector2 dot = new Vector2(
690  MathUtils.Dot(perp, vB - vA) + s2 * wB - s1 * wA,
691  wB - wA
692  );
693 
694  Vector2 df = k.Solve(-dot);
695  impulse += df;
696 
697  Vector2 p = df.X * perp;
698  float la = df.X * s1 + df.Y;
699  float lb = df.X * s2 + df.Y;
700 
701  vA -= mA * p;
702  wA -= iA * la;
703 
704  vB += mB * p;
705  wB += iB * lb;
706  }
707 
708  data.Velocities[indexA].V = vA;
709  data.Velocities[indexA].W = wA;
710  data.Velocities[indexB].V = vB;
711  data.Velocities[indexB].W = wB;
712  }

728  internal override bool SolvePositionConstraints(ref SolverData data)
729  {
730  Vector2 cA = data.Positions[indexA].C;
731  float aA = data.Positions[indexA].A;
732  Vector2 cB = data.Positions[indexB].C;
733  float aB = data.Positions[indexB].A;
734 
735  Rotation qA = new Rotation(aA), qB = new Rotation(aB);
736 
737  float mA = invMassA, mB = invMassB;
738  float iA = invIa, iB = invIb;
739 
740  Vector2 rA = MathUtils.Mul(qA, LocalAnchorA - localCenterA);
741  Vector2 rB = MathUtils.Mul(qB, LocalAnchorB - localCenterB);
742  Vector2 d = cB + rB - cA - rA;
743 
744  Vector2 axisLocal = MathUtils.Mul(qA, LocalXAxisA);
745  float a1Local = MathUtils.Cross(d + rA, axisLocal);
746  float a2Local = MathUtils.Cross(rB, axisLocal);
747  Vector2 perpLocal = MathUtils.Mul(qA, localYAxisA);
748 
749  float s1Local = MathUtils.Cross(d + rA, perpLocal);
750  float s2Local = MathUtils.Cross(rB, perpLocal);
751 
752  Vector3 impulseLocal;
753  Vector2 c1 = new Vector2(
754  Vector2.Dot(perpLocal, d),
755  aB - aA - ReferenceAngle
756  );
757 
758  float linearError = MathUtils.Abs(c1.X);
759  float angularError = MathUtils.Abs(c1.Y);
760 
761  bool active = false;
762  float c2 = 0.0f;
763  if (enableLimit)
764  {
765  float translationLocal = MathUtils.Dot(axisLocal, d);
766  if (MathUtils.Abs(upperTranslation - lowerTranslation) < 2.0f * Settings.LinearSlop)
767  {
768  c2 = translationLocal;
769  linearError = MathUtils.Max(linearError, MathUtils.Abs(translationLocal));
770  active = true;
771  }
772  else if (translationLocal <= lowerTranslation)
773  {
774  c2 = MathUtils.Min(translationLocal - lowerTranslation, 0.0f);
775  linearError = MathUtils.Max(linearError, lowerTranslation - translationLocal);
776  active = true;
777  }
778  else if (translationLocal >= upperTranslation)
779  {
780  c2 = MathUtils.Max(translationLocal - upperTranslation, 0.0f);
781  linearError = MathUtils.Max(linearError, translationLocal - upperTranslation);
782  active = true;
783  }
784  }
785 
786  if (active)
787  {
788  float k11 = mA + mB + iA * s1Local * s1Local + iB * s2Local * s2Local;
789  float k12 = iA * s1Local + iB * s2Local;
790  float k13 = iA * s1Local * a1Local + iB * s2Local * a2Local;
791  float k22 = iA + iB;
792  if (k22 == 0.0f)
793  {
794  k22 = 1.0f;
795  }
796 
797  float k23 = iA * a1Local + iB * a2Local;
798  float k33 = mA + mB + iA * a1Local * a1Local + iB * a2Local * a2Local;
799 
800  Matrix3X3F matrix3X3F = new Matrix3X3F(
801  new Vector3(k11, k12, k13),
802  new Vector3(k12, k22, k23),
803  new Vector3(k13, k23, k33)
804  )
805  {
806  Ex = new Vector3(k11, k12, k13),
807  Ey = new Vector3(k12, k22, k23),
808  Ez = new Vector3(k13, k23, k33)
809  };
810 
811  Vector3 c = new Vector3(
812  c1.X,
813  c1.Y,
814  c2
815  );
816 
817  impulseLocal = matrix3X3F.Solve33(-c);
818  }
819  else
820  {
821  float k11 = mA + mB + iA * s1Local * s1Local + iB * s2Local * s2Local;
822  float k12 = iA * s1Local + iB * s2Local;
823  float k22 = iA + iB;
824  if (k22 == 0.0f)
825  {
826  k22 = 1.0f;
827  }
828 
829  Matrix2X2F matrix2X2F = new Matrix2X2F
830  {
831  Ex = new Vector2(k11, k12),
832  Ey = new Vector2(k12, k22)
833  };
834 
835  Vector2 impulse1 = matrix2X2F.Solve(-c1);
836  impulseLocal = new Vector3(
837  impulse1.X,
838  impulse1.Y,
839  0.0f
840  );
841  }
842 
843  Vector2 p = impulseLocal.X * perpLocal + impulseLocal.Z * axisLocal;
844  float la = impulseLocal.X * s1Local + impulseLocal.Y + impulseLocal.Z * a1Local;
845  float lb = impulseLocal.X * s2Local + impulseLocal.Y + impulseLocal.Z * a2Local;
846 
847  cA -= mA * p;
848  aA -= iA * la;
849  cB += mB * p;
850  aB += iB * lb;
851 
852  data.Positions[indexA].C = cA;
853  data.Positions[indexA].A = aA;
854  data.Positions[indexB].C = cB;
855  data.Positions[indexB].A = aB;
856 
857  return (linearError <= Settings.LinearSlop) && (angularError <= Settings.AngularSlop);
858  }
