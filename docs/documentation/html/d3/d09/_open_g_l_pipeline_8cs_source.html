305  private void ProcessResourceSetLayouts(ResourceLayout[] layouts)
306  {
307  int resourceLayoutCount = layouts.Length;
308  _setInfos = new SetBindingsInfo[resourceLayoutCount];
309  int lastTextureLocation = -1;
310  int relativeTextureIndex = -1;
311  int relativeImageIndex = -1;
312  uint storageBlockIndex = 0; // Tracks OpenGL ES storage buffers.
313  for (uint setSlot = 0; setSlot < resourceLayoutCount; setSlot++)
314  {
315  ResourceLayout setLayout = layouts[setSlot];
316  OpenGLResourceLayout glSetLayout = Util.AssertSubtype<ResourceLayout, OpenGLResourceLayout>(setLayout);
317  ResourceLayoutElementDescription[] resources = glSetLayout.Elements;
318 
319  Dictionary<uint, OpenGLUniformBinding> uniformBindings = new Dictionary<uint, OpenGLUniformBinding>();
320  Dictionary<uint, OpenGLTextureBindingSlotInfo> textureBindings = new Dictionary<uint, OpenGLTextureBindingSlotInfo>();
321  Dictionary<uint, OpenGLSamplerBindingSlotInfo> samplerBindings = new Dictionary<uint, OpenGLSamplerBindingSlotInfo>();
322  Dictionary<uint, OpenGLShaderStorageBinding> storageBufferBindings = new Dictionary<uint, OpenGLShaderStorageBinding>();
323 
324  List<int> samplerTrackedRelativeTextureIndices = new List<int>();
325  for (uint i = 0; i < resources.Length; i++)
326  {
327  ResourceLayoutElementDescription resource = resources[i];
328  if (resource.Kind == ResourceKind.UniformBuffer)
329  {
330  uint blockIndex = GetUniformBlockIndex(resource.Name);
331  if (blockIndex != GL_INVALID_INDEX)
332  {
333  int blockSize;
334  glGetActiveUniformBlockiv(_program, blockIndex, ActiveUniformBlockParameter.UniformBlockDataSize, &blockSize);
335  CheckLastError();
336  uniformBindings[i] = new OpenGLUniformBinding(_program, blockIndex, (uint)blockSize);
337  }
338  }
339  else if (resource.Kind == ResourceKind.TextureReadOnly)
340  {
341  int location = GetUniformLocation(resource.Name);
342  relativeTextureIndex += 1;
343  textureBindings[i] = new OpenGLTextureBindingSlotInfo() { RelativeIndex = relativeTextureIndex, UniformLocation = location };
344  lastTextureLocation = location;
345  samplerTrackedRelativeTextureIndices.Add(relativeTextureIndex);
346  }
347  else if (resource.Kind == ResourceKind.TextureReadWrite)
348  {
349  int location = GetUniformLocation(resource.Name);
350  relativeImageIndex += 1;
351  textureBindings[i] = new OpenGLTextureBindingSlotInfo() { RelativeIndex = relativeImageIndex, UniformLocation = location };
352  }
353  else if (resource.Kind == ResourceKind.StructuredBufferReadOnly
354  || resource.Kind == ResourceKind.StructuredBufferReadWrite)
355  {
356  uint storageBlockBinding;
357  if (_gd.BackendType == GraphicsBackend.OpenGL)
358  {
359  storageBlockBinding = GetProgramResourceIndex(resource.Name, ProgramInterface.ShaderStorageBlock);
360  }
361  else
362  {
363  storageBlockBinding = storageBlockIndex;
364  storageBlockIndex += 1;
365  }
366 
367  storageBufferBindings[i] = new OpenGLShaderStorageBinding(storageBlockBinding);
368  }
369  else
370  {
371  Debug.Assert(resource.Kind == ResourceKind.Sampler);
372 
373  int[] relativeIndices = samplerTrackedRelativeTextureIndices.ToArray();
374  samplerTrackedRelativeTextureIndices.Clear();
375  samplerBindings[i] = new OpenGLSamplerBindingSlotInfo()
376  {
377  RelativeIndices = relativeIndices
378  };
379  }
380  }
381 
382  _setInfos[setSlot] = new SetBindingsInfo(uniformBindings, textureBindings, samplerBindings, storageBufferBindings);
383  }

391  uint GetUniformBlockIndex(string resourceName)
392  {
393  int byteCount = Encoding.UTF8.GetByteCount(resourceName) + 1;
394  byte* resourceNamePtr = stackalloc byte[byteCount];
395  fixed (char* charPtr = resourceName)
396  {
397  int bytesWritten = Encoding.UTF8.GetBytes(charPtr, resourceName.Length, resourceNamePtr, byteCount);
398  Debug.Assert(bytesWritten == byteCount - 1);
399  }
400  resourceNamePtr[byteCount - 1] = 0; // Add null terminator.
401 
402  uint blockIndex = glGetUniformBlockIndex(_program, resourceNamePtr);
403  CheckLastError();
404 #if DEBUG && GL_VALIDATE_SHADER_RESOURCE_NAMES
405  if (blockIndex == GL_INVALID_INDEX)
406  {
407  uint uniformBufferIndex = 0;
408  uint bufferNameByteCount = 64;
409  byte* bufferNamePtr = stackalloc byte[(int)bufferNameByteCount];
410  var names = new List<string>();
411  while (true)
412  {
413  uint actualLength;
414  glGetActiveUniformBlockName(_program, uniformBufferIndex, bufferNameByteCount, &actualLength, bufferNamePtr);
415 
416  if (glGetError() != 0)
417  {
418  break;
419  }
420 
421  string name = Encoding.UTF8.GetString(bufferNamePtr, (int)actualLength);
422  names.Add(name);
423  uniformBufferIndex++;
424  }
425 
426  throw new VeldridException($"Unable to bind uniform buffer \"{resourceName}\" by name. Valid names for this pipeline are: {string.Join(", ", names)}");
427  }
428 #endif
429  return blockIndex;
430  }
