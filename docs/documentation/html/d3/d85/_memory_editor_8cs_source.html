35  private static string FixedHex(int v, int count)
36  {
37  return v.ToString("X").PadLeft(count, '0');

40  private static bool TryHexParse(byte[] bytes, out int result)
41  {
42  string input = System.Text.Encoding.UTF8.GetString(bytes).ToString();
43  return int.TryParse(input, NumberStyles.AllowHexSpecifier, CultureInfo.CurrentCulture, out result);
44  }

46  private static void ReplaceChars(byte[] bytes, string input)
47  {
48  var address = System.Text.Encoding.ASCII.GetBytes(input);
49  for (int i = 0; i < bytes.Length; i++)
50  {
51  bytes[i] = (i < address.Length) ? address[i] : (byte)0;
52  }

55  public unsafe void Draw(string title, byte[] mem_data, int mem_size, int base_display_addr = 0)
56  {
57  ImGui.SetNextWindowSize(new Vector2(500, 350), ImGuiCond.FirstUseEver);
58  if (!ImGui.Begin(title))
59  {
60  ImGui.End();
61  return;
62  }
63 
64  float line_height = ImGuiNative.igGetTextLineHeight();
65  int line_total_count = (mem_size + Rows - 1) / Rows;
66 
67  ImGuiNative.igSetNextWindowContentSize(new Vector2(0.0f, line_total_count * line_height));
68  ImGui.BeginChild("##scrolling", new Vector2(0, -ImGuiNative.igGetFrameHeightWithSpacing()), false, 0);
69 
70  ImGui.PushStyleVar(ImGuiStyleVar.FramePadding, new Vector2(0, 0));
71  ImGui.PushStyleVar(ImGuiStyleVar.ItemSpacing, new Vector2(0, 0));
72 
73  int addr_digits_count = 0;
74  for (int n = base_display_addr + mem_size - 1; n > 0; n >>= 4)
75  addr_digits_count++;
76 
77  float glyph_width = ImGui.CalcTextSize("F").X;
78  float cell_width = glyph_width * 3; // "FF " we include trailing space in the width to easily catch clicks everywhere
79 
80  var clipper = new ImGuiListClipper2(line_total_count, line_height);
81  int visible_start_addr = clipper.DisplayStart * Rows;
82  int visible_end_addr = clipper.DisplayEnd * Rows;
83 
84  bool data_next = false;
85 
86  if (!AllowEdits || DataEditingAddr >= mem_size)
87  DataEditingAddr = -1;
88 
89  int data_editing_addr_backup = DataEditingAddr;
90 
91  if (DataEditingAddr != -1)
92  {
93  if (ImGui.IsKeyPressed(ImGui.GetKeyIndex(ImGuiKey.UpArrow)) && DataEditingAddr >= Rows) { DataEditingAddr -= Rows; DataEditingTakeFocus = true; }
94  else if (ImGui.IsKeyPressed(ImGui.GetKeyIndex(ImGuiKey.DownArrow)) && DataEditingAddr < mem_size - Rows) { DataEditingAddr += Rows; DataEditingTakeFocus = true; }
95  else if (ImGui.IsKeyPressed(ImGui.GetKeyIndex(ImGuiKey.LeftArrow)) && DataEditingAddr > 0) { DataEditingAddr -= 1; DataEditingTakeFocus = true; }
96  else if (ImGui.IsKeyPressed(ImGui.GetKeyIndex(ImGuiKey.RightArrow)) && DataEditingAddr < mem_size - 1) { DataEditingAddr += 1; DataEditingTakeFocus = true; }
97  }
98  if ((DataEditingAddr / Rows) != (data_editing_addr_backup / Rows))
99  {
100  // Track cursor movements
101  float scroll_offset = ((DataEditingAddr / Rows) - (data_editing_addr_backup / Rows)) * line_height;
102  bool scroll_desired = (scroll_offset < 0.0f && DataEditingAddr < visible_start_addr + Rows * 2) || (scroll_offset > 0.0f && DataEditingAddr > visible_end_addr - Rows * 2);
103  if (scroll_desired)
104  ImGuiNative.igSetScrollY_Float(ImGuiNative.igGetScrollY() + scroll_offset);
105  }
106 
107  for (int line_i = clipper.DisplayStart; line_i < clipper.DisplayEnd; line_i++) // display only visible items
108  {
109  int addr = line_i * Rows;
110  ImGui.Text(FixedHex(base_display_addr + addr, addr_digits_count) + ": ");
111  ImGui.SameLine();
112 
113  // Draw Hexadecimal
114  float line_start_x = ImGuiNative.igGetCursorPosX();
115  for (int n = 0; n < Rows && addr < mem_size; n++, addr++)
116  {
117  ImGui.SameLine(line_start_x + cell_width * n);
118 
119  if (DataEditingAddr == addr)
120  {
121  // Display text input on current byte
122  ImGui.PushID(addr);
123 
124  // FIXME: We should have a way to retrieve the text edit cursor position more easily in the API, this is rather tedious.
125  ImGuiInputTextCallback callback = (data) =>
126  {
127  int* p_cursor_pos = (int*)data->UserData;
128 
129  if (ImGuiNative.ImGuiInputTextCallbackData_HasSelection(data) == 0)
130  *p_cursor_pos = data->CursorPos;
131  return 0;
132  };
133  int cursor_pos = -1;
134  bool data_write = false;
135  if (DataEditingTakeFocus)
136  {
137  ImGui.SetKeyboardFocusHere();
138  ReplaceChars(DataInput, FixedHex(mem_data[addr], 2));
139  ReplaceChars(AddrInput, FixedHex(base_display_addr + addr, addr_digits_count));
140  }
141  ImGui.PushItemWidth(ImGui.CalcTextSize("FF").X);
142 
143  var flags = ImGuiInputTextFlags.CharsHexadecimal | ImGuiInputTextFlags.EnterReturnsTrue | ImGuiInputTextFlags.AutoSelectAll | ImGuiInputTextFlags.NoHorizontalScroll | ImGuiInputTextFlags.CallbackAlways;
144 
145  if (ImGui.InputText("##data", DataInput, 32, flags, callback, (IntPtr)(&cursor_pos)))
146  data_write = data_next = true;
147  else if (!DataEditingTakeFocus && !ImGui.IsItemActive())
148  DataEditingAddr = -1;
149 
150  DataEditingTakeFocus = false;
151  ImGui.PopItemWidth();
152  if (cursor_pos >= 2)
153  data_write = data_next = true;
154  if (data_write)
155  {
156  int data;
157  if (TryHexParse(DataInput, out data))
158  mem_data[addr] = (byte)data;
159  }
160  ImGui.PopID();
161  }
162  else
163  {
164  ImGui.Text(FixedHex(mem_data[addr], 2));
165  if (AllowEdits && ImGui.IsItemHovered() && ImGui.IsMouseClicked(0))
166  {
167  DataEditingTakeFocus = true;
168  DataEditingAddr = addr;
169  }
170  }
171  }
172 
173  ImGui.SameLine(line_start_x + cell_width * Rows + glyph_width * 2);
174  //separator line drawing replaced by printing a pipe char
175 
176  // Draw ASCII values
177  addr = line_i * Rows;
178  var asciiVal = new System.Text.StringBuilder(2 + Rows);
179  asciiVal.Append("| ");
180  for (int n = 0; n < Rows && addr < mem_size; n++, addr++)
181  {
182  int c = mem_data[addr];
183  asciiVal.Append((c >= 32 && c < 128) ? Convert.ToChar(c) : '.');
184  }
185  ImGui.TextUnformatted(asciiVal.ToString()); //use unformatted, so string can contain the '%' character
186  }
187  //clipper.End(); //not implemented
188  ImGui.PopStyleVar(2);
189 
190  ImGui.EndChild();
191 
192  if (data_next && DataEditingAddr < mem_size)
193  {
194  DataEditingAddr = DataEditingAddr + 1;
195  DataEditingTakeFocus = true;
196  }
197 
198  ImGui.Separator();
199 
200  ImGuiNative.igAlignTextToFramePadding();
201  ImGui.PushItemWidth(50);
202  ImGui.PushAllowKeyboardFocus(true);
203  int rows_backup = Rows;
204  if (ImGui.DragInt("##rows", ref Rows, 0.2f, 4, 32, "%.0f rows"))
205  {
206  if (Rows <= 0) Rows = 4;
207  Vector2 new_window_size = ImGui.GetWindowSize();
208  new_window_size.X += (Rows - rows_backup) * (cell_width + glyph_width);
209  ImGui.SetWindowSize(new_window_size);
210  }
211  ImGui.PopAllowKeyboardFocus();
212  ImGui.PopItemWidth();
213  ImGui.SameLine();
214  ImGui.Text(string.Format(" Range {0}..{1} ", FixedHex(base_display_addr, addr_digits_count),
215  FixedHex(base_display_addr + mem_size - 1, addr_digits_count)));
216  ImGui.SameLine();
217  ImGui.PushItemWidth(70);
218  if (ImGui.InputText("##addr", AddrInput, 32, ImGuiInputTextFlags.CharsHexadecimal | ImGuiInputTextFlags.EnterReturnsTrue, null))
219  {
220  int goto_addr;
221  if (TryHexParse(AddrInput, out goto_addr))
222  {
223  goto_addr -= base_display_addr;
224  if (goto_addr >= 0 && goto_addr < mem_size)
225  {
226  ImGui.BeginChild("##scrolling");
227  ImGui.SetScrollFromPosY(ImGui.GetCursorStartPos().Y + (goto_addr / Rows) * ImGuiNative.igGetTextLineHeight());
228  ImGui.EndChild();
229  DataEditingAddr = goto_addr;
230  DataEditingTakeFocus = true;
231  }
232  }
233  }
234  ImGui.PopItemWidth();
235 
236  ImGui.End();
237  }
238  }
