57  public static List<Vertices> ConvexPartition(Vertices vertices, TriangulationAlgorithm algorithm,
58  bool discardAndFixInvalid = true, float tolerance = 0.001f)
59  {
60  if (vertices.Count <= 3)
61  {
62  return new List<Vertices> {vertices};
63  }
64 
65  List<Vertices> results = null;
66 
67  switch (algorithm)
68  {
69  case TriangulationAlgorithm.Earclip:
70  if (!Settings.SkipSanityChecks)
71  {
72  if (vertices.IsCounterClockWise())
73  {
74  Vertices temp = new Vertices(vertices);
75  temp.Reverse();
76  results = EarclipDecomposer.ConvexPartition(temp, tolerance);
77  }
78  else
79  {
80  results = EarclipDecomposer.ConvexPartition(vertices, tolerance);
81  }
82  }
83 
84  break;
85  case TriangulationAlgorithm.Bayazit:
86  if (!Settings.SkipSanityChecks)
87  {
88  if (!vertices.IsCounterClockWise())
89  {
90  Vertices temp = new Vertices(vertices);
91  temp.Reverse();
92  results = BayazitDecomposer.ConvexPartition(temp);
93  }
94  else
95  {
96  results = BayazitDecomposer.ConvexPartition(vertices);
97  }
98  }
99 
100  break;
101  case TriangulationAlgorithm.Flipcode:
102  if (!Settings.SkipSanityChecks)
103  {
104  if (!vertices.IsCounterClockWise())
105  {
106  Vertices temp = new Vertices(vertices);
107  temp.Reverse();
108  results = FlipcodeDecomposer.ConvexPartition(temp);
109  }
110  else
111  {
112  results = FlipcodeDecomposer.ConvexPartition(vertices);
113  }
114  }
115 
116  break;
117  case TriangulationAlgorithm.Seidel:
118  results = SeidelDecomposer.ConvexPartition(vertices, tolerance);
119  break;
120  case TriangulationAlgorithm.SeidelTrapezoids:
121  results = SeidelDecomposer.ConvexPartitionTrapezoid(vertices, tolerance);
122  break;
123  case TriangulationAlgorithm.Delauny:
124  results = CdtDecomposer.ConvexPartition(vertices);
125  break;
126  default:
127  throw new ArgumentOutOfRangeException(nameof(algorithm));
128  }
129 
130  if (discardAndFixInvalid)
131  {
132  for (int i = results.Count - 1; i >= 0; i--)
133  {
134  Vertices polygon = results[i];
135 
136  if (!ValidatePolygon(polygon))
137  {
138  results.RemoveAt(i);
139  }
140  }
141  }
142 
143  return results;
144  }

151  private static bool ValidatePolygon(Vertices polygon)
152  {
153  PolygonError errorCode = polygon.CheckPolygon();
154 
155  if (errorCode == PolygonError.InvalidAmountOfVertices || errorCode == PolygonError.AreaTooSmall ||
156  errorCode == PolygonError.SideTooSmall || errorCode == PolygonError.NotSimple)
157  {
158  return false;
159  }
160 
161  if (
162  errorCode ==
163  PolygonError
164  .NotCounterClockWise) //NotCounterCloseWise is the last check in CheckPolygon(), thus we don't need to call ValidatePolygon again.
165  {
166  polygon.Reverse();
167  }
168 
169  if (errorCode == PolygonError.NotConvex)
170  {
171  polygon = GiftWrap.GetConvexHull(polygon);
172  return ValidatePolygon(polygon);
173  }
174 
175  return true;
176  }
177  }
