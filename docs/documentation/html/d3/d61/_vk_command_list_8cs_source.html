463  private void FlushNewResourceSets(
464  BoundResourceSetInfo[] resourceSets,
465  bool[] resourceSetsChanged,
466  uint resourceSetCount,
467  VkPipelineBindPoint bindPoint,
468  VkPipelineLayout pipelineLayout)
469  {
470  VkPipeline pipeline = bindPoint == VkPipelineBindPoint.Graphics ? _currentGraphicsPipeline : _currentComputePipeline;
471 
472  VkDescriptorSet* descriptorSets = stackalloc VkDescriptorSet[(int)resourceSetCount];
473  uint* dynamicOffsets = stackalloc uint[pipeline.DynamicOffsetsCount];
474  uint currentBatchCount = 0;
475  uint currentBatchFirstSet = 0;
476  uint currentBatchDynamicOffsetCount = 0;
477 
478  for (uint currentSlot = 0; currentSlot < resourceSetCount; currentSlot++)
479  {
480  bool batchEnded = !resourceSetsChanged[currentSlot] || currentSlot == resourceSetCount - 1;
481 
482  if (resourceSetsChanged[currentSlot])
483  {
484  resourceSetsChanged[currentSlot] = false;
485  VkResourceSet vkSet = Util.AssertSubtype<ResourceSet, VkResourceSet>(resourceSets[currentSlot].Set);
486  descriptorSets[currentBatchCount] = vkSet.DescriptorSet;
487  currentBatchCount += 1;
488 
489  ref SmallFixedOrDynamicArray curSetOffsets = ref resourceSets[currentSlot].Offsets;
490  for (uint i = 0; i < curSetOffsets.Count; i++)
491  {
492  dynamicOffsets[currentBatchDynamicOffsetCount] = curSetOffsets.Get(i);
493  currentBatchDynamicOffsetCount += 1;
494  }
495 
496  // Increment ref count on first use of a set.
497  _currentStagingInfo.Resources.Add(vkSet.RefCount);
498  for (int i = 0; i < vkSet.RefCounts.Count; i++)
499  {
500  _currentStagingInfo.Resources.Add(vkSet.RefCounts[i]);
501  }
502  }
503 
504  if (batchEnded)
505  {
506  if (currentBatchCount != 0)
507  {
508  // Flush current batch.
509  vkCmdBindDescriptorSets(
510  _cb,
511  bindPoint,
512  pipelineLayout,
513  currentBatchFirstSet,
514  currentBatchCount,
515  descriptorSets,
516  currentBatchDynamicOffsetCount,
517  dynamicOffsets);
518  }
519 
520  currentBatchCount = 0;
521  currentBatchFirstSet = currentSlot + 1;
522  }

735  private void BeginCurrentRenderPass()
736  {
737  Debug.Assert(_activeRenderPass == VkRenderPass.Null);
738  Debug.Assert(_currentFramebuffer != null);
739  _currentFramebufferEverActive = true;
740 
741  uint attachmentCount = _currentFramebuffer.AttachmentCount;
742  bool haveAnyAttachments = _currentFramebuffer.ColorTargets.Count > 0 || _currentFramebuffer.DepthTarget != null;
743  bool haveAllClearValues = _depthClearValue.HasValue || _currentFramebuffer.DepthTarget == null;
744  bool haveAnyClearValues = _depthClearValue.HasValue;
745  for (int i = 0; i < _currentFramebuffer.ColorTargets.Count; i++)
746  {
747  if (!_validColorClearValues[i])
748  {
749  haveAllClearValues = false;
750  }
751  else
752  {
753  haveAnyClearValues = true;
754  }
755  }
756 
757  VkRenderPassBeginInfo renderPassBI = VkRenderPassBeginInfo.New();
758  renderPassBI.renderArea = new VkRect2D(_currentFramebuffer.RenderableWidth, _currentFramebuffer.RenderableHeight);
759  renderPassBI.framebuffer = _currentFramebuffer.CurrentFramebuffer;
760 
761  if (!haveAnyAttachments || !haveAllClearValues)
762  {
763  renderPassBI.renderPass = _newFramebuffer
764  ? _currentFramebuffer.RenderPassNoClear_Init
765  : _currentFramebuffer.RenderPassNoClear_Load;
766  vkCmdBeginRenderPass(_cb, ref renderPassBI, VkSubpassContents.Inline);
767  _activeRenderPass = renderPassBI.renderPass;
768 
769  if (haveAnyClearValues)
770  {
771  if (_depthClearValue.HasValue)
772  {
773  ClearDepthStencilCore(_depthClearValue.Value.depthStencil.depth, (byte)_depthClearValue.Value.depthStencil.stencil);
774  _depthClearValue = null;
775  }
776 
777  for (uint i = 0; i < _currentFramebuffer.ColorTargets.Count; i++)
778  {
779  if (_validColorClearValues[i])
780  {
781  _validColorClearValues[i] = false;
782  VkClearValue vkClearValue = _clearValues[i];
783  RgbaFloat clearColor = new RgbaFloat(
784  vkClearValue.color.float32_0,
785  vkClearValue.color.float32_1,
786  vkClearValue.color.float32_2,
787  vkClearValue.color.float32_3);
788  ClearColorTarget(i, clearColor);
789  }
790  }
791  }
792  }
793  else
794  {
795  // We have clear values for every attachment.
796  renderPassBI.renderPass = _currentFramebuffer.RenderPassClear;
797  fixed (VkClearValue* clearValuesPtr = &_clearValues[0])
798  {
799  renderPassBI.clearValueCount = attachmentCount;
800  renderPassBI.pClearValues = clearValuesPtr;
801  if (_depthClearValue.HasValue)
802  {
803  _clearValues[_currentFramebuffer.ColorTargets.Count] = _depthClearValue.Value;
804  _depthClearValue = null;
805  }
806  vkCmdBeginRenderPass(_cb, ref renderPassBI, VkSubpassContents.Inline);
807  _activeRenderPass = _currentFramebuffer.RenderPassClear;
808  Util.ClearArray(_validColorClearValues);
809  }
810  }
811 
812  _newFramebuffer = false;
813  }
