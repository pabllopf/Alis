462  private void FlushNewResourceSets(
463  BoundResourceSetInfo[] resourceSets,
464  bool[] resourceSetsChanged,
465  uint resourceSetCount,
466  VkPipelineBindPoint bindPoint,
467  VkPipelineLayout pipelineLayout)
468  {
469  VkPipeline pipeline = bindPoint == VkPipelineBindPoint.Graphics ? _currentGraphicsPipeline : _currentComputePipeline;
470 
471  VkDescriptorSet* descriptorSets = stackalloc VkDescriptorSet[(int)resourceSetCount];
472  uint* dynamicOffsets = stackalloc uint[pipeline.DynamicOffsetsCount];
473  uint currentBatchCount = 0;
474  uint currentBatchFirstSet = 0;
475  uint currentBatchDynamicOffsetCount = 0;
476 
477  for (uint currentSlot = 0; currentSlot < resourceSetCount; currentSlot++)
478  {
479  bool batchEnded = !resourceSetsChanged[currentSlot] || currentSlot == resourceSetCount - 1;
480 
481  if (resourceSetsChanged[currentSlot])
482  {
483  resourceSetsChanged[currentSlot] = false;
484  VkResourceSet vkSet = Util.AssertSubtype<ResourceSet, VkResourceSet>(resourceSets[currentSlot].Set);
485  descriptorSets[currentBatchCount] = vkSet.DescriptorSet;
486  currentBatchCount += 1;
487 
488  ref SmallFixedOrDynamicArray curSetOffsets = ref resourceSets[currentSlot].Offsets;
489  for (uint i = 0; i < curSetOffsets.Count; i++)
490  {
491  dynamicOffsets[currentBatchDynamicOffsetCount] = curSetOffsets.Get(i);
492  currentBatchDynamicOffsetCount += 1;
493  }
494 
495  // Increment ref count on first use of a set.
496  _currentStagingInfo.Resources.Add(vkSet.RefCount);
497  for (int i = 0; i < vkSet.RefCounts.Count; i++)
498  {
499  _currentStagingInfo.Resources.Add(vkSet.RefCounts[i]);
500  }
501  }
502 
503  if (batchEnded)
504  {
505  if (currentBatchCount != 0)
506  {
507  // Flush current batch.
508  vkCmdBindDescriptorSets(
509  _cb,
510  bindPoint,
511  pipelineLayout,
512  currentBatchFirstSet,
513  currentBatchCount,
514  descriptorSets,
515  currentBatchDynamicOffsetCount,
516  dynamicOffsets);
517  }
518 
519  currentBatchCount = 0;
520  currentBatchFirstSet = currentSlot + 1;
521  }

734  private void BeginCurrentRenderPass()
735  {
736  Debug.Assert(_activeRenderPass == VkRenderPass.Null);
737  Debug.Assert(_currentFramebuffer != null);
738  _currentFramebufferEverActive = true;
739 
740  uint attachmentCount = _currentFramebuffer.AttachmentCount;
741  bool haveAnyAttachments = _currentFramebuffer.ColorTargets.Count > 0 || _currentFramebuffer.DepthTarget != null;
742  bool haveAllClearValues = _depthClearValue.HasValue || _currentFramebuffer.DepthTarget == null;
743  bool haveAnyClearValues = _depthClearValue.HasValue;
744  for (int i = 0; i < _currentFramebuffer.ColorTargets.Count; i++)
745  {
746  if (!_validColorClearValues[i])
747  {
748  haveAllClearValues = false;
749  }
750  else
751  {
752  haveAnyClearValues = true;
753  }
754  }
755 
756  VkRenderPassBeginInfo renderPassBI = VkRenderPassBeginInfo.New();
757  renderPassBI.renderArea = new VkRect2D(_currentFramebuffer.RenderableWidth, _currentFramebuffer.RenderableHeight);
758  renderPassBI.framebuffer = _currentFramebuffer.CurrentFramebuffer;
759 
760  if (!haveAnyAttachments || !haveAllClearValues)
761  {
762  renderPassBI.renderPass = _newFramebuffer
763  ? _currentFramebuffer.RenderPassNoClear_Init
764  : _currentFramebuffer.RenderPassNoClear_Load;
765  vkCmdBeginRenderPass(_cb, ref renderPassBI, VkSubpassContents.Inline);
766  _activeRenderPass = renderPassBI.renderPass;
767 
768  if (haveAnyClearValues)
769  {
770  if (_depthClearValue.HasValue)
771  {
772  ClearDepthStencilCore(_depthClearValue.Value.depthStencil.depth, (byte)_depthClearValue.Value.depthStencil.stencil);
773  _depthClearValue = null;
774  }
775 
776  for (uint i = 0; i < _currentFramebuffer.ColorTargets.Count; i++)
777  {
778  if (_validColorClearValues[i])
779  {
780  _validColorClearValues[i] = false;
781  VkClearValue vkClearValue = _clearValues[i];
782  RgbaFloat clearColor = new RgbaFloat(
783  vkClearValue.color.float32_0,
784  vkClearValue.color.float32_1,
785  vkClearValue.color.float32_2,
786  vkClearValue.color.float32_3);
787  ClearColorTarget(i, clearColor);
788  }
789  }
790  }
791  }
792  else
793  {
794  // We have clear values for every attachment.
795  renderPassBI.renderPass = _currentFramebuffer.RenderPassClear;
796  fixed (VkClearValue* clearValuesPtr = &_clearValues[0])
797  {
798  renderPassBI.clearValueCount = attachmentCount;
799  renderPassBI.pClearValues = clearValuesPtr;
800  if (_depthClearValue.HasValue)
801  {
802  _clearValues[_currentFramebuffer.ColorTargets.Count] = _depthClearValue.Value;
803  _depthClearValue = null;
804  }
805  vkCmdBeginRenderPass(_cb, ref renderPassBI, VkSubpassContents.Inline);
806  _activeRenderPass = _currentFramebuffer.RenderPassClear;
807  Util.ClearArray(_validColorClearValues);
808  }
809  }
810 
811  _newFramebuffer = false;
812  }
