154  public static Quaternion operator /(Quaternion value1, Quaternion value2)
155  {
156  Quaternion ans = Zero;
157 
158  float q1X = value1.X;
159  float q1Y = value1.Y;
160  float q1Z = value1.Z;
161  float q1W = value1.W;
162 
163  //-------------------------------------
164  // Inverse part.
165  float ls = value2.X * value2.X + value2.Y * value2.Y +
166  value2.Z * value2.Z + value2.W * value2.W;
167  float invNorm = 1.0f / ls;
168 
169  float q2X = -value2.X * invNorm;
170  float q2Y = -value2.Y * invNorm;
171  float q2Z = -value2.Z * invNorm;
172  float q2W = value2.W * invNorm;
173 
174  //-------------------------------------
175  // Multiply part.
176 
177  // cross(av, bv)
178  float cx = q1Y * q2Z - q1Z * q2Y;
179  float cy = q1Z * q2X - q1X * q2Z;
180  float cz = q1X * q2Y - q1Y * q2X;
181 
182  float dot = q1X * q2X + q1Y * q2Y + q1Z * q2Z;
183 
184  ans.X = q1X * q2W + q2X * q1W + cx;
185  ans.Y = q1Y * q2W + q2Y * q1W + cy;
186  ans.Z = q1Z * q2W + q2Z * q1W + cz;
187  ans.W = q1W * q2W - dot;
188 
189  return ans;
190  }

396  public static Quaternion CreateFromRotationMatrix(Matrix4X4F matrix)
397  {
398  float trace = matrix.M11 + matrix.M22 + matrix.M33;
399 
400  Quaternion q = default(Quaternion);
401 
402  if (trace > 0.0f)
403  {
404  float s = MathF.Sqrt(trace + 1.0f);
405  q.W = s * 0.5f;
406  s = 0.5f / s;
407  q.X = (matrix.M23 - matrix.M32) * s;
408  q.Y = (matrix.M31 - matrix.M13) * s;
409  q.Z = (matrix.M12 - matrix.M21) * s;
410  }
411  else
412  {
413  if ((matrix.M11 >= matrix.M22) && (matrix.M11 >= matrix.M33))
414  {
415  float s = MathF.Sqrt(1.0f + matrix.M11 - matrix.M22 - matrix.M33);
416  float invS = 0.5f / s;
417  q.X = 0.5f * s;
418  q.Y = (matrix.M12 + matrix.M21) * invS;
419  q.Z = (matrix.M13 + matrix.M31) * invS;
420  q.W = (matrix.M23 - matrix.M32) * invS;
421  }
422  else if (matrix.M22 > matrix.M33)
423  {
424  float s = MathF.Sqrt(1.0f + matrix.M22 - matrix.M11 - matrix.M33);
425  float invS = 0.5f / s;
426  q.X = (matrix.M21 + matrix.M12) * invS;
427  q.Y = 0.5f * s;
428  q.Z = (matrix.M32 + matrix.M23) * invS;
429  q.W = (matrix.M31 - matrix.M13) * invS;
430  }
431  else
432  {
433  float s = MathF.Sqrt(1.0f + matrix.M33 - matrix.M11 - matrix.M22);
434  float invS = 0.5f / s;
435  q.X = (matrix.M31 + matrix.M13) * invS;
436  q.Y = (matrix.M32 + matrix.M23) * invS;
437  q.Z = 0.5f * s;
438  q.W = (matrix.M12 - matrix.M21) * invS;
439  }
440  }
441 
442  return q;
443  }

450  public static Quaternion CreateFromYawPitchRoll(float yaw, float pitch, float roll)
451  {
452  // Roll first, about axis the object is facing, then
453  // pitch upward, then yaw to face into the new heading
454  float sr, cr, sp, cp, sy, cy;
455 
456  float halfRoll = roll * 0.5f;
457  sr = MathF.Sin(halfRoll);
458  cr = MathF.Cos(halfRoll);
459 
460  float halfPitch = pitch * 0.5f;
461  sp = MathF.Sin(halfPitch);
462  cp = MathF.Cos(halfPitch);
463 
464  float halfYaw = yaw * 0.5f;
465  sy = MathF.Sin(halfYaw);
466  cy = MathF.Cos(halfYaw);
467 
468  Quaternion result = Zero;
469 
470  result.X = cy * sp * cr + sy * cp * sr;
471  result.Y = sy * cp * cr - cy * sp * sr;
472  result.Z = cy * cp * sr - sy * sp * cr;
473  result.W = cy * cp * cr + sy * sp * sr;
474 
475  return result;
476  }

524  public static Quaternion Lerp(Quaternion quaternion1, Quaternion quaternion2, float amount)
525  {
526  float t = amount;
527  float t1 = 1.0f - t;
528 
529  Quaternion r = default(Quaternion);
530 
531  float dot = quaternion1.X * quaternion2.X + quaternion1.Y * quaternion2.Y +
532  quaternion1.Z * quaternion2.Z + quaternion1.W * quaternion2.W;
533 
534  if (dot >= 0.0f)
535  {
536  r.X = t1 * quaternion1.X + t * quaternion2.X;
537  r.Y = t1 * quaternion1.Y + t * quaternion2.Y;
538  r.Z = t1 * quaternion1.Z + t * quaternion2.Z;
539  r.W = t1 * quaternion1.W + t * quaternion2.W;
540  }
541  else
542  {
543  r.X = t1 * quaternion1.X - t * quaternion2.X;
544  r.Y = t1 * quaternion1.Y - t * quaternion2.Y;
545  r.Z = t1 * quaternion1.Z - t * quaternion2.Z;
546  r.W = t1 * quaternion1.W - t * quaternion2.W;
547  }
548 
549  // Normalize it.
550  float ls = r.X * r.X + r.Y * r.Y + r.Z * r.Z + r.W * r.W;
551  float invNorm = 1.0f / MathF.Sqrt(ls);
552 
553  r.X *= invNorm;
554  r.Y *= invNorm;
555  r.Z *= invNorm;
556  r.W *= invNorm;
557 
558  return r;
559  }

608  public static Quaternion Slerp(Quaternion quaternion1, Quaternion quaternion2, float amount)
609  {
610  float t = amount;
611 
612  float cosOmega = quaternion1.X * quaternion2.X + quaternion1.Y * quaternion2.Y +
613  quaternion1.Z * quaternion2.Z + quaternion1.W * quaternion2.W;
614 
615  bool flip = false;
616 
617  if (cosOmega < 0.0f)
618  {
619  flip = true;
620  cosOmega = -cosOmega;
621  }
622 
623  float s1, s2;
624 
625  if (cosOmega > 1.0f - SlerpEpsilon)
626  {
627  // Too close, do straight linear interpolation.
628  s1 = 1.0f - t;
629  s2 = flip ? -t : t;
630  }
631  else
632  {
633  float omega = MathF.Acos(cosOmega);
634  float invSinOmega = 1 / MathF.Sin(omega);
635 
636  s1 = MathF.Sin((1.0f - t) * omega) * invSinOmega;
637  s2 = flip
638  ? -MathF.Sin(t * omega) * invSinOmega
639  : MathF.Sin(t * omega) * invSinOmega;
640  }
641 
642  Quaternion ans = Zero;
643 
644  ans.X = s1 * quaternion1.X + s2 * quaternion2.X;
645  ans.Y = s1 * quaternion1.Y + s2 * quaternion2.Y;
646  ans.Z = s1 * quaternion1.Z + s2 * quaternion2.Z;
647  ans.W = s1 * quaternion1.W + s2 * quaternion2.W;
648 
649  return ans;
650  }
