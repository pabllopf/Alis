128  public static Quaternion operator /(Quaternion value1, Quaternion value2)
129  {
130  Quaternion ans;
131 
132  float q1x = value1.X;
133  float q1y = value1.Y;
134  float q1z = value1.Z;
135  float q1w = value1.W;
136 
137  //-------------------------------------
138  // Inverse part.
139  float ls = value2.X * value2.X + value2.Y * value2.Y +
140  value2.Z * value2.Z + value2.W * value2.W;
141  float invNorm = 1.0f / ls;
142 
143  float q2x = -value2.X * invNorm;
144  float q2y = -value2.Y * invNorm;
145  float q2z = -value2.Z * invNorm;
146  float q2w = value2.W * invNorm;
147 
148  //-------------------------------------
149  // Multiply part.
150 
151  // cross(av, bv)
152  float cx = q1y * q2z - q1z * q2y;
153  float cy = q1z * q2x - q1x * q2z;
154  float cz = q1x * q2y - q1y * q2x;
155 
156  float dot = q1x * q2x + q1y * q2y + q1z * q2z;
157 
158  ans.X = q1x * q2w + q2x * q1w + cx;
159  ans.Y = q1y * q2w + q2y * q1w + cy;
160  ans.Z = q1z * q2w + q2z * q1w + cz;
161  ans.W = q1w * q2w - dot;
162 
163  return ans;
164  }

370  public static Quaternion CreateFromRotationMatrix(Matrix4x4 matrix)
371  {
372  float trace = matrix.M11 + matrix.M22 + matrix.M33;
373 
374  Quaternion q = default(Quaternion);
375 
376  if (trace > 0.0f)
377  {
378  float s = MathF.Sqrt(trace + 1.0f);
379  q.W = s * 0.5f;
380  s = 0.5f / s;
381  q.X = (matrix.M23 - matrix.M32) * s;
382  q.Y = (matrix.M31 - matrix.M13) * s;
383  q.Z = (matrix.M12 - matrix.M21) * s;
384  }
385  else
386  {
387  if ((matrix.M11 >= matrix.M22) && (matrix.M11 >= matrix.M33))
388  {
389  float s = MathF.Sqrt(1.0f + matrix.M11 - matrix.M22 - matrix.M33);
390  float invS = 0.5f / s;
391  q.X = 0.5f * s;
392  q.Y = (matrix.M12 + matrix.M21) * invS;
393  q.Z = (matrix.M13 + matrix.M31) * invS;
394  q.W = (matrix.M23 - matrix.M32) * invS;
395  }
396  else if (matrix.M22 > matrix.M33)
397  {
398  float s = MathF.Sqrt(1.0f + matrix.M22 - matrix.M11 - matrix.M33);
399  float invS = 0.5f / s;
400  q.X = (matrix.M21 + matrix.M12) * invS;
401  q.Y = 0.5f * s;
402  q.Z = (matrix.M32 + matrix.M23) * invS;
403  q.W = (matrix.M31 - matrix.M13) * invS;
404  }
405  else
406  {
407  float s = MathF.Sqrt(1.0f + matrix.M33 - matrix.M11 - matrix.M22);
408  float invS = 0.5f / s;
409  q.X = (matrix.M31 + matrix.M13) * invS;
410  q.Y = (matrix.M32 + matrix.M23) * invS;
411  q.Z = 0.5f * s;
412  q.W = (matrix.M12 - matrix.M21) * invS;
413  }
414  }
415 
416  return q;
417  }

424  public static Quaternion CreateFromYawPitchRoll(float yaw, float pitch, float roll)
425  {
426  // Roll first, about axis the object is facing, then
427  // pitch upward, then yaw to face into the new heading
428  float sr, cr, sp, cp, sy, cy;
429 
430  float halfRoll = roll * 0.5f;
431  sr = MathF.Sin(halfRoll);
432  cr = MathF.Cos(halfRoll);
433 
434  float halfPitch = pitch * 0.5f;
435  sp = MathF.Sin(halfPitch);
436  cp = MathF.Cos(halfPitch);
437 
438  float halfYaw = yaw * 0.5f;
439  sy = MathF.Sin(halfYaw);
440  cy = MathF.Cos(halfYaw);
441 
442  Quaternion result;
443 
444  result.X = cy * sp * cr + sy * cp * sr;
445  result.Y = sy * cp * cr - cy * sp * sr;
446  result.Z = cy * cp * sr - sy * sp * cr;
447  result.W = cy * cp * cr + sy * sp * sr;
448 
449  return result;
450  }

498  public static Quaternion Lerp(Quaternion quaternion1, Quaternion quaternion2, float amount)
499  {
500  float t = amount;
501  float t1 = 1.0f - t;
502 
503  Quaternion r = default(Quaternion);
504 
505  float dot = quaternion1.X * quaternion2.X + quaternion1.Y * quaternion2.Y +
506  quaternion1.Z * quaternion2.Z + quaternion1.W * quaternion2.W;
507 
508  if (dot >= 0.0f)
509  {
510  r.X = t1 * quaternion1.X + t * quaternion2.X;
511  r.Y = t1 * quaternion1.Y + t * quaternion2.Y;
512  r.Z = t1 * quaternion1.Z + t * quaternion2.Z;
513  r.W = t1 * quaternion1.W + t * quaternion2.W;
514  }
515  else
516  {
517  r.X = t1 * quaternion1.X - t * quaternion2.X;
518  r.Y = t1 * quaternion1.Y - t * quaternion2.Y;
519  r.Z = t1 * quaternion1.Z - t * quaternion2.Z;
520  r.W = t1 * quaternion1.W - t * quaternion2.W;
521  }
522 
523  // Normalize it.
524  float ls = r.X * r.X + r.Y * r.Y + r.Z * r.Z + r.W * r.W;
525  float invNorm = 1.0f / MathF.Sqrt(ls);
526 
527  r.X *= invNorm;
528  r.Y *= invNorm;
529  r.Z *= invNorm;
530  r.W *= invNorm;
531 
532  return r;
533  }

582  public static Quaternion Slerp(Quaternion quaternion1, Quaternion quaternion2, float amount)
583  {
584  float t = amount;
585 
586  float cosOmega = quaternion1.X * quaternion2.X + quaternion1.Y * quaternion2.Y +
587  quaternion1.Z * quaternion2.Z + quaternion1.W * quaternion2.W;
588 
589  bool flip = false;
590 
591  if (cosOmega < 0.0f)
592  {
593  flip = true;
594  cosOmega = -cosOmega;
595  }
596 
597  float s1, s2;
598 
599  if (cosOmega > 1.0f - SlerpEpsilon)
600  {
601  // Too close, do straight linear interpolation.
602  s1 = 1.0f - t;
603  s2 = flip ? -t : t;
604  }
605  else
606  {
607  float omega = MathF.Acos(cosOmega);
608  float invSinOmega = 1 / MathF.Sin(omega);
609 
610  s1 = MathF.Sin((1.0f - t) * omega) * invSinOmega;
611  s2 = flip
612  ? -MathF.Sin(t * omega) * invSinOmega
613  : MathF.Sin(t * omega) * invSinOmega;
614  }
615 
616  Quaternion ans;
617 
618  ans.X = s1 * quaternion1.X + s2 * quaternion2.X;
619  ans.Y = s1 * quaternion1.Y + s2 * quaternion2.Y;
620  ans.Z = s1 * quaternion1.Z + s2 * quaternion2.Z;
621  ans.W = s1 * quaternion1.W + s2 * quaternion2.W;
622 
623  return ans;
624  }
