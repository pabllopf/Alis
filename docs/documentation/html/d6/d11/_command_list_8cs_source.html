254  public void SetGraphicsResourceSet(uint slot, ResourceSet rs, uint dynamicOffsetsCount, ref uint dynamicOffsets)
255  {
256 #if VALIDATE_USAGE

260  }

262  int layoutsCount = _graphicsPipeline.ResourceLayouts.Length;
263  if (layoutsCount <= slot)
264  {
265  throw new VeldridException(
266  $"Failed to bind ResourceSet to slot {slot}. The active graphics Pipeline only contains {layoutsCount} ResourceLayouts.");
267  }
268 
269  ResourceLayout layout = _graphicsPipeline.ResourceLayouts[slot];
270  int pipelineLength = layout.Description.Elements.Length;
271  ResourceLayoutDescription layoutDesc = rs.Layout.Description;
272  int setLength = layoutDesc.Elements.Length;
273  if (pipelineLength != setLength)
274  {
275  throw new VeldridException($"Failed to bind ResourceSet to slot {slot}. The number of resources in the ResourceSet ({setLength}) does not match the number expected by the active Pipeline ({pipelineLength}).");
276  }
277 
278  for (int i = 0; i < pipelineLength; i++)
279  {
280  ResourceKind pipelineKind = layout.Description.Elements[i].Kind;
281  ResourceKind setKind = layoutDesc.Elements[i].Kind;
282  if (pipelineKind != setKind)
283  {
284  throw new VeldridException(
285  $"Failed to bind ResourceSet to slot {slot}. Resource element {i} was of the incorrect type. The bound Pipeline expects {pipelineKind}, but the ResourceSet contained {setKind}.");
286  }
287  }
288 
289  if (rs.Layout.DynamicBufferCount != dynamicOffsetsCount)
290  {
291  throw new VeldridException(
292  $"A dynamic offset must be provided for each resource that specifies " +
293  $"{nameof(ResourceLayoutElementOptions)}.{nameof(ResourceLayoutElementOptions.DynamicBinding)}. " +
294  $"{rs.Layout.DynamicBufferCount} offsets were expected, but only {dynamicOffsetsCount} were provided.");
295  }
296 
297  uint dynamicOffsetIndex = 0;
298  for (uint i = 0; i < layoutDesc.Elements.Length; i++)
299  {
300  if ((layoutDesc.Elements[i].Options & ResourceLayoutElementOptions.DynamicBinding) != 0)
301  {
302  uint requiredAlignment = layoutDesc.Elements[i].Kind == ResourceKind.UniformBuffer
303  ? _uniformBufferAlignment
304  : _structuredBufferAlignment;
305  uint desiredOffset = Unsafe.Add(ref dynamicOffsets, (int)dynamicOffsetIndex);
306  dynamicOffsetIndex += 1;
307  DeviceBufferRange range = Util.GetBufferRange(rs.Resources[i], desiredOffset);
308 
309  if ((range.Offset % requiredAlignment) != 0)
310  {
311  throw new VeldridException(
312  $"The effective offset of the buffer in slot {i} does not meet the alignment " +
313  $"requirements of this device. The offset must be a multiple of {requiredAlignment}, but it is " +
314  $"{range.Offset}");

1135  throw new VeldridException($"{nameof(layerCount)} must be greater than 0.");

1138  uint srcBlockSize = FormatHelpers.IsCompressedFormat(source.Format) ? 4u : 1u;
1139  uint roundedSrcWidth = (srcWidth + srcBlockSize - 1) / srcBlockSize * srcBlockSize;
1140  uint roundedSrcHeight = (srcHeight + srcBlockSize - 1) / srcBlockSize * srcBlockSize;
1141  if (srcX + width > roundedSrcWidth || srcY + height > roundedSrcHeight || srcZ + depth > srcDepth)
1142  {
1143  throw new VeldridException($"The given copy region is not valid for the source Texture.");
1144  }
1145  Util.GetMipDimensions(destination, dstMipLevel, out uint dstWidth, out uint dstHeight, out uint dstDepth);

1147  uint roundedDstWidth = (dstWidth + dstBlockSize - 1) / dstBlockSize * dstBlockSize;
1148  uint roundedDstHeight = (dstHeight + dstBlockSize - 1) / dstBlockSize * dstBlockSize;
1149  if (dstX + width > roundedDstWidth || dstY + height > roundedDstHeight || dstZ + depth > dstDepth)
1150  {
1151  throw new VeldridException($"The given copy region is not valid for the destination Texture.");
1152  }
1153  if (srcMipLevel >= source.MipLevels)
1154  {
1155  throw new VeldridException($"{nameof(srcMipLevel)} must be less than the number of mip levels in the source Texture.");
1156  }
