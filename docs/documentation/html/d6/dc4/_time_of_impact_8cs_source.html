71  public static void CalculateTimeOfImpact(ref ToiInput input, out ToiOutput output)
72  {
73  ++ToiCalls;
74 
75  output = new ToiOutput
76  {
77  State = ToiOutputState.Unknown,
78  T = input.Max
79  };
80 
81  Sweep sweepA = input.SweepA;
82  Sweep sweepB = input.SweepB;
83 
84  // Large rotations can make the root finder fail, so we normalize the
85  // sweep angles.
86  sweepA.Normalize();
87  sweepB.Normalize();
88 
89  float tMax = input.Max;
90 
91  float totalRadius = input.ProxyA.Radius + input.ProxyB.Radius;
92  float target = Math.Max(Settings.LinearSlop, totalRadius - 3.0f * Settings.LinearSlop);
93  float tolerance = 0.25f * Settings.LinearSlop;
94  Debug.Assert(target > tolerance);
95 
96  float t1 = 0.0f;
97  const int kMaxIterations = 20;
98  int iter = 0;
99 
100  // Prepare input for distance query.
101  DistanceInput distanceInput = new DistanceInput
102  {
103  ProxyA = input.ProxyA,
104  ProxyB = input.ProxyB,
105  UseRadii = false
106  };
107 
108  // The outer loop progressively attempts to compute new separating axes.
109  // This loop terminates when an axis is repeated (no progress is made).
110  for (;;)
111  {
112  sweepA.GetTransform(out Transform xfA, t1);
113  sweepB.GetTransform(out Transform xfB, t1);
114 
115  // Get the distance between shapes. We can also use the results
116  // to get a separating axis.
117  distanceInput.TransformA = xfA;
118  distanceInput.TransformB = xfB;
119  DistanceGjk.ComputeDistance(ref distanceInput, out DistanceOutput distanceOutput,
120  out SimplexCache cache);
121 
122  // If the shapes are overlapped, we give up on continuous collision.
123  if (distanceOutput.Distance <= 0.0f)
124  {
125  // Failure!
126  output.State = ToiOutputState.Overlapped;
127  output.T = 0.0f;
128  break;
129  }
130 
131  if (distanceOutput.Distance < target + tolerance)
132  {
133  // Victory!
134  output.State = ToiOutputState.Touching;
135  output.T = t1;
136  break;
137  }
138 
139  SeparationFunction.Initialize(ref cache, input.ProxyA, ref sweepA, input.ProxyB, ref sweepB, t1,
140  out Vector2F axis, out Vector2F localPoint, out SeparationFunctionType type);
141 
142  // Compute the TOI on the separating axis. We do this by successively
143  // resolving the deepest point. This loop is bounded by the number of vertices.
144  bool done = false;
145  float t2 = tMax;
146  int pushBackIter = 0;
147  for (;;)
148  {
149  // Find the deepest point at t2. Store the witness point indices.
150  float s2 = SeparationFunction.FindMinSeparation(out int indexA, out int indexB, t2, input.ProxyA,
151  ref sweepA, input.ProxyB, ref sweepB, ref axis, ref localPoint, type);
152 
153  // Is the final configuration separated?
154  if (s2 > target + tolerance)
155  {
156  // Victory!
157  output.State = ToiOutputState.Seperated;
158  output.T = tMax;
159  done = true;
160  break;
161  }
162 
163  // Has the separation reached tolerance?
164  if (s2 > target - tolerance)
165  {
166  // Advance the sweeps
167  t1 = t2;
168  break;
169  }
170 
171  // Compute the initial separation of the witness points.
172  float s1 = SeparationFunction.Evaluate(indexA, indexB, t1, input.ProxyA, ref sweepA, input.ProxyB,
173  ref sweepB, ref axis, ref localPoint, type);
174 
175  // Check for initial overlap. This might happen if the root finder
176  // runs out of iterations.
177  if (s1 < target - tolerance)
178  {
179  output.State = ToiOutputState.Failed;
180  output.T = t1;
181  done = true;
182  break;
183  }
184 
185  // Check for touching
186  if (s1 <= target + tolerance)
187  {
188  // Victory! t1 should hold the TOI (could be 0.0).
189  output.State = ToiOutputState.Touching;
190  output.T = t1;
191  done = true;
192  break;
193  }
194 
195  // Compute 1D root of: f(x) - target = 0
196  int rootIterCount = 0;
197  float a1 = t1, a2 = t2;
198  for (;;)
199  {
200  // Use a mix of the secant rule and bisection.
201  float t;
202  if ((rootIterCount & 1) != 0)
203  {
204  // Secant rule to improve convergence.
205  t = a1 + (target - s1) * (a2 - a1) / (s2 - s1);
206  }
207  else
208  {
209  // Bisection to guarantee progress.
210  t = 0.5f * (a1 + a2);
211  }
212 
213  ++rootIterCount;
214  ++ToiRootIters;
215 
216  float s = SeparationFunction.Evaluate(indexA, indexB, t, input.ProxyA, ref sweepA, input.ProxyB,
217  ref sweepB, ref axis, ref localPoint, type);
218 
219  if (Math.Abs(s - target) < tolerance)
220  {
221  // t2 holds a tentative value for t1
222  t2 = t;
223  break;
224  }
225 
226  // Ensure we continue to bracket the root.
227  if (s > target)
228  {
229  a1 = t;
230  s1 = s;
231  }
232  else
233  {
234  a2 = t;
235  s2 = s;
236  }
237 
238  if (rootIterCount == 50)
239  {
240  break;
241  }
242  }
243 
244  ToiMaxRootIters = Math.Max(ToiMaxRootIters, rootIterCount);
245 
246  ++pushBackIter;
247 
248  if (pushBackIter == Settings.PolygonVertices)
249  {
250  break;
251  }
252  }
253 
254  ++iter;
255  ++ToiIters;
256 
257  if (done)
258  {
259  break;
260  }
261 
262  if (iter == kMaxIterations)
263  {
264  // Root finder got stuck. Semi-victory.
265  output.State = ToiOutputState.Failed;
266  output.T = t1;
267  break;
268  }
269  }
270 
271  ToiMaxIters = Math.Max(ToiMaxIters, iter);
272  }
273  }
274 }
