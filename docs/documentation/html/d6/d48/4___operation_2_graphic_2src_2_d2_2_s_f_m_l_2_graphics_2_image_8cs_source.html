148  public Image(Color[,] pixels) :
149  base(IntPtr.Zero)
150  {
151  uint width = (uint) pixels.GetLength(0);
152  uint height = (uint) pixels.GetLength(1);
153 
154  // Transpose the array (.Net gives dimensions in reverse order of what SFML expects)
155  Color[,] transposed = new Color[height, width];
156  for (int x = 0; x < width; ++x)
157  {
158  for (int y = 0; y < height; ++y)
159  {
160  transposed[y, x] = pixels[x, y];
161  }
162  }
163 
164  unsafe
165  {
166  fixed (Color* pixelsPtr = transposed)
167  {
168  CPointer = sfImage_createFromPixels(width, height, (byte*) pixelsPtr);
169  }
170  }
171 
172  if (CPointer == IntPtr.Zero)
173  {
174  throw new LoadingFailedException("image");
175  }
176  }
