636  private static T _<T>() where T : class
637  {
638  string method = "gl" + typeof(T).Name;
639  IntPtr ptr = SDL_GL_GetProcAddress(method);
640  if (ptr == IntPtr.Zero)
641  {
642  throw new Exception($"nogo: {method} from {typeof(T).Name}");
643  }
644 
645  return Marshal.GetDelegateForFunctionPointer(ptr, typeof(T)) as T;
646  }
647 
652  private static T _Del<T>() where T : class
653  {
654  string method = "gl" + typeof(T).Name.Substring(0, typeof(T).Name.Length - 3);
655  IntPtr ptr = SDL_GL_GetProcAddress(method);
656  if (ptr == IntPtr.Zero)
657  {
658  throw new Exception($"nogo: {method} from {typeof(T).Name}");
659  }
660 
661  return Marshal.GetDelegateForFunctionPointer(ptr, typeof(T)) as T;
662  }
