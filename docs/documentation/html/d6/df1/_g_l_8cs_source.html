337  private static T _<T>(string command) where T : class
338  {
339  IntPtr ptr = Sdl.GlGetProcAddress(command);
340  if (ptr == IntPtr.Zero)
341  {
342  throw new ExternalException($"{command} from {typeof(T).Name}");
343  }
344 
345  return Marshal.GetDelegateForFunctionPointer(ptr, typeof(T)) as T;
346  }

353  public static string GlGetString(StringName pname)
354  {
355  IntPtr ptr = GetString(pname);
356  if (ptr == IntPtr.Zero)
357  {
358  return string.Empty;
359  }
360 
361  int length = 0;
362  while (Marshal.ReadByte(ptr, length) != 0)
363  {
364  length++;
365  }
366 
367  byte[] buffer = new byte[length];
368  Marshal.Copy(ptr, buffer, 0, length);
369 
370  return Encoding.ASCII.GetString(buffer);
371  }
