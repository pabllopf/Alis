54  public static bool RayCastEdge(ref Vector2 start, ref Vector2 end, bool oneSided, ref RayCastInput input,
55  ref Transform transform, out RayCastOutput output)
56  {
57  // p = p1 + t * d
58  // v = v1 + s * e
59  // p1 + t * d = v1 + s * e
60  // s * e - t * d = p1 - v1
61 
62  output = new RayCastOutput();
63 
64  // Put the ray into the edge's frame of reference.
65  Vector2 p1 = MathUtils.MulT(transform.Q, input.Point1 - transform.P);
66  Vector2 p2 = MathUtils.MulT(transform.Q, input.Point2 - transform.P);
67  Vector2 d = p2 - p1;
68 
69  Vector2 v1 = start;
70  Vector2 v2 = end;
71  Vector2 e = v2 - v1;
72 
73  // Normal points to the right, looking from v1 at v2
74  Vector2 normal = new Vector2(e.Y, -e.X);
75  normal = Vector2.Normalize(normal);
76 
77  // q = p1 + t * d
78  // dot(normal, q - v1) = 0
79  // dot(normal, p1 - v1) + t * dot(normal, d) = 0
80  float numerator = Vector2.Dot(normal, v1 - p1);
81  if (oneSided && numerator > 0.0f)
82  {
83  return false;
84  }
85 
86  float denominator = Vector2.Dot(normal, d);
87 
88  if (denominator == 0.0f)
89  {
90  return false;
91  }
92 
93  float t = numerator / denominator;
94  if (t < 0.0f || input.MaxFraction < t)
95  {
96  return false;
97  }
98 
99  Vector2 q = p1 + t * d;
100 
101  // q = v1 + s * r
102  // s = dot(q - v1, r) / dot(r, r)
103  Vector2 r = v2 - v1;
104  float rr = Vector2.Dot(r, r);
105  if (rr == 0.0f)
106  {
107  return false;
108  }
109 
110  float s = Vector2.Dot(q - v1, r) / rr;
111  if (s < 0.0f || 1.0f < s)
112  {
113  return false;
114  }
115 
116  output.Fraction = t;
117  if (numerator > 0.0f)
118  {
119  output.Normal = -MathUtils.MulT(transform.Q, normal);
120  }
121  else
122  {
123  output.Normal = MathUtils.MulT(transform.Q, normal);
124  }
125 
126  return true;
127  }

138  public static bool RayCastCircle(ref Vector2 pos, float radius, ref RayCastInput input, ref Transform transform,
139  out RayCastOutput output)
140  {
141  // Collision Detection in Interactive 3D Environments by Gino van den Bergen
142  // From Section 3.1.2
143  // x = s + a * r
144  // norm(x) = radius
145 
146  output = new RayCastOutput();
147 
148  Vector2 position = transform.P + MathUtils.Mul(transform.Q, pos);
149  Vector2 s = input.Point1 - position;
150  float b = Vector2.Dot(s, s) - radius * radius;
151 
152  // Solve quadratic equation.
153  Vector2 r = input.Point2 - input.Point1;
154  float c = Vector2.Dot(s, r);
155  float rr = Vector2.Dot(r, r);
156  float sigma = c * c - rr * b;
157 
158  // Check for negative discriminant and short segment.
159  if (sigma < 0.0f || rr < MathConstants.Epsilon)
160  {
161  return false;
162  }
163 
164  // Find the point of intersection of the line with the circle.
165  float a = -(c + (float) Math.Sqrt(sigma));
166 
167  // Is the intersection point on the segment?
168  if (0.0f <= a && a <= input.MaxFraction * rr)
169  {
170  a /= rr;
171  output.Fraction = a;
172  output.Normal = s + a * r;
173  output.Normal = Vector2.Normalize(output.Normal);
174  return true;
175  }
176 
177  return false;
178  }

189  public static bool RayCastPolygon(Vertices vertices, Vertices normals, ref RayCastInput input,
190  ref Transform transform, out RayCastOutput output)
191  {
192  output = new RayCastOutput();
193 
194  // Put the ray into the polygon's frame of reference.
195  Vector2 p1 = MathUtils.MulT(transform.Q, input.Point1 - transform.P);
196  Vector2 p2 = MathUtils.MulT(transform.Q, input.Point2 - transform.P);
197  Vector2 d = p2 - p1;
198 
199  float lower = 0.0f, upper = input.MaxFraction;
200 
201  int index = -1;
202 
203  for (int i = 0; i < vertices.Count; ++i)
204  {
205  // p = p1 + a * d
206  // dot(normal, p - v) = 0
207  // dot(normal, p1 - v) + a * dot(normal, d) = 0
208  float numerator = Vector2.Dot(normals[i], vertices[i] - p1);
209  float denominator = Vector2.Dot(normals[i], d);
210 
211  if (denominator == 0.0f)
212  {
213  if (numerator < 0.0f)
214  {
215  return false;
216  }
217  }
218  else
219  {
220  // Note: we want this predicate without division:
221  // lower < numerator / denominator, where denominator < 0
222  // Since denominator < 0, we have to flip the inequality:
223  // lower < numerator / denominator <==> denominator * lower > numerator.
224  if (denominator < 0.0f && numerator < lower * denominator)
225  {
226  // Increase lower.
227  // The segment enters this half-space.
228  lower = numerator / denominator;
229  index = i;
230  }
231  else if (denominator > 0.0f && numerator < upper * denominator)
232  {
233  // Decrease upper.
234  // The segment exits this half-space.
235  upper = numerator / denominator;
236  }
237  }
238 
239  // The use of epsilon here causes the assert on lower to trip
240  // in some cases. Apparently the use of epsilon was to make edge
241  // shapes work, but now those are handled separately.
242  //if (upper < lower - b2_epsilon)
243  if (upper < lower)
244  {
245  return false;
246  }
247  }
248 
249  Debug.Assert(0.0f <= lower && lower <= input.MaxFraction);
250 
251  if (index >= 0)
252  {
253  output.Fraction = lower;
254  output.Normal = MathUtils.Mul(transform.Q, normals[index]);
255  return true;
256  }
257 
258  return false;
259  }
260  }
