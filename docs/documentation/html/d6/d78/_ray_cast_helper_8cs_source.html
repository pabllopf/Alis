56  public static bool RayCastEdge(ref Vector2 start, ref Vector2 end, bool oneSided, ref RayCastInput input,
57  ref Transform transform, out RayCastOutput output)
58  {
59  // p = p1 + t * d
60  // v = v1 + s * e
61  // p1 + t * d = v1 + s * e
62  // s * e - t * d = p1 - v1
63 
64  output = new RayCastOutput();
65 
66  // Put the ray into the edge's frame of reference.
67  Vector2 p1 = MathUtils.MulT(transform.Rotation, input.Point1 - transform.Position);
68  Vector2 p2 = MathUtils.MulT(transform.Rotation, input.Point2 - transform.Position);
69  Vector2 d = p2 - p1;
70 
71  Vector2 v1 = start;
72  Vector2 v2 = end;
73  Vector2 e = v2 - v1;
74 
75  // Normal points to the right, looking from v1 at v2
76  Vector2 normal = new Vector2(e.Y, -e.X);
77  normal = Vector2.Normalize(normal);
78 
79  // q = p1 + t * d
80  // dot(normal, q - v1) = 0
81  // dot(normal, p1 - v1) + t * dot(normal, d) = 0
82  float numerator = Vector2.Dot(normal, v1 - p1);
83  if (oneSided && (numerator > 0.0f))
84  {
85  return false;
86  }
87 
88  float denominator = Vector2.Dot(normal, d);
89 
90  if (denominator == 0.0f)
91  {
92  return false;
93  }
94 
95  float t = numerator / denominator;
96  if (t < 0.0f || input.Fraction < t)
97  {
98  return false;
99  }
100 
101  Vector2 q = p1 + t * d;
102 
103  // q = v1 + s * r
104  // s = dot(q - v1, r) / dot(r, r)
105  Vector2 r = v2 - v1;
106  float rr = Vector2.Dot(r, r);
107  if (rr == 0.0f)
108  {
109  return false;
110  }
111 
112  float s = Vector2.Dot(q - v1, r) / rr;
113  if (s < 0.0f || 1.0f < s)
114  {
115  return false;
116  }
117 
118  output.Fraction = t;
119  if (numerator > 0.0f)
120  {
121  output.Normal = -MathUtils.MulT(transform.Rotation, normal);
122  }
123  else
124  {
125  output.Normal = MathUtils.MulT(transform.Rotation, normal);
126  }
127 
128  return true;
129  }

140  public static bool RayCastCircle(ref Vector2 pos, float radius, ref RayCastInput input, ref Transform transform,
141  out RayCastOutput output)
142  {
143  // Collision Detection in Interactive 3D Environments by Gino van den Bergen
144  // From Section 3.1.2
145  // x = s + a * r
146  // norm(x) = radius
147 
148  output = new RayCastOutput();
149 
150  Vector2 position = transform.Position + MathUtils.Mul(transform.Rotation, pos);
151  Vector2 s = input.Point1 - position;
152  float b = Vector2.Dot(s, s) - radius * radius;
153 
154  // Solve quadratic equation.
155  Vector2 r = input.Point2 - input.Point1;
156  float c = Vector2.Dot(s, r);
157  float rr = Vector2.Dot(r, r);
158  float sigma = c * c - rr * b;
159 
160  // Check for negative discriminant and short segment.
161  if (sigma < 0.0f || rr < Constant.Epsilon)
162  {
163  return false;
164  }
165 
166  // Find the point of intersection of the line with the circle.
167  float a = -(c + (float) Math.Sqrt(sigma));
168 
169  // Is the intersection point on the segment?
170  if ((0.0f <= a) && (a <= input.Fraction * rr))
171  {
172  a /= rr;
173  output.Fraction = a;
174  output.Normal = s + a * r;
175  output.Normal = Vector2.Normalize(output.Normal);
176  return true;
177  }
178 
179  return false;
180  }

191  public static bool RayCastPolygon(Vertices vertices, Vertices normals, ref RayCastInput input,
192  ref Transform transform, out RayCastOutput output)
193  {
194  output = new RayCastOutput();
195 
196  // Put the ray into the polygon's frame of reference.
197  Vector2 p1 = MathUtils.MulT(transform.Rotation, input.Point1 - transform.Position);
198  Vector2 p2 = MathUtils.MulT(transform.Rotation, input.Point2 - transform.Position);
199  Vector2 d = p2 - p1;
200 
201  float lower = 0.0f, upper = input.Fraction;
202 
203  int index = -1;
204 
205  for (int i = 0; i < vertices.Count; ++i)
206  {
207  // p = p1 + a * d
208  // dot(normal, p - v) = 0
209  // dot(normal, p1 - v) + a * dot(normal, d) = 0
210  float numerator = Vector2.Dot(normals[i], vertices[i] - p1);
211  float denominator = Vector2.Dot(normals[i], d);
212 
213  if (denominator == 0.0f)
214  {
215  if (numerator < 0.0f)
216  {
217  return false;
218  }
219  }
220  else
221  {
222  // Note: we want this predicate without division:
223  // lower < numerator / denominator, where denominator < 0
224  // Since denominator < 0, we have to flip the inequality:
225  // lower < numerator / denominator <==> denominator * lower > numerator.
226  if ((denominator < 0.0f) && (numerator < lower * denominator))
227  {
228  // Increase lower.
229  // The segment enters this half-space.
230  lower = numerator / denominator;
231  index = i;
232  }
233  else if ((denominator > 0.0f) && (numerator < upper * denominator))
234  {
235  // Decrease upper.
236  // The segment exits this half-space.
237  upper = numerator / denominator;
238  }
239  }
240 
241  // The use of epsilon here causes the assert on lower to trip
242  // in some cases. Apparently the use of epsilon was to make edge
243  // shapes work, but now those are handled separately.
244  //if (upper < lower - b2_epsilon)
245  if (upper < lower)
246  {
247  return false;
248  }
249  }
250 
251  Debug.Assert((0.0f <= lower) && (lower <= input.Fraction));
252 
253  if (index >= 0)
254  {
255  output.Fraction = lower;
256  output.Normal = MathUtils.Mul(transform.Rotation, normals[index]);
257  return true;
258  }
259 
260  return false;
261  }
262  }
