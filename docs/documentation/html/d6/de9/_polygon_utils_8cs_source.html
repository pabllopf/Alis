90  public static Vertices CreateRoundedRectangle(float width, float height, float xRadius, float yRadius,
91  int segments)
92  {
93  if (yRadius > height / 2 || xRadius > width / 2)
94  {
95  throw new Exception("Rounding amount can't be more than half the height and width respectively.");
96  }
97 
98  if (segments < 0)
99  {
100  throw new Exception("Segments must be zero or more.");
101  }
102 
103  Vertices vertices = new Vertices();
104  if (segments == 0)
105  {
106  vertices.Add(new Vector2(width * .5f - xRadius, -height * .5f));
107  vertices.Add(new Vector2(width * .5f, -height * .5f + yRadius));
108 
109  vertices.Add(new Vector2(width * .5f, height * .5f - yRadius));
110  vertices.Add(new Vector2(width * .5f - xRadius, height * .5f));
111 
112  vertices.Add(new Vector2(-width * .5f + xRadius, height * .5f));
113  vertices.Add(new Vector2(-width * .5f, height * .5f - yRadius));
114 
115  vertices.Add(new Vector2(-width * .5f, -height * .5f + yRadius));
116  vertices.Add(new Vector2(-width * .5f + xRadius, -height * .5f));
117  }
118  else
119  {
120  int numberOfEdges = segments * 4 + 8;
121 
122  float stepSize = MathConstants.TwoPi / (numberOfEdges - 4);
123  int perPhase = numberOfEdges / 4;
124 
125  Vector2 posOffset = new Vector2(width / 2 - xRadius, height / 2 - yRadius);
126  vertices.Add(posOffset + new Vector2(xRadius, -yRadius + yRadius));
127  short phase = 0;
128  for (int i = 1; i < numberOfEdges; i++)
129  {
130  if (i - perPhase == 0 || i - perPhase * 3 == 0)
131  {
132  posOffset.X *= -1;
133  phase--;
134  }
135  else if (i - perPhase * 2 == 0)
136  {
137  posOffset.Y *= -1;
138  phase--;
139  }
140 
141  vertices.Add(posOffset + new Vector2(xRadius * (float) Math.Cos(stepSize * -(i + phase)),
142  -yRadius * (float) Math.Sin(stepSize * -(i + phase))));
143  }
144  }
145 
146  return vertices;
147  }

175  public static Vertices CreateEllipse(float xRadius, float yRadius, int numberOfEdges)
176  {
177  Vertices vertices = new Vertices();
178 
179  float stepSize = MathConstants.TwoPi / numberOfEdges;
180 
181  vertices.Add(new Vector2(xRadius, 0));
182  for (int i = numberOfEdges - 1; i > 0; --i)
183  {
184  vertices.Add(new Vector2(xRadius * (float) Math.Cos(stepSize * i),
185  -yRadius * (float) Math.Sin(stepSize * i)));
186  }
187 
188  return vertices;

248  public static Vertices CreateCapsule(float height, float topRadius, int topEdges, float bottomRadius,
249  int bottomEdges)
250  {
251  if (height <= 0)
252  {
253  throw new ArgumentException("Height must be longer than 0", nameof(height));
254  }
255 
256  if (topRadius <= 0)
257  {
258  throw new ArgumentException("The top radius must be more than 0", nameof(topRadius));
259  }
260 
261  if (topEdges <= 0)
262  {
263  throw new ArgumentException("Top edges must be more than 0", nameof(topEdges));
264  }
265 
266  if (bottomRadius <= 0)
267  {
268  throw new ArgumentException("The bottom radius must be more than 0", nameof(bottomRadius));
269  }
270 
271  if (bottomEdges <= 0)
272  {
273  throw new ArgumentException("Bottom edges must be more than 0", nameof(bottomEdges));
274  }
275 
276  if (topRadius >= height / 2)
277  {
278  throw new ArgumentException(
279  "The top radius must be lower than height / 2. Higher values of top radius would create a circle, and not a half circle.",
280  nameof(topRadius));
281  }
282 
283  if (bottomRadius >= height / 2)
284  {
285  throw new ArgumentException(
286  "The bottom radius must be lower than height / 2. Higher values of bottom radius would create a circle, and not a half circle.",
287  nameof(bottomRadius));
288  }
289 
290  Vertices vertices = new Vertices();
291 
292  float newHeight = (height - topRadius - bottomRadius) * 0.5f;
293 
294  // top
295  vertices.Add(new Vector2(topRadius, newHeight));
296 
297  float stepSize = MathConstants.Pi / topEdges;
298  for (int i = 1; i < topEdges; i++)
299  {
300  vertices.Add(new Vector2(topRadius * (float) Math.Cos(stepSize * i),
301  topRadius * (float) Math.Sin(stepSize * i) + newHeight));
302  }
303 
304  vertices.Add(new Vector2(-topRadius, newHeight));
305 
306  // bottom
307  vertices.Add(new Vector2(-bottomRadius, -newHeight));
308 
309  stepSize = MathConstants.Pi / bottomEdges;
310  for (int i = 1; i < bottomEdges; i++)
311  {
312  vertices.Add(new Vector2(-bottomRadius * (float) Math.Cos(stepSize * i),
313  -bottomRadius * (float) Math.Sin(stepSize * i) - newHeight));
314  }
315 
316  vertices.Add(new Vector2(bottomRadius, -newHeight));
317 
318  return vertices;
319  }

326  public static Vertices CreateGear(float radius, int numberOfTeeth, float tipPercentage, float toothHeight)
327  {
328  Vertices vertices = new Vertices();
329 
330  float stepSize = MathConstants.TwoPi / numberOfTeeth;
331  tipPercentage /= 100f;
332  MathHelper.Clamp(tipPercentage, 0f, 1f);
333  float toothTipStepSize = stepSize / 2f * tipPercentage;
334 
335  float toothAngleStepSize = (stepSize - toothTipStepSize * 2f) / 2f;
336 
337  for (int i = numberOfTeeth - 1; i >= 0; --i)
338  {
339  if (toothTipStepSize > 0f)
340  {
341  vertices.Add(
342  new Vector2(radius *
343  (float) Math.Cos(stepSize * i + toothAngleStepSize * 2f + toothTipStepSize),
344  -radius *
345  (float) Math.Sin(stepSize * i + toothAngleStepSize * 2f + toothTipStepSize)));
346 
347  vertices.Add(
348  new Vector2((radius + toothHeight) *
349  (float) Math.Cos(stepSize * i + toothAngleStepSize + toothTipStepSize),
350  -(radius + toothHeight) *
351  (float) Math.Sin(stepSize * i + toothAngleStepSize + toothTipStepSize)));
352  }
353 
354  vertices.Add(new Vector2((radius + toothHeight) *
355  (float) Math.Cos(stepSize * i + toothAngleStepSize),
356  -(radius + toothHeight) *
357  (float) Math.Sin(stepSize * i + toothAngleStepSize)));
358 
359  vertices.Add(new Vector2(radius * (float) Math.Cos(stepSize * i),
360  -radius * (float) Math.Sin(stepSize * i)));
361  }
362 
363  return vertices;
364  }
