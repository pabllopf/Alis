336  {
337  terminatorWritePtr = null;
338  if (!_currentBlock.Alloc(size, out void* ptr))
339  {
340  int currentBlockIndex = _blocks.IndexOf(_currentBlock);
341  bool anyWorked = false;
342  for (int i = currentBlockIndex + 1; i < _blocks.Count; i++)
343  {
344  EntryStorageBlock nextBlock = _blocks[i];
345  if (nextBlock.Alloc(size, out ptr))
346  {
