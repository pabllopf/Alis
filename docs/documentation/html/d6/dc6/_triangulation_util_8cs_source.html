153  public static bool InScanArea(TriangulationPoint pa, TriangulationPoint pb, TriangulationPoint pc,
154  TriangulationPoint pd)
155  {
156  double oadb = (pa.X - pb.X) * (pd.Y - pb.Y) - (pd.X - pb.X) * (pa.Y - pb.Y);
157  if (oadb >= -Epsilon)
158  {
159  return false;
160  }
161 
162  double oadc = (pa.X - pc.X) * (pd.Y - pc.Y) - (pd.X - pc.X) * (pa.Y - pc.Y);
163  if (oadc <= Epsilon)
164  {
165  return false;
166  }
167 
168  return true;
169  }
