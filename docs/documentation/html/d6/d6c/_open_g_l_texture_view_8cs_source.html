263  private void CreateGLResources()
264  {
265  if (!_needsTextureView)
266  {
267  TextureTarget = Target.TextureTarget;
268  return;
269  }
270 
271  glGenTextures(1, out _textureView);
272  CheckLastError();
273 
274  TextureTarget originalTarget = Target.TextureTarget;
275  var effectiveArrayLayers = ArrayLayers;
276  if (originalTarget == TextureTarget.Texture1D)
277  {
278  TextureTarget = TextureTarget.Texture1D;
279  }
280  else if (originalTarget == TextureTarget.Texture1DArray)
281  {
282  if (ArrayLayers > 1)
283  {
284  TextureTarget = TextureTarget.Texture1DArray;
285  }
286  else
287  {
288  TextureTarget = TextureTarget.Texture1D;
289  }
290  }
291  else if (originalTarget == TextureTarget.Texture2D)
292  {
293  TextureTarget = TextureTarget.Texture2D;
294  }
295  else if (originalTarget == TextureTarget.Texture2DArray)
296  {
297  if (ArrayLayers > 1)
298  {
299  TextureTarget = TextureTarget.Texture2DArray;
300  }
301  else
302  {
303  TextureTarget = TextureTarget.Texture2D;
304  }
305  }
306  else if (originalTarget == TextureTarget.Texture2DMultisample)
307  {
308  TextureTarget = TextureTarget.Texture2DMultisample;
309  }
310  else if (originalTarget == TextureTarget.Texture2DMultisampleArray)
311  {
312  if (ArrayLayers > 1)
313  {
314  TextureTarget = TextureTarget.Texture2DMultisampleArray;
315  }
316  else
317  {
318  TextureTarget = TextureTarget.Texture2DMultisample;
319  }
320  }
321  else if (originalTarget == TextureTarget.Texture3D)
322  {
323  TextureTarget = TextureTarget.Texture3D;
324  }
325  else if (originalTarget == TextureTarget.TextureCubeMap)
326  {
327  if (ArrayLayers > 1)
328  {
329  TextureTarget = TextureTarget.TextureCubeMap;
330  effectiveArrayLayers *= 6;
331  }
332  else
333  {
334  TextureTarget = TextureTarget.TextureCubeMapArray;
335  effectiveArrayLayers *= 6;
336  }
337  }
338  else
339  {
340  throw new VeldridException("The given TextureView parameters are not supported with the OpenGL backend.");
341  }
342 
343  PixelInternalFormat internalFormat = (PixelInternalFormat)OpenGLFormats.VdToGLSizedInternalFormat(
344  Format,
345  (Target.Usage & TextureUsage.DepthStencil) == TextureUsage.DepthStencil);
346  Debug.Assert(Target.Created);
347  glTextureView(
348  _textureView,
349  TextureTarget,
350  Target.Texture,
351  internalFormat,
352  BaseMipLevel,
353  MipLevels,
354  BaseArrayLayer,
355  effectiveArrayLayers);
356  CheckLastError();
357  }
