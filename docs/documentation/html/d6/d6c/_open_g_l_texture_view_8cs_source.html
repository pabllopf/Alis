262  private void CreateGLResources()
263  {
264  if (!_needsTextureView)
265  {
266  TextureTarget = Target.TextureTarget;
267  return;
268  }
269 
270  glGenTextures(1, out _textureView);
271  CheckLastError();
272 
273  TextureTarget originalTarget = Target.TextureTarget;
274  var effectiveArrayLayers = ArrayLayers;
275  if (originalTarget == TextureTarget.Texture1D)
276  {
277  TextureTarget = TextureTarget.Texture1D;
278  }
279  else if (originalTarget == TextureTarget.Texture1DArray)
280  {
281  if (ArrayLayers > 1)
282  {
283  TextureTarget = TextureTarget.Texture1DArray;
284  }
285  else
286  {
287  TextureTarget = TextureTarget.Texture1D;
288  }
289  }
290  else if (originalTarget == TextureTarget.Texture2D)
291  {
292  TextureTarget = TextureTarget.Texture2D;
293  }
294  else if (originalTarget == TextureTarget.Texture2DArray)
295  {
296  if (ArrayLayers > 1)
297  {
298  TextureTarget = TextureTarget.Texture2DArray;
299  }
300  else
301  {
302  TextureTarget = TextureTarget.Texture2D;
303  }
304  }
305  else if (originalTarget == TextureTarget.Texture2DMultisample)
306  {
307  TextureTarget = TextureTarget.Texture2DMultisample;
308  }
309  else if (originalTarget == TextureTarget.Texture2DMultisampleArray)
310  {
311  if (ArrayLayers > 1)
312  {
313  TextureTarget = TextureTarget.Texture2DMultisampleArray;
314  }
315  else
316  {
317  TextureTarget = TextureTarget.Texture2DMultisample;
318  }
319  }
320  else if (originalTarget == TextureTarget.Texture3D)
321  {
322  TextureTarget = TextureTarget.Texture3D;
323  }
324  else if (originalTarget == TextureTarget.TextureCubeMap)
325  {
326  if (ArrayLayers > 1)
327  {
328  TextureTarget = TextureTarget.TextureCubeMap;
329  effectiveArrayLayers *= 6;
330  }
331  else
332  {
333  TextureTarget = TextureTarget.TextureCubeMapArray;
334  effectiveArrayLayers *= 6;
335  }
336  }
337  else
338  {
339  throw new VeldridException("The given TextureView parameters are not supported with the OpenGL backend.");
340  }
341 
342  PixelInternalFormat internalFormat = (PixelInternalFormat)OpenGLFormats.VdToGLSizedInternalFormat(
343  Format,
344  (Target.Usage & TextureUsage.DepthStencil) == TextureUsage.DepthStencil);
345  Debug.Assert(Target.Created);
346  glTextureView(
347  _textureView,
348  TextureTarget,
349  Target.Texture,
350  internalFormat,
351  BaseMipLevel,
352  MipLevels,
353  BaseArrayLayer,
354  effectiveArrayLayers);
355  CheckLastError();
356  }
