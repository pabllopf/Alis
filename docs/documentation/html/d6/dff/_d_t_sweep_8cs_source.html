70  public static void Triangulate(DtSweepContext tcx)
71  {
72  tcx.CreateAdvancingFront();
73 
74  Sweep(tcx);
75 
76  // Finalize triangulation
77  if (tcx.TriangulationMode == TriangulationMode.Polygon)
78  {

80  }
81  else

90  private static void Sweep(DtSweepContext tcx)
91  {
92  List<TriangulationPoint> points = tcx.Points;
93 
94  for (int i = 1; i < points.Count; i++)
95  {
96  TriangulationPoint point = points[i];
97 
98  AdvancingFrontNode node = PointEvent(tcx, point);
99 
100  if (point.HasEdges)
101  {
102  foreach (DtSweepConstraint e in point.Edges)
103  {
104  EdgeEvent(tcx, e, node);
105  }
106  }
107 
108  tcx.Update(null);
109  }
110  }

113  private static void FinalizationConvexHull(DtSweepContext tcx)
114  {
115  DelaunayTriangle t1, t2;
116 
117  AdvancingFrontNode n1 = tcx.AFront.Head.Next;
118  AdvancingFrontNode n2 = n1.Next;
119 
120  TurnAdvancingFrontConvex(tcx, n1, n2);
121 
122  n1 = tcx.AFront.Tail.Prev;
123  if (n1.Triangle.Contains(n1.Next.Point) && n1.Triangle.Contains(n1.Prev.Point))
124  {
125  t1 = n1.Triangle.NeighborAcross(n1.Point);
126  RotateTrianglePair(n1.Triangle, n1.Point, t1, t1.OppositePoint(n1.Triangle, n1.Point));
127  tcx.MapTriangleToNodes(n1.Triangle);
128  tcx.MapTriangleToNodes(t1);
129  }
130 
131  n1 = tcx.AFront.Head.Next;
132  if (n1.Triangle.Contains(n1.Prev.Point) && n1.Triangle.Contains(n1.Next.Point))
133  {
134  t1 = n1.Triangle.NeighborAcross(n1.Point);
135  RotateTrianglePair(n1.Triangle, n1.Point, t1, t1.OppositePoint(n1.Triangle, n1.Point));
136  tcx.MapTriangleToNodes(n1.Triangle);
137  tcx.MapTriangleToNodes(t1);
138  }
139 
140  // Lower right boundary
141  TriangulationPoint first = tcx.AFront.Head.Point;
142  n2 = tcx.AFront.Tail.Prev;
143  t1 = n2.Triangle;
144  TriangulationPoint p1 = n2.Point;
145  n2.Triangle = null;
146  do
147  {
148  tcx.RemoveFromList(t1);
149  p1 = t1.PointCcw(p1);
150  if (p1 == first)
151  {
152  break;
153  }
154 
155  t2 = t1.NeighborCcw(p1);
156  t1.Clear();
157  t1 = t2;
158  } while (true);
159 
160  // Lower left boundary
161  first = tcx.AFront.Head.Next.Point;
162  p1 = t1.PointCw(tcx.AFront.Head.Point);
163  t2 = t1.NeighborCw(tcx.AFront.Head.Point);
164  t1.Clear();
165  t1 = t2;
166  while (p1 != first)
167  {
168  tcx.RemoveFromList(t1);
169  p1 = t1.PointCcw(p1);
170  t2 = t1.NeighborCcw(p1);
171  t1.Clear();
172  t1 = t2;
173  }
174 
175  // Remove current head and tail node now that we have removed all triangles attached
176  // to them. Then set new head and tail node points
177  tcx.AFront.Head = tcx.AFront.Head.Next;
178  tcx.AFront.Head.Prev = null;
179  tcx.AFront.Tail = tcx.AFront.Tail.Prev;
180  tcx.AFront.Tail.Next = null;
181 
182  tcx.FinalizeTriangulation();
183  }

186  private static void TurnAdvancingFrontConvex(DtSweepContext tcx, AdvancingFrontNode b, AdvancingFrontNode c)
187  {
188  AdvancingFrontNode first = b;
189  while (c != tcx.AFront.Tail)
190  {
191  if (TriangulationUtil.Orient2d(b.Point, c.Point, c.Next.Point) == Orientation.Ccw)
192  {
193  // [b,c,d] Concave - fill around c
194  Fill(tcx, c);
195  c = c.Next;
196  }
197  else
198  {
199  // [b,c,d] Convex
200  if ((b != first) && (TriangulationUtil.Orient2d(b.Prev.Point, b.Point, c.Point) == Orientation.Ccw))
201  {
202  // [a,b,c] Concave - fill around b
203  Fill(tcx, b);
204  b = b.Prev;
205  }
206  else
207  {
208  // [a,b,c] Convex - nothing to fill
209  b = c;
210  c = c.Next;
211  }
212  }
213  }

238  private static AdvancingFrontNode PointEvent(DtSweepContext tcx, TriangulationPoint point)
239  {
240  AdvancingFrontNode node = tcx.LocateNode(point);
241  AdvancingFrontNode newNode = NewFrontTriangle(tcx, point, node);
242 
243  // Only need to check +epsilon since point never have smaller
244  // x value than node due to how we fetch nodes from the front
245  if (point.X <= node.Point.X + TriangulationUtil.Epsilon)
246  {
247  Fill(tcx, node);
248  }

251 
252  FillAdvancingFront(tcx, newNode);

289  {

568  private static void EdgeEvent(DtSweepContext tcx, TriangulationPoint ep, TriangulationPoint eq,
569  DelaunayTriangle triangle, TriangulationPoint point)
570  {
571  if (IsEdgeSideOfTriangle(triangle, ep, eq))
572  {
573  return;
574  }
575 
576  TriangulationPoint p1 = triangle.PointCcw(point);
577  Orientation o1 = TriangulationUtil.Orient2d(eq, p1, ep);
578  if (o1 == Orientation.Collinear)
579  {
580  if (triangle.Contains(eq, p1))
581  {
582  triangle.MarkConstrainedEdge(eq, p1);
583 
584  // We are modifying the constraint maybe it would be better to
585  // not change the given constraint and just keep a variable for the new constraint
586  tcx.EdgeEvent.ConstrainedEdge.Q = p1;
587  triangle = triangle.NeighborAcross(point);
588  EdgeEvent(tcx, ep, p1, triangle, p1);
589  }
590  else
591  {
592  throw new PointOnEdgeException("EdgeEvent - Point on constrained edge not supported yet");
593  }
594 
595  if (tcx.IsDebugEnabled)
596  {
597  Debug.WriteLine("EdgeEvent - Point on constrained edge");
598  }
599 
600  return;
601  }
602 
603  TriangulationPoint p2 = triangle.PointCw(point);
604  Orientation o2 = TriangulationUtil.Orient2d(eq, p2, ep);
605  if (o2 == Orientation.Collinear)
606  {
607  if (triangle.Contains(eq, p2))
608  {
609  triangle.MarkConstrainedEdge(eq, p2);
610 
611  // We are modifying the constraint maybe it would be better to
612  // not change the given constraint and just keep a variable for the new constraint
613  tcx.EdgeEvent.ConstrainedEdge.Q = p2;
614  triangle = triangle.NeighborAcross(point);
615  EdgeEvent(tcx, ep, p2, triangle, p2);
616  }
617  else
618  {
619  throw new PointOnEdgeException("EdgeEvent - Point on constrained edge not supported yet");
620  }
621 
622  if (tcx.IsDebugEnabled)
623  {
624  Debug.WriteLine("EdgeEvent - Point on constrained edge");
625  }
626 
627  return;
628  }
629 
630  if (o1 == o2)
631  {
632  // Need to decide if we are rotating CW or CCW to get to a triangle
633  // that will cross edge
634  if (o1 == Orientation.Cw)
635  {
636  triangle = triangle.NeighborCcw(point);
637  }
638  else
639  {
640  triangle = triangle.NeighborCw(point);
641  }
642 
643  EdgeEvent(tcx, ep, eq, triangle, point);
644  }
645  else
646  {
647  // This triangle crosses constraint so lets flippin start!
648  FlipEdgeEvent(tcx, ep, eq, triangle, point);
649  }
650  }

661  private static void FlipEdgeEvent(DtSweepContext tcx, TriangulationPoint ep, TriangulationPoint eq,
662  DelaunayTriangle t, TriangulationPoint p)
663  {
664  DelaunayTriangle ot = t.NeighborAcross(p);
665 
666 
667  if (t.GetConstrainedEdgeAcross(p))
668  {
669  throw new Exception("Intersecting Constraints");
670  }
671 
672  TriangulationPoint op = ot.OppositePoint(t, p);
673 
674  bool inScanArea = TriangulationUtil.InScanArea(p, t.PointCcw(p), t.PointCw(p), op);
675  if (inScanArea)
676  {
677  // Lets rotate shared edge one vertex CW
678  RotateTrianglePair(t, p, ot, op);
679  tcx.MapTriangleToNodes(t);
680  tcx.MapTriangleToNodes(ot);
681 
682  if ((p == eq) && (op == ep))
683  {
684  if ((eq == tcx.EdgeEvent.ConstrainedEdge.Q)
685  && (ep == tcx.EdgeEvent.ConstrainedEdge.P))
686  {
687  if (tcx.IsDebugEnabled)
688  {
689  Logger.Log("[FLIP] - constrained edge done");
690  }
691 
692  t.MarkConstrainedEdge(ep, eq);
693  ot.MarkConstrainedEdge(ep, eq);
694  Legalize(tcx, t);
695  Legalize(tcx, ot);
696  }
697  else
698  {
699  if (tcx.IsDebugEnabled)
700  {
701  Logger.Log("[FLIP] - subedge done");
702  }
703  }
704  }
705  else
706  {
707  if (tcx.IsDebugEnabled)
708  {
709  Logger.Log("[FLIP] - flipping and continuing with triangle still crossing edge");
710  }
711 
712 
713  Orientation o = TriangulationUtil.Orient2d(eq, op, ep);
714  t = NextFlipTriangle(tcx, o, t, ot, p, op);
715  FlipEdgeEvent(tcx, ep, eq, t, p);
716  }
717  }
718  else
719  {
720  TriangulationPoint newP = NextFlipPoint(ep, eq, ot, op);
721  FlipScanEdgeEvent(tcx, ep, eq, t, ot, newP);
722  EdgeEvent(tcx, ep, eq, t, p);
723  }
724  }

813  private static void FillAdvancingFront(DtSweepContext tcx, AdvancingFrontNode n)
814  {
815  double angle;
816 
817  // Fill right holes
818  AdvancingFrontNode node = n.Next;
819  while (node.HasNext)
820  {
821  // if HoleAngle exceeds 90 degrees then break.
822  if (LargeHole_DontFill(node))
823  {
824  break;
825  }
826 
827  Fill(tcx, node);
828  node = node.Next;
829  }
830 
831  // Fill left holes
832  node = n.Prev;
833  while (node.HasPrev)
834  {
835  // if HoleAngle exceeds 90 degrees then break.
836  if (LargeHole_DontFill(node))
837  {
838  break;
839  }
840 
841  angle = HoleAngle(node);
842  if (angle > PiDiv2 || angle < -PiDiv2)
843  {

936  private static double Angle(TriangulationPoint origin, TriangulationPoint pa, TriangulationPoint pb)
937  {
938  /* Complex plane
939  * ab = cosA +i*sinA
940  * ab = (ax + ay*i)(bx + by*i) = (ax*bx + ay*by) + i(ax*by-ay*bx)
941  * atan2(y,x) computes the principal value of the argument function
942  * applied to the complex number x+iy
943  * Where x = ax*bx + ay*by
944  * y = ax*by - ay*bx
945  */

947  double py = origin.Y;
948  double ax = pa.X - px;
949  double ay = pa.Y - py;

951  double by = pb.Y - py;
952  double x = ax * by - ay * bx;
953  double y = ax * bx + ay * by;
954  double angle = Math.Atan2(x, y);

1144  // violate the Delaunay condition
