130  private void GetParams()
131  {
132  shaderParams = new Dictionary<string, GlShaderProgramParam>();
133 
134  int[] resources = new int[1];
135  int[] actualLength = new int[1];
136  int[] arraySize = new int[1];
137 
138  glGetProgramiv(ProgramId, ProgramParameter.ActiveAttributes, resources);
139 
140  for (uint i = 0; i < resources[0]; i++)
141  {
142  ActiveAttribType[] type = new ActiveAttribType[1];
143  StringBuilder sb = new StringBuilder(256);
144  glGetActiveAttrib(ProgramId, i, 256, actualLength, arraySize, type, sb);
145 
146  if (!shaderParams.ContainsKey(sb.ToString()))
147  {
148  GlShaderProgramParam param = new GlShaderProgramParam(TypeFromAttributeType(type[0]), ParamType.Attribute, sb.ToString());
149  shaderParams.Add(param.Name, param);
150  param.GetLocation(this);
151  }
152  }
153 
154  glGetProgramiv(ProgramId, ProgramParameter.ActiveUniforms, resources);
155 
156  for (uint i = 0; i < resources[0]; i++)
157  {
158  ActiveUniformType[] type = new ActiveUniformType[1];
159  StringBuilder sb = new StringBuilder(256);
160  glGetActiveUniform(ProgramId, i, 256, actualLength, arraySize, type, sb);
161 
162  if (!shaderParams.ContainsKey(sb.ToString()))
163  {
164  GlShaderProgramParam param = new GlShaderProgramParam(TypeFromUniformType(type[0]), ParamType.Uniform, sb.ToString());
165  shaderParams.Add(param.Name, param);
166  param.GetLocation(this);
167  }
168  }
169  }

177  private Type TypeFromAttributeType(ActiveAttribType type)
178  {
179  switch (type)
180  {
181  case ActiveAttribType.Float: return typeof(float);
182  case ActiveAttribType.FloatMat2: return typeof(float[]);
183  case ActiveAttribType.FloatMat3: throw new Exception();
