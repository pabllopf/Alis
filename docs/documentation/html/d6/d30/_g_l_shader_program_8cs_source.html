132  private void GetParams()
133  {
134  shaderParams = new Dictionary<string, GlShaderProgramParam>();
135 
136  int[] resources = new int[1];
137  int[] actualLength = new int[1];
138  int[] arraySize = new int[1];
139 
140  GlGetProgramiv(ProgramId, ProgramParameter.ActiveAttributes, resources);
141 
142  for (uint i = 0; i < resources[0]; i++)
143  {
144  ActiveAttribType[] type = new ActiveAttribType[1];
145  StringBuilder sb = new StringBuilder(256);
146  GlGetActiveAttrib(ProgramId, i, 256, actualLength, arraySize, type, sb);
147 
148  if (!shaderParams.ContainsKey(sb.ToString()))
149  {
150  GlShaderProgramParam param = new GlShaderProgramParam(TypeFromAttributeType(type[0]), ParamType.Attribute, sb.ToString());
151  shaderParams.Add(param.Name, param);
152  param.GetLocation(this);
153  }
154  }
155 
156  GlGetProgramiv(ProgramId, ProgramParameter.ActiveUniforms, resources);
157 
158  for (uint i = 0; i < resources[0]; i++)
159  {
160  ActiveUniformType[] type = new ActiveUniformType[1];
161  StringBuilder sb = new StringBuilder(256);
162  GlGetActiveUniform(ProgramId, i, 256, actualLength, arraySize, type, sb);
163 
164  if (!shaderParams.ContainsKey(sb.ToString()))
165  {
166  GlShaderProgramParam param = new GlShaderProgramParam(TypeFromUniformType(type[0]), ParamType.Uniform, sb.ToString());
167  shaderParams.Add(param.Name, param);
168  param.GetLocation(this);
169  }
170  }
171  }
