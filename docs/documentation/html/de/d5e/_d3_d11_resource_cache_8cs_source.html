115  private ID3D11BlendState CreateNewBlendState(ref BlendStateDescription description)
116  {
117  BlendAttachmentDescription[] attachmentStates = description.AttachmentStates;
118  Vortice.Direct3D11.BlendDescription d3dBlendStateDesc = new Vortice.Direct3D11.BlendDescription();
119 
120  for (int i = 0; i < attachmentStates.Length; i++)
121  {
122  BlendAttachmentDescription state = attachmentStates[i];
123  d3dBlendStateDesc.RenderTarget[i].BlendEnable = state.BlendEnabled;
124  d3dBlendStateDesc.RenderTarget[i].RenderTargetWriteMask = D3D11Formats.VdToD3D11ColorWriteEnable(state.ColorWriteMask.GetOrDefault());
125  d3dBlendStateDesc.RenderTarget[i].SourceBlend = D3D11Formats.VdToD3D11Blend(state.SourceColorFactor);
126  d3dBlendStateDesc.RenderTarget[i].DestinationBlend = D3D11Formats.VdToD3D11Blend(state.DestinationColorFactor);
127  d3dBlendStateDesc.RenderTarget[i].BlendOperation = D3D11Formats.VdToD3D11BlendOperation(state.ColorFunction);
128  d3dBlendStateDesc.RenderTarget[i].SourceBlendAlpha = D3D11Formats.VdToD3D11Blend(state.SourceAlphaFactor);
129  d3dBlendStateDesc.RenderTarget[i].DestinationBlendAlpha = D3D11Formats.VdToD3D11Blend(state.DestinationAlphaFactor);
130  d3dBlendStateDesc.RenderTarget[i].BlendOperationAlpha = D3D11Formats.VdToD3D11BlendOperation(state.AlphaFunction);
131  }
132 
133  d3dBlendStateDesc.AlphaToCoverageEnable = description.AlphaToCoverageEnabled;
134  d3dBlendStateDesc.IndependentBlendEnable = true;
135 
136  return _device.CreateBlendState(d3dBlendStateDesc);
137  }

240  private ID3D11InputLayout GetInputLayout(VertexLayoutDescription[] vertexLayouts, byte[] vsBytecode)
241  {
242  Debug.Assert(Monitor.IsEntered(_lock));
243 
244  if (vsBytecode == null || vertexLayouts == null || vertexLayouts.Length == 0) { return null; }
245 
246  InputLayoutCacheKey tempKey = InputLayoutCacheKey.CreateTempKey(vertexLayouts);
247  if (!_inputLayouts.TryGetValue(tempKey, out ID3D11InputLayout inputLayout))
248  {
249  inputLayout = CreateNewInputLayout(vertexLayouts, vsBytecode);
250  InputLayoutCacheKey permanentKey = InputLayoutCacheKey.CreatePermanentKey(vertexLayouts);
251  _inputLayouts.Add(permanentKey, inputLayout);
252  }
253 
254  return inputLayout;
255  }

263  private ID3D11InputLayout CreateNewInputLayout(VertexLayoutDescription[] vertexLayouts, byte[] vsBytecode)
264  {
265  int totalCount = 0;
266  for (int i = 0; i < vertexLayouts.Length; i++)
267  {
268  totalCount += vertexLayouts[i].Elements.Length;
269  }
270 
271  int element = 0; // Total element index across slots.
272  InputElementDescription[] elements = new InputElementDescription[totalCount];
273  SemanticIndices si = new SemanticIndices();
274  for (int slot = 0; slot < vertexLayouts.Length; slot++)
275  {
276  VertexElementDescription[] elementDescs = vertexLayouts[slot].Elements;
277  uint stepRate = vertexLayouts[slot].InstanceStepRate;
278  int currentOffset = 0;
279  for (int i = 0; i < elementDescs.Length; i++)
280  {
281  VertexElementDescription desc = elementDescs[i];
282  elements[element] = new InputElementDescription(
283  GetSemanticString(desc.Semantic),
284  SemanticIndices.GetAndIncrement(ref si, desc.Semantic),
285  D3D11Formats.ToDxgiFormat(desc.Format),
286  desc.Offset != 0 ? (int)desc.Offset : currentOffset,
287  slot,
288  stepRate == 0 ? InputClassification.PerVertexData : InputClassification.PerInstanceData,
289  (int)stepRate);
290 
291  currentOffset += (int)FormatSizeHelpers.GetSizeInBytes(desc.Format);
292  element += 1;
293  }
294  }
295 
296  return _device.CreateInputLayout(elements, vsBytecode);
297  }
