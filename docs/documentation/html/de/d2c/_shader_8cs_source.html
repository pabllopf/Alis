102  public Shader(Stream vertexShaderStream, Stream geometryShaderStream, Stream fragmentShaderStream) :
103  base(IntPtr.Zero)
104  {
105  // using these funky conditional operators because StreamAdaptor doesn't have some method for dealing with
106  // its constructor argument being null
107  using (StreamAdaptor vertexAdaptor =
108  vertexShaderStream != null ? new StreamAdaptor(vertexShaderStream) : null,
109  geometryAdaptor = geometryShaderStream != null ? new StreamAdaptor(geometryShaderStream) : null,
110  fragmentAdaptor = fragmentShaderStream != null ? new StreamAdaptor(fragmentShaderStream) : null)
111  {
112  CPointer = sfShader_createFromStream(vertexAdaptor != null ? vertexAdaptor.InputStreamPtr : IntPtr.Zero,
113  geometryAdaptor != null ? geometryAdaptor.InputStreamPtr : IntPtr.Zero,
114  fragmentAdaptor != null ? fragmentAdaptor.InputStreamPtr : IntPtr.Zero);
115  }
116 
117  if (CPointer == IntPtr.Zero)
118  {
119  throw new LoadingFailedException("shader");
120  }
121  }
