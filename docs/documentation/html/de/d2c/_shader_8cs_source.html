101  public Shader(Stream vertexShaderStream, Stream geometryShaderStream, Stream fragmentShaderStream) :
102  base(IntPtr.Zero)
103  {
104  // using these funky conditional operators because StreamAdaptor doesn't have some method for dealing with
105  // its constructor argument being null
106  using (StreamAdaptor vertexAdaptor =
107  vertexShaderStream != null ? new StreamAdaptor(vertexShaderStream) : null,
108  geometryAdaptor = geometryShaderStream != null ? new StreamAdaptor(geometryShaderStream) : null,
109  fragmentAdaptor = fragmentShaderStream != null ? new StreamAdaptor(fragmentShaderStream) : null)
110  {
111  CPointer = sfShader_createFromStream(vertexAdaptor != null ? vertexAdaptor.InputStreamPtr : IntPtr.Zero,
112  geometryAdaptor != null ? geometryAdaptor.InputStreamPtr : IntPtr.Zero,
113  fragmentAdaptor != null ? fragmentAdaptor.InputStreamPtr : IntPtr.Zero);
114  }
115 
116  if (CPointer == IntPtr.Zero)
117  {
118  throw new LoadingFailedException("shader");
119  }
120  }
