184  private static float FindMaxSeparation(out int edgeIndex, PolygonShape poly1, ref Transform xf1,
185  PolygonShape poly2, ref Transform xf2)
186  {
187  int count1 = poly1.VerticesPrivate.Count;
188  int count2 = poly2.VerticesPrivate.Count;
189  Vertices n1S = poly1.NormalsPrivate;
190  Vertices v1S = poly1.VerticesPrivate;
191  Vertices v2S = poly2.VerticesPrivate;
192  Transform xf = MathUtils.MulT(xf2, xf1);
193 
194  int bestIndex = 0;
195  float maxSeparation = -MathConstants.MaxFloat;
196  for (int i = 0; i < count1; ++i)
197  {
198  // Get poly1 normal in frame2.
199  Vector2 n = MathUtils.Mul(ref xf.Q, n1S[i]);
200  Vector2 v1 = MathUtils.Mul(ref xf, v1S[i]);
201 
202  // Find deepest point for normal i.
203  float si = MathConstants.MaxFloat;
204  for (int j = 0; j < count2; ++j)
205  {
206  float sij = Vector2.Dot(n, v2S[j] - v1);
207  if (sij < si)
208  {
209  si = sij;
210  }
211  }
212 
213  if (si > maxSeparation)
214  {
215  maxSeparation = si;
216  bestIndex = i;
217  }
218  }
219 
220  edgeIndex = bestIndex;
221  return maxSeparation;
222  }

233  private static void FindIncidentEdge(out FixedArray2<ClipVertex> c, PolygonShape poly1, ref Transform xf1,
234  int edge1, PolygonShape poly2, ref Transform xf2)
235  {
236  Vertices normals1 = poly1.NormalsPrivate;
237 
238  int count2 = poly2.VerticesPrivate.Count;
239  Vertices vertices2 = poly2.VerticesPrivate;
240  Vertices normals2 = poly2.NormalsPrivate;
241 
242  Debug.Assert(0 <= edge1 && edge1 < poly1.VerticesPrivate.Count);
243 
244  // Get the normal of the reference edge in poly2's frame.
245  Vector2 normal1 = MathUtils.MulT(ref xf2.Q, MathUtils.Mul(ref xf1.Q, normals1[edge1]));
246 
247  // Find the incident edge on poly2.
248  int index = 0;
249  float minDot = MathConstants.MaxFloat;
250  for (int i = 0; i < count2; ++i)
251  {
252  float dot = Vector2.Dot(normal1, normals2[i]);
253  if (dot < minDot)
254  {
255  minDot = dot;
256  index = i;
257  }
258  }
259 
260  // Build the clip vertices for the incident edge.
261  int i1 = index;
262  int i2 = i1 + 1 < count2 ? i1 + 1 : 0;
263 
264  c = new FixedArray2<ClipVertex>();
265  c.Value0.V = MathUtils.Mul(ref xf2, vertices2[i1]);
266  c.Value0.Id.ContactFeature.IndexA = (byte) edge1;
267  c.Value0.Id.ContactFeature.IndexB = (byte) i1;
268  c.Value0.Id.ContactFeature.TypeA = ContactFeatureType.Face;
269  c.Value0.Id.ContactFeature.TypeB = ContactFeatureType.Vertex;
270 
271  c.Value1.V = MathUtils.Mul(ref xf2, vertices2[i2]);
272  c.Value1.Id.ContactFeature.IndexA = (byte) edge1;
273  c.Value1.Id.ContactFeature.IndexB = (byte) i2;
274  c.Value1.Id.ContactFeature.TypeA = ContactFeatureType.Face;
275  c.Value1.Id.ContactFeature.TypeB = ContactFeatureType.Vertex;
276  }
