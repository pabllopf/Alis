449  public static bool Invert(Matrix3X2F matrix, out Matrix3X2F result)
450  {
451  float det = matrix.M11 * matrix.M22 - matrix.M21 * matrix.M12;
452 
453  result = Identity;
454 
455  if (MathF.Abs(det) < float.Epsilon)
456  {
457  result = new Matrix3X2F(float.NaN, float.NaN, float.NaN, float.NaN, float.NaN, float.NaN);
458  return false;
459  }
460 
461  float invDet = 1.0f / det;
462 
463  result.M11 = matrix.M22 * invDet;
464  result.M12 = -matrix.M12 * invDet;
465 
466  result.M21 = -matrix.M21 * invDet;
467  result.M22 = matrix.M11 * invDet;
468 
469  result.M31 = (matrix.M21 * matrix.M32 - matrix.M31 * matrix.M22) * invDet;
470  result.M32 = (matrix.M31 * matrix.M12 - matrix.M11 * matrix.M32) * invDet;
471 
472  return true;
473  }

535  public float GetDeterminant() =>
536  // There isn't actually any such thing as a determinant for a non-square matrix,
537  // but this 3x2 type is really just an optimization of a 3x3 where we happen to
538  // know the rightmost column is always (0, 0, 1). So we expand to 3x3 format:
539  //
540  // [ M11, M12, 0 ]
541  // [ M21, M22, 0 ]
542  // [ M31, M32, 1 ]
543  //
544  // Sum the diagonal products:
545  // (M11 * M22 * 1) + (M12 * 0 * M31) + (0 * M21 * M32)
546  //
547  // Subtract the opposite diagonal products:
548  // (M31 * M22 * 0) + (M32 * 0 * M11) + (1 * M21 * M12)
549  //
550  // Collapse out the constants and oh look, this is just a 2x2 determinant!
551  M11 * M22 - M21 * M12;
