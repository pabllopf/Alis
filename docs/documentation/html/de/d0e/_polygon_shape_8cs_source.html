104  private void SetVertices(Vertices vertices)
105  {
106  Debug.Assert((vertices.Count >= 3) && (vertices.Count <= Settings.PolygonVertices));
107 
108  //Velcro: We throw an exception instead of setting the polygon to a box for safety reasons
109  if (vertices.Count < 3)
110  {
111  throw new InvalidOperationException("You can't create a polygon with less than 3 vertices");
112  }
113 
114  int n = MathUtils.Min(vertices.Count, Settings.PolygonVertices);
115 
116  // Perform welding and copy vertices into local buffer.
117  Vector2[] ps = new Vector2[n]; //Velcro: The temp buffer is n long instead of Settings.MaxPolygonVertices
118  int tempCount = 0;
119  for (int i = 0; i < n; ++i)
120  {
121  Vector2 v = vertices[i];
122 
123  bool unique = true;
124  for (int j = 0; j < tempCount; ++j)
125  {
126  Vector2 temp = ps[j];
127  if (MathUtils.DistanceSquared(ref v, ref temp) <
128  0.5f * Settings.LinearSlop * (0.5f * Settings.LinearSlop))
129  {
130  unique = false;
131  break;
132  }
133  }
134 
135  if (unique)
136  {
137  ps[tempCount++] = v;
138  }
139  }
140 
141  n = tempCount;
142  if (n < 3)
143  {
144  // Polygon is degenerate.
145  throw
146  new InvalidOperationException(
147  "Polygon is degenerate"); //Velcro: We throw exception here because we had invalid input
148  }
149 
150  // Create the convex hull using the Gift wrapping algorithm
151  // http://en.wikipedia.org/wiki/Gift_wrapping_algorithm
152 
153  // Find the right most point on the hull
154  int i0 = 0;
155  float x0 = ps[0].X;
156  for (int i = 1; i < n; ++i)
157  {
158  float x = ps[i].X;
159  if (x > x0 || ((x == x0) && (ps[i].Y < ps[i0].Y)))
160  {
161  i0 = i;
162  x0 = x;
163  }
164  }
165 
166  int[] hull = new int[Settings.PolygonVertices];
167  int m = 0;
168  int ih = i0;
169 
170  for (;;)
171  {
172  Debug.Assert(m < Settings.PolygonVertices);
173  hull[m] = ih;
174 
175  int ie = 0;
176  for (int j = 1; j < n; ++j)
177  {
178  if (ie == ih)
179  {
180  ie = j;
181  continue;
182  }
183 
184  Vector2 r = ps[ie] - ps[hull[m]];
185  Vector2 v = ps[j] - ps[hull[m]];
186  float c = MathUtils.Cross(r, v);
187  if (c < 0.0f)
188  {
189  ie = j;
190  }
191 
192  // Collinearity check
193  if ((c == 0.0f) && (v.LengthSquared() > r.LengthSquared()))
194  {
195  ie = j;
196  }
197  }
198 
199  ++m;
200  ih = ie;
201 
202  if (ie == i0)
203  {
204  break;
205  }
206  }
207 
208  if (m < 3)
209  {
210  // Polygon is degenerate.
211  throw
212  new InvalidOperationException(
213  "Polygon is degenerate"); //Velcro: We throw exception here because we had invalid input
214  }
215 
216  VerticesPrivate = new Vertices(m);
217 
218  // Copy vertices.
219  for (int i = 0; i < m; ++i)
220  {
221  VerticesPrivate.Add(ps[hull[i]]);
222  }
223 
224  NormalsPrivate = new Vertices(m);
225 
226  // Compute normals. Ensure the edges have non-zero length.
227  for (int i = 0; i < m; ++i)
228  {
229  int i1 = i;
230  int i2 = i + 1 < VerticesPrivate.Count ? i + 1 : 0;
231  Vector2 edge = VerticesPrivate[i2] - VerticesPrivate[i1];
232  Debug.Assert(edge.LengthSquared() > Constant.Epsilon * Constant.Epsilon);
233  Vector2 temp = MathUtils.Cross(edge, 1.0f);
234  temp = Vector2.Normalize(temp);
235  NormalsPrivate.Add(temp);
236  }
237 
238  //Velcro: We compute all the mass data properties up front
239  ComputeProperties();
240  }

302  protected sealed override void ComputeProperties()
303  {
304  // Polygon mass, centroid, and inertia.
305  // Let rho be the polygon density in mass per unit area.
306  // Then:
307  // mass = rho * int(dA)
308  // centroid.x = (1/mass) * rho * int(x * dA)
309  // centroid.y = (1/mass) * rho * int(y * dA)
310  // I = rho * int((x*x + y*y) * dA)
311  //
312  // We can compute these integrals by summing all the integrals
313  // for each triangle of the polygon. To evaluate the integral
314  // for a single triangle, we make a change of variables to
315  // the (u,v) coordinates of the triangle:
316  // x = x0 + e1x * u + e2x * v
317  // y = y0 + e1y * u + e2y * v
318  // where 0 <= u && 0 <= v && u + v <= 1.
319  //
320  // We integrate u from [0,1-v] and then v from [0,1].
321  // We also need to use the Jacobian of the transformation:
322  // D = cross(e1, e2)
323  //
324  // Simplification: triangle centroid = (1/3) * (p1 + p2 + p3)
325  //
326  // The rest of the derivation is handled by computer algebra.
327 
328  Debug.Assert(VerticesPrivate.Count >= 3);
329 
330  //Velcro: Early exit as polygons with 0 density does not have any properties.
331  if (DensityPrivate <= 0)
332  {
333  return;
334  }
335 
336  //Velcro: Consolidated the calculate centroid and mass code to a single method.
337  Vector2 center = Vector2.Zero;
338  float area = 0.0f;
339  float I = 0.0f;
340 
341  // Get a reference point for forming triangles.
342  // Use the first vertex to reduce round-off errors.
343  Vector2 s = VerticesPrivate[0];
344 
345  const float inv3 = 1.0f / 3.0f;
346 
347  int count = VerticesPrivate.Count;
348 
349  for (int i = 0; i < count; ++i)
350  {
351  // Triangle vertices.
352  Vector2 e1 = VerticesPrivate[i] - s;
353  Vector2 e2 = i + 1 < count ? VerticesPrivate[i + 1] - s : VerticesPrivate[0] - s;
354 
355  float d = MathUtils.Cross(e1, e2);
356 
357  float triangleArea = 0.5f * d;
358  area += triangleArea;
359 
360  // Area weighted centroid
361  center += triangleArea * inv3 * (e1 + e2);
362 
363  float ex1 = e1.X, ey1 = e1.Y;
364  float ex2 = e2.X, ey2 = e2.Y;
365 
366  float intx2 = ex1 * ex1 + ex2 * ex1 + ex2 * ex2;
367  float inty2 = ey1 * ey1 + ey2 * ey1 + ey2 * ey2;
368 
369  I += 0.25f * inv3 * d * (intx2 + inty2);
370  }
371 
372  //The area is too small for the engine to handle.
373  Debug.Assert(area > Constant.Epsilon);
374 
375  // We save the area
376  MassDataPrivate.Area = area;
377 
378  // Total mass
379  MassDataPrivate.Mass = DensityPrivate * area;
380 
381  // Center of mass
382  center *= 1.0f / area;
383  MassDataPrivate.Centroid = center + s;
384 
385  // Inertia tensor relative to the local origin (point s).
386  MassDataPrivate.Inertia = DensityPrivate * I;
387 
388  // Shift to center of mass then to original body origin.
389  MassDataPrivate.Inertia += MassDataPrivate.Mass *
390  (MathUtils.Dot(MassDataPrivate.Centroid, MassDataPrivate.Centroid) -
391  MathUtils.Dot(center, center));
392  }
