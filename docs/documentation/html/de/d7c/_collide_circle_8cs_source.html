45  public static void CollideCircles(ref Manifold manifold, CircleShape circleA, ref Transform xfA,
46  CircleShape circleB, ref Transform xfB)
47  {
48  manifold.PointCount = 0;
49 
50  Vector2F pA = MathUtils.Mul(ref xfA, circleA.Position);
51  Vector2F pB = MathUtils.Mul(ref xfB, circleB.Position);
52 
53  Vector2F d = pB - pA;
54  float distSqr = Vector2F.Dot(d, d);
55  float rA = circleA.RadiusPrivate, rB = circleB.RadiusPrivate;
56  float radius = rA + rB;
57  if (distSqr > radius * radius)
58  {
59  return;
60  }
61 
62  manifold.Type = ManifoldType.Circles;
63  manifold.LocalPoint = circleA.Position;
64  manifold.LocalNormal = Vector2F.Zero;
65  manifold.PointCount = 1;
66 
67  ManifoldPoint p0 = manifold.Points[0];
68  p0.LocalPoint = circleB.Position;
69  p0.Id.Key = 0;
70  manifold.Points[0] = p0;
71  }

79  public static void CollidePolygonAndCircle(ref Manifold manifold, PolygonShape polygonA, ref Transform xfA,
80  CircleShape circleB, ref Transform xfB)
81  {
82  manifold.PointCount = 0;
83 
84  // Compute circle position in the frame of the polygon.
85  Vector2F c = MathUtils.Mul(ref xfB, circleB.Position);
86  Vector2F cLocal = MathUtils.MulT(ref xfA, c);
87 
88  // Find the min separating edge.
89  int normalIndex = 0;
90  float separation = -float.MaxValue;
91  float radius = polygonA.RadiusPrivate + circleB.RadiusPrivate;
92  int vertexCount = polygonA.VerticesPrivate.Count;
93  Vertices vertices = polygonA.VerticesPrivate;
94  Vertices normals = polygonA.NormalsPrivate;
95 
96  for (int i = 0; i < vertexCount; ++i)
97  {
98  float s = Vector2F.Dot(normals[i], cLocal - vertices[i]);
99 
100  if (s > radius)
101  {
102  // Early out.
103  return;
104  }
105 
106  if (s > separation)
107  {
108  separation = s;
109  normalIndex = i;
110  }
111  }
112 
113  // Vertices that subtend the incident face.
114  int vertIndex1 = normalIndex;
115  int vertIndex2 = vertIndex1 + 1 < vertexCount ? vertIndex1 + 1 : 0;
116  Vector2F v1 = vertices[vertIndex1];
117  Vector2F v2 = vertices[vertIndex2];
118 
119  // If the center is inside the polygon ...
120  if (separation < Constant.Epsilon)
121  {
122  manifold.PointCount = 1;
123  manifold.Type = ManifoldType.FaceA;
124  manifold.LocalNormal = normals[normalIndex];
125  manifold.LocalPoint = 0.5f * (v1 + v2);
126  manifold.Points.Value0.LocalPoint = circleB.Position;
127  manifold.Points.Value0.Id.Key = 0;
128  return;
129  }
130 
131  // Compute barycentric coordinates
132  float u1 = Vector2F.Dot(cLocal - v1, v2 - v1);
133  float u2 = Vector2F.Dot(cLocal - v2, v1 - v2);
134 
135  if (u1 <= 0.0f)
136  {
137  if (Vector2F.DistanceSquared(cLocal, v1) > radius * radius)
138  {
139  return;
140  }
141 
142  manifold.PointCount = 1;
143  manifold.Type = ManifoldType.FaceA;
144  manifold.LocalNormal = cLocal - v1;
145  manifold.LocalNormal = Vector2F.Normalize(manifold.LocalNormal);
146  manifold.LocalPoint = v1;
147  manifold.Points.Value0.LocalPoint = circleB.Position;
148  manifold.Points.Value0.Id.Key = 0;
149  }
150  else if (u2 <= 0.0f)
151  {
152  if (Vector2F.DistanceSquared(cLocal, v2) > radius * radius)
153  {
154  return;
155  }
156 
157  manifold.PointCount = 1;
158  manifold.Type = ManifoldType.FaceA;
159  manifold.LocalNormal = cLocal - v2;
160  manifold.LocalNormal = Vector2F.Normalize(manifold.LocalNormal);
161  manifold.LocalPoint = v2;
162  manifold.Points.Value0.LocalPoint = circleB.Position;
163  manifold.Points.Value0.Id.Key = 0;
164  }
165  else
166  {
167  Vector2F faceCenter = 0.5f * (v1 + v2);
168  float s = Vector2F.Dot(cLocal - faceCenter, normals[vertIndex1]);
169  if (s > radius)
170  {
171  return;
172  }
173 
174  manifold.PointCount = 1;
175  manifold.Type = ManifoldType.FaceA;
176  manifold.LocalNormal = normals[vertIndex1];
177  manifold.LocalPoint = faceCenter;
178  manifold.Points.Value0.LocalPoint = circleB.Position;
179  manifold.Points.Value0.Id.Key = 0;
180  }
181  }
182  }
183 }
