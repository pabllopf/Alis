372  [MethodImpl(MethodImplOptions.AggressiveInlining)]
373  public static Vector2 Transform(Vector2 value, Quaternion rotation)
374  {
375  float x2 = rotation.X + rotation.X;
376  float y2 = rotation.Y + rotation.Y;
377  float z2 = rotation.Z + rotation.Z;
378 
379  float wz2 = rotation.W * z2;
380  float xx2 = rotation.X * x2;
381  float xy2 = rotation.X * y2;
382  float yy2 = rotation.Y * y2;
383  float zz2 = rotation.Z * z2;
384 
385  return new Vector2(
386  value.X * (1.0f - yy2 - zz2) + value.Y * (xy2 - wz2),
387  value.X * (xy2 + wz2) + value.Y * (1.0f - xx2 - zz2)
388  );
389  }
