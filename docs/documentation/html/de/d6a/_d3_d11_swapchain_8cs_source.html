214  {

217  foreach (D3D11CommandList cl in _referencedCLs)
218  {

220  }
221

223  }

225  bool resizeBuffers = false;
226 
227  if (_framebuffer != null)
228  {
229  resizeBuffers = true;
230  if (_depthTexture != null)
231  {
232  _depthTexture.Dispose();
233  }
234 
235  _framebuffer.Dispose();
236  }
237 
238  uint actualWidth = (uint)(width * _pixelScale);
239  uint actualHeight = (uint)(height * _pixelScale);
240  if (resizeBuffers)

242  _dxgiSwapChain.ResizeBuffers(2, (int)actualWidth, (int)actualHeight, _colorFormat, SwapChainFlags.None).CheckError();
243  }

247  {

261  D3D11Formats.ToVdFormat(_colorFormat));
262
