158  internal override void InitVelocityConstraints(ref SolverData data)
159  {
160  int indexA = BodyA.IslandIndex;

162 
163  float aW = data.Positions[indexA].A;
164  float bW = data.Positions[indexB].A;
165 
166  jointError = bW - aW - targetAngle;
167  bias = -biasFactor * data.Step.InvertedDeltaTime * jointError;
168  massFactor = (1 - softness) / (BodyA.InvI + BodyB.InvI);

176  {

179 
180  float p = (bias - data.Velocities[indexB].W + data.Velocities[indexA].W) * massFactor;
181 
182  data.Velocities[indexA].W -=
183  BodyA.InvI * MathUtils.Sign(p) * MathUtils.Min(MathUtils.Abs(p), maxImpulse);
184  data.Velocities[indexB].W +=
185  BodyB.InvI * MathUtils.Sign(p) * MathUtils.Min(MathUtils.Abs(p), maxImpulse);
