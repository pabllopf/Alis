160  internal override void InitVelocityConstraints(ref SolverData data)
161  {

164

166  float bW = data.Positions[indexB].A;

168  jointError = bW - aW - targetAngle;
169  bias = -biasFactor * data.Step.InvertedDeltaTime * jointError;

178  {

181 
182  float p = (bias - data.Velocities[indexB].W + data.Velocities[indexA].W) * massFactor;
183 
184  data.Velocities[indexA].W -=
185  BodyA.InvI * MathUtils.Sign(p) * MathUtils.Min(MathUtils.Abs(p), maxImpulse);
186  data.Velocities[indexB].W +=
187  BodyB.InvI * MathUtils.Sign(p) * MathUtils.Min(MathUtils.Abs(p), maxImpulse);
