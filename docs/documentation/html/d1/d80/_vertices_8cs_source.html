101  private float GetSignedArea()
102  {
103  //The simplest polygon which can exist in the Euclidean plane has 3 sides.
104  if (Count < 3)
105  {
106  return 0;
107  }
108 
109  int i;
110  float area = 0;
111 
112  for (i = 0; i < Count; i++)
113  {
114  int j = (i + 1) % Count;
115 
116  Vector2 vi = this[i];
117  Vector2 vj = this[j];
118 
119  area += vi.X * vj.Y;
120  area -= vi.Y * vj.X;
121  }
122 
123  area /= 2.0f;
124  return area;
125  }

137  public Vector2 GetCentroid()
138  {
139  //The simplest polygon which can exist in the Euclidean plane has 3 sides.
140  if (Count < 3)
141  {
142  return new Vector2(float.NaN, float.NaN);
143  }
144 
145  // Same algorithm is used by Box2D
146  Vector2 c = Vector2.Zero;
147  float area = 0.0f;
148  const float inv3 = 1.0f / 3.0f;
149 
150  for (int i = 0; i < Count; ++i)
151  {
152  // Triangle vertices.
153  Vector2 current = this[i];
154  Vector2 next = i + 1 < Count ? this[i + 1] : this[0];
155 
156  float triangleArea = 0.5f * (current.X * next.Y - current.Y * next.X);
157  area += triangleArea;
158 
159  // Area weighted centroid
160  c += triangleArea * inv3 * (current + next);
161  }
162 
163  // Centroid
164  c *= 1.0f / area;
165  return c;
166  }

291  public bool IsConvex()
292  {
293  //The simplest polygon which can exist in the Euclidean plane has 3 sides.
294  if (Count < 3)
295  {
296  return false;
297  }
298 
299  //Triangles are always convex
300  if (Count == 3)
301  {
302  return true;
303  }
304 
305  // Checks the polygon is convex and the interior is to the left of each edge.
306  for (int i = 0; i < Count; ++i)
307  {
308  int next = i + 1 < Count ? i + 1 : 0;
309  Vector2 edge = this[next] - this[i];
310 
311  for (int j = 0; j < Count; ++j)
312  {
313  // Don't check vertices on the current edge.
314  if (j == i || j == next)
315  {
316  continue;
317  }
318 
319  Vector2 r = this[j] - this[i];
320 
321  float s = edge.X * r.Y - edge.Y * r.X;
322 
323  if (s <= 0.0f)
324  {
325  return false;
326  }
327  }
328  }
329 
330  return true;
331  }

479  private int CalculateWindingNumber(Vector2 point)
480  {
481  int windingNumber = 0;
482 
483  for (int i = 0; i < Count; i++)
484  {
485  Vector2 p1 = this[i];
486  Vector2 p2 = this[NextIndex(i)];
487 
488  if (IsPointOnEdge(point, p1, p2))
489  {
490  return 0;
491  }
492 
493  if (IsEdgeIntersectingRay(point, p1, p2))
494  {
495  windingNumber += DetermineWindingDirection(point, p1, p2);
496  }
497  }
498 
499  return windingNumber;

509  private bool IsPointOnEdge(Vector2 point, Vector2 p1, Vector2 p2)
510  {
511  Vector2 edge = p2 - p1;
512  float area = MathUtils.Area(ref p1, ref p2, ref point);
513  return (area == 0f) && (Vector2.Dot(point - p1, edge) >= 0f) && (Vector2.Dot(point - p2, edge) <= 0f);
514  }

634  Vector2[] sourceArray,

637  Vector2[] destinationArray,
