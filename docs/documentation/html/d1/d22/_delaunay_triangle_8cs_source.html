414  public int EdgeIndex(TriangulationPoint p1, TriangulationPoint p2)
415  {
416  int i1 = Points.IndexOf(p1);
417  int i2 = Points.IndexOf(p2);
418 
419  // Points of this triangle in the edge p1-p2
420  bool a = i1 == 0 || i2 == 0;
421  bool b = i1 == 1 || i2 == 1;
422  bool c = i1 == 2 || i2 == 2;
423 
424  if (b && c)
425  {
426  return 0;
427  }
428 
429  if (a && c)
430  {
431  return 1;
432  }
433 
434  if (a && b)
435  {
436  return 2;
437  }
438 
439  return -1;
440  }
