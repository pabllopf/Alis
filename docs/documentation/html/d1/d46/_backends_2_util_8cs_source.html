338  uint blockSizeInBytes = blockSize > 1 ? FormatHelpers.GetBlockSizeInBytes(format) : FormatSizeHelpers.GetSizeInBytes(format);
339  uint compressedSrcX = srcX / blockSize;
340  uint compressedSrcY = srcY / blockSize;
341  uint compressedDstX = dstX / blockSize;
342  uint compressedDstY = dstY / blockSize;
343  uint numRows = FormatHelpers.GetNumRows(height, format);
344  uint rowSize = width / blockSize * blockSizeInBytes;
345 
346  if (srcRowPitch == dstRowPitch && srcDepthPitch == dstDepthPitch)
347  {
348  uint totalCopySize = depth * srcDepthPitch;
349  Buffer.MemoryCopy(
350  src,
351  dst,
352  totalCopySize,
353  totalCopySize);
354  }
355  else
356  {
357  for (uint zz = 0; zz < depth; zz++)
358  for (uint yy = 0; yy < numRows; yy++)
359  {
360  byte* rowCopyDst = (byte*)dst
361  + dstDepthPitch * (zz + dstZ)
362  + dstRowPitch * (yy + compressedDstY)
363  + blockSizeInBytes * compressedDstX;
364 
365  byte* rowCopySrc = (byte*)src
366  + srcDepthPitch * (zz + srcZ)
367  + srcRowPitch * (yy + compressedSrcY)
368  + blockSizeInBytes * compressedSrcX;
369 
370  Unsafe.CopyBlock(rowCopyDst, rowCopySrc, rowSize);
371  }
