400  [MethodImpl(MethodImplOptions.AggressiveInlining)]
401  public static Vector3 Transform(Vector3 value, Quaternion rotation)
402  {
403  float x2 = rotation.X + rotation.X;
404  float y2 = rotation.Y + rotation.Y;
405  float z2 = rotation.Z + rotation.Z;
406 
407  float wx2 = rotation.W * x2;
408  float wy2 = rotation.W * y2;
409  float wz2 = rotation.W * z2;
410  float xx2 = rotation.X * x2;
411  float xy2 = rotation.X * y2;
412  float xz2 = rotation.X * z2;
413  float yy2 = rotation.Y * y2;
414  float yz2 = rotation.Y * z2;
415  float zz2 = rotation.Z * z2;
416 
417  return new Vector3(
418  value.X * (1.0f - yy2 - zz2) + value.Y * (xy2 - wz2) + value.Z * (xz2 + wy2),
419  value.X * (xy2 + wz2) + value.Y * (1.0f - xx2 - zz2) + value.Z * (yz2 - wx2),
420  value.X * (xz2 - wy2) + value.Y * (yz2 + wx2) + value.Z * (1.0f - xx2 - yy2)
421  );
422  }
