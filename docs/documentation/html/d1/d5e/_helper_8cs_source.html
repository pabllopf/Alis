61  public static float CatmullRom(float value1, float value2, float value3, float value4, float amount)
62  {
63  // Using formula from http://www.mvps.org/directx/articles/catmull/
64  // Internally using doubles not to lose precission
65  double amountSquared = amount * amount;
66  double amountCubed = amountSquared * amount;
67  return (float) (0.5 * (2.0 * value2 +
68  (value3 - value1) * amount +
69  (2.0 * value1 - 5.0 * value2 + 4.0 * value3 - value4) * amountSquared +
70  (3.0 * value2 - value1 - 3.0 * value3 + value4) * amountCubed));
71  }

115  public static float Hermite(float value1, float tangent1, float value2, float tangent2, float amount)
116  {
117  // All transformed to double not to lose precission
118  // Otherwise, for high numbers of param:amount the result is NaN instead of Infinity
119  double v1 = value1, v2 = value2, t1 = tangent1, t2 = tangent2, s = amount, result;
120  double sCubed = s * s * s;
121  double sSquared = s * s;
122 
123  if (amount == 0f)
124  {
125  result = value1;
126  }
127  else if (amount == 1f)
128  {
129  result = value2;
130  }
131  else
132  {
133  result = (2 * v1 - 2 * v2 + t2 + t1) * sCubed +
134  (3 * v2 - 3 * v1 - 2 * t1 - t2) * sSquared +
135  t1 * s +
136  v1;
137  }
138 
139  return (float) result;
140  }
