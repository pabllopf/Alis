254  private bool CreateSwapchain(uint width, uint height)
255  {
256  // Obtain the surface capabilities first -- this will indicate whether the surface has been lost.
257  VkResult result = vkGetPhysicalDeviceSurfaceCapabilitiesKHR(_gd.PhysicalDevice, _surface, out VkSurfaceCapabilitiesKHR surfaceCapabilities);
258  if (result == VkResult.ErrorSurfaceLostKHR)
259  {
260  throw new VeldridException($"The Swapchain's underlying surface has been lost.");
261  }
262 
263  if (surfaceCapabilities.minImageExtent.width == 0 && surfaceCapabilities.minImageExtent.height == 0
264  && surfaceCapabilities.maxImageExtent.width == 0 && surfaceCapabilities.maxImageExtent.height == 0)
265  {
266  return false;
267  }
268 
269  if (_deviceSwapchain != VkSwapchainKHR.Null)
270  {
271  _gd.WaitForIdle();
272  }
273 
274  _currentImageIndex = 0;
275  uint surfaceFormatCount = 0;
276  result = vkGetPhysicalDeviceSurfaceFormatsKHR(_gd.PhysicalDevice, _surface, ref surfaceFormatCount, null);
277  CheckResult(result);
278  VkSurfaceFormatKHR[] formats = new VkSurfaceFormatKHR[surfaceFormatCount];
279  result = vkGetPhysicalDeviceSurfaceFormatsKHR(_gd.PhysicalDevice, _surface, ref surfaceFormatCount, out formats[0]);
280  CheckResult(result);
281 
282  VkFormat desiredFormat = _colorSrgb
283  ? VkFormat.B8g8r8a8Srgb
284  : VkFormat.B8g8r8a8Unorm;
285 
286  VkSurfaceFormatKHR surfaceFormat = new VkSurfaceFormatKHR();
287  if (formats.Length == 1 && formats[0].format == VkFormat.Undefined)
288  {
289  surfaceFormat = new VkSurfaceFormatKHR { colorSpace = VkColorSpaceKHR.SrgbNonlinearKHR, format = desiredFormat };
290  }
291  else
292  {
293  foreach (VkSurfaceFormatKHR format in formats)
294  {
295  if (format.colorSpace == VkColorSpaceKHR.SrgbNonlinearKHR && format.format == desiredFormat)
296  {
297  surfaceFormat = format;
298  break;
299  }
300  }
301  if (surfaceFormat.format == VkFormat.Undefined)
302  {
303  if (_colorSrgb && surfaceFormat.format != VkFormat.R8g8b8a8Srgb)
304  {
305  throw new VeldridException($"Unable to create an sRGB Swapchain for this surface.");
306  }
307 
308  surfaceFormat = formats[0];
309  }
310  }
311 
312  uint presentModeCount = 0;
313  result = vkGetPhysicalDeviceSurfacePresentModesKHR(_gd.PhysicalDevice, _surface, ref presentModeCount, null);
314  CheckResult(result);
315  VkPresentModeKHR[] presentModes = new VkPresentModeKHR[presentModeCount];
316  result = vkGetPhysicalDeviceSurfacePresentModesKHR(_gd.PhysicalDevice, _surface, ref presentModeCount, out presentModes[0]);
317  CheckResult(result);
318 
319  VkPresentModeKHR presentMode = VkPresentModeKHR.FifoKHR;
320 
321  if (_syncToVBlank)
322  {
323  if (presentModes.Contains(VkPresentModeKHR.FifoRelaxedKHR))
324  {
325  presentMode = VkPresentModeKHR.FifoRelaxedKHR;
326  }
327  }
328  else
329  {
330  if (presentModes.Contains(VkPresentModeKHR.MailboxKHR))
331  {
332  presentMode = VkPresentModeKHR.MailboxKHR;
333  }
334  else if (presentModes.Contains(VkPresentModeKHR.ImmediateKHR))
335  {
336  presentMode = VkPresentModeKHR.ImmediateKHR;
337  }
338  }
339 
340  uint maxImageCount = surfaceCapabilities.maxImageCount == 0 ? uint.MaxValue : surfaceCapabilities.maxImageCount;
341  uint imageCount = Math.Min(maxImageCount, surfaceCapabilities.minImageCount + 1);
342 
343  VkSwapchainCreateInfoKHR swapchainCI = VkSwapchainCreateInfoKHR.New();
344  swapchainCI.surface = _surface;
345  swapchainCI.presentMode = presentMode;
346  swapchainCI.imageFormat = surfaceFormat.format;
347  swapchainCI.imageColorSpace = surfaceFormat.colorSpace;
348  uint clampedWidth = Util.Clamp(width, surfaceCapabilities.minImageExtent.width, surfaceCapabilities.maxImageExtent.width);
349  uint clampedHeight = Util.Clamp(height, surfaceCapabilities.minImageExtent.height, surfaceCapabilities.maxImageExtent.height);
350  swapchainCI.imageExtent = new VkExtent2D { width = clampedWidth, height = clampedHeight };
351  swapchainCI.minImageCount = imageCount;
352  swapchainCI.imageArrayLayers = 1;
353  swapchainCI.imageUsage = VkImageUsageFlags.ColorAttachment | VkImageUsageFlags.TransferDst;
354 
355  FixedArray2<uint> queueFamilyIndices = new FixedArray2<uint>(_gd.GraphicsQueueIndex, _gd.PresentQueueIndex);
356 
357  if (_gd.GraphicsQueueIndex != _gd.PresentQueueIndex)
358  {
359  swapchainCI.imageSharingMode = VkSharingMode.Concurrent;
360  swapchainCI.queueFamilyIndexCount = 2;
361  swapchainCI.pQueueFamilyIndices = &queueFamilyIndices.First;
362  }
363  else
364  {
365  swapchainCI.imageSharingMode = VkSharingMode.Exclusive;
366  swapchainCI.queueFamilyIndexCount = 0;
367  }
368 
369  swapchainCI.preTransform = VkSurfaceTransformFlagsKHR.IdentityKHR;
370  swapchainCI.compositeAlpha = VkCompositeAlphaFlagsKHR.OpaqueKHR;
371  swapchainCI.clipped = true;
372 
373  VkSwapchainKHR oldSwapchain = _deviceSwapchain;
374  swapchainCI.oldSwapchain = oldSwapchain;
375 
376  result = vkCreateSwapchainKHR(_gd.Device, ref swapchainCI, null, out _deviceSwapchain);
377  CheckResult(result);
378  if (oldSwapchain != VkSwapchainKHR.Null)
379  {
380  vkDestroySwapchainKHR(_gd.Device, oldSwapchain, null);
381  }
382 
383  _framebuffer.SetNewSwapchain(_deviceSwapchain, width, height, surfaceFormat, swapchainCI.imageExtent);
384  return true;
385  }

392  private bool GetPresentQueueIndex(out uint queueFamilyIndex)
393  {
394  uint graphicsQueueIndex = _gd.GraphicsQueueIndex;
395  uint presentQueueIndex = _gd.PresentQueueIndex;
396 
397  if (QueueSupportsPresent(graphicsQueueIndex, _surface))
398  {

401  }
402  else if (graphicsQueueIndex != presentQueueIndex && QueueSupportsPresent(presentQueueIndex, _surface))

405  return true;
406  }
407 
408  queueFamilyIndex = 0;
409  return false;
410  }
