47  public static void Initialize(ref Manifold manifold, ref Transform xfA, float radiusA, ref Transform xfB,
48  float radiusB, out Vector2F normal, out FixedArray2<Vector2F> points, out FixedArray2<float> separations)
49  {
50  normal = Vector2F.Zero;
51  points = new FixedArray2<Vector2F>();
52  separations = new FixedArray2<float>();
53 
54  if (manifold.PointCount == 0)
55  {
56  return;
57  }
58 
59  switch (manifold.Type)
60  {
61  case ManifoldType.Circles:
62  {
63  normal = new Vector2F(1.0f, 0.0f);
64  Vector2F pointA = MathUtils.Mul(ref xfA, manifold.LocalPoint);
65  Vector2F pointB = MathUtils.Mul(ref xfB, manifold.Points.Value0.LocalPoint);
66  if (Vector2F.DistanceSquared(pointA, pointB) > Constant.Epsilon * Constant.Epsilon)
67  {
68  normal = pointB - pointA;
69  normal = Vector2F.Normalize(normal);
70  }
71 
72  Vector2F cA = pointA + radiusA * normal;
73  Vector2F cB = pointB - radiusB * normal;
74  points.Value0 = 0.5f * (cA + cB);
75  separations.Value0 = Vector2F.Dot(cB - cA, normal);
76  }
77  break;
78 
79  case ManifoldType.FaceA:
80  {
81  normal = MathUtils.Mul(xfA.Rotation, manifold.LocalNormal);
82  Vector2F planePoint = MathUtils.Mul(ref xfA, manifold.LocalPoint);
83 
84  for (int i = 0; i < manifold.PointCount; ++i)
85  {
86  Vector2F clipPoint = MathUtils.Mul(ref xfB, manifold.Points[i].LocalPoint);
87  Vector2F cA = clipPoint + (radiusA - Vector2F.Dot(clipPoint - planePoint, normal)) * normal;
88  Vector2F cB = clipPoint - radiusB * normal;
89  points[i] = 0.5f * (cA + cB);
90  separations[i] = Vector2F.Dot(cB - cA, normal);
91  }
92  }
93  break;
94 
95  case ManifoldType.FaceB:
96  {
97  normal = MathUtils.Mul(xfB.Rotation, manifold.LocalNormal);
98  Vector2F planePoint = MathUtils.Mul(ref xfB, manifold.LocalPoint);
99 
100  for (int i = 0; i < manifold.PointCount; ++i)
101  {
102  Vector2F clipPoint = MathUtils.Mul(ref xfA, manifold.Points[i].LocalPoint);
103  Vector2F cB = clipPoint + (radiusB - Vector2F.Dot(clipPoint - planePoint, normal)) * normal;
104  Vector2F cA = clipPoint - radiusA * normal;
105  points[i] = 0.5f * (cA + cB);
106  separations[i] = Vector2F.Dot(cA - cB, normal);
107  }
108 
109  // Ensure normal points from A to B.
110  normal = -normal;
111  }
112  break;
113  }
114  }
115  }
