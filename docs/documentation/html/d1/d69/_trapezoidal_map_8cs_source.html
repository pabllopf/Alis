175

186  }
187

201  cross = t.Top;
202

254  trapezoids[2].UpdateLeftRight(trapezoids[0], trapezoids[1], t.UpperRight, t.LowerRight);

259  // Create an AABB around segments
265  public Trapezoid BoundingBox(List<Edge> edges)
266  {
267  Point max = edges[0].P + margin;
268  Point min = edges[0].Q - margin;
269 
270  foreach (Edge e in edges)
271  {
272  if (e.P.X > max.X)
273  {
274  max = new Point(e.P.X + margin, max.Y);
275  }
276 
277  if (e.P.Y > max.Y)
278  {
279  max = new Point(max.X, e.P.Y + margin);
280  }
281 
282  if (e.Q.X > max.X)
283  {
284  max = new Point(e.Q.X + margin, max.Y);
285  }
286 
287  if (e.Q.Y > max.Y)
288  {
289  max = new Point(max.X, e.Q.Y + margin);
290  }
291 
292  if (e.P.X < min.X)
293  {
294  min = new Point(e.P.X - margin, min.Y);
295  }
296 
297  if (e.P.Y < min.Y)
298  {
299  min = new Point(min.X, e.P.Y - margin);
300  }
301 
302  if (e.Q.X < min.X)
303  {
304  min = new Point(e.Q.X - margin, min.Y);
305  }
306 
307  if (e.Q.Y < min.Y)
308  {
309  min = new Point(min.X, e.Q.Y - margin);
310  }
311  }
312 
313  Edge top = new Edge(new Point(min.X, max.Y), new Point(max.X, max.Y));
314  Edge bottom = new Edge(new Point(min.X, min.Y), new Point(max.X, min.Y));
315  Point left = bottom.P;
316  Point right = top.Q;
317 
318  return new Trapezoid(left, right, top, bottom);
319  }
