442  VkCommandBuffer cb,
443  uint baseMipLevel,
444  uint levelCount,
445  uint baseArrayLayer,
446  uint layerCount,
447  VkImageLayout newLayout)
448  {
449  if (_stagingBuffer != Vulkan.VkBuffer.Null)
450  {
451  return;
452  }
453 
454  VkImageLayout oldLayout = _imageLayouts[CalculateSubresource(baseMipLevel, baseArrayLayer)];
455 #if DEBUG
456  for (uint level = 0; level < levelCount; level++)
457  {
458  for (uint layer = 0; layer < layerCount; layer++)
459  {

461  {
462  throw new VeldridException("Unexpected image layout.");
463  }
464  }
465  }
466 #endif

468  {
469  VkImageAspectFlags aspectMask;

471  {

474  : VkImageAspectFlags.Depth;
475  }
476  else
477  {
478  aspectMask = VkImageAspectFlags.Color;
479  }
480  VulkanUtil.TransitionImageLayout(
481  cb,
482  OptimalDeviceImage,
483  baseMipLevel,
484  levelCount,
485  baseArrayLayer,
486  layerCount,
487  aspectMask,
488  _imageLayouts[CalculateSubresource(baseMipLevel, baseArrayLayer)],
489  newLayout);
490 
491  for (uint level = 0; level < levelCount; level++)
492  {
493  for (uint layer = 0; layer < layerCount; layer++)
494  {
495  _imageLayouts[CalculateSubresource(baseMipLevel + level, baseArrayLayer + layer)] = newLayout;
