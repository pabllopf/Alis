519  public static string UTF8_ToManaged(IntPtr s, bool freePtr = false)
520  {
521  if (s == IntPtr.Zero)
522  {
523  return null;
524  }
525 
526  int len = 0;
527  while (Marshal.ReadByte(s, len) != 0)
528  {
529  len++;
530  }
531 
532  if (len == 0)
533  {
534  return string.Empty;
535  }
536 
537  byte[] buffer = new byte[len];
538  Marshal.Copy(s, buffer, 0, len);
539  string result = Encoding.UTF8.GetString(buffer, 0, len);
540 
541  if (freePtr)
542  {
543  SdlMixerExtern.SDL_free(s);
544  }
545 
546  return result;
547  }
