520  public static string UTF8_ToManaged(IntPtr s, bool freePtr = false)
521  {
522  if (s == IntPtr.Zero)
523  {
524  return null;
525  }
526 
527  int len = 0;
528  while (Marshal.ReadByte(s, len) != 0)
529  {
530  len++;
531  }
532 
533  if (len == 0)
534  {
535  return string.Empty;
536  }
537 
538  byte[] buffer = new byte[len];
539  Marshal.Copy(s, buffer, 0, len);
540  string result = Encoding.UTF8.GetString(buffer, 0, len);
541 
542  if (freePtr)
543  {
544  SdlMixerExtern.SDL_free(s);
545  }
546 
547  return result;
548  }
