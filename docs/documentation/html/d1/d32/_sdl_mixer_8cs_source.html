539  /* Used for stack allocated string marshaling. */
545  internal static int Utf8Size(string str)
546  {
547  if (str == null)
548  {
549  return 0;
550  }
551 
552  return str.Length * 4 + 1;
553  }

562  internal static unsafe byte* Utf8Encode(string str, byte* buffer, int bufferSize)
563  {
564  if (str == null)
565  {
566  return (byte*) 0;
567  }
568 
569  fixed (char* strPtr = str)
570  {
571  Encoding.UTF8.GetBytes(strPtr, str.Length + 1, buffer, bufferSize);
572  }
573 
574  return buffer;
575  }
576 
577  /* Used for heap allocated string marshaling.
578  * Returned byte* must be free'd with FreeHGlobal.
579  */
585  internal static unsafe byte* Utf8EncodeHeap(string str)
586  {
587  if (str == null)
588  {
589  return (byte*) 0;
590  }
591 
592  int bufferSize = Utf8Size(str);
593  byte* buffer = (byte*) Marshal.AllocHGlobal(bufferSize);
594  fixed (char* strPtr = str)
595  {
596  Encoding.UTF8.GetBytes(strPtr, str.Length + 1, buffer, bufferSize);
597  }
598 
599  return buffer;
600  }

602  /* This is public because SDL_DropEvent needs it! */
609  public static unsafe string UTF8_ToManaged(IntPtr s, bool freePtr = false)
610  {
611  if (s == IntPtr.Zero)
612  {
613  return null;
614  }
615 
616  /* We get to do strlen ourselves! */
617  byte* ptr = (byte*) s;
618  while (*ptr != 0)
619  {
620  ptr++;
621  }
622 
623  /* TODO: This #ifdef is only here because the equivalent
624  * .NET 2.0 constructor appears to be less efficient?
625  * Here's the pretty version, maybe steal this instead:
626  *
627  string result = new string(
628  (sbyte*) s, // Also, why sbyte???
629  0,
630  (int) (ptr - (byte*) s),
631  System.Text.Encoding.UTF8
632  );
633  * See the CoreCLR source for more info.
634  * -flibit
635  */
636 #if NETSTANDARD2_0
637  /* Modern C# lets you just send the byte*, nice! */
638  string result = System.Text.Encoding.UTF8.GetString(
639  (byte*) s,
640  (int) (ptr - (byte*) s)
641  );
642 #else
643  /* Old C# requires an extra memcpy, bleh! */
644  int len = (int) (ptr - (byte*) s);
645  if (len == 0)
646  {
647  return string.Empty;
648  }
649 
650  char* chars = stackalloc char[len];
651  int strLen = Encoding.UTF8.GetChars((byte*) s, len, chars, len);
652  string result = new string(chars, 0, strLen);
653 #endif
654 
655  /* Some SDL functions will malloc, we have to free! */
656  if (freePtr)
657  {
658  SdlMixerExtern.SDL_free(s);
659  }
660 
661  return result;
662  }
