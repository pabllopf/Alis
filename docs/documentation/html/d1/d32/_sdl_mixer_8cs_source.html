563  public static string UTF8_ToManaged(IntPtr s, bool freePtr = false)
564  {
565  if (s == IntPtr.Zero)
566  {
567  return null;
568  }
569 
570  int len = 0;
571  while (Marshal.ReadByte(s, len) != 0)
572  {
573  len++;
574  }
575 
576  if (len == 0)
577  {
578  return string.Empty;
579  }
580 
581  byte[] buffer = new byte[len];
582  Marshal.Copy(s, buffer, 0, len);
583  string result = Encoding.UTF8.GetString(buffer, 0, len);
584 
585  if (freePtr)
586  {
587  SdlMixerExtern.SDL_free(s);
588  }
589 
590  return result;
591  }
