205  public override void PrepareTriangulation(ITriangulatable t)
206  {
207  base.PrepareTriangulation(t);
208 
209  double xmin;
210  double ymin;
211 
212  double xmax = xmin = Points[0].X;
213  double ymax = ymin = Points[0].Y;
214 
215  // Calculate bounds. Should be combined with the sorting
216  foreach (TriangulationPoint p in Points)
217  {
218  if (p.X > xmax)
219  {
220  xmax = p.X;
221  }
222 
223  if (p.X < xmin)
224  {
225  xmin = p.X;
226  }
227 
228  if (p.Y > ymax)
229  {
230  ymax = p.Y;
231  }
232 
233  if (p.Y < ymin)
234  {
235  ymin = p.Y;
236  }
237  }
238 
239  double deltaX = Alpha * (xmax - xmin);
240  double deltaY = Alpha * (ymax - ymin);
241  TriangulationPoint p1 = new TriangulationPoint(xmax + deltaX, ymin - deltaY);
242  TriangulationPoint p2 = new TriangulationPoint(xmin - deltaX, ymin - deltaY);
243 
244  Head = p1;
245  Tail = p2;
246 
247  // long time = System.nanoTime();
248  // Sort the points along y-axis
249  Points.Sort(comparator);
250 
251  // logger.info( "Triangulation setup [{}ms]", ( System.nanoTime() - time ) / 1e6 );
252  }
