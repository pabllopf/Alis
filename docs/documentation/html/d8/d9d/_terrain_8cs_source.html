137  public void Initialize()
138  {
139  // find top left of terrain in world space
140  topLeft = new Vector2(Center.X - Width * 0.5f, Center.Y - -Height * 0.5f);
141 
142  // convert the terrains size to a point cloud size
143  localWidth = Width * PointsPerUnit;
144  localHeight = Height * PointsPerUnit;
145 
146  terrainMap = new sbyte[(int) localWidth + 1, (int) localHeight + 1];
147 
148  for (int x = 0; x < localWidth; x++)
149  {
150  for (int y = 0; y < localHeight; y++)
151  {
152  terrainMap[x, y] = 1;
153  }
154  }
155 
156  xnum = (int) (localWidth / CellSize);
157  ynum = (int) (localHeight / CellSize);
158  bodyMap = new List<Body>[xnum, ynum];
159 
160  // make sure to mark the dirty area to an infinitely small box

188  public void ModifyTerrain(Vector2 location, sbyte value)
189  {
190  // find local position
191  // make position local to map space
192  Vector2 p = location - topLeft;
193 
194  // find map position for each axis
195  p.X = p.X * localWidth / Width;
196  p.Y = p.Y * -localHeight / Height;
197 
198  if (p.X >= 0 && p.X < localWidth && p.Y >= 0 && p.Y < localHeight)
199  {
200  terrainMap[(int) p.X, (int) p.Y] = value;
201 
202  unchecked
203  {
204  // expand dirty area
205  if (p.X < (int) dirtyArea.LowerBound.X)
206  {
207  dirtyArea.LowerBound.X = p.X;
208  }
209 
210  if (p.X > (int) dirtyArea.UpperBound.X)
211  {
212  dirtyArea.UpperBound.X = p.X;
213  }
214 
215  if (p.Y < (int) dirtyArea.LowerBound.Y)
216  {
217  dirtyArea.LowerBound.Y = p.Y;
218  }
219 
220  if (p.Y > (int) dirtyArea.UpperBound.Y)
221  {
222  dirtyArea.UpperBound.Y = p.Y;
223  }
224  }
225  }

229  public void RegenerateTerrain()
230  {
231  unchecked
232  {
233  //iterate effected cells
234  int xStart = (int) (dirtyArea.LowerBound.X / CellSize);
235  if (xStart < 0)
236  {
237  xStart = 0;
238  }
239 
240  int xEnd = (int) (dirtyArea.UpperBound.X / CellSize) + 1;
241  if (xEnd > xnum)
242  {
243  xEnd = xnum;
244  }
245 
246  int yStart = (int) (dirtyArea.LowerBound.Y / CellSize);
247  if (yStart < 0)
248  {
249  yStart = 0;
250  }
251 
252  int yEnd = (int) (dirtyArea.UpperBound.Y / CellSize) + 1;
253  if (yEnd > ynum)
254  {
255  yEnd = ynum;
256  }
257 
258  RemoveOldData(xStart, xEnd, yStart, yEnd);
259 
260  dirtyArea = new Aabb(new Vector2(float.MaxValue, float.MaxValue),
261  new Vector2(float.MinValue, float.MinValue));
262  }
263  }

272  private void RemoveOldData(int xStart, int xEnd, int yStart, int yEnd)
273  {
274  for (int x = xStart; x < xEnd; x++)
275  {
276  for (int y = yStart; y < yEnd; y++)
277  {
278  //remove old terrain object at grid cell
279  if (bodyMap[x, y] != null)
280  {
281  for (int i = 0; i < bodyMap[x, y].Count; i++)
282  {
283  World.RemoveBody(bodyMap[x, y][i]);
284  }
285  }
286 
287  bodyMap[x, y] = null;
288 
289  //generate new one
290  GenerateTerrain(x, y);
291  }
292  }

300  private void GenerateTerrain(int gx, int gy)
301  {
302  float ax = gx * CellSize;
303  float ay = gy * CellSize;
304 
305  List<Vertices> polys = MarchingSquares.DetectSquares(
306  new Aabb(new Vector2(ax, ay), new Vector2(ax + CellSize, ay + CellSize)), SubCellSize, SubCellSize,
307  terrainMap, Iterations, true);
308  if (polys.Count == 0)
309  {
310  return;
311  }
312 
313  bodyMap[gx, gy] = new List<Body>();
314 
315  // create the scale vector
316  Vector2 scale = new Vector2(1f / PointsPerUnit, 1f / -PointsPerUnit);
317 
318  // create physics object for this grid cell
319  foreach (Vertices item in polys)
320  {
321  // does this need to be negative?
322  item.Scale(ref scale);
323  item.Translate(ref topLeft);
324  Vertices simplified = SimplifyTools.CollinearSimplify(item);
325 
326  List<Vertices> decompPolys = Triangulate.ConvexPartition(simplified, Decomposer);
327 
328  foreach (Vertices poly in decompPolys)
329  {
330  if (poly.Count > 2)
331  {
332  bodyMap[gx, gy].Add(BodyFactory.CreatePolygon(World, poly, 1));
333  }
334  }
335  }
336  }
337  }
