77  public static void ExtractEmbeddedDlls(string dllName, byte[] resourceBytes)
78  {
79  Assembly assem = Assembly.GetExecutingAssembly();
80  string[] names = assem.GetManifestResourceNames();
81  AssemblyName an = assem.GetName();
82 
83  // The temporary folder holds one or more of the temporary DLLs
84  // It is made "unique" to avoid different versions of the DLL or architectures.
85  tempFolder = String.Format("{0}.{1}.{2}", an.Name, an.ProcessorArchitecture, an.Version);
86 
87  dirName = Path.Combine(Path.GetTempPath(), tempFolder);
88  dirName = Environment.CurrentDirectory;
89 
90 
91  if (!Directory.Exists(dirName))
92  {
93  Directory.CreateDirectory(dirName);
94  }
95 
96  // Add the temporary dirName to the PATH environment variable (at the head!)
97  string path = Environment.GetEnvironmentVariable("PATH");
98  string[] pathPieces = path.Split(';');
99  bool found = false;
100  foreach (string pathPiece in pathPieces)
101  {
102  if (pathPiece == dirName)
103  {
104  found = true;
105  break;
106  }
107  }
108  if (!found)
109  {
110  Environment.SetEnvironmentVariable("PATH", dirName + ";" + path);
111  }
112 
113  // See if the file exists, avoid rewriting it if not necessary
114  string dllPath = Path.Combine(dirName, dllName);
115  if (File.Exists(dllPath)) {
116  File.Delete(dllPath);
117  }
118  File.WriteAllBytes(dllPath, resourceBytes);
119  }
