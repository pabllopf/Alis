48  public static void ExtractEmbeddedDlls(string dllName, Dictionary<(OSPlatform Platform, Architecture Arch), byte[]> dllBytes)
49  {
50  string extension = GetDllExtension();
51 
52  string dllPath = Path.Combine(Environment.CurrentDirectory, $"{dllName}.{extension}");
53 
54  if (!File.Exists(dllPath))
55  {
56  OSPlatform currentPlatform = GetCurrentPlatform();
57  Architecture currentArchitecture = RuntimeInformation.ProcessArchitecture;
58 
59  if (dllBytes.TryGetValue((currentPlatform, currentArchitecture), out byte[] resourceBytes))
60  {
61  File.WriteAllBytes(dllPath, resourceBytes);
62  Console.WriteLine($"dllPath={dllPath}");
63  }
64  }
65  }

72  private static string GetDllExtension()
73  {
74  OSPlatform currentPlatform = GetCurrentPlatform();
75 
76  if (currentPlatform == OSPlatform.Windows)
77  {
78  return "dll";
79  }
80 
81  if (currentPlatform == OSPlatform.OSX || currentPlatform == OSPlatform.Create("IOS"))

83  return "dylib";
84  }
85 
86  if (currentPlatform == OSPlatform.Linux || currentPlatform == OSPlatform.Create("Android"))
87  {
88  return "so";
