49  public static void ExtractEmbeddedDlls(string dllName, Dictionary<(OSPlatform Platform, Architecture Arch), byte[]> dllBytes)
50  {
51  string extension = GetDllExtension();
52 
53  string dllPath = Path.Combine(Environment.CurrentDirectory, $"{dllName}.{extension}");
54 
55  if (!File.Exists(dllPath))
56  {
57  OSPlatform currentPlatform = GetCurrentPlatform();
58  Architecture currentArchitecture = RuntimeInformation.ProcessArchitecture;
59 
60  if (dllBytes.TryGetValue((currentPlatform, currentArchitecture), out byte[] resourceBytes))
61  {
62  ExtractZipFile(dllPath,$"{dllName}.{extension}" ,resourceBytes);
63  Console.WriteLine($"OSPlatform={currentPlatform} | Architecture={currentArchitecture} -> lib: {dllName}");
64  }
65  }
66  }

73  private static string GetDllExtension()
74  {
75  OSPlatform currentPlatform = GetCurrentPlatform();
76 
77  if (currentPlatform == OSPlatform.Windows)
78  {
79  return "dll";
80  }
81 
82  if (currentPlatform == OSPlatform.OSX || currentPlatform == OSPlatform.Create("IOS"))
83  {
84  return "dylib";
85  }
86 
87  if (currentPlatform == OSPlatform.Linux || currentPlatform == OSPlatform.Create("Android"))
88  {
89  return "so";
90  }
91 
92  throw new PlatformNotSupportedException("Unsupported platform.");
93  }

137  private static void ExtractZipFile(string filePath, string fileName, byte[] zipData)
138  {
139  using (MemoryStream ms = new MemoryStream(zipData))
140  {
141  using (ZipArchive archive = new ZipArchive(ms))
142  {
143  foreach (ZipArchiveEntry entry in archive.Entries)
144  {
145  string entryPath = Path.Combine(Environment.CurrentDirectory, fileName);
146 
147  if (File.Exists(entryPath))
148  {
149  continue; // Skip if the file already exists
150  }
151 
152  Directory.CreateDirectory(Path.GetDirectoryName(entryPath));
153  using (Stream entryStream = entry.Open())
154  using (FileStream fs = File.Create(entryPath))
155  {
156  entryStream.CopyTo(fs);
157  }
158  }
159  }
160  }
161  }
