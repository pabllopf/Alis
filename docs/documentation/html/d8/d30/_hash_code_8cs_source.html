105  public static int Combine<T1>(T1 value1)
106  {
107  // Provide a way of diffusing bits from something with a limited
108  // input hash space. For example, many enums only have a few
109  // possible hashes, only using the bottom few bits of the code. Some
110  // collections are built on the assumption that hashes are spread
111  // over a larger space, so diffusing the bits may help the
112  // collection work more efficiently.
113 
114  uint hc1 = (uint) (value1?.GetHashCode() ?? 0);
115 
116  uint hash = MixEmptyState();
117  hash += 4;
118 
119  hash = QueueRound(hash, hc1);
120 
121  hash = MixFinal(hash);
122  return (int) hash;
123  }

485  private void Add(int value)
486  {
487  // The original xxHash works as follows:
488  // 0. Initialize immediately. We can't do this in a struct (no
489  // default ctor).
490  // 1. Accumulate blocks of length 16 (4 uints) into 4 accumulators.
491  // 2. Accumulate remaining blocks of length 4 (1 uint) into the
492  // hash.
493  // 3. Accumulate remaining blocks of length 1 into the hash.
494 
495  // There is no need for #3 as this type only accepts ints. _queue1,
496  // _queue2 and _queue3 are basically a buffer so that when
497  // ToHashCode is called we can execute #2 correctly.
498 
499  // We need to initialize the xxHash32 state (_v1 to _v4) lazily (see
500  // #0) nd the last place that can be done if you look at the
501  // original code is just before the first block of 16 bytes is mixed
502  // in. The xxHash32 state is never used for streams containing fewer
503  // than 16 bytes.
504 
505  // To see what's really going on here, have a look at the Combine
506  // methods.
507 
508  uint val = (uint) value;
509 
510  // Storing the value of _length locally shaves of quite a few bytes
511  // in the resulting machine code.
512  uint previousLength = _length++;
513  uint position = previousLength % 4;
514 
515  // Switch can't be inlined.
516 
517  if (position == 0)
518  {
519  _queue1 = val;
520  }
521  else if (position == 1)
522  {
523  _queue2 = val;
524  }
525  else if (position == 2)
526  {
527  _queue3 = val;
528  }
529  else // position == 3
530  {
531  if (previousLength == 3)
532  {
533  Initialize(out _v1, out _v2, out _v3, out _v4);
534  }
535 
536  _v1 = Round(_v1, _queue1);
537  _v2 = Round(_v2, _queue2);
538  _v3 = Round(_v3, _queue3);
539  _v4 = Round(_v4, val);
540  }
541  }

547  public int ToHashCode()
548  {
549  // Storing the value of _length locally shaves of quite a few bytes
550  // in the resulting machine code.
551  uint length = _length;
552 
553  // position refers to the *next* queue position in this method, so
554  // position == 1 means that _queue1 is populated; _queue2 would have
555  // been populated on the next call to Add.
556  uint position = length % 4;
557 
558  // If the length is less than 4, _v1 to _v4 don't contain anything
559  // yet. xxHash32 treats this differently.
560 
561  uint hash = length < 4 ? MixEmptyState() : MixState(_v1, _v2, _v3, _v4);
562 
563  // _length is incremented once per Add(Int32) and is therefore 4
564  // times too small (xxHash length is in bytes, not ints).
565 
566  hash += length * 4;
567 
568  // Mix what remains in the queue
569 
570  // Switch can't be inlined right now, so use as few branches as
571  // possible by manually excluding impossible scenarios (position > 1
572  // is always false if position is not > 0).
573  if (position > 0)
574  {
575  hash = QueueRound(hash, _queue1);
576  if (position > 1)
577  {
578  hash = QueueRound(hash, _queue2);
579  if (position > 2)
580  {
581  hash = QueueRound(hash, _queue3);
582  }
583  }
584  }
585 
586  hash = MixFinal(hash);
587  return (int) hash;
588  }
