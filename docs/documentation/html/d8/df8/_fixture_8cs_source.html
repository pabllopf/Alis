269  private void Refilter()
270  {
271  // Flag associated contacts for filtering.
272  ContactEdge edge = Body.ContactList;
273  while (edge != null)
274  {
275  Contact contact = edge.Contact;
276  Fixture fixtureA = contact.FixtureA;
277  Fixture fixtureB = contact.FixtureB;
278  if (fixtureA == this || fixtureB == this)
279  {
280  contact.Flags |= ContactFlags.FilterFlag;
281  }
282 
283  edge = edge.Next;
284  }
285 
286  // Touch each proxy so that new pairs may be created
287  IBroadPhase broadPhase = ContactManager.Current.BroadPhase;
288  for (int i = 0; i < ProxyCount; ++i)
289  {
290  broadPhase.TouchProxy(Proxies[i].ProxyId);
291  }
292  }

338  // These support body activation/deactivation.
344  internal void CreateProxies(IBroadPhase broadPhase, ref Transform xf)
345  {
346  //Debug.Assert(ProxyCount == 0);
347 
348  // Create proxies in the broad-phase.
349  ProxyCount = Shape.ChildCount;
350 
351  for (int i = 0; i < ProxyCount; ++i)
352  {
353  FixtureProxy proxy = new FixtureProxy();
354  Shape.ComputeAabb(ref xf, i, out proxy.Aabb);
355  proxy.Fixture = this;
356  proxy.ChildIndex = i;
357 
358  //Velcro note: This line needs to be after the previous two because FixtureProxy is a struct
359  proxy.ProxyId = broadPhase.AddProxy(ref proxy);
360 
361  Proxies[i] = proxy;
362  }
363  }
364 
369  internal void DestroyProxies(IBroadPhase broadPhase)
370  {
371  // Destroy proxies in the broad-phase.
372  for (int i = 0; i < ProxyCount; ++i)
373  {
374  FixtureProxy proxy = Proxies[i];
375  broadPhase.RemoveProxy(proxy.ProxyId);
376  proxy.ProxyId = DynamicTreeBroadPhase.NullProxy;
377  }
378 
379  ProxyCount = 0;
380  }

388  internal void Synchronize(IBroadPhase broadPhase, ref Transform transform1, ref Transform transform2)
389  {
390  if (ProxyCount == 0)
391  {
392  return;
393  }
394 
395  for (int i = 0; i < ProxyCount; ++i)
396  {
397  FixtureProxy proxy = Proxies[i];
398 
399  // Compute an AABB that covers the swept Shape (may miss some rotation effect).
400  Shape.ComputeAabb(ref transform1, proxy.ChildIndex, out Aabb aabb1);
401  Shape.ComputeAabb(ref transform2, proxy.ChildIndex, out Aabb aabb2);
402 
403  proxy.Aabb.Combine(ref aabb1, ref aabb2);
404 
405 
406  Vector2F displacement = aabb2.Center - aabb1.Center;
407 
408  broadPhase.MoveProxy(proxy.ProxyId, ref proxy.Aabb, displacement);
409  }
410  }
