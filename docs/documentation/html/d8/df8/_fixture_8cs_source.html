327  private void Refilter()
328  {
329  // Flag associated contacts for filtering.
330  ContactEdge edge = Bodyprivate.ContactList;
331  while (edge != null)
332  {
333  Contact contact = edge.Contact;
334  Fixture fixtureA = contact.FixtureA;
335  Fixture fixtureB = contact.FixtureB;
336  if (fixtureA == this || fixtureB == this)
337  {
338  contact.Flags |= ContactFlags.FilterFlag;
339  }
340 
341  edge = edge.Next;
342  }
343 
344  World world = Bodyprivate.World;
345 
346  if (world == null)
347  {
348  return;
349  }
350 
351  // Touch each proxy so that new pairs may be created
352  IBroadPhase broadPhase = world.ContactManager.BroadPhase;
353  for (int i = 0; i < proxyCount; ++i)
354  {
355  broadPhase.TouchProxy(proxies[i].ProxyId);
356  }
357  }

403  // These support body activation/deactivation.
409  internal void CreateProxies(IBroadPhase broadPhase, ref Transform xf)
410  {
411  Debug.Assert(proxyCount == 0);
412 
413  // Create proxies in the broad-phase.
414  proxyCount = ShapePrivate.ChildCount;
415 
416  for (int i = 0; i < proxyCount; ++i)
417  {
418  FixtureProxy proxy = new FixtureProxy();
419  ShapePrivate.ComputeAabb(ref xf, i, out proxy.Aabb);
420  proxy.Fixture = this;
421  proxy.ChildIndex = i;
422 
423  //Velcro note: This line needs to be after the previous two because FixtureProxy is a struct
424  proxy.ProxyId = broadPhase.AddProxy(ref proxy);
425 
426  proxies[i] = proxy;
427  }
428  }
429 
434  internal void DestroyProxies(IBroadPhase broadPhase)
435  {
436  // Destroy proxies in the broad-phase.
437  for (int i = 0; i < proxyCount; ++i)
438  {
439  FixtureProxy proxy = proxies[i];
440  broadPhase.RemoveProxy(proxy.ProxyId);
441  proxy.ProxyId = DynamicTreeBroadPhase.NullProxy;
442  }
443 
444  proxyCount = 0;
445  }
446 
453  internal void Synchronize(IBroadPhase broadPhase, ref Transform transform1, ref Transform transform2)
454  {
455  if (proxyCount == 0)
456  {
457  return;
458  }
459 
460  for (int i = 0; i < proxyCount; ++i)
461  {
462  FixtureProxy proxy = proxies[i];
463 
464  // Compute an AABB that covers the swept Shape (may miss some rotation effect).
465  Shape.ComputeAabb(ref transform1, proxy.ChildIndex, out Aabb aabb1);
466  Shape.ComputeAabb(ref transform2, proxy.ChildIndex, out Aabb aabb2);
467 
468  proxy.Aabb.Combine(ref aabb1, ref aabb2);
469 
470  Vector2 displacement = aabb2.Center - aabb1.Center;
471 
472  broadPhase.MoveProxy(proxy.ProxyId, ref proxy.Aabb, displacement);
473  }
474  }
475  }
