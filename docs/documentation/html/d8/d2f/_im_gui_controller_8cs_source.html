799  private void UpdateImGuiInput(InputSnapshot snapshot)
800  {
801  ImGuiIOPtr io = ImGui.GetIO();
802 
803  Vector2 mousePosition = snapshot.MousePosition;
804 
805  // Determine if any of the mouse buttons were pressed during this snapshot period, even if they are no longer held.
806  bool leftPressed = false;
807  bool middlePressed = false;
808  bool rightPressed = false;
809  foreach (MouseEvent me in snapshot.MouseEvents)
810  {
811  if (me.Down)
812  {
813  switch (me.MouseButton)
814  {
815  case MouseButton.Left:
816  leftPressed = true;
817  break;
818  case MouseButton.Middle:
819  middlePressed = true;
820  break;
821  case MouseButton.Right:
822  rightPressed = true;
823  break;
824  }
825  }
826  }
827 
828  io.MouseDown[0] = leftPressed || snapshot.IsMouseDown(MouseButton.Left);
829  io.MouseDown[1] = middlePressed || snapshot.IsMouseDown(MouseButton.Right);
830  io.MouseDown[2] = rightPressed || snapshot.IsMouseDown(MouseButton.Middle);
831 
832  if (p_sdl_GetGlobalMouseState == null)
833  {
834  p_sdl_GetGlobalMouseState = Sdl2Native.LoadFunction<SDL_GetGlobalMouseState_t>("SDL_GetGlobalMouseState");
835  }
836 
837  int x, y;
838  unsafe
839  {
840  uint buttons = p_sdl_GetGlobalMouseState(&x, &y);
841  io.MouseDown[0] = (buttons & 0b0001) != 0;
842  io.MouseDown[1] = (buttons & 0b0010) != 0;
843  io.MouseDown[2] = (buttons & 0b0100) != 0;
844  }
845 
846  io.MousePos = new Vector2(x, y);
847  io.MouseWheel = snapshot.WheelDelta;
848 
849  IReadOnlyList<char> keyCharPresses = snapshot.KeyCharPresses;
850  for (int i = 0; i < keyCharPresses.Count; i++)
851  {
852  char c = keyCharPresses[i];
853  io.AddInputCharacter(c);
854  }
855 
856  IReadOnlyList<KeyEvent> keyEvents = snapshot.KeyEvents;
857  for (int i = 0; i < keyEvents.Count; i++)
858  {
859  KeyEvent keyEvent = keyEvents[i];
860  io.KeysDown[(int)keyEvent.Key] = keyEvent.Down;
861  if (keyEvent.Key == Key.ControlLeft)
862  {
863  _controlDown = keyEvent.Down;
864  }
865  if (keyEvent.Key == Key.ShiftLeft)
866  {
867  _shiftDown = keyEvent.Down;
868  }
869  if (keyEvent.Key == Key.AltLeft)
870  {
871  _altDown = keyEvent.Down;
872  }
873  if (keyEvent.Key == Key.WinLeft)
874  {
875  _winKeyDown = keyEvent.Down;
876  }
877  }
878 
879  io.KeyCtrl = _controlDown;
880  io.KeyAlt = _altDown;
881  io.KeyShift = _shiftDown;
882  io.KeySuper = _winKeyDown;
883 
884  ImVector<ImGuiViewportPtr> viewports = ImGui.GetPlatformIO().Viewports;
885  for (int i = 1; i < viewports.Size; i++)
886  {
887  ImGuiViewportPtr v = viewports[i];
888  VeldridImGuiWindow window = ((VeldridImGuiWindow)GCHandle.FromIntPtr(v.PlatformUserData).Target);
889  window.Update();
890  }

932  uint indexOffsetInElements = 0;
