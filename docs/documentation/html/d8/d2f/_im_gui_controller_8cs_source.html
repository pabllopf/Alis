798  private void UpdateImGuiInput(InputSnapshot snapshot)
799  {
800  ImGuiIOPtr io = ImGui.GetIO();
801 
802  Vector2 mousePosition = snapshot.MousePosition;
803 
804  // Determine if any of the mouse buttons were pressed during this snapshot period, even if they are no longer held.
805  bool leftPressed = false;
806  bool middlePressed = false;
807  bool rightPressed = false;
808  foreach (MouseEvent me in snapshot.MouseEvents)
809  {
810  if (me.Down)
811  {
812  switch (me.MouseButton)
813  {
814  case MouseButton.Left:
815  leftPressed = true;
816  break;
817  case MouseButton.Middle:
818  middlePressed = true;
819  break;
820  case MouseButton.Right:
821  rightPressed = true;
822  break;
823  }
824  }
825  }
826 
827  io.MouseDown[0] = leftPressed || snapshot.IsMouseDown(MouseButton.Left);
828  io.MouseDown[1] = middlePressed || snapshot.IsMouseDown(MouseButton.Right);
829  io.MouseDown[2] = rightPressed || snapshot.IsMouseDown(MouseButton.Middle);
830 
831  if (p_sdl_GetGlobalMouseState == null)
832  {
833  p_sdl_GetGlobalMouseState = Sdl2Native.LoadFunction<SDL_GetGlobalMouseState_t>("SDL_GetGlobalMouseState");
834  }
835 
836  int x, y;
837  unsafe
838  {
839  uint buttons = p_sdl_GetGlobalMouseState(&x, &y);
840  io.MouseDown[0] = (buttons & 0b0001) != 0;
841  io.MouseDown[1] = (buttons & 0b0010) != 0;
842  io.MouseDown[2] = (buttons & 0b0100) != 0;
843  }
844 
845  io.MousePos = new Vector2(x, y);
846  io.MouseWheel = snapshot.WheelDelta;
847 
848  IReadOnlyList<char> keyCharPresses = snapshot.KeyCharPresses;
849  for (int i = 0; i < keyCharPresses.Count; i++)
850  {
851  char c = keyCharPresses[i];
852  io.AddInputCharacter(c);
853  }
854 
855  IReadOnlyList<KeyEvent> keyEvents = snapshot.KeyEvents;
856  for (int i = 0; i < keyEvents.Count; i++)
857  {
858  KeyEvent keyEvent = keyEvents[i];
859  io.KeysDown[(int)keyEvent.Key] = keyEvent.Down;
860  if (keyEvent.Key == Key.ControlLeft)
861  {
862  _controlDown = keyEvent.Down;
863  }
864  if (keyEvent.Key == Key.ShiftLeft)
865  {
866  _shiftDown = keyEvent.Down;
867  }
868  if (keyEvent.Key == Key.AltLeft)
869  {
870  _altDown = keyEvent.Down;
871  }
872  if (keyEvent.Key == Key.WinLeft)
873  {
874  _winKeyDown = keyEvent.Down;
875  }
876  }
877 
878  io.KeyCtrl = _controlDown;
879  io.KeyAlt = _altDown;
880  io.KeyShift = _shiftDown;
881  io.KeySuper = _winKeyDown;
882 
883  ImVector<ImGuiViewportPtr> viewports = ImGui.GetPlatformIO().Viewports;
884  for (int i = 1; i < viewports.Size; i++)
885  {
886  ImGuiViewportPtr v = viewports[i];
887  VeldridImGuiWindow window = ((VeldridImGuiWindow)GCHandle.FromIntPtr(v.PlatformUserData).Target);
888  window.Update();
889  }

928  private void RenderImDrawData(ImDrawDataPtr draw_data, GraphicsDevice gd, CommandList cl)
929  {
930  uint vertexOffsetInVertices = 0;
931  uint indexOffsetInElements = 0;
932 
933  if (draw_data.CmdListsCount == 0)
934  {
935  return;
936  }
937 
938  uint totalVBSize = (uint)(draw_data.TotalVtxCount * Unsafe.SizeOf<ImDrawVert>());
939  if (totalVBSize > _vertexBuffer.SizeInBytes)
940  {
941  gd.DisposeWhenIdle(_vertexBuffer);
942  _vertexBuffer = gd.ResourceFactory.CreateBuffer(new BufferDescription((uint)(totalVBSize * 1.5f), BufferUsage.VertexBuffer | BufferUsage.Dynamic));
943  }
944 
945  uint totalIBSize = (uint)(draw_data.TotalIdxCount * sizeof(ushort));
946  if (totalIBSize > _indexBuffer.SizeInBytes)
947  {
948  gd.DisposeWhenIdle(_indexBuffer);
949  _indexBuffer = gd.ResourceFactory.CreateBuffer(new BufferDescription((uint)(totalIBSize * 1.5f), BufferUsage.IndexBuffer | BufferUsage.Dynamic));
950  }
951 
952  Vector2 pos = draw_data.DisplayPos;
953  for (int i = 0; i < draw_data.CmdListsCount; i++)
954  {
955  ImDrawListPtr cmd_list = draw_data.CmdListsRange[i];
956 
957  cl.UpdateBuffer(
958  _vertexBuffer,
959  vertexOffsetInVertices * (uint)Unsafe.SizeOf<ImDrawVert>(),
960  cmd_list.VtxBuffer.Data,
961  (uint)(cmd_list.VtxBuffer.Size * Unsafe.SizeOf<ImDrawVert>()));
962 
963  cl.UpdateBuffer(
964  _indexBuffer,
965  indexOffsetInElements * sizeof(ushort),
966  cmd_list.IdxBuffer.Data,
967  (uint)(cmd_list.IdxBuffer.Size * sizeof(ushort)));
968 
969  vertexOffsetInVertices += (uint)cmd_list.VtxBuffer.Size;
970  indexOffsetInElements += (uint)cmd_list.IdxBuffer.Size;
971  }
972 
973  // Setup orthographic projection matrix into our constant buffer
974  ImGuiIOPtr io = ImGui.GetIO();
975  Matrix4x4 mvp = Matrix4x4.CreateOrthographicOffCenter(
976  pos.X,
977  pos.X + draw_data.DisplaySize.X,
978  pos.Y + draw_data.DisplaySize.Y,
979  pos.Y,
980  -1.0f,
981  1.0f);
982 
983  cl.UpdateBuffer(_projMatrixBuffer, 0, ref mvp);
984 
985  cl.SetVertexBuffer(0, _vertexBuffer);
986  cl.SetIndexBuffer(_indexBuffer, IndexFormat.UInt16);
987  cl.SetPipeline(_pipeline);
988  cl.SetGraphicsResourceSet(0, _mainResourceSet);
989 
990  draw_data.ScaleClipRects(io.DisplayFramebufferScale);
991 
992  // Render command lists
993  int vtx_offset = 0;
994  int idx_offset = 0;
995  for (int n = 0; n < draw_data.CmdListsCount; n++)
996  {
997  ImDrawListPtr cmd_list = draw_data.CmdListsRange[n];
998  for (int cmd_i = 0; cmd_i < cmd_list.CmdBuffer.Size; cmd_i++)
999  {
1000  ImDrawCmdPtr pcmd = cmd_list.CmdBuffer[cmd_i];
1001  if (pcmd.UserCallback != IntPtr.Zero)
1002  {
1003  throw new NotImplementedException();
1004  }
1005  else
1006  {
1007  if (pcmd.TextureId != IntPtr.Zero)
1008  {
1009  if (pcmd.TextureId == _fontAtlasID)
1010  {
1011  cl.SetGraphicsResourceSet(1, _fontTextureResourceSet);
1012  }
1013  else
1014  {
1015  cl.SetGraphicsResourceSet(1, GetImageResourceSet(pcmd.TextureId));
1016  }
1017  }
1018 
1019  cl.SetScissorRect(
1020  0,
1021  (uint)(pcmd.ClipRect.X - pos.X),
1022  (uint)(pcmd.ClipRect.Y - pos.Y),
1023  (uint)(pcmd.ClipRect.Z - pcmd.ClipRect.X),
1024  (uint)(pcmd.ClipRect.W - pcmd.ClipRect.Y));
1025 
1026  cl.DrawIndexed(pcmd.ElemCount, 1, pcmd.IdxOffset + (uint)idx_offset, (int)pcmd.VtxOffset + vtx_offset, 0);
1027  }
1028  }
1029  vtx_offset += cmd_list.VtxBuffer.Size;
1030  idx_offset += cmd_list.IdxBuffer.Size;
1031  }
