70  public static List<Vertices> ConvexPartition(Vertices vertices)
71  {
72  Debug.Assert(vertices.Count > 3);
73  Debug.Assert(vertices.IsCounterClockWise());
74 
75  int[] polygon = new int[vertices.Count];
76 
77  for (int v = 0; v < vertices.Count; v++)
78  {
79  polygon[v] = v;
80  }
81 
82  int nv = vertices.Count;
83 
84  // Remove nv-2 Vertices, creating 1 triangle every time
85  int count = 2 * nv; /* error detection */
86 
87  List<Vertices> result = new List<Vertices>();
88 
89  for (int v = nv - 1; nv > 2;)
90  {
91  // If we loop, it is probably a non-simple polygon
92  if (0 >= count--)
93  {
94  // Triangulate: ERROR - probable bad polygon!
95  return new List<Vertices>();
96  }
97 
98  // Three consecutive vertices in current polygon, <u,v,w>
99  int u = v;
100  if (nv <= u)
101  {
102  u = 0; // Previous
103  }
104 
105  v = u + 1;
106  if (nv <= v)
107  {
108  v = 0; // New v
109  }
110 
111  int w = v + 1;
112  if (nv <= w)
113  {
114  w = 0; // GetNext
115  }
116 
117  _tmpA = vertices[polygon[u]];
118  _tmpB = vertices[polygon[v]];
119  _tmpC = vertices[polygon[w]];
120 
121  if (Snip(vertices, u, v, w, nv, polygon))
122  {
123  int s, t;
124 
125  // Output Triangle
126  Vertices triangle = new Vertices(3)
127  {
128  _tmpA,
129  _tmpB,
130  _tmpC
131  };
132  result.Add(triangle);
133 
134  // Remove v from remaining polygon
135  for (s = v, t = v + 1; t < nv; s++, t++)
136  {
137  polygon[s] = polygon[t];
138  }
139 
140  nv--;
141 
142  // Reset error detection counter
143  count = 2 * nv;
144  }
145  }
146 
147  return result;
148  }

156  private static bool InsideTriangle(ref Vector2 a, ref Vector2 b, ref Vector2 c, ref Vector2 p)
157  {
158  //A cross bp
159  float abp = (c.X - b.X) * (p.Y - b.Y) - (c.Y - b.Y) * (p.X - b.X);
160 
161  //A cross ap
162  float aap = (b.X - a.X) * (p.Y - a.Y) - (b.Y - a.Y) * (p.X - a.X);
163 
164  //b cross cp
165  float bcp = (a.X - c.X) * (p.Y - c.Y) - (a.Y - c.Y) * (p.X - c.X);
166 
167  return abp >= 0.0f && bcp >= 0.0f && aap >= 0.0f;
168  }

178  private static bool Snip(Vertices contour, int u, int v, int w, int n, int[] vv)
179  {
180  if (MathConstants.Epsilon > MathUtils.Area(ref _tmpA, ref _tmpB, ref _tmpC))
181  {
182  return false;
183  }
184 
185  for (int p = 0; p < n; p++)
186  {
187  if (p == u || p == v || p == w)
188  {
189  continue;
190  }
191 
192  Vector2 point = contour[vv[p]];
193 
194  if (InsideTriangle(ref _tmpA, ref _tmpB, ref _tmpC, ref point))
195  {
196  return false;
197  }
198  }
199 
200  return true;
201  }
