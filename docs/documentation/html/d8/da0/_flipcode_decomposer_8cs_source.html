71  public static List<Vertices> ConvexPartition(Vertices vertices)
72  {
73  Debug.Assert(vertices.Count > 3);
74  Debug.Assert(vertices.IsCounterClockWise());
75 
76  int[] polygon = new int[vertices.Count];
77 
78  for (int v = 0; v < vertices.Count; v++)
79  {
80  polygon[v] = v;
81  }
82 
83  int nv = vertices.Count;
84 
85  // Remove nv-2 Vertices, creating 1 triangle every time
86  int count = 2 * nv; /* error detection */
87 
88  List<Vertices> result = new List<Vertices>();
89 
90  for (int v = nv - 1; nv > 2;)
91  {
92  // If we loop, it is probably a non-simple polygon
93  if (0 >= count--)
94  {
95  // Triangulate: ERROR - probable bad polygon!
96  return new List<Vertices>();
97  }
98 
99  // Three consecutive vertices in current polygon, <u,v,w>
100  int u = v;
101  if (nv <= u)
102  {
103  u = 0; // Previous
104  }
105 
106  v = u + 1;
107  if (nv <= v)
108  {
109  v = 0; // New v
110  }
111 
112  int w = v + 1;
113  if (nv <= w)
114  {
115  w = 0; // GetNext
116  }
117 
118  _tmpA = vertices[polygon[u]];
119  _tmpB = vertices[polygon[v]];
120  _tmpC = vertices[polygon[w]];
121 
122  if (Snip(vertices, u, v, w, nv, polygon))
123  {
124  int s, t;
125 
126  // Output Triangle
127  Vertices triangle = new Vertices(3)
128  {
129  _tmpA,
130  _tmpB,
131  _tmpC
132  };
133  result.Add(triangle);
134 
135  // Remove v from remaining polygon
136  for (s = v, t = v + 1; t < nv; s++, t++)
137  {
138  polygon[s] = polygon[t];
139  }
140 
141  nv--;
142 
143  // Reset error detection counter
144  count = 2 * nv;
145  }
146  }
147 
148  return result;
149  }

157  private static bool InsideTriangle(ref Vector2 a, ref Vector2 b, ref Vector2 c, ref Vector2 p)
158  {
159  //A cross bp
160  float abp = (c.X - b.X) * (p.Y - b.Y) - (c.Y - b.Y) * (p.X - b.X);
161 
162  //A cross ap
163  float aap = (b.X - a.X) * (p.Y - a.Y) - (b.Y - a.Y) * (p.X - a.X);
164 
165  //b cross cp
166  float bcp = (a.X - c.X) * (p.Y - c.Y) - (a.Y - c.Y) * (p.X - c.X);
167 
168  return (abp >= 0.0f) && (bcp >= 0.0f) && (aap >= 0.0f);
169  }

179  private static bool Snip(Vertices contour, int u, int v, int w, int n, int[] vv)
180  {
181  if (Constant.Epsilon > MathUtils.Area(ref _tmpA, ref _tmpB, ref _tmpC))
182  {
183  return false;
184  }
185 
186  for (int p = 0; p < n; p++)
187  {
188  if (p == u || p == v || p == w)
189  {
190  continue;
191  }
192 
193  Vector2 point = contour[vv[p]];
194 
195  if (InsideTriangle(ref _tmpA, ref _tmpB, ref _tmpC, ref point))
196  {
197  return false;
198  }
199  }
200 
201  return true;
202  }
203  }
