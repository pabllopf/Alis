223  public static Rot Mul(Rot q, Rot r)
224  {
225  // [qc -qs] * [rc -rs] = [qc*rc-qs*rs -qc*rs-qs*rc]
226  // [qs qc] [rs rc] [qs*rc+qc*rs -qs*rs+qc*rc]
227  // s = qs * rc + qc * rs
228  // c = qc * rc - qs * rs
229  Rot qr;
230  qr.S = q.S * r.C + q.C * r.S;
231  qr.C = q.C * r.C - q.S * r.S;
232  return qr;
233  }

252  public static Rot MulT(Rot q, Rot r)
253  {
254  // [ qc qs] * [rc -rs] = [qc*rc+qs*rs -qc*rs+qs*rc]
255  // [-qs qc] [rs rc] [-qs*rc+qc*rs qs*rs+qc*rc]

359  {

361  double theta2 = System.Math.Atan2(p2.Y, p2.X);
362  double dtheta = theta2 - theta1;
363 
364  while (dtheta > MathConstants.Pi)
365  {
366  dtheta -= MathConstants.TwoPi;
367  }

369  while (dtheta < -MathConstants.Pi)
370  {
371  dtheta += MathConstants.TwoPi;
372  }
373 
374  return dtheta;
