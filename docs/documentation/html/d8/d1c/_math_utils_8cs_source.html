228  public static Rotation Mul(Rotation q, Rotation r)
229  {
230  // [qc -qs] * [rc -rs] = [qc*rc-qs*rs -qc*rs-qs*rc]
231  // [qs qc] [rs rc] [qs*rc+qc*rs -qs*rs+qc*rc]
232  // s = qs * rc + qc * rs
233  // c = qc * rc - qs * rs
234  Rotation qr;
235  qr.Sine = q.Sine * r.Cosine + q.Cosine * r.Sine;
236  qr.Cosine = q.Cosine * r.Cosine - q.Sine * r.Sine;
