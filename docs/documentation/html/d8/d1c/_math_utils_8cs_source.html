227  public static Rotation Mul(Rotation q, Rotation r)
228  {
229  // [qc -qs] * [rc -rs] = [qc*rc-qs*rs -qc*rs-qs*rc]
230  // [qs qc] [rs rc] [qs*rc+qc*rs -qs*rs+qc*rc]
231  // s = qs * rc + qc * rs
232  // c = qc * rc - qs * rs
233  Rotation qr;
234  qr.Sine = q.Sine * r.Cosine + q.Cosine * r.Sine;
235  qr.Cosine = q.Cosine * r.Cosine - q.Sine * r.Sine;
