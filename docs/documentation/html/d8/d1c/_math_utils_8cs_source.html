225  public static Rot Mul(Rot q, Rot r)
226  {
227  // [qc -qs] * [rc -rs] = [qc*rc-qs*rs -qc*rs-qs*rc]
228  // [qs qc] [rs rc] [qs*rc+qc*rs -qs*rs+qc*rc]
229  // s = qs * rc + qc * rs
230  // c = qc * rc - qs * rs
231  Rot qr;
232  qr.S = q.S * r.C + q.C * r.S;
233  qr.C = q.C * r.C - q.S * r.S;
234  return qr;
235  }
