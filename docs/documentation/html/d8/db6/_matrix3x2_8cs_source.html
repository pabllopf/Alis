434  public static bool Invert(Matrix3x2 matrix, out Matrix3x2 result)
435  {
436  float det = matrix.M11 * matrix.M22 - matrix.M21 * matrix.M12;
437 
438  result = Identity;
439 
440  if (MathF.Abs(det) < float.Epsilon)
441  {
442  result = new Matrix3x2(float.NaN, float.NaN, float.NaN, float.NaN, float.NaN, float.NaN);
443  return false;
444  }
445 
446  float invDet = 1.0f / det;
447 
448  result.M11 = matrix.M22 * invDet;
449  result.M12 = -matrix.M12 * invDet;
450 
451  result.M21 = -matrix.M21 * invDet;
452  result.M22 = matrix.M11 * invDet;
453 
454  result.M31 = (matrix.M21 * matrix.M32 - matrix.M31 * matrix.M22) * invDet;
455  result.M32 = (matrix.M31 * matrix.M12 - matrix.M11 * matrix.M32) * invDet;
456 
457  return true;
458  }

520  public float GetDeterminant() =>
521  // There isn't actually any such thing as a determinant for a non-square matrix,
522  // but this 3x2 type is really just an optimization of a 3x3 where we happen to
523  // know the rightmost column is always (0, 0, 1). So we expand to 3x3 format:
524  //
525  // [ M11, M12, 0 ]
526  // [ M21, M22, 0 ]
527  // [ M31, M32, 1 ]
528  //
529  // Sum the diagonal products:
530  // (M11 * M22 * 1) + (M12 * 0 * M31) + (0 * M21 * M32)
531  //
532  // Subtract the opposite diagonal products:
533  // (M31 * M22 * 0) + (M32 * 0 * M11) + (1 * M21 * M12)
534  //
535  // Collapse out the constants and oh look, this is just a 2x2 determinant!
536  M11 * M22 - M21 * M12;
