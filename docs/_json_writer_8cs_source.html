79  internal static State[][] BuildStateArray()
80  {
81  List<State[]> allStates = StateArrayTempate.ToList();
82  State[] errorStates = StateArrayTempate[0];
83  State[] valueStates = StateArrayTempate[7];
84 
85  EnumInfo enumValuesAndNames = EnumUtils.GetEnumValuesAndNames(typeof(JsonToken));
86 
87  foreach (ulong valueToken in enumValuesAndNames.Values)
88  {
89  if (allStates.Count <= (int)valueToken)
90  {
91  JsonToken token = (JsonToken)valueToken;
92  switch (token)
93  {
94  case JsonToken.Integer:
95  case JsonToken.Float:
96  case JsonToken.String:
97  case JsonToken.Boolean:
98  case JsonToken.Null:
99  case JsonToken.Undefined:
100  case JsonToken.Date:
101  case JsonToken.Bytes:
102  allStates.Add(valueStates);
103  break;
104  default:
105  allStates.Add(errorStates);
106  break;
107  }
108  }
109  }
110 
111  return allStates.ToArray();
112  }

525  public void WriteToken(JsonToken token, object? value)
526  {
527  switch (token)
528  {
529  case JsonToken.None:
530  // read to next
531  break;
532  case JsonToken.StartObject:
533  WriteStartObject();
534  break;
535  case JsonToken.StartArray:
536  WriteStartArray();
537  break;
538  case JsonToken.StartConstructor:
539  ValidationUtils.ArgumentNotNull(value, nameof(value));
540  WriteStartConstructor(value.ToString());
541  break;
542  case JsonToken.PropertyName:
543  ValidationUtils.ArgumentNotNull(value, nameof(value));
544  WritePropertyName(value.ToString());
545  break;
546  case JsonToken.Comment:
547  WriteComment(value?.ToString());
548  break;
549  case JsonToken.Integer:
550  ValidationUtils.ArgumentNotNull(value, nameof(value));
551 #if HAVE_BIG_INTEGER
552  if (value is BigInteger integer)
553  {
554  WriteValue(integer);
555  }
556  else
557 #endif
558  {
559  WriteValue(Convert.ToInt64(value, CultureInfo.InvariantCulture));
560  }
561  break;
562  case JsonToken.Float:
563  ValidationUtils.ArgumentNotNull(value, nameof(value));
564  if (value is decimal decimalValue)
565  {
566  WriteValue(decimalValue);
567  }
568  else if (value is double doubleValue)
569  {
570  WriteValue(doubleValue);
571  }
572  else if (value is float floatValue)
573  {
574  WriteValue(floatValue);
575  }
576  else
577  {
578  WriteValue(Convert.ToDouble(value, CultureInfo.InvariantCulture));
579  }
580  break;
581  case JsonToken.String:
582  ValidationUtils.ArgumentNotNull(value, nameof(value));
583  WriteValue(value.ToString());
584  break;
585  case JsonToken.Boolean:
586  ValidationUtils.ArgumentNotNull(value, nameof(value));
587  WriteValue(Convert.ToBoolean(value, CultureInfo.InvariantCulture));
588  break;
589  case JsonToken.Null:
590  WriteNull();
591  break;
592  case JsonToken.Undefined:
593  WriteUndefined();
594  break;
595  case JsonToken.EndObject:
596  WriteEndObject();
597  break;
598  case JsonToken.EndArray:
599  WriteEndArray();
600  break;
601  case JsonToken.EndConstructor:
602  WriteEndConstructor();
603  break;
604  case JsonToken.Date:
605  ValidationUtils.ArgumentNotNull(value, nameof(value));
606 #if HAVE_DATE_TIME_OFFSET
607  if (value is DateTimeOffset dt)
608  {
609  WriteValue(dt);
610  }
611  else
612 #endif
613  {
614  WriteValue(Convert.ToDateTime(value, CultureInfo.InvariantCulture));
615  }
616  break;
617  case JsonToken.Raw:
618  WriteRawValue(value?.ToString());
619  break;
620  case JsonToken.Bytes:
621  ValidationUtils.ArgumentNotNull(value, nameof(value));
622  if (value is Guid guid)
623  {
624  WriteValue(guid);
625  }
626  else
627  {
628  WriteValue((byte[])value!);
629  }
630  break;
631  default:
632  throw MiscellaneousUtils.CreateArgumentOutOfRangeException(nameof(token), token, "Unexpected token type.");
633  }
634  }

645  internal virtual void WriteToken(JsonReader reader, bool writeChildren, bool writeDateConstructorAsDate, bool writeComments)
646  {
647  int initialDepth = CalculateWriteTokenInitialDepth(reader);
648 
649  do
650  {
651  // write a JValue date when the constructor is for a date
652  if (writeDateConstructorAsDate && reader.TokenType == JsonToken.StartConstructor && string.Equals(reader.Value?.ToString(), "Date", StringComparison.Ordinal))
653  {
654  WriteConstructorDate(reader);
655  }
656  else
657  {
658  if (writeComments || reader.TokenType != JsonToken.Comment)
659  {
660  WriteToken(reader.TokenType, reader.Value);
661  }
662  }
663  } while (
664  // stop if we have reached the end of the token being read
665  initialDepth - 1 < reader.Depth - (JsonTokenUtils.IsEndToken(reader.TokenType) ? 1 : 0)
666  && writeChildren
667  && reader.Read());
668 
669  if (IsWriteTokenIncomplete(reader, writeChildren, initialDepth))
670  {
671  throw JsonWriterException.Create(this, "Unexpected end when reading token.", null);
672  }
673  }
674 
675  private bool IsWriteTokenIncomplete(JsonReader reader, bool writeChildren, int initialDepth)
676  {
677  int finalDepth = CalculateWriteTokenFinalDepth(reader);
678  return initialDepth < finalDepth ||
679  (writeChildren && initialDepth == finalDepth && JsonTokenUtils.IsStartToken(reader.TokenType));
680  }

683  {
684  JsonToken type = reader.TokenType;

755  private void AutoCompleteClose(JsonContainerType type)
756  {
757  int levelsToComplete = CalculateLevelsToComplete(type);
758 
759  for (int i = 0; i < levelsToComplete; i++)
760  {
761  JsonToken token = GetCloseTokenForType(Pop());
762 
763  if (_currentState == State.Property)
764  {
765  WriteNull();
766  }
767 
768  if (_formatting == Formatting.Indented)
769  {
770  if (_currentState != State.ObjectStart && _currentState != State.ArrayStart)
771  {
772  WriteIndent();
773  }
774  }
775 
776  WriteEnd(token);
777 
778  UpdateCurrentState();
779  }
780  }

782  private int CalculateLevelsToComplete(JsonContainerType type)
783  {
784  int levelsToComplete = 0;
785 
786  if (_currentPosition.Type == type)
787  {
788  levelsToComplete = 1;
789  }
790  else
791  {
792  int top = Top - 2;
793  for (int i = top; i >= 0; i--)
794  {
795  int currentLevel = top - i;
796 
797  if (_stack![currentLevel].Type == type)
798  {
799  levelsToComplete = i + 2;
800  break;
801  }
802  }
803  }
804 
805  if (levelsToComplete == 0)
806  {
807  throw JsonWriterException.Create(this, "No token to close.", null);
808  }
809 
810  return levelsToComplete;
811  }

813  private void UpdateCurrentState()
814  {
815  JsonContainerType currentLevelType = Peek();
816 
817  switch (currentLevelType)
818  {
819  case JsonContainerType.Object:

821  break;
822  case JsonContainerType.Array:
823  _currentState = State.Array;
824  break;
825  case JsonContainerType.Constructor:
826  _currentState = State.Array;
827  break;
828  case JsonContainerType.None:
829  _currentState = State.Start;
830  break;
831  default:
832  throw JsonWriterException.Create(this, "Unknown JsonType: " + currentLevelType, null);
833  }

865  internal void AutoComplete(JsonToken tokenBeingWritten)
866  {
867  // gets new state based on the current state and what is being written
868  State newState = StateArray[(int)tokenBeingWritten][(int)_currentState];
869 
870  if (newState == State.Error)
871  {
872  throw JsonWriterException.Create(this, "Token {0} in state {1} would result in an invalid JSON object.".FormatWith(CultureInfo.InvariantCulture, tokenBeingWritten.ToString(), _currentState.ToString()), null);
873  }
874 
875  if ((_currentState == State.Object || _currentState == State.Array || _currentState == State.Constructor) && tokenBeingWritten != JsonToken.Comment)
876  {
877  WriteValueDelimiter();
878  }
879 
880  if (_formatting == Formatting.Indented)
881  {
882  if (_currentState == State.Property)
883  {
884  WriteIndentSpace();
885  }
886 
887  // don't indent a property when it is the first token to be written (i.e. at the start)
888  if ((_currentState == State.Array || _currentState == State.ArrayStart || _currentState == State.Constructor || _currentState == State.ConstructorStart)
889  || (tokenBeingWritten == JsonToken.PropertyName && _currentState != State.Start))
890  {
891  WriteIndent();
892  }
893  }
894 
895  _currentState = newState;
896  }
