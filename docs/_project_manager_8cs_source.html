295  private void CreateProject(string name, string directory, string mode)
296  {
297  if (!Directory.Exists(directory + "/" + name))
298  {
299  string dir = directory + "/" + name;
300  string assetPath = directory + "/" + name + "/Assets";
301  string configPath = directory + "/" + name + "/Config";
302  string dataPath = directory + "/" + name + "/Data";
303  string libPath = directory + "/" + name + "/Lib";
304 
305  Project project = new Project(name, dir);
306  Logger.Log("Create project " + " name: " + name + " directory: " + dir);
307 
308 
309  projects.Add(project);
310  LocalData.Save<List<Project>>("Projects", projects);
311  Logger.Log("Saves the project in file");
312 
313 
314 
315  Directory.CreateDirectory(dir);
316  Directory.CreateDirectory(assetPath);
317  Directory.CreateDirectory(configPath);
318  Directory.CreateDirectory(dataPath);
319  Directory.CreateDirectory(libPath);
320 
321  Logger.Log("Create directorys of project.");
322 
323  string projectFile = File.ReadAllText(Application.ProjectFolder + "/Resources/DefaultPr.txt", Encoding.UTF8);
324  File.WriteAllText(dir + "/" + name + ".csproj", projectFile, Encoding.UTF8);
325  Logger.Log("Created .csproj");
326 
327  string solutionFile = File.ReadAllText(Application.ProjectFolder + "/Resources/DefaultSl.txt", Encoding.UTF8).Replace("Example", name);
328  File.WriteAllText(dir + "/" + name + ".sln", solutionFile, Encoding.UTF8);
329  Logger.Log("Created .sln");
330 
331  string program = File.ReadAllText(Application.ProjectFolder + "/Resources/Program.txt", Encoding.UTF8);
332  File.WriteAllText(dir + "/" + "Program" + ".cs", program, Encoding.UTF8);
333  Logger.Log("Created Program.cs");
334 
335  File.Copy(Application.ProjectFolder + "/Resources/Core-SFML.dll", libPath + "/" + "Core-SFML" + ".dll");
336  File.Copy(Application.ProjectFolder + "/Resources/Core.dll", libPath + "/" + "Core" + ".dll");
337  File.Copy(Application.ProjectFolder + "/Resources/Tools.dll", libPath + "/" + "Tools" + ".dll");
338  Logger.Log("Copied .DLLS");
339 
340  VideoGame game = VideoGame.Builder()
341  .Config(Config.Builder().Name(name).Build())
342  .SceneManager(SceneManager.Builder().Scene(new Scene("Default")).Build())
343  .Build();
344 
345  Logger.Log("Created default game");
346 
347  LocalData.Save<VideoGame>("Data", dataPath, game);
348 
349  Logger.Log("Saved default game");
350 
351  Project.Current = project;
352  Project.VideoGame = game;
353 
354  Close();
355  }
