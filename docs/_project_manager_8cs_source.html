385  private void GenerateFiles(Project project)
386  {
387  string dir = project.Directory + "/" + project.Name;
388  Directory.CreateDirectory(dir);
389  Directory.CreateDirectory(project.AssetPath);
390  Directory.CreateDirectory(project.ConfigPath1);
391  Directory.CreateDirectory(project.DataPath1);
392  Directory.CreateDirectory(project.LibPath);
393 
394  string projectFile = File.ReadAllText(Application.ProjectFolder + "/Resources/DefaultPr.txt", Encoding.UTF8);
395  File.WriteAllText(dir + "/" + project.Name + ".csproj", projectFile, Encoding.UTF8);
396  Logger.Log("Created .csproj");
397 
398  string solutionFile = File.ReadAllText(Application.ProjectFolder + "/Resources/DefaultSl.txt", Encoding.UTF8).Replace("Example", project.Name);
399  File.WriteAllText(dir + "/" + project.Name + ".sln", solutionFile, Encoding.UTF8);
400  Logger.Log("Created .sln");
401 
402  string program = File.ReadAllText(Application.ProjectFolder + "/Resources/Program.txt", Encoding.UTF8);
403  File.WriteAllText(dir + "/" + "Program" + ".cs", program, Encoding.UTF8);
404  Logger.Log("Created Program.cs");
405 
406  File.Copy(Environment.CurrentDirectory + "/Core-SFML.dll", project.LibPath + "/" + "Core-SFML" + ".dll");
407  File.Copy(Environment.CurrentDirectory + "/Core.dll", project.LibPath + "/" + "Core" + ".dll");
408  File.Copy(Environment.CurrentDirectory + "/Tools.dll", project.LibPath + "/" + "Tools" + ".dll");
409 
410  File.Copy(Environment.CurrentDirectory + "/Resources/Segoe.ttf", project.AssetPath + "/" + "Segoe.ttf");
411 
412  VideoGame game = VideoGame.Builder()
413  .Config(Config.Builder().Name("Alis Game").Build())
414  .SceneManager(SceneManager.Builder().Scene(
415  Scene.Builder().GameObject(new GameObject("Default")).Build()).Build())
416  .Build();
417 
418  LocalData.Save("Data", project.DataPath1, game);
419 
420  LocalData.Save("Project", dir, project);
421 
422  Logger.Log("Saved default game");
423 
424  Project.Set(project);
425  Project.VideoGame = game;
426 
427 
428 
429  Close();
430  }
