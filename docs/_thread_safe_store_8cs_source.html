49  public ThreadSafeStore(Func<TKey, TValue> creator)
50  {
51  ValidationUtils.ArgumentNotNull(creator, nameof(creator));
52 
53  _creator = creator;
54 #if HAVE_CONCURRENT_DICTIONARY
55  _concurrentStore = new ConcurrentDictionary<TKey, TValue>();
56 #else
57  _store = new Dictionary<TKey, TValue>();
58 #endif
59  }

61  public TValue Get(TKey key)
62  {
63 #if HAVE_CONCURRENT_DICTIONARY
64  return _concurrentStore.GetOrAdd(key, _creator);
65 #else
66  if (!_store.TryGetValue(key, out TValue value))
67  {
68  return AddValue(key);
69  }
70 
71  return value;
72 #endif
73  }
74 
75 #if !HAVE_CONCURRENT_DICTIONARY
76  private TValue AddValue(TKey key)
77  {
78  TValue value = _creator(key);
79 
80  lock (_lock)
81  {
82  if (_store == null)
83  {
84  _store = new Dictionary<TKey, TValue>();
85  _store[key] = value;
86  }
87  else
88  {
89  // double check locking
90  if (_store.TryGetValue(key, out TValue checkValue))
91  {
92  return checkValue;
93  }
94 
95  Dictionary<TKey, TValue> newStore = new Dictionary<TKey, TValue>(_store);
96  newStore[key] = value;
97 
98 #if HAVE_MEMORY_BARRIER
99  Thread.MemoryBarrier();
100 #endif
101  _store = newStore;
102  }
103 
104  return value;
105  }
106  }
107 #endif
108  }
