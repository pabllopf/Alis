91  private Expression BuildMethodCall(MethodBase method, Type type, ParameterExpression? targetParameterExpression, ParameterExpression argsParameterExpression)
92  {
93  ParameterInfo[] parametersInfo = method.GetParameters();
94 
95  Expression[] argsExpression;
96  IList<ByRefParameter> refParameterMap;
97  if (parametersInfo.Length == 0)
98  {
99  argsExpression = CollectionUtils.ArrayEmpty<Expression>();
100  refParameterMap = CollectionUtils.ArrayEmpty<ByRefParameter>();
101  }
102  else
103  {
104  argsExpression = new Expression[parametersInfo.Length];
105  refParameterMap = new List<ByRefParameter>();
106 
107  for (int i = 0; i < parametersInfo.Length; i++)
108  {
109  ParameterInfo parameter = parametersInfo[i];
110  Type parameterType = parameter.ParameterType;
111  bool isByRef = false;
112  if (parameterType.IsByRef)
113  {
114  parameterType = parameterType.GetElementType();
115  isByRef = true;
116  }
117 
118  Expression indexExpression = Expression.Constant(i);
119 
120  Expression paramAccessorExpression = Expression.ArrayIndex(argsParameterExpression, indexExpression);
121 
122  Expression argExpression = EnsureCastExpression(paramAccessorExpression, parameterType, !isByRef);
123 
124  if (isByRef)
125  {
126  ParameterExpression variable = Expression.Variable(parameterType);
127  refParameterMap.Add(new ByRefParameter(argExpression, variable, parameter.IsOut));
128 
129  argExpression = variable;
130  }
131 
132  argsExpression[i] = argExpression;
133  }
134  }
135 
136  Expression callExpression;
137  if (method.IsConstructor)
138  {
139  callExpression = Expression.New((ConstructorInfo)method, argsExpression);
140  }
141  else if (method.IsStatic)
142  {
143  callExpression = Expression.Call((MethodInfo)method, argsExpression);
144  }
145  else
146  {
147  Expression readParameter = EnsureCastExpression(targetParameterExpression!, method.DeclaringType);
148 
149  callExpression = Expression.Call(readParameter, (MethodInfo)method, argsExpression);
150  }
151 
152  if (method is MethodInfo m)
153  {
154  if (m.ReturnType != typeof(void))
155  {
156  callExpression = EnsureCastExpression(callExpression, type);
157  }
158  else
159  {
160  callExpression = Expression.Block(callExpression, Expression.Constant(null));
161  }
162  }
163  else
164  {
165  callExpression = EnsureCastExpression(callExpression, type);
166  }
167 
168  if (refParameterMap.Count > 0)
169  {
170  IList<ParameterExpression> variableExpressions = new List<ParameterExpression>();
171  IList<Expression> bodyExpressions = new List<Expression>();
172  foreach (ByRefParameter p in refParameterMap)
173  {
174  if (!p.IsOut)
175  {
176  bodyExpressions.Add(Expression.Assign(p.Variable, p.Value));
177  }
178 
179  variableExpressions.Add(p.Variable);
180  }
181 
182  bodyExpressions.Add(callExpression);
183 
184  callExpression = Expression.Block(variableExpressions, bodyExpressions);
185  }
186 
187  return callExpression;
188  }

190  public override Func<T> CreateDefaultConstructor<T>(Type type)
191  {
192  ValidationUtils.ArgumentNotNull(type, "type");
193 
194  // avoid error from expressions compiler because of abstract class
195  if (type.IsAbstract())
196  {
197  return () => (T)Activator.CreateInstance(type);
198  }
199 
200  try
201  {
202  Type resultType = typeof(T);
203 
204  Expression expression = Expression.New(type);
205 
206  expression = EnsureCastExpression(expression, resultType);
207 
208  LambdaExpression lambdaExpression = Expression.Lambda(typeof(Func<T>), expression);
209 
210  Func<T> compiled = (Func<T>)lambdaExpression.Compile();
211  return compiled;
212  }
213  catch
214  {
215  // an error can be thrown if constructor is not valid on Win8
216  // will have INVOCATION_FLAGS_NON_W8P_FX_API invocation flag
217  return () => (T)Activator.CreateInstance(type);
218  }
219  }

221  public override Func<T, object?> CreateGet<T>(PropertyInfo propertyInfo)
222  {
223  ValidationUtils.ArgumentNotNull(propertyInfo, nameof(propertyInfo));
224 
225  Type instanceType = typeof(T);
226  Type resultType = typeof(object);
227 
228  ParameterExpression parameterExpression = Expression.Parameter(instanceType, "instance");
229  Expression resultExpression;
230 
231  MethodInfo? getMethod = propertyInfo.GetGetMethod(true);
232  if (getMethod == null)
233  {
234  throw new ArgumentException("Property does not have a getter.");
235  }
236 
237  if (getMethod.IsStatic)
238  {
239  resultExpression = Expression.MakeMemberAccess(null, propertyInfo);
240  }
241  else
242  {
243  Expression readParameter = EnsureCastExpression(parameterExpression, propertyInfo.DeclaringType);
244 
245  resultExpression = Expression.MakeMemberAccess(readParameter, propertyInfo);
246  }
247 
248  resultExpression = EnsureCastExpression(resultExpression, resultType);
249 
250  LambdaExpression lambdaExpression = Expression.Lambda(typeof(Func<T, object>), resultExpression, parameterExpression);
251 
252  Func<T, object?> compiled = (Func<T, object?>)lambdaExpression.Compile();
253  return compiled;

280  public override Action<T, object?> CreateSet<T>(FieldInfo fieldInfo)
281  {
282  ValidationUtils.ArgumentNotNull(fieldInfo, nameof(fieldInfo));
283 
284  // use reflection for structs
285  // expression doesn't correctly set value
286  if (fieldInfo.DeclaringType.IsValueType() || fieldInfo.IsInitOnly)
287  {
288  return LateBoundReflectionDelegateFactory.Instance.CreateSet<T>(fieldInfo);
289  }
290 
291  ParameterExpression sourceParameterExpression = Expression.Parameter(typeof(T), "source");
292  ParameterExpression valueParameterExpression = Expression.Parameter(typeof(object), "value");
293 
294  Expression fieldExpression;
295  if (fieldInfo.IsStatic)
296  {
297  fieldExpression = Expression.Field(null, fieldInfo);
298  }
299  else
300  {
301  Expression sourceExpression = EnsureCastExpression(sourceParameterExpression, fieldInfo.DeclaringType);
302 
303  fieldExpression = Expression.Field(sourceExpression, fieldInfo);
304  }
305 
306  Expression valueExpression = EnsureCastExpression(valueParameterExpression, fieldExpression.Type);
307 
308  BinaryExpression assignExpression = Expression.Assign(fieldExpression, valueExpression);
309 
310  LambdaExpression lambdaExpression = Expression.Lambda(typeof(Action<T, object>), assignExpression, sourceParameterExpression, valueParameterExpression);
311 
312  Action<T, object?> compiled = (Action<T, object?>)lambdaExpression.Compile();
313  return compiled;

316  public override Action<T, object?> CreateSet<T>(PropertyInfo propertyInfo)
317  {
318  ValidationUtils.ArgumentNotNull(propertyInfo, nameof(propertyInfo));
319 
320  // use reflection for structs
321  // expression doesn't correctly set value
322  if (propertyInfo.DeclaringType.IsValueType())
323  {
324  return LateBoundReflectionDelegateFactory.Instance.CreateSet<T>(propertyInfo);
325  }
326 
327  Type instanceType = typeof(T);
328  Type valueType = typeof(object);
329 
330  ParameterExpression instanceParameter = Expression.Parameter(instanceType, "instance");
331 
332  ParameterExpression valueParameter = Expression.Parameter(valueType, "value");
333  Expression readValueParameter = EnsureCastExpression(valueParameter, propertyInfo.PropertyType);
334 
335  MethodInfo? setMethod = propertyInfo.GetSetMethod(true);
336  if (setMethod == null)
337  {
338  throw new ArgumentException("Property does not have a setter.");
339  }
340 
341  Expression setExpression;
342  if (setMethod.IsStatic)
343  {
344  setExpression = Expression.Call(setMethod, readValueParameter);
345  }
346  else
347  {
348  Expression readInstanceParameter = EnsureCastExpression(instanceParameter, propertyInfo.DeclaringType);
349 
350  setExpression = Expression.Call(readInstanceParameter, setMethod, readValueParameter);
351  }
352 
353  LambdaExpression lambdaExpression = Expression.Lambda(typeof(Action<T, object?>), setExpression, instanceParameter, valueParameter);
354 
355  Action<T, object?> compiled = (Action<T, object?>)lambdaExpression.Compile();
356  return compiled;

359  private Expression EnsureCastExpression(Expression expression, Type targetType, bool allowWidening = false)
360  {
361  Type expressionType = expression.Type;
362 
363  // check if a cast or conversion is required
364  if (expressionType == targetType || (!expressionType.IsValueType() && targetType.IsAssignableFrom(expressionType)))
365  {
366  return expression;
367  }
368 
369  if (targetType.IsValueType())
370  {
371  Expression convert = Expression.Unbox(expression, targetType);
372 
373  if (allowWidening && targetType.IsPrimitive())
374  {
375  MethodInfo toTargetTypeMethod = typeof(Convert)
376  .GetMethod("To" + targetType.Name, new[] { typeof(object) });
377 
378  if (toTargetTypeMethod != null)
379  {
380  convert = Expression.Condition(
381  Expression.TypeIs(expression, targetType),
382  convert,
383  Expression.Call(toTargetTypeMethod, expression));
384  }
385  }
386 
387  return Expression.Condition(
388  Expression.Equal(expression, Expression.Constant(null, typeof(object))),
389  Expression.Default(targetType),
390  convert);
391  }
392 
393  return Expression.Convert(expression, targetType);
394  }
395  }
