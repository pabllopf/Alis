49  public static bool ValueEquals(object? objA, object? objB)
50  {
51  if (objA == objB)
52  {
53  return true;
54  }
55  if (objA == null || objB == null)
56  {
57  return false;
58  }
59 
60  // comparing an Int32 and Int64 both of the same value returns false
61  // make types the same then compare
62  if (objA.GetType() != objB.GetType())
63  {
64  if (ConvertUtils.IsInteger(objA) && ConvertUtils.IsInteger(objB))
65  {
66  return Convert.ToDecimal(objA, CultureInfo.CurrentCulture).Equals(Convert.ToDecimal(objB, CultureInfo.CurrentCulture));
67  }
68  else if ((objA is double || objA is float || objA is decimal) && (objB is double || objB is float || objB is decimal))
69  {
70  return MathUtils.ApproxEquals(Convert.ToDouble(objA, CultureInfo.CurrentCulture), Convert.ToDouble(objB, CultureInfo.CurrentCulture));
71  }
72  else
73  {
74  return false;
75  }
76  }
77 
78  return objA.Equals(objB);
79  }

132  public static void GetQualifiedNameParts(string qualifiedName, out string? prefix, out string localName)
133  {
134  int colonPosition = qualifiedName.IndexOf(':');
135 
136  if ((colonPosition == -1 || colonPosition == 0) || (qualifiedName.Length - 1) == colonPosition)
137  {
138  prefix = null;
139  localName = qualifiedName;
140  }
141  else
142  {
143  prefix = qualifiedName.Substring(0, colonPosition);
144  localName = qualifiedName.Substring(colonPosition + 1);
145  }
146  }
