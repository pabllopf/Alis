39  internal async Task ReadTokenFromAsync(JsonReader reader, JsonLoadSettings? options, CancellationToken cancellationToken = default)
40  {
41  ValidationUtils.ArgumentNotNull(reader, nameof(reader));
42  int startDepth = reader.Depth;
43 
44  if (!await reader.ReadAsync(cancellationToken).ConfigureAwait(false))
45  {
46  throw JsonReaderException.Create(reader, "Error reading {0} from JsonReader.".FormatWith(CultureInfo.InvariantCulture, GetType().Name));
47  }
48 
49  await ReadContentFromAsync(reader, options, cancellationToken).ConfigureAwait(false);
50 
51  if (reader.Depth > startDepth)
52  {
53  throw JsonReaderException.Create(reader, "Unexpected end of content while loading {0}.".FormatWith(CultureInfo.InvariantCulture, GetType().Name));
54  }
55  }

57  private async Task ReadContentFromAsync(JsonReader reader, JsonLoadSettings? settings, CancellationToken cancellationToken = default)
58  {
59  IJsonLineInfo? lineInfo = reader as IJsonLineInfo;
60 
61  JContainer? parent = this;
62 
63  do
64  {
65  if (parent is JProperty p && p.Value != null)
66  {
67  if (parent == this)
68  {
69  return;
70  }
71 
72  parent = parent.Parent;
73  }
74 
75  MiscellaneousUtils.Assert(parent != null);
76 
77  switch (reader.TokenType)
78  {
79  case JsonToken.None:
80  // new reader. move to actual content
81  break;
82  case JsonToken.StartArray:
83  JArray a = new JArray();
84  a.SetLineInfo(lineInfo, settings);
85  parent.Add(a);
86  parent = a;
87  break;
88 
89  case JsonToken.EndArray:
90  if (parent == this)
91  {
92  return;
93  }
94 
95  parent = parent.Parent;
96  break;
97  case JsonToken.StartObject:
98  JObject o = new JObject();
99  o.SetLineInfo(lineInfo, settings);
100  parent.Add(o);
101  parent = o;
102  break;
103  case JsonToken.EndObject:
104  if (parent == this)
105  {
106  return;
107  }
108 
109  parent = parent.Parent;
110  break;
111  case JsonToken.StartConstructor:
112  JConstructor constructor = new JConstructor(reader.Value!.ToString());
113  constructor.SetLineInfo(lineInfo, settings);
114  parent.Add(constructor);
115  parent = constructor;
116  break;
117  case JsonToken.EndConstructor:
118  if (parent == this)
119  {
120  return;
121  }
122 
123  parent = parent.Parent;
124  break;
125  case JsonToken.String:
126  case JsonToken.Integer:
127  case JsonToken.Float:
128  case JsonToken.Date:
129  case JsonToken.Boolean:
130  case JsonToken.Bytes:
131  JValue v = new JValue(reader.Value);
132  v.SetLineInfo(lineInfo, settings);
133  parent.Add(v);
134  break;
135  case JsonToken.Comment:
136  if (settings != null && settings.CommentHandling == CommentHandling.Load)
137  {
138  v = JValue.CreateComment(reader.Value!.ToString());
139  v.SetLineInfo(lineInfo, settings);
140  parent.Add(v);
141  }
142  break;
143  case JsonToken.Null:
144  v = JValue.CreateNull();
145  v.SetLineInfo(lineInfo, settings);
146  parent.Add(v);
147  break;
148  case JsonToken.Undefined:
149  v = JValue.CreateUndefined();
150  v.SetLineInfo(lineInfo, settings);
151  parent.Add(v);
152  break;
153  case JsonToken.PropertyName:
154  JProperty? property = ReadProperty(reader, settings, lineInfo, parent);
155  if (property != null)
156  {
157  parent = property;
158  }
159  else
160  {
161  await reader.SkipAsync().ConfigureAwait(false);
162  }
163  break;
164  default:
165  throw new InvalidOperationException("The JsonReader should not be on a token of type {0}.".FormatWith(CultureInfo.InvariantCulture, reader.TokenType));
166  }
167  } while (await reader.ReadAsync(cancellationToken).ConfigureAwait(false));
168  }
169  }
170 }
