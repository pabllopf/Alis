41  public TraceJsonWriter(JsonWriter innerWriter)
42  {
43  _innerWriter = innerWriter;
44 
45  _sw = new StringWriter(CultureInfo.InvariantCulture);
46  // prefix the message in the stringwriter to avoid concat with a potentially large JSON string
47  _sw.Write("Serialized JSON: " + Environment.NewLine);
48 
49  _textWriter = new JsonTextWriter(_sw);
50  _textWriter.Formatting = Formatting.Indented;
51  _textWriter.Culture = innerWriter.Culture;
52  _textWriter.DateFormatHandling = innerWriter.DateFormatHandling;
53  _textWriter.DateFormatString = innerWriter.DateFormatString;
54  _textWriter.DateTimeZoneHandling = innerWriter.DateTimeZoneHandling;
55  _textWriter.FloatFormatHandling = innerWriter.FloatFormatHandling;
56  }
