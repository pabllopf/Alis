500  private void RunCommand(string message, string fileName, string commandBuild, string WorkingDirectory, bool messages)
501  {
502  if (messages)
503  {
504  BottomMenu.Loading(true, message);
505  }
506 
507  Process buildProcess = new Process();
508 
509  buildProcess.StartInfo.FileName = fileName;
510  buildProcess.StartInfo.WorkingDirectory = WorkingDirectory;
511  buildProcess.StartInfo.CreateNoWindow = false;
512  buildProcess.StartInfo.RedirectStandardInput = true;
513  buildProcess.StartInfo.RedirectStandardOutput = true;
514  buildProcess.StartInfo.UseShellExecute = false;
515  buildProcess.Start();
516  buildProcess.StandardInput.WriteLine(commandBuild);
517  buildProcess.StandardInput.Flush();
518  buildProcess.StandardInput.Close();
519  buildProcess.WaitForExit();
520 
521  string[] sentences = buildProcess.StandardOutput.ReadToEnd().Split("\n");
522  for (int i = 0; i < sentences.Length; i++)
523  {
524  if (sentences[i].Contains("warning"))
525  {
526  Console.Warning(sentences[i]);
527  continue;
528  }
529 
530  if (sentences[i].Contains("error") || sentences[i].Contains("failed") || sentences[i].Contains("exception"))
531  {
532  Console.Error(sentences[i]);
533  continue;
534  }
535 
536  Console.Log(sentences[i], false);
537  }
538 
539  if (messages)
540  {
541  BottomMenu.Loading(false, "");
542  }
543  }
