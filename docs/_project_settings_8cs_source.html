110  private void ShowConfig(Config config)
111  {
112  if (ImGui.TreeNodeEx(Icon.WRENCH + " General", ImGuiTreeNodeFlags.AllowItemOverlap | ImGuiTreeNodeFlags.DefaultOpen))
113  {
114  foreach (PropertyInfo property in config.GetType().GetProperties())
115  {
116  ImGui.PushStyleVar(ImGuiStyleVar.FrameBorderSize, 0f);
117  ImGui.PushStyleColor(ImGuiCol.Button, childBackground);
118 
119  foreach (KeyValuePair<Type, Action<Config, PropertyInfo>> field in fields)
120  {
121  if (field.Key.Equals(property.PropertyType) && property.CanWrite)
122  {
123  field.Value.Invoke(config, property);
124  }
125  }
126 
127  ImGui.PopStyleVar();
128  ImGui.PopStyleColor();
129  }
