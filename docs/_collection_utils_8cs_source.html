126  public static ConstructorInfo? ResolveEnumerableCollectionConstructor(Type collectionType, Type collectionItemType, Type constructorArgumentType)
127  {
128  Type genericEnumerable = typeof(IEnumerable<>).MakeGenericType(collectionItemType);
129  ConstructorInfo? match = null;
130 
131  foreach (ConstructorInfo constructor in collectionType.GetConstructors(BindingFlags.Public | BindingFlags.Instance))
132  {
133  IList<ParameterInfo> parameters = constructor.GetParameters();
134 
135  if (parameters.Count == 1)
136  {
137  Type parameterType = parameters[0].ParameterType;
138 
139  if (genericEnumerable == parameterType)
140  {
141  // exact match
142  match = constructor;
143  break;
144  }
145 
146  // in case we can't find an exact match, use first inexact
147  if (match == null)
148  {
149  if (parameterType.IsAssignableFrom(constructorArgumentType))
150  {
151  match = constructor;
152  }
153  }
154  }
155  }
156 
157  return match;
158  }

272  private static IList<int> GetDimensions(IList values, int dimensionsCount)
273  {
274  IList<int> dimensions = new List<int>();
275 
276  IList currentArray = values;
277  while (true)
278  {
279  dimensions.Add(currentArray.Count);
280 
281  // don't keep calculating dimensions for arrays inside the value array
282  if (dimensions.Count == dimensionsCount)
283  {
284  break;
285  }
286 
287  if (currentArray.Count == 0)
288  {
289  break;
290  }
291 
292  object v = currentArray[0];
293  if (v is IList list)
294  {
295  currentArray = list;
296  }
297  else
298  {
299  break;
300  }
301  }
302 
303  return dimensions;
304  }

306  private static void CopyFromJaggedToMultidimensionalArray(IList values, Array multidimensionalArray, int[] indices)
307  {
308  int dimension = indices.Length;
309  if (dimension == multidimensionalArray.Rank)
310  {
311  multidimensionalArray.SetValue(JaggedArrayGetValue(values, indices), indices);
312  return;
313  }
314 
315  int dimensionLength = multidimensionalArray.GetLength(dimension);
316  IList list = (IList)JaggedArrayGetValue(values, indices);
317  int currentValuesLength = list.Count;
318  if (currentValuesLength != dimensionLength)
319  {
320  throw new Exception("Cannot deserialize non-cubical array as multidimensional array.");
321  }
322 
323  int[] newIndices = new int[dimension + 1];
324  for (int i = 0; i < dimension; i++)
325  {
326  newIndices[i] = indices[i];
327  }
328 
329  for (int i = 0; i < multidimensionalArray.GetLength(dimension); i++)
330  {
331  newIndices[dimension] = i;
332  CopyFromJaggedToMultidimensionalArray(values, multidimensionalArray, newIndices);
333  }
334  }

336  private static object JaggedArrayGetValue(IList values, int[] indices)
337  {
338  IList currentList = values;
339  for (int i = 0; i < indices.Length; i++)
340  {
341  int index = indices[i];
342  if (i == indices.Length - 1)
343  {
344  return currentList[index];
345  }
346  else
347  {
348  currentList = (IList)currentList[index];
349  }
350  }
351  return currentList;
352  }

354  public static Array ToMultidimensionalArray(IList values, Type type, int rank)
355  {
356  IList<int> dimensions = GetDimensions(values, rank);
357 
358  while (dimensions.Count < rank)
359  {
360  dimensions.Add(0);
361  }
362 
363  Array multidimensionalArray = Array.CreateInstance(type, dimensions.ToArray());
364  CopyFromJaggedToMultidimensionalArray(values, multidimensionalArray, ArrayEmpty<int>());
