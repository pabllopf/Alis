118  public static DataMemberAttribute? GetDataMemberAttribute(MemberInfo memberInfo)
119  {
120  // DataMemberAttribute does not have inheritance
121 
122  // can't override a field
123  if (memberInfo.MemberType() == MemberTypes.Field)
124  {
125  return CachedAttributeGetter<DataMemberAttribute>.GetAttribute(memberInfo);
126  }
127 
128  // search property and then search base properties if nothing is returned and the property is virtual
129  PropertyInfo propertyInfo = (PropertyInfo)memberInfo;
130  DataMemberAttribute? result = CachedAttributeGetter<DataMemberAttribute>.GetAttribute(propertyInfo);
131  if (result == null)
132  {
133  if (propertyInfo.IsVirtual())
134  {
135  Type currentType = propertyInfo.DeclaringType;
136 
137  while (result == null && currentType != null)
138  {
139  PropertyInfo baseProperty = (PropertyInfo)ReflectionUtils.GetMemberInfoFromType(currentType, propertyInfo);
140  if (baseProperty != null && baseProperty.IsVirtual())
141  {
142  result = CachedAttributeGetter<DataMemberAttribute>.GetAttribute(baseProperty);
143  }
144 
145  currentType = currentType.BaseType();
146  }
147  }
148  }
149 
150  return result;
151  }

230  private static Func<object[]?, object> GetCreator(Type type)
231  {
232  Func<object>? defaultConstructor = (ReflectionUtils.HasDefaultConstructor(type, false))
233  ? ReflectionDelegateFactory.CreateDefaultConstructor<object>(type)
234  : null;
235 
236  return (parameters) =>
237  {
238  try
239  {
240  if (parameters != null)
241  {
242  Type[] paramTypes = parameters.Select(param =>
243  {
244  if (param == null)
245  {
246  throw new InvalidOperationException("Cannot pass a null parameter to the constructor.");
247  }
248 
249  return param.GetType();
250  }).ToArray();
251  ConstructorInfo parameterizedConstructorInfo = type.GetConstructor(paramTypes);
252 
253  if (parameterizedConstructorInfo != null)
254  {
255  ObjectConstructor<object> parameterizedConstructor = ReflectionDelegateFactory.CreateParameterizedConstructor(parameterizedConstructorInfo);
256  return parameterizedConstructor(parameters);
257  }
258  else
259  {
260  throw new JsonException("No matching parameterized constructor found for '{0}'.".FormatWith(CultureInfo.InvariantCulture, type));
261  }
262  }
263 
264  if (defaultConstructor == null)
265  {
266  throw new JsonException("No parameterless constructor defined for '{0}'.".FormatWith(CultureInfo.InvariantCulture, type));
267  }
268 
269  return defaultConstructor();
270  }
271  catch (Exception ex)
272  {
273  throw new JsonException("Error creating '{0}'.".FormatWith(CultureInfo.InvariantCulture, type), ex);
274  }
275  };
276  }
