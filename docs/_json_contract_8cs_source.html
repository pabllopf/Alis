239  internal JsonContract(Type underlyingType)
240  {
241  ValidationUtils.ArgumentNotNull(underlyingType, nameof(underlyingType));
242 
243  UnderlyingType = underlyingType;
244 
245  // resolve ByRef types
246  // typically comes from in and ref parameters on methods/ctors
247  underlyingType = ReflectionUtils.EnsureNotByRefType(underlyingType);
248 
249  IsNullable = ReflectionUtils.IsNullable(underlyingType);
250 
251  NonNullableUnderlyingType = (IsNullable && ReflectionUtils.IsNullableType(underlyingType)) ? Nullable.GetUnderlyingType(underlyingType) : underlyingType;
252 
253  _createdType = CreatedType = NonNullableUnderlyingType;
254 
255  IsConvertable = ConvertUtils.IsConvertible(NonNullableUnderlyingType);
256  IsEnum = NonNullableUnderlyingType.IsEnum();
257 
258  InternalReadType = ReadType.Read;
259  }
