151  private void UpdateGameObject(GameObject obj)
152  {
153  if (Project.Get().DLL1 != null)
154  {
155  List<Component> tempList = obj.Components.ToList();
156  Type[] ty = Project.Get().DLL1.GetTypes();
157 
158  for (int i = 0; i < tempList.Count; i++)
159  {
160  if (tempList[i] != null)
161  {
162  Component compo = tempList[i];
163  Type type = typeof(Component);
164  IEnumerable<Type> types = ty
165  .Where(p => type.IsAssignableFrom(p));
166 
167  if (types.Any(i => i.Name.Equals(compo.GetType().Name)))
168  {
169  Type final = types.First(i => i.Name.Equals(compo.GetType().Name));
170 
171  if (final.FullName.Equals(obj.Components[i].GetType().FullName) && !final.Assembly.Equals(obj.Components[i].GetType().Assembly))
172  {
173  Console.Warning("Create new intencie of " + final.FullName + " on " + obj.Name);
174 
175  Component tempCompo = (Component)Activator.CreateInstance(final);
176 
177  foreach (PropertyInfo property in tempCompo.GetType().GetProperties())
178  {
179  PropertyInfo info = compo.GetType().GetProperty(property.Name);
180  if (info != null)
181  {
182  if (property.CanWrite && info.CanWrite && info.Name.Equals(property.Name))
183  {
184  property.SetValue(tempCompo, info.GetValue(compo));
185  }
186  }
187 
188 
189  }
190 
191  obj.Set(tempCompo, i);
192  }
193  }
194  }
195  }
196  }
