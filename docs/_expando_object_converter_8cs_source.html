67  private object? ReadValue(JsonReader reader)
68  {
69  if (!reader.MoveToContent())
70  {
71  throw JsonSerializationException.Create(reader, "Unexpected end when reading ExpandoObject.");
72  }
73 
74  switch (reader.TokenType)
75  {
76  case JsonToken.StartObject:
77  return ReadObject(reader);
78  case JsonToken.StartArray:
79  return ReadList(reader);
80  default:
81  if (JsonTokenUtils.IsPrimitiveToken(reader.TokenType))
82  {
83  return reader.Value;
84  }
85 
86  throw JsonSerializationException.Create(reader, "Unexpected token when converting ExpandoObject: {0}".FormatWith(CultureInfo.InvariantCulture, reader.TokenType));
87  }
88  }
89 
90  private object ReadList(JsonReader reader)
91  {
92  IList<object?> list = new List<object?>();
93 
94  while (reader.Read())
95  {
96  switch (reader.TokenType)
97  {
98  case JsonToken.Comment:
99  break;
100  default:
101  object? v = ReadValue(reader);
102 
103  list.Add(v);
104  break;
105  case JsonToken.EndArray:
106  return list;
107  }
108  }
109 
110  throw JsonSerializationException.Create(reader, "Unexpected end when reading ExpandoObject.");
111  }

113  private object ReadObject(JsonReader reader)
114  {
115  IDictionary<string, object?> expandoObject = new ExpandoObject();
116 
117  while (reader.Read())
118  {
119  switch (reader.TokenType)
120  {
121  case JsonToken.PropertyName:
122  string propertyName = reader.Value!.ToString();
123 
124  if (!reader.Read())
125  {
126  throw JsonSerializationException.Create(reader, "Unexpected end when reading ExpandoObject.");
127  }
128 
129  object? v = ReadValue(reader);
130 
131  expandoObject[propertyName] = v;
132  break;
133  case JsonToken.Comment:
134  break;
135  case JsonToken.EndObject:
136  return expandoObject;
137  }
138  }
139 
140  throw JsonSerializationException.Create(reader, "Unexpected end when reading ExpandoObject.");
141  }
