79  public override void WriteJson(JsonWriter writer, object? value, JsonSerializer serializer)
80  {
81  string text;
82 
83  if (value is DateTime dateTime)
84  {
85  if ((_dateTimeStyles & DateTimeStyles.AdjustToUniversal) == DateTimeStyles.AdjustToUniversal
86  || (_dateTimeStyles & DateTimeStyles.AssumeUniversal) == DateTimeStyles.AssumeUniversal)
87  {
88  dateTime = dateTime.ToUniversalTime();
89  }
90 
91  text = dateTime.ToString(_dateTimeFormat ?? DefaultDateTimeFormat, Culture);
92  }
93 #if HAVE_DATE_TIME_OFFSET
94  else if (value is DateTimeOffset dateTimeOffset)
95  {
96  if ((_dateTimeStyles & DateTimeStyles.AdjustToUniversal) == DateTimeStyles.AdjustToUniversal
97  || (_dateTimeStyles & DateTimeStyles.AssumeUniversal) == DateTimeStyles.AssumeUniversal)
98  {
99  dateTimeOffset = dateTimeOffset.ToUniversalTime();
100  }
101 
102  text = dateTimeOffset.ToString(_dateTimeFormat ?? DefaultDateTimeFormat, Culture);
103  }
104 #endif
105  else
106  {
107  throw new JsonSerializationException("Unexpected value when converting date. Expected DateTime or DateTimeOffset, got {0}.".FormatWith(CultureInfo.InvariantCulture, ReflectionUtils.GetObjectType(value)!));
108  }
109 
110  writer.WriteValue(text);
111  }

121  public override object? ReadJson(JsonReader reader, Type objectType, object? existingValue, JsonSerializer serializer)
122  {
123  bool nullable = ReflectionUtils.IsNullableType(objectType);
124  if (reader.TokenType == JsonToken.Null)
125  {
126  if (!nullable)
127  {
128  throw JsonSerializationException.Create(reader, "Cannot convert null value to {0}.".FormatWith(CultureInfo.InvariantCulture, objectType));
129  }
130 
131  return null;
132  }
133 
134 #if HAVE_DATE_TIME_OFFSET
135  Type t = (nullable)
136  ? Nullable.GetUnderlyingType(objectType)
137  : objectType;
138 #endif
139 
140  if (reader.TokenType == JsonToken.Date)
141  {
142 #if HAVE_DATE_TIME_OFFSET
143  if (t == typeof(DateTimeOffset))
144  {
145  return (reader.Value is DateTimeOffset) ? reader.Value : new DateTimeOffset((DateTime)reader.Value!);
146  }
147 
148  // converter is expected to return a DateTime
149  if (reader.Value is DateTimeOffset offset)
150  {
151  return offset.DateTime;
152  }
153 #endif
154 
155  return reader.Value;
156  }
157 
158  if (reader.TokenType != JsonToken.String)
159  {
160  throw JsonSerializationException.Create(reader, "Unexpected token parsing date. Expected String, got {0}.".FormatWith(CultureInfo.InvariantCulture, reader.TokenType));
161  }
162 
163  string? dateText = reader.Value?.ToString();
164 
165  if (StringUtils.IsNullOrEmpty(dateText) && nullable)
166  {
167  return null;
168  }
169 
170 #if HAVE_DATE_TIME_OFFSET
171  if (t == typeof(DateTimeOffset))
172  {
173  if (!StringUtils.IsNullOrEmpty(_dateTimeFormat))
174  {
175  return DateTimeOffset.ParseExact(dateText, _dateTimeFormat, Culture, _dateTimeStyles);
176  }
177  else
178  {
179  return DateTimeOffset.Parse(dateText, Culture, _dateTimeStyles);
180  }
181  }
182 #endif
183 
184  if (!StringUtils.IsNullOrEmpty(_dateTimeFormat))
185  {
186  return DateTime.ParseExact(dateText, _dateTimeFormat, Culture, _dateTimeStyles);
187  }
188  else
189  {
190  return DateTime.Parse(dateText, Culture, _dateTimeStyles);
191  }
192  }
193  }
