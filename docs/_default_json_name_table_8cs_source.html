62  public override string? Get(char[] key, int start, int length)
63  {
64  if (length == 0)
65  {
66  return string.Empty;
67  }
68 
69  int hashCode = length + HashCodeRandomizer;
70  hashCode += (hashCode << 7) ^ key[start];
71  int end = start + length;
72  for (int i = start + 1; i < end; i++)
73  {
74  hashCode += (hashCode << 7) ^ key[i];
75  }
76  hashCode -= hashCode >> 17;
77  hashCode -= hashCode >> 11;
78  hashCode -= hashCode >> 5;
79 
80  // make sure index is evaluated before accessing _entries, otherwise potential race condition causing IndexOutOfRangeException
81  var index = hashCode & _mask;
82  var entries = _entries;
83 
84  for (Entry entry = entries[index]; entry != null; entry = entry.Next)
85  {
86  if (entry.HashCode == hashCode && TextEquals(entry.Value, key, start, length))
87  {
88  return entry.Value;
89  }
90  }
91 
92  return null;
93  }

101  public string Add(string key)
102  {
103  if (key == null)
104  {
105  throw new ArgumentNullException(nameof(key));
106  }
107 
108  int length = key.Length;
109  if (length == 0)
110  {
111  return string.Empty;
112  }
113 
114  int hashCode = length + HashCodeRandomizer;
115  for (int i = 0; i < key.Length; i++)
116  {
117  hashCode += (hashCode << 7) ^ key[i];
118  }
119  hashCode -= hashCode >> 17;
120  hashCode -= hashCode >> 11;
121  hashCode -= hashCode >> 5;
122  for (Entry entry = _entries[hashCode & _mask]; entry != null; entry = entry.Next)
123  {
124  if (entry.HashCode == hashCode && entry.Value.Equals(key, StringComparison.Ordinal))
125  {

127  }
128  }

133  private string AddEntry(string str, int hashCode)
134  {
135  int index = hashCode & _mask;
136  Entry entry = new Entry(str, hashCode, _entries[index]);
137  _entries[index] = entry;

145  private void Grow()
146  {
147  Entry[] entries = _entries;
148  int newMask = (_mask * 2) + 1;
149  Entry[] newEntries = new Entry[newMask + 1];
150 
151  for (int i = 0; i < entries.Length; i++)
152  {
153  Entry next;
154  for (Entry entry = entries[i]; entry != null; entry = next)
155  {
156  int index = entry.HashCode & newMask;
157  next = entry.Next;
158  entry.Next = newEntries[index];
159  newEntries[index] = entry;
160  }
161  }
162  _entries = newEntries;
163  _mask = newMask;
164  }
