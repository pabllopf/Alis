61  public void Serialize(JsonWriter jsonWriter, object? value, Type? objectType)
62  {
63  if (jsonWriter == null)
64  {
65  throw new ArgumentNullException(nameof(jsonWriter));
66  }
67 
68  _rootType = objectType;
69  _rootLevel = _serializeStack.Count + 1;
70 
71  JsonContract? contract = GetContractSafe(value);
72 
73  try
74  {
75  if (ShouldWriteReference(value, null, contract, null, null))
76  {
77  WriteReference(jsonWriter, value!);
78  }
79  else
80  {
81  SerializeValue(jsonWriter, value, contract, null, null, null);
82  }
83  }
84  catch (Exception ex)
85  {
86  if (IsErrorHandled(null, contract, null, null, jsonWriter.Path, ex))
87  {
88  HandleError(jsonWriter, 0);
89  }
90  else
91  {
92  // clear context in case serializer is being used inside a converter
93  // if the converter wraps the error then not clearing the context will cause this error:
94  // "Current error context error is different to requested error."
95  ClearErrorContext();
96  throw;
97  }
98  }
99  finally
100  {
101  // clear root contract to ensure that if level was > 1 then it won't
102  // accidentally be used for non root values
103  _rootType = null;
104  }
105  }

154  private void SerializeValue(JsonWriter writer, object? value, JsonContract? valueContract, JsonProperty? member, JsonContainerContract? containerContract, JsonProperty? containerProperty)
155  {
156  if (value == null)
157  {
158  writer.WriteNull();
159  return;
160  }
161 
162  MiscellaneousUtils.Assert(valueContract != null);
163 
164  JsonConverter? converter =
165  member?.Converter ??
166  containerProperty?.ItemConverter ??
167  containerContract?.ItemConverter ??
168  valueContract.Converter ??
169  Serializer.GetMatchingConverter(valueContract.UnderlyingType) ??
170  valueContract.InternalConverter;
171 
172  if (converter != null && converter.CanWrite)
173  {
174  SerializeConvertable(writer, converter, value, valueContract, containerContract, containerProperty);
175  return;
176  }
177 
178  switch (valueContract.ContractType)
179  {
180  case JsonContractType.Object:
181  SerializeObject(writer, value, (JsonObjectContract)valueContract, member, containerContract, containerProperty);
182  break;
183  case JsonContractType.Array:
184  JsonArrayContract arrayContract = (JsonArrayContract)valueContract;
185  if (!arrayContract.IsMultidimensionalArray)
186  {
187  SerializeList(writer, (IEnumerable)value, arrayContract, member, containerContract, containerProperty);
188  }
189  else
190  {
191  SerializeMultidimensionalArray(writer, (Array)value, arrayContract, member, containerContract, containerProperty);
192  }
193  break;
194  case JsonContractType.Primitive:
195  SerializePrimitive(writer, value, (JsonPrimitiveContract)valueContract, member, containerContract, containerProperty);
196  break;
197  case JsonContractType.String:
198  SerializeString(writer, value, (JsonStringContract)valueContract);
199  break;
200  case JsonContractType.Dictionary:
201  JsonDictionaryContract dictionaryContract = (JsonDictionaryContract)valueContract;
202  SerializeDictionary(writer, (value is IDictionary dictionary) ? dictionary : dictionaryContract.CreateWrapper(value), dictionaryContract, member, containerContract, containerProperty);
203  break;
204 #if HAVE_DYNAMIC
205  case JsonContractType.Dynamic:
206  SerializeDynamic(writer, (IDynamicMetaObjectProvider)value, (JsonDynamicContract)valueContract, member, containerContract, containerProperty);
207  break;
208 #endif
209 #if HAVE_BINARY_SERIALIZATION
210  case JsonContractType.Serializable:
211  SerializeISerializable(writer, (ISerializable)value, (JsonISerializableContract)valueContract, member, containerContract, containerProperty);
212  break;
213 #endif
214  case JsonContractType.Linq:
215  ((JToken)value).WriteTo(writer, Serializer.Converters.ToArray());
216  break;
217  }

220  private bool? ResolveIsReference(JsonContract contract, JsonProperty? property, JsonContainerContract? collectionContract, JsonProperty? containerProperty)
221  {
222  bool? isReference = null;
223 
224  // value could be coming from a dictionary or array and not have a property
225  if (property != null)
226  {
227  isReference = property.IsReference;
228  }
229 
230  if (isReference == null && containerProperty != null)
231  {
232  isReference = containerProperty.ItemIsReference;
233  }
234 
235  if (isReference == null && collectionContract != null)
236  {
237  isReference = collectionContract.ItemIsReference;
238  }
239 
240  if (isReference == null)
241  {
242  isReference = contract.IsReference;
243  }
244 
245  return isReference;
246  }

248  private bool ShouldWriteReference(object? value, JsonProperty? property, JsonContract? valueContract, JsonContainerContract? collectionContract, JsonProperty? containerProperty)
249  {
250  if (value == null)
251  {
252  return false;
253  }
254 
255  MiscellaneousUtils.Assert(valueContract != null);
256 
257  if (valueContract.ContractType == JsonContractType.Primitive || valueContract.ContractType == JsonContractType.String)
258  {
259  return false;
260  }
261 
262  bool? isReference = ResolveIsReference(valueContract, property, collectionContract, containerProperty);
263 
264  if (isReference == null)
265  {
266  if (valueContract.ContractType == JsonContractType.Array)
267  {
268  isReference = HasFlag(Serializer._preserveReferencesHandling, PreserveReferencesHandling.Arrays);
269  }
270  else
271  {
272  isReference = HasFlag(Serializer._preserveReferencesHandling, PreserveReferencesHandling.Objects);
273  }
274  }
275 
276  if (!isReference.GetValueOrDefault())
277  {
278  return false;
279  }
280 
281  return Serializer.GetReferenceResolver().IsReferenced(this, value);
282  }

284  private bool ShouldWriteProperty(object? memberValue, JsonObjectContract? containerContract, JsonProperty property)
285  {
286  if (memberValue == null && ResolvedNullValueHandling(containerContract, property) == NullValueHandling.Ignore)

293  {
294  return false;
295  }
296 
297  return true;

300  private bool CheckForCircularReference(JsonWriter writer, object? value, JsonProperty? property, JsonContract? contract, JsonContainerContract? containerContract, JsonProperty? containerProperty)
301  {
302  if (value == null)
303  {
304  return true;
305  }
306 
307  MiscellaneousUtils.Assert(contract != null);
308 
309  if (contract.ContractType == JsonContractType.Primitive || contract.ContractType == JsonContractType.String)
310  {
311  return true;
312  }
313 
314  ReferenceLoopHandling? referenceLoopHandling = null;
315 
316  if (property != null)
317  {
318  referenceLoopHandling = property.ReferenceLoopHandling;
319  }
320 
321  if (referenceLoopHandling == null && containerProperty != null)
322  {
323  referenceLoopHandling = containerProperty.ItemReferenceLoopHandling;
324  }
325 
326  if (referenceLoopHandling == null && containerContract != null)
327  {
328  referenceLoopHandling = containerContract.ItemReferenceLoopHandling;
329  }
330 
331  bool exists = (Serializer._equalityComparer != null)
332  ? _serializeStack.Contains(value, Serializer._equalityComparer)
333  : _serializeStack.Contains(value);
334 
335  if (exists)
336  {
337  string message = "Self referencing loop detected";
338  if (property != null)
339  {
340  message += " for property '{0}'".FormatWith(CultureInfo.InvariantCulture, property.PropertyName);
341  }
342  message += " with type '{0}'.".FormatWith(CultureInfo.InvariantCulture, value.GetType());
343 
344  switch (referenceLoopHandling.GetValueOrDefault(Serializer._referenceLoopHandling))
345  {
346  case ReferenceLoopHandling.Error:
347  throw JsonSerializationException.Create(null, writer.ContainerPath, message, null);
348  case ReferenceLoopHandling.Ignore:
349  if (TraceWriter != null && TraceWriter.LevelFilter >= TraceLevel.Verbose)
350  {
351  TraceWriter.Trace(TraceLevel.Verbose, JsonPosition.FormatMessage(null, writer.Path, message + ". Skipping serializing self referenced value."), null);
352  }
353 
354  return false;
355  case ReferenceLoopHandling.Serialize:
356  if (TraceWriter != null && TraceWriter.LevelFilter >= TraceLevel.Verbose)
357  {
358  TraceWriter.Trace(TraceLevel.Verbose, JsonPosition.FormatMessage(null, writer.Path, message + ". Serializing self referenced value."), null);
359  }
360 
361  return true;
362  }
363  }
364 
365  return true;
366  }

398  {
399 #if HAVE_TYPE_DESCRIPTOR

455  private void SerializeObject(JsonWriter writer, object value, JsonObjectContract contract, JsonProperty? member, JsonContainerContract? collectionContract, JsonProperty? containerProperty)
456  {
457  OnSerializing(writer, contract, value);
458 
459  _serializeStack.Add(value);
460 
461  WriteObjectStart(writer, value, contract, member, collectionContract, containerProperty);
462 
463  int initialDepth = writer.Top;
464 
465  for (int index = 0; index < contract.Properties.Count; index++)
466  {
467  JsonProperty property = contract.Properties[index];
468  try
469  {
470  if (!CalculatePropertyValues(writer, value, contract, member, property, out JsonContract? memberContract, out object? memberValue))
471  {
472  continue;
473  }
474 
475  property.WritePropertyName(writer);
476  SerializeValue(writer, memberValue, memberContract, property, contract, member);
477  }
478  catch (Exception ex)
479  {
480  if (IsErrorHandled(value, contract, property.PropertyName, null, writer.ContainerPath, ex))
481  {
482  HandleError(writer, initialDepth);
483  }
484  else
485  {
486  throw;
487  }
488  }
489  }
490 
491  IEnumerable<KeyValuePair<object, object>>? extensionData = contract.ExtensionDataGetter?.Invoke(value);
492  if (extensionData != null)
493  {
494  foreach (KeyValuePair<object, object> e in extensionData)
495  {
496  JsonContract keyContract = GetContract(e.Key);
497  JsonContract? valueContract = GetContractSafe(e.Value);
498 
499  string propertyName = GetPropertyName(writer, e.Key, keyContract, out _);
500 
501  propertyName = (contract.ExtensionDataNameResolver != null)
502  ? contract.ExtensionDataNameResolver(propertyName)
503  : propertyName;
504 
505  if (ShouldWriteReference(e.Value, null, valueContract, contract, member))
506  {
507  writer.WritePropertyName(propertyName);
508  WriteReference(writer, e.Value!);
509  }
510  else
511  {
512  if (!CheckForCircularReference(writer, e.Value, null, valueContract, contract, member))
513  {
514  continue;
515  }
516 
517  writer.WritePropertyName(propertyName);
518 
519  SerializeValue(writer, e.Value, valueContract, null, contract, member);
520  }
521  }
522  }
523 
524  writer.WriteEndObject();
525 
526  _serializeStack.RemoveAt(_serializeStack.Count - 1);
527 
528  OnSerialized(writer, contract, value);
529  }

531  private bool CalculatePropertyValues(JsonWriter writer, object value, JsonContainerContract contract, JsonProperty? member, JsonProperty property, [NotNullWhen(true)]out JsonContract? memberContract, [NotNullWhen(true)]out object? memberValue)
532  {
533  if (!property.Ignored && property.Readable && ShouldSerialize(writer, property, value) && IsSpecified(writer, property, value))
534  {
535  if (property.PropertyContract == null)
536  {
537  property.PropertyContract = Serializer._contractResolver.ResolveContract(property.PropertyType!);
538  }
539 
540  memberValue = property.ValueProvider!.GetValue(value);
541  memberContract = (property.PropertyContract.IsSealed) ? property.PropertyContract : GetContractSafe(memberValue);
542 
543  if (ShouldWriteProperty(memberValue, contract as JsonObjectContract, property))
544  {
545  if (ShouldWriteReference(memberValue, property, memberContract!, contract, member))
546  {
547  property.WritePropertyName(writer);
548  WriteReference(writer, memberValue!);
549  return false;
550  }
551 
552  if (!CheckForCircularReference(writer, memberValue, property, memberContract!, contract, member))
553  {
554  return false;
555  }
556 
557  if (memberValue == null)
558  {
559  JsonObjectContract? objectContract = contract as JsonObjectContract;
560  Required resolvedRequired = property._required ?? objectContract?.ItemRequired ?? Required.Default;
561  if (resolvedRequired == Required.Always)
562  {
563  throw JsonSerializationException.Create(null, writer.ContainerPath, "Cannot write a null value for property '{0}'. Property requires a value.".FormatWith(CultureInfo.InvariantCulture, property.PropertyName), null);
564  }
565  if (resolvedRequired == Required.DisallowNull)
566  {
567  throw JsonSerializationException.Create(null, writer.ContainerPath, "Cannot write a null value for property '{0}'. Property requires a non-null value.".FormatWith(CultureInfo.InvariantCulture, property.PropertyName), null);
568  }
569  }
570 
571  return true;
572  }
573  }
574 
575  memberContract = null;
576  memberValue = null;
577  return false;
578  }

678  private void SerializeList(JsonWriter writer, IEnumerable values, JsonArrayContract contract, JsonProperty? member, JsonContainerContract? collectionContract, JsonProperty? containerProperty)
679  {
680  object underlyingList = values is IWrappedCollection wrappedCollection ? wrappedCollection.UnderlyingCollection : values;
681 
682  OnSerializing(writer, contract, underlyingList);
683 
684  _serializeStack.Add(underlyingList);
685 
686  bool hasWrittenMetadataObject = WriteStartArray(writer, underlyingList, contract, member, collectionContract, containerProperty);
687 
688  writer.WriteStartArray();
689 
690  int initialDepth = writer.Top;
691 
692  int index = 0;
693  // note that an error in the IEnumerable won't be caught
694  foreach (object value in values)
695  {
696  try
697  {
698  JsonContract? valueContract = contract.FinalItemContract ?? GetContractSafe(value);
699 
700  if (ShouldWriteReference(value, null, valueContract, contract, member))
701  {
702  WriteReference(writer, value);
703  }
704  else
705  {
706  if (CheckForCircularReference(writer, value, null, valueContract, contract, member))
707  {
708  SerializeValue(writer, value, valueContract, null, contract, member);
709  }
710  }
711  }
712  catch (Exception ex)
713  {
714  if (IsErrorHandled(underlyingList, contract, index, null, writer.ContainerPath, ex))
715  {
716  HandleError(writer, initialDepth);
717  }
718  else
719  {
720  throw;
721  }
722  }
723  finally
724  {
725  index++;
726  }
727  }
728 
729  writer.WriteEndArray();
730 
731  if (hasWrittenMetadataObject)
732  {
733  writer.WriteEndObject();
734  }
735 
736  _serializeStack.RemoveAt(_serializeStack.Count - 1);
737 
738  OnSerialized(writer, contract, underlyingList);
739  }

743  OnSerializing(writer, contract, values);
