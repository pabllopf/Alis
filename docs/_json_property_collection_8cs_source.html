74  public void AddProperty(JsonProperty property)
75  {
76  MiscellaneousUtils.Assert(property.PropertyName != null);
77 
78  if (Contains(property.PropertyName))
79  {
80  // don't overwrite existing property with ignored property
81  if (property.Ignored)
82  {
83  return;
84  }
85 
86  JsonProperty existingProperty = this[property.PropertyName];
87  bool duplicateProperty = true;
88 
89  if (existingProperty.Ignored)
90  {
91  // remove ignored property so it can be replaced in collection
92  Remove(existingProperty);
93  duplicateProperty = false;
94  }
95  else
96  {
97  if (property.DeclaringType != null && existingProperty.DeclaringType != null)
98  {
99  if (property.DeclaringType.IsSubclassOf(existingProperty.DeclaringType)
100  || (existingProperty.DeclaringType.IsInterface() && property.DeclaringType.ImplementInterface(existingProperty.DeclaringType)))
101  {
102  // current property is on a derived class and hides the existing
103  Remove(existingProperty);
104  duplicateProperty = false;
105  }
106  if (existingProperty.DeclaringType.IsSubclassOf(property.DeclaringType)
107  || (property.DeclaringType.IsInterface() && existingProperty.DeclaringType.ImplementInterface(property.DeclaringType)))
108  {
109  // current property is hidden by the existing so don't add it
110  return;
111  }
112 
113  if (_type.ImplementInterface(existingProperty.DeclaringType) && _type.ImplementInterface(property.DeclaringType))
114  {
115  // current property was already defined on another interface
116  return;
117  }
118  }
119  }
120 
121  if (duplicateProperty)
122  {
123  throw new JsonSerializationException("A member with the name '{0}' already exists on '{1}'. Use the JsonPropertyAttribute to specify another name.".FormatWith(CultureInfo.InvariantCulture, property.PropertyName, _type));
124  }
125  }
126 
127  Add(property);
128  }

165  public JsonProperty? GetProperty(string propertyName, StringComparison comparisonType)
166  {

168  if (comparisonType == StringComparison.Ordinal)
169  {
170  if (TryGetValue(propertyName, out JsonProperty? property))
171  {
172  return property;
173  }
174 
175  return null;
176  }
177 
178  for (int i = 0; i < _list.Count; i++)
179  {
180  JsonProperty property = _list[i];
181  if (string.Equals(propertyName, property.PropertyName, comparisonType))
182  {
183  return property;
184  }
185  }
186
