123  private bool ParseTime(ref int start)
124  {
125  if (!(Parse2Digit(start, out Hour)
126  && Hour <= 24
127  && ParseChar(start + LzHH, ':')
128  && Parse2Digit(start + LzHH_, out Minute)

134  {
135  return false;
136  }
