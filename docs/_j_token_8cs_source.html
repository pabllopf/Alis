184  public string Path
185  {
186  get
187  {
188  if (Parent == null)
189  {
190  return string.Empty;
191  }
192 
193  List<JsonPosition> positions = new List<JsonPosition>();
194  JToken? previous = null;
195  for (JToken? current = this; current != null; current = current.Parent)
196  {
197  switch (current.Type)
198  {
199  case JTokenType.Property:
200  JProperty property = (JProperty)current;
201  positions.Add(new JsonPosition(JsonContainerType.Object) { PropertyName = property.Name });
202  break;
203  case JTokenType.Array:
204  case JTokenType.Constructor:
205  if (previous != null)
206  {
207  int index = ((IList<JToken>)current).IndexOf(previous);
208 
209  positions.Add(new JsonPosition(JsonContainerType.Array) { Position = index });
210  }
211  break;
212  }
213 
214  previous = current;
215  }
216 
217 #if HAVE_FAST_REVERSE
218  positions.FastReverse();
219 #else
220  positions.Reverse();
221 #endif
222 
223  return JsonPosition.BuildPath(positions, null);
224  }
