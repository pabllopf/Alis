170  private static readonly IList<JsonSchemaModel> EmptySchemaList = new List<JsonSchemaModel>();
171 
172  private IList<JsonSchemaModel> CurrentMemberSchemas
173  {
174  get
175  {
176  if (_currentScope == null)
177  {
178  return new List<JsonSchemaModel>(new[] { _model });
179  }
180 
181  if (_currentScope.Schemas == null || _currentScope.Schemas.Count == 0)
182  {
183  return EmptySchemaList;
184  }
185 
186  switch (_currentScope.TokenType)
187  {
188  case JTokenType.None:
189  return _currentScope.Schemas;
190  case JTokenType.Object:
191  {
192  if (_currentScope.CurrentPropertyName == null)
193  {
194  throw new JsonReaderException("CurrentPropertyName has not been set on scope.");
195  }
196 
197  IList<JsonSchemaModel> schemas = new List<JsonSchemaModel>();
198 
199  foreach (JsonSchemaModel schema in CurrentSchemas)
200  {
201  if (schema.Properties != null && schema.Properties.TryGetValue(_currentScope.CurrentPropertyName, out JsonSchemaModel propertySchema))
202  {
203  schemas.Add(propertySchema);
204  }
205  if (schema.PatternProperties != null)
206  {
207  foreach (KeyValuePair<string, JsonSchemaModel> patternProperty in schema.PatternProperties)
208  {
209  if (Regex.IsMatch(_currentScope.CurrentPropertyName, patternProperty.Key))
210  {
211  schemas.Add(patternProperty.Value);
212  }
213  }
214  }
215 
216  if (schemas.Count == 0 && schema.AllowAdditionalProperties && schema.AdditionalProperties != null)
217  {
218  schemas.Add(schema.AdditionalProperties);
219  }
220  }
221 
222  return schemas;
223  }
224  case JTokenType.Array:
225  {
226  IList<JsonSchemaModel> schemas = new List<JsonSchemaModel>();
227 
228  foreach (JsonSchemaModel schema in CurrentSchemas)
229  {
230  if (!schema.PositionalItemsValidation)
231  {
232  if (schema.Items != null && schema.Items.Count > 0)
233  {
234  schemas.Add(schema.Items[0]);
235  }
236  }
237  else
238  {
239  if (schema.Items != null && schema.Items.Count > 0)
240  {
241  if (schema.Items.Count > (_currentScope.ArrayItemCount - 1))
242  {
243  schemas.Add(schema.Items[_currentScope.ArrayItemCount - 1]);
244  }
245  }
246 
247  if (schema.AllowAdditionalItems && schema.AdditionalItems != null)
248  {
249  schemas.Add(schema.AdditionalItems);
250  }
251  }
252  }
253 
254  return schemas;
255  }
256  case JTokenType.Constructor:
257  return EmptySchemaList;
258  default:
259  throw new ArgumentOutOfRangeException("TokenType", "Unexpected token type: {0}".FormatWith(CultureInfo.InvariantCulture, _currentScope.TokenType));
260  }
261  }
262  }

500  private void ValidateCurrentToken()
501  {
502  // first time validate has been called. build model
503  if (_model == null)
504  {
505  JsonSchemaModelBuilder builder = new JsonSchemaModelBuilder();
506  _model = builder.Build(_schema);
507 
508  if (!JsonTokenUtils.IsStartToken(_reader.TokenType))
509  {
510  Push(new SchemaScope(JTokenType.None, CurrentMemberSchemas));
511  }
512  }
513 
514  switch (_reader.TokenType)
515  {
516  case JsonToken.StartObject:
517  ProcessValue();
518  IList<JsonSchemaModel> objectSchemas = CurrentMemberSchemas.Where(ValidateObject).ToList();
519  Push(new SchemaScope(JTokenType.Object, objectSchemas));
520  WriteToken(CurrentSchemas);
521  break;
522  case JsonToken.StartArray:
523  ProcessValue();
524  IList<JsonSchemaModel> arraySchemas = CurrentMemberSchemas.Where(ValidateArray).ToList();
525  Push(new SchemaScope(JTokenType.Array, arraySchemas));
526  WriteToken(CurrentSchemas);
527  break;
528  case JsonToken.StartConstructor:
529  ProcessValue();
530  Push(new SchemaScope(JTokenType.Constructor, null));
531  WriteToken(CurrentSchemas);
532  break;
533  case JsonToken.PropertyName:
534  WriteToken(CurrentSchemas);
535  foreach (JsonSchemaModel schema in CurrentSchemas)
536  {
537  ValidatePropertyName(schema);
538  }
539  break;
540  case JsonToken.Raw:
541  ProcessValue();
542  break;
543  case JsonToken.Integer:
544  ProcessValue();
545  WriteToken(CurrentMemberSchemas);
546  foreach (JsonSchemaModel schema in CurrentMemberSchemas)
547  {
548  ValidateInteger(schema);
549  }
550  break;
551  case JsonToken.Float:
552  ProcessValue();
553  WriteToken(CurrentMemberSchemas);
554  foreach (JsonSchemaModel schema in CurrentMemberSchemas)
555  {
556  ValidateFloat(schema);
557  }
558  break;
559  case JsonToken.String:
560  ProcessValue();
561  WriteToken(CurrentMemberSchemas);
562  foreach (JsonSchemaModel schema in CurrentMemberSchemas)
563  {
564  ValidateString(schema);
565  }
566  break;
567  case JsonToken.Boolean:
568  ProcessValue();
569  WriteToken(CurrentMemberSchemas);
570  foreach (JsonSchemaModel schema in CurrentMemberSchemas)
571  {
572  ValidateBoolean(schema);
573  }
574  break;
575  case JsonToken.Null:
576  ProcessValue();
577  WriteToken(CurrentMemberSchemas);
578  foreach (JsonSchemaModel schema in CurrentMemberSchemas)
579  {
580  ValidateNull(schema);
581  }
582  break;
583  case JsonToken.EndObject:
584  WriteToken(CurrentSchemas);
585  foreach (JsonSchemaModel schema in CurrentSchemas)
586  {
587  ValidateEndObject(schema);
588  }
589  Pop();
590  break;
591  case JsonToken.EndArray:
592  WriteToken(CurrentSchemas);
593  foreach (JsonSchemaModel schema in CurrentSchemas)
594  {
595  ValidateEndArray(schema);
596  }
597  Pop();
598  break;
599  case JsonToken.EndConstructor:
600  WriteToken(CurrentSchemas);
601  Pop();
602  break;
603  case JsonToken.Undefined:
604  case JsonToken.Date:
605  case JsonToken.Bytes:
606  // these have no equivalent in JSON schema
607  WriteToken(CurrentMemberSchemas);
608  break;
609  case JsonToken.None:
610  // no content, do nothing
611  break;
612  default:
613  throw new ArgumentOutOfRangeException();
614  }
615  }
616 
617  private void WriteToken(IList<JsonSchemaModel> schemas)
618  {
619  foreach (SchemaScope schemaScope in _stack)
620  {
621  bool isInUniqueArray = (schemaScope.TokenType == JTokenType.Array && schemaScope.IsUniqueArray && schemaScope.ArrayItemCount > 0);
622 
623  if (isInUniqueArray || schemas.Any(s => s.Enum != null))
624  {
625  if (schemaScope.CurrentItemWriter == null)
626  {
627  if (JsonTokenUtils.IsEndToken(_reader.TokenType))
628  {
629  continue;
630  }
631 
632  schemaScope.CurrentItemWriter = new JTokenWriter();
633  }
634 
635  schemaScope.CurrentItemWriter.WriteToken(_reader, false);
636 
637  // finished writing current item
638  if (schemaScope.CurrentItemWriter.Top == 0 && _reader.TokenType != JsonToken.PropertyName)
639  {
640  JToken finishedItem = schemaScope.CurrentItemWriter.Token;
641 
642  // start next item with new writer
643  schemaScope.CurrentItemWriter = null;
644 
645  if (isInUniqueArray)
646  {
647  if (schemaScope.UniqueArrayItems.Contains(finishedItem, JToken.EqualityComparer))
648  {
649  RaiseError("Non-unique array item at index {0}.".FormatWith(CultureInfo.InvariantCulture, schemaScope.ArrayItemCount - 1), schemaScope.Schemas.First(s => s.UniqueItems));
650  }
651 
652  schemaScope.UniqueArrayItems.Add(finishedItem);
653  }
654  else if (schemas.Any(s => s.Enum != null))
655  {
656  foreach (JsonSchemaModel schema in schemas)
657  {
658  if (schema.Enum != null)
659  {
660  if (!schema.Enum.ContainsValue(finishedItem, JToken.EqualityComparer))
661  {
662  StringWriter sw = new StringWriter(CultureInfo.InvariantCulture);
663  finishedItem.WriteTo(new JsonTextWriter(sw));
664 
665  RaiseError("Value {0} is not defined in enum.".FormatWith(CultureInfo.InvariantCulture, sw.ToString()), schema);
666  }
667  }
668  }
669  }
670  }
671  }
672  }
673  }

675  private void ValidateEndObject(JsonSchemaModel schema)
676  {
677  if (schema == null)
678  {
679  return;
680  }
681 
682  Dictionary<string, bool> requiredProperties = _currentScope.RequiredProperties;
683 
684  if (requiredProperties != null && requiredProperties.Values.Any(v => !v))
685  {
686  IEnumerable<string> unmatchedRequiredProperties = requiredProperties.Where(kv => !kv.Value).Select(kv => kv.Key);
687  RaiseError("Required properties are missing from object: {0}.".FormatWith(CultureInfo.InvariantCulture, string.Join(", ", unmatchedRequiredProperties
688 #if !HAVE_STRING_JOIN_WITH_ENUMERABLE
689  .ToArray()
690 #endif
691  )), schema);
692  }
693  }

695  private void ValidateEndArray(JsonSchemaModel schema)
696  {
697  if (schema == null)
698  {
699  return;
700  }
701 
702  int arrayItemCount = _currentScope.ArrayItemCount;
703 
704  if (schema.MaximumItems != null && arrayItemCount > schema.MaximumItems)
705  {
706  RaiseError("Array item count {0} exceeds maximum count of {1}.".FormatWith(CultureInfo.InvariantCulture, arrayItemCount, schema.MaximumItems), schema);
707  }
708 
709  if (schema.MinimumItems != null && arrayItemCount < schema.MinimumItems)
710  {
711  RaiseError("Array item count {0} is less than minimum count of {1}.".FormatWith(CultureInfo.InvariantCulture, arrayItemCount, schema.MinimumItems), schema);
712  }
713  }

745  private void ValidateString(JsonSchemaModel schema)
746  {
747  if (schema == null)
748  {
749  return;
750  }
751 
752  if (!TestType(schema, JsonSchemaType.String))
753  {
754  return;
755  }
756 
757  ValidateNotDisallowed(schema);
758 
759  string value = _reader.Value.ToString();
760 
761  if (schema.MaximumLength != null && value.Length > schema.MaximumLength)
762  {
763  RaiseError("String '{0}' exceeds maximum length of {1}.".FormatWith(CultureInfo.InvariantCulture, value, schema.MaximumLength), schema);
764  }
765 
766  if (schema.MinimumLength != null && value.Length < schema.MinimumLength)
767  {
768  RaiseError("String '{0}' is less than minimum length of {1}.".FormatWith(CultureInfo.InvariantCulture, value, schema.MinimumLength), schema);
769  }
770 
771  if (schema.Patterns != null)
772  {
773  foreach (string pattern in schema.Patterns)
774  {
775  if (!Regex.IsMatch(value, pattern))
776  {
777  RaiseError("String '{0}' does not match regex pattern '{1}'.".FormatWith(CultureInfo.InvariantCulture, value, pattern), schema);
778  }
779  }
780  }
781  }

783  private void ValidateInteger(JsonSchemaModel schema)
784  {
785  if (schema == null)
786  {
787  return;
788  }
789 
790  if (!TestType(schema, JsonSchemaType.Integer))
791  {
792  return;
793  }
794 
795  ValidateNotDisallowed(schema);
796 
797  object value = _reader.Value;
798 
799  if (schema.Maximum != null)
800  {
801  if (JValue.Compare(JTokenType.Integer, value, schema.Maximum) > 0)
802  {
803  RaiseError("Integer {0} exceeds maximum value of {1}.".FormatWith(CultureInfo.InvariantCulture, value, schema.Maximum), schema);
804  }
805  if (schema.ExclusiveMaximum && JValue.Compare(JTokenType.Integer, value, schema.Maximum) == 0)
806  {
807  RaiseError("Integer {0} equals maximum value of {1} and exclusive maximum is true.".FormatWith(CultureInfo.InvariantCulture, value, schema.Maximum), schema);
808  }
809  }
810 
811  if (schema.Minimum != null)
812  {
813  if (JValue.Compare(JTokenType.Integer, value, schema.Minimum) < 0)
814  {
815  RaiseError("Integer {0} is less than minimum value of {1}.".FormatWith(CultureInfo.InvariantCulture, value, schema.Minimum), schema);
816  }
817  if (schema.ExclusiveMinimum && JValue.Compare(JTokenType.Integer, value, schema.Minimum) == 0)
818  {
819  RaiseError("Integer {0} equals minimum value of {1} and exclusive minimum is true.".FormatWith(CultureInfo.InvariantCulture, value, schema.Minimum), schema);
820  }
821  }
822 
823  if (schema.DivisibleBy != null)
824  {
825  bool notDivisible;
826 #if HAVE_BIG_INTEGER
827  if (value is BigInteger i)
828  {
829  // not that this will lose any decimal point on DivisibleBy
830  // so manually raise an error if DivisibleBy is not an integer and value is not zero
831  bool divisibleNonInteger = !Math.Abs(schema.DivisibleBy.Value - Math.Truncate(schema.DivisibleBy.Value)).Equals(0);
832  if (divisibleNonInteger)
833  {
834  notDivisible = i != 0;
835  }
836  else
837  {
838  notDivisible = i % new BigInteger(schema.DivisibleBy.Value) != 0;
839  }
840  }
841  else
842 #endif
843  {
844  notDivisible = !IsZero(Convert.ToInt64(value, CultureInfo.InvariantCulture) % schema.DivisibleBy.GetValueOrDefault());
845  }
846 
847  if (notDivisible)
848  {
849  RaiseError("Integer {0} is not evenly divisible by {1}.".FormatWith(CultureInfo.InvariantCulture, JsonConvert.ToString(value), schema.DivisibleBy), schema);
850  }
851  }
852  }
853 
854  private void ProcessValue()
855  {
856  if (_currentScope != null && _currentScope.TokenType == JTokenType.Array)
857  {
858  _currentScope.ArrayItemCount++;
859 
860  foreach (JsonSchemaModel currentSchema in CurrentSchemas)
861  {
862  // if there is positional validation and the array index is past the number of item validation schemas and there are no additional items then error
863  if (currentSchema != null
864  && currentSchema.PositionalItemsValidation
865  && !currentSchema.AllowAdditionalItems
866  && (currentSchema.Items == null || _currentScope.ArrayItemCount - 1 >= currentSchema.Items.Count))
867  {
868  RaiseError("Index {0} has not been defined and the schema does not allow additional items.".FormatWith(CultureInfo.InvariantCulture, _currentScope.ArrayItemCount), currentSchema);
869  }
870  }
871  }
872  }

874  private void ValidateFloat(JsonSchemaModel schema)
875  {
876  if (schema == null)
877  {
878  return;
879  }
880 
881  if (!TestType(schema, JsonSchemaType.Float))
882  {
883  return;
884  }
885 
886  ValidateNotDisallowed(schema);
887 
888  double value = Convert.ToDouble(_reader.Value, CultureInfo.InvariantCulture);
889 
890  if (schema.Maximum != null)
891  {
892  if (value > schema.Maximum)
893  {
894  RaiseError("Float {0} exceeds maximum value of {1}.".FormatWith(CultureInfo.InvariantCulture, JsonConvert.ToString(value), schema.Maximum), schema);
895  }
896  if (schema.ExclusiveMaximum && value == schema.Maximum)
897  {
898  RaiseError("Float {0} equals maximum value of {1} and exclusive maximum is true.".FormatWith(CultureInfo.InvariantCulture, JsonConvert.ToString(value), schema.Maximum), schema);
899  }
900  }
901 
902  if (schema.Minimum != null)
903  {
904  if (value < schema.Minimum)
905  {
