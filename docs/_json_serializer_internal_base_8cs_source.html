94  private ErrorContext GetErrorContext(object? currentObject, object? member, string path, Exception error)
95  {
96  if (_currentErrorContext == null)

98  _currentErrorContext = new ErrorContext(currentObject, member, path, error);

119  protected bool IsErrorHandled(object? currentObject, JsonContract? contract, object? keyValue, IJsonLineInfo? lineInfo, string path, Exception ex)
120  {
121  ErrorContext errorContext = GetErrorContext(currentObject, keyValue, path, ex);
122 
123  if (TraceWriter != null && TraceWriter.LevelFilter >= TraceLevel.Error && !errorContext.Traced)
124  {
125  // only write error once
126  errorContext.Traced = true;
127 
128  // kind of a hack but meh. might clean this up later
129  string message = (GetType() == typeof(JsonSerializerInternalWriter)) ? "Error serializing" : "Error deserializing";
130  if (contract != null)
131  {
132  message += " " + contract.UnderlyingType;
133  }
134  message += ". " + ex.Message;
135 
136  // add line information to non-json.net exception message
137  if (!(ex is JsonException))
138  {
139  message = JsonPosition.FormatMessage(lineInfo, path, message);
140  }
141 
142  TraceWriter.Trace(TraceLevel.Error, message, ex);
143  }
144 
145  // attribute method is non-static so don't invoke if no object
146  if (contract != null && currentObject != null)
147  {
148  contract.InvokeOnError(currentObject, Serializer.Context, errorContext);
149  }
150 
151  if (!errorContext.Handled)
152  {
153  Serializer.OnError(new ErrorEventArgs(currentObject, errorContext));
154  }
155 
156  return errorContext.Handled;
157  }
158  }
