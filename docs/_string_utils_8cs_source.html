90  public static bool IsWhiteSpace(string s)
91  {
92  if (s == null)
93  {
94  throw new ArgumentNullException(nameof(s));
95  }
96 
97  if (s.Length == 0)
98  {
99  return false;
100  }
101 
102  for (int i = 0; i < s.Length; i++)
103  {
104  if (!char.IsWhiteSpace(s[i]))
105  {
106  return false;
107  }
108  }
109 
110  return true;

131  public static TSource ForgivingCaseSensitiveFind<TSource>(this IEnumerable<TSource> source, Func<TSource, string> valueSelector, string testValue)
132  {
133  if (source == null)
134  {
135  throw new ArgumentNullException(nameof(source));
136  }
137  if (valueSelector == null)
138  {
139  throw new ArgumentNullException(nameof(valueSelector));
140  }
141 
142  IEnumerable<TSource> caseInsensitiveResults = source.Where(s => string.Equals(valueSelector(s), testValue, StringComparison.OrdinalIgnoreCase));
143  if (caseInsensitiveResults.Count() <= 1)
144  {
145  return caseInsensitiveResults.SingleOrDefault();
146  }
147  else
148  {
149  // multiple results returned. now filter using case sensitivity
150  IEnumerable<TSource> caseSensitiveResults = source.Where(s => string.Equals(valueSelector(s), testValue, StringComparison.Ordinal));
151  return caseSensitiveResults.SingleOrDefault();
152  }

155  public static string ToCamelCase(string s)
156  {
157  if (StringUtils.IsNullOrEmpty(s) || !char.IsUpper(s[0]))
158  {
159  return s;
160  }
161 
162  char[] chars = s.ToCharArray();
163 
164  for (int i = 0; i < chars.Length; i++)
165  {
166  if (i == 1 && !char.IsUpper(chars[i]))
167  {
168  break;
169  }
170 
171  bool hasNext = (i + 1 < chars.Length);
172  if (i > 0 && hasNext && !char.IsUpper(chars[i + 1]))
173  {
174  // if the next character is a space, which is not considered uppercase
175  // (otherwise we wouldn't be here...)
176  // we want to ensure that the following:
177  // 'FOO bar' is rewritten as 'foo bar', and not as 'foO bar'
178  // The code was written in such a way that the first word in uppercase
179  // ends when if finds an uppercase letter followed by a lowercase letter.
180  // now a ' ' (space, (char)32) is considered not upper
181  // but in that case we still want our current character to become lowercase
182  if (char.IsSeparator(chars[i + 1]))
183  {
184  chars[i] = ToLower(chars[i]);
185  }
186 
187  break;
188  }
189 
190  chars[i] = ToLower(chars[i]);
191  }
192 
193  return new string(chars);
194  }

218  private static string ToSeparatedCase(string s, char separator)
219  {
220  if (StringUtils.IsNullOrEmpty(s))
221  {
222  return s;
223  }
224 
225  StringBuilder sb = new StringBuilder();
226  SeparatedCaseState state = SeparatedCaseState.Start;
227 
228  for (int i = 0; i < s.Length; i++)
229  {
230  if (s[i] == ' ')
231  {
232  if (state != SeparatedCaseState.Start)
233  {
234  state = SeparatedCaseState.NewWord;
235  }
236  }
237  else if (char.IsUpper(s[i]))
238  {
239  switch (state)
240  {
241  case SeparatedCaseState.Upper:
242  bool hasNext = (i + 1 < s.Length);
243  if (i > 0 && hasNext)
244  {
245  char nextChar = s[i + 1];
246  if (!char.IsUpper(nextChar) && nextChar != separator)
247  {
248  sb.Append(separator);
249  }
250  }
251  break;
252  case SeparatedCaseState.Lower:
253  case SeparatedCaseState.NewWord:
254  sb.Append(separator);
255  break;
256  }
257 
258  char c;
259 #if HAVE_CHAR_TO_LOWER_WITH_CULTURE
260  c = char.ToLower(s[i], CultureInfo.InvariantCulture);
261 #else
262  c = char.ToLowerInvariant(s[i]);
263 #endif
264  sb.Append(c);
265 
266  state = SeparatedCaseState.Upper;
267  }
268  else if (s[i] == separator)
269  {
270  sb.Append(separator);
271  state = SeparatedCaseState.Start;
272  }
273  else
274  {
275  if (state == SeparatedCaseState.NewWord)
276  {
277  sb.Append(separator);
278  }
279 
280  sb.Append(s[i]);
281  state = SeparatedCaseState.Lower;
282  }
283  }
284 
285  return sb.ToString();
286  }

316  public static string Trim(this string s, int start, int length)
317  {
318  // References: https://referencesource.microsoft.com/#mscorlib/system/string.cs,2691
319  // https://referencesource.microsoft.com/#mscorlib/system/string.cs,1226
320  if (s == null)
321  {
322  throw new ArgumentNullException();
323  }
324  if (start < 0)
325  {
326  throw new ArgumentOutOfRangeException(nameof(start));
327  }
328  if (length < 0)
329  {
330  throw new ArgumentOutOfRangeException(nameof(length));
331  }
332  int end = start + length - 1;
333  if (end >= s.Length)
334  {
335  throw new ArgumentOutOfRangeException(nameof(length));
336  }
337  for (; start < end; start++)
338  {
339  if (!char.IsWhiteSpace(s[start]))
340  {
341  break;
342  }
343  }
344  for (; end >= start; end--)
345  {
346  if (!char.IsWhiteSpace(s[end]))
347  {
348  break;
349  }
350  }
351  return s.Substring(start, end - start + 1);
352  }
353  }
