127  public override void WriteJson(JsonWriter writer, object? value, JsonSerializer serializer)
128  {
129  if (value == null)
130  {
131  writer.WriteNull();
132  return;
133  }
134 
135  DefaultContractResolver? resolver = serializer.ContractResolver as DefaultContractResolver;
136 
137  Type unionType = UnionTypeLookupCache.Get(value.GetType());
138  Union union = UnionCache.Get(unionType);
139 
140  int tag = (int)union.TagReader.Invoke(value);
141  UnionCase caseInfo = union.Cases.Single(c => c.Tag == tag);
142 
143  writer.WriteStartObject();
144  writer.WritePropertyName((resolver != null) ? resolver.GetResolvedPropertyName(CasePropertyName) : CasePropertyName);
145  writer.WriteValue(caseInfo.Name);
146  if (caseInfo.Fields != null && caseInfo.Fields.Length > 0)
147  {
148  object[] fields = (object[])caseInfo.FieldReader.Invoke(value)!;
149 
150  writer.WritePropertyName((resolver != null) ? resolver.GetResolvedPropertyName(FieldsPropertyName) : FieldsPropertyName);
151  writer.WriteStartArray();
152  foreach (object field in fields)
153  {
154  serializer.Serialize(writer, field);
155  }
156  writer.WriteEndArray();
157  }
158  writer.WriteEndObject();
159  }

169  public override object? ReadJson(JsonReader reader, Type objectType, object? existingValue, JsonSerializer serializer)
170  {
171  if (reader.TokenType == JsonToken.Null)
172  {
173  return null;
174  }
175 
176  UnionCase? caseInfo = null;
177  string? caseName = null;
178  JArray? fields = null;
179 
180  // start object
181  reader.ReadAndAssert();
182 
183  while (reader.TokenType == JsonToken.PropertyName)
184  {
185  string propertyName = reader.Value!.ToString();
186  if (string.Equals(propertyName, CasePropertyName, StringComparison.OrdinalIgnoreCase))
187  {
188  reader.ReadAndAssert();
189 
190  Union union = UnionCache.Get(objectType);
191 
192  caseName = reader.Value!.ToString();
193 
194  caseInfo = union.Cases.SingleOrDefault(c => c.Name == caseName);
195 
196  if (caseInfo == null)
197  {
198  throw JsonSerializationException.Create(reader, "No union type found with the name '{0}'.".FormatWith(CultureInfo.InvariantCulture, caseName));
199  }
200  }
201  else if (string.Equals(propertyName, FieldsPropertyName, StringComparison.OrdinalIgnoreCase))
202  {
203  reader.ReadAndAssert();
204  if (reader.TokenType != JsonToken.StartArray)
205  {
206  throw JsonSerializationException.Create(reader, "Union fields must been an array.");
207  }
208 
209  fields = (JArray)JToken.ReadFrom(reader);
210  }
211  else
212  {
213  throw JsonSerializationException.Create(reader, "Unexpected property '{0}' found when reading union.".FormatWith(CultureInfo.InvariantCulture, propertyName));
214  }
215 
216  reader.ReadAndAssert();
217  }
218 
219  if (caseInfo == null)
220  {
221  throw JsonSerializationException.Create(reader, "No '{0}' property with union name found.".FormatWith(CultureInfo.InvariantCulture, CasePropertyName));
222  }
223 
224  object?[] typedFieldValues = new object?[caseInfo.Fields.Length];
225 
226  if (caseInfo.Fields.Length > 0 && fields == null)
227  {
228  throw JsonSerializationException.Create(reader, "No '{0}' property with union fields found.".FormatWith(CultureInfo.InvariantCulture, FieldsPropertyName));
229  }
230 
231  if (fields != null)
232  {
233  if (caseInfo.Fields.Length != fields.Count)
234  {
235  throw JsonSerializationException.Create(reader, "The number of field values does not match the number of properties defined by union '{0}'.".FormatWith(CultureInfo.InvariantCulture, caseName));
236  }
237 
238  for (int i = 0; i < fields.Count; i++)
239  {
240  JToken t = fields[i];
241  PropertyInfo fieldProperty = caseInfo.Fields[i];
242 
243  typedFieldValues[i] = t.ToObject(fieldProperty.PropertyType, serializer);
244  }
245  }
246 
247  object[] args = { typedFieldValues };
248 
249  return caseInfo.Constructor.Invoke(args);
250  }

259  public override bool CanConvert(Type objectType)
260  {
261  if (typeof(IEnumerable).IsAssignableFrom(objectType))
262  {
263  return false;
264  }
265 
266  // all fsharp objects have CompilationMappingAttribute
267  // get the fsharp assembly from the attribute and initialize latebound methods
268  object[] attributes;
269 #if HAVE_FULL_REFLECTION
270  attributes = objectType.GetCustomAttributes(true);
271 #else
272  attributes = objectType.GetTypeInfo().GetCustomAttributes(true).ToArray();
273 #endif
274 
275  bool isFSharpType = false;
276  foreach (object attribute in attributes)
277  {
278  Type attributeType = attribute.GetType();
279  if (attributeType.FullName == "Microsoft.FSharp.Core.CompilationMappingAttribute")
280  {
281  FSharpUtils.EnsureInitialized(attributeType.Assembly());
282 
283  isFSharpType = true;
284  break;
285  }
286  }
287 
288  if (!isFSharpType)
289  {
290  return false;
291  }
292 
293  return (bool)FSharpUtils.Instance.IsUnion(null, objectType, null);
294  }
