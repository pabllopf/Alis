77 #pragma warning disable 618
78  private void WriteBson(BsonWriter writer, Regex regex)
79  {
80  // Regular expression - The first cstring is the regex pattern, the second
81  // is the regex options string. Options are identified by characters, which
82  // must be stored in alphabetical order. Valid options are 'i' for case
83  // insensitive matching, 'm' for multiline matching, 'x' for verbose mode,
84  // 'l' to make \w, \W, etc. locale dependent, 's' for dotall mode
85  // ('.' matches everything), and 'u' to make \w, \W, etc. match unicode.
86 
87  string? options = null;
88 
89  if (HasFlag(regex.Options, RegexOptions.IgnoreCase))
90  {
91  options += "i";
92  }
93 
94  if (HasFlag(regex.Options, RegexOptions.Multiline))
95  {
96  options += "m";
97  }
98 
99  if (HasFlag(regex.Options, RegexOptions.Singleline))
100  {
101  options += "s";
102  }
103 
104  options += "u";
105 
106  if (HasFlag(regex.Options, RegexOptions.ExplicitCapture))
107  {
108  options += "x";
109  }
110 
111  writer.WriteRegex(regex.ToString(), options);
112  }

115  private void WriteJson(JsonWriter writer, Regex regex, JsonSerializer serializer)
116  {
117  DefaultContractResolver? resolver = serializer.ContractResolver as DefaultContractResolver;
118 
119  writer.WriteStartObject();
120  writer.WritePropertyName((resolver != null) ? resolver.GetResolvedPropertyName(PatternName) : PatternName);
121  writer.WriteValue(regex.ToString());
122  writer.WritePropertyName((resolver != null) ? resolver.GetResolvedPropertyName(OptionsName) : OptionsName);
123  serializer.Serialize(writer, regex.Options);
124  writer.WriteEndObject();
125  }

135  public override object? ReadJson(JsonReader reader, Type objectType, object? existingValue, JsonSerializer serializer)
136  {

150  private object ReadRegexString(JsonReader reader)
151  {
152  string regexText = (string)reader.Value!;
153 
154  if (regexText.Length > 0 && regexText[0] == '/')
155  {
156  int patternOptionDelimiterIndex = regexText.LastIndexOf('/');
157 
158  if (patternOptionDelimiterIndex > 0)
159  {
160  string patternText = regexText.Substring(1, patternOptionDelimiterIndex - 1);
161  string optionsText = regexText.Substring(patternOptionDelimiterIndex + 1);
162 
163  RegexOptions options = MiscellaneousUtils.GetRegexOptions(optionsText);
164 
165  return new Regex(patternText, options);
166  }
167  }
168 
169  throw JsonSerializationException.Create(reader, "Regex pattern must be enclosed by slashes.");
170  }
171 
172  private Regex ReadRegexObject(JsonReader reader, JsonSerializer serializer)
173  {
174  string? pattern = null;
175  RegexOptions? options = null;
176 
177  while (reader.Read())
178  {
179  switch (reader.TokenType)
180  {
181  case JsonToken.PropertyName:
182  string propertyName = reader.Value!.ToString();
183 
184  if (!reader.Read())
185  {
186  throw JsonSerializationException.Create(reader, "Unexpected end when reading Regex.");
187  }
188 
189  if (string.Equals(propertyName, PatternName, StringComparison.OrdinalIgnoreCase))
190  {
191  pattern = (string?)reader.Value;
192  }
193  else if (string.Equals(propertyName, OptionsName, StringComparison.OrdinalIgnoreCase))
194  {
195  options = serializer.Deserialize<RegexOptions>(reader);
196  }
197  else
198  {
199  reader.Skip();
200  }
201  break;
202  case JsonToken.Comment:
203  break;
204  case JsonToken.EndObject:
205  if (pattern == null)
206  {
207  throw JsonSerializationException.Create(reader, "Error deserializing Regex. No pattern found.");
208  }
209 
210  return new Regex(pattern, options ?? RegexOptions.None);
211  }
212  }
213 
214  throw JsonSerializationException.Create(reader, "Unexpected end when reading Regex.");
215  }
