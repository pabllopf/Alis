55  {
56  switch (Operator)

108  public override bool IsMatch(JToken root, JToken t)
109  {
110  if (Operator == QueryOperator.Exists)
111  {
112  return GetResult(root, t, Left).Any();
113  }
114 
115  using (IEnumerator<JToken> leftResults = GetResult(root, t, Left).GetEnumerator())
116  {
117  if (leftResults.MoveNext())
118  {
119  IEnumerable<JToken> rightResultsEn = GetResult(root, t, Right);
120  ICollection<JToken> rightResults = rightResultsEn as ICollection<JToken> ?? rightResultsEn.ToList();
121 
122  do
123  {
124  JToken leftResult = leftResults.Current;
125  foreach (JToken rightResult in rightResults)
126  {
127  if (MatchTokens(leftResult, rightResult))
128  {
129  return true;
130  }
131  }
132  } while (leftResults.MoveNext());
133  }
134  }
135 
136  return false;
137  }

139  private bool MatchTokens(JToken leftResult, JToken rightResult)
140  {
141  if (leftResult is JValue leftValue && rightResult is JValue rightValue)
142  {
143  switch (Operator)
144  {
145  case QueryOperator.RegexEquals:
146  if (RegexEquals(leftValue, rightValue))
147  {
148  return true;
149  }
150  break;
151  case QueryOperator.Equals:
152  if (EqualsWithStringCoercion(leftValue, rightValue))
153  {
154  return true;
155  }
156  break;
157  case QueryOperator.StrictEquals:
158  if (EqualsWithStrictMatch(leftValue, rightValue))
159  {
160  return true;
161  }
162  break;
163  case QueryOperator.NotEquals:
164  if (!EqualsWithStringCoercion(leftValue, rightValue))
165  {
166  return true;
167  }
168  break;
169  case QueryOperator.StrictNotEquals:
170  if (!EqualsWithStrictMatch(leftValue, rightValue))
171  {
172  return true;
173  }
174  break;
175  case QueryOperator.GreaterThan:
176  if (leftValue.CompareTo(rightValue) > 0)
177  {
178  return true;
179  }
180  break;
181  case QueryOperator.GreaterThanOrEquals:
182  if (leftValue.CompareTo(rightValue) >= 0)
183  {
184  return true;
185  }
186  break;
187  case QueryOperator.LessThan:
188  if (leftValue.CompareTo(rightValue) < 0)
189  {
190  return true;
191  }
192  break;
193  case QueryOperator.LessThanOrEquals:
194  if (leftValue.CompareTo(rightValue) <= 0)
195  {
196  return true;
197  }
198  break;
199  case QueryOperator.Exists:
200  return true;
201  }
202  }
203  else
204  {
205  switch (Operator)
206  {
207  case QueryOperator.Exists:
208  // you can only specify primitive types in a comparison
209  // notequals will always be true
210  case QueryOperator.NotEquals:
211  return true;
212  }
213  }
214 
215  return false;
216  }

219  {

234  internal static bool EqualsWithStringCoercion(JValue value, JValue queryValue)
235  {
236  if (value.Equals(queryValue))
237  {
238  return true;
239  }
240 
241  // Handle comparing an integer with a float
242  // e.g. Comparing 1 and 1.0
243  if ((value.Type == JTokenType.Integer && queryValue.Type == JTokenType.Float)
244  || (value.Type == JTokenType.Float && queryValue.Type == JTokenType.Integer))
245  {
246  return JValue.Compare(value.Type, value.Value, queryValue.Value) == 0;
247  }
248 
249  if (queryValue.Type != JTokenType.String)
250  {
251  return false;
252  }
253 
254  string queryValueString = (string)queryValue.Value!;
255 
256  string currentValueString;
257 
258  // potential performance issue with converting every value to string?
259  switch (value.Type)
260  {
261  case JTokenType.Date:
262  using (StringWriter writer = StringUtils.CreateStringWriter(64))
263  {
264 #if HAVE_DATE_TIME_OFFSET
265  if (value.Value is DateTimeOffset offset)
266  {
267  DateTimeUtils.WriteDateTimeOffsetString(writer, offset, DateFormatHandling.IsoDateFormat, null, CultureInfo.InvariantCulture);
268  }
269  else
270 #endif
271  {
272  DateTimeUtils.WriteDateTimeString(writer, (DateTime)value.Value!, DateFormatHandling.IsoDateFormat, null, CultureInfo.InvariantCulture);
273  }
274 
275  currentValueString = writer.ToString();
276  }
277  break;
278  case JTokenType.Bytes:
279  currentValueString = Convert.ToBase64String((byte[])value.Value!);
280  break;
281  case JTokenType.Guid:
282  case JTokenType.TimeSpan:
283  currentValueString = value.Value!.ToString();
284  break;
285  case JTokenType.Uri:
286  currentValueString = ((Uri)value.Value!).OriginalString;
287  break;
288  default:
289  return false;
290  }
291 
292  return string.Equals(currentValueString, queryValueString, StringComparison.Ordinal);
293  }

295  internal static bool EqualsWithStrictMatch(JValue value, JValue queryValue)
296  {
297  MiscellaneousUtils.Assert(value != null);
298  MiscellaneousUtils.Assert(queryValue != null);
299 
300  // Handle comparing an integer with a float
301  // e.g. Comparing 1 and 1.0
302  if ((value.Type == JTokenType.Integer && queryValue.Type == JTokenType.Float)
303  || (value.Type == JTokenType.Float && queryValue.Type == JTokenType.Integer))
304  {
305  return JValue.Compare(value.Type, value.Value, queryValue.Value) == 0;
306  }
307 
308  // we handle floats and integers the exact same way, so they are pseudo equivalent
309  if (value.Type != queryValue.Type)
310  {
311  return false;
312  }
313 
314  return value.Equals(queryValue);
315  }
