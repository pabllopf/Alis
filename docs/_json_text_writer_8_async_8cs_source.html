95  internal Task DoWriteEndAsync(JsonToken token, CancellationToken cancellationToken)
96  {
97  switch (token)
98  {
99  case JsonToken.EndObject:
100  return _writer.WriteAsync('}', cancellationToken);
101  case JsonToken.EndArray:
102  return _writer.WriteAsync(']', cancellationToken);
103  case JsonToken.EndConstructor:
104  return _writer.WriteAsync(')', cancellationToken);
105  default:
106  throw JsonWriterException.Create(this, "Invalid JsonToken: " + token, null);
107  }
108  }

123  internal async Task DoCloseAsync(CancellationToken cancellationToken)
124  {
125  if (Top == 0) // otherwise will happen in calls to WriteEndAsync
126  {
127  cancellationToken.ThrowIfCancellationRequested();
128  }
129 
130  while (Top > 0)
131  {
132  await WriteEndAsync(cancellationToken).ConfigureAwait(false);
133  }
134 
135  CloseBufferAndWriter();
136  }

162  internal Task DoWriteIndentAsync(CancellationToken cancellationToken)
163  {
164  // levels of indentation multiplied by the indent count
165  int currentIndentCount = Top * _indentation;
166 
167  int newLineLen = SetIndentChars();
168  MiscellaneousUtils.Assert(_indentChars != null);
169 
170  if (currentIndentCount <= IndentCharBufferSize)
171  {
172  return _writer.WriteAsync(_indentChars, 0, newLineLen + currentIndentCount, cancellationToken);
173  }
174 
175  return WriteIndentAsync(currentIndentCount, newLineLen, cancellationToken);
176  }
177 
178  private async Task WriteIndentAsync(int currentIndentCount, int newLineLen, CancellationToken cancellationToken)
179  {
180  MiscellaneousUtils.Assert(_indentChars != null);
181 
182  await _writer.WriteAsync(_indentChars, 0, newLineLen + Math.Min(currentIndentCount, IndentCharBufferSize), cancellationToken).ConfigureAwait(false);
183 
184  while ((currentIndentCount -= IndentCharBufferSize) > 0)
185  {
186  await _writer.WriteAsync(_indentChars, newLineLen, Math.Min(currentIndentCount, IndentCharBufferSize), cancellationToken).ConfigureAwait(false);
187  }
188  }
