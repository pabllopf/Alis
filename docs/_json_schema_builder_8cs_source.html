95  private JsonSchema ResolveReferences(JsonSchema schema)
96  {
97  if (schema.DeferredReference != null)
98  {
99  string reference = schema.DeferredReference;
100 
101  bool locationReference = (reference.StartsWith("#", StringComparison.Ordinal));
102  if (locationReference)
103  {
104  reference = UnescapeReference(reference);
105  }
106 
107  JsonSchema resolvedSchema = _resolver.GetSchema(reference);
108 
109  if (resolvedSchema == null)
110  {
111  if (locationReference)
112  {
113  string[] escapedParts = schema.DeferredReference.TrimStart('#').Split(new[] { '/' }, StringSplitOptions.RemoveEmptyEntries);
114  JToken currentToken = _rootSchema;
115  foreach (string escapedPart in escapedParts)
116  {
117  string part = UnescapeReference(escapedPart);
118 
119  if (currentToken.Type == JTokenType.Object)
120  {
121  currentToken = currentToken[part];
122  }
123  else if (currentToken.Type == JTokenType.Array || currentToken.Type == JTokenType.Constructor)
124  {
125  if (int.TryParse(part, out int index) && index >= 0 && index < currentToken.Count())
126  {
127  currentToken = currentToken[index];
128  }
129  else
130  {
131  currentToken = null;
132  }
133  }
134 
135  if (currentToken == null)
136  {
137  break;
138  }
139  }
140 
141  if (currentToken != null)
142  {
143  resolvedSchema = BuildSchema(currentToken);
144  }
145  }
146 
147  if (resolvedSchema == null)
148  {
149  throw new JsonException("Could not resolve schema reference '{0}'.".FormatWith(CultureInfo.InvariantCulture, schema.DeferredReference));
150  }
151  }
152 
153  schema = resolvedSchema;
154  }
155 
156  if (schema.ReferencesResolved)
157  {
158  return schema;
159  }
160 
161  schema.ReferencesResolved = true;
162 
163  if (schema.Extends != null)
164  {
165  for (int i = 0; i < schema.Extends.Count; i++)
166  {
167  schema.Extends[i] = ResolveReferences(schema.Extends[i]);
168  }
169  }
170 
171  if (schema.Items != null)
172  {
173  for (int i = 0; i < schema.Items.Count; i++)
174  {
175  schema.Items[i] = ResolveReferences(schema.Items[i]);
176  }
177  }
178 
179  if (schema.AdditionalItems != null)
180  {
181  schema.AdditionalItems = ResolveReferences(schema.AdditionalItems);
182  }
183 
184  if (schema.PatternProperties != null)
185  {
186  foreach (KeyValuePair<string, JsonSchema> patternProperty in schema.PatternProperties.ToList())
187  {
188  schema.PatternProperties[patternProperty.Key] = ResolveReferences(patternProperty.Value);
189  }
190  }
191 
192  if (schema.Properties != null)
193  {
194  foreach (KeyValuePair<string, JsonSchema> property in schema.Properties.ToList())
195  {
196  schema.Properties[property.Key] = ResolveReferences(property.Value);
197  }
198  }
199 
200  if (schema.AdditionalProperties != null)
201  {
202  schema.AdditionalProperties = ResolveReferences(schema.AdditionalProperties);
203  }
204 
205  return schema;
206  }

208  private JsonSchema BuildSchema(JToken token)
209  {
210  if (!(token is JObject schemaObject))
211  {
212  throw JsonException.Create(token, token.Path, "Expected object while parsing schema object, got {0}.".FormatWith(CultureInfo.InvariantCulture, token.Type));
213  }
214 
215  if (schemaObject.TryGetValue(JsonTypeReflector.RefPropertyName, out JToken referenceToken))
216  {
217  JsonSchema deferredSchema = new JsonSchema();
218  deferredSchema.DeferredReference = (string)referenceToken;
219 
220  return deferredSchema;
221  }
222 
223  string location = token.Path.Replace(".", "/").Replace("[", "/").Replace("]", string.Empty);
224  if (!StringUtils.IsNullOrEmpty(location))
225  {
226  location = "/" + location;
227  }
228  location = "#" + location;
229 
230  if (_documentSchemas.TryGetValue(location, out JsonSchema existingSchema))
231  {
232  return existingSchema;
233  }
234 
235  Push(new JsonSchema { Location = location });
236 
237  ProcessSchemaProperties(schemaObject);
238 
239  return Pop();
240  }

339  private void ProcessExtends(JToken token)
340  {
341  IList<JsonSchema> schemas = new List<JsonSchema>();
342 
343  if (token.Type == JTokenType.Array)
344  {
345  foreach (JToken schemaObject in token)
346  {
347  schemas.Add(BuildSchema(schemaObject));
348  }
349  }
350  else
351  {
352  JsonSchema schema = BuildSchema(token);
353  if (schema != null)
354  {
355  schemas.Add(schema);
356  }
357  }
358 
359  if (schemas.Count > 0)
360  {
361  CurrentSchema.Extends = schemas;
362  }
363  }

365  private void ProcessEnum(JToken token)
366  {
367  if (token.Type != JTokenType.Array)
368  {
369  throw JsonException.Create(token, token.Path, "Expected Array token while parsing enum values, got {0}.".FormatWith(CultureInfo.InvariantCulture, token.Type));

372  CurrentSchema.Enum = new List<JToken>();
373 
374  foreach (JToken enumValue in token)

404  private IDictionary<string, JsonSchema> ProcessProperties(JToken token)
405  {
406  IDictionary<string, JsonSchema> properties = new Dictionary<string, JsonSchema>();
407 
408  if (token.Type != JTokenType.Object)
409  {
410  throw JsonException.Create(token, token.Path, "Expected Object token while parsing schema properties, got {0}.".FormatWith(CultureInfo.InvariantCulture, token.Type));
411  }
412 
413  foreach (JProperty propertyToken in token)
414  {
415  if (properties.ContainsKey(propertyToken.Name))
416  {
417  throw new JsonException("Property {0} has already been defined in schema.".FormatWith(CultureInfo.InvariantCulture, propertyToken.Name));
418  }
419 
420  properties.Add(propertyToken.Name, BuildSchema(propertyToken.Value));
421  }
422 
423  return properties;
424  }
425 
426  private void ProcessItems(JToken token)
427  {
428  CurrentSchema.Items = new List<JsonSchema>();
429 
430  switch (token.Type)
431  {
432  case JTokenType.Object:
433  CurrentSchema.Items.Add(BuildSchema(token));
434  CurrentSchema.PositionalItemsValidation = false;
435  break;
436  case JTokenType.Array:
437  CurrentSchema.PositionalItemsValidation = true;
438  foreach (JToken schemaToken in token)
439  {
440  CurrentSchema.Items.Add(BuildSchema(schemaToken));
441  }
442  break;
443  default:
444  throw JsonException.Create(token, token.Path, "Expected array or JSON schema object, got {0}.".FormatWith(CultureInfo.InvariantCulture, token.Type));
445  }
446  }
447 
448  private JsonSchemaType? ProcessType(JToken token)
449  {
450  switch (token.Type)
451  {
452  case JTokenType.Array:
453  // ensure type is in blank state before ORing values
454  JsonSchemaType? type = JsonSchemaType.None;
455 
456  foreach (JToken typeToken in token)
457  {
458  if (typeToken.Type != JTokenType.String)
459  {
460  throw JsonException.Create(typeToken, typeToken.Path, "Expected JSON schema type string token, got {0}.".FormatWith(CultureInfo.InvariantCulture, token.Type));
461  }
462 
463  type = type | MapType((string)typeToken);
464  }
465 
466  return type;
467  case JTokenType.String:
468  return MapType((string)token);
469  default:
470  throw JsonException.Create(token, token.Path, "Expected array or JSON schema type string token, got {0}.".FormatWith(CultureInfo.InvariantCulture, token.Type));
471  }
472  }

474  internal static JsonSchemaType MapType(string type)
475  {
476  if (!JsonSchemaConstants.JsonSchemaTypeMapping.TryGetValue(type, out JsonSchemaType mappedType))
477  {
478  throw new JsonException("Invalid JSON schema type: {0}".FormatWith(CultureInfo.InvariantCulture, type));
479  }
480 
481  return mappedType;
482  }
483 
484  internal static string MapType(JsonSchemaType type)
485  {
486  return JsonSchemaConstants.JsonSchemaTypeMapping.Single(kv => kv.Value == type).Key;
487  }
488  }
