311  public void CopyTo(KeyValuePair<TKey, TValue>[] array, int arrayIndex)
312  {
313  if (_dictionary != null)
314  {
315  // Manual use of IDictionaryEnumerator instead of foreach to avoid DictionaryEntry box allocations.
316  IDictionaryEnumerator e = _dictionary.GetEnumerator();
317  try
318  {
319  while (e.MoveNext())
320  {
321  DictionaryEntry entry = e.Entry;
322  array[arrayIndex++] = new KeyValuePair<TKey, TValue>((TKey)entry.Key, (TValue)entry.Value);
323  }
324  }
325  finally
326  {
327  (e as IDisposable)?.Dispose();
328  }
329  }
330 #if HAVE_READ_ONLY_COLLECTIONS
331  else if (_readOnlyDictionary != null)
332  {
333  throw new NotSupportedException();
334  }
335 #endif
336  else
337  {
338  GenericDictionary.CopyTo(array, arrayIndex);
339  }
340  }
