116  public JsonArrayContract(Type underlyingType)
117  : base(underlyingType)
118  {
119  ContractType = JsonContractType.Array;
120 
121  // netcoreapp3.0 uses EmptyPartition for empty enumerable. Treat as an empty array.
122  IsArray = CreatedType.IsArray ||
123  (NonNullableUnderlyingType.IsGenericType() && NonNullableUnderlyingType.GetGenericTypeDefinition().FullName == "System.Linq.EmptyPartition`1");
124 
125  bool canDeserialize;
126 
127  Type? tempCollectionType;
128  if (IsArray)
129  {
130  CollectionItemType = ReflectionUtils.GetCollectionItemType(UnderlyingType);
131  IsReadOnlyOrFixedSize = true;
132  _genericCollectionDefinitionType = typeof(List<>).MakeGenericType(CollectionItemType);
133 
134  canDeserialize = true;
135  IsMultidimensionalArray = (CreatedType.IsArray && UnderlyingType.GetArrayRank() > 1);
136  }
137  else if (typeof(IList).IsAssignableFrom(NonNullableUnderlyingType))
138  {
139  if (ReflectionUtils.ImplementsGenericDefinition(NonNullableUnderlyingType, typeof(ICollection<>), out _genericCollectionDefinitionType))
140  {
141  CollectionItemType = _genericCollectionDefinitionType.GetGenericArguments()[0];
142  }
143  else
144  {
145  CollectionItemType = ReflectionUtils.GetCollectionItemType(NonNullableUnderlyingType);
146  }
147 
148  if (NonNullableUnderlyingType == typeof(IList))
149  {
150  CreatedType = typeof(List<object>);
151  }
152 
153  if (CollectionItemType != null)
154  {
155  _parameterizedConstructor = CollectionUtils.ResolveEnumerableCollectionConstructor(NonNullableUnderlyingType, CollectionItemType);
156  }
157 
158  IsReadOnlyOrFixedSize = ReflectionUtils.InheritsGenericDefinition(NonNullableUnderlyingType, typeof(ReadOnlyCollection<>));
159  canDeserialize = true;
160  }
161  else if (ReflectionUtils.ImplementsGenericDefinition(NonNullableUnderlyingType, typeof(ICollection<>), out _genericCollectionDefinitionType))
162  {
163  CollectionItemType = _genericCollectionDefinitionType.GetGenericArguments()[0];
164 
165  if (ReflectionUtils.IsGenericDefinition(NonNullableUnderlyingType, typeof(ICollection<>))
166  || ReflectionUtils.IsGenericDefinition(NonNullableUnderlyingType, typeof(IList<>)))
167  {
168  CreatedType = typeof(List<>).MakeGenericType(CollectionItemType);
169  }
170 
171 #if HAVE_ISET
172  if (ReflectionUtils.IsGenericDefinition(NonNullableUnderlyingType, typeof(ISet<>)))
173  {
174  CreatedType = typeof(HashSet<>).MakeGenericType(CollectionItemType);
175  }
176 #endif
177 
178  _parameterizedConstructor = CollectionUtils.ResolveEnumerableCollectionConstructor(NonNullableUnderlyingType, CollectionItemType);
179  canDeserialize = true;
180  ShouldCreateWrapper = true;
181  }
182 #if HAVE_READ_ONLY_COLLECTIONS
183  else if (ReflectionUtils.ImplementsGenericDefinition(NonNullableUnderlyingType, typeof(IReadOnlyCollection<>), out tempCollectionType))
184  {
185  CollectionItemType = tempCollectionType.GetGenericArguments()[0];
186 
187  if (ReflectionUtils.IsGenericDefinition(NonNullableUnderlyingType, typeof(IReadOnlyCollection<>))
188  || ReflectionUtils.IsGenericDefinition(NonNullableUnderlyingType, typeof(IReadOnlyList<>)))
189  {
190  CreatedType = typeof(ReadOnlyCollection<>).MakeGenericType(CollectionItemType);
191  }
192 
193  _genericCollectionDefinitionType = typeof(List<>).MakeGenericType(CollectionItemType);
194  _parameterizedConstructor = CollectionUtils.ResolveEnumerableCollectionConstructor(CreatedType, CollectionItemType);
195 
196 #if HAVE_FSHARP_TYPES
197  StoreFSharpListCreatorIfNecessary(NonNullableUnderlyingType);
198 #endif
199 
200  IsReadOnlyOrFixedSize = true;
201  canDeserialize = HasParameterizedCreatorInternal;
202  }
203 #endif
204  else if (ReflectionUtils.ImplementsGenericDefinition(NonNullableUnderlyingType, typeof(IEnumerable<>), out tempCollectionType))
205  {
206  CollectionItemType = tempCollectionType.GetGenericArguments()[0];
207 
208  if (ReflectionUtils.IsGenericDefinition(UnderlyingType, typeof(IEnumerable<>)))
209  {
210  CreatedType = typeof(List<>).MakeGenericType(CollectionItemType);
211  }
212 
213  _parameterizedConstructor = CollectionUtils.ResolveEnumerableCollectionConstructor(NonNullableUnderlyingType, CollectionItemType);
214 
215 #if HAVE_FSHARP_TYPES
216  StoreFSharpListCreatorIfNecessary(NonNullableUnderlyingType);
217 #endif
218 
219  if (NonNullableUnderlyingType.IsGenericType() && NonNullableUnderlyingType.GetGenericTypeDefinition() == typeof(IEnumerable<>))
220  {
221  _genericCollectionDefinitionType = tempCollectionType;
222 
223  IsReadOnlyOrFixedSize = false;
224  ShouldCreateWrapper = false;
225  canDeserialize = true;
226  }
227  else
228  {
229  _genericCollectionDefinitionType = typeof(List<>).MakeGenericType(CollectionItemType);
230 
231  IsReadOnlyOrFixedSize = true;
232  ShouldCreateWrapper = true;
233  canDeserialize = HasParameterizedCreatorInternal;
234  }
235  }
236  else
237  {
238  // types that implement IEnumerable and nothing else
239  canDeserialize = false;
240  ShouldCreateWrapper = true;
241  }
242 
243  CanDeserialize = canDeserialize;
244 
245 #if (NET20 || NET35)
246  if (CollectionItemType != null && ReflectionUtils.IsNullableType(CollectionItemType))
247  {
248  // bug in .NET 2.0 & 3.5 that List<Nullable<T>> throws an error when adding null via IList.Add(object)
249  // wrapper will handle calling Add(T) instead
250  if (ReflectionUtils.InheritsGenericDefinition(CreatedType, typeof(List<>), out tempCollectionType)
251  || (IsArray && !IsMultidimensionalArray))
252  {
253  ShouldCreateWrapper = true;
254  }
255  }
256 #endif
257 
258  if (CollectionItemType != null &&
259  ImmutableCollectionsUtils.TryBuildImmutableForArrayContract(
260  NonNullableUnderlyingType,
261  CollectionItemType,
262  out Type? immutableCreatedType,
263  out ObjectConstructor<object>? immutableParameterizedCreator))
264  {
265  CreatedType = immutableCreatedType;
266  _parameterizedCreator = immutableParameterizedCreator;
267  IsReadOnlyOrFixedSize = true;
268  CanDeserialize = true;
269  }
270  }

272  internal IWrappedCollection CreateWrapper(object list)
273  {
274  if (_genericWrapperCreator == null)
275  {
276  MiscellaneousUtils.Assert(_genericCollectionDefinitionType != null);
277 
278  _genericWrapperType = typeof(CollectionWrapper<>).MakeGenericType(CollectionItemType);
279 
280  Type constructorArgument;
281 
282  if (ReflectionUtils.InheritsGenericDefinition(_genericCollectionDefinitionType, typeof(List<>))
283  || _genericCollectionDefinitionType.GetGenericTypeDefinition() == typeof(IEnumerable<>))
284  {
285  constructorArgument = typeof(ICollection<>).MakeGenericType(CollectionItemType);
286  }
287  else
288  {
289  constructorArgument = _genericCollectionDefinitionType;
290  }
291 
292  ConstructorInfo genericWrapperConstructor = _genericWrapperType.GetConstructor(new[] { constructorArgument });
293  _genericWrapperCreator = JsonTypeReflector.ReflectionDelegateFactory.CreateParameterizedConstructor(genericWrapperConstructor);
294  }
295 
296  return (IWrappedCollection)_genericWrapperCreator(list);
297  }

299  internal IList CreateTemporaryCollection()
300  {
301  if (_genericTemporaryCollectionCreator == null)
302  {
303  // multidimensional array will also have array instances in it
304  Type collectionItemType = (IsMultidimensionalArray || CollectionItemType == null)
305  ? typeof(object)
306  : CollectionItemType;
307 
308  Type temporaryListType = typeof(List<>).MakeGenericType(collectionItemType);
309  _genericTemporaryCollectionCreator = JsonTypeReflector.ReflectionDelegateFactory.CreateDefaultConstructor<object>(temporaryListType);
310  }
311 
312  return (IList)_genericTemporaryCollectionCreator();
313  }
