134  private static MethodInfo GetMethodWithNonPublicFallback(Type type, string methodName, BindingFlags bindingFlags)
135  {
136  MethodInfo methodInfo = type.GetMethod(methodName, bindingFlags);
137 
138  // if no matching method then attempt to find with nonpublic flag
139  // this is required because in WinApps some methods are private but always using NonPublic breaks medium trust
140  // https://github.com/JamesNK/Newtonsoft.Json/pull/649
141  // https://github.com/JamesNK/Newtonsoft.Json/issues/821
142  if (methodInfo == null && (bindingFlags & BindingFlags.NonPublic) != BindingFlags.NonPublic)
143  {
144  methodInfo = type.GetMethod(methodName, bindingFlags | BindingFlags.NonPublic);
145  }
146 
147  return methodInfo!;
148  }

150  private static MethodCall<object?, object> CreateFSharpFuncCall(Type type, string methodName)
151  {
152  MethodInfo innerMethodInfo = GetMethodWithNonPublicFallback(type, methodName, BindingFlags.Public | BindingFlags.Static);
153  MethodInfo invokeFunc = innerMethodInfo.ReturnType.GetMethod("Invoke", BindingFlags.Public | BindingFlags.Instance);
154 
155  MethodCall<object?, object?> call = JsonTypeReflector.ReflectionDelegateFactory.CreateMethodCall<object?>(innerMethodInfo);
156  MethodCall<object?, object> invoke = JsonTypeReflector.ReflectionDelegateFactory.CreateMethodCall<object?>(invokeFunc)!;
157 
158  MethodCall<object?, object> createFunction = (target, args) =>
159  {
160  object? result = call(target, args);
161 
162  FSharpFunction f = new FSharpFunction(result, invoke);
163  return f;
164  };
165 
166  return createFunction;
167  }

185  public ObjectConstructor<object> BuildMapCreator<TKey, TValue>()
186  {
187  Type genericMapType = _mapType.MakeGenericType(typeof(TKey), typeof(TValue));
188  ConstructorInfo ctor = genericMapType.GetConstructor(new[] { typeof(IEnumerable<Tuple<TKey, TValue>>) });
189  ObjectConstructor<object> ctorDelegate = JsonTypeReflector.ReflectionDelegateFactory.CreateParameterizedConstructor(ctor);
190 
191  ObjectConstructor<object> creator = args =>
192  {
193  // convert dictionary KeyValuePairs to Tuples
194  IEnumerable<KeyValuePair<TKey, TValue>> values = (IEnumerable<KeyValuePair<TKey, TValue>>)args[0]!;
195  IEnumerable<Tuple<TKey, TValue>> tupleValues = values.Select(kv => new Tuple<TKey, TValue>(kv.Key, kv.Value));
196 
197  return ctorDelegate(tupleValues);
198  };
199 
200  return creator;
201  }
