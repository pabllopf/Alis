457  public static string ToString(object? value)
458  {
459  if (value == null)
460  {
461  return Null;
462  }
463 
464  PrimitiveTypeCode typeCode = ConvertUtils.GetTypeCode(value.GetType());
465 
466  switch (typeCode)
467  {
468  case PrimitiveTypeCode.String:
469  return ToString((string)value);
470  case PrimitiveTypeCode.Char:
471  return ToString((char)value);
472  case PrimitiveTypeCode.Boolean:
473  return ToString((bool)value);
474  case PrimitiveTypeCode.SByte:
475  return ToString((sbyte)value);
476  case PrimitiveTypeCode.Int16:
477  return ToString((short)value);
478  case PrimitiveTypeCode.UInt16:
479  return ToString((ushort)value);
480  case PrimitiveTypeCode.Int32:
481  return ToString((int)value);
482  case PrimitiveTypeCode.Byte:
483  return ToString((byte)value);
484  case PrimitiveTypeCode.UInt32:
485  return ToString((uint)value);
486  case PrimitiveTypeCode.Int64:
487  return ToString((long)value);
488  case PrimitiveTypeCode.UInt64:
489  return ToString((ulong)value);
490  case PrimitiveTypeCode.Single:
491  return ToString((float)value);
492  case PrimitiveTypeCode.Double:
493  return ToString((double)value);
494  case PrimitiveTypeCode.DateTime:
495  return ToString((DateTime)value);
496  case PrimitiveTypeCode.Decimal:
497  return ToString((decimal)value);
498 #if HAVE_DB_NULL_TYPE_CODE
499  case PrimitiveTypeCode.DBNull:
500  return Null;
501 #endif
502 #if HAVE_DATE_TIME_OFFSET
503  case PrimitiveTypeCode.DateTimeOffset:
504  return ToString((DateTimeOffset)value);
505 #endif
506  case PrimitiveTypeCode.Guid:
507  return ToString((Guid)value);
508  case PrimitiveTypeCode.Uri:
509  return ToString((Uri)value);
510  case PrimitiveTypeCode.TimeSpan:
511  return ToString((TimeSpan)value);
512 #if HAVE_BIG_INTEGER
513  case PrimitiveTypeCode.BigInteger:
514  return ToStringInternal((BigInteger)value);
515 #endif
516  }
517 
518  throw new ArgumentException("Unsupported type: {0}. Use the JsonSerializer class to get the object's JSON representation.".FormatWith(CultureInfo.InvariantCulture, value.GetType()));
519  }

823  public static object? DeserializeObject(string value, Type? type, JsonSerializerSettings? settings)
824  {
825  ValidationUtils.ArgumentNotNull(value, nameof(value));
826 
827  JsonSerializer jsonSerializer = JsonSerializer.CreateDefault(settings);
828 
829  // by default DeserializeObject should check for additional content
830  if (!jsonSerializer.IsCheckAdditionalContentSet())
831  {
832  jsonSerializer.CheckAdditionalContent = true;
833  }
834 
835  using (JsonTextReader reader = new JsonTextReader(new StringReader(value)))
836  {
837  return jsonSerializer.Deserialize(reader, type);
838  }
839  }

863  public static void PopulateObject(string value, object target, JsonSerializerSettings? settings)
864  {
865  JsonSerializer jsonSerializer = JsonSerializer.CreateDefault(settings);
866 
867  using (JsonReader jsonReader = new JsonTextReader(new StringReader(value)))
868  {
869  jsonSerializer.Populate(jsonReader, target);
870 
871  if (settings != null && settings.CheckAdditionalContent)
872  {
873  while (jsonReader.Read())
874  {
875  if (jsonReader.TokenType != JsonToken.Comment)
876  {
877  throw JsonSerializationException.Create(jsonReader, "Additional text found in JSON string after finishing deserializing object.");
878  }
879  }
880  }
