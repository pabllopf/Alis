184  internal static IEnumerable<U> Values<T, U>(this IEnumerable<T> source, object? key) where T : JToken
185  {
186  ValidationUtils.ArgumentNotNull(source, nameof(source));
187 
188  if (key == null)
189  {
190  foreach (T token in source)
191  {
192  if (token is JValue value)
193  {
194  yield return Convert<JValue, U>(value);
195  }
196  else
197  {
198  foreach (JToken t in token.Children())
199  {
200  yield return t.Convert<JToken, U>();
201  }
202  }
203  }
204  }
205  else
206  {
207  foreach (T token in source)
208  {
209  JToken? value = token[key];
210  if (value != null)
211  {
212  yield return value.Convert<JToken, U>();
213  }
214  }
215  }

243  internal static IEnumerable<U> Convert<T, U>(this IEnumerable<T> source) where T : JToken
244  {

253  [return: MaybeNull]
254  internal static U Convert<T, U>(this T token) where T : JToken?
255  {
256  if (token == null)
257  {
258 #pragma warning disable CS8653 // A default expression introduces a null value for a type parameter.
259  return default;
260 #pragma warning restore CS8653 // A default expression introduces a null value for a type parameter.
261  }
262 
263  if (token is U castValue
264  // don't want to cast JValue to its interfaces, want to get the internal value
265  && typeof(U) != typeof(IComparable) && typeof(U) != typeof(IFormattable))
266  {
267  return castValue;
268  }
269  else
270  {
271  if (!(token is JValue value))
272  {
273  throw new InvalidCastException("Cannot cast {0} to {1}.".FormatWith(CultureInfo.InvariantCulture, token.GetType(), typeof(T)));
274  }
275 
276  if (value.Value is U u)
277  {
278  return u;
279  }
280 
281  Type targetType = typeof(U);
282 
283  if (ReflectionUtils.IsNullableType(targetType))
284  {
285  if (value.Value == null)
286  {
287 #pragma warning disable CS8653 // A default expression introduces a null value for a type parameter.
288  return default;
289 #pragma warning restore CS8653 // A default expression introduces a null value for a type parameter.
290  }
291 
292  targetType = Nullable.GetUnderlyingType(targetType);
293  }
294 
295  return (U)System.Convert.ChangeType(value.Value, targetType, CultureInfo.InvariantCulture);
296  }
297  }
