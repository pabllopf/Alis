184  internal static IEnumerable<U> Values<T, U>(this IEnumerable<T> source, object? key) where T : JToken
185  {
186  ValidationUtils.ArgumentNotNull(source, nameof(source));
187 
188  if (key == null)
189  {
190  foreach (T token in source)
191  {
192  if (token is JValue value)
193  {
194  yield return Convert<JValue, U>(value);
195  }
196  else
197  {
198  foreach (JToken t in token.Children())
199  {
200  yield return t.Convert<JToken, U>();
201  }
202  }
203  }
204  }
205  else
206  {
207  foreach (T token in source)
208  {
209  JToken? value = token[key];
210  if (value != null)
211  {
212  yield return value.Convert<JToken, U>();
213  }
214  }
215  }

256  [return: MaybeNull]
257  internal static U Convert<T, U>(this T token) where T : JToken?
258  {
259  if (token == null)
260  {
261 #pragma warning disable CS8653 // A default expression introduces a null value for a type parameter.
262  return default;
263 #pragma warning restore CS8653 // A default expression introduces a null value for a type parameter.
264  }
265 
266  if (token is U castValue
267  // don't want to cast JValue to its interfaces, want to get the internal value
268  && typeof(U) != typeof(IComparable) && typeof(U) != typeof(IFormattable))
269  {
270  return castValue;
271  }
272  else
273  {
274  if (!(token is JValue value))
275  {
276  throw new InvalidCastException("Cannot cast {0} to {1}.".FormatWith(CultureInfo.InvariantCulture, token.GetType(), typeof(T)));
277  }
278 
279  if (value.Value is U u)
280  {
281  return u;
282  }
283 
284  Type targetType = typeof(U);
285 
286  if (ReflectionUtils.IsNullableType(targetType))
287  {
288  if (value.Value == null)
289  {
290 #pragma warning disable CS8653 // A default expression introduces a null value for a type parameter.
291  return default;
292 #pragma warning restore CS8653 // A default expression introduces a null value for a type parameter.
293  }
294 
295  targetType = Nullable.GetUnderlyingType(targetType);
296  }
297 
298  return (U)System.Convert.ChangeType(value.Value, targetType, CultureInfo.InvariantCulture);
299  }
300  }
