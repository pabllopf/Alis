54  public override void WriteJson(JsonWriter writer, object? value, JsonSerializer serializer)
55  {
56  if (value == null)
57  {
58  writer.WriteNull();
59  return;
60  }
61 
62  EnsureReflectionObject(value.GetType());
63  MiscellaneousUtils.Assert(_reflectionObject != null);
64 
65  DefaultContractResolver? resolver = serializer.ContractResolver as DefaultContractResolver;
66 
67  string keyName = (string)_reflectionObject.GetValue(value, KeyPropertyName)!;
68  object? keyValue = _reflectionObject.GetValue(value, ValuePropertyName);
69 
70  Type? keyValueType = keyValue?.GetType();
71 
72  writer.WriteStartObject();
73  writer.WritePropertyName((resolver != null) ? resolver.GetResolvedPropertyName(KeyPropertyName) : KeyPropertyName);
74  writer.WriteValue(keyName);
75  writer.WritePropertyName((resolver != null) ? resolver.GetResolvedPropertyName(TypePropertyName) : TypePropertyName);
76  writer.WriteValue(keyValueType?.FullName);
77 
78  writer.WritePropertyName((resolver != null) ? resolver.GetResolvedPropertyName(ValuePropertyName) : ValuePropertyName);
79 
80  if (keyValueType != null)
81  {
82  if (JsonSerializerInternalWriter.TryConvertToString(keyValue!, keyValueType, out string? valueJson))
83  {
84  writer.WriteValue(valueJson);
85  }
86  else
87  {
88  writer.WriteValue(keyValue);
89  }
90  }
91  else
92  {
93  writer.WriteNull();
94  }
95 
96  writer.WriteEndObject();
97  }

99  private static void ReadAndAssertProperty(JsonReader reader, string propertyName)
100  {
101  reader.ReadAndAssert();
102 
103  if (reader.TokenType != JsonToken.PropertyName || !string.Equals(reader.Value?.ToString(), propertyName, StringComparison.OrdinalIgnoreCase))
104  {
105  throw new JsonSerializationException("Expected JSON property '{0}'.".FormatWith(CultureInfo.InvariantCulture, propertyName));
106  }
107  }
108 
117  public override object? ReadJson(JsonReader reader, Type objectType, object? existingValue, JsonSerializer serializer)
118  {
119  EnsureReflectionObject(objectType);
120  MiscellaneousUtils.Assert(_reflectionObject != null);
121 
122  object entityKeyMember = _reflectionObject.Creator!();
123 
124  ReadAndAssertProperty(reader, KeyPropertyName);
125  reader.ReadAndAssert();
126  _reflectionObject.SetValue(entityKeyMember, KeyPropertyName, reader.Value?.ToString());
127 
128  ReadAndAssertProperty(reader, TypePropertyName);
129  reader.ReadAndAssert();
130  string? type = reader.Value?.ToString();
131 
132  Type t = Type.GetType(type);
133 
134  ReadAndAssertProperty(reader, ValuePropertyName);
135  reader.ReadAndAssert();
136  _reflectionObject.SetValue(entityKeyMember, ValuePropertyName, serializer.Deserialize(reader, t));
137 
138  reader.ReadAndAssert();
139 
140  return entityKeyMember;
141  }
142 
143  private static void EnsureReflectionObject(Type objectType)
144  {
145  if (_reflectionObject == null)
146  {
147  _reflectionObject = ReflectionObject.Create(objectType, KeyPropertyName, ValuePropertyName);
148  }
149  }
150 
158  public override bool CanConvert(Type objectType)
159  {
160  return objectType.AssignableToTypeName(EntityKeyMemberFullTypeName, false);
161  }
