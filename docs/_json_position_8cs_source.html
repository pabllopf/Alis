60  internal int CalculateLength()
61  {
62  switch (Type)
63  {
64  case JsonContainerType.Object:
65  return PropertyName!.Length + 5;
66  case JsonContainerType.Array:
67  case JsonContainerType.Constructor:

74  internal void WriteTo(StringBuilder sb, ref StringWriter? writer, ref char[]? buffer)
75  {
76  switch (Type)
77  {
78  case JsonContainerType.Object:
79  string propertyName = PropertyName!;
80  if (propertyName.IndexOfAny(SpecialCharacters) != -1)
81  {
82  sb.Append(@"['");
83 
84  if (writer == null)
85  {
86  writer = new StringWriter(sb);
87  }
88 
89  JavaScriptUtils.WriteEscapedJavaScriptString(writer, propertyName, '\'', false, JavaScriptUtils.SingleQuoteCharEscapeFlags, StringEscapeHandling.Default, null, ref buffer);
90 
91  sb.Append(@"']");
92  }
93  else
94  {
95  if (sb.Length > 0)
96  {
97  sb.Append('.');
98  }
99 
100  sb.Append(propertyName);
101  }
102  break;
103  case JsonContainerType.Array:
104  case JsonContainerType.Constructor:
105  sb.Append('[');
106  sb.Append(Position);
107  sb.Append(']');
108  break;
109  }
110  }

117  internal static string BuildPath(List<JsonPosition> positions, JsonPosition? currentPosition)
118  {
119  int capacity = 0;
120  if (positions != null)
121  {
122  for (int i = 0; i < positions.Count; i++)
123  {
124  capacity += positions[i].CalculateLength();
125  }
126  }
127  if (currentPosition != null)
128  {
129  capacity += currentPosition.GetValueOrDefault().CalculateLength();
130  }
131 
132  StringBuilder sb = new StringBuilder(capacity);
133  StringWriter? writer = null;
134  char[]? buffer = null;
135  if (positions != null)
136  {
137  foreach (JsonPosition state in positions)
138  {
139  state.WriteTo(sb, ref writer, ref buffer);
140  }
141  }
142  if (currentPosition != null)
143  {
144  currentPosition.GetValueOrDefault().WriteTo(sb, ref writer, ref buffer);
145  }
146 
147  return sb.ToString();
148  }

150  internal static string FormatMessage(IJsonLineInfo? lineInfo, string path, string message)
151  {
152  // don't add a fullstop and space when message ends with a new line
153  if (!message.EndsWith(Environment.NewLine, StringComparison.Ordinal))
154  {
155  message = message.Trim();
156 
157  if (!message.EndsWith('.'))
158  {
159  message += ".";
160  }
161 
162  message += " ";
163  }
164 
165  message += "Path '{0}'".FormatWith(CultureInfo.InvariantCulture, path);
166 
167  if (lineInfo != null && lineInfo.HasLineInfo())
168  {
169  message += ", line {0}, position {1}".FormatWith(CultureInfo.InvariantCulture, lineInfo.LineNumber, lineInfo.LinePosition);
170  }
171 
172  message += ".";
173 
174  return message;
