53  // Loads a wave/riff audio file.
66  public static byte[] LoadWave(Stream stream, out int channels, out int bits, out int rate)
67  {
68  if (stream == null)
69  {
70  throw new ArgumentNullException("stream");
71  }
72 
73  using (BinaryReader reader = new BinaryReader(stream))
74  {
75  // RIFF header
76  string signature = new string(reader.ReadChars(4));
77  if (signature != "RIFF")
78  {
79  throw new NotSupportedException("Specified stream is not a wave file.");
80  }
81 
82  int riff_chunck_size = reader.ReadInt32();
83 
84  string format = new string(reader.ReadChars(4));
85  if (format != "WAVE")
86  {
87  throw new NotSupportedException("Specified stream is not a wave file.");
88  }
89 
90  // WAVE header
91  string format_signature = new string(reader.ReadChars(4));
92  if (format_signature != "fmt ")
93  {
94  throw new NotSupportedException("Specified wave file is not supported. fmt header not found.");
95  }
96 
97  int format_chunk_size = reader.ReadInt32();
98  int audio_format = reader.ReadInt16();
99  int num_channels = reader.ReadInt16();
100  int sample_rate = reader.ReadInt32();
101  int byte_rate = reader.ReadInt32();
102  int block_align = reader.ReadInt16();
103  int bits_per_sample = reader.ReadInt16();
104 
105  string data_signature = new string(reader.ReadChars(4));
106  //if (data_signature != "data")
107  //throw new NotSupportedException("Specified wave file is not supported. data data_signature ");
108 
109  int data_chunk_size = reader.ReadInt32();
110 
111  channels = num_channels;
112  bits = bits_per_sample;
113  rate = sample_rate;
114 
115  return reader.ReadBytes((int) reader.BaseStream.Length);
116  }
117  }

126  public static ALFormat GetSoundFormat(int channels, int bits)
127  {
128  switch (channels)
129  {
130  case 1: return bits == 8 ? ALFormat.Mono8 : ALFormat.Mono16;
131  case 2: return bits == 8 ? ALFormat.Stereo8 : ALFormat.Stereo16;
132  default: throw new NotSupportedException("The specified sound format is not supported.");
133  }
134  }

163  ALC.GetInteger(device, AlcGetInteger.MajorVersion, 1, out int alcMajorVersion);
164  ALC.GetInteger(device, AlcGetInteger.MinorVersion, 1, out int alcMinorVersion);

175  Console.WriteLine(

177 
178

216  } while ((ALSourceState) state == ALSourceState.Playing);
217 
218  Trace.WriteLine("");

222  AL.DeleteBuffer(buffer);
223 
224 
225  ALC.MakeContextCurrent(ALContext.Null);
226  ALC.DestroyContext(context);
227  ALC.CloseDevice(device);
228 
229  Console.WriteLine("Goodbye!");
