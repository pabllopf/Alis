862  public void RemoveFixture(Fixture fixture)
863  {
864  Debug.Assert(!World.IsLocked);
865  if (World.IsLocked)
866  {
867  return;
868  }
869 
870  if (fixture == null)
871  {
872  return;
873  }
874 
875  Debug.Assert(fixture.Body == this);
876 
877  // Remove the fixture from this body's singly linked list.
878  Debug.Assert(FixtureList.Count > 0);
879 
880  // You tried to remove a fixture that not present in the fixturelist.
881  Debug.Assert(FixtureList.Contains(fixture));
882 
883  // Destroy any contacts associated with the fixture.
884  ContactEdge edge = ContactList;
885  while (edge != null)
886  {
887  Contact c = edge.Contact;
888  edge = edge.Next;
889 
890  Fixture fixtureA = c.FixtureA;
891  Fixture fixtureB = c.FixtureB;
892 
893  if (fixture == fixtureA || fixture == fixtureB)
894  {
895  // This destroys the contact and removes it from
896  // this body's contact list.
897  World.ContactManager.Remove(c);
898  }
899  }
900 
901  if ((Flags & BodyFlags.Enabled) == BodyFlags.Enabled)
902  {
903  IBroadPhase broadPhase = World.ContactManager.BroadPhase;
904  fixture.DestroyProxies(broadPhase);
905  }
906 
907  FixtureList.Remove(fixture);
908  fixture.Destroy();
909  fixture.Bodyprivate = null;
910 
911  ResetMassData();
912  }

1119  public void ResetMassData()
1120  {
1121  // Compute mass data from shapes. Each shape has its own density.
1122  mass = 0.0f;
1123  InvMass = 0.0f;
1124  inertia = 0.0f;
1125  InvI = 0.0f;
1126  Sweep.LocalCenter = Vector2.Zero;
1127 
1128  //Velcro: We have mass on static bodies to support attaching joints to them
1129  // Kinematic bodies have zero mass.
1130  if (Type == BodyType.Kinematic)
1131  {
1132  Sweep.C0 = Xf.P;
1133  Sweep.C = Xf.P;
1134  Sweep.A0 = Sweep.A;
1135  return;
1136  }
1137 
1138  Debug.Assert(Type == BodyType.Dynamic || Type == BodyType.Static);
1139 
1140  // Accumulate mass over all fixtures.
1141  Vector2 localCenter = Vector2.Zero;
1142  foreach (Fixture f in FixtureList)
1143  {
1144  if (f.Shape.DensityPrivate == 0.0f)
1145  {
1146  continue;
