339  {
340  return;
341  }
342 
343  // Delete the attached joints.
344  JointEdge jn = null;
345  if (b.JointList != null)
346  {
347  jn = b.JointList;
348  }
349 
350  while (jn != null)
351  {
352  JointEdge jn0 = jn;
353  jn = jn.Next;
354 
355  if (destructionListener != null)
356  {

364  ControllerEdge ce = b.ControllerList;
365  while (ce != null)
366  {
367  ControllerEdge ce0 = ce;
368  ce = ce.NextController;

744  public void Step(float dt, int velocityIterations, int positionIteration)
745  {
746  Lock = true;
747 
748  TimeStep step = new TimeStep();
749  step.Dt = dt;
750  step.VelocityIterations = velocityIterations;
751  step.PositionIterations = positionIteration;
752  if (dt > 0.0f)
753  {
754  step.InvDt = 1.0f / dt;
755  }
756  else
757  {
758  step.InvDt = 0.0f;
759  }
760 
761  step.DtRatio = invDt0 * dt;
762 
763  step.WarmStarting = warmStarting;
764 
765  // Update contacts.
766  contactManager.Collide();
767 
768  // Integrate velocities, solve velocity constraints, and integrate positions.
769  if (step.Dt > 0.0f)
770  {
771  Solve(step);
772  }
773 
774  // Handle TOI events.
775  if (continuousPhysics && step.Dt > 0.0f)
776  {
777  SolveToi(step);
778  }
779 
780  // Draw debug information.
781  DrawDebugData();
782 
783  invDt0 = step.InvDt;
784  Lock = false;
785  }

885  // Find islands, integrate and solve constraints, solve position constraints
890  private void Solve(TimeStep step)
891  {
892  // Step all controlls
893  for (Controller controller = controllerList; controller != null; controller = controller.Next)
894  {
895  controller.Step(step);
896  }
897 
898  // Size the island for the worst case.
899  Island island = new Island(bodyCount, ContactCount, jointCount, ContactListener);
900 
901  // Clear all the island flags.
902  for (Body b = bodyList; b != null; b = b.Next)
903  {
904  b.Flags &= ~BodyFlags.Island;
905  }
906 
907  for (Contact c = ContactList; c != null; c = c.Next)
908  {
909  c.Flags &= ~Contact.CollisionFlags.Island;
910  }
911 
912  for (Joint j = jointList; j != null; j = j.Next)
913  {
914  j.IslandFlag = false;
915  }
916 
917  // Build and simulate all awake islands.
918  int stackSize = bodyCount;
919  {
920  Body[] stack = new Body[stackSize];
921 
922  for (Body seed = bodyList; seed != null; seed = seed.Next)
923  {
924  if ((seed.Flags & (BodyFlags.Island | BodyFlags.Sleep | BodyFlags.Frozen)) != 0)
925  {
926  continue;
927  }
928 
929  if (seed.IsStatic())
930  {
931  continue;
932  }
933 
934  // Reset island and stack.
935  island.Clear();
936  int stackCount = 0;
937  stack[stackCount++] = seed;
938  seed.Flags |= BodyFlags.Island;
939 
940  // Perform a depth first search (DFS) on the constraint graph.
941  while (stackCount > 0)
942  {
943  // Grab the next body off the stack and add it to the island.
944  Body b = stack[--stackCount];
945  island.Add(b);
946 
947  // Make sure the body is awake.
948  b.Flags &= ~BodyFlags.Sleep;
949 
950  // To keep islands as small as possible, we don't
951  // propagate islands across static bodies.
952  if (b.IsStatic())
953  {
954  continue;
955  }
956 
957  // Search all contacts connected to this body.
958  for (ContactEdge cn = b.ContactList; cn != null; cn = cn.Next)
959  {
960  // Has this contact already been added to an island?
961  if ((cn.Contact.Flags &
962  (Contact.CollisionFlags.Island | Contact.CollisionFlags.NonSolid)) != 0)
963  {
964  continue;
965  }
966 
967  // Is this contact touching?
968  if ((cn.Contact.Flags & Contact.CollisionFlags.Touch) == 0)
969  {
970  continue;
971  }
972 
973  island.Add(cn.Contact);
974  cn.Contact.Flags |= Contact.CollisionFlags.Island;
975 
976  Body other = cn.Other;
977 
978  // Was the other body already added to this island?
979  if ((other.Flags & BodyFlags.Island) != 0)
980  {
981  continue;
982  }
983 
984  Box2DxDebug.Assert(stackCount < stackSize);
985  stack[stackCount++] = other;
986  other.Flags |= BodyFlags.Island;
987  }
988 
989  // Search all joints connect to this body.
990  for (JointEdge jn = b.JointList; jn != null; jn = jn.Next)
991  {
992  if (jn.Joint.IslandFlag)
993  {
994  continue;
995  }
996 
997  island.Add(jn.Joint);
998  jn.Joint.IslandFlag = true;
999 
1000  Body other = jn.Other;
1001  if ((other.Flags & BodyFlags.Island) != 0)
1002  {
1003  continue;
1004  }
1005 
1006  Box2DxDebug.Assert(stackCount < stackSize);
1007  stack[stackCount++] = other;
1008  other.Flags |= BodyFlags.Island;
1009  }
1010  }
1011 
1012  island.Solve(step, gravity, allowSleep);
1013 
1014  // Post solve cleanup.
1015  for (int i = 0; i < island.BodyCount; ++i)
1016  {
1017  // Allow static bodies to participate in other islands.
1018  Body b = island.Bodies[i];
1019  if (b.IsStatic())
1020  {
1021  b.Flags &= ~BodyFlags.Island;
1022  }
1023  }
1024  }
1025 
1026  stack = null;
1027  }
1028 
1029  // Synchronize shapes, check for out of range bodies.
1030  for (Body b = bodyList; b != null; b = b.GetNext())
1031  {
1032  if ((b.Flags & (BodyFlags.Sleep | BodyFlags.Frozen)) != 0)
1033  {
1034  continue;
1035  }
1036 
1037  if (b.IsStatic())
1038  {
1039  continue;
1040  }
1041 
1042  // Update shapes (for broad-phase). If the shapes go out of
1043  // the world AABB then shapes and contacts may be destroyed,
1044  // including contacts that are
1045  bool inRange = b.SynchronizeFixtures();
1046 
1047  // Did the body's shapes leave the world?
1048  if (inRange == false && boundaryListener != null)
1049  {
1050  boundaryListener.Violation(b);
1051  }
1052  }
1053 
1054  // Commit shape proxy movements to the broad-phase so that new contacts are created.
1055  // Also, some contacts can be destroyed.
1056  BroadPhase.Commit();
1057  }

1059  // Find TOI contacts and solve them.
1064  private void SolveToi(TimeStep step)
1065  {
1066  // Reserve an island and a queue for TOI island solution.
1067  Island island = new Island(bodyCount, Settings.MaxToiContactsPerIsland, Settings.MaxToiJointsPerIsland,
1068  ContactListener);
1069 
1070  //Simple one pass queue
1071  //Relies on the fact that we're only making one pass
1072  //through and each body can only be pushed/popped once.
1073  //To push:
1074  // queue[queueStart+queueSize++] = newElement;
1075  //To pop:
1076  // poppedElement = queue[queueStart++];
1077  // --queueSize;
1078  int queueCapacity = bodyCount;
1079  Body[] queue = new Body[queueCapacity];
1080 
1081  for (Body b = bodyList; b != null; b = b.Next)
1082  {
1083  b.Flags &= ~BodyFlags.Island;
1084  b.Sweep.T0 = 0.0f;
1085  }
1086 
1087  for (Contact c = ContactList; c != null; c = c.Next)
1088  {
1089  // Invalidate TOI
1090  c.Flags &= ~(Contact.CollisionFlags.Toi | Contact.CollisionFlags.Island);
1091  }
1092 
1093  for (Joint j = jointList; j != null; j = j.Next)
1094  {
1095  j.IslandFlag = false;
1096  }
1097 
1098  // Find TOI events and solve them.
1099  for (;;)
1100  {
1101  // Find the first TOI.
1102  Contact minContact = null;
1103  float minToi = 1.0f;
1104 
1105  for (Contact c = ContactList; c != null; c = c.Next)
1106  {
1107  if ((int) (c.Flags & (Contact.CollisionFlags.Slow | Contact.CollisionFlags.NonSolid)) == 1)
1108  {
1109  continue;
1110  }
1111 
1112  // TODO_ERIN keep a counter on the contact, only respond to M TOIs per contact.
1113 
1114  float toi = 1.0f;
1115  if ((int) (c.Flags & Contact.CollisionFlags.Toi) == 1)
1116  {
1117  // This contact has a valid cached TOI.
1118  toi = c.Toi;
1119  }
1120  else
1121  {
1122  // Compute the TOI for this contact.
1123  Fixture s1 = c.FixtureA;
1124  Fixture s2 = c.FixtureB;
1125  Body b1 = s1.Body;
