130  public static int ClipSegmentToLine(out ClipVertex[ /*2*/] vOut, ClipVertex[ /*2*/] vIn, Vector2 normal,
131  float offset)
132  {
133  vOut = new ClipVertex[2];
134 
135  // Start with no output points
136  int numOut = 0;
137 
138  // Calculate the distance of end points to the line
139  float distance0 = Vector2.Dot(normal, vIn[0].V) - offset;
140  float distance1 = Vector2.Dot(normal, vIn[1].V) - offset;
141 
142  // If the points are behind the plane
143  if (distance0 <= 0.0f)
144  {
145  vOut[numOut++] = vIn[0];
146  }
147 
148  if (distance1 <= 0.0f)
149  {
150  vOut[numOut++] = vIn[1];
151  }
152 
153  // If the points are on different sides of the plane
154  if (distance0 * distance1 < 0.0f)
155  {
156  // Find intersection point of edge and plane
157  float interp = distance0 / (distance0 - distance1);
158  vOut[numOut].V = vIn[0].V + interp * (vIn[1].V - vIn[0].V);
159  if (distance0 > 0.0f)
160  {
161  vOut[numOut].Id = vIn[0].Id;
162  }
163  else
164  {
165  vOut[numOut].Id = vIn[1].Id;
166  }
167 
168  ++numOut;
169  }
170 
171  return numOut;
172  }
