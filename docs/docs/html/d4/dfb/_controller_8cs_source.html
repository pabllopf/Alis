149  public void RemoveBody(Body body)
150  {

152  Box2DxDebug.Assert(BodyCount > 0);
153 
154  //Find the corresponding edge
155  ControllerEdge edge = BodyList;
156  while (edge != null && edge.Body != body)
157  {
158  edge = edge.NextBody;
159  }
160 
161  //Assert that we are removing a body that is currently attached to the controller
162  Box2DxDebug.Assert(edge != null);
163 
164  //Remove edge from controller list
165  if (edge.PrevBody != null)
166  {
167  edge.PrevBody.NextBody = edge.NextBody;
168  }
169 
170  if (edge.NextBody != null)
171  {
172  edge.NextBody.PrevBody = edge.PrevBody;
173  }
174 
175  if (edge == BodyList)
176  {
177  BodyList = edge.NextBody;
178  }
179 
180  --BodyCount;
181 
182  //Remove edge from body list

184  {
185  edge.PrevController.NextController = edge.NextController;
186  }
187

191  }
192
