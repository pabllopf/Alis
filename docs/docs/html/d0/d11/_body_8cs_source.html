465  public void SetMassFromShapes()
466  {
467  Box2DxDebug.Assert(world.Lock == false);
468  if (world.Lock)
469  {
470  return;
471  }
472 
473  // Compute mass data from shapes. Each shape has its own density.
474  Mass = 0.0f;
475  InvMass = 0.0f;
476  I = 0.0f;
477  InvI = 0.0f;
478 
479  Vector2 center = Vector2.Zero;
480  for (Fixture f = FixtureList; f != null; f = f.Next)
481  {
482  MassData massData;
483  f.ComputeMass(out massData);
484  Mass += massData.Mass;
485  center += massData.Mass * massData.Center;
486  I += massData.I;
487  }
488 
489  // Compute center of mass, and shift the origin to the COM.
490  if (Mass > 0.0f)
491  {
492  InvMass = 1.0f / Mass;
493  center *= InvMass;
494  }
495 
496  if (I > 0.0f && (Flags & BodyFlags.FixedRotation) == 0)
497  {
498  // Center the inertia about the center of mass.
499  I -= Mass * Vector2.Dot(center, center);
500  Box2DxDebug.Assert(I > 0.0f);
501  InvI = 1.0f / I;
502  }
503  else
504  {
505  I = 0.0f;
506  InvI = 0.0f;
507  }
508 
509  // Move center of mass.
510  Sweep.LocalCenter = center;
511  Sweep.C0 = Sweep.C = Math.Mul(Xf, Sweep.LocalCenter);
512 
513  BodyType oldType = type;
514  if (InvMass == 0.0f && InvI == 0.0f)
515  {
516  type = BodyType.Static;
517  }
518  else
519  {
520  type = BodyType.Dynamic;
521  }
522 
523  // If the body type changed, we need to refilter the broad-phase proxies.
524  if (oldType != type)
525  {
526  for (Fixture f = FixtureList; f != null; f = f.Next)
527  {
528  f.RefilterProxy(world.BroadPhase, Xf);
529  }
530  }
531  }

546  public bool SetXForm(Vector2 position, float angle)
547  {
548  Box2DxDebug.Assert(world.Lock == false);
549  if (world.Lock)
550  {
551  return true;
552  }
553 
554  if (IsFrozen())
555  {
556  return false;
557  }
558 
559  Xf.R.Set(angle);
560  Xf.Position = position;
561 
562  Sweep.C0 = Sweep.C = Math.Mul(Xf, Sweep.LocalCenter);
563  Sweep.A0 = Sweep.A = angle;
564 
565  bool freeze = false;
566  for (Fixture f = FixtureList; f != null; f = f.Next)
567  {
568  bool inRange = f.Synchronize(world.BroadPhase, Xf, Xf);
569 
570  if (inRange == false)
571  {
572  freeze = true;
573  break;
574  }
575  }
576 
577  if (freeze)
578  {
579  Flags |= BodyFlags.Frozen;
580  LinearVelocity.SetZero();
581  AngularVelocity = 0.0f;
582 
583  // Failure
584  return false;
585  }
586 
587  // Success
588  world.BroadPhase.Commit();
589  return true;
590  }
