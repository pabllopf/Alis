171  internal override void InitVelocityConstraints(TimeStep step)
172  {
173  Body body2 = Body2;
174 
175  float body2Mass = body2.GetMass();
176 
177  // Frequency
178  float omega = 2.0f * Settings.Pi * FrequencyHz;
179 
180  // Damping coefficient
181  float coefficient = 2.0f * body2Mass * DampingRatio * omega;
182 
183  // Spring stiffness
184  float stiffness = body2Mass * (omega * omega);
185 
186  // magic formulas
187  // gamma has units of inverse mass.
188  // beta has units of inverse time.
189  Box2DxDebug.Assert(coefficient + step.Dt * stiffness > Settings.FltEpsilon);
190  Gamma = 1.0f / (step.Dt * (coefficient + step.Dt * stiffness));
191  Beta = step.Dt * stiffness * Gamma;
192 
193  // Compute the effective mass matrix.
194  Vector2 effectiveMass = Math.Mul(body2.GetXForm().R, LocalAnchor - body2.GetLocalCenter());
195 
196  // K = [(1/m1 + 1/m2) * eye(2) - skew(r1) * invI1 * skew(r1) - skew(r2) * invI2 * skew(r2)]
197  // = [1/m1+1/m2 0 ] + invI1 * [r1.y*r1.y -r1.x*r1.y] + invI2 * [r1.y*r1.y -r1.x*r1.y]
198  // [ 0 1/m1+1/m2] [-r1.x*r1.y r1.x*r1.x] [-r1.x*r1.y r1.x*r1.x]
199  float invMass = body2.InvMass;
200  float invI = body2.InvI;
201 
202  Matrix22 k1 = new Matrix22
203  {
204  Col1 = new Vector2(invMass, 0.0f),
205  Col2 = new Vector2(0.0f, invMass)
206  };
207 
208  Matrix22 k2 = new Matrix22
209  {
210  Col1 = new Vector2(invI * effectiveMass.Y * effectiveMass.Y, -invI * effectiveMass.X * effectiveMass.Y),
211  Col2 = new Vector2(-invI * effectiveMass.X * effectiveMass.Y, invI * effectiveMass.X * effectiveMass.X)
212  };
213 
214  Matrix22 k = k1 + k2;
215  k.Col1.X += Gamma;
216  k.Col2.Y += Gamma;
217 
218  Mass = k.GetInverse();
219 
220  C = body2.Sweep.C + effectiveMass - Target;
221 
222  // Cheat with some damping
223  body2.AngularVelocity *= 0.98f;
224 
225  // Warm starting.
226  Impulse *= step.DtRatio;
227  body2.LinearVelocity += invMass * Impulse;
228  body2.AngularVelocity += invI * Vector2.Cross(effectiveMass, Impulse);
229  }

235  internal override void SolveVelocityConstraints(TimeStep step)
236  {
237  Body b = Body2;
238 
239  Vector2 r = Math.Mul(b.GetXForm().R, LocalAnchor - b.GetLocalCenter());
240 
241  // Cdot = v + cross(w, r)
242  Vector2 cdot = b.LinearVelocity + Vector2.Cross(b.AngularVelocity, r);
243  Vector2 impulse = Math.Mul(Mass, -(cdot + Beta * C + Gamma * Impulse));
244 
245  Vector2 oldImpulse = Impulse;
246  Impulse += impulse;
247  float maxImpulse = step.Dt * MaxForce;
248  if (Impulse.LengthSquared() > maxImpulse * maxImpulse)
249  {
250  Impulse *= maxImpulse / Impulse.Length();
251  }
252 
253  impulse = Impulse - oldImpulse;
254 
255  b.LinearVelocity += b.InvMass * impulse;
256  b.AngularVelocity += b.InvI * Vector2.Cross(r, impulse);
257  }
