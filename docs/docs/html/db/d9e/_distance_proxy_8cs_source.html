56  public DistanceProxy(Shape shape, int index)
57  {
58  switch (shape.ShapeType)
59  {
60  case ShapeType.Circle:
61  {
62  CircleShape circle = (CircleShape) shape;
63  Vertices = new Vector2[1];
64  Vertices[0] = circle.Positionprivate;
65  Radius = circle.RadiusPrivate;
66  }
67  break;
68 
69  case ShapeType.Polygon:
70  {
71  PolygonShape polygon = (PolygonShape) shape;
72  Vertices = new Vector2[polygon.VerticesPrivate.Count];
73 
74  for (int i = 0; i < polygon.VerticesPrivate.Count; i++)
75  {
76  Vertices[i] = polygon.VerticesPrivate[i];
77  }
78 
79  Radius = polygon.RadiusPrivate;
80  }
81  break;
82 
83  case ShapeType.Chain:
84  {
85  ChainShape chain = (ChainShape) shape;
86  Debug.Assert(0 <= index && index < chain.Vertices.Count);
87 
88  Vertices = new Vector2[2];
89  Vertices[0] = chain.Vertices[index];
90  Vertices[1] = index + 1 < chain.Vertices.Count ? chain.Vertices[index + 1] : chain.Vertices[0];
91 
92  Radius = chain.RadiusPrivate;
93  }
94  break;
95 
96  case ShapeType.Edge:
97  {
98  EdgeShape edge = (EdgeShape) shape;
99  Vertices = new Vector2[2];
100  Vertices[0] = edge.Vertex1;
101  Vertices[1] = edge.Vertex2;
102  Radius = edge.RadiusPrivate;
103  }
104  break;
105 
106  default:
107  throw new NotSupportedException();
108  }
109  }

124  public int GetSupport(Vector2 direction)
125  {
126  int bestIndex = 0;
127  float bestValue = Vector2.Dot(Vertices[0], direction);
128  for (int i = 1; i < Vertices.Length; ++i)
129  {
130  float value = Vector2.Dot(Vertices[i], direction);
131  if (value > bestValue)
132  {
133  bestIndex = i;
134  bestValue = value;
135  }
136  }
137 
138  return bestIndex;
139  }
