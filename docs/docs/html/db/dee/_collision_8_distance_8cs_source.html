50  public static unsafe void Distance(out DistanceOutput output, ref SimplexCache cache, ref DistanceInput input,
51  Shape shapeA, Shape shapeB)
52  {
53  output = new DistanceOutput();
54 
55  XForm transformA = input.TransformA;
56  XForm transformB = input.TransformB;
57 
58  // Initialize the simplex.
59  Simplex simplex = new Simplex();
60  fixed (SimplexCache* sPtr = &cache)
61  {
62  simplex.ReadCache(sPtr, shapeA, transformA, shapeB, transformB);
63  }
64 
65  // Get simplex vertices as an array.
66  SimplexVertex* vertices = &simplex.V1;
67 
68  // These store the vertices of the last simplex so that we
69  // can check for duplicates and prevent cycling.
70  int* lastA = stackalloc int[4], lastB = stackalloc int[4];
71  int lastCount;
72 
73  // Main iteration loop.
74  int iter = 0;
75  const int kMaxIterationCount = 20;
76  while (iter < kMaxIterationCount)
77  {
78  // Copy simplex so we can identify duplicates.
79  lastCount = simplex.Count;
80  int i;
81  for (i = 0; i < lastCount; ++i)
82  {
83  lastA[i] = vertices[i].IndexA;
84  lastB[i] = vertices[i].IndexB;
85  }
86 
87  switch (simplex.Count)
88  {
89  case 1:
90  break;
91 
92  case 2:
93  simplex.Solve2();
94  break;
95 
96  case 3:
97  simplex.Solve3();
98  break;
99 
100  default:
101 #if DEBUG
102  Box2DxDebug.Assert(false);
103 #endif
104  break;
105  }
106 
107  // If we have 3 points, then the origin is in the corresponding triangle.
108  if (simplex.Count == 3)
109  {
110  break;
111  }
112 
113  // Compute closest point.
114  Vector2 p = simplex.GetClosestPoint();
115  float distanceSqr = p.LengthSquared();
116 
117  // Ensure the search direction is numerically fit.
118  if (distanceSqr < Settings.FltEpsilonSquared)
119  {
120  // The origin is probably contained by a line segment
121  // or triangle. Thus the shapes are overlapped.
122 
123  // We can't return zero here even though there may be overlap.
124  // In case the simplex is a point, segment, or triangle it is difficult
125  // to determine if the origin is contained in the CSO or very close to it.
126  break;
127  }
128 
129  // Compute a tentative new simplex vertex using support points.
130  SimplexVertex* vertex = vertices + simplex.Count;
131  vertex->IndexA = shapeA.GetSupport(Math.MulT(transformA.R, p));
132  vertex->Wa = Math.Mul(transformA, shapeA.GetVertex(vertex->IndexA));
133  //Vec2 wBLocal;
134  vertex->IndexB = shapeB.GetSupport(Math.MulT(transformB.R, -p));
135  vertex->Wb = Math.Mul(transformB, shapeB.GetVertex(vertex->IndexB));
136  vertex->W = vertex->Wb - vertex->Wa;
137 
138  // Iteration count is equated to the number of support point calls.
139  ++iter;
140 
141  // Check for convergence.
142  float lowerBound = Vector2.Dot(p, vertex->W);
143  float upperBound = distanceSqr;
144  const float kRelativeTolSqr = 0.01f * 0.01f; // 1:100
145  if (upperBound - lowerBound <= kRelativeTolSqr * upperBound)
146  {
147  // Converged!
148  break;
149  }
150 
151  // Check for duplicate support points.
152  bool duplicate = false;
153  for (i = 0; i < lastCount; ++i)
154  {
155  if (vertex->IndexA == lastA[i] && vertex->IndexB == lastB[i])
156  {
157  duplicate = true;
158  break;
159  }
160  }
161 
162  // If we found a duplicate support point we must exit to avoid cycling.
163  if (duplicate)
164  {
165  break;
166  }
167 
168  // New vertex is ok and needed.
169  ++simplex.Count;
170  }
171 
172 
173  fixed (DistanceOutput* doPtr = &output)
174  {
175  // Prepare output.
176  simplex.GetWitnessPoints(&doPtr->PointA, &doPtr->PointB);
177  doPtr->Distance = Vector2.Distance(doPtr->PointA, doPtr->PointB);
178  doPtr->Iterations = iter;
179  }
180 
181  fixed (SimplexCache* sPtr = &cache)
182  {
183  // Cache the simplex.
184  simplex.WriteCache(sPtr);
185  }
186 
187  // Apply radii if requested.
188  if (input.UseRadii)
189  {
190  float rA = shapeA.Radius;
191  float rB = shapeB.Radius;
192 
193  if (output.Distance > rA + rB && output.Distance > Settings.FltEpsilon)
194  {
195  // Shapes are still no overlapped.
196  // Move the witness points to the outer surface.
197  output.Distance -= rA + rB;
198  Vector2 normal = output.PointB - output.PointA;
199  normal.Normalize();
200  output.PointA += rA * normal;
201  output.PointB -= rB * normal;
202  }
203  else
204  {
205  // Shapes are overlapped when radii are considered.
206  // Move the witness points to the middle.
207  Vector2 p = 0.5f * (output.PointA + output.PointB);
208  output.PointA = p;
209  output.PointB = p;
210  output.Distance = 0.0f;
211  }
212  }
213  }
214  }
215 }
