30 // Point-to-point constraint
31 // C = p2 - p1
32 // Cdot = v2 - v1
33 // = v2 + cross(w2, r2) - v1 - cross(w1, r1)
34 // J = [-I -r1_skew I r2_skew ]
35 // Identity used:
36 // w k % (rx i + ry j) = w * (-ry i + rx j)
37 
38 // Motor constraint
39 // Cdot = w2 - w1
40 // J = [0 0 -1 0 0 1]
41 // K = invI1 + invI2

273  internal override void InitVelocityConstraints(TimeStep step)
274  {
275  Body body1 = Body1;
276  Body body2 = Body2;
277 
278  if (IsMotorEnabled || IsLimitEnabled)
279  {
280  // You cannot create a rotation limit between bodies that
281  // both have fixed rotation.
282  Box2DxDebug.Assert(body1.InvI > 0.0f || body2.InvI > 0.0f);
283  }
284 
285  // Compute the effective mass matrix.
286  Vector2 mulR1 = Box2DXMath.Mul(body1.GetXForm().R, LocalAnchor1 - body1.GetLocalCenter());
287  Vector2 mulR2 = Box2DXMath.Mul(body2.GetXForm().R, LocalAnchor2 - body2.GetLocalCenter());
288 
289  // J = [-I -r1_skew I r2_skew]
290  // [ 0 -1 0 1]
291  // r_skew = [-ry; rx]
292 
293  // Matlab
294  // K = [ m1+r1y^2*i1+m2+r2y^2*i2, -r1y*i1*r1x-r2y*i2*r2x, -r1y*i1-r2y*i2]
295  // [ -r1y*i1*r1x-r2y*i2*r2x, m1+r1x^2*i1+m2+r2x^2*i2, r1x*i1+r2x*i2]
296  // [ -r1y*i1-r2y*i2, r1x*i1+r2x*i2, i1+i2]
297 
298  float m1 = body1.InvMass, m2 = body2.InvMass;
299  float i1 = body1.InvI, i2 = body2.InvI;
300 
301 
302  float col1X = m1 + m2 + mulR1.Y * mulR1.Y * i1 + mulR2.Y * mulR2.Y * i2;
303  float col2X = -mulR1.Y * mulR1.X * i1 - mulR2.Y * mulR2.X * i2;
304  float col3X = -mulR1.Y * i1 - mulR2.Y * i2;
305 
306  float col1Y = Mass.Col2.X;
307  float col2Y = m1 + m2 + mulR1.X * mulR1.X * i1 + mulR2.X * mulR2.X * i2;
308  float col3Y = mulR1.X * i1 + mulR2.X * i2;
309 
310  float col1Z = Mass.Col3.X;
311  float col2Z = Mass.Col3.Y;
312  float col3Z = i1 + i2;
313 
314  Mass = new Matrix33(new Vector3(col1X, col1Y, col1Z), new Vector3(col2X, col2Y, col2Z),
315  new Vector3(col3X, col3Y, col3Z));
316 
317  /*
318  _mass.Col1.X = m1 + m2 + r1.Y * r1.Y * i1 + r2.Y * r2.Y * i2;
319  _mass.Col2.X = -r1.Y * r1.X * i1 - r2.Y * r2.X * i2;
320  _mass.Col3.X = -r1.Y * i1 - r2.Y * i2;
321  _mass.Col1.Y = _mass.Col2.X;
322  _mass.Col2.Y = m1 + m2 + r1.X * r1.X * i1 + r2.X * r2.X * i2;
323  _mass.Col3.Y = r1.X * i1 + r2.X * i2;
324  _mass.Col1.Z = _mass.Col3.X;
325  _mass.Col2.Z = _mass.Col3.Y;
326  _mass.Col3.Z = i1 + i2;
327 */
328  MotorMass = 1.0f / (i1 + i2);
329 
330  if (IsMotorEnabled == false)
331  {
332  MotorTorque = 0.0f;
333  }
334 
335  if (IsLimitEnabled)
336  {
337  float jointAngle = body2.Sweep.A - body1.Sweep.A - ReferenceAngle;
338  if (Box2DXMath.Abs(UpperLimit - LowerLimit) < 2.0f * Settings.AngularSlop)
339  {
340  State = LimitState.EqualLimits;
341  }
342  else if (jointAngle <= LowerLimit)
343  {
344  if (State != LimitState.AtLowerLimit)
345  {
346  Impulse = new Vector3(Impulse.X, Impulse.Y, 0.0f);
347  }
348 
349  State = LimitState.AtLowerLimit;
350  }
351  else if (jointAngle >= UpperLimit)
352  {
353  if (State != LimitState.AtUpperLimit)
354  {
355  Impulse = new Vector3(Impulse.X, Impulse.Y, 0.0f);
356  }
357 
358  State = LimitState.AtUpperLimit;
359  }
360  else
361  {
362  State = LimitState.InactiveLimit;
363  Impulse = new Vector3(Impulse.X, Impulse.Y, 0.0f);
364  }
365  }
366  else
367  {
368  State = LimitState.InactiveLimit;
369  }
370 
371  if (step.WarmStarting)
372  {
373  // Scale impulses to support a variable time step.
374  Impulse *= step.DtRatio;
375  MotorTorque *= step.DtRatio;
376 
377  Vector2 p = new Vector2(Impulse.X, Impulse.Y);
378 
379  body1.LinearVelocity -= m1 * p;
380  body1.AngularVelocity -= i1 * (Vector2.Cross(mulR1, p) + MotorTorque + Impulse.Z);
381 
382  body2.LinearVelocity += m2 * p;
383  body2.AngularVelocity += i2 * (Vector2.Cross(mulR2, p) + MotorTorque + Impulse.Z);
384  }
385  else
386  {
387  Impulse.SetZero();
388  MotorTorque = 0.0f;
389  }
390  }

396  internal override void SolveVelocityConstraints(TimeStep step)
397  {
398  Body b1 = Body1;
399  Body b2 = Body2;
400 
401  Vector2 v1 = b1.LinearVelocity;
402  float w1 = b1.AngularVelocity;
403  Vector2 v2 = b2.LinearVelocity;
404  float w2 = b2.AngularVelocity;
405 
406  float m1 = b1.InvMass, m2 = b2.InvMass;
407  float i1 = b1.InvI, i2 = b2.InvI;
408 
409  //Solve motor constraint.
410  if (IsMotorEnabled && State != LimitState.EqualLimits)
411  {
412  float cdot = w2 - w1 - MotorSpeed;
413  float impulse = MotorMass * -cdot;
414  float oldImpulse = MotorTorque;
415  float maxImpulse = step.Dt * MaxMotorTorque;
416  MotorTorque = Box2DXMath.Clamp(MotorTorque + impulse, -maxImpulse, maxImpulse);
417  impulse = MotorTorque - oldImpulse;
418 
419  w1 -= i1 * impulse;
420  w2 += i2 * impulse;
421  }
422 
423  //Solve limit constraint.
424  if (IsLimitEnabled && State != LimitState.InactiveLimit)
425  {
426  Vector2 r1 = Box2DXMath.Mul(b1.GetXForm().R, LocalAnchor1 - b1.GetLocalCenter());
427  Vector2 r2 = Box2DXMath.Mul(b2.GetXForm().R, LocalAnchor2 - b2.GetLocalCenter());
428 
429  // Solve point-to-point constraint
430  Vector2 cdot1 = v2 + Vector2.Cross(w2, r2) - v1 - Vector2.Cross(w1, r1);
431  float cdot2 = w2 - w1;
432  Vector3 cdot = new Vector3(cdot1.X, cdot1.Y, cdot2);
433 
434  Vector3 impulse = Mass.Solve33(-cdot);
435 
436  if (State == LimitState.EqualLimits)
437  {
438  Impulse += impulse;
439  }
440  else if (State == LimitState.AtLowerLimit)
441  {
442  float newImpulse = Impulse.Z + impulse.Z;
443  if (newImpulse < 0.0f)
444  {
445  Vector2 reduced = Mass.Solve22(-cdot1);
446  impulse.X = reduced.X;
447  impulse.Y = reduced.Y;
448  impulse.Z = -Impulse.Z;
449  Impulse = new Vector3(reduced.X, reduced.Y, 0.0f);
450  }
451  }
452  else if (State == LimitState.AtUpperLimit)
453  {
454  float newImpulse = Impulse.Z + impulse.Z;
455  if (newImpulse > 0.0f)
456  {
457  Vector2 reduced = Mass.Solve22(-cdot1);
458  impulse.X = reduced.X;
459  impulse.Y = reduced.Y;
460  impulse.Z = -Impulse.Z;
461  Impulse = new Vector3(reduced.X, reduced.Y, 0.0f);
462  }
463  }
464 
465  Vector2 p = new Vector2(impulse.X, impulse.Y);
466 
467  v1 -= m1 * p;
468  w1 -= i1 * (Vector2.Cross(r1, p) + impulse.Z);
469 
470  v2 += m2 * p;
471  w2 += i2 * (Vector2.Cross(r2, p) + impulse.Z);
472  }
473  else
474  {
475  Vector2 r1 = Box2DXMath.Mul(b1.GetXForm().R, LocalAnchor1 - b1.GetLocalCenter());
476  Vector2 r2 = Box2DXMath.Mul(b2.GetXForm().R, LocalAnchor2 - b2.GetLocalCenter());
477 
478  // Solve point-to-point constraint
479  Vector2 cdot = v2 + Vector2.Cross(w2, r2) - v1 - Vector2.Cross(w1, r1);
480  Vector2 impulse = Mass.Solve22(-cdot);
481 
482  Impulse = new Vector3(impulse.X, impulse.Y, Impulse.Z);
483 
484  v1 -= m1 * impulse;
485  w1 -= i1 * Vector2.Cross(r1, impulse);
486 
487  v2 += m2 * impulse;
488  w2 += i2 * Vector2.Cross(r2, impulse);
489  }
490 
491  b1.LinearVelocity = v1;
492  b1.AngularVelocity = w1;
493  b2.LinearVelocity = v2;
494  b2.AngularVelocity = w2;
495  }

502  internal override bool SolvePositionConstraints(float baumgarte)
503  {
504  // TODO_ERIN block solve with limit.
505 
506  Body body1 = Body1;
507  Body body2 = Body2;
508 
509  float angularError = 0.0f;
510  float positionError = 0.0f;
511 
512  // Solve angular limit constraint.
513  if (IsLimitEnabled && State != LimitState.InactiveLimit)
514  {
515  float angle = body2.Sweep.A - body1.Sweep.A - ReferenceAngle;
516  float limitImpulse = 0.0f;
517 
518  if (State == LimitState.EqualLimits)
519  {
520  // Prevent large angular corrections
521  float c = Box2DXMath.Clamp(angle, -Settings.MaxAngularCorrection, Settings.MaxAngularCorrection);
522  limitImpulse = -MotorMass * c;
523  angularError = Box2DXMath.Abs(c);
524  }
525  else if (State == LimitState.AtLowerLimit)
526  {
527  float c = angle - LowerLimit;
528  angularError = -c;
529 
530  // Prevent large angular corrections and allow some slop.
531  c = Box2DXMath.Clamp(c + Settings.AngularSlop, -Settings.MaxAngularCorrection, 0.0f);
532  limitImpulse = -MotorMass * c;
533  }
534  else if (State == LimitState.AtUpperLimit)
535  {
536  float c = angle - UpperLimit;
537  angularError = c;
538 
539  // Prevent large angular corrections and allow some slop.
540  c = Box2DXMath.Clamp(c - Settings.AngularSlop, 0.0f, Settings.MaxAngularCorrection);
541  limitImpulse = -MotorMass * c;
542  }
543 
544  body1.Sweep.A -= body1.InvI * limitImpulse;
545  body2.Sweep.A += body2.InvI * limitImpulse;
546 
547  body1.SynchronizeTransform();
548  body2.SynchronizeTransform();
549  }
550 
551  // Solve point-to-point constraint.
552  {
553  Vector2 mulR1 = Box2DXMath.Mul(body1.GetXForm().R, LocalAnchor1 - body1.GetLocalCenter());
554  Vector2 mulR2 = Box2DXMath.Mul(body2.GetXForm().R, LocalAnchor2 - body2.GetLocalCenter());
555 
556  Vector2 body2SweepC = body2.Sweep.C + mulR2 - body1.Sweep.C - mulR1;
557  positionError = body2SweepC.Length();
558 
559  float invMass1 = body1.InvMass, invMass2 = body2.InvMass;
560  float invI1 = body1.InvI, invI2 = body2.InvI;
561 
562  // Handle large detachment.
563  float kAllowedStretch = 10.0f * Settings.LinearSlop;
564  if (body2SweepC.LengthSquared() > kAllowedStretch * kAllowedStretch)
565  {
566  // Use a particle solution (no rotation).
567  Vector2 sweepC = body2SweepC;
568  sweepC.Normalize();
569  float mass12 = invMass1 + invMass2;
570  Box2DxDebug.Assert(mass12 > Settings.FltEpsilon);
571  float divideMass12 = 1.0f / mass12;
572  Vector2 impulseLocal = divideMass12 * -body2SweepC;
573  float kBeta = 0.5f;
574  body1.Sweep.C -= kBeta * invMass1 * impulseLocal;
575  body2.Sweep.C += kBeta * invMass2 * impulseLocal;
576 
577  body2SweepC = body2.Sweep.C + mulR2 - body1.Sweep.C - mulR1;
578  }
579 
580  Matrix22 k1 = new Matrix22
581  {
582  Col1 = new Vector2(invMass1 + invMass2, 0.0f),
583  Col2 = new Vector2(0.0f, invMass1 + invMass2)
584  };
585 
586  Matrix22 k2 = new Matrix22();
587  k2.Col1.X = invI1 * mulR1.Y * mulR1.Y;
588  k2.Col2.X = -invI1 * mulR1.X * mulR1.Y;
589  k2.Col1.Y = -invI1 * mulR1.X * mulR1.Y;
590  k2.Col2.Y = invI1 * mulR1.X * mulR1.X;
591 
592  Matrix22 k3 = new Matrix22();
593  k3.Col1.X = invI2 * mulR2.Y * mulR2.Y;
594  k3.Col2.X = -invI2 * mulR2.X * mulR2.Y;
595  k3.Col1.Y = -invI2 * mulR2.X * mulR2.Y;
596  k3.Col2.Y = invI2 * mulR2.X * mulR2.X;
597 
598  Matrix22 k = k1 + k2 + k3;
599  Vector2 impulse = k.Solve(-body2SweepC);
600 
601  body1.Sweep.C -= body1.InvMass * impulse;
602  body1.Sweep.A -= body1.InvI * Vector2.Cross(mulR1, impulse);
603 
604  body2.Sweep.C += body2.InvMass * impulse;
605  body2.Sweep.A += body2.InvI * Vector2.Cross(mulR2, impulse);
606 
607  body1.SynchronizeTransform();
608  body2.SynchronizeTransform();
609  }
610 
611  return positionError <= Settings.LinearSlop && angularError <= Settings.AngularSlop;
612  }
613  }
