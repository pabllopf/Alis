72  // Collision Detection in Interactive 3D Environments by Gino van den Bergen
73  // From Section 3.1.2
74  // x = s + a * r
75  // norm(x) = radius
85  public override SegmentCollide TestSegment(XForm transform, out float lambda, out Vector2 normal, Segment segment,
86  float maxLambda)
87  {
88  lambda = 0f;
89  normal = Vector2.Zero;
90 
91  Vector2 position = transform.Position + Math.Mul(transform.R, Position);
92  Vector2 s = segment.P1 - position;
93  float b = Vector2.Dot(s, s) - Radius * Radius;
94 
95  // Does the segment start inside the circle?
96  if (b < 0.0f)
97  {
98  lambda = 0f;
99  return SegmentCollide.StartInsideCollide;
100  }
101 
102  // Solve quadratic equation.
103  Vector2 r = segment.P2 - segment.P1;
104  float c = Vector2.Dot(s, r);
105  float rr = Vector2.Dot(r, r);
106  float sigma = c * c - rr * b;
107 
108  // Check for negative discriminant and short segment.
109  if (sigma < 0.0f || rr < Settings.FltEpsilon)
110  {
111  return SegmentCollide.MissCollide;
112  }
113 
114  // Find the point of intersection of the line with the circle.
115  float a = -(c + Math.Sqrt(sigma));
116 
117  // Is the intersection point on the segment?
118  if (0.0f <= a && a <= maxLambda * rr)
119  {
120  a /= rr;
121  lambda = a;
122  normal = s + a * r;
123  normal.Normalize();
124  return SegmentCollide.HitCollide;
125  }
126 
127  return SegmentCollide.MissCollide;
128  }

168  public override float ComputeSubmergedArea(Vector2 normal, float offset, XForm xf, out Vector2 c)
169  {
170  Vector2 p = Math.Mul(xf, Position);
171  float l = -(Vector2.Dot(normal, p) - offset);
172  if (l < -Radius + Settings.FltEpsilon)
173  {
174  //Completely dry
175  c = new Vector2();
176  return 0;
177  }
178 
179  if (l > Radius)
180  {
181  //Completely wet
182  c = p;
183  return Settings.Pi * Radius * Radius;
184  }
185 
186  //Magic
187  float r2 = Radius * Radius;
188  float l2 = l * l;
189  float area = r2 * ((float) System.Math.Asin(l / Radius) + Settings.Pi / 2) +
190  l * Math.Sqrt(r2 - l2);
191  float com = -2.0f / 3.0f * (float) System.Math.Pow(r2 - l2, 1.5f) / area;
192 
193  c.X = p.X + normal.X * com;
194  c.Y = p.Y + normal.Y * com;
195 
196  return area;
197  }
