39  // Collision Detection in Interactive 3D Environments by Gino van den Bergen
40  // From Section 3.4.1
41  // x = mu1 * p1 + mu2 * p2
42  // mu1 + mu2 = 1 && mu1 >= 0 && mu2 >= 0
43  // mu1 = 1 - mu2;
44  // x = (1 - mu2) * p1 + mu2 * p2
45  // = p1 + mu2 * (p2 - p1)
46  // x = s + a * r (s := start, r := end - start)
47  // s + a * r = p1 + mu2 * d (d := p2 - p1)
48  // -a * r + mu2 * d = b (b := s - p1)
49  // [-r d] * [a; mu2] = b
50  // Cramer's rule:
51  // denom = det[-r d]
52  // a = det[b d] / denom
53  // mu2 = det[-r b] / denom
57  public bool TestSegment(out float lambda, out Vector2 normal, Segment segment, float maxLambda)
58  {
59  lambda = 0f;
60  normal = new Vector2();
61 
62  Vector2 s = segment.P1;
63  Vector2 r = segment.P2 - s;
64  Vector2 d = P2 - P1;
65  Vector2 n = Vector2.Cross(d, 1.0f);
66 
67  float kSlop = 100.0f * Settings.FltEpsilon;
68  float denom = -Vector2.Dot(r, n);
69 
70  // Cull back facing collision and ignore parallel segments.
71  if (denom > kSlop)
72  {
73  // Does the segment intersect the infinite line associated with this segment?
74  Vector2 b = s - P1;
75  float a = Vector2.Dot(b, n);
76 
77  if (0.0f <= a && a <= maxLambda * denom)
78  {
79  float mu2 = -r.X * b.Y + r.Y * b.X;
80 
81  // Does the segment intersect this segment?
82  if (-kSlop * denom <= mu2 && mu2 <= denom * (1.0f + kSlop))
83  {
84  a /= denom;
85  n.Normalize();
86  lambda = a;
87  normal = n;
88  return true;
89  }
90  }
91  }
92 
93  return false;
94  }
