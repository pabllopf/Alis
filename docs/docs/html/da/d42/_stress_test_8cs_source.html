108  public async Task Run()
109  {
110  // NOTE: if the service is so busy that it cannot respond to a PING within the KeepAliveInterval interval the websocket connection will be closed
111  // To run extreme tests it is best to set the KeepAliveInterval to TimeSpan.Zero to disable ping pong
112  WebSocketClientOptions options = new WebSocketClientOptions
113  {
114  NoDelay = true, KeepAliveInterval = TimeSpan.FromSeconds(2), SecWebSocketProtocol = "chatV2, chatV1"
115  };
116  using (_webSocket = await _clientFactory.ConnectAsync(_uri, options))
117  {
118  CancellationTokenSource source = new CancellationTokenSource();
119  _token = source.Token;
120 
121  Random rand = new Random(_seed);
122  _expectedValues = new byte[50][];
123  for (int i = 0; i < _expectedValues.Length; i++)
124  {
125  int numBytes = rand.Next(_minNumBytesPerMessage, _maxNumBytesPerMessage);
126  byte[] bytes = new byte[numBytes];
127  rand.NextBytes(bytes);
128  _expectedValues[i] = bytes;
129  }
130 
131  Task recTask = Task.Run(ReceiveLoop);
132  byte[] sendBuffer = new byte[_maxNumBytesPerMessage];
133  for (int i = 0; i < _numItems; i++)
134  {
135  int index = i % _expectedValues.Length;
136  byte[] bytes = _expectedValues[index];
137  Buffer.BlockCopy(bytes, 0, sendBuffer, 0, bytes.Length);
138  ArraySegment<byte> buffer = new ArraySegment<byte>(sendBuffer, 0, bytes.Length);
139  await _webSocket.SendAsync(buffer, WebSocketMessageType.Binary, true, source.Token);
140  }
141 
142  await _webSocket.CloseAsync(WebSocketCloseStatus.NormalClosure, null, source.Token);
143  recTask.Wait();
144  }
145  }

154  private static bool AreEqual(byte[] actual, byte[] expected, int countActual)
155  {
156  if (countActual != expected.Length)
157  {
158  return false;
159  }
160 
161  for (int i = 0; i < countActual; i++)
162  {
163  if (actual[i] != expected[i])
164  {
165  return false;
166  }
167  }
168 
169  return true;
170  }

177  private async Task ReceiveLoop()
178  {
179  // the recArray should be large enough to at least receive control frames like Ping and Close frames (with payload)
180  const int minBufferSize = 510;
181  int size = _maxNumBytesPerMessage < minBufferSize ? minBufferSize : _maxNumBytesPerMessage;
182  byte[] recArray = new byte[size];
183  ArraySegment<byte> recBuffer = new ArraySegment<byte>(recArray);
184 
185  int i = 0;
186  while (true)
187  {
188  WebSocketReceiveResult result = await _webSocket.ReceiveAsync(recBuffer, _token);
189 
190  if (!result.EndOfMessage)
191  {
192  throw new Exception("Multi frame messages not supported");
193  }
194 
195  if (result.MessageType == WebSocketMessageType.Close || _token.IsCancellationRequested)
196  {
197  return;
198  }
199 
200  if (result.Count == 0)
201  {
202  await _webSocket.CloseOutputAsync(WebSocketCloseStatus.InvalidPayloadData, "Zero bytes in payload",
203  _token);
204  return;
205  }
206 
207  byte[] valueActual = recBuffer.Array;
208  int index = i % _expectedValues.Length;
209  i++;
210  byte[] valueExpected = _expectedValues[index];
211 
212  if (!AreEqual(valueActual, valueExpected, result.Count))
213  {
214  await _webSocket.CloseOutputAsync(WebSocketCloseStatus.InvalidPayloadData,
215  "Value actual does not equal value expected", _token);
216  throw new Exception(
217  $"Expected: {valueExpected.Length} bytes Actual: {result.Count} bytes. Contents different.");
218  }
219  }
220  }
221  }
