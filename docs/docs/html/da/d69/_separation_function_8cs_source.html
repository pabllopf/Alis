70  internal unsafe void Initialize(SimplexCache* cache,
71  Shape shapeA, XForm transformA,
72  Shape shapeB, XForm transformB)
73  {
74  ShapeA = shapeA;
75  ShapeB = shapeB;
76  int count = cache->Count;
77  Box2DxDebug.Assert(0 < count && count < 3);
78 
79  if (count == 1)
80  {
81  FaceType = Type.Points;
82  Vector2 localPointA = ShapeA.GetVertex(cache->IndexA[0]);
83  Vector2 localPointB = ShapeB.GetVertex(cache->IndexB[0]);
84  Vector2 pointA = Math.Mul(transformA, localPointA);
85  Vector2 pointB = Math.Mul(transformB, localPointB);
86  Axis = pointB - pointA;
87  Axis.Normalize();
88  }
89  else if (cache->IndexB[0] == cache->IndexB[1])
90  {
91  // Two points on A and one on B
92  FaceType = Type.FaceA;
93  Vector2 localPointA1 = ShapeA.GetVertex(cache->IndexA[0]);
94  Vector2 localPointA2 = ShapeA.GetVertex(cache->IndexA[1]);
95  Vector2 localPointB = ShapeB.GetVertex(cache->IndexB[0]);
96  LocalPoint = 0.5f * (localPointA1 + localPointA2);
97  Axis = Vector2.Cross(localPointA2 - localPointA1, 1.0f);
98  Axis.Normalize();
99 
100  Vector2 normal = Math.Mul(transformA.R, Axis);
101  Vector2 pointA = Math.Mul(transformA, LocalPoint);
102  Vector2 pointB = Math.Mul(transformB, localPointB);
103 
104  float s = Vector2.Dot(pointB - pointA, normal);
105  if (s < 0.0f)
106  {
107  Axis = -Axis;
108  }
109  }
110  else
111  {
112  // Two points on B and one or two points on A.
113  // We ignore the second point on A.
114  FaceType = Type.FaceB;
115  Vector2 localPointA = shapeA.GetVertex(cache->IndexA[0]);
116  Vector2 localPointB1 = shapeB.GetVertex(cache->IndexB[0]);
117  Vector2 localPointB2 = shapeB.GetVertex(cache->IndexB[1]);
118  LocalPoint = 0.5f * (localPointB1 + localPointB2);
119  Axis = Vector2.Cross(localPointB2 - localPointB1, 1.0f);
120  Axis.Normalize();
121 
122  Vector2 normal = Math.Mul(transformB.R, Axis);
123  Vector2 pointB = Math.Mul(transformB, LocalPoint);
124  Vector2 pointA = Math.Mul(transformA, localPointA);
125 
126  float s = Vector2.Dot(pointA - pointB, normal);
127  if (s < 0.0f)
128  {
129  Axis = -Axis;
130  }
131  }
132  }

140  internal float Evaluate(XForm transformA, XForm transformB)
141  {
142  switch (FaceType)
143  {
144  case Type.Points:
145  {
146  Vector2 axisA = Math.MulT(transformA.R, Axis);
147  Vector2 axisB = Math.MulT(transformB.R, -Axis);
148  Vector2 localPointA = ShapeA.GetSupportVertex(axisA);
149  Vector2 localPointB = ShapeB.GetSupportVertex(axisB);
150  Vector2 pointA = Math.Mul(transformA, localPointA);
151  Vector2 pointB = Math.Mul(transformB, localPointB);
152  float separation = Vector2.Dot(pointB - pointA, Axis);
153  return separation;
154  }
155 
156  case Type.FaceA:
157  {
158  Vector2 normal = Math.Mul(transformA.R, Axis);
159  Vector2 pointA = Math.Mul(transformA, LocalPoint);
160 
161  Vector2 axisB = Math.MulT(transformB.R, -normal);
162 
163  Vector2 localPointB = ShapeB.GetSupportVertex(axisB);
164  Vector2 pointB = Math.Mul(transformB, localPointB);
165 
166  float separation = Vector2.Dot(pointB - pointA, normal);
167  return separation;
168  }
169 
170  case Type.FaceB:
171  {
172  Vector2 normal = Math.Mul(transformB.R, Axis);
173  Vector2 pointB = Math.Mul(transformB, LocalPoint);
174 
175  Vector2 axisA = Math.MulT(transformA.R, -normal);
176 
177  Vector2 localPointA = ShapeA.GetSupportVertex(axisA);
178  Vector2 pointA = Math.Mul(transformA, localPointA);
179 
180  float separation = Vector2.Dot(pointA - pointB, normal);
181  return separation;
182  }
183 
184  default:
185  Box2DxDebug.Assert(false);
186  return 0.0f;
187  }
188  }
