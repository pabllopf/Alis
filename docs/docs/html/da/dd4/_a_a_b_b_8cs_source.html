89  public void RayCast(out RayCastOutput output, RayCastInput input)
90  {
91  float tmin = -Settings.FltMax;
92  float tmax = Settings.FltMax;
93 
94  output = new RayCastOutput();
95 
96  output.Hit = false;
97 
98  Vector2 p = input.P1;
99  Vector2 d = input.P2 - input.P1;
100  Vector2 absD = Math.Abs(d);
101 
102  Vector2 normal = new Vector2(0);
103 
104  for (int i = 0; i < 2; ++i)
105  {
106  if (absD[i] < Settings.FltEpsilon)
107  {
108  // Parallel.
109  if (p[i] < LowerBound[i] || UpperBound[i] < p[i])
110  {
111  return;
112  }
113  }
114  else
115  {
116  float invD = 1.0f / d[i];
117  float t1 = (LowerBound[i] - p[i]) * invD;
118  float t2 = (UpperBound[i] - p[i]) * invD;
119 
120  // Sign of the normal vector.
121  float s = -1.0f;
122 
123  if (t1 > t2)
124  {
125  Math.Swap(ref t1, ref t2);
126  s = 1.0f;
127  }
128 
129  // Push the min up
130  if (t1 > tmin)
131  {
132  normal.SetZero();
133  normal[i] = s;
134  tmin = t1;
135  }
136 
137  // Pull the max down
138  tmax = Math.Min(tmax, t2);
139 
140  if (tmin > tmax)
141  {
142  return;
143  }
144  }
145  }
146 
147  // Does the ray start inside the box?
148  // Does the ray intersect beyond the max fraction?
149  if (tmin < 0.0f || input.MaxFraction < tmin)
150  {
151  return;
152  }
153 
154  // Intersection.
155  output.Fraction = tmin;
156  output.Normal = normal;
157  output.Hit = true;
158  }
