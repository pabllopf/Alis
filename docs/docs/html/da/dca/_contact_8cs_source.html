298  public void Update(IContactListener listener)
299  {
300  Manifold oldManifold = Manifold.Clone();
301 
302  Evaluate();
303 
304  Body bodyA = FixtureA.Body;
305  Body bodyB = FixtureB.Body;
306 
307  int oldCount = oldManifold.PointCount;
308  int newCount = Manifold.PointCount;
309 
310  if (newCount == 0 && oldCount > 0)
311  {
312  bodyA.WakeUp();
313  bodyB.WakeUp();
314  }
315 
316  // Slow contacts don't generate TOI events.
317  if (bodyA.IsStatic() || bodyA.IsBullet() || bodyB.IsStatic() || bodyB.IsBullet())
318  {
319  Flags &= ~CollisionFlags.Slow;
320  }
321  else
322  {
323  Flags |= CollisionFlags.Slow;
324  }
325 
326  // Match old contact ids to new contact ids and copy the
327  // stored impulses to warm start the solver.
328  for (int i = 0; i < Manifold.PointCount; ++i)
329  {
330  ManifoldPoint mp2 = Manifold.Points[i];
331  mp2.NormalImpulse = 0.0f;
332  mp2.TangentImpulse = 0.0f;
333  ContactId id2 = mp2.Id;
334 
335  for (int j = 0; j < oldManifold.PointCount; ++j)
336  {
337  ManifoldPoint mp1 = oldManifold.Points[j];
338 
339  if (mp1.Id.Key == id2.Key)
340  {
341  mp2.NormalImpulse = mp1.NormalImpulse;
342  mp2.TangentImpulse = mp1.TangentImpulse;
343  break;
344  }
345  }
346  }
347 
348  if (oldCount == 0 && newCount > 0)
349  {
350  Flags |= CollisionFlags.Touch;
351  if (listener != null)
352  {
353  listener.BeginContact(this);
354  }
355  }
356 
357  if (oldCount > 0 && newCount == 0)
358  {
359  Flags &= ~CollisionFlags.Touch;
360  if (listener != null)
361  {
362  listener.EndContact(this);
363  }
364  }
365 
366  if ((Flags & CollisionFlags.NonSolid) == 0)
367  {
368  if (listener != null)
369  {
370  listener.PreSolve(this, oldManifold);
371  }
372 
373  // The user may have disabled contact.
374  if (Manifold.PointCount == 0)
375  {
376  Flags &= ~CollisionFlags.Touch;
377  }
378  }
379  }
