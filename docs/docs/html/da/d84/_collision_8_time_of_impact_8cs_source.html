51  // CCD via the secant method.
63  public static float TimeOfImpact(ToiInput input, Shape shapeA, Shape shapeB)
64  {
65  Sweep sweepA = input.SweepA;
66  Sweep sweepB = input.SweepB;
67 
68  Box2DxDebug.Assert(sweepA.T0 == sweepB.T0);
69  Box2DxDebug.Assert(1.0f - sweepA.T0 > Settings.FltEpsilon);
70 
71  float radius = shapeA.Radius + shapeB.Radius;
72  float tolerance = input.Tolerance;
73 
74  float alpha = 0.0f;
75 
76  const int kMaxIterations = 1000; // TODO_ERIN b2Settings
77  int iter = 0;
78  float target = 0.0f;
79 
80  // Prepare input for distance query.
81  SimplexCache cache = new SimplexCache();
82  cache.Count = 0;
83  DistanceInput distanceInput;
84  distanceInput.UseRadii = false;
85 
86  for (;;)
87  {
88  XForm xfA, xfB;
89  sweepA.GetTransform(out xfA, alpha);
90  sweepB.GetTransform(out xfB, alpha);
91 
92  // Get the distance between shapes.
93  distanceInput.TransformA = xfA;
94  distanceInput.TransformB = xfB;
95  DistanceOutput distanceOutput;
96  Distance(out distanceOutput, ref cache, ref distanceInput, shapeA, shapeB);
97 
98  if (distanceOutput.Distance <= 0.0f)
99  {
100  alpha = 1.0f;
101  break;
102  }
103 
104  SeparationFunction fcn = new SeparationFunction();
105  unsafe
106  {
107  fcn.Initialize(&cache, shapeA, xfA, shapeB, xfB);
108  }
109 
110  float separation = fcn.Evaluate(xfA, xfB);
111  if (separation <= 0.0f)
112  {
113  alpha = 1.0f;
114  break;
115  }
116 
117  if (iter == 0)
118  {
119  // Compute a reasonable target distance to give some breathing room
120  // for conservative advancement. We take advantage of the shape radii
121  // to create additional clearance.
122  if (separation > radius)
123  {
124  target = Math.Max(radius - tolerance, 0.75f * radius);
125  }
126  else
127  {
128  target = Math.Max(separation - tolerance, 0.02f * radius);
129  }
130  }
131 
132  if (separation - target < 0.5f * tolerance)
133  {
134  if (iter == 0)
135  {
136  alpha = 1.0f;
137  }
138 
139  break;
140  }
141 
142 #if _FALSE
143  // Dump the curve seen by the root finder
144  {
145  const int32 N = 100;
146  float32 dx = 1.0f / N;
147  float32 xs[N+1];
148  float32 fs[N+1];
149 
150  float32 x = 0.0f;
151 
152  for (int32 i = 0; i <= N; ++i)
153  {
154  sweepA.GetTransform(&xfA, x);
155  sweepB.GetTransform(&xfB, x);
156  float32 f = fcn.Evaluate(xfA, xfB) - target;
157 
158  printf("%g %g\n", x, f);
159 
160  xs[i] = x;
161  fs[i] = f;
162 
163  x += dx;
164  }
165  }
166 #endif
167 
168  // Compute 1D root of: f(x) - target = 0
169  float newAlpha = alpha;
170  {
171  float x1 = alpha, x2 = 1.0f;
172 
173  float f1 = separation;
174 
175  sweepA.GetTransform(out xfA, x2);
176  sweepB.GetTransform(out xfB, x2);
177  float f2 = fcn.Evaluate(xfA, xfB);
178 
179  // If intervals don't overlap at t2, then we are done.
180  if (f2 >= target)
181  {
182  alpha = 1.0f;
183  break;
184  }
185 
186  // Determine when intervals intersect.
187  int rootIterCount = 0;
188  for (;;)
189  {
190  // Use a mix of the secant rule and bisection.
191  float x;
192  if ((rootIterCount & 1) != 0)
193  {
194  // Secant rule to improve convergence.
195  x = x1 + (target - f1) * (x2 - x1) / (f2 - f1);
196  }
197  else
198  {
199  // Bisection to guarantee progress.
200  x = 0.5f * (x1 + x2);
201  }
202 
203  sweepA.GetTransform(out xfA, x);
204  sweepB.GetTransform(out xfB, x);
205 
206  float f = fcn.Evaluate(xfA, xfB);
207 
208  if (Math.Abs(f - target) < 0.025f * tolerance)
209  {
210  newAlpha = x;
211  break;
212  }
213 
214  // Ensure we continue to bracket the root.
215  if (f > target)
216  {
217  x1 = x;
218  f1 = f;
219  }
220  else
221  {
222  x2 = x;
223  f2 = f;
224  }
225 
226  ++rootIterCount;
227 
228  Box2DxDebug.Assert(rootIterCount < 50);
229  }
230 
231  MaxToiRootIters = Math.Max(MaxToiRootIters, rootIterCount);
232  }
233 
234  // Ensure significant advancement.
235  if (newAlpha < (1.0f + 100.0f * Settings.FltEpsilon) * alpha)
236  {
237  break;
238  }
239 
240  alpha = newAlpha;
241 
242  ++iter;
243 
244  if (iter == kMaxIterations)
245  {
246  break;
247  }
248  }
249 
250  MaxToiIters = Math.Max(MaxToiIters, iter);
251 
252  return alpha;
253  }
254  }
