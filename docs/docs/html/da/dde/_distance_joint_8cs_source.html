32 // v2 + (beta/h) * x1 + gamma * lambda = 0, gamma has units of inverse mass.

160  internal override void InitVelocityConstraints(TimeStep step)
161  {
162  Body b1 = Body1;
163  Body b2 = Body2;
164 
165  // Compute the effective mass matrix.
166  Vector2 r1 = Math.Mul(b1.GetXForm().R, LocalAnchor1 - b1.GetLocalCenter());
167  Vector2 r2 = Math.Mul(b2.GetXForm().R, LocalAnchor2 - b2.GetLocalCenter());
168  U = b2.Sweep.C + r2 - b1.Sweep.C - r1;
169 
170  // Handle singularity.
171  float length = U.Length();
172  if (length > Settings.LinearSlop)
173  {
174  U *= 1.0f / length;
175  }
176  else
177  {
178  U.Set(0.0f, 0.0f);
179  }
180 
181  float cr1U = Vector2.Cross(r1, U);
182  float cr2U = Vector2.Cross(r2, U);
183  float invMass = b1.InvMass + b1.InvI * cr1U * cr1U + b2.InvMass + b2.InvI * cr2U * cr2U;
184  Box2DxDebug.Assert(invMass > Settings.FltEpsilon);
185  Mass = 1.0f / invMass;
186 
187  if (FrequencyHz > 0.0f)
188  {
189  float c = length - this.length;
190 
191  // Frequency
192  float omega = 2.0f * Settings.Pi * FrequencyHz;
193 
194  // Damping coefficient
195  float d = 2.0f * Mass * DampingRatio * omega;
196 
197  // Spring stiffness
198  float k = Mass * omega * omega;
199 
200  // magic formulas
201  Gamma = 1.0f / (step.Dt * (d + step.Dt * k));
202  Bias = c * step.Dt * k * Gamma;
203 
204  Mass = 1.0f / (invMass + Gamma);
205  }
206 
207  if (step.WarmStarting)
208  {
209  //Scale the inpulse to support a variable timestep.
210  Impulse *= step.DtRatio;
211  Vector2 p = Impulse * U;
212  b1.LinearVelocity -= b1.InvMass * p;
213  b1.AngularVelocity -= b1.InvI * Vector2.Cross(r1, p);
214  b2.LinearVelocity += b2.InvMass * p;
215  b2.AngularVelocity += b2.InvI * Vector2.Cross(r2, p);
216  }
217  else
218  {
219  Impulse = 0.0f;
220  }
221  }

228  internal override bool SolvePositionConstraints(float baumgarte)
229  {
230  if (FrequencyHz > 0.0f)
231  {
232  //There is no possition correction for soft distace constraint.
233  return true;
234  }
235 
236  Body b1 = Body1;
237  Body b2 = Body2;
238 
239  Vector2 r1 = Math.Mul(b1.GetXForm().R, LocalAnchor1 - b1.GetLocalCenter());
240  Vector2 r2 = Math.Mul(b2.GetXForm().R, LocalAnchor2 - b2.GetLocalCenter());
241 
242  Vector2 d = b2.Sweep.C + r2 - b1.Sweep.C - r1;
243 
244  float length = d.Normalize();
245  float c = length - this.length;
246  c = Math.Clamp(c, -Settings.MaxLinearCorrection, Settings.MaxLinearCorrection);
247 
248  float impulse = -Mass * c;
249  U = d;
250  Vector2 p = impulse * U;
251 
252  b1.Sweep.C -= b1.InvMass * p;
253  b1.Sweep.A -= b1.InvI * Vector2.Cross(r1, p);
254  b2.Sweep.C += b2.InvMass * p;
255  b2.Sweep.A += b2.InvI * Vector2.Cross(r2, p);
256 
257  b1.SynchronizeTransform();
258  b2.SynchronizeTransform();
259 
260  return System.Math.Abs(c) < Settings.LinearSlop;
261  }
