163  public override SegmentCollide TestSegment(XForm transform, out float lambda, out Vector2 normal, Segment segment,
164  float maxLambda)
165  {
166  Vector2 r = segment.P2 - segment.P1;
167  Vector2 v1 = Math.Mul(transform, Vertex1);
168  Vector2 d = Math.Mul(transform, Vertex2) - v1;
169  Vector2 n = Vector2.Cross(d, 1.0f);
170 
171  float kSlop = 100.0f * Settings.FltEpsilon;
172  float denom = -Vector2.Dot(r, n);
173 
174  // Cull back facing collision and ignore parallel segments.
175  if (denom > kSlop)
176  {
177  // Does the segment intersect the infinite line associated with this segment?
178  Vector2 b = segment.P1 - v1;
179  float a = Vector2.Dot(b, n);
180 
181  if (0.0f <= a && a <= maxLambda * denom)
182  {
183  float mu2 = -r.X * b.Y + r.Y * b.X;
184 
185  // Does the segment intersect this segment?
186  if (-kSlop * denom <= mu2 && mu2 <= denom * (1.0f + kSlop))
187  {
188  a /= denom;
189  n.Normalize();
190  lambda = a;
191  normal = n;
192  return SegmentCollide.HitCollide;
193  }
194  }
195  }
196 
197  lambda = 0;
198  normal = new Vector2();
199  return SegmentCollide.MissCollide;
200  }

263  public override float ComputeSubmergedArea(Vector2 normal, float offset, XForm xf, out Vector2 c)
264  {
265  //Note that v0 is independent of any details of the specific edge
266  //We are relying on v0 being consistent between multiple edges of the same body
267  Vector2 v0 = offset * normal;
268  //b2Vec2 v0 = xf.position + (offset - b2Dot(normal, xf.position)) * normal;
269 
270  Vector2 v1 = Math.Mul(xf, Vertex1);
271  Vector2 v2 = Math.Mul(xf, Vertex2);
272 
273  float d1 = Vector2.Dot(normal, v1) - offset;
274  float d2 = Vector2.Dot(normal, v2) - offset;
275 
276  if (d1 > 0.0f)
277  {
278  if (d2 > 0.0f)
279  {
280  c = new Vector2();
281  return 0.0f;
282  }
283 
284  v1 = -d2 / (d1 - d2) * v1 + d1 / (d1 - d2) * v2;
285  }
286  else
287  {
288  if (d2 > 0.0f)
289  {
290  v2 = -d2 / (d1 - d2) * v1 + d1 / (d1 - d2) * v2;
291  }
292  }
293 
294  // v0,v1,v2 represents a fully submerged triangle
295  float kInv3 = 1.0f / 3.0f;
296 
297  // Area weighted centroid
298  c = kInv3 * (v0 + v1 + v2);
299 
300  Vector2 e1 = v1 - v0;
301  Vector2 e2 = v2 - v0;
302 
303  return 0.5f * Vector2.Cross(e1, e2);
304  }
