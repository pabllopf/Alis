30 // Linear constraint (point-to-line)
31 // d = p2 - p1 = x2 + r2 - x1 - r1
32 // C = dot(perp, d)
33 // Cdot = dot(d, cross(w1, perp)) + dot(perp, v2 + cross(w2, r2) - v1 - cross(w1, r1))
34 // = -dot(perp, v1) - dot(cross(d + r1, perp), w1) + dot(perp, v2) + dot(cross(r2, perp), v2)
35 // J = [-perp, -cross(d + r1, perp), perp, cross(r2,perp)]
36 //
37 // K = J * invM * JT
38 //
39 // J = [-a -s1 a s2]
40 // a = perp
41 // s1 = cross(d + r1, a) = cross(p2 - x1, a)
42 // s2 = cross(r2, a) = cross(p2 - x2, a)
43 
44 
45 // Motor/Limit linear constraint
46 // C = dot(ax1, d)
47 // Cdot = = -dot(ax1, v1) - dot(cross(d + r1, ax1), w1) + dot(ax1, v2) + dot(cross(r2, ax1), v2)
48 // J = [-ax1 -cross(d+r1,ax1) ax1 cross(r2,ax1)]
49 
50 // Block Solver
51 // We develop a block solver that includes the joint limit. This makes the limit stiff (inelastic) even
52 // when the mass has poor distribution (leading to large torques about the joint anchor points).
53 //
54 // The Jacobian has 3 rows:
55 // J = [-uT -s1 uT s2] // linear
56 // [-vT -a1 vT a2] // limit
57 //
58 // u = perp
59 // v = axis
60 // s1 = cross(d + r1, u), s2 = cross(r2, u)
61 // a1 = cross(d + r1, v), a2 = cross(r2, v)
62 
63 // M * (v2 - v1) = JT * df
64 // J * v2 = bias
65 //
66 // v2 = v1 + invM * JT * df
67 // J * (v1 + invM * JT * df) = bias
68 // K * df = bias - J * v1 = -Cdot
69 // K = J * invM * JT
70 // Cdot = J * v1 - bias
71 //
72 // Now solve for f2.
73 // df = f2 - f1
74 // K * (f2 - f1) = -Cdot
75 // f2 = invK * (-Cdot) + f1
76 //
77 // Clamp accumulated limit impulse.
78 // lower: f2(2) = max(f2(2), 0)
79 // upper: f2(2) = min(f2(2), 0)
80 //
81 // Solve for correct f2(1)
82 // K(1,1) * f2(1) = -Cdot(1) - K(1,2) * f2(2) + K(1,1:2) * f1
83 // = -Cdot(1) - K(1,2) * f2(2) + K(1,1) * f1(1) + K(1,2) * f1(2)
84 // K(1,1) * f2(1) = -Cdot(1) - K(1,2) * (f2(2) - f1(2)) + K(1,1) * f1(1)
85 // f2(1) = invK(1,1) * (-Cdot(1) - K(1,2) * (f2(2) - f1(2))) + f1(1)
86 //
87 // Now compute impulse to be applied:
88 // df = f2 - f1

413  internal override void InitVelocityConstraints(TimeStep step)
414  {
415  Body b1 = Body1;
416  Body b2 = Body2;
417 
418  LocalCenter1 = b1.GetLocalCenter();
419  LocalCenter2 = b2.GetLocalCenter();
420 
421  XForm xf1 = b1.GetXForm();
422  XForm xf2 = b2.GetXForm();
423 
424  // Compute the effective masses.
425  Vector2 r1 = Math.Mul(xf1.R, LocalAnchor1 - LocalCenter1);
426  Vector2 r2 = Math.Mul(xf2.R, LocalAnchor2 - LocalCenter2);
427  Vector2 d = b2.Sweep.C + r2 - b1.Sweep.C - r1;
428 
429  InvMass1 = b1.InvMass;
430  InvI1 = b1.InvI;
431  InvMass2 = b2.InvMass;
432  InvI2 = b2.InvI;
433 
434  // Compute motor Jacobian and effective mass.
435  {
436  Axis = Math.Mul(xf1.R, LocalXAxis1);
437  A1 = Vector2.Cross(d + r1, Axis);
438  A2 = Vector2.Cross(r2, Axis);
439 
440  MotorMass = InvMass1 + InvMass2 + InvI1 * A1 * A1 + InvI2 * A2 * A2;
441  Box2DxDebug.Assert(MotorMass > Settings.FltEpsilon);
442  MotorMass = 1.0f / MotorMass;
443  }
444 
445  // Prismatic constraint.
446  {
447  Perp = Math.Mul(xf1.R, LocalYAxis1);
448 
449  S1 = Vector2.Cross(d + r1, Perp);
450  s2 = Vector2.Cross(r2, Perp);
451 
452  float m1 = InvMass1, m2 = InvMass2;
453  float i1 = InvI1, i2 = InvI2;
454 
455  float k11 = m1 + m2 + i1 * S1 * S1 + i2 * s2 * s2;
456  float k12 = i1 * S1 * A1 + i2 * s2 * A2;
457  float k22 = m1 + m2 + i1 * A1 * A1 + i2 * A2 * A2;
458 
459  K.Col1.Set(k11, k12);
460  K.Col2.Set(k12, k22);
461  }
462 
463  // Compute motor and limit terms.
464  if (EnableLimitx)
465  {
466  float jointTranslation = Vector2.Dot(Axis, d);
467  if (Math.Abs(UpperTranslation - LowerTranslation) < 2.0f * Settings.LinearSlop)
468  {
469  LimitState = LimitState.EqualLimits;
470  }
471  else if (jointTranslation <= LowerTranslation)
472  {
473  if (LimitState != LimitState.AtLowerLimit)
474  {
475  LimitState = LimitState.AtLowerLimit;
476  Impulse.Y = 0.0f;
477  }
478  }
479  else if (jointTranslation >= UpperTranslation)
480  {
481  if (LimitState != LimitState.AtUpperLimit)
482  {
483  LimitState = LimitState.AtUpperLimit;
484  Impulse.Y = 0.0f;
485  }
486  }
487  else
488  {
489  LimitState = LimitState.InactiveLimit;
490  Impulse.Y = 0.0f;
491  }
492  }
493  else
494  {
495  LimitState = LimitState.InactiveLimit;
496  }
497 
498  if (EnableMotorx == false)
499  {
500  MotorImpulse = 0.0f;
501  }
502 
503  if (step.WarmStarting)
504  {
505  // Account for variable time step.
506  Impulse *= step.DtRatio;
507  MotorImpulse *= step.DtRatio;
508 
509  Vector2 p = Impulse.X * Perp + (MotorImpulse + Impulse.Y) * Axis;
510  float l1 = Impulse.X * S1 + (MotorImpulse + Impulse.Y) * A1;
511  float l2 = Impulse.X * s2 + (MotorImpulse + Impulse.Y) * A2;
512 
513  b1.LinearVelocity -= InvMass1 * p;
514  b1.AngularVelocity -= InvI1 * l1;
515 
516  b2.LinearVelocity += InvMass2 * p;
517  b2.AngularVelocity += InvI2 * l2;
518  }
519  else
520  {
521  Impulse.SetZero();
522  MotorImpulse = 0.0f;
523  }
524  }

530  internal override void SolveVelocityConstraints(TimeStep step)
531  {
532  Body b1 = Body1;
533  Body b2 = Body2;
534 
535  Vector2 v1 = b1.LinearVelocity;
536  float w1 = b1.AngularVelocity;
537  Vector2 v2 = b2.LinearVelocity;
538  float w2 = b2.AngularVelocity;
539 
540  // Solve linear motor constraint.
541  if (EnableMotorx && LimitState != LimitState.EqualLimits)
542  {
543  float cdot = Vector2.Dot(Axis, v2 - v1) + A2 * w2 - A1 * w1;
544  float impulse = MotorMass * (MotorSpeed - cdot);
545  float oldImpulse = MotorImpulse;
546  float maxImpulse = step.Dt * MaxMotorForce;
547  MotorImpulse = Math.Clamp(MotorImpulse + impulse, -maxImpulse, maxImpulse);
548  impulse = MotorImpulse - oldImpulse;
549 
550  Vector2 p = impulse * Axis;
551  float l1 = impulse * A1;
552  float l2 = impulse * A2;
553 
554  v1 -= InvMass1 * p;
555  w1 -= InvI1 * l1;
556 
557  v2 += InvMass2 * p;
558  w2 += InvI2 * l2;
559  }
560 
561  float cdot1 = Vector2.Dot(Perp, v2 - v1) + s2 * w2 - S1 * w1;
562 
563  if (EnableLimitx && LimitState != LimitState.InactiveLimit)
564  {
565  // Solve prismatic and limit constraint in block form.
566  float cdot2 = Vector2.Dot(Axis, v2 - v1) + A2 * w2 - A1 * w1;
567  Vector2 cdot = new Vector2(cdot1, cdot2);
568 
569  Vector2 f1 = Impulse;
570  Vector2 df = K.Solve(-cdot);
571  Impulse += df;
572 
573  if (LimitState == LimitState.AtLowerLimit)
574  {
575  Impulse.Y = Math.Max(Impulse.Y, 0.0f);
576  }
577  else if (LimitState == LimitState.AtUpperLimit)
578  {
579  Impulse.Y = Math.Min(Impulse.Y, 0.0f);
580  }
581 
582  // f2(1) = invK(1,1) * (-Cdot(1) - K(1,2) * (f2(2) - f1(2))) + f1(1)
583  float b = -cdot1 - (Impulse.Y - f1.Y) * K.Col2.X;
584  float f2R = b / K.Col1.X + f1.X;
585  Impulse.X = f2R;
586 
587  df = Impulse - f1;
588 
589  Vector2 p = df.X * Perp + df.Y * Axis;
590  float l1 = df.X * S1 + df.Y * A1;
591  float l2 = df.X * s2 + df.Y * A2;
592 
593  v1 -= InvMass1 * p;
594  w1 -= InvI1 * l1;
595 
596  v2 += InvMass2 * p;
597  w2 += InvI2 * l2;
598  }
599  else
600  {
601  // Limit is inactive, just solve the prismatic constraint in block form.
602  float df = -cdot1 / K.Col1.X;
603  Impulse.X += df;
604 
605  Vector2 p = df * Perp;
606  float l1 = df * S1;
607  float l2 = df * s2;
608 
609  v1 -= InvMass1 * p;
610  w1 -= InvI1 * l1;
611 
612  v2 += InvMass2 * p;
613  w2 += InvI2 * l2;
614  }
615 
616  b1.LinearVelocity = v1;
617  b1.AngularVelocity = w1;
618  b2.LinearVelocity = v2;
619  b2.AngularVelocity = w2;
620  }

627  internal override bool SolvePositionConstraints(float baumgarte)
628  {
629  Body b1 = Body1;
630  Body b2 = Body2;
631 
632  Vector2 invMass1 = b1.Sweep.C;
633  float a1 = b1.Sweep.A;
634 
635  Vector2 invMass2 = b2.Sweep.C;
636  float a2 = b2.Sweep.A;
637 
638  // Solve linear limit constraint.
639  float linearError = 0.0f, angularError = 0.0f;
640  bool active = false;
641  float clamp = 0.0f;
642 
643  var mat22 = new Matrix22(a1);
644  var mat23 = new Matrix22(a2);
645 
646  Vector2 r1 = Math.Mul(mat22, LocalAnchor1 - LocalCenter1);
647  Vector2 r2 = Math.Mul(mat23, LocalAnchor2 - LocalCenter2);
648  Vector2 d = invMass2 + r2 - invMass1 - r1;
649 
650  if (EnableLimitx)
651  {
652  Axis = Math.Mul(mat22, LocalXAxis1);
653 
654  A1 = Vector2.Cross(d + r1, Axis);
655  A2 = Vector2.Cross(r2, Axis);
656 
657  float translation = Vector2.Dot(Axis, d);
658  if (Math.Abs(UpperTranslation - LowerTranslation) < 2.0f * Settings.LinearSlop)
659  {
660  // Prevent large angular corrections
661  clamp = Math.Clamp(translation, -Settings.MaxLinearCorrection, Settings.MaxLinearCorrection);
662  linearError = Math.Abs(translation);
663  active = true;
664  }
665  else if (translation <= LowerTranslation)
666  {
667  // Prevent large linear corrections and allow some slop.
668  clamp = Math.Clamp(translation - LowerTranslation + Settings.LinearSlop,
669  -Settings.MaxLinearCorrection, 0.0f);
670  linearError = LowerTranslation - translation;
671  active = true;
672  }
673  else if (translation >= UpperTranslation)
674  {
675  // Prevent large linear corrections and allow some slop.
676  clamp = Math.Clamp(translation - UpperTranslation - Settings.LinearSlop, 0.0f,
677  Settings.MaxLinearCorrection);
678  linearError = translation - UpperTranslation;
679  active = true;
680  }
681  }
682 
683  Perp = Math.Mul(mat22, LocalYAxis1);
684 
685  S1 = Vector2.Cross(d + r1, Perp);
686  s2 = Vector2.Cross(r2, Perp);
687 
688  Vector2 impulse;
689  float c1;
690  c1 = Vector2.Dot(Perp, d);
691 
692  linearError = Math.Max(linearError, Math.Abs(c1));
693  angularError = 0.0f;
694 
695  if (active)
696  {
697  float m1 = InvMass1, m2 = InvMass2;
698  float i1 = InvI1, i2 = InvI2;
699 
700  float k11 = m1 + m2 + i1 * S1 * S1 + i2 * s2 * s2;
701  float k12 = i1 * S1 * A1 + i2 * s2 * A2;
702  float k22 = m1 + m2 + i1 * A1 * A1 + i2 * A2 * A2;
703 
704  K.Col1.Set(k11, k12);
705  K.Col2.Set(k12, k22);
706 
707  Vector2 c = new Vector2();
708  c.X = c1;
709  c.Y = clamp;
710 
711  impulse = K.Solve(-c);
712  }
713  else
714  {
715  float m1 = InvMass1, m2 = InvMass2;
716  float i1 = InvI1, i2 = InvI2;
717 
718  float k11 = m1 + m2 + i1 * S1 * S1 + i2 * s2 * s2;
719 
720  float impulse1 = -c1 / k11;
721  impulse.X = impulse1;
722  impulse.Y = 0.0f;
723  }
724 
725  Vector2 p = impulse.X * Perp + impulse.Y * Axis;
726  float l1 = impulse.X * S1 + impulse.Y * A1;
727  float l2 = impulse.X * s2 + impulse.Y * A2;
728 
729  invMass1 -= InvMass1 * p;
730  a1 -= InvI1 * l1;
731  invMass2 += InvMass2 * p;
732  a2 += InvI2 * l2;
733 
734  // TODO_ERIN remove need for this.
735  b1.Sweep.C = invMass1;
736  b1.Sweep.A = a1;
737  b2.Sweep.C = invMass2;
738  b2.Sweep.A = a2;
739  b1.SynchronizeTransform();
740  b2.SynchronizeTransform();
741 
742  return linearError <= Settings.LinearSlop && angularError <= Settings.AngularSlop;
743  }
744  }
