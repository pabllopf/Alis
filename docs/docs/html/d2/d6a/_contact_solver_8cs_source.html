69  public ContactSolver(TimeStep step, Contact[] contacts, int contactCount)
70  {
71  this.step = step;
72  ConstraintCount = contactCount;
73 
74  Constraints = new ContactConstraint[ConstraintCount];
75  for (int i = 0; i < ConstraintCount; i++)
76  {
77  Constraints[i] = new ContactConstraint();
78  }
79 
80  //int count = 0;
81  for (int i = 0; i < ConstraintCount; ++i)
82  {
83  Contact contact = contacts[i];
84 
85  Fixture fixtureA = contact.FixtureA;
86  Fixture fixtureB = contact.FixtureB;
87  Shape shapeA = fixtureA.Shape;
88  Shape shapeB = fixtureB.Shape;
89  float radiusA = shapeA.Radius;
90  float radiusB = shapeB.Radius;
91  Body bodyA = fixtureA.Body;
92  Body bodyB = fixtureB.Body;
93  Manifold manifold = contact.Manifold;
94 
95  float friction = Settings.MixFriction(fixtureA.Friction, fixtureB.Friction);
96  float restitution = Settings.MixRestitution(fixtureA.Restitution, fixtureB.Restitution);
97 
98  Vector2 vA = bodyA.LinearVelocity;
99  Vector2 vB = bodyB.LinearVelocity;
100  float wA = bodyA.AngularVelocity;
101  float wB = bodyB.AngularVelocity;
102 
103  Box2DxDebug.Assert(manifold.PointCount > 0);
104 
105  WorldManifold worldManifold = new WorldManifold();
106  worldManifold.Initialize(manifold, bodyA.Xf, radiusA, bodyB.Xf, radiusB);
107 
108  ContactConstraint cc = Constraints[i];
109  cc.BodyA = bodyA;
110  cc.BodyB = bodyB;
111  cc.Manifold = manifold;
112  cc.Normal = worldManifold.Normal;
113  cc.PointCount = manifold.PointCount;
114  cc.Friction = friction;
115  cc.Restitution = restitution;
116 
117  cc.LocalPlaneNormal = manifold.LocalPlaneNormal;
118  cc.LocalPoint = manifold.LocalPoint;
119  cc.Radius = radiusA + radiusB;
120  cc.Type = manifold.Type;
121 
122  unsafe
123  {
124  fixed (ContactConstraintPoint* ccPointsPtr = cc.Points)
125  {
126  for (int j = 0; j < cc.PointCount; ++j)
127  {
128  ManifoldPoint cp = manifold.Points[j];
129  ContactConstraintPoint* ccp = &ccPointsPtr[j];
130 
131  ccp->NormalImpulse = cp.NormalImpulse;
132  ccp->TangentImpulse = cp.TangentImpulse;
133 
134  ccp->LocalPoint = cp.LocalPoint;
135 
136  ccp->Ra = worldManifold.Points[j] - bodyA.Sweep.C;
137  ccp->Rb = worldManifold.Points[j] - bodyB.Sweep.C;
138 
139  float rnA = Vector2.Cross(ccp->Ra, cc.Normal);
140  float rnB = Vector2.Cross(ccp->Rb, cc.Normal);
141  rnA *= rnA;
142  rnB *= rnB;
143 
144  float kNormal = bodyA.InvMass + bodyB.InvMass + bodyA.InvI * rnA + bodyB.InvI * rnB;
145 
146  Box2DxDebug.Assert(kNormal > Settings.FltEpsilon);
147  ccp->NormalMass = 1.0f / kNormal;
148 
149  float kEqualized = bodyA.Mass * bodyA.InvMass + bodyB.Mass * bodyB.InvMass;
150  kEqualized += bodyA.Mass * bodyA.InvI * rnA + bodyB.Mass * bodyB.InvI * rnB;
151 
152  Box2DxDebug.Assert(kEqualized > Settings.FltEpsilon);
153  ccp->EqualizedMass = 1.0f / kEqualized;
154 
155  Vector2 tangent = Vector2.Cross(cc.Normal, 1.0f);
156 
157  float rtA = Vector2.Cross(ccp->Ra, tangent);
158  float rtB = Vector2.Cross(ccp->Rb, tangent);
159  rtA *= rtA;
160  rtB *= rtB;
161 
162  float kTangent = bodyA.InvMass + bodyB.InvMass + bodyA.InvI * rtA + bodyB.InvI * rtB;
163 
164  Box2DxDebug.Assert(kTangent > Settings.FltEpsilon);
165  ccp->TangentMass = 1.0f / kTangent;
166 
167  // Setup a velocity bias for restitution.
168  ccp->VelocityBias = 0.0f;
169  float vRel = Vector2.Dot(cc.Normal,
170  vB + Vector2.Cross(wB, ccp->Rb) - vA - Vector2.Cross(wA, ccp->Ra));
171  if (vRel < -Settings.VelocityThreshold)
172  {
173  ccp->VelocityBias = -cc.Restitution * vRel;
174  }
175  }
176 
177  // If we have two points, then prepare the block solver.
178  if (cc.PointCount == 2)
179  {
180  ContactConstraintPoint* ccp1 = &ccPointsPtr[0];
181  ContactConstraintPoint* ccp2 = &ccPointsPtr[1];
182 
183  float invMassA = bodyA.InvMass;
184  float invIa = bodyA.InvI;
185  float invMassB = bodyB.InvMass;
186  float invIb = bodyB.InvI;
187 
188  float rn1A = Vector2.Cross(ccp1->Ra, cc.Normal);
189  float rn1B = Vector2.Cross(ccp1->Rb, cc.Normal);
190  float rn2A = Vector2.Cross(ccp2->Ra, cc.Normal);
191  float rn2B = Vector2.Cross(ccp2->Rb, cc.Normal);
192 
193  float k11 = invMassA + invMassB + invIa * rn1A * rn1A + invIb * rn1B * rn1B;
194  float k22 = invMassA + invMassB + invIa * rn2A * rn2A + invIb * rn2B * rn2B;
195  float k12 = invMassA + invMassB + invIa * rn1A * rn2A + invIb * rn1B * rn2B;
196 
197  // Ensure a reasonable condition number.
198  const float kMaxConditionNumber = 100.0f;
199  if (k11 * k11 < kMaxConditionNumber * (k11 * k22 - k12 * k12))
200  {
201  // K is safe to invert.
202  cc.K.Col1.Set(k11, k12);
203  cc.K.Col2.Set(k12, k22);
204  cc.NormalMass = cc.K.GetInverse();
205  }
206  else
207  {
208  // The constraints are redundant, just use one.
209  // TODO_ERIN use deepest?
210  cc.PointCount = 1;
211  }
212  }
213  }
214  }
215  }

235  public void InitVelocityConstraints(TimeStep step)
236  {
237  unsafe
238  {
239  // Warm start.
240  for (int i = 0; i < ConstraintCount; ++i)
241  {
242  ContactConstraint c = Constraints[i];
243 
244  Body bodyA = c.BodyA;
245  Body bodyB = c.BodyB;
246  float invMassA = bodyA.InvMass;
247  float invIa = bodyA.InvI;
248  float invMassB = bodyB.InvMass;
249  float invIb = bodyB.InvI;
250  Vector2 normal = c.Normal;
251  Vector2 tangent = Vector2.Cross(normal, 1.0f);
252 
253  fixed (ContactConstraintPoint* pointsPtr = c.Points)
254  {
255  if (step.WarmStarting)
256  {
257  for (int j = 0; j < c.PointCount; ++j)
258  {
259  ContactConstraintPoint* ccp = &pointsPtr[j];
260  ccp->NormalImpulse *= step.DtRatio;
261  ccp->TangentImpulse *= step.DtRatio;
262  Vector2 p = ccp->NormalImpulse * normal + ccp->TangentImpulse * tangent;
263  bodyA.AngularVelocity -= invIa * Vector2.Cross(ccp->Ra, p);
264  bodyA.LinearVelocity -= invMassA * p;
265  bodyB.AngularVelocity += invIb * Vector2.Cross(ccp->Rb, p);
266  bodyB.LinearVelocity += invMassB * p;
267  }
268  }
269  else
270  {
271  for (int j = 0; j < c.PointCount; ++j)
272  {
273  ContactConstraintPoint* ccp = &pointsPtr[j];
274  ccp->NormalImpulse = 0.0f;
275  ccp->TangentImpulse = 0.0f;
276  }
277  }
278  }
279  }
280  }

286  public void SolveVelocityConstraints()
287  {
288  for (int i = 0; i < ConstraintCount; ++i)
289  {
290  ContactConstraint c = Constraints[i];
291  Body bodyA = c.BodyA;
292  Body bodyB = c.BodyB;
293  float wA = bodyA.AngularVelocity;
294  float wB = bodyB.AngularVelocity;
295  Vector2 vA = bodyA.LinearVelocity;
296  Vector2 vB = bodyB.LinearVelocity;
297  float invMassA = bodyA.InvMass;
298  float invIa = bodyA.InvI;
299  float invMassB = bodyB.InvMass;
300  float invIb = bodyB.InvI;
301  Vector2 normal = c.Normal;
302  Vector2 tangent = Vector2.Cross(normal, 1.0f);
303  float friction = c.Friction;
304 
305  Box2DxDebug.Assert(c.PointCount == 1 || c.PointCount == 2);
306 
307  unsafe
308  {
309  fixed (ContactConstraintPoint* pointsPtr = c.Points)
310  {
311  // Solve tangent constraints
312  for (int j = 0; j < c.PointCount; ++j)
313  {
314  ContactConstraintPoint* ccp = &pointsPtr[j];
315 
316  // Relative velocity at contact
317  Vector2 dv = vB + Vector2.Cross(wB, ccp->Rb) - vA - Vector2.Cross(wA, ccp->Ra);
318 
319  // Compute tangent force
320  float vt = Vector2.Dot(dv, tangent);
321  float lambda = ccp->TangentMass * -vt;
322 
323  // b2Clamp the accumulated force
324  float maxFriction = friction * ccp->NormalImpulse;
325  float newImpulse = Math.Clamp(ccp->TangentImpulse + lambda, -maxFriction, maxFriction);
326  lambda = newImpulse - ccp->TangentImpulse;
327 
328  // Apply contact impulse
329  Vector2 p = lambda * tangent;
330 
331  vA -= invMassA * p;
332  wA -= invIa * Vector2.Cross(ccp->Ra, p);
333 
334  vB += invMassB * p;
335  wB += invIb * Vector2.Cross(ccp->Rb, p);
336 
337  ccp->TangentImpulse = newImpulse;
338  }
339 
340  // Solve normal constraints
341  if (c.PointCount == 1)
342  {
343  ContactConstraintPoint ccp = c.Points[0];
344 
345  // Relative velocity at contact
346  Vector2 dv = vB + Vector2.Cross(wB, ccp.Rb) - vA - Vector2.Cross(wA, ccp.Ra);
347 
348  // Compute normal impulse
349  float vn = Vector2.Dot(dv, normal);
350  float lambda = -ccp.NormalMass * (vn - ccp.VelocityBias);
351 
352  // Clamp the accumulated impulse
353  float newImpulse = Math.Max(ccp.NormalImpulse + lambda, 0.0f);
354  lambda = newImpulse - ccp.NormalImpulse;
355 
356  // Apply contact impulse
357  Vector2 p = lambda * normal;
358  vA -= invMassA * p;
359  wA -= invIa * Vector2.Cross(ccp.Ra, p);
360 
361  vB += invMassB * p;
362  wB += invIb * Vector2.Cross(ccp.Rb, p);
363  ccp.NormalImpulse = newImpulse;
364  }
365  else
366  {
367  // Block solver developed in collaboration with Dirk Gregorius (back in 01/07 on Box2D_Lite).
368  // Build the mini LCP for this contact patch
369  //
370  // vn = A * x + b, vn >= 0, , vn >= 0, x >= 0 and vn_i * x_i = 0 with i = 1..2
371  //
372  // A = J * W * JT and J = ( -n, -r1 x n, n, r2 x n )
373  // b = vn_0 - velocityBias
374  //
375  // The system is solved using the "Total enumeration method" (s. Murty). The complementary constraint vn_i * x_i
376  // implies that we must have in any solution either vn_i = 0 or x_i = 0. So for the 2D contact problem the cases
377  // vn1 = 0 and vn2 = 0, x1 = 0 and x2 = 0, x1 = 0 and vn2 = 0, x2 = 0 and vn1 = 0 need to be tested. The first valid
378  // solution that satisfies the problem is chosen.
379  //
380  // In order to account of the accumulated impulse 'a' (because of the iterative nature of the solver which only requires
381  // that the accumulated impulse is clamped and not the incremental impulse) we change the impulse variable (x_i).
382  //
383  // Substitute:
384  //
385  // x = x' - a
386  //
387  // Plug into above equation:
388  //
389  // vn = A * x + b
390  // = A * (x' - a) + b
391  // = A * x' + b - A * a
392  // = A * x' + b'
393  // b' = b - A * a;
394 
395  ContactConstraintPoint* cp1 = &pointsPtr[0];
396  ContactConstraintPoint* cp2 = &pointsPtr[1];
397 
398  Vector2 a = new Vector2(cp1->NormalImpulse, cp2->NormalImpulse);
399  Box2DxDebug.Assert(a.X >= 0.0f && a.Y >= 0.0f);
400 
401  // Relative velocity at contact
402  Vector2 dv1 = vB + Vector2.Cross(wB, cp1->Rb) - vA - Vector2.Cross(wA, cp1->Ra);
403  Vector2 dv2 = vB + Vector2.Cross(wB, cp2->Rb) - vA - Vector2.Cross(wA, cp2->Ra);
404 
405  // Compute normal velocity
406  float vn1 = Vector2.Dot(dv1, normal);
407  float vn2 = Vector2.Dot(dv2, normal);
408 
409  Vector2 b;
410  b.X = vn1 - cp1->VelocityBias;
411  b.Y = vn2 - cp2->VelocityBias;
412  b -= Math.Mul(c.K, a);
413 
414  //const float k_errorTol = 1e-3f;
415  //B2_NOT_USED(k_errorTol);
416 
417  for (;;)
418  {
419  //
420  // Case 1: vn = 0
421  //
422  // 0 = A * x' + b'
423  //
424  // Solve for x':
425  //
426  // x' = - inv(A) * b'
427  //
428  Vector2 x = -Math.Mul(c.NormalMass, b);
429 
430  if (x.X >= 0.0f && x.Y >= 0.0f)
431  {
432  // Resubstitute for the incremental impulse
433  Vector2 d = x - a;
434 
435  // Apply incremental impulse
436  Vector2 p1 = d.X * normal;
437  Vector2 p2 = d.Y * normal;
438  vA -= invMassA * (p1 + p2);
439  wA -= invIa * (Vector2.Cross(cp1->Ra, p1) + Vector2.Cross(cp2->Ra, p2));
440 
441  vB += invMassB * (p1 + p2);
442  wB += invIb * (Vector2.Cross(cp1->Rb, p1) + Vector2.Cross(cp2->Rb, p2));
443 
444  // Accumulate
445  cp1->NormalImpulse = x.X;
446  cp2->NormalImpulse = x.Y;
447 
448 #if DEBUG_SOLVER
449  // Postconditions
450  dv1 = vB + Vec2.Cross(wB, cp1->RB) - vA - Vec2.Cross(wA, cp1->RA);
451  dv2 = vB + Vec2.Cross(wB, cp2->RB) - vA - Vec2.Cross(wA, cp2->RA);
452 
453  // Compute normal velocity
454  vn1 = Vec2.Dot(dv1, normal);
455  vn2 = Vec2.Dot(dv2, normal);
456 
457  Box2DXDebug.Assert(Common.Math.Abs(vn1 - cp1.VelocityBias) < k_errorTol);
458  Box2DXDebug.Assert(Common.Math.Abs(vn2 - cp2.VelocityBias) < k_errorTol);
459 #endif
460  break;
461  }
462 
463  //
464  // Case 2: vn1 = 0 and x2 = 0
465  //
466  // 0 = a11 * x1' + a12 * 0 + b1'
467  // vn2 = a21 * x1' + a22 * 0 + b2'
468  //
469  x.X = -cp1->NormalMass * b.X;
470  x.Y = 0.0f;
471 /*
472  vn1 = 0.0f;
473 */
474  vn2 = c.K.Col1.Y * x.X + b.Y;
475 
476  if (x.X >= 0.0f && vn2 >= 0.0f)
477  {
478  // Resubstitute for the incremental impulse
479  Vector2 d = x - a;
480 
481  // Apply incremental impulse
482  Vector2 p1 = d.X * normal;
483  Vector2 p2 = d.Y * normal;
484  vA -= invMassA * (p1 + p2);
485  wA -= invIa * (Vector2.Cross(cp1->Ra, p1) + Vector2.Cross(cp2->Ra, p2));
486 
487  vB += invMassB * (p1 + p2);
488  wB += invIb * (Vector2.Cross(cp1->Rb, p1) + Vector2.Cross(cp2->Rb, p2));
489 
490  // Accumulate
491  cp1->NormalImpulse = x.X;
492  cp2->NormalImpulse = x.Y;
493 
494 #if DEBUG_SOLVER
495  // Postconditions
496  dv1 = vB + Vec2.Cross(wB, cp1->RB) - vA - Vec2.Cross(wA, cp1->RA);
497 
498  // Compute normal velocity
499  vn1 = Vec2.Dot(dv1, normal);
500 
501  Box2DXDebug.Assert(Common.Math.Abs(vn1 - cp1.VelocityBias) < k_errorTol);
502 #endif
503  break;
504  }
505 
506 
507  //
508  // Case 3: w2 = 0 and x1 = 0
509  //
510  // vn1 = a11 * 0 + a12 * x2' + b1'
511  // 0 = a21 * 0 + a22 * x2' + b2'
512  //
513  x.X = 0.0f;
514  x.Y = -cp2->NormalMass * b.Y;
515  vn1 = c.K.Col2.X * x.Y + b.X;
516 /*
517  vn2 = 0.0f;
518 */
519 
520  if (x.Y >= 0.0f && vn1 >= 0.0f)
521  {
522  // Resubstitute for the incremental impulse
523  Vector2 d = x - a;
524 
525  // Apply incremental impulse
526  Vector2 p1 = d.X * normal;
527  Vector2 p2 = d.Y * normal;
528  vA -= invMassA * (p1 + p2);
529  wA -= invIa * (Vector2.Cross(cp1->Ra, p1) + Vector2.Cross(cp2->Ra, p2));
530 
531  vB += invMassB * (p1 + p2);
532  wB += invIb * (Vector2.Cross(cp1->Rb, p1) + Vector2.Cross(cp2->Rb, p2));
533 
534  // Accumulate
535  cp1->NormalImpulse = x.X;
536  cp2->NormalImpulse = x.Y;
537 
538 #if DEBUG_SOLVER
539  // Postconditions
540  dv2 = vB + Vec2.Cross(wB, cp2->RB) - vA - Vec2.Cross(wA, cp2->RA);
541 
542  // Compute normal velocity
543  vn2 = Vec2.Dot(dv2, normal);
544 
545  Box2DXDebug.Assert(Common.Math.Abs(vn2 - cp2.VelocityBias) < k_errorTol);
546 #endif
547  break;
548  }
549 
550  //
551  // Case 4: x1 = 0 and x2 = 0
552  //
553  // vn1 = b1
554  // vn2 = b2;
555  x.X = 0.0f;
556  x.Y = 0.0f;
557  vn1 = b.X;
558  vn2 = b.Y;
559 
560  if (vn1 >= 0.0f && vn2 >= 0.0f)
561  {
562  // Resubstitute for the incremental impulse
563  Vector2 d = x - a;
564 
565  // Apply incremental impulse
566  Vector2 p1 = d.X * normal;
567  Vector2 p2 = d.Y * normal;
568  vA -= invMassA * (p1 + p2);
569  wA -= invIa * (Vector2.Cross(cp1->Ra, p1) + Vector2.Cross(cp2->Ra, p2));
570 
571  vB += invMassB * (p1 + p2);
572  wB += invIb * (Vector2.Cross(cp1->Rb, p1) + Vector2.Cross(cp2->Rb, p2));
573 
574  // Accumulate
575  cp1->NormalImpulse = x.X;
576  cp2->NormalImpulse = x.Y;
577  }
578 
579  // No solution, give up. This is hit sometimes, but it doesn't seem to matter.
580  break;
581  }
582  }
583 
584  bodyA.LinearVelocity = vA;
585  bodyA.AngularVelocity = wA;
586  bodyB.LinearVelocity = vB;
587  bodyB.AngularVelocity = wB;
588  }
589  }
590  }

616  public bool SolvePositionConstraints(float baumgarte)
617  {
618  float minSeparation = 0.0f;
619 
620  for (int i = 0; i < ConstraintCount; ++i)
621  {
622  ContactConstraint c = Constraints[i];
623  Body bodyA = c.BodyA;
624  Body bodyB = c.BodyB;
625 
626  float invMassA = bodyA.Mass * bodyA.InvMass;
627  float invIa = bodyA.Mass * bodyA.InvI;
628  float invMassB = bodyB.Mass * bodyB.InvMass;
629  float invIb = bodyB.Mass * bodyB.InvI;
630 
631  SPositionSolverManifold.Initialize(c);
632  Vector2 normal = SPositionSolverManifold.Normal;
633 
634  // Solver normal constraints
635  for (int j = 0; j < c.PointCount; ++j)
636  {
637  Vector2 point = SPositionSolverManifold.Points[j];
638  float separation = SPositionSolverManifold.Separations[j];
639 
640  Vector2 rA = point - bodyA.Sweep.C;
641  Vector2 rB = point - bodyB.Sweep.C;
642 
643  // Track max constraint error.
644  minSeparation = Math.Min(minSeparation, separation);
645 
646  // Prevent large corrections and allow slop.
647  float clamp = baumgarte * Math.Clamp(separation + Settings.LinearSlop,
648  -Settings.MaxLinearCorrection,
649  0.0f);
650 
651  // Compute normal impulse
652  float impulse = -c.Points[j].EqualizedMass * clamp;
653 
654  Vector2 p = impulse * normal;
655 
656  bodyA.Sweep.C -= invMassA * p;
657  bodyA.Sweep.A -= invIa * Vector2.Cross(rA, p);
658  bodyA.SynchronizeTransform();
659 
660  bodyB.Sweep.C += invMassB * p;
661  bodyB.Sweep.A += invIb * Vector2.Cross(rB, p);
662  bodyB.SynchronizeTransform();
663  }
664  }
665 
666  // We can't expect minSpeparation >= -Settings.LinearSlop because we don't
667  // push the separation above -Settings.LinearSlop.
