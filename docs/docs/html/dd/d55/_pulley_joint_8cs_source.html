30 // Pulley:
31 // length1 = norm(p1 - s1)
32 // length2 = norm(p2 - s2)
33 // C0 = (length1 + ratio * length2)_initial
34 // C = C0 - (length1 + ratio * length2) >= 0
35 // u1 = (p1 - s1) / norm(p1 - s1)
36 // u2 = (p2 - s2) / norm(p2 - s2)
37 // Cdot = -dot(u1, v1 + cross(w1, r1)) - ratio * dot(u2, v2 + cross(w2, r2))
38 // J = -[u1 cross(r1, u1) ratio * u2 ratio * cross(r2, u2)]
39 // K = J * invM * JT
40 // = invMass1 + invI1 * cross(r1, u1)^2 + ratio^2 * (invMass2 + invI2 * cross(r2, u2)^2)
41 //
42 // Limit:
43 // C = maxLength - length
44 // u = (p - s) / norm(p - s)
45 // Cdot = -dot(u, v + cross(w, r))
46 // K = invMass + invI * cross(r, u)^2
47 // 0 <= impulse

275  internal override void InitVelocityConstraints(TimeStep step)
276  {
277  Body body1 = Body1;
278  Body body2 = Body2;
279 
280  Vector2 mulR1 = Box2DXMath.Mul(body1.GetXForm().R, LocalAnchor1 - body1.GetLocalCenter());
281  Vector2 mulR2 = Box2DXMath.Mul(body2.GetXForm().R, LocalAnchor2 - body2.GetLocalCenter());
282 
283  Vector2 body1SweepC = body1.Sweep.C + mulR1;
284  Vector2 body2SweepC = body2.Sweep.C + mulR2;
285 
286  Vector2 groundAnchor1 = Ground.GetXForm().Position + GroundAnchor1;
287  Vector2 groundAnchor2 = Ground.GetXForm().Position + GroundAnchor2;
288 
289  // Get the pulley axes.
290  U1 = body1SweepC - groundAnchor1;
291  U2 = body2SweepC - groundAnchor2;
292 
293  float length1 = U1.Length();
294  float length2 = U2.Length();
295 
296  if (length1 > Settings.LinearSlop)
297  {
298  U1 *= 1.0f / length1;
299  }
300  else
301  {
302  U1.SetZero();
303  }
304 
305  if (length2 > Settings.LinearSlop)
306  {
307  U2 *= 1.0f / length2;
308  }
309  else
310  {
311  U2.SetZero();
312  }
313 
314  float c = Constant - length1 - Ratio * length2;
315  if (c > 0.0f)
316  {
317  State = LimitState.InactiveLimit;
318  Impulse = 0.0f;
319  }
320  else
321  {
322  State = LimitState.AtUpperLimit;
323  }
324 
325  if (length1 < MaxLength1)
326  {
327  LimitState1 = LimitState.InactiveLimit;
328  LimitImpulse1 = 0.0f;
329  }
330  else
331  {
332  LimitState1 = LimitState.AtUpperLimit;
333  }
334 
335  if (length2 < MaxLength2)
336  {
337  LimitState2 = LimitState.InactiveLimit;
338  LimitImpulse2 = 0.0f;
339  }
340  else
341  {
342  LimitState2 = LimitState.AtUpperLimit;
343  }
344 
345  // Compute effective mass.
346  float cr1U1 = Vector2.Cross(mulR1, U1);
347  float cr2U2 = Vector2.Cross(mulR2, U2);
348 
349  LimitMass1 = body1.InvMass + body1.InvI * cr1U1 * cr1U1;
350  LimitMass2 = body2.InvMass + body2.InvI * cr2U2 * cr2U2;
351  PulleyMass = LimitMass1 + Ratio * Ratio * LimitMass2;
352  Box2DxDebug.Assert(LimitMass1 > Settings.FltEpsilon);
353  Box2DxDebug.Assert(LimitMass2 > Settings.FltEpsilon);
354  Box2DxDebug.Assert(PulleyMass > Settings.FltEpsilon);
355  LimitMass1 = 1.0f / LimitMass1;
356  LimitMass2 = 1.0f / LimitMass2;
357  PulleyMass = 1.0f / PulleyMass;
358 
359  if (step.WarmStarting)
360  {
361  // Scale impulses to support variable time steps.
362  Impulse *= step.DtRatio;
363  LimitImpulse1 *= step.DtRatio;
364  LimitImpulse2 *= step.DtRatio;
365 
366  // Warm starting.
367  Vector2 p1 = -(Impulse + LimitImpulse1) * U1;
368  Vector2 p2 = (-Ratio * Impulse - LimitImpulse2) * U2;
369  body1.LinearVelocity += body1.InvMass * p1;
370  body1.AngularVelocity += body1.InvI * Vector2.Cross(mulR1, p1);
371  body2.LinearVelocity += body2.InvMass * p2;
372  body2.AngularVelocity += body2.InvI * Vector2.Cross(mulR2, p2);
373  }
374  else
375  {
376  Impulse = 0.0f;
377  LimitImpulse1 = 0.0f;
378  LimitImpulse2 = 0.0f;
379  }
380  }

386  internal override void SolveVelocityConstraints(TimeStep step)
387  {
388  Body b1 = Body1;
389  Body b2 = Body2;
390 
391  Vector2 r1 = Box2DXMath.Mul(b1.GetXForm().R, LocalAnchor1 - b1.GetLocalCenter());
392  Vector2 r2 = Box2DXMath.Mul(b2.GetXForm().R, LocalAnchor2 - b2.GetLocalCenter());
393 
394  if (State == LimitState.AtUpperLimit)
395  {
396  Vector2 v1 = b1.LinearVelocity + Vector2.Cross(b1.AngularVelocity, r1);
397  Vector2 v2 = b2.LinearVelocity + Vector2.Cross(b2.AngularVelocity, r2);
398 
399  float cdot = -Vector2.Dot(U1, v1) - Ratio * Vector2.Dot(U2, v2);
400  float impulse = PulleyMass * -cdot;
401  float oldImpulse = Impulse;
402  Impulse = Box2DXMath.Max(0.0f, Impulse + impulse);
403  impulse = Impulse - oldImpulse;
404 
405  Vector2 p1 = -impulse * U1;
406  Vector2 p2 = -Ratio * impulse * U2;
407  b1.LinearVelocity += b1.InvMass * p1;
408  b1.AngularVelocity += b1.InvI * Vector2.Cross(r1, p1);
409  b2.LinearVelocity += b2.InvMass * p2;
410  b2.AngularVelocity += b2.InvI * Vector2.Cross(r2, p2);
411  }
412 
413  if (LimitState1 == LimitState.AtUpperLimit)
414  {
415  Vector2 v1 = b1.LinearVelocity + Vector2.Cross(b1.AngularVelocity, r1);
416 
417  float cdot = -Vector2.Dot(U1, v1);
418  float impulse = -LimitMass1 * cdot;
419  float oldImpulse = LimitImpulse1;
420  LimitImpulse1 = Box2DXMath.Max(0.0f, LimitImpulse1 + impulse);
421  impulse = LimitImpulse1 - oldImpulse;
422 
423  Vector2 p1 = -impulse * U1;
424  b1.LinearVelocity += b1.InvMass * p1;
425  b1.AngularVelocity += b1.InvI * Vector2.Cross(r1, p1);
426  }
427 
428  if (LimitState2 == LimitState.AtUpperLimit)
429  {
430  Vector2 v2 = b2.LinearVelocity + Vector2.Cross(b2.AngularVelocity, r2);
431 
432  float cdot = -Vector2.Dot(U2, v2);
433  float impulse = -LimitMass2 * cdot;
434  float oldImpulse = LimitImpulse2;
435  LimitImpulse2 = Box2DXMath.Max(0.0f, LimitImpulse2 + impulse);
436  impulse = LimitImpulse2 - oldImpulse;
437 
438  Vector2 p2 = -impulse * U2;
439  b2.LinearVelocity += b2.InvMass * p2;
440  b2.AngularVelocity += b2.InvI * Vector2.Cross(r2, p2);
441  }
442  }

449  internal override bool SolvePositionConstraints(float baumgarte)
450  {
451  Body body1 = Body1;
452  Body body2 = Body2;
453 
454  Vector2 groundAnchor1 = Ground.GetXForm().Position + GroundAnchor1;
455  Vector2 groundAnchor2 = Ground.GetXForm().Position + GroundAnchor2;
456 
457  float linearError = 0.0f;
458 
459  if (State == LimitState.AtUpperLimit)
460  {
461  Vector2 mulR1 = Box2DXMath.Mul(body1.GetXForm().R, LocalAnchor1 - body1.GetLocalCenter());
462  Vector2 mulR2 = Box2DXMath.Mul(body2.GetXForm().R, LocalAnchor2 - body2.GetLocalCenter());
463 
464  Vector2 body1SweepC = body1.Sweep.C + mulR1;
465  Vector2 body2SweepC = body2.Sweep.C + mulR2;
466 
467  // Get the pulley axes.
468  U1 = body1SweepC - groundAnchor1;
469  U2 = body2SweepC - groundAnchor2;
470 
471  float length1 = U1.Length();
472  float length2 = U2.Length();
473 
474  if (length1 > Settings.LinearSlop)
475  {
476  U1 *= 1.0f / length1;
477  }
478  else
479  {
480  U1.SetZero();
481  }
482 
483  if (length2 > Settings.LinearSlop)
484  {
485  U2 *= 1.0f / length2;
486  }
487  else
488  {
489  U2.SetZero();
490  }
491 
492  float c = Constant - length1 - Ratio * length2;
493  linearError = Box2DXMath.Max(linearError, -c);
494 
495  c = Box2DXMath.Clamp(c + Settings.LinearSlop, -Settings.MaxLinearCorrection, 0.0f);
496  float impulse = -PulleyMass * c;
497 
498  Vector2 p1 = -impulse * U1;
499  Vector2 p2 = -Ratio * impulse * U2;
500 
501  body1.Sweep.C += body1.InvMass * p1;
502  body1.Sweep.A += body1.InvI * Vector2.Cross(mulR1, p1);
503  body2.Sweep.C += body2.InvMass * p2;
504  body2.Sweep.A += body2.InvI * Vector2.Cross(mulR2, p2);
505 
506  body1.SynchronizeTransform();
507  body2.SynchronizeTransform();
508  }
509 
510  if (LimitState1 == LimitState.AtUpperLimit)
511  {
512  Vector2 mulR1 = Box2DXMath.Mul(body1.GetXForm().R, LocalAnchor1 - body1.GetLocalCenter());
513  Vector2 body1SweepC = body1.Sweep.C + mulR1;
514 
515  U1 = body1SweepC - groundAnchor1;
516  float length1 = U1.Length();
517 
518  if (length1 > Settings.LinearSlop)
519  {
520  U1 *= 1.0f / length1;
521  }
522  else
523  {
524  U1.SetZero();
525  }
526 
527  float c = MaxLength1 - length1;
528  linearError = Box2DXMath.Max(linearError, -c);
529  c = Box2DXMath.Clamp(c + Settings.LinearSlop, -Settings.MaxLinearCorrection, 0.0f);
530  float impulse = -LimitMass1 * c;
531 
532  Vector2 p1 = -impulse * U1;
533  body1.Sweep.C += body1.InvMass * p1;
534  body1.Sweep.A += body1.InvI * Vector2.Cross(mulR1, p1);
535 
536  body1.SynchronizeTransform();
537  }
538 
539  if (LimitState2 == LimitState.AtUpperLimit)
540  {
541  Vector2 mulR2 = Box2DXMath.Mul(body2.GetXForm().R, LocalAnchor2 - body2.GetLocalCenter());
542  Vector2 body2SweepC = body2.Sweep.C + mulR2;
543 
544  U2 = body2SweepC - groundAnchor2;
545  float length2 = U2.Length();
546 
547  if (length2 > Settings.LinearSlop)
548  {
549  U2 *= 1.0f / length2;
550  }
551  else
552  {
553  U2.SetZero();
554  }
555 
556  float c = MaxLength2 - length2;
557  linearError = Box2DXMath.Max(linearError, -c);
558  c = Box2DXMath.Clamp(c + Settings.LinearSlop, -Settings.MaxLinearCorrection, 0.0f);
559  float impulse = -LimitMass2 * c;
560 
561  Vector2 p2 = -impulse * U2;
562  body2.Sweep.C += body2.InvMass * p2;
563  body2.Sweep.A += body2.InvI * Vector2.Cross(mulR2, p2);
564 
565  body2.SynchronizeTransform();
566  }
567 
568  return linearError < Settings.LinearSlop;
569  }
