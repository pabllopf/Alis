88  public override void Step(TimeStep step)
89  {
90  //B2_NOT_USED(step);
91  if (BodyList == null)
92  {
93  return;
94  }
95 
96  if (UseWorldGravity)
97  {
98  Gravity = World.Gravity;
99  }
100 
101  for (ControllerEdge i = BodyList; i != null; i = i.NextBody)
102  {
103  Body body = i.Body;
104  if (body.IsSleeping())
105  {
106  //Buoyancy force is just a function of position,
107  //so unlike most forces, it is safe to ignore sleeping bodes
108  continue;
109  }
110 
111  Vector2 areac = new Vector2(0, 0);
112  Vector2 massc = new Vector2(0, 0);
113  float area = 0;
114  float mass = 0;
115  for (Fixture shape = body.GetFixtureList(); shape != null; shape = shape.Next)
116  {
117  Vector2 sc;
118  float sarea = shape.ComputeSubmergedArea(Normal, Offset, out sc);
119  area += sarea;
120  areac.X += sarea * sc.X;
121  areac.Y += sarea * sc.Y;
122  float shapeDensity = 0;
123  if (UseDensity)
124  {
125  //TODO: Expose density publicly
126  shapeDensity = shape.Density;
127  }
128  else
129  {
130  shapeDensity = 1;
131  }
132 
133  mass += sarea * shapeDensity;
134  massc.X += sarea * sc.X * shapeDensity;
135  massc.Y += sarea * sc.Y * shapeDensity;
136  }
137 
138  areac.X /= area;
139  areac.Y /= area;
140  //Vec2 localCentroid = Math.MulT(body.GetXForm(), areac);
141  massc.X /= mass;
142  massc.Y /= mass;
143  if (area < Settings.FltEpsilon)
144  {
145  continue;
146  }
147 
148  //Buoyancy
149  Vector2 buoyancyForce = -Density * area * Gravity;
150  body.ApplyForce(buoyancyForce, massc);
151  //Linear drag
152  Vector2 dragForce = body.GetLinearVelocityFromWorldPoint(areac) - Velocity;
153  dragForce *= -LinearDrag * area;
154  body.ApplyForce(dragForce, areac);
155  //Angular drag
156  //TODO: Something that makes more physical sense?
157  body.ApplyTorque(-body.GetInertia() / body.GetMass() * area * body.GetAngularVelocity() * angularDrag);
158  }
159  }
