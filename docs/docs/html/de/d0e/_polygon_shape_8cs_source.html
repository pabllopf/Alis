311  public override SegmentCollide TestSegment(XForm xf, out float lambda, out Vector2 normal, Segment segment,
312  float maxLambda)
313  {
314  lambda = 0f;
315  normal = Vector2.Zero;
316 
317  float lower = 0.0f, upper = maxLambda;
318 
319  Vector2 p1 = Math.MulT(xf.R, segment.P1 - xf.Position);
320  Vector2 p2 = Math.MulT(xf.R, segment.P2 - xf.Position);
321  Vector2 d = p2 - p1;
322  int index = -1;
323 
324  for (int i = 0; i < VertexCount; ++i)
325  {
326  // p = p1 + a * d
327  // dot(normal, p - v) = 0
328  // dot(normal, p1 - v) + a * dot(normal, d) = 0
329  float numerator = Vector2.Dot(Normals[i], Vertices[i] - p1);
330  float denominator = Vector2.Dot(Normals[i], d);
331 
332  if (denominator == 0.0f)
333  {
334  if (numerator < 0.0f)
335  {
336  return SegmentCollide.MissCollide;
337  }
338  }
339  else
340  {
341  // Note: we want this predicate without division:
342  // lower < numerator / denominator, where denominator < 0
343  // Since denominator < 0, we have to flip the inequality:
344  // lower < numerator / denominator <==> denominator * lower > numerator.
345  if (denominator < 0.0f && numerator < lower * denominator)
346  {
347  // Increase lower.
348  // The segment enters this half-space.
349  lower = numerator / denominator;
350  index = i;
351  }
352  else if (denominator > 0.0f && numerator < upper * denominator)
353  {
354  // Decrease upper.
355  // The segment exits this half-space.
356  upper = numerator / denominator;
357  }
358  }
359 
360  if (upper < lower)
361  {
362  return SegmentCollide.MissCollide;
363  }
364  }
365 
366  Box2DxDebug.Assert(0.0f <= lower && lower <= maxLambda);
367 
368  if (index >= 0)
369  {
370  lambda = lower;
371  normal = Math.Mul(xf.R, Normals[index]);
372  return SegmentCollide.HitCollide;
373  }
374 
375  lambda = 0f;
376  return SegmentCollide.StartInsideCollide;
377  }

406  public override void ComputeMass(out MassData massData, float denstity)
407  {
408  // Polygon mass, centroid, and inertia.
409  // Let rho be the polygon density in mass per unit area.
410  // Then:
411  // mass = rho * int(dA)
412  // centroid.x = (1/mass) * rho * int(x * dA)
413  // centroid.y = (1/mass) * rho * int(y * dA)
414  // I = rho * int((x*x + y*y) * dA)
415  //
416  // We can compute these integrals by summing all the integrals
417  // for each triangle of the polygon. To evaluate the integral
418  // for a single triangle, we make a change of variables to
419  // the (u,v) coordinates of the triangle:
420  // x = x0 + e1x * u + e2x * v
421  // y = y0 + e1y * u + e2y * v
422  // where 0 <= u && 0 <= v && u + v <= 1.
423  //
424  // We integrate u from [0,1-v] and then v from [0,1].
425  // We also need to use the Jacobian of the transformation:
426  // D = cross(e1, e2)
427  //
428  // Simplification: triangle centroid = (1/3) * (p1 + p2 + p3)
429  //
430  // The rest of the derivation is handled by computer algebra.
431 
432  Box2DxDebug.Assert(VertexCount >= 3);
433 
434  Vector2 center = new Vector2(0);
435  float area = 0.0f;
436  float I = 0.0f;
437 
438  // pRef is the reference point for forming triangles.
439  // It's location doesn't change the result (except for rounding error).
440  Vector2 pRef = new Vector2(0);
441 
442 #if O
443  // This code would put the reference point inside the polygon.
444  for (int i = 0; i < vCount; ++i)
445  {
446  pRef += _vertices[i];
447  }
448  pRef *= 1.0f / count;
449 #endif
450 
451  const float kInv3 = 1.0f / 3.0f;
452 
453  for (int i = 0; i < VertexCount; ++i)
454  {
455  // Triangle vertices.
456  Vector2 p1 = pRef;
457  Vector2 p2 = Vertices[i];
458  Vector2 p3 = i + 1 < VertexCount ? Vertices[i + 1] : Vertices[0];
459 
460  Vector2 e1 = p2 - p1;
461  Vector2 e2 = p3 - p1;
462 
463  float d = Vector2.Cross(e1, e2);
464 
465  float triangleArea = 0.5f * d;
466  area += triangleArea;
467 
468  // Area weighted centroid
469  center += triangleArea * kInv3 * (p1 + p2 + p3);
470 
471  float px = p1.X, py = p1.Y;
472  float ex1 = e1.X, ey1 = e1.Y;
473  float ex2 = e2.X, ey2 = e2.Y;
474 
475  float intx2 = kInv3 * (0.25f * (ex1 * ex1 + ex2 * ex1 + ex2 * ex2) + (px * ex1 + px * ex2)) +
476  0.5f * px * px;
477  float inty2 = kInv3 * (0.25f * (ey1 * ey1 + ey2 * ey1 + ey2 * ey2) + (py * ey1 + py * ey2)) +
478  0.5f * py * py;
479 
480  I += d * (intx2 + inty2);
481  }
482 
483  // Total mass
484  massData.Mass = denstity * area;
485 
486  // Center of mass
487  Box2DxDebug.Assert(area > Settings.FltEpsilon);
488  center *= 1.0f / area;
489  massData.Center = center;
490 
491  // Inertia tensor relative to the local origin.
492  massData.I = denstity * I;
493  }

503  public override float ComputeSubmergedArea(Vector2 normal, float offset, XForm xf, out Vector2 c)
504  {
505  //Transform plane into shape co-ordinates
506  Vector2 normalL = Math.MulT(xf.R, normal);
507  float offsetL = offset - Vector2.Dot(normal, xf.Position);
508 
509  float[] depths = new float[Settings.MaxPolygonVertices];
510  int diveCount = 0;
511  int intoIndex = -1;
512  int outoIndex = -1;
513 
514  bool lastSubmerged = false;
515  int i;
516  for (i = 0; i < VertexCount; i++)
517  {
518  depths[i] = Vector2.Dot(normalL, Vertices[i]) - offsetL;
519  bool isSubmerged = depths[i] < -Settings.FltEpsilon;
520  if (i > 0)
521  {
522  if (isSubmerged)
523  {
524  if (!lastSubmerged)
525  {
526  intoIndex = i - 1;
527  diveCount++;
528  }
529  }
530  else
531  {
532  if (lastSubmerged)
533  {
534  outoIndex = i - 1;
535  diveCount++;
536  }
537  }
538  }
539 
540  lastSubmerged = isSubmerged;
541  }
542 
543  switch (diveCount)
544  {
545  case 0:
546  if (lastSubmerged)
547  {
548  //Completely submerged
549  MassData md;
550  ComputeMass(out md, 1f);
551  c = Math.Mul(xf, md.Center);
552  return md.Mass;
553  }
554 
555  // Completely dry
556  // TODO: Shouldn't return break only?
557  c = new Vector2();
558  return 0;
559  case 1:
560  if (intoIndex == -1)
561  {
562  intoIndex = VertexCount - 1;
563  }
564  else
565  {
566  outoIndex = VertexCount - 1;
567  }
568 
569  break;
570  }
571 
572  int intoIndex2 = (intoIndex + 1) % VertexCount;
573  int outoIndex2 = (outoIndex + 1) % VertexCount;
574 
575  float intoLambda = (0 - depths[intoIndex]) / (depths[intoIndex2] - depths[intoIndex]);
576  float outoLambda = (0 - depths[outoIndex]) / (depths[outoIndex2] - depths[outoIndex]);
577 
578  Vector2 intoVec = new Vector2(Vertices[intoIndex].X * (1 - intoLambda) + Vertices[intoIndex2].X * intoLambda,
579  Vertices[intoIndex].Y * (1 - intoLambda) + Vertices[intoIndex2].Y * intoLambda);
580  Vector2 outoVec = new Vector2(Vertices[outoIndex].X * (1 - outoLambda) + Vertices[outoIndex2].X * outoLambda,
581  Vertices[outoIndex].Y * (1 - outoLambda) + Vertices[outoIndex2].Y * outoLambda);
582 
583  //Initialize accumulator
584  float area = 0;
585  Vector2 center = new Vector2(0);
586  Vector2 p2 = Vertices[intoIndex2];
587  Vector2 p3;
588 
589  const float kInv3 = 1.0f / 3.0f;
590 
591  //An awkward loop from intoIndex2+1 to outIndex2
592  i = intoIndex2;
593  while (i != outoIndex2)
594  {
595  i = (i + 1) % VertexCount;
596  if (i == outoIndex2)
597  {
598  p3 = outoVec;
599  }
600  else
601  {
602  p3 = Vertices[i];
603  }
604 
605  //Add the triangle formed by intoVec,p2,p3
606  {
607  Vector2 e1 = p2 - intoVec;
608  Vector2 e2 = p3 - intoVec;
609 
610  float d = Vector2.Cross(e1, e2);
611 
612  float triangleArea = 0.5f * d;
613 
614  area += triangleArea;
615 
616  // Area weighted centroid
617  center += triangleArea * kInv3 * (intoVec + p2 + p3);
618  }
619  //
620  p2 = p3;
621  }
622 
623  //Normalize and transform centroid
624  center *= 1.0f / area;
625 
626  c = Math.Mul(xf, center);
627 
628  return area;
629  }

708  public static Vector2 ComputeCentroid(Vector2[] vs, int count)
709  {
710  Box2DxDebug.Assert(count >= 3);
711 
712  Vector2 c = new Vector2(0f);
713  float area = 0f;
714 
715  // pRef is the reference point for forming triangles.
716  // It's location doesn't change the result (except for rounding error).
717  Vector2 pRef = new Vector2(0f);
718 #if O
719  // This code would put the reference point inside the polygon.
720  for (int i = 0; i < count; ++i)
721  {
722  pRef += vs[i];
723  }
724  pRef *= 1.0f / count;
725 #endif
726 
727  const float inv3 = 1.0f / 3.0f;
728 
729  for (int i = 0; i < count; ++i)
730  {
731  // Triangle vertices.
732  Vector2 p1 = pRef;
733  Vector2 p2 = vs[i];
734  Vector2 p3 = i + 1 < count ? vs[i + 1] : vs[0];
735 
736  Vector2 e1 = p2 - p1;
737  Vector2 e2 = p3 - p1;
738 
739  float d = Vector2.Cross(e1, e2);
740 
741  float triangleArea = 0.5f * d;
742  area += triangleArea;
743 
744  // Area weighted centroid
745  c += triangleArea * inv3 * (p1 + p2 + p3);
746  }
747 
748  // Centroid
749  Box2DxDebug.Assert(area > Settings.FltEpsilon);
750  c *= 1.0f / area;
751  return c;
752  }
