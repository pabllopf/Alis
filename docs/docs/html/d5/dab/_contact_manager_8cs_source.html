65  // This is a callback from the broadphase when two AABB proxies begin
66  // to overlap. We create a Contact to manage the narrow phase.
73  public override object PairAdded(object proxyUserDataA, object proxyUserDataB)
74  {
75  Fixture fixtureA = proxyUserDataA as Fixture;
76  Fixture fixtureB = proxyUserDataB as Fixture;
77 
78  Body bodyA = fixtureA.Body;
79  Body bodyB = fixtureB.Body;
80 
81  if (bodyA.IsStatic() && bodyB.IsStatic())
82  {
83  return NullContact;
84  }
85 
86  if (fixtureA.Body == fixtureB.Body)
87  {
88  return NullContact;
89  }
90 
91  if (bodyB.IsConnected(bodyA))
92  {
93  return NullContact;
94  }
95 
96  if (World.ContactFilter != null && World.ContactFilter.ShouldCollide(fixtureA, fixtureB) == false)
97  {
98  return NullContact;
99  }
100 
101  // Call the factory.
102  Contact c = Contact.Create(fixtureA, fixtureB);
103 
104  if (c == null)
105  {
106  return NullContact;
107  }
108 
109  // Contact creation may swap shapes.
110  fixtureA = c.FixtureA;
111  fixtureB = c.FixtureB;
112  bodyA = fixtureA.Body;
113  bodyB = fixtureB.Body;
114 
115  // Insert into the world.
116  c.Prev = null;
117  c.Next = World.ContactList;
118  if (World.ContactList != null)
119  {
120  World.ContactList.Prev = c;
121  }
122 
123  World.ContactList = c;
124 
125  // Connect to island graph.
126 
127  // Connect to body 1
128  c.NodeA.Contact = c;
129  c.NodeA.Other = bodyB;
130 
131  c.NodeA.Prev = null;
132  c.NodeA.Next = bodyA.ContactList;
133  if (bodyA.ContactList != null)
134  {
135  bodyA.ContactList.Prev = c.NodeA;
136  }
137 
138  bodyA.ContactList = c.NodeA;
139 
140  // Connect to body 2
141  c.NodeB.Contact = c;
142  c.NodeB.Other = bodyA;
143 
144  c.NodeB.Prev = null;
145  c.NodeB.Next = bodyB.ContactList;
146  if (bodyB.ContactList != null)
147  {
148  bodyB.ContactList.Prev = c.NodeB;
149  }
150 
151  bodyB.ContactList = c.NodeB;
152 
153  ++World.ContactCount;
154  return c;
155  }

157  // This is a callback from the broadphase when two AABB proxies cease

165  public override void PairRemoved(object proxyUserData1, object proxyUserData2, object pairUserData)
166  {
167  //B2_NOT_USED(proxyUserData1);

169 
170  if (pairUserData == null)
171  {
172  return;

176  if (c == NullContact)
177  {
178  return;
179  }
180 
181  // An attached body is being destroyed, we must destroy this contact
182  // immediately to avoid orphaned shape pointers.
183  Destroy(c);
184  }

190  public void Destroy(Contact c)
191  {
192  Fixture fixtureA = c.FixtureA;
193  Fixture fixtureB = c.FixtureB;
194  Body bodyA = fixtureA.Body;
195  Body bodyB = fixtureB.Body;
196 
197  if (c.Manifold.PointCount > 0)
198  {
199  if (World.ContactListener != null)
200  {
201  World.ContactListener.EndContact(c);
202  }
203  }
204 
205  // Remove from the world.
206  if (c.Prev != null)
207  {
208  c.Prev.Next = c.Next;
209  }
210 
211  if (c.Next != null)
212  {
213  c.Next.Prev = c.Prev;
214  }
215 
216  if (c == World.ContactList)
217  {
218  World.ContactList = c.Next;
219  }
220 
221  // Remove from body 1
222  if (c.NodeA.Prev != null)
223  {
224  c.NodeA.Prev.Next = c.NodeA.Next;
225  }
226 
227  if (c.NodeA.Next != null)
228  {
229  c.NodeA.Next.Prev = c.NodeA.Prev;
230  }
231 
232  if (c.NodeA == bodyA.ContactList)
233  {
234  bodyA.ContactList = c.NodeA.Next;
235  }
236 
237  // Remove from body 2
238  if (c.NodeB.Prev != null)
239  {
240  c.NodeB.Prev.Next = c.NodeB.Next;
241  }
242 
243  if (c.NodeB.Next != null)
244  {
245  c.NodeB.Next.Prev = c.NodeB.Prev;
246  }
247 
248  if (c.NodeB == bodyB.ContactList)
249  {
250  bodyB.ContactList = c.NodeB.Next;
251  }
252 
253  // Call the factory.
254  Contact.Destroy(ref c);
255  --World.ContactCount;
256  }

258  // This is the top level collision call for the time step. Here
259  // all the narrow phase collision is processed for the world
260  // contact list.
264  public void Collide()
265  {
266  // Update awake contacts.
267  for (Contact c = World.ContactList; c != null; c = c.GetNext())
268  {
269  Body bodyA = c.FixtureA.Body;
270  Body bodyB = c.FixtureB.Body;
271  if (bodyA.IsSleeping() && bodyB.IsSleeping())
272  {
273  continue;
274  }
275 
276  c.Update(World.ContactListener);
277  }
278  }
