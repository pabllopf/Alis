99  public float GetSignedArea()
100  {
101  //The simplest polygon which can exist in the Euclidean plane has 3 sides.
102  if (Count < 3)
103  {
104  return 0;
105  }
106 
107  int i;
108  float area = 0;
109 
110  for (i = 0; i < Count; i++)
111  {
112  int j = (i + 1) % Count;
113 
114  Vector2 vi = this[i];
115  Vector2 vj = this[j];
116 
117  area += vi.X * vj.Y;
118  area -= vi.Y * vj.X;
119  }
120 
121  area /= 2.0f;
122  return area;
123  }

135  public Vector2 GetCentroid()
136  {
137  //The simplest polygon which can exist in the Euclidean plane has 3 sides.
138  if (Count < 3)
139  {
140  return new Vector2(float.NaN, float.NaN);
141  }
142 
143  // Same algorithm is used by Box2D
144  Vector2 c = Vector2.Zero;
145  float area = 0.0f;
146  const float inv3 = 1.0f / 3.0f;
147 
148  for (int i = 0; i < Count; ++i)
149  {
150  // Triangle vertices.
151  Vector2 current = this[i];
152  Vector2 next = i + 1 < Count ? this[i + 1] : this[0];
153 
154  float triangleArea = 0.5f * (current.X * next.Y - current.Y * next.X);
155  area += triangleArea;
156 
157  // Area weighted centroid
158  c += triangleArea * inv3 * (current + next);
159  }
160 
161  // Centroid
162  c *= 1.0f / area;
163  return c;
164  }

289  public bool IsConvex()
290  {
291  //The simplest polygon which can exist in the Euclidean plane has 3 sides.
292  if (Count < 3)
293  {
294  return false;
295  }
296 
297  //Triangles are always convex
298  if (Count == 3)
299  {
300  return true;
301  }
302 
303  // Checks the polygon is convex and the interior is to the left of each edge.
304  for (int i = 0; i < Count; ++i)
305  {
306  int next = i + 1 < Count ? i + 1 : 0;
307  Vector2 edge = this[next] - this[i];
308 
309  for (int j = 0; j < Count; ++j)
310  {
311  // Don't check vertices on the current edge.
312  if (j == i || j == next)
313  {
314  continue;
315  }
316 
317  Vector2 r = this[j] - this[i];
318 
319  float s = edge.X * r.Y - edge.Y * r.X;
320 
321  if (s <= 0.0f)
322  {
323  return false;
324  }
325  }
326  }
327 
328  return true;
329  }

465  public int PointInPolygon(ref Vector2 point)
466  {
467  // Winding number
468  int wn = 0;
469 
470  // Iterate through polygon's edges
471  for (int i = 0; i < Count; i++)
472  {
473  // Get points
474  Vector2 p1 = this[i];
475  Vector2 p2 = this[NextIndex(i)];
476 
477  // Test if a point is directly on the edge
478  Vector2 edge = p2 - p1;
479  float area = MathUtils.Area(ref p1, ref p2, ref point);
480  if (area == 0f && Vector2.Dot(point - p1, edge) >= 0f && Vector2.Dot(point - p2, edge) <= 0f)
481  {
482  return 0;
483  }
484 
485  // Test edge for intersection with ray from point
486  if (p1.Y <= point.Y)
487  {
488  if (p2.Y > point.Y && area > 0f)
489  {
490  ++wn;
491  }
492  }
493  else
494  {
495  if (p2.Y <= point.Y && area < 0f)
496  {
497  --wn;
498  }
499  }
500  }
501 
502  return wn == 0 ? -1 : 1;
503  }

585  Vector2[] sourceArray,
586  int sourceIndex,
587  ref Matrix4x4 matrix,
588  Vector2[] destinationArray,
589  int destinationIndex,
590  int length)
591  {
592  if (sourceArray == null)
593  {

595  }
596 
597  if (destinationArray == null)
598  {
599  throw new ArgumentNullException(nameof(destinationArray));
600  }
601

603  {
604  throw new ArgumentException("Source array length is lesser than sourceIndex + length");
605  }
606 
607  if (destinationArray.Length < destinationIndex + length)
608  {
609  throw new ArgumentException("Destination array length is lesser than destinationIndex + length");
610  }
611

613  {
614  Vector2 position = sourceArray[sourceIndex + x];
615  Vector2 destination = destinationArray[destinationIndex + x];
616  destination.X = position.X * matrix.M11 + position.Y * matrix.M21 + matrix.M41;
617  destination.Y = position.X * matrix.M12 + position.Y * matrix.M22 + matrix.M42;
618  destinationArray[destinationIndex + x] = destination;
