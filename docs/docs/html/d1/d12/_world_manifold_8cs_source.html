65  public void Initialize(Manifold manifold, XForm xfA, float radiusA, XForm xfB, float radiusB)
66  {
67  if (manifold.PointCount == 0)
68  {
69  return;
70  }
71 
72  switch (manifold.Type)
73  {
74  case ManifoldType.Circles:
75  {
76  Vector2 pointA = Math.Mul(xfA, manifold.LocalPoint);
77  Vector2 pointB = Math.Mul(xfB, manifold.Points[0].LocalPoint);
78  Vector2 normal = new Vector2(1.0f, 0.0f);
79  if (Vector2.DistanceSquared(pointA, pointB) > Settings.FltEpsilonSquared)
80  {
81  normal = pointB - pointA;
82  normal.Normalize();
83  }
84 
85  Normal = normal;
86 
87  Vector2 cA = pointA + radiusA * normal;
88  Vector2 cB = pointB - radiusB * normal;
89  Points[0] = 0.5f * (cA + cB);
90  }
91  break;
92 
93  case ManifoldType.FaceA:
94  {
95  Vector2 normal = Math.Mul(xfA.R, manifold.LocalPlaneNormal);
96  Vector2 planePoint = Math.Mul(xfA, manifold.LocalPoint);
97 
98  // Ensure normal points from A to B.
99  Normal = normal;
100 
101  for (int i = 0; i < manifold.PointCount; ++i)
102  {
103  Vector2 clipPoint = Math.Mul(xfB, manifold.Points[i].LocalPoint);
104  Vector2 cA = clipPoint + (radiusA - Vector2.Dot(clipPoint - planePoint, normal)) * normal;
105  Vector2 cB = clipPoint - radiusB * normal;
106  Points[i] = 0.5f * (cA + cB);
107  }
108  }
109  break;
110 
111  case ManifoldType.FaceB:
112  {
113  Vector2 normal = Math.Mul(xfB.R, manifold.LocalPlaneNormal);
114  Vector2 planePoint = Math.Mul(xfB, manifold.LocalPoint);
115 
116  // Ensure normal points from A to B.
117  Normal = -normal;
118 
119  for (int i = 0; i < manifold.PointCount; ++i)
120  {
121  Vector2 clipPoint = Math.Mul(xfA, manifold.Points[i].LocalPoint);
122  Vector2 cA = clipPoint - radiusA * normal;
123  Vector2 cB = clipPoint + (radiusB - Vector2.Dot(clipPoint - planePoint, normal)) * normal;
124  Points[i] = 0.5f * (cA + cB);
125  }
126  }
127  break;
128  }
129  }
130  }
