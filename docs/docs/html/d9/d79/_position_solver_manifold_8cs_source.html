60  internal void Initialize(ContactConstraint cc)
61  {
62  Box2DxDebug.Assert(cc.PointCount > 0);
63 
64  switch (cc.Type)
65  {
66  case ManifoldType.Circles:
67  {
68  Vector2 pointA = cc.BodyA.GetWorldPoint(cc.LocalPoint);
69  Vector2 pointB = cc.BodyB.GetWorldPoint(cc.Points[0].LocalPoint);
70  if (Vector2.DistanceSquared(pointA, pointB) > Settings.FltEpsilonSquared)
71  {
72  Normal = pointB - pointA;
73  Normal.Normalize();
74  }
75  else
76  {
77  Normal.Set(1.0f, 0.0f);
78  }
79 
80  Points[0] = 0.5f * (pointA + pointB);
81  Separations[0] = Vector2.Dot(pointB - pointA, Normal) - cc.Radius;
82  }
83  break;
84 
85  case ManifoldType.FaceA:
86  {
87  Normal = cc.BodyA.GetWorldVector(cc.LocalPlaneNormal);
88  Vector2 planePoint = cc.BodyA.GetWorldPoint(cc.LocalPoint);
89 
90  for (int i = 0; i < cc.PointCount; ++i)
91  {
92  Vector2 clipPoint = cc.BodyB.GetWorldPoint(cc.Points[i].LocalPoint);
93  Separations[i] = Vector2.Dot(clipPoint - planePoint, Normal) - cc.Radius;
94  Points[i] = clipPoint;
95  }
96  }
97  break;
98 
99  case ManifoldType.FaceB:
100  {
101  Normal = cc.BodyB.GetWorldVector(cc.LocalPlaneNormal);
102  Vector2 planePoint = cc.BodyB.GetWorldPoint(cc.LocalPoint);
103 
104  for (int i = 0; i < cc.PointCount; ++i)
105  {
106  Vector2 clipPoint = cc.BodyA.GetWorldPoint(cc.Points[i].LocalPoint);
107  Separations[i] = Vector2.Dot(clipPoint - planePoint, Normal) - cc.Radius;
108  Points[i] = clipPoint;
109  }
110 
111  // Ensure normal points from A to B
112  Normal = -Normal;
113  }
114  break;
115  }
116  }
117  }
