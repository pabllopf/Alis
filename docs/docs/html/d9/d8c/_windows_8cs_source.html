37  // Be warned, there is a LOT of stuff here. It might seem complicated, but just take it slow and you'll be fine.
38  // OpenGL's initial hurdle is quite large, but once you get past that, things will start making more sense.
43  public class Window : GameWindow
44  {
45  // Create the vertices for our triangle. These are listed in normalized device coordinates (NDC)
46  // In NDC, (0, 0) is the center of the screen.
47  // Negative X coordinates move to the left, positive X move to the right.
48  // Negative Y coordinates move to the bottom, positive Y move to the top.
49  // OpenGL only supports rendering in 3D, so to create a flat triangle, the Z coordinate will be kept as 0.
53  private readonly float[] _vertices =
54  {
55  -0.5f, -0.5f, 0.0f, // Bottom-left vertex
56  0.5f, -0.5f, 0.0f, // Bottom-right vertex
57  0.0f, 0.5f, 0.0f // Top vertex
58  };
59 
63  private int _vertexArrayObject;
64 
65  // These are the handles to OpenGL objects. A handle is an integer representing where the object lives on the
66  // graphics card. Consider them sort of like a pointer; we can't do anything with them directly, but we can
67  // send them to OpenGL functions that need them.
68 
69  // What these objects are will be explained in OnLoad.
73  private int _vertexBufferObject;

86  // Now, we start initializing OpenGL.
90  protected override void OnLoad()
91  {
92  base.OnLoad();
93 
94  // This will be the color of the background after we clear it, in normalized colors.
95  // Normalized colors are mapped on a range of 0.0 to 1.0, with 0.0 representing black, and 1.0 representing
96  // the largest possible value for that channel.
97  // This is a deep green.
98  GL.ClearColor(0.2f, 0.3f, 0.3f, 1.0f);
99 
100  // We need to send our vertices over to the graphics card so OpenGL can use them.
101  // To do this, we need to create what's called a Vertex Buffer Object (VBO).
102  // These allow you to upload a bunch of data to a buffer, and send the buffer to the graphics card.
103  // This effectively sends all the vertices at the same time.
104 
105  // First, we need to create a buffer. This function returns a handle to it, but as of right now, it's empty.
106  _vertexBufferObject = GL.GenBuffer();
107 
108  // Now, bind the buffer. OpenGL uses one global state, so after calling this,
109  // all future calls that modify the VBO will be applied to this buffer until another buffer is bound instead.
110  // The first argument is an enum, specifying what type of buffer we're binding. A VBO is an ArrayBuffer.
111  // There are multiple types of buffers, but for now, only the VBO is necessary.
112  // The second argument is the handle to our buffer.
113  GL.BindBuffer(BufferTarget.ArrayBuffer, _vertexBufferObject);
114 
115  // Finally, upload the vertices to the buffer.
116  // Arguments:
117  // Which buffer the data should be sent to.
118  // How much data is being sent, in bytes. You can generally set this to the length of your array, multiplied by sizeof(array type).
119  // The vertices themselves.
120  // How the buffer will be used, so that OpenGL can write the data to the proper memory space on the GPU.
121  // There are three different BufferUsageHints for drawing:
122  // StaticDraw: This buffer will rarely, if ever, update after being initially uploaded.
123  // DynamicDraw: This buffer will change frequently after being initially uploaded.
124  // StreamDraw: This buffer will change on every frame.
125  // Writing to the proper memory space is important! Generally, you'll only want StaticDraw,
126  // but be sure to use the right one for your use case.
127  GL.BufferData(BufferTarget.ArrayBuffer, _vertices.Length * sizeof(float), _vertices, BufferUsageHint.StaticDraw);
128 
129  // One notable thing about the buffer we just loaded data into is that it doesn't have any structure to it. It's just a bunch of floats (which are actaully just bytes).
130  // The opengl driver doesn't know how this data should be interpreted or how it should be divided up into vertices. To do this opengl introduces the idea of a
131  // Vertex Array Obejct (VAO) which has the job of keeping track of what parts or what buffers correspond to what data. In this example we want to set our VAO up so that
132  // it tells opengl that we want to interpret 12 bytes as 3 floats and divide the buffer into vertices using that.
133  // To do this we generate and bind a VAO (which looks deceptivly similar to creating and binding a VBO, but they are different!).
134  _vertexArrayObject = GL.GenVertexArray();
135  GL.BindVertexArray(_vertexArrayObject);
136 
137  // Now, we need to setup how the vertex shader will interpret the VBO data; you can send almost any C datatype (and a few non-C ones too) to it.
138  // While this makes them incredibly flexible, it means we have to specify how that data will be mapped to the shader's input variables.
139 
140  // To do this, we use the GL.VertexAttribPointer function
141  // This function has two jobs, to tell opengl about the format of the data, but also to associate the current array buffer with the VAO.
142  // This means that after this call, we have setup this attribute to source data from the current array buffer and interpret it in the way we specified.
143  // Arguments:
144  // Location of the input variable in the shader. the layout(location = 0) line in the vertex shader explicitly sets it to 0.
145  // How many elements will be sent to the variable. In this case, 3 floats for every vertex.
146  // The data type of the elements set, in this case float.
147  // Whether or not the data should be converted to normalized device coordinates. In this case, false, because that's already done.
148  // The stride; this is how many bytes are between the last element of one vertex and the first element of the next. 3 * sizeof(float) in this case.
149  // The offset; this is how many bytes it should skip to find the first element of the first vertex. 0 as of right now.
150  // Stride and Offset are just sort of glossed over for now, but when we get into texture coordinates they'll be shown in better detail.
151  GL.VertexAttribPointer(0, 3, VertexAttribPointerType.Float, false, 3 * sizeof(float), 0);
152 
153  // Enable variable 0 in the shader.
154  GL.EnableVertexAttribArray(0);
155 
156 
157  // Setup is now complete! Now we move to the OnRenderFrame function to finally draw the triangle.
158  }
159 
160  // Now that initialization is done, let's create our render loop.
165  protected override void OnRenderFrame(FrameEventArgs e)
166  {
167  base.OnRenderFrame(e);
168 
169  // This clears the image, using what you set as GL.ClearColor earlier.
170  // OpenGL provides several different types of data that can be rendered.
171  // You can clear multiple buffers by using multiple bit flags.
172  // However, we only modify the color, so ColorBufferBit is all we need to clear.
173  GL.Clear(ClearBufferMask.ColorBufferBit);
174 
175  // To draw an object in OpenGL, it's typically as simple as binding your shader,
176  // setting shader uniforms (not done here, will be shown in a future tutorial)
177  // binding the VAO,
178  // and then calling an OpenGL function to render.
179 
180 
181  // Bind the VAO
182  GL.BindVertexArray(_vertexArrayObject);
183 
184  // And then call our drawing function.
185  // For this tutorial, we'll use GL.DrawArrays, which is a very simple rendering function.
186  // Arguments:
187  // Primitive type; What sort of geometric primitive the vertices represent.
188  // OpenGL used to support many different primitive types, but almost all of the ones still supported
189  // is some variant of a triangle. Since we just want a single triangle, we use Triangles.
190  // Starting index; this is just the start of the data you want to draw. 0 here.
191  // How many vertices you want to draw. 3 for a triangle.
192  GL.DrawArrays(PrimitiveType.Triangles, 0, 3);
193 
194  // OpenTK windows are what's known as "double-buffered". In essence, the window manages two buffers.
195  // One is rendered to while the other is currently displayed by the window.
196  // This avoids screen tearing, a visual artifact that can happen if the buffer is modified while being displayed.
197  // After drawing, call this function to swap the buffers. If you don't, it won't display what you've rendered.
198  SwapBuffers();
199 
200  // And that's all you have to do for rendering! You should now see a yellow triangle on a black screen.
201  }

232  // Now, for cleanup.
233  // You should generally not do cleanup of opengl resources when exiting an application
234  // as that is handled by the driver and operating system when the application exits.
235  //
236  // There are reasons to delete opengl resources but exiting the application is not one of them.
237  // This is provided here as a reference on how resoruce cleanup is done in opengl but
238  // should not be done when exiting the application.
239  //
240  // Places where cleanup is appropriate would be to delete textures that are no
241  // longer used for whatever reason (e.g. a new scene is loaded that doesn't use a texture).
242  // This would free up video ram (VRAM) that can be used for new textures.
243  //
244  // The comming chapters will not have this code.
248  protected override void OnUnload()
249  {
250  // Unbind all the resources by binding the targets to 0/null.
251  GL.BindBuffer(BufferTarget.ArrayBuffer, 0);
252  GL.BindVertexArray(0);
253  GL.UseProgram(0);
254 
255  // Delete all the resources.
256  GL.DeleteBuffer(_vertexBufferObject);
257  GL.DeleteVertexArray(_vertexArrayObject);
258 
259  base.OnUnload();
260  }
261  }
