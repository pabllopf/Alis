54  public static void SplitShape(Fixture fixture, Vector2 entryPoint, Vector2 exitPoint, out Vertices first,
55  out Vertices second)
56  {
57  Vector2 localEntryPoint = fixture.Body.GetLocalPoint(ref entryPoint);
58  Vector2 localExitPoint = fixture.Body.GetLocalPoint(ref exitPoint);
59 
60  //We can only cut polygons at the moment
61  if (!(fixture.Shape is PolygonShape shape))
62  {
63  first = new Vertices();
64  second = new Vertices();
65  return;
66  }
67 
68  //Offset the entry and exit points if they are too close to the vertices
69  foreach (Vector2 vertex in shape.VerticesPrivate)
70  {
71  if (vertex.Equals(localEntryPoint))
72  {
73  localEntryPoint -= new Vector2(0, MathConstants.Epsilon);
74  }
75 
76  if (vertex.Equals(localExitPoint))
77  {
78  localExitPoint += new Vector2(0, MathConstants.Epsilon);
79  }
80  }
81 
82  Vertices vertices = new Vertices(shape.VerticesPrivate);
83  Vertices[] newPolygon = new Vertices[2];
84 
85  for (int i = 0; i < newPolygon.Length; i++)
86  {
87  newPolygon[i] = new Vertices(vertices.Count);
88  }
89 
90  int[] cutAdded = {-1, -1};
91  int last = -1;
92  for (int i = 0; i < vertices.Count; i++)
93  {
94  int n;
95 
96  //Find out if this vertex is on the old or new shape.
97  if (Vector2.Dot(MathUtils.Cross(localExitPoint - localEntryPoint, 1), vertices[i] - localEntryPoint) >
98  MathConstants.Epsilon)
99  {
100  n = 0;
101  }
102  else
103  {
104  n = 1;
105  }
106 
107  if (last != n)
108  {
109  //If we switch from one shape to the other add the cut vertices.
110  if (last == 0)
111  {
112  Debug.Assert(cutAdded[0] == -1);
113  cutAdded[0] = newPolygon[last].Count;
114  newPolygon[last].Add(localExitPoint);
115  newPolygon[last].Add(localEntryPoint);
116  }
117 
118  if (last == 1)
119  {
120  Debug.Assert(cutAdded[last] == -1);
121  cutAdded[last] = newPolygon[last].Count;
122  newPolygon[last].Add(localEntryPoint);
123  newPolygon[last].Add(localExitPoint);
124  }
125  }
126 
127  newPolygon[n].Add(vertices[i]);
128  last = n;
129  }
130 
131  //Add the cut in case it has not been added yet.
132  if (cutAdded[0] == -1)
133  {
134  cutAdded[0] = newPolygon[0].Count;
135  newPolygon[0].Add(localExitPoint);
136  newPolygon[0].Add(localEntryPoint);
137  }
138 
139  if (cutAdded[1] == -1)
140  {
141  cutAdded[1] = newPolygon[1].Count;
142  newPolygon[1].Add(localEntryPoint);
143  newPolygon[1].Add(localExitPoint);
144  }
145 
146  for (int n = 0; n < 2; n++)
147  {
148  Vector2 offset;
149  if (cutAdded[n] > 0)
150  {
151  offset = newPolygon[n][cutAdded[n] - 1] - newPolygon[n][cutAdded[n]];
152  }
153  else
154  {
155  offset = newPolygon[n][newPolygon[n].Count - 1] - newPolygon[n][0];
156  }
157 
158  offset = Vector2.Normalize(offset);
159 
160  if (!offset.IsValid())
161  {
162  offset = Vector2.One;
163  }
164 
165  newPolygon[n][cutAdded[n]] += MathConstants.Epsilon * offset;
166 
167  if (cutAdded[n] < newPolygon[n].Count - 2)
168  {
169  offset = newPolygon[n][cutAdded[n] + 2] - newPolygon[n][cutAdded[n] + 1];
170  }
171  else
172  {
173  offset = newPolygon[n][0] - newPolygon[n][newPolygon[n].Count - 1];
174  }
175 
176  offset = Vector2.Normalize(offset);
177 
178  if (!offset.IsValid())
179  {
180  offset = Vector2.One;
181  }
182 
183  newPolygon[n][cutAdded[n] + 1] += MathConstants.Epsilon * offset;
184  }
185 
186  first = newPolygon[0];
187  second = newPolygon[1];
188  }

198  public static bool Cut(World world, Vector2 start, Vector2 end)
199  {
200  List<Fixture> fixtures = new List<Fixture>();
201  List<Vector2> entryPoints = new List<Vector2>();
202  List<Vector2> exitPoints = new List<Vector2>();
203 
204  //We don't support cutting when the start or end is inside a shape.
205  if (world.TestPointAll(start) != null || world.TestPointAll(end) != null)
206  {
207  return false;
208  }
209 
210  //Get the entry points
211  world.RayCast((f, p, n, fr) =>
212  {
213  fixtures.Add(f);
214  entryPoints.Add(p);
215  return 1;
216  }, start, end);
217 
218  //Reverse the ray to get the exitpoints
219  world.RayCast((f, p, n, fr) =>
220  {
221  exitPoints.Add(p);
222  return 1;
223  }, end, start);
224 
225  //We only have a single point. We need at least 2
226  if (entryPoints.Count + exitPoints.Count < 2)
227  {
228  return false;
229  }
230 
231  for (int i = 0; i < fixtures.Count; i++)
232  {
233  // can't cut circles or edges yet !
234  if (fixtures[i].Shape.ShapeType != ShapeType.Polygon)
235  {
236  continue;
237  }
238 
239  if (fixtures[i].Body.BodyType != BodyType.Static)
240  {
241  //Split the shape up into two shapes
242  SplitShape(fixtures[i], entryPoints[i], exitPoints[i], out Vertices first, out Vertices second);
243 
244  //Delete the original shape and create two new. Retain the properties of the body.
245  if (first.CheckPolygon() == PolygonError.NoError)
246  {
247  Body firstFixture = BodyFactory.CreatePolygon(world, first, fixtures[i].Shape.DensityPrivate,
248  fixtures[i].Body.Position);
249  firstFixture.Rotation = fixtures[i].Body.Rotation;
250  firstFixture.LinearVelocity = fixtures[i].Body.LinearVelocity;
251  firstFixture.AngularVelocity = fixtures[i].Body.AngularVelocity;
252  firstFixture.BodyType = BodyType.Dynamic;
253  }
254 
255  if (second.CheckPolygon() == PolygonError.NoError)
256  {
257  Body secondFixture = BodyFactory.CreatePolygon(world, second, fixtures[i].Shape.DensityPrivate,
258  fixtures[i].Body.Position);
259  secondFixture.Rotation = fixtures[i].Body.Rotation;
260  secondFixture.LinearVelocity = fixtures[i].Body.LinearVelocity;
261  secondFixture.AngularVelocity = fixtures[i].Body.AngularVelocity;
262  secondFixture.BodyType = BodyType.Dynamic;
263  }
264 
265  world.RemoveBody(fixtures[i].Body);
266  }
267  }
268 
269  return true;
270  }
271  }
