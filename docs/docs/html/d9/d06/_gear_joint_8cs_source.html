30 // Gear Joint:
31 // C0 = (coordinate1 + ratio * coordinate2)_initial
32 // C = C0 - (cordinate1 + ratio * coordinate2) = 0
33 // Cdot = -(Cdot1 + ratio * Cdot2)
34 // J = -[J1 ratio * J2]
35 // K = J * invM * JT
36 // = J1 * invM1 * J1T + ratio * ratio * J2 * invM2 * J2T
37 //
38 // Revolute:
39 // coordinate = rotation
40 // Cdot = angularVelocity
41 // J = [0 0 1]
42 // K = J * invM * JT = invI
43 //
44 // Prismatic:
45 // coordinate = dot(p - pg, ug)
46 // Cdot = dot(v + cross(w, r), ug)
47 // J = [ug cross(r, ug)]
48 // K = J * invM * JT = invMass + invI * cross(r, ug)^2

255  internal override void InitVelocityConstraints(TimeStep step)
256  {
257  Body g1 = Ground1;
258  Body g2 = Ground2;
259  Body b1 = Body1;
260  Body b2 = Body2;
261 
262  float k = 0.0f;
263  jacobian.SetZero();
264 
265  if (Revolute1 != null)
266  {
267  jacobian.Angular1 = -1.0f;
268  k += b1.InvI;
269  }
270  else
271  {
272  Vector2 ug = Math.Mul(g1.GetXForm().R, Prismatic1.LocalXAxis1);
273  Vector2 r = Math.Mul(b1.GetXForm().R, LocalAnchor1 - b1.GetLocalCenter());
274  float crug = Vector2.Cross(r, ug);
275  jacobian.Linear1 = -ug;
276  jacobian.Angular1 = -crug;
277  k += b1.InvMass + b1.InvI * crug * crug;
278  }
279 
280  if (Revolute2 != null)
281  {
282  jacobian.Angular2 = -Ratio;
283  k += Ratio * Ratio * b2.InvI;
284  }
285  else
286  {
287  Vector2 ug = Math.Mul(g2.GetXForm().R, Prismatic2.LocalXAxis1);
288  Vector2 r = Math.Mul(b2.GetXForm().R, LocalAnchor2 - b2.GetLocalCenter());
289  float crug = Vector2.Cross(r, ug);
290  jacobian.Linear2 = -Ratio * ug;
291  jacobian.Angular2 = -Ratio * crug;
292  k += Ratio * Ratio * (b2.InvMass + b2.InvI * crug * crug);
293  }
294 
295  // Compute effective mass.
296  Box2DxDebug.Assert(k > 0.0f);
297  Mass = 1.0f / k;
298 
299  if (step.WarmStarting)
300  {
301  // Warm starting.
302  b1.LinearVelocity += b1.InvMass * Impulse * jacobian.Linear1;
303  b1.AngularVelocity += b1.InvI * Impulse * jacobian.Angular1;
304  b2.LinearVelocity += b2.InvMass * Impulse * jacobian.Linear2;
305  b2.AngularVelocity += b2.InvI * Impulse * jacobian.Angular2;
306  }
307  else
308  {
309  Impulse = 0.0f;
310  }
311  }
