130  private string GetSubProtocol(IList<string> requestedSubProtocols)
131  {
132  foreach (string subProtocol in requestedSubProtocols)

134  // match the first sub protocol that we support (the client should pass the most preferable sub protocols first)
135  if (_supportedSubProtocols.Contains(subProtocol))
136  {

156  private async Task ProcessTcpClientAsync(TcpClient tcpClient)
157  {
158  CancellationTokenSource source = new CancellationTokenSource();
159 
160  try
161  {
162  if (_isDisposed)
163  {
164  return;
165  }
166 
167  // this worker thread stays alive until either of the following happens:
168  // Client sends a close conection request OR
169  // An unhandled exception is thrown OR
170  // The server is disposed
171  Debug.Print("Server: Connection opened. Reading Http header from stream");
172 
173  // get a secure or insecure stream
174  Stream stream = tcpClient.GetStream();
175  WebSocketHttpContext context = await _webSocketServerFactory.ReadHttpHeaderFromStreamAsync(stream);
176  if (context.IsWebSocketRequest)
177  {
178  string subProtocol = GetSubProtocol(context.WebSocketRequestedProtocols);
179  WebSocketServerOptions options =
180  new WebSocketServerOptions(TimeSpan.FromSeconds(30),
181  subProtocol);
182  Debug.Print(
183  "Http header has requested an upgrade to Web Socket protocol. Negotiating Web Socket handshake");
184 
185  WebSocket webSocket = await _webSocketServerFactory.AcceptWebSocketAsync(context, options);
186 
187  Debug.Print("Web Socket handshake response sent. Stream ready.");
188  await RespondToWebSocketRequestAsync(webSocket, source.Token);
189  }
190  else
191  {
192  Debug.Print("Http header contains no web socket upgrade request. Ignoring");
193  }
194 
195  Debug.Print("Server: Connection closed");
196  }
197  catch (ObjectDisposedException)
198  {
199  // do nothing. This will be thrown if the Listener has been stopped
200  }
201  catch (Exception ex)
202  {
203  Debug.Print(ex.ToString());
204  }
205  finally
206  {
207  try
208  {
209  tcpClient.Client.Close();
210  tcpClient.Close();
211  source.Cancel();
212  }
213  catch (Exception ex)
214  {
215  Debug.Print($"Failed to close TCP connection: {ex}");
216  }
217  }
218  }

225  public async Task RespondToWebSocketRequestAsync(WebSocket webSocket, CancellationToken token)
226  {
227  ArraySegment<byte> buffer = new ArraySegment<byte>(new byte[BufferSize]);
228 
229  while (true)
230  {
231  WebSocketReceiveResult result = await webSocket.ReceiveAsync(buffer, token);
232  if (result.MessageType == WebSocketMessageType.Close)
233  {
234  Debug.Print(
235  $"Client initiated close. Status: {result.CloseStatus} Description: {result.CloseStatusDescription}");
236  break;
237  }
238 
239  if (result.Count > BufferSize)
240  {
241  await webSocket.CloseAsync(WebSocketCloseStatus.MessageTooBig,
242  $"Web socket frame cannot exceed buffer size of {BufferSize:#,##0} bytes. Send multiple frames instead.",
243  token);
244  break;
245  }
246 
247  // just echo the message back to the client
248  ArraySegment<byte> toSend = new ArraySegment<byte>(buffer.Array, buffer.Offset, result.Count);
249  await webSocket.SendAsync(toSend, WebSocketMessageType.Binary, true, token);
250  }
251  }
