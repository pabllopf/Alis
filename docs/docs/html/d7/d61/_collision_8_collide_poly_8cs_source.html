44  public static float EdgeSeparation(PolygonShape poly1, XForm xf1, int edge1, PolygonShape poly2, XForm xf2)
45  {
46  int count1 = poly1.VertexCount;
47  Vector2[] vertices1 = poly1.Vertices;
48  Vector2[] normals1 = poly1.Normals;
49 
50  int count2 = poly2.VertexCount;
51  Vector2[] vertices2 = poly2.Vertices;
52 
53  Box2DxDebug.Assert(0 <= edge1 && edge1 < count1);
54 
55  // Convert normal from poly1's frame into poly2's frame.
56  Vector2 normal1World = Math.Mul(xf1.R, normals1[edge1]);
57  Vector2 normal1 = Math.MulT(xf2.R, normal1World);
58 
59  // Find support vertex on poly2 for -normal.
60  int index = 0;
61  float minDot = Settings.FltMax;
62  for (int i = 0; i < count2; ++i)
63  {
64  float dot = Vector2.Dot(vertices2[i], normal1);
65  if (dot < minDot)
66  {
67  minDot = dot;
68  index = i;
69  }
70  }
71 
72  Vector2 v1 = Math.Mul(xf1, vertices1[edge1]);
73  Vector2 v2 = Math.Mul(xf2, vertices2[index]);
74  float separation = Vector2.Dot(v2 - v1, normal1World);
75  return separation;
76  }

81  public static float FindMaxSeparation(ref int edgeIndex, PolygonShape poly1, XForm xf1, PolygonShape poly2,
82  XForm xf2)
83  {
84  int count1 = poly1.VertexCount;
85  Vector2[] normals1 = poly1.Normals;
86 
87  // Vector pointing from the centroid of poly1 to the centroid of poly2.
88  Vector2 d = Math.Mul(xf2, poly2.Centroid) - Math.Mul(xf1, poly1.Centroid);
89  Vector2 dLocal1 = Math.MulT(xf1.R, d);
90 
91  // Find edge normal on poly1 that has the largest projection onto d.
92  int edge = 0;
93  float maxDot = -Settings.FltMax;
94  for (int i = 0; i < count1; ++i)
95  {
96  float dot = Vector2.Dot(normals1[i], dLocal1);
97  if (dot > maxDot)
98  {
99  maxDot = dot;
100  edge = i;
101  }
102  }
103 
104  // Get the separation for the edge normal.
105  float s = EdgeSeparation(poly1, xf1, edge, poly2, xf2);
106 
107  // Check the separation for the previous edge normal.
108  int prevEdge = edge - 1 >= 0 ? edge - 1 : count1 - 1;
109  float sPrev = EdgeSeparation(poly1, xf1, prevEdge, poly2, xf2);
110 
111  // Check the separation for the next edge normal.
112  int nextEdge = edge + 1 < count1 ? edge + 1 : 0;
113  float sNext = EdgeSeparation(poly1, xf1, nextEdge, poly2, xf2);
114 
115  // Find the best edge and the search direction.
116  int bestEdge;
117  float bestSeparation;
118  int increment;
119  if (sPrev > s && sPrev > sNext)
120  {
121  increment = -1;
122  bestEdge = prevEdge;
123  bestSeparation = sPrev;
124  }
125  else if (sNext > s)
126  {
127  increment = 1;
128  bestEdge = nextEdge;
129  bestSeparation = sNext;
130  }
131  else
132  {
133  edgeIndex = edge;
134  return s;
135  }
136 
137  // Perform a local search for the best edge normal.
138  for (;;)
139  {
140  if (increment == -1)
141  {
142  edge = bestEdge - 1 >= 0 ? bestEdge - 1 : count1 - 1;
143  }
144  else
145  {
146  edge = bestEdge + 1 < count1 ? bestEdge + 1 : 0;
147  }
148 
149  s = EdgeSeparation(poly1, xf1, edge, poly2, xf2);
150 
151  if (s > bestSeparation)
152  {
153  bestEdge = edge;
154  bestSeparation = s;
155  }
156  else
157  {
158  break;
159  }
160  }
161 
162  edgeIndex = bestEdge;
163  return bestSeparation;
164  }

175  public static void FindIncidentEdge(out ClipVertex[] c,
176  PolygonShape poly1, XForm xf1, int edge1, PolygonShape poly2, XForm xf2)
177  {
178  int count1 = poly1.VertexCount;
179  Vector2[] normals1 = poly1.Normals;
180 
181  int count2 = poly2.VertexCount;
182  Vector2[] vertices2 = poly2.Vertices;
183  Vector2[] normals2 = poly2.Normals;
184 
185  Box2DxDebug.Assert(0 <= edge1 && edge1 < count1);
186 
187  // Get the normal of the reference edge in poly2's frame.
188  Vector2 normal1 = Math.MulT(xf2.R, Math.Mul(xf1.R, normals1[edge1]));
189 
190  // Find the incident edge on poly2.
191  int index = 0;
192  float minDot = Settings.FltMax;
193  for (int i = 0; i < count2; ++i)
194  {
195  float dot = Vector2.Dot(normal1, normals2[i]);
196  if (dot < minDot)
197  {
198  minDot = dot;
199  index = i;
200  }
201  }
202 
203  // Build the clip vertices for the incident edge.
204  int i1 = index;
205  int i2 = i1 + 1 < count2 ? i1 + 1 : 0;
206 
207  c = new ClipVertex[2];
208 
209  c[0].V = Math.Mul(xf2, vertices2[i1]);
210  c[0].Id.Features.ReferenceEdge = (byte) edge1;
211  c[0].Id.Features.IncidentEdge = (byte) i1;
212  c[0].Id.Features.IncidentVertex = 0;
213 
214  c[1].V = Math.Mul(xf2, vertices2[i2]);
215  c[1].Id.Features.ReferenceEdge = (byte) edge1;
216  c[1].Id.Features.IncidentEdge = (byte) i2;
217  c[1].Id.Features.IncidentVertex = 1;
218  }

234  public static void CollidePolygons(ref Manifold manifold,
235  PolygonShape polyA, XForm xfA, PolygonShape polyB, XForm xfB)
236  {
237  manifold.PointCount = 0;
238  float totalRadius = polyA.Radius + polyB.Radius;
239 
240  int edgeA = 0;
241  float separationA = FindMaxSeparation(ref edgeA, polyA, xfA, polyB, xfB);
242  if (separationA > totalRadius)
243  {
244  return;
245  }
246 
247  int edgeB = 0;
248  float separationB = FindMaxSeparation(ref edgeB, polyB, xfB, polyA, xfA);
249  if (separationB > totalRadius)
250  {
251  return;
252  }
253 
254  PolygonShape poly1; // reference poly
255  PolygonShape poly2; // incident poly
256  XForm xf1, xf2;
257  int edge1; // reference edge
258  byte flip;
259  const float kRelativeTol = 0.98f;
260  const float kAbsoluteTol = 0.001f;
261 
262  if (separationB > kRelativeTol * separationA + kAbsoluteTol)
263  {
264  poly1 = polyB;
265  poly2 = polyA;
266  xf1 = xfB;
267  xf2 = xfA;
268  edge1 = edgeB;
269  manifold.Type = ManifoldType.FaceB;
270  flip = 1;
271  }
272  else
273  {
274  poly1 = polyA;
275  poly2 = polyB;
276  xf1 = xfA;
277  xf2 = xfB;
278  edge1 = edgeA;
279  manifold.Type = ManifoldType.FaceA;
280  flip = 0;
281  }
282 
283  ClipVertex[] incidentEdge;
284  FindIncidentEdge(out incidentEdge, poly1, xf1, edge1, poly2, xf2);
285 
286  int count1 = poly1.VertexCount;
287  Vector2[] vertices1 = poly1.Vertices;
288 
289  Vector2 v11 = vertices1[edge1];
290  Vector2 v12 = edge1 + 1 < count1 ? vertices1[edge1 + 1] : vertices1[0];
291 
292  Vector2 dv = v12 - v11;
293 
294  Vector2 localNormal = Vector2.Cross(dv, 1.0f);
295  localNormal.Normalize();
296  Vector2 planePoint = 0.5f * (v11 + v12);
297 
298  Vector2 sideNormal = Math.Mul(xf1.R, v12 - v11);
299  sideNormal.Normalize();
300  Vector2 frontNormal = Vector2.Cross(sideNormal, 1.0f);
301 
302  v11 = Math.Mul(xf1, v11);
303  v12 = Math.Mul(xf1, v12);
304 
305  float frontOffset = Vector2.Dot(frontNormal, v11);
306  float sideOffset1 = -Vector2.Dot(sideNormal, v11);
307  float sideOffset2 = Vector2.Dot(sideNormal, v12);
308 
309  // Clip incident edge against extruded edge1 side edges.
310  ClipVertex[] clipPoints1;
311  ClipVertex[] clipPoints2;
312  int np;
313 
314  // Clip to box side 1
315  np = ClipSegmentToLine(out clipPoints1, incidentEdge, -sideNormal, sideOffset1);
316 
317  if (np < 2)
318  {
319  return;
320  }
321 
322  // Clip to negative box side 1
323  np = ClipSegmentToLine(out clipPoints2, clipPoints1, sideNormal, sideOffset2);
324 
325  if (np < 2)
326  {
327  return;
328  }
329 
330  // Now clipPoints2 contains the clipped points.
331  manifold.LocalPlaneNormal = localNormal;
332  manifold.LocalPoint = planePoint;
333 
334  int pointCount = 0;
335  for (int i = 0; i < Settings.MaxManifoldPoints; ++i)
336  {
337  float separation = Vector2.Dot(frontNormal, clipPoints2[i].V) - frontOffset;
338 
339  if (separation <= totalRadius)
340  {
341  ManifoldPoint cp = manifold.Points[pointCount];
342  cp.LocalPoint = Math.MulT(xf2, clipPoints2[i].V);
343  cp.Id = clipPoints2[i].Id;
344  cp.Id.Features.Flip = flip;
345  ++pointCount;
346  }
347  }
348 
349  manifold.PointCount = pointCount;
350  }
351  }
