30 /*
31 This broad phase uses the Sweep and Prune algorithm as described in:
32 Collision Detection in Interactive 3D Environments by Gino van den Bergen
33 Also, some ideas, such as using integral values for fast compares comes from
34 Bullet (http:/www.bulletphysics.com).
35 */
36 
37 // Notes:
38 // - we use bound arrays instead of linked lists for cache coherence.
39 // - we use quantized integral values for fast compares.
40 // - we use short indices rather than pointers to save memory.
41 // - we use a stabbing count for fast overlap queries (less than order N).
42 // - we also use a time stamp on each proxy to speed up the registration of
43 // overlap query results.
44 // - where possible, we compare bound indices instead of values to reduce
45 // cache misses (TODO_ERIN).
46 // - no broadphase is perfect and neither is this one: it is not great for huge
47 // worlds (use a multi-SAP instead), it is not great for large objects.

146  public BroadPhase(Aabb worldAabb, PairCallback callback)
147  {
148  PairManager = new PairManager();
149  PairManager.Initialize(this, callback);
150 
151  Box2DxDebug.Assert(worldAabb.IsValid);
152  WorldAabb = worldAabb;
153  ProxyCount = 0;
154 
155  Vector2 d = worldAabb.UpperBound - worldAabb.LowerBound;
156  QuantizationFactor.X = BroadphaseMax / d.X;
157  QuantizationFactor.Y = BroadphaseMax / d.Y;
158 
159  for (ushort i = 0; i < Settings.MaxProxies - 1; ++i)
160  {
161  ProxyPool[i] = new Proxy();
162  ProxyPool[i].Next = (ushort) (i + 1);
163  ProxyPool[i].TimeStamp = 0;
164  ProxyPool[i].OverlapCount = Invalid;
165  ProxyPool[i].UserData = null;
166  }
167 
168  ProxyPool[Settings.MaxProxies - 1] = new Proxy();
169  ProxyPool[Settings.MaxProxies - 1].Next = PairManager.NullProxy;
170  ProxyPool[Settings.MaxProxies - 1].TimeStamp = 0;
171  ProxyPool[Settings.MaxProxies - 1].OverlapCount = Invalid;
172  ProxyPool[Settings.MaxProxies - 1].UserData = null;
173  freeProxy = 0;
174 
175  TimeStamp = 1;
176  QueryResultCount = 0;
177 
178  for (int i = 0; i < 2; i++)
179  {
180  Bounds[i] = new Bound[2 * Settings.MaxProxies];
181  }
182 
183  int bCount = 2 * Settings.MaxProxies;
184  for (int j = 0; j < 2; j++)
185  for (int k = 0; k < bCount; k++)
186  {
187  Bounds[j][k] = new Bound();

191  // Use this to see if your proxy is in range. If it is not in range,
192  // it should be destroyed. Otherwise you may get O(m^2) pairs, where m
193  // is the number of proxies that are out of range.
199  public bool InRange(Aabb aabb)
200  {
201  Vector2 d = Math.Max(aabb.LowerBound - WorldAabb.UpperBound, WorldAabb.LowerBound - aabb.UpperBound);
202  return Math.Max(d.X, d.Y) < 0.0f;

205  // Create and destroy proxies. These call Flush first.
212  public ushort CreateProxy(Aabb aabb, object userData)
213  {
214  Box2DxDebug.Assert(ProxyCount < Settings.MaxProxies);
215  Box2DxDebug.Assert(freeProxy != PairManager.NullProxy);
216 
217  ushort proxyId = freeProxy;
218  Proxy proxy1 = ProxyPool[proxyId];
219  freeProxy = proxy1.Next;
220 
221  proxy1.OverlapCount = 0;
222  proxy1.UserData = userData;
223 
224  int boundCount = 2 * ProxyCount;
225 
226  ushort[] lowerValues;
227  ushort[] upperValues;
228  ComputeBounds(out lowerValues, out upperValues, aabb);
229 
230  for (int axis = 0; axis < 2; ++axis)
231  {
232  Bound[] bounds = Bounds[axis];
233  int lowerIndex, upperIndex;
234  Query(out lowerIndex, out upperIndex, lowerValues[axis], upperValues[axis], bounds, boundCount, axis);
235 
236 
237  Bound[] tmp = new Bound[boundCount - upperIndex];
238  for (int i = 0; i < boundCount - upperIndex; i++)
239  {
240  tmp[i] = bounds[upperIndex + i].Clone();
241  }
242 
243  for (int i = 0; i < boundCount - upperIndex; i++)
244  {
245  bounds[upperIndex + 2 + i] = tmp[i];
246  }
247 
248 
249  tmp = new Bound[upperIndex - lowerIndex];
250  for (int i = 0; i < upperIndex - lowerIndex; i++)
251  {
252  tmp[i] = bounds[lowerIndex + i].Clone();
253  }
254 
255  for (int i = 0; i < upperIndex - lowerIndex; i++)
256  {
257  bounds[lowerIndex + 1 + i] = tmp[i];
258  }
259 
260  // The upper index has increased because of the lower bound insertion.
261  ++upperIndex;
262 
263  // Copy in the new bounds.
264  bounds[lowerIndex].Value = lowerValues[axis];
265  bounds[lowerIndex].ProxyId = proxyId;
266  bounds[upperIndex].Value = upperValues[axis];
267  bounds[upperIndex].ProxyId = proxyId;
268 
269  bounds[lowerIndex].StabbingCount = lowerIndex == 0 ? (ushort) 0 : bounds[lowerIndex - 1].StabbingCount;
270  bounds[upperIndex].StabbingCount = bounds[upperIndex - 1].StabbingCount;
271 
272  // Adjust the stabbing count between the new bounds.
273  for (int index = lowerIndex; index < upperIndex; ++index)
274  {
275  ++bounds[index].StabbingCount;
276  }
277 
278  // Adjust the all the affected bound indices.
279  for (int index = lowerIndex; index < boundCount + 2; ++index)
280  {
281  Proxy proxy = ProxyPool[bounds[index].ProxyId];
282  if (bounds[index].IsLower)
283  {
284  proxy.LowerBounds[axis] = (ushort) index;
285  }
286  else
287  {
288  proxy.UpperBounds[axis] = (ushort) index;
289  }
290  }
291  }
292 
293  ++ProxyCount;
294 
295  Box2DxDebug.Assert(QueryResultCount < Settings.MaxProxies);
296 
297  // Create pairs if the AABB is in range.
298  for (int i = 0; i < QueryResultCount; ++i)
299  {
300  Box2DxDebug.Assert(QueryResults[i] < Settings.MaxProxies);
301  Box2DxDebug.Assert(ProxyPool[QueryResults[i]].IsValid);
302 
303  PairManager.AddBufferedPair(proxyId, QueryResults[i]);
304  }
305 
306  PairManager.Commit();
307 
308  if (IsValidate)
309  {
310  Validate();
311  }
312 
313  // Prepare for next query.
314  QueryResultCount = 0;
315  IncrementTimeStamp();
316 
317  return proxyId;
318  }

324  public void DestroyProxy(int proxyId)
325  {
326  Box2DxDebug.Assert(0 < ProxyCount && ProxyCount <= Settings.MaxProxies);
327  Proxy proxy1 = ProxyPool[proxyId];
328  Box2DxDebug.Assert(proxy1.IsValid);
329 
330  int boundCount = 2 * ProxyCount;
331 
332  for (int axis = 0; axis < 2; ++axis)
333  {
334  Bound[] bounds = Bounds[axis];
335 
336  int lowerIndex = proxy1.LowerBounds[axis];
337  int upperIndex = proxy1.UpperBounds[axis];
338  ushort lowerValue = bounds[lowerIndex].Value;
339  ushort upperValue = bounds[upperIndex].Value;
340 
341 
342  Bound[] tmp = new Bound[upperIndex - lowerIndex - 1];
343  for (int i = 0; i < upperIndex - lowerIndex - 1; i++)
344  {
345  tmp[i] = bounds[lowerIndex + 1 + i].Clone();
346  }
347 
348  for (int i = 0; i < upperIndex - lowerIndex - 1; i++)
349  {
350  bounds[lowerIndex + i] = tmp[i];
351  }
352 
353 
354  tmp = new Bound[boundCount - upperIndex - 1];
355  for (int i = 0; i < boundCount - upperIndex - 1; i++)
356  {
357  tmp[i] = bounds[upperIndex + 1 + i].Clone();
358  }
359 
360  for (int i = 0; i < boundCount - upperIndex - 1; i++)
361  {
362  bounds[upperIndex - 1 + i] = tmp[i];
363  }
364 
365  // Fix bound indices.
366  for (int index = lowerIndex; index < boundCount - 2; ++index)
367  {
368  Proxy proxy = ProxyPool[bounds[index].ProxyId];
369  if (bounds[index].IsLower)
370  {
371  proxy.LowerBounds[axis] = (ushort) index;
372  }
373  else
374  {
375  proxy.UpperBounds[axis] = (ushort) index;
376  }
377  }
378 
379  // Fix stabbing count.
380  for (int index = lowerIndex; index < upperIndex - 1; ++index)
381  {
382  --bounds[index].StabbingCount;
383  }
384 
385  // Query for pairs to be removed. lowerIndex and upperIndex are not needed.
386  Query(out lowerIndex, out upperIndex, lowerValue, upperValue, bounds, boundCount - 2, axis);
387  }
388 
389  Box2DxDebug.Assert(QueryResultCount < Settings.MaxProxies);
390 
391  for (int i = 0; i < QueryResultCount; ++i)
392  {
393  Box2DxDebug.Assert(ProxyPool[QueryResults[i]].IsValid);
394  PairManager.RemoveBufferedPair(proxyId, QueryResults[i]);
395  }
396 
397  PairManager.Commit();
398 
399  // Prepare for next query.
400  QueryResultCount = 0;
401  IncrementTimeStamp();
402 
403  // Return the proxy to the pool.
404  proxy1.UserData = null;
405  proxy1.OverlapCount = Invalid;
406  proxy1.LowerBounds[0] = Invalid;
407  proxy1.LowerBounds[1] = Invalid;
408  proxy1.UpperBounds[0] = Invalid;
409  proxy1.UpperBounds[1] = Invalid;
410 
411  proxy1.Next = freeProxy;
412  freeProxy = (ushort) proxyId;
413  --ProxyCount;
414 
415  if (IsValidate)
416  {
417  Validate();
418  }
419  }

421  // Call MoveProxy as many times as you like, then when you are done
422  // call Commit to finalized the proxy pairs (for your time step).
428  public void MoveProxy(int proxyId, Aabb aabb)
429  {
430  if (proxyId == PairManager.NullProxy || Settings.MaxProxies <= proxyId)
431  {
432  Box2DxDebug.Assert(false);
433  return;
434  }
435 
436  if (aabb.IsValid == false)
437  {
438  Box2DxDebug.Assert(false);
439  return;
440  }
441 
442  int boundCount = 2 * ProxyCount;
443 
444  Proxy proxy1 = ProxyPool[proxyId];
445 
446  // Get new bound values
447  BoundValues newValues = new BoundValues();
448 
449  ComputeBounds(out newValues.LowerValues, out newValues.UpperValues, aabb);
450 
451  // Get old bound values
452  BoundValues oldValues = new BoundValues();
453  for (int axis = 0; axis < 2; ++axis)
454  {
455  oldValues.LowerValues[axis] = Bounds[axis][proxy1.LowerBounds[axis]].Value;
456  oldValues.UpperValues[axis] = Bounds[axis][proxy1.UpperBounds[axis]].Value;
457  }
458 
459  for (int axis = 0; axis < 2; ++axis)
460  {
461  Bound[] bounds = Bounds[axis];
462 
463  int lowerIndex = proxy1.LowerBounds[axis];
464  int upperIndex = proxy1.UpperBounds[axis];
465 
466  ushort lowerValue = newValues.LowerValues[axis];
467  ushort upperValue = newValues.UpperValues[axis];
468 
469  int deltaLower = lowerValue - bounds[lowerIndex].Value;
470  int deltaUpper = upperValue - bounds[upperIndex].Value;
471 
472  bounds[lowerIndex].Value = lowerValue;
473  bounds[upperIndex].Value = upperValue;
474 
475  //
476  // Expanding adds overlaps
477  //
478 
479  // Should we move the lower bound down?
480  if (deltaLower < 0)
481  {
482  int index = lowerIndex;
483  while (index > 0 && lowerValue < bounds[index - 1].Value)
484  {
485  Bound bound = bounds[index];
486  Bound prevBound = bounds[index - 1];
487 
488  int prevProxyId = prevBound.ProxyId;
489  Proxy prevProxy = ProxyPool[prevBound.ProxyId];
490 
491  ++prevBound.StabbingCount;
492 
493  if (prevBound.IsUpper)
494  {
495  if (TestOverlap(newValues, prevProxy))
496  {
497  PairManager.AddBufferedPair(proxyId, prevProxyId);
498  }
499 
500  ++prevProxy.UpperBounds[axis];
501  ++bound.StabbingCount;
502  }
503  else
504  {
505  ++prevProxy.LowerBounds[axis];
506  --bound.StabbingCount;
507  }
508 
509  --proxy1.LowerBounds[axis];
510  Math.Swap(ref bounds[index], ref bounds[index - 1]);
511  --index;
512  }
513  }
514 
515  // Should we move the upper bound up?
516  if (deltaUpper > 0)
517  {
518  int index = upperIndex;
519  while (index < boundCount - 1 && bounds[index + 1].Value <= upperValue)
520  {
521  Bound bound = bounds[index];
522  Bound nextBound = bounds[index + 1];
523  int nextProxyId = nextBound.ProxyId;
524  Proxy nextProxy = ProxyPool[nextProxyId];
525 
526  ++nextBound.StabbingCount;
527 
528  if (nextBound.IsLower)
529  {
530  if (TestOverlap(newValues, nextProxy))
531  {
532  PairManager.AddBufferedPair(proxyId, nextProxyId);
533  }
534 
535  --nextProxy.LowerBounds[axis];
536  ++bound.StabbingCount;
537  }
538  else
539  {
540  --nextProxy.UpperBounds[axis];
541  --bound.StabbingCount;
542  }
543 
544  ++proxy1.UpperBounds[axis];
545  Math.Swap(ref bounds[index], ref bounds[index + 1]);
546  ++index;
547  }
548  }
549 
550  //
551  // Shrinking removes overlaps
552  //
553 
554  // Should we move the lower bound up?
555  if (deltaLower > 0)
556  {
557  int index = lowerIndex;
558  while (index < boundCount - 1 && bounds[index + 1].Value <= lowerValue)
559  {
560  Bound bound = bounds[index];
561  Bound nextBound = bounds[index + 1];
562 
563  int nextProxyId = nextBound.ProxyId;
564  Proxy nextProxy = ProxyPool[nextProxyId];
565 
566  --nextBound.StabbingCount;
567 
568  if (nextBound.IsUpper)
569  {
570  if (TestOverlap(oldValues, nextProxy))
571  {
572  PairManager.RemoveBufferedPair(proxyId, nextProxyId);
573  }
574 
575  --nextProxy.UpperBounds[axis];
576  --bound.StabbingCount;
577  }
578  else
579  {
580  --nextProxy.LowerBounds[axis];
581  ++bound.StabbingCount;
582  }
583 
584  ++proxy1.LowerBounds[axis];
585  Math.Swap(ref bounds[index], ref bounds[index + 1]);
586  ++index;
587  }
588  }
589 
590  // Should we move the upper bound down?
591  if (deltaUpper < 0)
592  {
593  int index = upperIndex;
594  while (index > 0 && upperValue < bounds[index - 1].Value)
595  {
596  Bound bound = bounds[index];
597  Bound prevBound = bounds[index - 1];
598 
599  int prevProxyId = prevBound.ProxyId;
600  Proxy prevProxy = ProxyPool[prevProxyId];
601 
602  --prevBound.StabbingCount;
603 
604  if (prevBound.IsLower)
605  {
606  if (TestOverlap(oldValues, prevProxy))
607  {
608  PairManager.RemoveBufferedPair(proxyId, prevProxyId);
609  }
610 
611  ++prevProxy.LowerBounds[axis];
612  --bound.StabbingCount;
613  }
614  else
615  {
616  ++prevProxy.UpperBounds[axis];
617  ++bound.StabbingCount;
618  }
619 
620  --proxy1.UpperBounds[axis];
621  Math.Swap(ref bounds[index], ref bounds[index - 1]);
622  --index;
623  }
624  }
625  }
626 
627  if (IsValidate)
628  {
629  Validate();
630  }
631  }

657  // Query an AABB for overlapping proxies, returns the user data and
658  // the count, up to the supplied maximum count.
666  public int Query(Aabb aabb, object[] userData, int maxCount)
667  {
668  ushort[] lowerValues;
669  ushort[] upperValues;
670  ComputeBounds(out lowerValues, out upperValues, aabb);
671 
672  int lowerIndex, upperIndex;
673 
674  Query(out lowerIndex, out upperIndex, lowerValues[0], upperValues[0], Bounds[0], 2 * ProxyCount, 0);
675  Query(out lowerIndex, out upperIndex, lowerValues[1], upperValues[1], Bounds[1], 2 * ProxyCount, 1);
676 
677  Box2DxDebug.Assert(QueryResultCount < Settings.MaxProxies);
678 
679  int count = 0;
680  for (int i = 0; i < QueryResultCount && count < maxCount; ++i, ++count)
681  {
682  Box2DxDebug.Assert(QueryResults[i] < Settings.MaxProxies);
683  Proxy proxy = ProxyPool[QueryResults[i]];
684  Box2DxDebug.Assert(proxy.IsValid);
685  userData[i] = proxy.UserData;
686  }
687 
688  // Prepare for next query.
689  QueryResultCount = 0;
690  IncrementTimeStamp();
691 
692  return count;
693  }

710  public
711 #if ALLOWUNSAFE
712  unsafe
713 #endif //#if ALLOWUNSAFE
714  int QuerySegment(Segment segment, object[] userData, int maxCount, SortKeyFunc sortKey)
715  {
716  float maxLambda = 1;
717 
718  float dx = (segment.P2.X - segment.P1.X) * QuantizationFactor.X;
719  float dy = (segment.P2.Y - segment.P1.Y) * QuantizationFactor.Y;
720 
721  int sx = dx < -Settings.FltEpsilon ? -1 : dx > Settings.FltEpsilon ? 1 : 0;
722  int sy = dy < -Settings.FltEpsilon ? -1 : dy > Settings.FltEpsilon ? 1 : 0;
723 
724  Box2DxDebug.Assert(sx != 0 || sy != 0);
725 
726  float p1X = (segment.P1.X - WorldAabb.LowerBound.X) * QuantizationFactor.X;
727  float p1Y = (segment.P1.Y - WorldAabb.LowerBound.Y) * QuantizationFactor.Y;
728 #if ALLOWUNSAFE
729  ushort* startValues = stackalloc ushort[2];
730  ushort* startValues2 = stackalloc ushort[2];
731 #else
732  ushort[] startValues = new ushort[2];
733  ushort[] startValues2 = new ushort[2];
734 #endif
735 
736  int xIndex;
737  int yIndex;
738 
739  ushort proxyId;
740 
741  // TODO_ERIN implement fast float to ushort conversion.
742  startValues[0] = (ushort) ((ushort) p1X & (BroadphaseMax - 1));
743  startValues2[0] = (ushort) ((ushort) p1X | 1);
744 
745  startValues[1] = (ushort) ((ushort) p1Y & (BroadphaseMax - 1));
746  startValues2[1] = (ushort) ((ushort) p1Y | 1);
747 
748  //First deal with all the proxies that contain segment.p1
749  int lowerIndex;
750  int upperIndex;
751  Query(out lowerIndex, out upperIndex, startValues[0], startValues2[0], Bounds[0], 2 * ProxyCount, 0);
752  if (sx >= 0)
753  {
754  xIndex = upperIndex - 1;
755  }
756  else
757  {
758  xIndex = lowerIndex;
759  }
760 
761  Query(out lowerIndex, out upperIndex, startValues[1], startValues2[1], Bounds[1], 2 * ProxyCount, 1);
762  if (sy >= 0)
763  {
764  yIndex = upperIndex - 1;
765  }
766  else
767  {
768  yIndex = lowerIndex;
769  }
770 
771  //If we are using sortKey, then sort what we have so far, filtering negative keys
772  if (sortKey != null)
773  {
774  //Fill keys
775  for (int j = 0; j < QueryResultCount; j++)
776  {
777  QuerySortKeys[j] = sortKey(ProxyPool[QueryResults[j]].UserData);
778  }
779 
780  //Bubble sort keys
781  //Sorting negative values to the top, so we can easily remove them
782  int i = 0;
783  while (i < QueryResultCount - 1)
784  {
785  float a = QuerySortKeys[i];
786  float b = QuerySortKeys[i + 1];
787  if (a < 0 ? b >= 0 : a > b && b >= 0)
788  {
789  QuerySortKeys[i + 1] = a;
790  QuerySortKeys[i] = b;
791  ushort tempValue = QueryResults[i + 1];
792  QueryResults[i + 1] = QueryResults[i];
793  QueryResults[i] = tempValue;
794  i--;
795  if (i == -1)
796  {
797  i = 1;
798  }
799  }
800  else
801  {
802  i++;
803  }
804  }
805 
806  //Skim off negative values
807  while (QueryResultCount > 0 && QuerySortKeys[QueryResultCount - 1] < 0)
808  {
809  QueryResultCount--;
810  }
811  }
812 
813  //Now work through the rest of the segment
814  for (;;)
815  {
816  float xProgress = 0;
817  float yProgress = 0;
818  if (xIndex < 0 || xIndex >= ProxyCount * 2)
819  {
820  break;
821  }
822 
823  if (yIndex < 0 || yIndex >= ProxyCount * 2)
824  {
825  break;
826  }
827 
828  if (sx != 0)
829  {
830  //Move on to the next bound
831  if (sx > 0)
832  {
833  xIndex++;
834  if (xIndex == ProxyCount * 2)
835  {
836  break;
837  }
838  }
839  else
840  {
841  xIndex--;
842  if (xIndex < 0)
843  {
844  break;
845  }
846  }
847 
848  xProgress = (Bounds[0][xIndex].Value - p1X) / dx;
849  }
850 
851  if (sy != 0)
