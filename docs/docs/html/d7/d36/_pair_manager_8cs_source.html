147  /*
148  As proxies are created and moved, many pairs are created and destroyed. Even worse, the same
149  pair may be added and removed multiple times in a single time step of the physics engine. To reduce
150  traffic in the pair manager, we try to avoid destroying pairs in the pair manager until the
151  end of the physics step. This is done by buffering all the RemovePair requests. AddPair
152  requests are processed immediately because we need the hash table entry for quick lookup.
153 
154  All user user callbacks are delayed until the buffered pairs are confirmed in Commit.
155  This is very important because the user callbacks may be very expensive and client logic
156  may be harmed if pairs are added and removed within the same time step.
157 
158  Buffer a pair for addition.
159  We may add a pair that is not in the pair manager or pair buffer.
160  We may add a pair that is already in the pair manager and pair buffer.
161  If the added pair is not a new pair, then it must be in the pair buffer (because RemovePair was called).
162  */
168  public void AddBufferedPair(int id1, int id2)
169  {
170  Box2DxDebug.Assert(id1 != NullProxy && id2 != NullProxy);
171  Box2DxDebug.Assert(PairBufferCount < Settings.MaxPairs);
172 
173  Pair pair = AddPair(id1, id2);
174 
175  // If this pair is not in the pair buffer ...
176  if (pair.IsBuffered() == false)
177  {
178  // This must be a newly added pair.
179  Box2DxDebug.Assert(pair.IsFinal() == false);
180 
181  // Add it to the pair buffer.
182  pair.SetBuffered();
183  PairBuffer[PairBufferCount].ProxyId1 = pair.ProxyId1;
184  PairBuffer[PairBufferCount].ProxyId2 = pair.ProxyId2;
185  ++PairBufferCount;
186 
187  Box2DxDebug.Assert(PairBufferCount <= PairCount);
188  }
189 
190  // Confirm this pair for the subsequent call to Commit.
191  pair.ClearRemoved();
192 
193  if (BroadPhase.IsValidate)
194  {
195  ValidateBuffer();
196  }
197  }
198 
199  // Buffer a pair for removal.
205  public void RemoveBufferedPair(int id1, int id2)
206  {
207  Box2DxDebug.Assert(id1 != NullProxy && id2 != NullProxy);
208  Box2DxDebug.Assert(PairBufferCount < Settings.MaxPairs);
209 
210  Pair pair = Find(id1, id2);
211 
212  if (pair == null)
213  {
214  // The pair never existed. This is legal (due to collision filtering).
215  return;
216  }
217 
218  // If this pair is not in the pair buffer ...
219  if (pair.IsBuffered() == false)
220  {
221  // This must be an old pair.
222  Box2DxDebug.Assert(pair.IsFinal());
223 
224  pair.SetBuffered();
225  PairBuffer[PairBufferCount].ProxyId1 = pair.ProxyId1;
226  PairBuffer[PairBufferCount].ProxyId2 = pair.ProxyId2;
227  ++PairBufferCount;
228 
229  Box2DxDebug.Assert(PairBufferCount <= PairCount);
230  }
231 
232  pair.SetRemoved();
233 
234  if (BroadPhase.IsValidate)
235  {
236  ValidateBuffer();
237  }
238  }

243  public void Commit()
244  {
245  int removeCount = 0;
246 
247  Proxy[] proxies = BroadPhase.ProxyPool;
248 
249  for (int i = 0; i < PairBufferCount; ++i)
250  {
251  Pair pair = Find(PairBuffer[i].ProxyId1, PairBuffer[i].ProxyId2);
252  Box2DxDebug.Assert(pair.IsBuffered());
253  pair.ClearBuffered();
254 
255  Box2DxDebug.Assert(pair.ProxyId1 < Settings.MaxProxies && pair.ProxyId2 < Settings.MaxProxies);
256 
257  Proxy proxy1 = proxies[pair.ProxyId1];
258  Proxy proxy2 = proxies[pair.ProxyId2];
259 
260  Box2DxDebug.Assert(proxy1.IsValid);
261  Box2DxDebug.Assert(proxy2.IsValid);
262 
263  if (pair.IsRemoved())
264  {
265  // It is possible a pair was added then removed before a commit. Therefore,
266  // we should be careful not to tell the user the pair was removed when the
267  // the user didn't receive a matching add.
268  if (pair.IsFinal())
269  {
270  Callback.PairRemoved(proxy1.UserData, proxy2.UserData, pair.UserData);
271  }
272 
273  // Store the ids so we can actually remove the pair below.
274  PairBuffer[removeCount].ProxyId1 = pair.ProxyId1;
275  PairBuffer[removeCount].ProxyId2 = pair.ProxyId2;
276  ++removeCount;
277  }
278  else
279  {
280  Box2DxDebug.Assert(BroadPhase.TestOverlap(proxy1, proxy2));
281 
282  if (pair.IsFinal() == false)
283  {
284  pair.UserData = Callback.PairAdded(proxy1.UserData, proxy2.UserData);
285  pair.SetFinal();
286  }
287  }
288  }
289 
290  for (int i = 0; i < removeCount; ++i)
291  {
292  RemovePair(PairBuffer[i].ProxyId1, PairBuffer[i].ProxyId2);
293  }
294 
295  PairBufferCount = 0;
296 
297  if (BroadPhase.IsValidate)
298  {
299  ValidateTable();
300  }
301  }

309  private Pair Find(int proxyId1, int proxyId2)
310  {
311  if (proxyId1 > proxyId2)
312  {
313  Math.Swap(ref proxyId1, ref proxyId2);
314  }
315 
316  uint hash = (uint) (Hash((uint) proxyId1, (uint) proxyId2) & TableMask);
317 
318  return Find(proxyId1, proxyId2, hash);
319  }

328  private Pair Find(int proxyId1, int proxyId2, uint hash)
329  {
330  int index = HashTable[hash];
331 
332  while (index != NullPair && Equals(Pairs[index], proxyId1, proxyId2) == false)
333  {
334  index = Pairs[index].Next;
335  }
336 
337  if (index == NullPair)
338  {
339  return null;
340  }
341 
342  Box2DxDebug.Assert(index < Settings.MaxPairs);
343 
344  return Pairs[index];
345  }

347  // Returns existing pair or creates a new one.
354  private Pair AddPair(int proxyId1, int proxyId2)
355  {
356  if (proxyId1 > proxyId2)
357  {
358  Math.Swap(ref proxyId1, ref proxyId2);
359  }
360 
361  int hash = (int) (Hash((uint) proxyId1, (uint) proxyId2) & TableMask);
362 
363  Pair pair = Find(proxyId1, proxyId2, (uint) hash);
364  if (pair != null)
365  {
366  return pair;
367  }
368 
369  Box2DxDebug.Assert(PairCount < Settings.MaxPairs && FreePair != NullPair);
370 
371  ushort pairIndex = FreePair;
372  pair = Pairs[pairIndex];
373  FreePair = pair.Next;
374 
375  pair.ProxyId1 = (ushort) proxyId1;
376  pair.ProxyId2 = (ushort) proxyId2;
377  pair.Status = 0;
378  pair.UserData = null;
379  pair.Next = HashTable[hash];
380 
381  HashTable[hash] = pairIndex;
382 
383  ++PairCount;
384 
385  return pair;
386  }

388  // Removes a pair. The pair must exist.
395  private object RemovePair(int proxyId1, int proxyId2)
396  {
397  Box2DxDebug.Assert(PairCount > 0);
398 
399  if (proxyId1 > proxyId2)
400  {
401  Math.Swap(ref proxyId1, ref proxyId2);
402  }
403 
404  int hash = (int) (Hash((uint) proxyId1, (uint) proxyId2) & TableMask);
405 
406  //uint16* node = &m_hashTable[hash];
407  ushort node = HashTable[hash];
408  bool ion = false;
409  int ni = 0;
410  while (node != NullPair)
411  {
412  if (Equals(Pairs[node], proxyId1, proxyId2))
413  {
414  //uint16 index = *node;
415  //*node = m_pairs[*node].next;
416 
417  ushort index = node;
418  node = Pairs[node].Next;
419  if (ion)
420  {
421  Pairs[ni].Next = node;
422  }
423  else
424  {
425  HashTable[hash] = node;
426  }
427 
428  Pair pair = Pairs[index];
429  object userData = pair.UserData;
430 
431  // Scrub
432  pair.Next = FreePair;
433  pair.ProxyId1 = NullProxy;
434  pair.ProxyId2 = NullProxy;
435  pair.UserData = null;
436  pair.Status = 0;
437 
438  FreePair = index;
439  --PairCount;
440  return userData;
441  }
442 
443  //node = &m_pairs[*node].next;
444  ni = node;
445  node = Pairs[ni].Next;
446  ion = true;
447  }
448 
449  Box2DxDebug.Assert(false);
450  return null;
451  }

456  private void ValidateBuffer()
457  {
458 #if DEBUG
459  Box2DxDebug.Assert(PairBufferCount <= PairCount);
460 
461  //std::sort(m_pairBuffer, m_pairBuffer + m_pairBufferCount);
462  BufferedPair[] tmp = new BufferedPair[PairBufferCount];
463  Array.Copy(PairBuffer, 0, tmp, 0, PairBufferCount);

467  for (int i = 0; i < PairBufferCount; ++i)

524  // Thomas Wang's hash, see: http://www.concentric.net/~Ttwang/tech/inthash.htm
525  // This assumes proxyId1 and proxyId2 are 16-bit.
532  private uint Hash(uint proxyId1, uint proxyId2)
533  {
534  uint key = (proxyId2 << 16) | proxyId1;
535  key = ~key + (key << 15);
536  key = key ^ (key >> 12);
537  key = key + (key << 2);
538  key = key ^ (key >> 4);
539  key = key * 2057;
540  key = key ^ (key >> 16);
541  return key;
542  }

573  public static int BufferedPairSortPredicate(BufferedPair pair1, BufferedPair pair2)
574  {
575  if (pair1.ProxyId1 < pair2.ProxyId1)
576  {

581  {
582  return -1;

584

586  {
587  return 1;
588  }
