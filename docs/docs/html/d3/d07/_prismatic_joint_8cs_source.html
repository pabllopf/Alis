30 // Linear constraint (point-to-line)
31 // d = p2 - p1 = x2 + r2 - x1 - r1
32 // C = dot(perp, d)
33 // Cdot = dot(d, cross(w1, perp)) + dot(perp, v2 + cross(w2, r2) - v1 - cross(w1, r1))
34 // = -dot(perp, v1) - dot(cross(d + r1, perp), w1) + dot(perp, v2) + dot(cross(r2, perp), v2)
35 // J = [-perp, -cross(d + r1, perp), perp, cross(r2,perp)]
36 //
37 // Angular constraint
38 // C = a2 - a1 + a_initial
39 // Cdot = w2 - w1
40 // J = [0 0 -1 0 0 1]
41 //
42 // K = J * invM * JT
43 //
44 // J = [-a -s1 a s2]
45 // [0 -1 0 1]
46 // a = perp
47 // s1 = cross(d + r1, a) = cross(p2 - x1, a)
48 // s2 = cross(r2, a) = cross(p2 - x2, a)
49 
50 
51 // Motor/Limit linear constraint
52 // C = dot(ax1, d)
53 // Cdot = = -dot(ax1, v1) - dot(cross(d + r1, ax1), w1) + dot(ax1, v2) + dot(cross(r2, ax1), v2)
54 // J = [-ax1 -cross(d+r1,ax1) ax1 cross(r2,ax1)]
55 
56 // Block Solver
57 // We develop a block solver that includes the joint limit. This makes the limit stiff (inelastic) even
58 // when the mass has poor distribution (leading to large torques about the joint anchor points).
59 //
60 // The Jacobian has 3 rows:
61 // J = [-uT -s1 uT s2] // linear
62 // [0 -1 0 1] // angular
63 // [-vT -a1 vT a2] // limit
64 //
65 // u = perp
66 // v = axis
67 // s1 = cross(d + r1, u), s2 = cross(r2, u)
68 // a1 = cross(d + r1, v), a2 = cross(r2, v)
69 
70 // M * (v2 - v1) = JT * df
71 // J * v2 = bias
72 //
73 // v2 = v1 + invM * JT * df
74 // J * (v1 + invM * JT * df) = bias
75 // K * df = bias - J * v1 = -Cdot
76 // K = J * invM * JT
77 // Cdot = J * v1 - bias
78 //
79 // Now solve for f2.
80 // df = f2 - f1
81 // K * (f2 - f1) = -Cdot
82 // f2 = invK * (-Cdot) + f1
83 //
84 // Clamp accumulated limit impulse.
85 // lower: f2(3) = max(f2(3), 0)
86 // upper: f2(3) = min(f2(3), 0)
87 //
88 // Solve for correct f2(1:2)
89 // K(1:2, 1:2) * f2(1:2) = -Cdot(1:2) - K(1:2,3) * f2(3) + K(1:2,1:3) * f1
90 // = -Cdot(1:2) - K(1:2,3) * f2(3) + K(1:2,1:2) * f1(1:2) + K(1:2,3) * f1(3)
91 // K(1:2, 1:2) * f2(1:2) = -Cdot(1:2) - K(1:2,3) * (f2(3) - f1(3)) + K(1:2,1:2) * f1(1:2)
92 // f2(1:2) = invK(1:2,1:2) * (-Cdot(1:2) - K(1:2,3) * (f2(3) - f1(3))) + f1(1:2)
93 //
94 // Now compute impulse to be applied:
95 // df = f2 - f1

390  internal override void InitVelocityConstraints(TimeStep step)
391  {
392  Body b1 = Body1;
393  Body b2 = Body2;
394 
395  // You cannot create a prismatic joint between bodies that
396  // both have fixed rotation.
397  Box2DxDebug.Assert(b1.InvI > 0.0f || b2.InvI > 0.0f);
398 
399  LocalCenter1 = b1.GetLocalCenter();
400  LocalCenter2 = b2.GetLocalCenter();
401 
402  XForm xf1 = b1.GetXForm();
403  XForm xf2 = b2.GetXForm();
404 
405  // Compute the effective masses.
406  Vector2 r1 = Box2DXMath.Mul(xf1.R, LocalAnchor1 - LocalCenter1);
407  Vector2 r2 = Box2DXMath.Mul(xf2.R, LocalAnchor2 - LocalCenter2);
408  Vector2 d = b2.Sweep.C + r2 - b1.Sweep.C - r1;
409 
410  InvMass1 = b1.InvMass;
411  InvI1 = b1.InvI;
412  InvMass2 = b2.InvMass;
413  InvI2 = b2.InvI;
414 
415  // Compute motor Jacobian and effective mass.
416  {
417  Axis = Box2DXMath.Mul(xf1.R, LocalXAxis1);
418  a1 = Vector2.Cross(d + r1, Axis);
419  A2 = Vector2.Cross(r2, Axis);
420 
421  MotorMass = InvMass1 + InvMass2 + InvI1 * a1 * a1 + InvI2 * A2 * A2;
422  Box2DxDebug.Assert(MotorMass > Settings.FltEpsilon);
423  MotorMass = 1.0f / MotorMass;
424  }
425 
426  // Prismatic constraint.
427  {
428  Perp = Box2DXMath.Mul(xf1.R, LocalYAxis1);
429 
430  s1 = Vector2.Cross(d + r1, Perp);
431  s2 = Vector2.Cross(r2, Perp);
432 
433  float m1 = InvMass1, m2 = InvMass2;
434  float i1 = InvI1, i2 = InvI2;
435 
436  float k11 = m1 + m2 + i1 * s1 * s1 + i2 * s2 * s2;
437  float k12 = i1 * s1 + i2 * s2;
438  float k13 = i1 * s1 * a1 + i2 * s2 * A2;
439  float k22 = i1 + i2;
440  float k23 = i1 * a1 + i2 * A2;
441  float k33 = m1 + m2 + i1 * a1 * a1 + i2 * A2 * A2;
442 
443  K.Col1.Set(k11, k12, k13);
444  K.Col2.Set(k12, k22, k23);
445  K.Col3.Set(k13, k23, k33);
446  }
447 
448  // Compute motor and limit terms.
449  if (IsLimitEnabled)
450  {
451  float jointTranslation = Vector2.Dot(Axis, d);
452  if (Box2DXMath.Abs(UpperLimit - LowerLimit) < 2.0f * Settings.LinearSlop)
453  {
454  LimitState = LimitState.EqualLimits;
455  }
456  else if (jointTranslation <= LowerLimit)
457  {
458  if (LimitState != LimitState.AtLowerLimit)
459  {
460  LimitState = LimitState.AtLowerLimit;
461  Impulse.Z = 0.0f;
462  }
463  }
464  else if (jointTranslation >= UpperLimit)
465  {
466  if (LimitState != LimitState.AtUpperLimit)
467  {
468  LimitState = LimitState.AtUpperLimit;
469  Impulse.Z = 0.0f;
470  }
471  }
472  else
473  {
474  LimitState = LimitState.InactiveLimit;
475  Impulse.Z = 0.0f;
476  }
477  }
478  else
479  {
480  LimitState = LimitState.InactiveLimit;
481  }
482 
483  if (IsMotorEnabled == false)
484  {
485  MotorForce = 0.0f;
486  }
487 
488  if (step.WarmStarting)
489  {
490  // Account for variable time step.
491  Impulse *= step.DtRatio;
492  MotorForce *= step.DtRatio;
493 
494  Vector2 p = Impulse.X * Perp + (MotorForce + Impulse.Z) * Axis;
495  float l1 = Impulse.X * s1 + Impulse.Y + (MotorForce + Impulse.Z) * a1;
496  float l2 = Impulse.X * s2 + Impulse.Y + (MotorForce + Impulse.Z) * A2;
497 
498  b1.LinearVelocity -= InvMass1 * p;
499  b1.AngularVelocity -= InvI1 * l1;
500 
501  b2.LinearVelocity += InvMass2 * p;
502  b2.AngularVelocity += InvI2 * l2;
503  }
504  else
505  {
506  Impulse.SetZero();
507  MotorForce = 0.0f;
508  }
509  }

515  internal override void SolveVelocityConstraints(TimeStep step)
516  {
517  Body b1 = Body1;
518  Body b2 = Body2;
519 
520  Vector2 v1 = b1.LinearVelocity;
521  float w1 = b1.AngularVelocity;
522  Vector2 v2 = b2.LinearVelocity;
523  float w2 = b2.AngularVelocity;
524 
525  // Solve linear motor constraint.
526  if (IsMotorEnabled && LimitState != LimitState.EqualLimits)
527  {
528  float cdot = Vector2.Dot(Axis, v2 - v1) + A2 * w2 - a1 * w1;
529  float impulse = MotorMass * (motorSpeedx - cdot);
530  float oldImpulse = MotorForce;
531  float maxImpulse = step.Dt * MaxMotorForce;
532  MotorForce = Box2DXMath.Clamp(MotorForce + impulse, -maxImpulse, maxImpulse);
533  impulse = MotorForce - oldImpulse;
534 
535  Vector2 p = impulse * Axis;
536  float l1 = impulse * a1;
537  float l2 = impulse * A2;
538 
539  v1 -= InvMass1 * p;
540  w1 -= InvI1 * l1;
541 
542  v2 += InvMass2 * p;
543  w2 += InvI2 * l2;
544  }
545 
546  Vector2 cdot1;
547  cdot1.X = Vector2.Dot(Perp, v2 - v1) + s2 * w2 - s1 * w1;
548  cdot1.Y = w2 - w1;
549 
550  if (IsLimitEnabled && LimitState != LimitState.InactiveLimit)
551  {
552  // Solve prismatic and limit constraint in block form.
553  float cdot2;
554  cdot2 = Vector2.Dot(Axis, v2 - v1) + A2 * w2 - a1 * w1;
555  Vector3 cdot = new Vector3(cdot1.X, cdot1.Y, cdot2);
556 
557  Vector3 f1 = Impulse;
558  Vector3 df = K.Solve33(-cdot);
559  Impulse += df;
560 
561  if (LimitState == LimitState.AtLowerLimit)
562  {
563  Impulse.Z = Box2DXMath.Max(Impulse.Z, 0.0f);
564  }
565  else if (LimitState == LimitState.AtUpperLimit)
566  {
567  Impulse.Z = Box2DXMath.Min(Impulse.Z, 0.0f);
568  }
569 
570  // f2(1:2) = invK(1:2,1:2) * (-Cdot(1:2) - K(1:2,3) * (f2(3) - f1(3))) + f1(1:2)
571  Vector2 b = -cdot1 - (Impulse.Z - f1.Z) * new Vector2(K.Col3.X, K.Col3.Y);
572  Vector2 f2R = K.Solve22(b) + new Vector2(f1.X, f1.Y);
573  Impulse.X = f2R.X;
574  Impulse.Y = f2R.Y;
575 
576  df = Impulse - f1;
577 
578  Vector2 p = df.X * Perp + df.Z * Axis;
579  float l1 = df.X * s1 + df.Y + df.Z * a1;
580  float l2 = df.X * s2 + df.Y + df.Z * A2;
581 
582  v1 -= InvMass1 * p;
583  w1 -= InvI1 * l1;
584 
585  v2 += InvMass2 * p;
586  w2 += InvI2 * l2;
587  }
588  else
589  {
590  // Limit is inactive, just solve the prismatic constraint in block form.
591  Vector2 df = K.Solve22(-cdot1);
592  Impulse.X += df.X;
593  Impulse.Y += df.Y;
594 
595  Vector2 p = df.X * Perp;
596  float l1 = df.X * s1 + df.Y;
597  float l2 = df.X * s2 + df.Y;
598 
599  v1 -= InvMass1 * p;
600  w1 -= InvI1 * l1;
601 
602  v2 += InvMass2 * p;
603  w2 += InvI2 * l2;
604  }
605 
606  b1.LinearVelocity = v1;
607  b1.AngularVelocity = w1;
608  b2.LinearVelocity = v2;
609  b2.AngularVelocity = w2;
610  }

617  internal override bool SolvePositionConstraints(float baumgarte)
618  {
619  Body body1 = Body1;
620  Body body2 = Body2;
621 
622  Vector2 body1SweepC = body1.Sweep.C;
623  float body1SweepA = body1.Sweep.A;
624 
625  Vector2 body2SweepC = body2.Sweep.C;
626  float body2SweepA = body2.Sweep.A;
627 
628  // Solve linear limit constraint.
629  var linearError = 0.0f;
630  float angularError;
631  bool active = false;
632  float c2 = 0.0f;
633 
634  var mat22R1 = new Matrix22(body1SweepA);
635  var mat22R2 = new Matrix22(body2SweepA);
636 
637  Vector2 r1 = Box2DXMath.Mul(mat22R1, LocalAnchor1 - LocalCenter1);
638  Vector2 r2 = Box2DXMath.Mul(mat22R2, LocalAnchor2 - LocalCenter2);
639  Vector2 distance = body2SweepC + r2 - body1SweepC - r1;
640 
641  if (IsLimitEnabled)
642  {
643  Axis = Box2DXMath.Mul(mat22R1, LocalXAxis1);
644 
645  a1 = Vector2.Cross(distance + r1, Axis);
646  A2 = Vector2.Cross(r2, Axis);
647 
648  float translation = Vector2.Dot(Axis, distance);
649  if (Box2DXMath.Abs(UpperLimit - LowerLimit) < 2.0f * Settings.LinearSlop)
650  {
651  // Prevent large angular corrections
652  c2 = Box2DXMath.Clamp(translation, -Settings.MaxLinearCorrection, Settings.MaxLinearCorrection);
653  linearError = Box2DXMath.Abs(translation);
654  active = true;
655  }
656  else if (translation <= LowerLimit)
657  {
658  // Prevent large linear corrections and allow some slop.
659  c2 = Box2DXMath.Clamp(translation - LowerLimit + Settings.LinearSlop,
660  -Settings.MaxLinearCorrection, 0.0f);
661  linearError = LowerLimit - translation;
662  active = true;
663  }
664  else if (translation >= UpperLimit)
665  {
666  // Prevent large linear corrections and allow some slop.
667  c2 = Box2DXMath.Clamp(translation - UpperLimit - Settings.LinearSlop, 0.0f,
668  Settings.MaxLinearCorrection);
669  linearError = translation - UpperLimit;
670  active = true;
671  }
672  }
673 
674  Perp = Box2DXMath.Mul(mat22R1, LocalYAxis1);
675 
676  s1 = Vector2.Cross(distance + r1, Perp);
677  s2 = Vector2.Cross(r2, Perp);
678 
679  Vector3 impulse;
680  Vector2 c1 = new Vector2();
681  c1.X = Vector2.Dot(Perp, distance);
682  c1.Y = body2SweepA - body1SweepA - refAngle;
683 
684  linearError = Box2DXMath.Max(linearError, Box2DXMath.Abs(c1.X));
685  angularError = Box2DXMath.Abs(c1.Y);
686 
687  if (active)
688  {
689  float m1 = InvMass1, m2 = InvMass2;
690  float i1 = InvI1, i2 = InvI2;
691 
692  float k11 = m1 + m2 + i1 * s1 * s1 + i2 * s2 * s2;
693  float k12 = i1 * s1 + i2 * s2;
694  float k13 = i1 * s1 * a1 + i2 * s2 * A2;
695  float k22 = i1 + i2;
696  float k23 = i1 * a1 + i2 * A2;
697  float k33 = m1 + m2 + i1 * a1 * a1 + i2 * A2 * A2;
698 
699  K.Col1.Set(k11, k12, k13);
700  K.Col2.Set(k12, k22, k23);
701  K.Col3.Set(k13, k23, k33);
702 
703  Vector3 c = new Vector3();
704  c.X = c1.X;
705  c.Y = c1.Y;
706  c.Z = c2;
707 
708  impulse = K.Solve33(-c);
709  }
710  else
711  {
712  float m1 = InvMass1, m2 = InvMass2;
713  float i1 = InvI1, i2 = InvI2;
714 
715  float k11 = m1 + m2 + i1 * s1 * s1 + i2 * s2 * s2;
716  float k12 = i1 * s1 + i2 * s2;
717  float k22 = i1 + i2;
718 
719  K.Col1.Set(k11, k12, 0.0f);
720  K.Col2.Set(k12, k22, 0.0f);
721 
722  Vector2 impulse1 = K.Solve22(-c1);
723  impulse.X = impulse1.X;
724  impulse.Y = impulse1.Y;
725  impulse.Z = 0.0f;
726  }
727 
728  Vector2 p = impulse.X * Perp + impulse.Z * Axis;
729  float l1 = impulse.X * s1 + impulse.Y + impulse.Z * a1;
730  float l2 = impulse.X * s2 + impulse.Y + impulse.Z * A2;
731 
732  body1SweepC -= InvMass1 * p;
733  body1SweepA -= InvI1 * l1;
734  body2SweepC += InvMass2 * p;
735  body2SweepA += InvI2 * l2;
736 
737  // TODO_ERIN remove need for this.
738  body1.Sweep.C = body1SweepC;
739  body1.Sweep.A = body1SweepA;
740  body2.Sweep.C = body2SweepC;
741  body2.Sweep.A = body2SweepA;
742  body1.SynchronizeTransform();
743  body2.SynchronizeTransform();
744 
745  return linearError <= Settings.LinearSlop && angularError <= Settings.AngularSlop;
746  }
747  }
