40  // This implements 2-sided edge vs circle collision.
49  public static void CollideEdgeAndCircle(ref Manifold manifold, EdgeShape edge, XForm transformA,
50  CircleShape circle, XForm transformB)
51  {
52  manifold.PointCount = 0;
53  Vector2 cLocal = Math.MulT(transformA, Math.Mul(transformB, circle.Position));
54  Vector2 normal = edge.NormalVector;
55  Vector2 v1 = edge.Vertex1;
56  Vector2 v2 = edge.Vertex2;
57  float radius = edge.Radius + circle.Radius;
58 
59  // Barycentric coordinates
60  float u1 = Vector2.Dot(cLocal - v1, v2 - v1);
61  float u2 = Vector2.Dot(cLocal - v2, v1 - v2);
62 
63  if (u1 <= 0.0f)
64  {
65  // Behind v1
66  if (Vector2.DistanceSquared(cLocal, v1) > radius * radius)
67  {
68  return;
69  }
70 
71  manifold.PointCount = 1;
72  manifold.Type = ManifoldType.FaceA;
73  manifold.LocalPlaneNormal = cLocal - v1;
74  manifold.LocalPlaneNormal.Normalize();
75  manifold.LocalPoint = v1;
76  manifold.Points[0].LocalPoint = circle.Position;
77  manifold.Points[0].Id.Key = 0;
78  }
79  else if (u2 <= 0.0f)
80  {
81  // Ahead of v2
82  if (Vector2.DistanceSquared(cLocal, v2) > radius * radius)
83  {
84  return;
85  }
86 
87  manifold.PointCount = 1;
88  manifold.Type = ManifoldType.FaceA;
89  manifold.LocalPlaneNormal = cLocal - v2;
90  manifold.LocalPlaneNormal.Normalize();
91  manifold.LocalPoint = v2;
92  manifold.Points[0].LocalPoint = circle.Position;
93  manifold.Points[0].Id.Key = 0;
94  }
95  else
96  {
97  float separation = Vector2.Dot(cLocal - v1, normal);
98  if (separation < -radius || radius < separation)
99  {
100  return;
101  }
102 
103  manifold.PointCount = 1;
104  manifold.Type = ManifoldType.FaceA;
105  manifold.LocalPlaneNormal = separation < 0.0f ? -normal : normal;
106  manifold.LocalPoint = 0.5f * (v1 + v2);
107  manifold.Points[0].LocalPoint = circle.Position;
108  manifold.Points[0].Id.Key = 0;
109  }
110  }
