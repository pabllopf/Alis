40  public class Program

46  public static void Main(string[] args)
47  {
48  // Define the size of the world. Simulation will still work
49  // if bodies reach the end of the world, but it will be slower.
50  Aabb worldAabb = new Aabb();
51  worldAabb.LowerBound.Set(-100.0f);
52  worldAabb.UpperBound.Set(100.0f);
53 
54  // Define the gravity vector.
55  Vector2 gravity = new Vector2(0.0f, -10.0f);
56 
57  // Do we want to let bodies sleep?
58  const bool doSleep = true;
59 
60  // Construct a world object, which will hold and simulate the rigid bodies.
61  World world = new World(worldAabb, gravity, doSleep);
62 
63  // Define the ground body.
64  BodyDef groundBodyDef = new BodyDef();
65  groundBodyDef.Position.Set(0.0f, -10.0f);
66 
67  // Call the body factory which creates the ground box shape.
68  // The body is also added to the world.
69  Body groundBody = world.CreateBody(groundBodyDef);
70 
71  // Define the ground box shape.
72  PolygonDef groundShapeDef = new PolygonDef();
73 
74  // The extents are the half-widths of the box.
75  groundShapeDef.SetAsBox(50.0f, 10.0f);
76 
77  // Add the ground shape to the ground body.
78  groundBody.CreateFixture(groundShapeDef);
79 
80  // Define the dynamic body. We set its position and call the body factory.
81  BodyDef bodyDef = new BodyDef();
82  bodyDef.Position.Set(0.0f, 4.0f);
83  Body body = world.CreateBody(bodyDef);
84 
85  // Define another box shape for our dynamic body.
86  PolygonDef shapeDef = new PolygonDef();
87  shapeDef.SetAsBox(1.0f, 1.0f);
88 
89  // Set the box density to be non-zero, so it will be dynamic.
90  shapeDef.Density = 1.0f;
91 
92  // Override the default friction.
93  shapeDef.Friction = 0.3f;
94 
95  // Add the shape to the body.
96  body.CreateFixture(shapeDef);
97 
98  // Now tell the dynamic body to compute it's mass properties base
99  // on its shape.
100  body.SetMassFromShapes();
101 
102  // Prepare for simulation. Typically we use a time step of 1/60 of a
103  // second (60Hz) and 10 iterations. This provides a high quality simulation
104  // in most game scenarios.
105  const float timeStep = 1.0f / 60.0f;
106  const int velocityIterations = 8;
107  const int positionIterations = 1;
108 
109  // This is our little game loop.
110  for (int i = 0; i < 100; ++i)
111  {
112  // Instruct the world to perform a single step of simulation. It is
113  // generally best to keep the time step and iterations fixed.
114  world.Step(timeStep, velocityIterations, positionIterations);
115 
116  // Now print the position and angle of the body.
117  Vector2 position = body.GetPosition();
118  float angle = body.GetAngle();
119 
120  Console.WriteLine(
121  "Step: {3} - X: {0}, Y: {1}, Angle: {2}", position.X.ToString(), position.Y.ToString(),
122  angle.ToString(), i.ToString());
123  }
124 
125  // When the world destructor is called, all bodies and joints are freed. This can
126  // create orphaned pointers, so be careful about your world management.
127 
128  //Console.ReadLine();
129  }
130  }
