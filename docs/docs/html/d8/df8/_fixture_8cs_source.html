119  public void Create(BroadPhase broadPhase, Body body, XForm xf, FixtureDef def)
120  {
121  UserData = def.UserData;
122  Friction = def.Friction;
123  Restitution = def.Restitution;
124  Density = def.Density;
125 
126  Body = body;
127  Next = null;
128 
129  Filter = def.Filter;
130 
131  IsSensor = def.IsSensor;
132 
133  ShapeType = def.Type;
134 
135  // Allocate and initialize the child shape.
136  switch (ShapeType)
137  {
138  case ShapeType.CircleShape:
139  {
140  CircleShape circle = new CircleShape();
141  CircleDef circleDef = (CircleDef) def;
142  circle.Position = circleDef.LocalPosition;
143  circle.Radius = circleDef.Radius;
144  Shape = circle;
145  }
146  break;
147 
148  case ShapeType.PolygonShape:
149  {
150  PolygonShape polygon = new PolygonShape();
151  PolygonDef polygonDef = (PolygonDef) def;
152  polygon.Set(polygonDef.Vertices, polygonDef.VertexCount);
153  Shape = polygon;
154  }
155  break;
156 
157  case ShapeType.EdgeShape:
158  {
159  EdgeShape edge = new EdgeShape();
160  EdgeDef edgeDef = (EdgeDef) def;
161  edge.Set(edgeDef.Vertex1, edgeDef.Vertex2);
162  Shape = edge;
163  }
164  break;
165 
166  default:
167  Box2DxDebug.Assert(false);
168  break;
169  }
170 
171  // Create proxy in the broad-phase.
172  Aabb aabb;
173  Shape.ComputeAabb(out aabb, xf);
174 
175  bool inRange = broadPhase.InRange(aabb);
176 
177  // You are creating a shape outside the world box.
178  Box2DxDebug.Assert(inRange);
179 
180  if (inRange)
181  {
182  ProxyId = broadPhase.CreateProxy(aabb, this);
183  }
184  else
185  {
186  ProxyId = PairManager.NullProxy;
187  }
188  }
189 
194  public void Destroy(BroadPhase broadPhase)
195  {
196  // Remove proxy from the broad-phase.
197  if (ProxyId != PairManager.NullProxy)
198  {
199  broadPhase.DestroyProxy(ProxyId);
200  ProxyId = PairManager.NullProxy;
201  }
202 
203  // Free the child shape.
204  Shape.Dispose();
205  Shape = null;
206  }

215  internal bool Synchronize(BroadPhase broadPhase, XForm transform1, XForm transform2)
216  {
217  if (ProxyId == PairManager.NullProxy)
218  {
219  return false;
220  }
221 
222  // Compute an AABB that covers the swept shape (may miss some rotation effect).
223  Aabb aabb1, aabb2;
224  Shape.ComputeAabb(out aabb1, transform1);
225  Shape.ComputeAabb(out aabb2, transform2);
226 
227  Aabb aabb = new Aabb();
228  aabb.Combine(aabb1, aabb2);
229 
230  if (broadPhase.InRange(aabb))
231  {
232  broadPhase.MoveProxy(ProxyId, aabb);
233  return true;
234  }
235 
236  return false;
237  }
238 
244  internal void RefilterProxy(BroadPhase broadPhase, XForm transform)
245  {
246  if (ProxyId == PairManager.NullProxy)
247  {
248  return;
249  }
250 
251  broadPhase.DestroyProxy(ProxyId);
252 
253  Aabb aabb;
254  Shape.ComputeAabb(out aabb, transform);
255 
256  bool inRange = broadPhase.InRange(aabb);
257 
258  if (inRange)
259  {
260  ProxyId = broadPhase.CreateProxy(aabb, this);
261  }
262  else
263  {
264  ProxyId = PairManager.NullProxy;
265  }
266  }
