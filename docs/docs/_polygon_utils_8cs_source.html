89  public static Vertices CreateRoundedRectangle(float width, float height, float xRadius, float yRadius,
90  int segments)
91  {
92  if (yRadius > height / 2 || xRadius > width / 2)
93  {
94  throw new Exception("Rounding amount can't be more than half the height and width respectively.");
95  }
96 
97  if (segments < 0)
98  {
99  throw new Exception("Segments must be zero or more.");
100  }
101 
102  //TODO: TBM
103  //We need at least 8 vertices to create a rounded rectangle
104  //Debug.Assert(Settings.MaxPolygonVertices >= 8);
105 
106  Vertices vertices = new Vertices();
107  if (segments == 0)
108  {
109  vertices.Add(new Vector2(width * .5f - xRadius, -height * .5f));
110  vertices.Add(new Vector2(width * .5f, -height * .5f + yRadius));
111 
112  vertices.Add(new Vector2(width * .5f, height * .5f - yRadius));
113  vertices.Add(new Vector2(width * .5f - xRadius, height * .5f));
114 
115  vertices.Add(new Vector2(-width * .5f + xRadius, height * .5f));
116  vertices.Add(new Vector2(-width * .5f, height * .5f - yRadius));
117 
118  vertices.Add(new Vector2(-width * .5f, -height * .5f + yRadius));
119  vertices.Add(new Vector2(-width * .5f + xRadius, -height * .5f));
120  }
121  else
122  {
123  int numberOfEdges = segments * 4 + 8;
124 
125  float stepSize = MathConstants.TwoPi / (numberOfEdges - 4);
126  int perPhase = numberOfEdges / 4;
127 
128  Vector2 posOffset = new Vector2(width / 2 - xRadius, height / 2 - yRadius);
129  vertices.Add(posOffset + new Vector2(xRadius, -yRadius + yRadius));
130  short phase = 0;
131  for (int i = 1; i < numberOfEdges; i++)
132  {
133  if (i - perPhase == 0 || i - perPhase * 3 == 0)
134  {
135  posOffset.X *= -1;
136  phase--;
137  }
138  else if (i - perPhase * 2 == 0)
139  {
140  posOffset.Y *= -1;
141  phase--;
142  }
143 
144  vertices.Add(posOffset + new Vector2(xRadius * (float) Math.Cos(stepSize * -(i + phase)),
145  -yRadius * (float) Math.Sin(stepSize * -(i + phase))));
146  }
147  }
148 
149  return vertices;
150  }

178  public static Vertices CreateEllipse(float xRadius, float yRadius, int numberOfEdges)
179  {
180  Vertices vertices = new Vertices();
181 
182  float stepSize = MathConstants.TwoPi / numberOfEdges;
183 
184  vertices.Add(new Vector2(xRadius, 0));

186  {
187  vertices.Add(new Vector2(xRadius * (float) Math.Cos(stepSize * i),
188  -yRadius * (float) Math.Sin(stepSize * i)));

251  public static Vertices CreateCapsule(float height, float topRadius, int topEdges, float bottomRadius,
252  int bottomEdges)
253  {
254  if (height <= 0)
255  {
256  throw new ArgumentException("Height must be longer than 0", nameof(height));
257  }
258 
259  if (topRadius <= 0)
260  {
261  throw new ArgumentException("The top radius must be more than 0", nameof(topRadius));
262  }
263 
264  if (topEdges <= 0)
265  {
266  throw new ArgumentException("Top edges must be more than 0", nameof(topEdges));
267  }
268 
269  if (bottomRadius <= 0)
270  {
271  throw new ArgumentException("The bottom radius must be more than 0", nameof(bottomRadius));
272  }
273 
274  if (bottomEdges <= 0)
275  {
276  throw new ArgumentException("Bottom edges must be more than 0", nameof(bottomEdges));
277  }
278 
279  if (topRadius >= height / 2)
280  {
281  throw new ArgumentException(
282  "The top radius must be lower than height / 2. Higher values of top radius would create a circle, and not a half circle.",
283  nameof(topRadius));
284  }
285 
286  if (bottomRadius >= height / 2)
287  {
288  throw new ArgumentException(
289  "The bottom radius must be lower than height / 2. Higher values of bottom radius would create a circle, and not a half circle.",
290  nameof(bottomRadius));
291  }
292 
293  Vertices vertices = new Vertices();
294 
295  float newHeight = (height - topRadius - bottomRadius) * 0.5f;
296 
297  // top
298  vertices.Add(new Vector2(topRadius, newHeight));
299 
300  float stepSize = MathConstants.Pi / topEdges;
301  for (int i = 1; i < topEdges; i++)
302  {
303  vertices.Add(new Vector2(topRadius * (float) Math.Cos(stepSize * i),
304  topRadius * (float) Math.Sin(stepSize * i) + newHeight));
305  }
306 
307  vertices.Add(new Vector2(-topRadius, newHeight));
308 
309  // bottom
310  vertices.Add(new Vector2(-bottomRadius, -newHeight));
311 
312  stepSize = MathConstants.Pi / bottomEdges;
313  for (int i = 1; i < bottomEdges; i++)
314  {
315  vertices.Add(new Vector2(-bottomRadius * (float) Math.Cos(stepSize * i),
316  -bottomRadius * (float) Math.Sin(stepSize * i) - newHeight));
317  }
318 
319  vertices.Add(new Vector2(bottomRadius, -newHeight));
320 
321  return vertices;
322  }

329  public static Vertices CreateGear(float radius, int numberOfTeeth, float tipPercentage, float toothHeight)
330  {
331  Vertices vertices = new Vertices();
332 
333  float stepSize = MathConstants.TwoPi / numberOfTeeth;
334  tipPercentage /= 100f;
335  MathHelper.Clamp(tipPercentage, 0f, 1f);
336  float toothTipStepSize = stepSize / 2f * tipPercentage;
337 
338  float toothAngleStepSize = (stepSize - toothTipStepSize * 2f) / 2f;
339 
340  for (int i = numberOfTeeth - 1; i >= 0; --i)
341  {
342  if (toothTipStepSize > 0f)
343  {
344  vertices.Add(
345  new Vector2(radius *
346  (float) Math.Cos(stepSize * i + toothAngleStepSize * 2f + toothTipStepSize),
347  -radius *
348  (float) Math.Sin(stepSize * i + toothAngleStepSize * 2f + toothTipStepSize)));
349 
350  vertices.Add(
351  new Vector2((radius + toothHeight) *
352  (float) Math.Cos(stepSize * i + toothAngleStepSize + toothTipStepSize),
353  -(radius + toothHeight) *
354  (float) Math.Sin(stepSize * i + toothAngleStepSize + toothTipStepSize)));
355  }
356 
357  vertices.Add(new Vector2((radius + toothHeight) *
358  (float) Math.Cos(stepSize * i + toothAngleStepSize),
359  -(radius + toothHeight) *
360  (float) Math.Sin(stepSize * i + toothAngleStepSize)));
361 
362  vertices.Add(new Vector2(radius * (float) Math.Cos(stepSize * i),
363  -radius * (float) Math.Sin(stepSize * i)));
364  }
365 
366  return vertices;
367  }
