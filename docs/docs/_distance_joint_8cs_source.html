340  internal override void InitVelocityConstraints(ref SolverData data)
341  {
342  indexA = BodyA.IslandIndex;
343  indexB = BodyB.IslandIndex;
344  localCenterA = BodyA.Sweep.LocalCenter;
345  localCenterB = BodyB.Sweep.LocalCenter;
346  invMassA = BodyA.InvMass;
347  invMassB = BodyB.InvMass;
348  invIa = BodyA.InvI;
349  invIb = BodyB.InvI;
350 
351  Vector2 cA = data.Positions[indexA].C;
352  float aA = data.Positions[indexA].A;
353  Vector2 vA = data.Velocities[indexA].V;
354  float wA = data.Velocities[indexA].W;
355 
356  Vector2 cB = data.Positions[indexB].C;
357  float aB = data.Positions[indexB].A;
358  Vector2 vB = data.Velocities[indexB].V;
359  float wB = data.Velocities[indexB].W;
360 
361  Rot qA = new Rot(aA), qB = new Rot(aB);
362 
363  rA = MathUtils.Mul(ref qA, localAnchorA - localCenterA);
364  rB = MathUtils.Mul(ref qB, localAnchorB - localCenterB);
365  u = cB + rB - cA - rA;
366 
367  // Handle singularity.
368  currentLength = u.Length();
369  if (currentLength > Settings.LinearSlop)
370  {
371  u *= 1.0f / currentLength;
372  }
373  else
374  {
375  u = Vector2.Zero;
376  mass = 0.0f;
377  impulse = 0.0f;
378  lowerImpulse = 0.0f;
379  upperImpulse = 0.0f;
380  }
381 
382  float crAu = MathUtils.Cross(rA, u);
383  float crBu = MathUtils.Cross(rB, u);
384  float invMass = invMassA + invIa * crAu * crAu + invMassB + invIb * crBu * crBu;
385  mass = invMass != 0.0f ? 1.0f / invMass : 0.0f;
386 
387  if (stiffness > 0.0f && minLength < maxLength)
388  {
389  // soft
390  float c = currentLength - length;
391 
392  float d = damping;
393  float k = stiffness;
394 
395  // magic formulas
396  float h = data.Step.DeltaTime;
397 
398  // gamma = 1 / (h * (d + h * k))
399  // the extra factor of h in the denominator is since the lambda is an impulse, not a force
400  gamma = h * (d + h * k);
401  gamma = gamma != 0.0f ? 1.0f / gamma : 0.0f;
402  bias = c * h * k * gamma;
403 
404  invMass += gamma;
405  softMass = invMass != 0.0f ? 1.0f / invMass : 0.0f;
406  }
407  else
408  {
409  // rigid
410  gamma = 0.0f;
411  bias = 0.0f;
412  softMass = mass;
413  }
414 
415  if (data.Step.WarmStarting)
416  {
417  // Scale the impulse to support a variable time step.
418  impulse *= data.Step.DeltaTimeRatio;
419  lowerImpulse *= data.Step.DeltaTimeRatio;
420  upperImpulse *= data.Step.DeltaTimeRatio;
421 
422  Vector2 p = (impulse + lowerImpulse - upperImpulse) * u;
423  vA -= invMassA * p;
424  wA -= invIa * MathUtils.Cross(ref rA, ref p);
425  vB += invMassB * p;
426  wB += invIb * MathUtils.Cross(ref rB, ref p);
427  }
428  else
429  {
430  impulse = 0.0f;
431  }
432 
433  data.Velocities[indexA].V = vA;
434  data.Velocities[indexA].W = wA;
435  data.Velocities[indexB].V = vB;
436  data.Velocities[indexB].W = wB;
437  }

443  internal override void SolveVelocityConstraints(ref SolverData data)
444  {
445  Vector2 vA = data.Velocities[indexA].V;
446  float wA = data.Velocities[indexA].W;
447  Vector2 vB = data.Velocities[indexB].V;
448  float wB = data.Velocities[indexB].W;
449 
450  if (minLength < maxLength)
451  {
452  if (stiffness > 0.0f)
453  {
454  // Cdot = dot(u, v + cross(w, r))
455  Vector2 vpA = vA + MathUtils.Cross(wA, rA);
456  Vector2 vpB = vB + MathUtils.Cross(wB, rB);
457  float cdot = MathUtils.Dot(u, vpB - vpA);
458 
459  float impulse = -softMass * (cdot + bias + gamma * this.impulse);
460  this.impulse += impulse;
461 
462  Vector2 p = impulse * u;
463  vA -= invMassA * p;
464  wA -= invIa * MathUtils.Cross(rA, p);
465  vB += invMassB * p;
466  wB += invIb * MathUtils.Cross(rB, p);
467  }
468 
469  // lower
470  {
471  float c = currentLength - minLength;
472  float bias = MathUtils.Max(0.0f, c) * data.Step.InvertedDeltaTime;
473 
474  Vector2 vpA = vA + MathUtils.Cross(wA, rA);
475  Vector2 vpB = vB + MathUtils.Cross(wB, rB);
476  float cdot = MathUtils.Dot(u, vpB - vpA);
477 
478  float impulse = -mass * (cdot + bias);
479  float oldImpulse = lowerImpulse;
480  lowerImpulse = MathUtils.Max(0.0f, lowerImpulse + impulse);
481  impulse = lowerImpulse - oldImpulse;
482  Vector2 p = impulse * u;
483 
484  vA -= invMassA * p;
485  wA -= invIa * MathUtils.Cross(rA, p);
486  vB += invMassB * p;
487  wB += invIb * MathUtils.Cross(rB, p);
488  }
489 
490  // upper
491  {
492  float c = maxLength - currentLength;
493  float bias = MathUtils.Max(0.0f, c) * data.Step.InvertedDeltaTime;
494 
495  Vector2 vpA = vA + MathUtils.Cross(wA, rA);
496  Vector2 vpB = vB + MathUtils.Cross(wB, rB);
497  float cdot = MathUtils.Dot(u, vpA - vpB);
498 
499  float impulse = -mass * (cdot + bias);
500  float oldImpulse = upperImpulse;
501  upperImpulse = MathUtils.Max(0.0f, upperImpulse + impulse);
502  impulse = upperImpulse - oldImpulse;
503  Vector2 p = -impulse * u;
504 
505  vA -= invMassA * p;
506  wA -= invIa * MathUtils.Cross(rA, p);
507  vB += invMassB * p;
508  wB += invIb * MathUtils.Cross(rB, p);
509  }
510  }
511  else
512  {
513  // Equal limits
514 
515  // Cdot = dot(u, v + cross(w, r))
516  Vector2 vpA = vA + MathUtils.Cross(wA, rA);
517  Vector2 vpB = vB + MathUtils.Cross(wB, rB);
518  float cdot = MathUtils.Dot(u, vpB - vpA);
519 
520  float impulse = -mass * cdot;
521  this.impulse += impulse;
522 
523  Vector2 p = impulse * u;
524  vA -= invMassA * p;
525  wA -= invIa * MathUtils.Cross(rA, p);
526  vB += invMassB * p;
527  wB += invIb * MathUtils.Cross(rB, p);
528  }
529 
530  data.Velocities[indexA].V = vA;
531  data.Velocities[indexA].W = wA;
532  data.Velocities[indexB].V = vB;
533  data.Velocities[indexB].W = wB;
534  }

541  internal override bool SolvePositionConstraints(ref SolverData data)
542  {
543  Vector2 cA = data.Positions[indexA].C;
544  float aA = data.Positions[indexA].A;
545  Vector2 cB = data.Positions[indexB].C;
546  float aB = data.Positions[indexB].A;
547 
548  Rot qA = new Rot(aA), qB = new Rot(aB);
549 
550  Vector2 rA = MathUtils.Mul(qA, localAnchorA - localCenterA);
551  Vector2 rB = MathUtils.Mul(qB, localAnchorB - localCenterB);
552  Vector2 u = cB + rB - cA - rA;
553 
554  float length = MathUtils.Normalize(ref u);
555  float c;
556  if (minLength == maxLength)
557  {
558  c = length - minLength;
559  }
560  else if (length < minLength)
561  {
562  c = length - minLength;
563  }
564  else if (maxLength < length)
565  {
566  c = length - maxLength;
567  }
568  else
569  {
570  return true;
571  }
572 
573  float impulse = -mass * c;
574  Vector2 p = impulse * u;
575 
576  cA -= invMassA * p;
577  aA -= invIa * MathUtils.Cross(rA, p);
578  cB += invMassB * p;
579  aB += invIb * MathUtils.Cross(rB, p);
580 
581  data.Positions[indexA].C = cA;
582  data.Positions[indexA].A = aA;
583  data.Positions[indexB].C = cB;
584  data.Positions[indexB].A = aB;
585 
586  return MathUtils.Abs(c) < Settings.LinearSlop;
587  }
588  }
