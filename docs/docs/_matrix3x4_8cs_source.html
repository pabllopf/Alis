318  public static void CreateFromAxisAngle(Vector3 axis, float angle, out Matrix3x4 result)
319  {
320  axis.Normalize();
321  float axisX = axis.X, axisY = axis.Y, axisZ = axis.Z;
322 
323  float cos = MathF.Cos(angle);
324  float sin = MathF.Sin(angle);
325  float t = 1.0f - cos;
326 
327  float tXX = t * axisX * axisX;
328  float tXY = t * axisX * axisY;
329  float tXZ = t * axisX * axisZ;
330  float tYY = t * axisY * axisY;
331  float tYZ = t * axisY * axisZ;
332  float tZZ = t * axisZ * axisZ;
333 
334  float sinX = sin * axisX;
335  float sinY = sin * axisY;
336  float sinZ = sin * axisZ;
337 
338  result.Row0.X = tXX + cos;
339  result.Row0.Y = tXY - sinZ;
340  result.Row0.Z = tXZ + sinY;
341  result.Row0.W = 0;
342  result.Row1.X = tXY + sinZ;
343  result.Row1.Y = tYY + cos;
344  result.Row1.Z = tYZ - sinX;
345  result.Row1.W = 0;
346  result.Row2.X = tXZ - sinY;
347  result.Row2.Y = tYZ + sinX;
348  result.Row2.Z = tZZ + cos;
349  result.Row2.W = 0;
350  }

370  public static void CreateFromQuaternion(in Quaternion q, out Matrix3x4 result)
371  {
372  float x = q.X;
373  float y = q.Y;
374  float z = q.Z;
375  float w = q.W;
376  float tx = 2 * x;
377  float ty = 2 * y;
378  float tz = 2 * z;
379  float txx = tx * x;
380  float tyy = ty * y;
381  float tzz = tz * z;
382  float txy = tx * y;
383  float txz = tx * z;
384  float tyz = ty * z;
385  float txw = tx * w;
386  float tyw = ty * w;
387  float tzw = tz * w;
388 
389  result.Row0.X = 1f - (tyy + tzz);
390  result.Row0.Y = txy + tzw;
391  result.Row0.Z = txz - tyw;
392  result.Row0.W = 0f;
393  result.Row1.X = txy - tzw;
394  result.Row1.Y = 1f - (txx + tzz);
395  result.Row1.Z = tyz + txw;
396  result.Row1.W = 0f;
397  result.Row2.X = txz + tyw;
398  result.Row2.Y = tyz - txw;
399  result.Row2.Z = 1f - (txx + tyy);
400  result.Row2.W = 0f;
401  }

654  public static void Mult(in Matrix3x4 left, in Matrix4x3 right, out Matrix3 result)
655  {
656  float leftM11 = left.Row0.X;
657  float leftM12 = left.Row0.Y;
658  float leftM13 = left.Row0.Z;
659  float leftM14 = left.Row0.W;
660  float leftM21 = left.Row1.X;
661  float leftM22 = left.Row1.Y;
662  float leftM23 = left.Row1.Z;
663  float leftM24 = left.Row1.W;
664  float leftM31 = left.Row2.X;
665  float leftM32 = left.Row2.Y;
666  float leftM33 = left.Row2.Z;
667  float leftM34 = left.Row2.W;
668  float rightM11 = right.Row0.X;
669  float rightM12 = right.Row0.Y;
670  float rightM13 = right.Row0.Z;
671  float rightM21 = right.Row1.X;
672  float rightM22 = right.Row1.Y;
673  float rightM23 = right.Row1.Z;
674  float rightM31 = right.Row2.X;
675  float rightM32 = right.Row2.Y;
676  float rightM33 = right.Row2.Z;
677  float rightM41 = right.Row3.X;
678  float rightM42 = right.Row3.Y;
679  float rightM43 = right.Row3.Z;
680 
681  result.Row0.X = leftM11 * rightM11 + leftM12 * rightM21 + leftM13 * rightM31 + leftM14 * rightM41;
682  result.Row0.Y = leftM11 * rightM12 + leftM12 * rightM22 + leftM13 * rightM32 + leftM14 * rightM42;
683  result.Row0.Z = leftM11 * rightM13 + leftM12 * rightM23 + leftM13 * rightM33 + leftM14 * rightM43;
684  result.Row1.X = leftM21 * rightM11 + leftM22 * rightM21 + leftM23 * rightM31 + leftM24 * rightM41;
685  result.Row1.Y = leftM21 * rightM12 + leftM22 * rightM22 + leftM23 * rightM32 + leftM24 * rightM42;
686  result.Row1.Z = leftM21 * rightM13 + leftM22 * rightM23 + leftM23 * rightM33 + leftM24 * rightM43;
687  result.Row2.X = leftM31 * rightM11 + leftM32 * rightM21 + leftM33 * rightM31 + leftM34 * rightM41;
688  result.Row2.Y = leftM31 * rightM12 + leftM32 * rightM22 + leftM33 * rightM32 + leftM34 * rightM42;
689  result.Row2.Z = leftM31 * rightM13 + leftM32 * rightM23 + leftM33 * rightM33 + leftM34 * rightM43;

849  public static void Invert(in Matrix3x4 mat, out Matrix3x4 result)
850  {
851  Matrix3 inverseRotation = new Matrix3(mat.Column0, mat.Column1, mat.Column2);
852  inverseRotation.Row0 /= inverseRotation.Row0.LengthSquared;
853  inverseRotation.Row1 /= inverseRotation.Row1.LengthSquared;
854  inverseRotation.Row2 /= inverseRotation.Row2.LengthSquared;
855 
856  Vector3 translation = new Vector3(mat.Row0.W, mat.Row1.W, mat.Row2.W);
857 
858  result.Row0 = new Vector4(inverseRotation.Row0, -Vector3.Dot(inverseRotation.Row0, translation));
859  result.Row1 = new Vector4(inverseRotation.Row1, -Vector3.Dot(inverseRotation.Row1, translation));
860  result.Row2 = new Vector4(inverseRotation.Row2, -Vector3.Dot(inverseRotation.Row2, translation));
861  }
