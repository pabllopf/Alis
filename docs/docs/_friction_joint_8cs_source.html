247  internal override void InitVelocityConstraints(ref SolverData data)
248  {
249  indexA = BodyA.IslandIndex;
250  indexB = BodyB.IslandIndex;
251  localCenterA = BodyA.Sweep.LocalCenter;
252  localCenterB = BodyB.Sweep.LocalCenter;
253  invMassA = BodyA.InvMass;
254  invMassB = BodyB.InvMass;
255  invIa = BodyA.InvI;
256  invIb = BodyB.InvI;
257 
258  float aA = data.Positions[indexA].A;
259  Vector2 vA = data.Velocities[indexA].V;
260  float wA = data.Velocities[indexA].W;
261 
262  float aB = data.Positions[indexB].A;
263  Vector2 vB = data.Velocities[indexB].V;
264  float wB = data.Velocities[indexB].W;
265 
266  Rot qA = new Rot(aA), qB = new Rot(aB);
267 
268  // Compute the effective mass matrix.
269  rA = MathUtils.Mul(qA, localAnchorA - localCenterA);
270  rB = MathUtils.Mul(qB, localAnchorB - localCenterB);
271 
272  // J = [-I -r1_skew I r2_skew]
273  // [ 0 -1 0 1]
274  // r_skew = [-ry; rx]
275 
276  // Matlab
277  // K = [ mA+r1y^2*iA+mB+r2y^2*iB, -r1y*iA*r1x-r2y*iB*r2x, -r1y*iA-r2y*iB]
278  // [ -r1y*iA*r1x-r2y*iB*r2x, mA+r1x^2*iA+mB+r2x^2*iB, r1x*iA+r2x*iB]
279  // [ -r1y*iA-r2y*iB, r1x*iA+r2x*iB, iA+iB]
280 
281  float mA = invMassA, mB = invMassB;
282  float iA = invIa, iB = invIb;
283 
284  Mat22 k = new Mat22();
285  k.Ex.X = mA + mB + iA * rA.Y * rA.Y + iB * rB.Y * rB.Y;
286  k.Ex.Y = -iA * rA.X * rA.Y - iB * rB.X * rB.Y;
287  k.Ey.X = k.Ex.Y;
288  k.Ey.Y = mA + mB + iA * rA.X * rA.X + iB * rB.X * rB.X;
289 
290  linearMass = k.Inverse;
291 
292  angularMass = iA + iB;
293  if (angularMass > 0.0f)
294  {
295  angularMass = 1.0f / angularMass;
296  }
297 
298  if (data.Step.WarmStarting)
299  {
300  // Scale impulses to support a variable time step.
301  linearImpulse *= data.Step.DeltaTimeRatio;
302  angularImpulse *= data.Step.DeltaTimeRatio;
303 
304  Vector2 p = new Vector2(linearImpulse.X, linearImpulse.Y);
305  vA -= mA * p;
306  wA -= iA * (MathUtils.Cross(rA, p) + angularImpulse);
307  vB += mB * p;
308  wB += iB * (MathUtils.Cross(rB, p) + angularImpulse);
309  }
310  else
311  {
312  linearImpulse = Vector2.Zero;
313  angularImpulse = 0.0f;
314  }
315 
316  data.Velocities[indexA].V = vA;
317  data.Velocities[indexA].W = wA;
318  data.Velocities[indexB].V = vB;
319  data.Velocities[indexB].W = wB;
320  }

326  internal override void SolveVelocityConstraints(ref SolverData data)
327  {
328  Vector2 vA = data.Velocities[indexA].V;
329  float wA = data.Velocities[indexA].W;
330  Vector2 vB = data.Velocities[indexB].V;
331  float wB = data.Velocities[indexB].W;
332 
333  float mA = invMassA, mB = invMassB;
334  float iA = invIa, iB = invIb;
335 
336  float h = data.Step.DeltaTime;
337 
338  // Solve angular friction
339  {
340  float cdot = wB - wA;
341  float impulse = -angularMass * cdot;
342 
343  float oldImpulse = angularImpulse;
344  float maxImpulse = h * MaxTorque;
345  angularImpulse = MathUtils.Clamp(angularImpulse + impulse, -maxImpulse, maxImpulse);
346  impulse = angularImpulse - oldImpulse;
347 
348  wA -= iA * impulse;
349  wB += iB * impulse;
350  }
351 
352  // Solve linear friction
353  {
354  Vector2 cdot = vB + MathUtils.Cross(wB, rB) - vA - MathUtils.Cross(wA, rA);
355 
356  Vector2 impulse = -MathUtils.Mul(ref linearMass, cdot);
357  Vector2 oldImpulse = linearImpulse;
358  linearImpulse += impulse;
359 
360  float maxImpulse = h * MaxForce;
361 
362  if (linearImpulse.LengthSquared() > maxImpulse * maxImpulse)
363  {
364  linearImpulse = Vector2.Normalize(linearImpulse);
365  linearImpulse *= maxImpulse;
366  }
367 
368  impulse = linearImpulse - oldImpulse;
369 
370  vA -= mA * impulse;
371  wA -= iA * MathUtils.Cross(rA, impulse);
372 
373  vB += mB * impulse;
374  wB += iB * MathUtils.Cross(rB, impulse);
375  }
376 
377  data.Velocities[indexA].V = vA;
378  data.Velocities[indexA].W = wA;
379  data.Velocities[indexB].V = vB;
380  data.Velocities[indexB].W = wB;
381  }
