41  // Gear Joint:
42  // C0 = (coordinate1 + ratio * coordinate2)_initial
43  // C = (coordinate1 + ratio * coordinate2) - C0 = 0
44  // J = [J1 ratio * J2]
45  // K = J * invM * JT
46  // = J1 * invM1 * J1T + ratio * ratio * J2 * invM2 * J2T
47  //
48  // Revolute:
49  // coordinate = rotation
50  // Cdot = angularVelocity
51  // J = [0 0 1]
52  // K = J * invM * JT = invI
53  //
54  // Prismatic:
55  // coordinate = dot(p - pg, ug)
56  // Cdot = dot(v + cross(w, r), ug)
57  // J = [ug cross(r, ug)]
58  // K = J * invM * JT = invMass + invI * cross(r, ug)^2

198  public GearJoint(GearJointDef def) : base(def)

207  Debug.Assert(typeB == JointType.Revolute || typeB == JointType.Prismatic);

214  BodyA = jointA.BodyB;
215 
216  // Body B on joint1 must be dynamic
217  Debug.Assert(BodyA.Type == BodyType.Dynamic);
218 
219  // Get geometry of joint1
220  Transform xfA = BodyA.Xf;
221  float aA = BodyA.Sweep.A;
222  Transform xfC = bodyC.Xf;
223  float aC = bodyC.Sweep.A;

449  internal override void InitVelocityConstraints(ref SolverData data)
450  {
451  indexA = BodyA.IslandIndex;
452  indexB = BodyB.IslandIndex;
453  indexC = bodyC.IslandIndex;
454  indexD = bodyD.IslandIndex;
455  lcA = BodyA.Sweep.LocalCenter;
456  lcB = BodyB.Sweep.LocalCenter;
457  lcC = bodyC.Sweep.LocalCenter;
458  lcD = bodyD.Sweep.LocalCenter;
459  mA = BodyA.InvMass;
460  mB = BodyB.InvMass;
461  mC = bodyC.InvMass;
462  mD = bodyD.InvMass;
463  iA = BodyA.InvI;
464  iB = BodyB.InvI;
465  iC = bodyC.InvI;
466  iD = bodyD.InvI;
467 
468  float aA = data.Positions[indexA].A;
469  Vector2 vA = data.Velocities[indexA].V;
470  float wA = data.Velocities[indexA].W;
471 
472  float aB = data.Positions[indexB].A;
473  Vector2 vB = data.Velocities[indexB].V;
474  float wB = data.Velocities[indexB].W;
475 
476  float aC = data.Positions[indexC].A;
477  Vector2 vC = data.Velocities[indexC].V;
478  float wC = data.Velocities[indexC].W;
479 
480  float aD = data.Positions[indexD].A;
481  Vector2 vD = data.Velocities[indexD].V;
482  float wD = data.Velocities[indexD].W;
483 
484  Rot qA = new Rot(aA), qB = new Rot(aB), qC = new Rot(aC), qD = new Rot(aD);
485 
486  mass = 0.0f;
487 
488  if (typeA == JointType.Revolute)
489  {
490  jvAc = Vector2.Zero;
491  jwA = 1.0f;
492  jwC = 1.0f;
493  mass += iA + iC;
494  }
495  else
496  {
497  Vector2 u = MathUtils.Mul(qC, localAxisC);
498  Vector2 rC = MathUtils.Mul(qC, localAnchorC - lcC);
499  Vector2 rA = MathUtils.Mul(qA, localAnchorA - lcA);
500  jvAc = u;
501  jwC = MathUtils.Cross(rC, u);
502  jwA = MathUtils.Cross(rA, u);
503  mass += mC + mA + iC * jwC * jwC + iA * jwA * jwA;
504  }
505 
506  if (typeB == JointType.Revolute)
507  {
508  jvBd = Vector2.Zero;
509  jwB = ratio;
510  jwD = ratio;
511  mass += ratio * ratio * (iB + iD);
512  }
513  else
514  {
515  Vector2 u = MathUtils.Mul(qD, localAxisD);
516  Vector2 rD = MathUtils.Mul(qD, localAnchorD - lcD);
517  Vector2 rB = MathUtils.Mul(qB, localAnchorB - lcB);
518  jvBd = ratio * u;
519  jwD = ratio * MathUtils.Cross(rD, u);
520  jwB = ratio * MathUtils.Cross(rB, u);
521  mass += ratio * ratio * (mD + mB) + iD * jwD * jwD + iB * jwB * jwB;
522  }
523 
524  // Compute effective mass.
525  mass = mass > 0.0f ? 1.0f / mass : 0.0f;
526 
527  if (data.Step.WarmStarting)
528  {
529  vA += mA * impulse * jvAc;
530  wA += iA * impulse * jwA;
531  vB += mB * impulse * jvBd;
532  wB += iB * impulse * jwB;
533  vC -= mC * impulse * jvAc;
534  wC -= iC * impulse * jwC;
535  vD -= mD * impulse * jvBd;
536  wD -= iD * impulse * jwD;
537  }
538  else
539  {
540  impulse = 0.0f;
541  }
542 
543  data.Velocities[indexA].V = vA;
544  data.Velocities[indexA].W = wA;
545  data.Velocities[indexB].V = vB;
546  data.Velocities[indexB].W = wB;
547  data.Velocities[indexC].V = vC;
548  data.Velocities[indexC].W = wC;
549  data.Velocities[indexD].V = vD;
550  data.Velocities[indexD].W = wD;
551  }

557  internal override void SolveVelocityConstraints(ref SolverData data)
558  {
559  Vector2 vA = data.Velocities[indexA].V;
560  float wA = data.Velocities[indexA].W;
561  Vector2 vB = data.Velocities[indexB].V;
562  float wB = data.Velocities[indexB].W;
563  Vector2 vC = data.Velocities[indexC].V;
564  float wC = data.Velocities[indexC].W;
565  Vector2 vD = data.Velocities[indexD].V;
566  float wD = data.Velocities[indexD].W;
567 
568  float cdot = Vector2.Dot(jvAc, vA - vC) + Vector2.Dot(jvBd, vB - vD);
569  cdot += jwA * wA - jwC * wC + (jwB * wB - jwD * wD);
570 
571  float impulse = -mass * cdot;
572  this.impulse += impulse;
573 
574  vA += mA * impulse * jvAc;
575  wA += iA * impulse * jwA;
576  vB += mB * impulse * jvBd;
577  wB += iB * impulse * jwB;
578  vC -= mC * impulse * jvAc;
579  wC -= iC * impulse * jwC;
580  vD -= mD * impulse * jvBd;
581  wD -= iD * impulse * jwD;
582 
583  data.Velocities[indexA].V = vA;
584  data.Velocities[indexA].W = wA;
585  data.Velocities[indexB].V = vB;
586  data.Velocities[indexB].W = wB;
587  data.Velocities[indexC].V = vC;
588  data.Velocities[indexC].W = wC;
589  data.Velocities[indexD].V = vD;
590  data.Velocities[indexD].W = wD;
591  }

598  internal override bool SolvePositionConstraints(ref SolverData data)
599  {
600  Vector2 cA = data.Positions[indexA].C;
601  float aA = data.Positions[indexA].A;
602  Vector2 cB = data.Positions[indexB].C;
603  float aB = data.Positions[indexB].A;
604  Vector2 cC = data.Positions[indexC].C;
605  float aC = data.Positions[indexC].A;
606  Vector2 cD = data.Positions[indexD].C;
607  float aD = data.Positions[indexD].A;
608 
609  Rot qA = new Rot(aA), qB = new Rot(aB), qC = new Rot(aC), qD = new Rot(aD);
610 
611  const float linearError = 0.0f;
612 
613  float coordinateA, coordinateB;
614 
615  Vector2 jvAc, jvBd;
616  float jwA, jwB, jwC, jwD;
617  float mass = 0.0f;
618 
619  if (typeA == JointType.Revolute)
620  {
621  jvAc = Vector2.Zero;
622  jwA = 1.0f;
623  jwC = 1.0f;
624  mass += iA + iC;
625 
626  coordinateA = aA - aC - referenceAngleA;
627  }
628  else
629  {
630  Vector2 u = MathUtils.Mul(qC, localAxisC);
631  Vector2 rC = MathUtils.Mul(qC, localAnchorC - lcC);
632  Vector2 rA = MathUtils.Mul(qA, localAnchorA - lcA);
633  jvAc = u;
634  jwC = MathUtils.Cross(rC, u);
635  jwA = MathUtils.Cross(rA, u);
636  mass += mC + mA + iC * jwC * jwC + iA * jwA * jwA;
637 
638  Vector2 pC = localAnchorC - lcC;
639  Vector2 pA = MathUtils.MulT(qC, rA + (cA - cC));
640  coordinateA = Vector2.Dot(pA - pC, localAxisC);
641  }
642 
643  if (typeB == JointType.Revolute)
644  {
645  jvBd = Vector2.Zero;
646  jwB = ratio;
647  jwD = ratio;
648  mass += ratio * ratio * (iB + iD);
649 
650  coordinateB = aB - aD - referenceAngleB;
651  }
652  else
653  {
654  Vector2 u = MathUtils.Mul(qD, localAxisD);
655  Vector2 rD = MathUtils.Mul(qD, localAnchorD - lcD);
656  Vector2 rB = MathUtils.Mul(qB, localAnchorB - lcB);
657  jvBd = ratio * u;
658  jwD = ratio * MathUtils.Cross(rD, u);
659  jwB = ratio * MathUtils.Cross(rB, u);
660  mass += ratio * ratio * (mD + mB) + iD * jwD * jwD + iB * jwB * jwB;
661 
662  Vector2 pD = localAnchorD - lcD;
663  Vector2 pB = MathUtils.MulT(qD, rB + (cB - cD));
664  coordinateB = Vector2.Dot(pB - pD, localAxisD);
665  }
666 
667  float c = coordinateA + ratio * coordinateB - constant;
668 
669  float impulse = 0.0f;
670  if (mass > 0.0f)
671  {
672  impulse = -c / mass;
673  }
674 
675  cA += mA * impulse * jvAc;
676  aA += iA * impulse * jwA;
677  cB += mB * impulse * jvBd;
678  aB += iB * impulse * jwB;
679  cC -= mC * impulse * jvAc;
680  aC -= iC * impulse * jwC;
681  cD -= mD * impulse * jvBd;
682  aD -= iD * impulse * jwD;
683 
684  data.Positions[indexA].C = cA;
685  data.Positions[indexA].A = aA;
686  data.Positions[indexB].C = cB;
687  data.Positions[indexB].A = aB;
688  data.Positions[indexC].C = cC;
689  data.Positions[indexC].A = aC;
690  data.Positions[indexD].C = cD;
691  data.Positions[indexD].A = aD;
692 
693  // TODO_ERIN not implemented
694  return linearError < Settings.LinearSlop;
695  }
696  }
