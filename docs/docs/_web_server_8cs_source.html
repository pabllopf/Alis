139  private string GetSubProtocol(IList<string> requestedSubProtocols)
140  {
141  foreach (string subProtocol in requestedSubProtocols)
142  {
143  // match the first sub protocol that we support (the client should pass the most preferable sub protocols first)
144  if (_supportedSubProtocols.Contains(subProtocol))
145  {
146  _logger.LogInformation($"Http header has requested sub protocol {subProtocol} which is supported");
147 
148  return subProtocol;
149  }
150  }
151 
152  if (requestedSubProtocols.Count > 0)
153  {
154  _logger.LogWarning(
155  $"Http header has requested the following sub protocols: {string.Join(", ", requestedSubProtocols)}. There are no supported protocols configured that match.");
156  }
157 
158  return null;

165  private async Task ProcessTcpClientAsync(TcpClient tcpClient)
166  {
167  CancellationTokenSource source = new CancellationTokenSource();
168 
169  try
170  {
171  if (_isDisposed)
172  {
173  return;
174  }
175 
176  // this worker thread stays alive until either of the following happens:
177  // Client sends a close conection request OR
178  // An unhandled exception is thrown OR
179  // The server is disposed
180  _logger.LogInformation("Server: Connection opened. Reading Http header from stream");
181 
182  // get a secure or insecure stream
183  Stream stream = tcpClient.GetStream();
184  WebSocketHttpContext context = await _webSocketServerFactory.ReadHttpHeaderFromStreamAsync(stream);
185  if (context.IsWebSocketRequest)
186  {
187  string subProtocol = GetSubProtocol(context.WebSocketRequestedProtocols);
188  WebSocketServerOptions options =
189  new WebSocketServerOptions(TimeSpan.FromSeconds(30),
190  subProtocol);
191  _logger.LogInformation(
192  "Http header has requested an upgrade to Web Socket protocol. Negotiating Web Socket handshake");
193 
194  WebSocket webSocket = await _webSocketServerFactory.AcceptWebSocketAsync(context, options);
195 
196  _logger.LogInformation("Web Socket handshake response sent. Stream ready.");
197  await RespondToWebSocketRequestAsync(webSocket, source.Token);
198  }
199  else
200  {
201  _logger.LogInformation("Http header contains no web socket upgrade request. Ignoring");
202  }
203 
204  _logger.LogInformation("Server: Connection closed");
205  }
206  catch (ObjectDisposedException)
207  {
208  // do nothing. This will be thrown if the Listener has been stopped
209  }
210  catch (Exception ex)
211  {
212  _logger.LogError(ex.ToString());
213  }
214  finally
215  {
216  try
217  {
218  tcpClient.Client.Close();
219  tcpClient.Close();
220  source.Cancel();
221  }
222  catch (Exception ex)
223  {
224  _logger.LogError($"Failed to close TCP connection: {ex}");
225  }
226  }
227  }

234  public async Task RespondToWebSocketRequestAsync(WebSocket webSocket, CancellationToken token)
235  {
236  ArraySegment<byte> buffer = new ArraySegment<byte>(new byte[BUFFER_SIZE]);
237 
238  while (true)
239  {
240  WebSocketReceiveResult result = await webSocket.ReceiveAsync(buffer, token);
241  if (result.MessageType == WebSocketMessageType.Close)
242  {
243  _logger.LogInformation(
244  $"Client initiated close. Status: {result.CloseStatus} Description: {result.CloseStatusDescription}");
245  break;
246  }
247 
248  if (result.Count > BUFFER_SIZE)
249  {
250  await webSocket.CloseAsync(WebSocketCloseStatus.MessageTooBig,
251  $"Web socket frame cannot exceed buffer size of {BUFFER_SIZE:#,##0} bytes. Send multiple frames instead.",
252  token);
253  break;
254  }
255 
256  // just echo the message back to the client
257  ArraySegment<byte> toSend = new ArraySegment<byte>(buffer.Array, buffer.Offset, result.Count);
258  await webSocket.SendAsync(toSend, WebSocketMessageType.Binary, true, token);
259  }
260  }
