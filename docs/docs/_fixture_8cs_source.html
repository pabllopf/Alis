330  private void Refilter()
331  {
332  // Flag associated contacts for filtering.
333  ContactEdge edge = Bodyprivate.ContactList;
334  while (edge != null)
335  {
336  Contact contact = edge.Contact;
337  Fixture fixtureA = contact.FixtureA;
338  Fixture fixtureB = contact.FixtureB;
339  if (fixtureA == this || fixtureB == this)
340  {
341  contact.Flags |= ContactFlags.FilterFlag;
342  }
343 
344  edge = edge.Next;
345  }
346 
347  World world = Bodyprivate.World;
348 
349  if (world == null)
350  {
351  return;
352  }
353 
354  // Touch each proxy so that new pairs may be created
355  IBroadPhase broadPhase = world.ContactManager.BroadPhase;
356  for (int i = 0; i < proxyCount; ++i)
357  {
358  broadPhase.TouchProxy(proxies[i].ProxyId);
359  }
360  }

406  // These support body activation/deactivation.
412  internal void CreateProxies(IBroadPhase broadPhase, ref Transform xf)
413  {
414  Debug.Assert(proxyCount == 0);
415 
416  // Create proxies in the broad-phase.
417  proxyCount = ShapePrivate.ChildCount;
418 
419  for (int i = 0; i < proxyCount; ++i)
420  {
421  FixtureProxy proxy = new FixtureProxy();
422  ShapePrivate.ComputeAabb(ref xf, i, out proxy.Aabb);
423  proxy.Fixture = this;
424  proxy.ChildIndex = i;
425 
426  //Velcro note: This line needs to be after the previous two because FixtureProxy is a struct
427  proxy.ProxyId = broadPhase.AddProxy(ref proxy);
428 
429  proxies[i] = proxy;
430  }
431  }
432 
437  internal void DestroyProxies(IBroadPhase broadPhase)
438  {
439  // Destroy proxies in the broad-phase.
440  for (int i = 0; i < proxyCount; ++i)
441  {
442  FixtureProxy proxy = proxies[i];
443  broadPhase.RemoveProxy(proxy.ProxyId);
444  proxy.ProxyId = DynamicTreeBroadPhase.NullProxy;
445  }
446 
447  proxyCount = 0;
448  }
449 
456  internal void Synchronize(IBroadPhase broadPhase, ref Transform transform1, ref Transform transform2)
457  {
458  if (proxyCount == 0)
459  {
460  return;
461  }
462 
463  for (int i = 0; i < proxyCount; ++i)
464  {
465  FixtureProxy proxy = proxies[i];
466 
467  // Compute an AABB that covers the swept Shape (may miss some rotation effect).
468  Shape.ComputeAabb(ref transform1, proxy.ChildIndex, out Aabb aabb1);
469  Shape.ComputeAabb(ref transform2, proxy.ChildIndex, out Aabb aabb2);
470 
471  proxy.Aabb.Combine(ref aabb1, ref aabb2);
472 
473  Vector2 displacement = aabb2.Center - aabb1.Center;
474 
475  broadPhase.MoveProxy(proxy.ProxyId, ref proxy.Aabb, displacement);
476  }
477  }
478  }
