654  internal static unsafe List<string> ALStringListToList(byte* alList)
655  {
656  if (alList == (byte*) 0)
657  {
658  return new List<string>();
659  }
660 
661  List<string> strings = new List<string>();
662 
663  byte* currentPos = alList;
664  while (true)
665  {
666  string? currentString = Marshal.PtrToStringAnsi(new IntPtr(currentPos));
667  if (string.IsNullOrEmpty(currentString))
668  {
669  break;
670  }
671 
672  strings.Add(currentString);
673  currentPos += currentString.Length + 1;
674  }
675 
676  return strings;
677  }
678  }
