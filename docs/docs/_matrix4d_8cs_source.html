567  [Pure]
568  public Quaterniond ExtractRotation(bool rowNormalize = true)
569  {
570  Vector3d row0 = Row0.Xyz;
571  Vector3d row1 = Row1.Xyz;
572  Vector3d row2 = Row2.Xyz;
573 
574  if (rowNormalize)
575  {
576  row0 = row0.Normalized();
577  row1 = row1.Normalized();
578  row2 = row2.Normalized();
579  }
580 
581  // code below adapted from Blender
582  Quaterniond q = default(Quaterniond);
583  double trace = 0.25 * (row0[0] + row1[1] + row2[2] + 1.0);
584 
585  if (trace > 0)
586  {
587  double sq = Math.Sqrt(trace);
588 
589  q.W = sq;
590  sq = 1.0 / (4.0 * sq);
591  q.X = (row1[2] - row2[1]) * sq;
592  q.Y = (row2[0] - row0[2]) * sq;
593  q.Z = (row0[1] - row1[0]) * sq;
594  }
595  else if (row0[0] > row1[1] && row0[0] > row2[2])
596  {
597  double sq = 2.0 * Math.Sqrt(1.0 + row0[0] - row1[1] - row2[2]);
598 
599  q.X = 0.25 * sq;
600  sq = 1.0 / sq;
601  q.W = (row2[1] - row1[2]) * sq;
602  q.Y = (row1[0] + row0[1]) * sq;
603  q.Z = (row2[0] + row0[2]) * sq;
604  }
605  else if (row1[1] > row2[2])
606  {
607  double sq = 2.0 * Math.Sqrt(1.0 + row1[1] - row0[0] - row2[2]);
608 
609  q.Y = 0.25 * sq;
610  sq = 1.0 / sq;
611  q.W = (row2[0] - row0[2]) * sq;
612  q.X = (row1[0] + row0[1]) * sq;
613  q.Z = (row2[1] + row1[2]) * sq;
614  }
615  else
616  {
617  double sq = 2.0 * Math.Sqrt(1.0 + row2[2] - row0[0] - row1[1]);
618 
619  q.Z = 0.25 * sq;
620  sq = 1.0 / sq;
621  q.W = (row1[0] - row0[1]) * sq;
622  q.X = (row2[0] + row0[2]) * sq;
623  q.Y = (row2[1] + row1[2]) * sq;
624  }
625 
626  q.Normalize();
627  return q;
628  }

642  public static void CreateFromAxisAngle(Vector3d axis, double angle, out Matrix4d result)
643  {
644  // normalize and create a local copy of the vector.
645  axis.Normalize();
646  double axisX = axis.X, axisY = axis.Y, axisZ = axis.Z;
647 
648  // calculate angles
649  double cos = Math.Cos(-angle);
650  double sin = Math.Sin(-angle);
651  double t = 1.0f - cos;
652 
653  // do the conversion math once
654  double tXX = t * axisX * axisX;
655  double tXY = t * axisX * axisY;
656  double tXZ = t * axisX * axisZ;
657  double tYY = t * axisY * axisY;
658  double tYZ = t * axisY * axisZ;
659  double tZZ = t * axisZ * axisZ;
660 
661  double sinX = sin * axisX;
662  double sinY = sin * axisY;
663  double sinZ = sin * axisZ;
664 
665  result.Row0.X = tXX + cos;
666  result.Row0.Y = tXY - sinZ;
667  result.Row0.Z = tXZ + sinY;
668  result.Row0.W = 0;
669  result.Row1.X = tXY + sinZ;
670  result.Row1.Y = tYY + cos;
671  result.Row1.Z = tYZ - sinX;
672  result.Row1.W = 0;
673  result.Row2.X = tXZ - sinY;
674  result.Row2.Y = tYZ + sinX;
675  result.Row2.Z = tZZ + cos;
676  result.Row2.W = 0;
677  result.Row3 = Vector4d.UnitW;
678  }

1026  public static void CreatePerspectiveOffCenter
1027  (
1028  double left,
1029  double right,
1030  double bottom,
1031  double top,
1032  double depthNear,
1033  double depthFar,
1034  out Matrix4d result
1035  )
1036  {
1037  if (depthNear <= 0)
1038  {
1039  throw new ArgumentOutOfRangeException(nameof(depthNear));
1040  }
1041 
1042  if (depthFar <= 0)
1043  {
1044  throw new ArgumentOutOfRangeException(nameof(depthFar));
1045  }
1046 
1047  if (depthNear >= depthFar)
1048  {
1049  throw new ArgumentOutOfRangeException(nameof(depthNear));
1050  }
1051 
1052  double x = 2.0 * depthNear / (right - left);
1053  double y = 2.0 * depthNear / (top - bottom);
1054  double a = (right + left) / (right - left);
1055  double b = (top + bottom) / (top - bottom);
1056  double c = -(depthFar + depthNear) / (depthFar - depthNear);
1057  double d = -(2.0 * depthFar * depthNear) / (depthFar - depthNear);
1058 
1059 #pragma warning disable SA1117 // Parameters should be on same line or separate lines
1060  result = new Matrix4d
1061  (
1062  x, 0, 0, 0,
1063  0, y, 0, 0,
1064  a, b, c, -1,
