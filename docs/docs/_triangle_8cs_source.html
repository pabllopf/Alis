76  Vector2 a = this[0];
77  Vector2 b = this[1];
78  Vector2 c = this[2];
79 
80  if (x < a.X && x < b.X && x < c.X)

87  return false;

91  {
92  return false;
93  }
94 
95  if (y > a.Y && y > b.Y && y > c.Y)
96  {
97  return false;
98  }
99 
100  float vx2 = x - a.X;
101  float vy2 = y - a.Y;
102  float vx1 = b.X - a.X;
103  float vy1 = b.Y - a.Y;
104  float vx0 = c.X - a.X;
105  float vy0 = c.Y - a.Y;
106 
107  float dot00 = vx0 * vx0 + vy0 * vy0;
108  float dot01 = vx0 * vx1 + vy0 * vy1;
109  float dot02 = vx0 * vx2 + vy0 * vy2;
110  float dot11 = vx1 * vx1 + vy1 * vy1;
111  float dot12 = vx1 * vx2 + vy1 * vy2;
112  float invDenom = 1.0f / (dot00 * dot11 - dot01 * dot01);
113  float u = (dot11 * dot02 - dot01 * dot12) * invDenom;
114  float v = (dot00 * dot12 - dot01 * dot02) * invDenom;
115 
116  return u > 0 && v > 0 && u + v < 1;
117  }
