93  // Almost Pi!
97  private const float PiSlop = 3.1f;

99  // Append a point to the list
104  public void Add(Point point)
105  {
106  if (size == 0)
107  {
108  head = point;
109  size = 1;
110  }
111  else if (size == 1)
112  {
113  // Keep repeat points out of the list
114  tail = point;
115  tail.Prev = head;
116  head.Next = tail;
117  size = 2;
118  }
119  else
120  {
121  // Keep repeat points out of the list
122  tail.Next = point;
123  point.Prev = tail;
124  tail = point;
125  size += 1;
126  }
127  }
128 
129  // Remove a point from the list
134  public void Remove(Point point)
135  {
136  Point next = point.Next;
137  Point prev = point.Prev;
138  point.Prev.Next = next;
139  point.Next.Prev = prev;
140  size -= 1;
141  }
142 
143  // Partition a x-monotone mountain into triangles O(n)
144  // See "Computational Geometry in C", 2nd edition, by Joseph O'Rourke, page 52
148  public void Process()
149  {
150  // Establish the proper sign
151  positive = AngleSign();
152 
153  // create monotone polygon - for dubug purposes
154  GenMonoPoly();
155 
156  // Initialize internal angles at each nonbase vertex
157  // Link strictly convex vertices into a list, ignore reflex vertices
158  Point p = head.Next;
159  while (p.Neq(tail))
160  {
161  float a = Angle(p);
162 
163  // If the point is almost colinear with it's neighbor, remove it!
164  if (a >= PiSlop || a <= -PiSlop || a == 0.0f)
165  {
166  Remove(p);
167  }
168  else if (IsConvex(p))
169  {
170  convexPoints.Add(p);
171  }
172 
173  p = p.Next;
174  }
175 
176  Triangulate();
177  }
178 
182  private void Triangulate()
183  {
184  while (convexPoints.Count != 0)
185  {
186  using (IEnumerator<Point> e = convexPoints.GetEnumerator())
187  {
188  e.MoveNext();
189  Point ear = e.Current;
190 
191  convexPoints.Remove(ear);
192  Point a = ear.Prev;
193  Point b = ear;
194  Point c = ear.Next;
195  List<Point> triangle = new List<Point>(3)
196  {
197  a,
198  b,
199  c
200  };
201 
202  Triangles.Add(triangle);
203 
204  // Remove ear, update angles and convex list
205  Remove(ear);
206  if (Valid(a))
207  {
208  convexPoints.Add(a);
209  }
210 
211  if (Valid(c))
212  {
213  convexPoints.Add(c);
214  }
215  }
216  }
217 
218  Debug.Assert(size <= 3, "Triangulation bug, please report");
219  }
220 
226  private bool Valid(Point p) => p.Neq(head) && p.Neq(tail) && IsConvex(p);
227 
228  // Create the monotone polygon
232  private void GenMonoPoly()
233  {
234  Point p = head;
235  while (p != null)
236  {
237  monoPoly.Add(p);
238  p = p.Next;
239  }
240  }
241 
247  private float Angle(Point p)
248  {
249  Point a = p.Next - p;
250  Point b = p.Prev - p;
251  return (float) Math.Atan2(a.Cross(b), a.Dot(b));
252  }
253 
258  private bool AngleSign()
259  {
260  Point a = head.Next - head;
261  Point b = tail - head;
262  return Math.Atan2(a.Cross(b), a.Dot(b)) >= 0;
263  }
264 
265  // Determines if the inslide angle is convex or reflex
271  private bool IsConvex(Point p)
272  {
273  if (positive != Angle(p) >= 0)
274  {
275  return false;
276  }
277 
278  return true;
279  }
280  }
