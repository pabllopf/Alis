53  public virtual bool IsActiveOn(Body body)
54  {
55  if (body == null || !body.Enabled || body.IsStatic)
56  {
57  return false;
58  }
59 
60  if (body.FixtureList == null)
61  {
62  return false;
63  }
64 
65  foreach (Fixture fixture in body.FixtureList)
66  {
67  //Disable
68  if (fixture.CollisionGroup == DisabledOnGroup && fixture.CollisionGroup != 0 && DisabledOnGroup != 0)
69  {
70  return false;
71  }
72 
73  if ((fixture.CollisionCategories & DisabledOnCategories) != Category.None)
74  {
75  return false;
76  }
77 
78  if (EnabledOnGroup != 0 || EnabledOnCategories != Category.All)
79  {
80  //Enable
81  if (fixture.CollisionGroup == EnabledOnGroup && fixture.CollisionGroup != 0 && EnabledOnGroup != 0)
82  {
83  return true;
84  }
85 
86  if ((fixture.CollisionCategories & EnabledOnCategories) != Category.None &&
87  EnabledOnCategories != Category.All)
88  {
89  return true;
90  }
91  }
92  else
93  {
94  return true;
95  }
96  }
97 
98  return false;
99  }
