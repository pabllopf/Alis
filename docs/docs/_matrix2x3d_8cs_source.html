390  public static void Mult(in Matrix2x3d left, in Matrix3x2 right, out Matrix2d result)
391  {
392  double leftM11 = left.Row0.X;
393  double leftM12 = left.Row0.Y;
394  double leftM13 = left.Row0.Z;
395  double leftM21 = left.Row1.X;
396  double leftM22 = left.Row1.Y;
397  double leftM23 = left.Row1.Z;
398  double rightM11 = right.Row0.X;
399  double rightM12 = right.Row0.Y;
400  double rightM21 = right.Row1.X;
401  double rightM22 = right.Row1.Y;
402  double rightM31 = right.Row2.X;
403  double rightM32 = right.Row2.Y;
404 
405  result.Row0.X = leftM11 * rightM11 + leftM12 * rightM21 + leftM13 * rightM31;
406  result.Row0.Y = leftM11 * rightM12 + leftM12 * rightM22 + leftM13 * rightM32;
407  result.Row1.X = leftM21 * rightM11 + leftM22 * rightM21 + leftM23 * rightM31;
408  result.Row1.Y = leftM21 * rightM12 + leftM22 * rightM22 + leftM23 * rightM32;
