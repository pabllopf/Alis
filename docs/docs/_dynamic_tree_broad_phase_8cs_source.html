192  public void UpdatePairs(BroadphaseHandler callback)
193  {
194  // Reset pair buffer
195  pairCount = 0;
196 
197  // Perform tree queries for all moving proxies.
198  for (int i = 0; i < moveCount; ++i)
199  {
200  queryProxyId = moveBuffer[i];
201  if (queryProxyId == NullProxy)
202  {
203  continue;
204  }
205 
206  // We have to query the tree with the fat AABB so that
207  // we don't fail to create a pair that may touch later.
208  tree.GetFatAabb(queryProxyId, out Aabb fatAabb);
209 
210  // Query tree, create pairs and add them pair buffer.
211  tree.Query(queryCallback, ref fatAabb);
212  }
213 
214  for (int i = 0; i < pairCount; ++i)
215  {
216  Pair primaryPair = pairBuffer[i];
217  FixtureProxy userDataA = tree.GetUserData(primaryPair.ProxyIdA);
218  FixtureProxy userDataB = tree.GetUserData(primaryPair.ProxyIdB);
219 
220  callback(ref userDataA, ref userDataB);
221  }
222 
223  // Clear move flags
224  for (int i = 0; i < moveCount; ++i)
225  {
226  int proxyId = moveBuffer[i];
227  if (proxyId == NullProxy)
228  {
229  continue;
230  }
231 
232  tree.ClearMoved(proxyId);
233  }
234 
235  // Reset move buffer
236  moveCount = 0;
237  }

273  private void BufferMove(int proxyId)
274  {
275  if (moveCount == moveCapacity)
276  {
277  int[] oldBuffer = moveBuffer;
278  moveCapacity *= 2;
279  moveBuffer = new int[moveCapacity];
280  Array.Copy(oldBuffer, moveBuffer, moveCount);
281  }
282 
283  moveBuffer[moveCount] = proxyId;
284  ++moveCount;
285  }
286 
291  private void UnBufferMove(int proxyId)
292  {
293  for (int i = 0; i < moveCount; ++i)
294  {
295  if (moveBuffer[i] == proxyId)
296  {
297  moveBuffer[i] = NullProxy;
298  }
299  }
300  }
301 
303  private bool QueryCallback(int proxyId)
304  {
305  // A proxy cannot form a pair with itself.
306  if (proxyId == queryProxyId)
307  {
308  return true;
309  }
310 
311  bool moved = tree.WasMoved(proxyId);
312  if (moved && proxyId > queryProxyId)
313  {
314  // Both proxies are moving. Avoid duplicate pairs.
315  return true;
316  }
317 
318  // Grow the pair buffer as needed.
319  if (pairCount == pairCapacity)
320  {
321  Pair[] oldBuffer = pairBuffer;
322  pairCapacity += pairCapacity >> 1;
323  pairBuffer = new Pair[pairCapacity];
324  Array.Copy(oldBuffer, pairBuffer, pairCount);
325  }
326 
327  pairBuffer[pairCount].ProxyIdA = Math.Min(proxyId, queryProxyId);
328  pairBuffer[pairCount].ProxyIdB = Math.Max(proxyId, queryProxyId);
329  ++pairCount;
330 
331  return true;
332  }
333  }
