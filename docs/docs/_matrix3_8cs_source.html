400  public Quaternion ExtractRotation(bool rowNormalize = true)
401  {
402  Vector3 row0 = Row0;
403  Vector3 row1 = Row1;
404  Vector3 row2 = Row2;
405 
406  if (rowNormalize)
407  {
408  row0 = row0.Normalized();
409  row1 = row1.Normalized();
410  row2 = row2.Normalized();
411  }
412 
413  // code below adapted from Blender
414  Quaternion q = default(Quaternion);
415  double trace = 0.25 * (row0[0] + row1[1] + row2[2] + 1.0);
416 
417  if (trace > 0)
418  {
419  double sq = Math.Sqrt(trace);
420 
421  q.W = (float) sq;
422  sq = 1.0 / (4.0 * sq);
423  q.X = (float) ((row1[2] - row2[1]) * sq);
424  q.Y = (float) ((row2[0] - row0[2]) * sq);
425  q.Z = (float) ((row0[1] - row1[0]) * sq);
426  }
427  else if (row0[0] > row1[1] && row0[0] > row2[2])
428  {
429  double sq = 2.0 * Math.Sqrt(1.0 + row0[0] - row1[1] - row2[2]);
430 
431  q.X = (float) (0.25 * sq);
432  sq = 1.0 / sq;
433  q.W = (float) ((row2[1] - row1[2]) * sq);
434  q.Y = (float) ((row1[0] + row0[1]) * sq);
435  q.Z = (float) ((row2[0] + row0[2]) * sq);
436  }
437  else if (row1[1] > row2[2])
438  {
439  double sq = 2.0 * Math.Sqrt(1.0 + row1[1] - row0[0] - row2[2]);
440 
441  q.Y = (float) (0.25 * sq);
442  sq = 1.0 / sq;
443  q.W = (float) ((row2[0] - row0[2]) * sq);
444  q.X = (float) ((row1[0] + row0[1]) * sq);
445  q.Z = (float) ((row2[1] + row1[2]) * sq);
446  }
447  else
448  {
449  double sq = 2.0 * Math.Sqrt(1.0 + row2[2] - row0[0] - row1[1]);
450 
451  q.Z = (float) (0.25 * sq);
452  sq = 1.0 / sq;
453  q.W = (float) ((row1[0] - row0[1]) * sq);
454  q.X = (float) ((row2[0] + row0[2]) * sq);
455  q.Y = (float) ((row2[1] + row1[2]) * sq);
456  }
457 
458  q.Normalize();
459  return q;
460  }

468  public static void CreateFromAxisAngle(Vector3 axis, float angle, out Matrix3 result)
469  {
470  // normalize and create a local copy of the vector.
471  axis.Normalize();
472  float axisX = axis.X, axisY = axis.Y, axisZ = axis.Z;
473 
474  // calculate angles
475  float cos = MathF.Cos(-angle);
476  float sin = MathF.Sin(-angle);
477  float t = 1.0f - cos;
478 
479  // do the conversion math once
480  float tXX = t * axisX * axisX;
481  float tXY = t * axisX * axisY;
482  float tXZ = t * axisX * axisZ;
483  float tYY = t * axisY * axisY;
484  float tYZ = t * axisY * axisZ;
485  float tZZ = t * axisZ * axisZ;
486 
487  float sinX = sin * axisX;
488  float sinY = sin * axisY;
489  float sinZ = sin * axisZ;
490 
491  result.Row0.X = tXX + cos;
492  result.Row0.Y = tXY - sinZ;
493  result.Row0.Z = tXZ + sinY;
494  result.Row1.X = tXY + sinZ;
495  result.Row1.Y = tYY + cos;
496  result.Row1.Z = tYZ - sinX;
497  result.Row2.X = tXZ - sinY;
498  result.Row2.Y = tYZ + sinX;
499  result.Row2.Z = tZZ + cos;
500  }

520  public static void CreateFromQuaternion(in Quaternion q, out Matrix3 result)
521  {
522  // Adapted from https://en.wikipedia.org/wiki/Quaternions_and_spatial_rotation#Quaternion-derived_rotation_matrix
523  // with the caviat that opentk uses row-major matrices so the matrix we create is transposed
524  float sqx = q.X * q.X;
525  float sqy = q.Y * q.Y;
526  float sqz = q.Z * q.Z;
527  float sqw = q.W * q.W;
528 
529  float xy = q.X * q.Y;
530  float xz = q.X * q.Z;
531  float xw = q.X * q.W;
532 
533  float yz = q.Y * q.Z;
534  float yw = q.Y * q.W;
535 
536  float zw = q.Z * q.W;
537 
538  float s2 = 2f / (sqx + sqy + sqz + sqw);
539 
540  result.Row0.X = 1f - s2 * (sqy + sqz);
541  result.Row1.Y = 1f - s2 * (sqx + sqz);
542  result.Row2.Z = 1f - s2 * (sqx + sqy);
543 
544  result.Row0.Y = s2 * (xy + zw);
545  result.Row1.X = s2 * (xy - zw);
546 
547  result.Row2.X = s2 * (xz + yw);
548  result.Row0.Z = s2 * (xz - yw);
549 
550  result.Row2.Y = s2 * (yz - xw);
551  result.Row1.Z = s2 * (yz + xw);
552  }

777  public static void Mult(in Matrix3 left, in Matrix3 right, out Matrix3 result)
778  {
779  float leftM11 = left.Row0.X;
780  float leftM12 = left.Row0.Y;
781  float leftM13 = left.Row0.Z;
782  float leftM21 = left.Row1.X;
783  float leftM22 = left.Row1.Y;
784  float leftM23 = left.Row1.Z;
785  float leftM31 = left.Row2.X;
786  float leftM32 = left.Row2.Y;
787  float leftM33 = left.Row2.Z;
788  float rightM11 = right.Row0.X;
789  float rightM12 = right.Row0.Y;
790  float rightM13 = right.Row0.Z;
791  float rightM21 = right.Row1.X;
792  float rightM22 = right.Row1.Y;
793  float rightM23 = right.Row1.Z;
794  float rightM31 = right.Row2.X;
795  float rightM32 = right.Row2.Y;
796  float rightM33 = right.Row2.Z;
797 
798  result.Row0.X = leftM11 * rightM11 + leftM12 * rightM21 + leftM13 * rightM31;
799  result.Row0.Y = leftM11 * rightM12 + leftM12 * rightM22 + leftM13 * rightM32;
800  result.Row0.Z = leftM11 * rightM13 + leftM12 * rightM23 + leftM13 * rightM33;
801  result.Row1.X = leftM21 * rightM11 + leftM22 * rightM21 + leftM23 * rightM31;
802  result.Row1.Y = leftM21 * rightM12 + leftM22 * rightM22 + leftM23 * rightM32;
803  result.Row1.Z = leftM21 * rightM13 + leftM22 * rightM23 + leftM23 * rightM33;
804  result.Row2.X = leftM31 * rightM11 + leftM32 * rightM21 + leftM33 * rightM31;
805  result.Row2.Y = leftM31 * rightM12 + leftM32 * rightM22 + leftM33 * rightM32;
806  result.Row2.Z = leftM31 * rightM13 + leftM32 * rightM23 + leftM33 * rightM33;

815  public static void Invert(in Matrix3 matrix, out Matrix3 result)
816  {
817  // Original implementation can be found here:
818  // https://github.com/niswegmann/small-matrix-inverse/blob/6eac02b84ad06870692abaf828638a391548502c/invert3x3_c.h
819  float row0X = matrix.Row0.X, row0Y = matrix.Row0.Y, row0Z = matrix.Row0.Z;
820  float row1X = matrix.Row1.X, row1Y = matrix.Row1.Y, row1Z = matrix.Row1.Z;
821  float row2X = matrix.Row2.X, row2Y = matrix.Row2.Y, row2Z = matrix.Row2.Z;
822 
823  // Compute the elements needed to calculate the determinant
824  // so that we can throw without writing anything to the out parameter.
825  float invRow0X = +row1Y * row2Z - row1Z * row2Y;
826  float invRow1X = -row1X * row2Z + row1Z * row2X;
827  float invRow2X = +row1X * row2Y - row1Y * row2X;
828 
829  // Compute determinant:
830  float det = row0X * invRow0X + row0Y * invRow1X + row0Z * invRow2X;
831 
832  if (det == 0f)
833  {
834  throw new InvalidOperationException("Matrix is singular and cannot be inverted.");
835  }
836 
837  // Compute adjoint:
838  result.Row0.X = invRow0X;
839  result.Row0.Y = -row0Y * row2Z + row0Z * row2Y;
840  result.Row0.Z = +row0Y * row1Z - row0Z * row1Y;
841  result.Row1.X = invRow1X;
842  result.Row1.Y = +row0X * row2Z - row0Z * row2X;
843  result.Row1.Z = -row0X * row1Z + row0Z * row1X;
844  result.Row2.X = invRow2X;
845  result.Row2.Y = -row0X * row2Y + row0Y * row2X;
846  result.Row2.Z = +row0X * row1Y - row0Y * row1X;
847 
848  // Multiply adjoint with reciprocal of determinant:
849  det = 1.0f / det;
850 
851  result.Row0.X *= det;
852  result.Row0.Y *= det;
853  result.Row0.Z *= det;
854  result.Row1.X *= det;
855  result.Row1.Y *= det;
856  result.Row1.Z *= det;
857  result.Row2.X *= det;
858  result.Row2.Y *= det;
859  result.Row2.Z *= det;
860  }
