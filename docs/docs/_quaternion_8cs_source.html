203  public Vector3 ToEulerAngles()
204  {
205  /*
206  reference
207  http://en.wikipedia.org/wiki/Conversion_between_quaternions_and_Euler_angles
208  http://www.euclideanspace.com/maths/geometry/rotations/conversions/quaternionToEuler/
209  */
210 
211  Quaternion q = this;
212 
213  Vector3 eulerAngles;
214 
215  // Threshold for the singularities found at the north/south poles.
216  const float SINGULARITY_THRESHOLD = 0.4999995f;
217 
218  float sqw = q.W * q.W;
219  float sqx = q.X * q.X;
220  float sqy = q.Y * q.Y;
221  float sqz = q.Z * q.Z;
222  float unit = sqx + sqy + sqz + sqw; // if normalised is one, otherwise is correction factor
223  float singularityTest = q.X * q.Z + q.W * q.Y;
224 
225  if (singularityTest > SINGULARITY_THRESHOLD * unit)
226  {
227  eulerAngles.Z = (float) (2 * Math.Atan2(q.X, q.W));
228  eulerAngles.Y = MathHelper.PiOver2;
229  eulerAngles.X = 0;
230  }
231  else if (singularityTest < -SINGULARITY_THRESHOLD * unit)
232  {
233  eulerAngles.Z = (float) (-2 * Math.Atan2(q.X, q.W));
234  eulerAngles.Y = -MathHelper.PiOver2;
235  eulerAngles.X = 0;
236  }
237  else
238  {
239  eulerAngles.Z = MathF.Atan2(2 * (q.W * q.Z - q.X * q.Y), sqw + sqx - sqy - sqz);
240  eulerAngles.Y = MathF.Asin(2 * singularityTest / unit);
241  eulerAngles.X = MathF.Atan2(2 * (q.W * q.X - q.Y * q.Z), sqw - sqx - sqy + sqz);
242  }
243 
244  return eulerAngles;
245  }

588  public static void FromMatrix(in Matrix3 matrix, out Quaternion result)
589  {
590  float trace = matrix.Trace;
591 
592  if (trace > 0)
593  {
594  float s = MathF.Sqrt(trace + 1) * 2;
595  float invS = 1f / s;
596 
597  result.W = s * 0.25f;
598  result.Xyz.X = (matrix.Row2.Y - matrix.Row1.Z) * invS;
599  result.Xyz.Y = (matrix.Row0.Z - matrix.Row2.X) * invS;
600  result.Xyz.Z = (matrix.Row1.X - matrix.Row0.Y) * invS;
601  }
602  else
603  {
604  float m00 = matrix.Row0.X, m11 = matrix.Row1.Y, m22 = matrix.Row2.Z;
605 
606  if (m00 > m11 && m00 > m22)
607  {
608  float s = MathF.Sqrt(1 + m00 - m11 - m22) * 2;
609  float invS = 1f / s;
610 
611  result.W = (matrix.Row2.Y - matrix.Row1.Z) * invS;
612  result.Xyz.X = s * 0.25f;
613  result.Xyz.Y = (matrix.Row0.Y + matrix.Row1.X) * invS;
614  result.Xyz.Z = (matrix.Row0.Z + matrix.Row2.X) * invS;
615  }
616  else if (m11 > m22)
617  {
618  float s = MathF.Sqrt(1 + m11 - m00 - m22) * 2;
619  float invS = 1f / s;
620 
621  result.W = (matrix.Row0.Z - matrix.Row2.X) * invS;
622  result.Xyz.X = (matrix.Row0.Y + matrix.Row1.X) * invS;
623  result.Xyz.Y = s * 0.25f;
624  result.Xyz.Z = (matrix.Row1.Z + matrix.Row2.Y) * invS;
625  }
626  else
627  {
628  float s = MathF.Sqrt(1 + m22 - m00 - m11) * 2;
629  float invS = 1f / s;
630 
631  result.W = (matrix.Row1.X - matrix.Row0.Y) * invS;
632  result.Xyz.X = (matrix.Row0.Z + matrix.Row2.X) * invS;
633  result.Xyz.Y = (matrix.Row1.Z + matrix.Row2.Y) * invS;
634  result.Xyz.Z = s * 0.25f;
635  }
636  }
637  }

646  [Pure]
647  public static Quaternion Slerp(Quaternion q1, Quaternion q2, float blend)
648  {
649  // if either input is zero, return the other.
650  if (q1.LengthSquared == 0.0f)
651  {
652  if (q2.LengthSquared == 0.0f)
653  {
654  return Identity;
655  }
656 
657  return q2;
658  }
659 
660  if (q2.LengthSquared == 0.0f)
661  {
662  return q1;
663  }
664 
665  float cosHalfAngle = q1.W * q2.W + Vector3.Dot(q1.Xyz, q2.Xyz);
666 
667  if (cosHalfAngle >= 1.0f || cosHalfAngle <= -1.0f)
668  {
669  // angle = 0.0f, so just return one input.
670  return q1;
671  }
672 
673  if (cosHalfAngle < 0.0f)
674  {
675  q2.Xyz = -q2.Xyz;
676  q2.W = -q2.W;
677  cosHalfAngle = -cosHalfAngle;
678  }
679 
680  float blendA;
681  float blendB;
682  if (cosHalfAngle < 0.99f)
683  {
684  // do proper slerp for big angles
685  float halfAngle = MathF.Acos(cosHalfAngle);
686  float sinHalfAngle = MathF.Sin(halfAngle);
687  float oneOverSinHalfAngle = 1.0f / sinHalfAngle;
688  blendA = MathF.Sin(halfAngle * (1.0f - blend)) * oneOverSinHalfAngle;
689  blendB = MathF.Sin(halfAngle * blend) * oneOverSinHalfAngle;
690  }
691  else
692  {
693  // do lerp if angle is really small.
694  blendA = 1.0f - blend;
695  blendB = blend;
696  }
697 
698  Quaternion result = new Quaternion(blendA * q1.Xyz + blendB * q2.Xyz, blendA * q1.W + blendB * q2.W);
699  if (result.LengthSquared > 0.0f)
700  {
701  return Normalize(result);
702  }
703 
704  return Identity;
705  }
