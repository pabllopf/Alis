159  [Pure]
160  private ushort SingleToHalf(int si32)
161  {
162  // Our floating point number, F, is represented by the bit pattern in integer i.
163  // Disassemble that bit pattern into the sign, S, the exponent, E, and the significand, M.
164  // Shift S into the position where it will go in in the resulting half number.
165  // Adjust E, accounting for the different exponent bias of float and half (127 versus 15).
166  int sign = (si32 >> 16) & 0x00008000;
167  int exponent = ((si32 >> 23) & 0x000000ff) - (127 - 15);
168  int mantissa = si32 & 0x007fffff;
169 
170  // Now reassemble S, E and M into a half:
171  if (exponent <= 0)
172  {
173  if (exponent < -10)
174  {
175  // E is less than -10. The absolute value of F is less than Half.MinValue
176  // (F may be a small normalized float, a denormalized float or a zero).
177  //
178  // We convert F to a half zero with the same sign as F.
179  return (ushort) sign;
180  }
181 
182  // E is between -10 and 0. F is a normalized float whose magnitude is less than Half.MinNormalizedValue.
183  //
184  // We convert F to a denormalized half.
185 
186  // Add an explicit leading 1 to the significand.
187  mantissa = mantissa | 0x00800000;
188 
189  // Round to M to the nearest (10+E)-bit value (with E between -10 and 0); in case of a tie, round to the nearest even value.
190  //
191  // Rounding may cause the significand to overflow and make our number normalized. Because of the way a half's bits
192  // are laid out, we don't have to treat this case separately; the code below will handle it correctly.
193  int t = 14 - exponent;
194  int a = (1 << (t - 1)) - 1;
195  int b = (mantissa >> t) & 1;
196 
197  mantissa = (mantissa + a + b) >> t;
198 
199  // Assemble the half from S, E (==zero) and M.
200  return (ushort) (sign | mantissa);
201  }
202 
203  if (exponent == 0xff - (127 - 15))
204  {
205  if (mantissa == 0)
206  {
207  // F is an infinity; convert F to a half infinity with the same sign as F.
208  return (ushort) (sign | 0x7c00);
209  }
210 
211  // F is a NAN; we produce a half NAN that preserves the sign bit and the 10 leftmost bits of the
212  // significand of F, with one exception: If the 10 leftmost bits are all zero, the NAN would turn
213  // into an infinity, so we have to set at least one bit in the significand.
214  mantissa >>= 13;
215  return (ushort) (sign | 0x7c00 | mantissa | (mantissa == 0 ? 1 : 0));
216  }
217 
218  // E is greater than zero. F is a normalized float. We try to convert F to a normalized half.
219 
220  // Round to M to the nearest 10-bit value. In case of a tie, round to the nearest even value.
221  mantissa = mantissa + 0x00000fff + ((mantissa >> 13) & 1);
222 
223  if ((mantissa & 0x00800000) != 0)
224  {
225  mantissa = 0; // overflow in significand,
226  exponent += 1; // adjust exponent
227  }
228 
229  // exponent overflow
230  if (exponent > 30)
231  {
232  throw new ArithmeticException("Half: Hardware floating-point overflow.");
233  }
234 
235  // Assemble the half from S, E and M.
236  return (ushort) (sign | (exponent << 10) | (mantissa >> 13));
237  }

256  [Pure]
257  private int HalfToFloat(ushort ui16)
258  {
259  int sign = (ui16 >> 15) & 0x00000001;
260  int exponent = (ui16 >> 10) & 0x0000001f;
261  int mantissa = ui16 & 0x000003ff;
262 
263  if (exponent == 0)
264  {
265  if (mantissa == 0)
266  {
267  // Plus or minus zero
268  return sign << 31;
269  }
270 
271  // Denormalized number -- renormalize it
272  while ((mantissa & 0x00000400) == 0)
273  {
274  mantissa <<= 1;
275  exponent -= 1;
276  }
277 
278  exponent += 1;
279  mantissa &= ~0x00000400;
280  }
281  else if (exponent == 31)
282  {
283  if (mantissa == 0)
284  {
285  // Positive or negative infinity
286  return (sign << 31) | 0x7f800000;
287  }
288 
289  // Nan -- preserve sign and significand bits
290  return (sign << 31) | 0x7f800000 | (mantissa << 13);
291  }
292 
293  // Normalized number
294  exponent = exponent + (127 - 15);
295  mantissa = mantissa << 13;
296 
297  // Assemble S, E and M.
298  return (sign << 31) | (exponent << 23) | mantissa;
299  }
