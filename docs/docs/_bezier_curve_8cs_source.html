218  public static Vector2 CalculatePoint(IList<Vector2> points, float t, float parallel)
219  {
220  Vector2 r = default(Vector2);
221  double c = 1.0d - t;
222  float temp;
223  int i = 0;
224 
225  foreach (Vector2 pt in points)
226  {
227  temp = MathHelper.BinomialCoefficient
228  (
229  points.Count - 1, i) * (float) (Math.Pow(t, i) * Math.Pow(c, points.Count - 1 - i)
230  );
231 
232  r.X += temp * pt.X;
233  r.Y += temp * pt.Y;
234  i++;
235  }
236 
237  if (parallel == 0.0f)
238  {
239  return r;
240  }
241 
242  Vector2 perpendicular;
243 
244  if (t != 0.0f)
245  {
246  perpendicular = r - CalculatePointOfDerivative(points, t);
247  }
248  else
249  {
250  perpendicular = points[1] - points[0];
251  }
252 
253  return r + Vector2.Normalize(perpendicular).PerpendicularRight * parallel;
254  }
