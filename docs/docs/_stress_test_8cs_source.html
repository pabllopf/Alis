100  public async Task Run()
101  {
102  // NOTE: if the service is so busy that it cannot respond to a PING within the KeepAliveInterval interval the websocket connection will be closed
103  // To run extreme tests it is best to set the KeepAliveInterval to TimeSpan.Zero to disable ping pong
104  WebSocketClientOptions options = new WebSocketClientOptions
105  {NoDelay = true, KeepAliveInterval = TimeSpan.FromSeconds(2), SecWebSocketProtocol = "chatV2, chatV1"};
106  using (_webSocket = await _clientFactory.ConnectAsync(_uri, options))
107  {
108  CancellationTokenSource source = new CancellationTokenSource();
109  _token = source.Token;
110 
111  Random rand = new Random(_seed);
112  _expectedValues = new byte[50][];
113  for (int i = 0; i < _expectedValues.Length; i++)
114  {
115  int numBytes = rand.Next(_minNumBytesPerMessage, _maxNumBytesPerMessage);
116  byte[] bytes = new byte[numBytes];
117  rand.NextBytes(bytes);
118  _expectedValues[i] = bytes;
119  }
120 
121  Task recTask = Task.Run(ReceiveLoop);
122  byte[] sendBuffer = new byte[_maxNumBytesPerMessage];
123  for (int i = 0; i < _numItems; i++)
124  {
125  int index = i % _expectedValues.Length;
126  byte[] bytes = _expectedValues[index];
127  Buffer.BlockCopy(bytes, 0, sendBuffer, 0, bytes.Length);
128  ArraySegment<byte> buffer = new ArraySegment<byte>(sendBuffer, 0, bytes.Length);
129  await _webSocket.SendAsync(buffer, WebSocketMessageType.Binary, true, source.Token);
130  }
131 
132  await _webSocket.CloseAsync(WebSocketCloseStatus.NormalClosure, null, source.Token);
133  recTask.Wait();
134  }
135  }

144  private static bool AreEqual(byte[] actual, byte[] expected, int countActual)
145  {
146  if (countActual != expected.Length)
147  {
148  return false;
149  }
150 
151  for (int i = 0; i < countActual; i++)
152  {
153  if (actual[i] != expected[i])
154  {
155  return false;
156  }
157  }
158 
159  return true;

167  private async Task ReceiveLoop()
168  {
169  // the recArray should be large enough to at least receive control frames like Ping and Close frames (with payload)
170  const int MIN_BUFFER_SIZE = 510;
171  int size = _maxNumBytesPerMessage < MIN_BUFFER_SIZE ? MIN_BUFFER_SIZE : _maxNumBytesPerMessage;
172  byte[] recArray = new byte[size];
173  ArraySegment<byte> recBuffer = new ArraySegment<byte>(recArray);
174 
175  int i = 0;
176  while (true)
177  {
178  WebSocketReceiveResult result = await _webSocket.ReceiveAsync(recBuffer, _token);
179 
180  if (!result.EndOfMessage)
181  {
182  throw new Exception("Multi frame messages not supported");
183  }
184 
185  if (result.MessageType == WebSocketMessageType.Close || _token.IsCancellationRequested)
186  {
187  return;
188  }
189 
190  if (result.Count == 0)
191  {
192  await _webSocket.CloseOutputAsync(WebSocketCloseStatus.InvalidPayloadData, "Zero bytes in payload",
193  _token);
194  return;
195  }
196 
197  byte[] valueActual = recBuffer.Array;
198  int index = i % _expectedValues.Length;
199  i++;
200  byte[] valueExpected = _expectedValues[index];
201 
202  if (!AreEqual(valueActual, valueExpected, result.Count))
203  {
204  await _webSocket.CloseOutputAsync(WebSocketCloseStatus.InvalidPayloadData,
205  "Value actual does not equal value expected", _token);
206  throw new Exception(
207  $"Expected: {valueExpected.Length} bytes Actual: {result.Count} bytes. Contents different.");
208  }
209  }
210  }
211  }
