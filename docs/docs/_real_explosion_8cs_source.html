46  /* Methodology:
47  * Force applied at a ray is inversely proportional to the square of distance from source
48  * AABB is used to query for shapes that may be affected
49  * For each RIGID BODY (not shape -- this is an optimization) that is matched, loop through its vertices to determine
50  * the extreme points -- if there is structure that contains outlining polygon, use that as an additional optimization
51  * Evenly cast a number of rays against the shape - number roughly proportional to the arc coverage
52  * - Something like every 3 degrees should do the trick although this can be altered depending on the distance (if really close don't need such a high density of rays)
53  * - There should be a minimum number of rays (3-5?) applied to each body so that small bodies far away are still accurately modeled
54  * - Be sure to have the forces of each ray be proportional to the average arc length covered by each.
55  * For each ray that actually intersects with the shape (non intersections indicate something blocking the path of explosion):
56  * - Apply the appropriate force dotted with the negative of the collision normal at the collision point
57  * - Optionally apply linear interpolation between aforementioned Normal force and the original explosion force in the direction of ray to simulate "surface friction" of sorts
58  */

115  public Dictionary<Fixture, Vector2> Activate(Vector2 pos, float radius, float maxForce)
116  {
117  Aabb aabb;
118  aabb.LowerBound = pos + new Vector2(-radius, -radius);
119  aabb.UpperBound = pos + new Vector2(radius, radius);
120  Fixture[] shapes = new Fixture[MaxShapes];
121 
122  // More than 5 shapes in an explosion could be possible, but still strange.
123  Fixture[] containedShapes = new Fixture[5];
124  bool exit = false;
125 
126  int shapeCount = 0;
127  int containedShapeCount = 0;
128 
129  // Query the world for overlapping shapes.
130  World.TestPointAllFixtures.ForEach(fixture =>
131  {
132  if (fixture.TestPoint(ref pos))
133  {
134  if (IgnoreWhenInsideShape)
135  {
136  exit = true;
137  }
138  else
139  {
140  containedShapes[containedShapeCount++] = fixture;
141  }
142  }
143  else
144  {
145  shapes[shapeCount++] = fixture;
146  }
147  });
148 
149  if (exit)
150  {
151  return new Dictionary<Fixture, Vector2>();
152  }
153 
154  Dictionary<Fixture, Vector2> exploded = new Dictionary<Fixture, Vector2>(shapeCount + containedShapeCount);
155 
156  // Per shape max/min angles for now.
157  float[] vals = new float[shapeCount * 2];
158  int valIndex = 0;
159  for (int i = 0; i < shapeCount; ++i)
160  {
161  PolygonShape ps;
162  if (shapes[i].Shape is CircleShape cs)
163  {
164  // We create a "diamond" approximation of the circle
165  Vertices v = new Vertices();
166  Vector2 vec = Vector2.Zero + new Vector2(cs.RadiusPrivate, 0);
167  v.Add(vec);
168  vec = Vector2.Zero + new Vector2(0, cs.RadiusPrivate);
169  v.Add(vec);
170  vec = Vector2.Zero + new Vector2(-cs.RadiusPrivate, cs.RadiusPrivate);
171  v.Add(vec);
172  vec = Vector2.Zero + new Vector2(0, -cs.RadiusPrivate);
173  v.Add(vec);
174  ps = new PolygonShape(v, 0);
175  }
176  else
177  {
178  ps = shapes[i].Shape as PolygonShape;
179  }
180 
181  if (shapes[i].Body.BodyType == BodyType.Dynamic && ps != null)
182  {
183  Vector2 toCentroid = shapes[i].Body.GetWorldPoint(ps.MassDataPrivate.Centroid) - pos;
184  float angleToCentroid = (float) Math.Atan2(toCentroid.Y, toCentroid.X);
185  float min = float.MaxValue;
186  float max = float.MinValue;
187  float minAbsolute = 0.0f;
188  float maxAbsolute = 0.0f;
189 
190  for (int j = 0; j < ps.VerticesPrivate.Count; ++j)
191  {
192  Vector2 toVertex = shapes[i].Body.GetWorldPoint(ps.VerticesPrivate[j]) - pos;
193  float newAngle = (float) Math.Atan2(toVertex.Y, toVertex.X);
194  float diff = newAngle - angleToCentroid;
195 
196  diff = (diff - MathConstants.Pi) % (2 * MathConstants.Pi);
197 
198  // the minus pi is important. It means cutoff for going other direction is at 180 deg where it needs to be
199 
200  if (diff < 0.0f)
201  {
202  diff += 2 * MathConstants.Pi; // correction for not handling negs
203  }
204 
205  diff -= MathConstants.Pi;
206 
207  if (Math.Abs(diff) > MathConstants.Pi)
208  {
209  continue; // Something's wrong, point not in shape but exists angle diff > 180
210  }
211 
212  if (diff > max)
213  {
214  max = diff;
215  maxAbsolute = newAngle;
216  }
217 
218  if (diff < min)
219  {
220  min = diff;
221  minAbsolute = newAngle;
222  }
223  }
224 
225  vals[valIndex] = minAbsolute;
226  ++valIndex;
227  vals[valIndex] = maxAbsolute;
228  ++valIndex;
229  }
230  }
231 
232  Array.Sort(vals, 0, valIndex, rdc);
233  data.Clear();
234  bool rayMissed = true;
235 
236  for (int i = 0; i < valIndex; ++i)
237  {
238  Fixture fixture = null;
239  float midpt;
240 
241  int iplus = i == valIndex - 1 ? 0 : i + 1;
242  if (vals[i] == vals[iplus])
243  {
244  continue;
245  }
246 
247  if (i == valIndex - 1)
248  {
249  // the single edgecase
250  midpt = vals[0] + MathConstants.Pi * 2 + vals[i];
251  }
252  else
253  {
254  midpt = vals[i + 1] + vals[i];
255  }
256 
257  midpt /= 2;
258 
259  Vector2 p1 = pos;
260  Vector2 p2 = radius * new Vector2((float) Math.Cos(midpt), (float) Math.Sin(midpt)) + pos;
261 
262  // RaycastOne
263  bool hitClosest = false;
264  World.RayCast((f, p, n, fr) =>
265  {
266  Body body = f.Body;
267 
268  if (!IsActiveOn(body))
269  {
270  return 0;
271  }
272 
273  hitClosest = true;
274  fixture = f;
275  return fr;
276  }, p1, p2);
277 
278  //draws radius points
279  if (hitClosest && fixture.Body.BodyType == BodyType.Dynamic)
280  {
281  if (data.Any() && data.Last().Body == fixture.Body && !rayMissed)
282  {
283  int laPos = data.Count - 1;
284  ShapeData la = data[laPos];
285  la.Max = vals[iplus];
286  data[laPos] = la;
287  }
288  else
289  {
290  // make new
291  ShapeData d;
292  d.Body = fixture.Body;
293  d.Min = vals[i];
294  d.Max = vals[iplus];
295  data.Add(d);
296  }
297 
298  if (data.Count > 1
299  && i == valIndex - 1
300  && data.Last().Body == data.First().Body
301  && data.Last().Max == data.First().Min)
302  {
303  ShapeData fi = data[0];
304  fi.Min = data.Last().Min;
305  data.RemoveAt(data.Count - 1);
306  data[0] = fi;
307  while (data.First().Min >= data.First().Max)
308  {
309  fi.Min -= MathConstants.Pi * 2;
310  data[0] = fi;
311  }
312  }
313 
314  int lastPos = data.Count - 1;
315  ShapeData last = data[lastPos];
316  while (data.Count > 0
317  && data.Last().Min >= data.Last().Max) // just making sure min<max
318  {
319  last.Min = data.Last().Min - 2 * MathConstants.Pi;
320  data[lastPos] = last;
321  }
322 
323  rayMissed = false;
324  }
325  else
326  {
327  rayMissed = true; // raycast did not find a shape
328  }
329  }
330 
331  for (int i = 0; i < data.Count; ++i)
332  {
333  if (!IsActiveOn(data[i].Body))
334  {
335  continue;
336  }
337 
338  float arclen = data[i].Max - data[i].Min;
339 
340  float first = MathHelper.Min(MaxEdgeOffset, EdgeRatio * arclen);
341  int insertedRays = (int) Math.Ceiling((arclen - 2.0f * first - (MinRays - 1) * MaxAngle) / MaxAngle);
342 
343  if (insertedRays < 0)
344  {
345  insertedRays = 0;
346  }
347 
348  float offset = (arclen - first * 2.0f) / ((float) MinRays + insertedRays - 1);
349 
350  //Note: This loop can go into infinite as it operates on floats.
351  //Added FloatEquals with a large epsilon.
352  for (float j = data[i].Min + first;
353  j < data[i].Max || MathUtils.FloatEquals(j, data[i].Max, 0.0001f);
354  j += offset)
355  {
356  Vector2 p1 = pos;
357  Vector2 p2 = pos + radius * new Vector2((float) Math.Cos(j), (float) Math.Sin(j));
358  Vector2 hitpoint = Vector2.Zero;
359  float minlambda = float.MaxValue;
360 
361  List<Fixture> fl = data[i].Body.FixtureList;
362  for (int x = 0; x < fl.Count; x++)
363  {
364  Fixture f = fl[x];
365  RayCastInput ri;
366  ri.Point1 = p1;
367  ri.Point2 = p2;
368  ri.MaxFraction = 50f;
369 
370  if (f.RayCast(out RayCastOutput ro, ref ri, 0))
371  {
372  if (minlambda > ro.Fraction)
373  {
374  minlambda = ro.Fraction;
375  hitpoint = ro.Fraction * p2 + (1 - ro.Fraction) * p1;
376  }
377  }
378 
379  // the force that is to be applied for this particular ray.
380  // offset is angular coverage. lambda*length of segment is distance.
381  float impulse = arclen / (MinRays + insertedRays) * maxForce * 180.0f / MathConstants.Pi *
382  (1.0f - Math.Min(1.0f, minlambda));
383 
384  // We Apply the impulse!!!
385  Vector2 vectImp =
386  Vector2.Dot(impulse * new Vector2((float) Math.Cos(j), (float) Math.Sin(j)), -ro.Normal) *
387  new Vector2((float) Math.Cos(j), (float) Math.Sin(j));
388  data[i].Body.ApplyLinearImpulse(ref vectImp, ref hitpoint);
389 
390  // We gather the fixtures for returning them
391  if (exploded.ContainsKey(f))
392  {
393  exploded[f] += vectImp;
394  }
395  else
396  {
397  exploded.Add(f, vectImp);
398  }
399 
400  if (minlambda > 1.0f)
401  {
402  hitpoint = p2;
403  }
404  }
405  }
406  }
407 
408  // We check contained shapes
409  for (int i = 0; i < containedShapeCount; ++i)
410  {
411  Fixture fix = containedShapes[i];
412 
413  if (!IsActiveOn(fix.Body))
414  {
415  continue;
416  }
417 
418  float impulse = MinRays * maxForce * 180.0f / MathConstants.Pi;
419  Vector2 hitPoint;
420 
421  if (fix.Shape is CircleShape circShape)
422  {
423  hitPoint = fix.Body.GetWorldPoint(circShape.Position);
424  }
425  else
426  {
427  PolygonShape shape = fix.Shape as PolygonShape;
428  hitPoint = fix.Body.GetWorldPoint(shape.MassDataPrivate.Centroid);
429  }
430 
431  Vector2 vectImp = impulse * (hitPoint - pos);
432 
433  fix.Body.ApplyLinearImpulse(ref vectImp, ref hitPoint);
434 
435  if (!exploded.ContainsKey(fix))
436  {
437  exploded.Add(fix, vectImp);
438  }
439  }
440 
441  return exploded;
442  }
443  }
444 }
