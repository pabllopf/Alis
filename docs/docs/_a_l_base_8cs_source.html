61  protected internal static TDelegate LoadDelegate<TDelegate>(string name) where TDelegate : Delegate
62  {
63  IntPtr ptr = AL.AL.GetProcAddress(name);
64  if (ptr == IntPtr.Zero)
65  {
66  // If we can't load the function for whatever reason we dynamically generate a delegate to
67  // give the user an error message that is actually understandable.
68  MethodInfo invoke = typeof(TDelegate).GetMethod("Invoke");
69  Type returnType = invoke.ReturnType;
70  Type[] parameters = invoke.GetParameters().Select(p => p.ParameterType).ToArray();
71  DynamicMethod method = new DynamicMethod(
72  "OpenAL_AL_Extension_Null_GetProcAddress_Exception_Delegate_" + Guid.NewGuid(),
73  returnType,
74  parameters);
75  ILGenerator generator = method.GetILGenerator();
76 
77  // Here we are generating a delegate that looks like this:
78  // ((<the arguments that the delegate type takes>) =>
79  // throw new Exception(<error string>);
80  generator.Emit(OpCodes.Ldstr,
81  "This OpenAL function could not be loaded. This likely means that this extension isn't present in the current context.");
82  generator.Emit(OpCodes.Newobj, typeof(Exception).GetConstructor(new[] {typeof(string)}));
83  generator.Emit(OpCodes.Throw);
84 
85  return (TDelegate) method.CreateDelegate(typeof(TDelegate));
86  }
87 
88  return Marshal.GetDelegateForFunctionPointer<TDelegate>(ptr);
89  }
90  }
