124  [Pure]
125  public Vector2 CalculatePoint(float t)
126  {
127  float c = 1.0f - t;
128 
129  float x = StartAnchor.X * c * c * c + FirstControlPoint.X * 3 * t * c * c +
130  SecondControlPoint.X * 3 * t * t * c + EndAnchor.X * t * t * t;
131 
132  float y = StartAnchor.Y * c * c * c + FirstControlPoint.Y * 3 * t * c * c +
133  SecondControlPoint.Y * 3 * t * t * c + EndAnchor.Y * t * t * t;
134 
135  Vector2 r = new Vector2(x, y);
136 
137  if (Parallel == 0.0f)
138  {
139  return r;
140  }
141 
142  Vector2 perpendicular;
143 
144  if (t == 0.0f)
145  {
146  perpendicular = FirstControlPoint - StartAnchor;
147  }
148  else
149  {
150  perpendicular = r - CalculatePointOfDerivative(t);
151  }
152 
153  return r + Vector2.Normalize(perpendicular).PerpendicularRight * Parallel;
154  }

161  [Pure]
162  private Vector2 CalculatePointOfDerivative(float t)
163  {
164  float c = 1.0f - t;
165  Vector2 r = new Vector2
166  (
167  c * c * StartAnchor.X + 2 * t * c * FirstControlPoint.X + t * t * SecondControlPoint.X,
168  c * c * StartAnchor.Y + 2 * t * c * FirstControlPoint.Y + t * t * SecondControlPoint.Y
169  );
170 
171  return r;
172  }
