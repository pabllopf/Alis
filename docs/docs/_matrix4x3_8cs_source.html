332  public static void CreateFromAxisAngle(Vector3 axis, float angle, out Matrix4x3 result)
333  {
334  axis.Normalize();
335  float axisX = axis.X, axisY = axis.Y, axisZ = axis.Z;
336 
337  float cos = MathF.Cos(-angle);
338  float sin = MathF.Sin(-angle);
339  float t = 1.0f - cos;
340 
341  float tXX = t * axisX * axisX;
342  float tXY = t * axisX * axisY;
343  float tXZ = t * axisX * axisZ;
344  float tYY = t * axisY * axisY;
345  float tYZ = t * axisY * axisZ;
346  float tZZ = t * axisZ * axisZ;
347 
348  float sinX = sin * axisX;
349  float sinY = sin * axisY;
350  float sinZ = sin * axisZ;
351 
352  result.Row0.X = tXX + cos;
353  result.Row0.Y = tXY - sinZ;
354  result.Row0.Z = tXZ + sinY;
355  result.Row1.X = tXY + sinZ;
356  result.Row1.Y = tYY + cos;
357  result.Row1.Z = tYZ - sinX;
358  result.Row2.X = tXZ - sinY;
359  result.Row2.Y = tYZ + sinX;
360  result.Row2.Z = tZZ + cos;
361  result.Row3.X = 0;
362  result.Row3.Y = 0;
363  result.Row3.Z = 0;
364  }

384  public static void CreateFromQuaternion(in Quaternion q, out Matrix4x3 result)
385  {
386  float x = q.X;
387  float y = q.Y;
388  float z = q.Z;
389  float w = q.W;
390  float tx = 2 * x;
391  float ty = 2 * y;
392  float tz = 2 * z;
393  float txx = tx * x;
394  float tyy = ty * y;
395  float tzz = tz * z;
396  float txy = tx * y;
397  float txz = tx * z;
398  float tyz = ty * z;
399  float twx = w * tx;
400  float twy = w * ty;
401  float twz = w * tz;
402 
403  result.Row0.X = 1f - tyy - tzz;
404  result.Row0.Y = txy - twz;
405  result.Row0.Z = txz + twy;
406  result.Row1.X = txy + twz;
407  result.Row1.Y = 1f - txx - tzz;
408  result.Row1.Z = tyz - twx;
409  result.Row2.X = txz - twy;
410  result.Row2.Y = tyz + twx;
411  result.Row2.Z = 1f - txx - tyy;
412  result.Row3.X = 0;
413  result.Row3.Y = 0;
414  result.Row3.Z = 0;
415  }

670  public static void Mult(in Matrix4x3 left, in Matrix3x4 right, out Matrix4 result)
671  {
672  float leftM11 = left.Row0.X;
673  float leftM12 = left.Row0.Y;
674  float leftM13 = left.Row0.Z;
675  float leftM21 = left.Row1.X;
676  float leftM22 = left.Row1.Y;
677  float leftM23 = left.Row1.Z;
678  float leftM31 = left.Row2.X;
679  float leftM32 = left.Row2.Y;
680  float leftM33 = left.Row2.Z;
681  float leftM41 = left.Row3.X;
682  float leftM42 = left.Row3.Y;
683  float leftM43 = left.Row3.Z;
684  float rightM11 = right.Row0.X;
685  float rightM12 = right.Row0.Y;
686  float rightM13 = right.Row0.Z;
687  float rightM14 = right.Row0.W;
688  float rightM21 = right.Row1.X;
689  float rightM22 = right.Row1.Y;
690  float rightM23 = right.Row1.Z;
691  float rightM24 = right.Row1.W;
692  float rightM31 = right.Row2.X;
693  float rightM32 = right.Row2.Y;
694  float rightM33 = right.Row2.Z;
695  float rightM34 = right.Row2.W;
696 
697  result.Row0.X = leftM11 * rightM11 + leftM12 * rightM21 + leftM13 * rightM31;
698  result.Row0.Y = leftM11 * rightM12 + leftM12 * rightM22 + leftM13 * rightM32;
699  result.Row0.Z = leftM11 * rightM13 + leftM12 * rightM23 + leftM13 * rightM33;
700  result.Row0.W = leftM11 * rightM14 + leftM12 * rightM24 + leftM13 * rightM34;
701  result.Row1.X = leftM21 * rightM11 + leftM22 * rightM21 + leftM23 * rightM31;
702  result.Row1.Y = leftM21 * rightM12 + leftM22 * rightM22 + leftM23 * rightM32;
703  result.Row1.Z = leftM21 * rightM13 + leftM22 * rightM23 + leftM23 * rightM33;
704  result.Row1.W = leftM21 * rightM14 + leftM22 * rightM24 + leftM23 * rightM34;
705  result.Row2.X = leftM31 * rightM11 + leftM32 * rightM21 + leftM33 * rightM31;
706  result.Row2.Y = leftM31 * rightM12 + leftM32 * rightM22 + leftM33 * rightM32;
707  result.Row2.Z = leftM31 * rightM13 + leftM32 * rightM23 + leftM33 * rightM33;
708  result.Row2.W = leftM31 * rightM14 + leftM32 * rightM24 + leftM33 * rightM34;
709  result.Row3.X = leftM41 * rightM11 + leftM42 * rightM21 + leftM43 * rightM31;
710  result.Row3.Y = leftM41 * rightM12 + leftM42 * rightM22 + leftM43 * rightM32;
711  result.Row3.Z = leftM41 * rightM13 + leftM42 * rightM23 + leftM43 * rightM33;
712  result.Row3.W = leftM41 * rightM14 + leftM42 * rightM24 + leftM43 * rightM34;
713  }

737  float leftM11 = left.Row0.X;
738  float leftM12 = left.Row0.Y;
739  float leftM13 = left.Row0.Z;

741  float leftM22 = left.Row1.Y;
742  float leftM23 = left.Row1.Z;
743  float leftM31 = left.Row2.X;
744  float leftM32 = left.Row2.Y;
745  float leftM33 = left.Row2.Z;
746  float leftM41 = left.Row3.X;
747  float leftM42 = left.Row3.Y;
748  float leftM43 = left.Row3.Z;
749  float rightM11 = right.Row0.X;
750  float rightM12 = right.Row0.Y;
751  float rightM13 = right.Row0.Z;
752  float rightM21 = right.Row1.X;

754  float rightM23 = right.Row1.Z;
755  float rightM31 = right.Row2.X;

758  float rightM41 = right.Row3.X;
759  float rightM42 = right.Row3.Y;
760  float rightM43 = right.Row3.Z;
761 
762  result.Row0.X = leftM11 * rightM11 + leftM12 * rightM21 + leftM13 * rightM31 + rightM41;
763  result.Row0.Y = leftM11 * rightM12 + leftM12 * rightM22 + leftM13 * rightM32 + rightM42;
764  result.Row0.Z = leftM11 * rightM13 + leftM12 * rightM23 + leftM13 * rightM33 + rightM43;
765  result.Row1.X = leftM21 * rightM11 + leftM22 * rightM21 + leftM23 * rightM31 + rightM41;
766  result.Row1.Y = leftM21 * rightM12 + leftM22 * rightM22 + leftM23 * rightM32 + rightM42;
767  result.Row1.Z = leftM21 * rightM13 + leftM22 * rightM23 + leftM23 * rightM33 + rightM43;
768  result.Row2.X = leftM31 * rightM11 + leftM32 * rightM21 + leftM33 * rightM31 + rightM41;
769  result.Row2.Y = leftM31 * rightM12 + leftM32 * rightM22 + leftM33 * rightM32 + rightM42;
770  result.Row2.Z = leftM31 * rightM13 + leftM32 * rightM23 + leftM33 * rightM33 + rightM43;
771  result.Row3.X = leftM41 * rightM11 + leftM42 * rightM21 + leftM43 * rightM31 + rightM41;
772  result.Row3.Y = leftM41 * rightM12 + leftM42 * rightM22 + leftM43 * rightM32 + rightM42;
773  result.Row3.Z = leftM41 * rightM13 + leftM42 * rightM23 + leftM43 * rightM33 + rightM43;

876  public static void Invert(in Matrix4x3 mat, out Matrix4x3 result)
877  {
878  Matrix3 inverseRotation = new Matrix3(mat.Column0.Xyz, mat.Column1.Xyz, mat.Column2.Xyz);
879  inverseRotation.Row0 /= inverseRotation.Row0.LengthSquared;
880  inverseRotation.Row1 /= inverseRotation.Row1.LengthSquared;
881  inverseRotation.Row2 /= inverseRotation.Row2.LengthSquared;
882 
883  Vector3 translation = mat.Row3;
884 
885  result.Row0 = inverseRotation.Row0;
886  result.Row1 = inverseRotation.Row1;
887  result.Row2 = inverseRotation.Row2;
888  result.Row3 = new Vector3
889  (
890  -Vector3.Dot(inverseRotation.Row0, translation),
891  -Vector3.Dot(inverseRotation.Row1, translation),
892  -Vector3.Dot(inverseRotation.Row2, translation)
893  );
894  }
