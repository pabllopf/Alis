1082  [Pure]
1083  public static bool ApproximatelyEqual(float a, float b, int maxDeltaBits)
1084  {
1085  // we use longs here, otherwise we run into a two's complement problem, causing this to fail with -2 and 2.0
1086  long k = BitConverter.SingleToInt32Bits(a);
1087  if (k < 0)
1088  {
1089  k = int.MinValue - k;
1090  }
1091 
1092  long l = BitConverter.SingleToInt32Bits(b);
1093  if (l < 0)
1094  {
1095  l = int.MinValue - l;
1096  }
1097 
1098  long intDiff = Math.Abs(k - l);
1099  return intDiff <= 1 << maxDeltaBits;

1117  public static bool ApproximatelyEqualEpsilon(double a, double b, double epsilon)
1118  {
1119  const double doubleNormal = (1L << 52) * double.Epsilon;
1120  double absA = Math.Abs(a);
1121  double absB = Math.Abs(b);
1122  double diff = Math.Abs(a - b);
1123 
1124  if (a == b)
1125  {
1126  // Shortcut, handles infinities
1127  return true;

1129

1133  // relative error is less meaningful here
1134  return diff < epsilon * doubleNormal;
1135  }
1136 
1137  // use relative error
1138  return diff / Math.Min(absA + absB, double.MaxValue) < epsilon;
