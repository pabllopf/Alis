225  public static Rot Mul(Rot q, Rot r)
226  {
227  // [qc -qs] * [rc -rs] = [qc*rc-qs*rs -qc*rs-qs*rc]
228  // [qs qc] [rs rc] [qs*rc+qc*rs -qs*rs+qc*rc]
229  // s = qs * rc + qc * rs
230  // c = qc * rc - qs * rs
231  Rot qr;
232  qr.S = q.S * r.C + q.C * r.S;
233  qr.C = q.C * r.C - q.S * r.S;
234  return qr;
235  }

254  public static Rot MulT(Rot q, Rot r)
255  {
256  // [ qc qs] * [rc -rs] = [qc*rc+qs*rs -qc*rs+qs*rc]
257  // [-qs qc] [rs rc] [-qs*rc+qc*rs qs*rs+qc*rc]
258  // s = qc * rs - qs * rc
259  // c = qc * rc + qs * rs
260  Rot qr;

262  qr.C = q.C * r.C + q.S * r.S;
263  return qr;

360  public static double VectorAngle(ref Vector2 p1, ref Vector2 p2)
361  {
362  double theta1 = Math.Atan2(p1.Y, p1.X);
363  double theta2 = Math.Atan2(p2.Y, p2.X);
364  double dtheta = theta2 - theta1;
365 
366  while (dtheta > MathConstants.Pi)
367  {
368  dtheta -= MathConstants.TwoPi;
369  }
370 
371  while (dtheta < -MathConstants.Pi)
372  {
373  dtheta += MathConstants.TwoPi;
374  }
375 
376  return dtheta;
377  }
