984  public static void Transform(in Vector3 vec, in Quaternion quat, out Vector3 result)
985  {
986  // Since vec.W == 0, we can optimize quat * vec * quat^-1 as follows:
987  // vec + 2.0 * cross(quat.xyz, cross(quat.xyz, vec) + quat.w * vec)
988  Vector3 xyz = quat.Xyz;
989  Cross(in xyz, in vec, out Vector3 temp);
990  Multiply(in vec, quat.W, out Vector3 temp2);
991  Add(in temp, in temp2, out temp);
992  Cross(in xyz, in temp, out temp2);
993  Multiply(in temp2, 2f, out temp2);
994  Add(in vec, in temp2, out result);
995  }

1094  [Pure]
1095  public static Vector3 Project
1096  (
1097  Vector3 vector,
1098  float x,
1099  float y,
1100  float width,
1101  float height,
1102  float minZ,
1103  float maxZ,
1104  Matrix4 worldViewProjection
1105  )
1106  {
1107  Vector4 result;
1108 
1109  result.X =
1110  vector.X * worldViewProjection.M11 +
1111  vector.Y * worldViewProjection.M21 +
1112  vector.Z * worldViewProjection.M31 +
1113  worldViewProjection.M41;
1114 
1115  result.Y =
1116  vector.X * worldViewProjection.M12 +
1117  vector.Y * worldViewProjection.M22 +
1118  vector.Z * worldViewProjection.M32 +
1119  worldViewProjection.M42;
1120 
1121  result.Z =
1122  vector.X * worldViewProjection.M13 +
1123  vector.Y * worldViewProjection.M23 +
1124  vector.Z * worldViewProjection.M33 +
1125  worldViewProjection.M43;
1126 
1127  result.W =
1128  vector.X * worldViewProjection.M14 +
1129  vector.Y * worldViewProjection.M24 +
1130  vector.Z * worldViewProjection.M34 +
1131  worldViewProjection.M44;
1132 
1133  result /= result.W;
1134 
1135  result.X = x + width * ((result.X + 1.0f) / 2.0f);
1136  result.Y = y + height * ((result.Y + 1.0f) / 2.0f);
1137  result.Z = minZ + (maxZ - minZ) * ((result.Z + 1.0f) / 2.0f);
1138 
1139  return new Vector3(result.X, result.Y, result.Z);
1140  }

1158  [Pure]
1159  public static Vector3 Unproject
1160  (
1161  Vector3 vector,
1162  float x,
1163  float y,
1164  float width,
1165  float height,
1166  float minZ,
1167  float maxZ,
1168  Matrix4 inverseWorldViewProjection
1169  )
1170  {
1171  float tempX = (vector.X - x) / width * 2.0f - 1.0f;
1172  float tempY = (vector.Y - y) / height * 2.0f - 1.0f;
1173  float tempZ = (vector.Z - minZ) / (maxZ - minZ) * 2.0f - 1.0f;
1174 
1175  Vector3 result;
1176  result.X =
1177  tempX * inverseWorldViewProjection.M11 +
1178  tempY * inverseWorldViewProjection.M21 +
1179  tempZ * inverseWorldViewProjection.M31 +
1180  inverseWorldViewProjection.M41;
