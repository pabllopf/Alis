559  public static void Invert(in Matrix2 mat, out Matrix2 result)
560  {
561  float det = mat.Row0.X * mat.Row1.Y - mat.Row0.Y * mat.Row1.X;
562 
563  if (det == 0)
564  {
565  throw new InvalidOperationException("Matrix is singular and cannot be inverted.");
566  }
567 
568  float invDet = 1f / det;
569 
570  // Because the c# jit assumes alias for byref types we need to
571  // save this value as the write to result.Row0.X could change the
572  // value of mat.Row0.X.
573  float row0x = mat.Row0.X;
574 
575  result.Row0.X = mat.Row1.Y * invDet;
576  result.Row0.Y = -mat.Row0.Y * invDet;
577  result.Row1.X = -mat.Row1.X * invDet;
578  result.Row1.Y = row0x * invDet;
579  }
