58  public static List<Vertices> ConvexPartition(Vertices vertices, TriangulationAlgorithm algorithm,
59  bool discardAndFixInvalid = true, float tolerance = 0.001f)
60  {
61  if (vertices.Count <= 3)
62  {
63  return new List<Vertices> {vertices};
64  }
65 
66  List<Vertices> results = null;
67 
68  switch (algorithm)
69  {
70  case TriangulationAlgorithm.Earclip:
71  if (Settings.SkipSanityChecks)
72  {
73  Debug.Assert(!vertices.IsCounterClockWise(),
74  "The Ear-clip algorithm expects the polygon to be clockwise.");
75  }
76  else
77  {
78  if (vertices.IsCounterClockWise())
79  {
80  Vertices temp = new Vertices(vertices);
81  temp.Reverse();
82  results = EarclipDecomposer.ConvexPartition(temp, tolerance);
83  }
84  else
85  {
86  results = EarclipDecomposer.ConvexPartition(vertices, tolerance);
87  }
88  }
89 
90  break;
91  case TriangulationAlgorithm.Bayazit:
92  if (Settings.SkipSanityChecks)
93  {
94  Debug.Assert(vertices.IsCounterClockWise(),
95  "The polygon is not counter clockwise. This is needed for Bayazit to work correctly.");
96  }
97  else
98  {
99  if (!vertices.IsCounterClockWise())
100  {
101  Vertices temp = new Vertices(vertices);
102  temp.Reverse();
103  results = BayazitDecomposer.ConvexPartition(temp);
104  }
105  else
106  {
107  results = BayazitDecomposer.ConvexPartition(vertices);
108  }
109  }
110 
111  break;
112  case TriangulationAlgorithm.Flipcode:
113  if (Settings.SkipSanityChecks)
114  {
115  Debug.Assert(vertices.IsCounterClockWise(),
116  "The polygon is not counter clockwise. This is needed for Bayazit to work correctly.");
117  }
118  else
119  {
120  if (!vertices.IsCounterClockWise())
121  {
122  Vertices temp = new Vertices(vertices);
123  temp.Reverse();
124  results = FlipcodeDecomposer.ConvexPartition(temp);
125  }
126  else
127  {
128  results = FlipcodeDecomposer.ConvexPartition(vertices);
129  }
130  }
131 
132  break;
133  case TriangulationAlgorithm.Seidel:
134  results = SeidelDecomposer.ConvexPartition(vertices, tolerance);
135  break;
136  case TriangulationAlgorithm.SeidelTrapezoids:
137  results = SeidelDecomposer.ConvexPartitionTrapezoid(vertices, tolerance);
138  break;
139  case TriangulationAlgorithm.Delauny:
140  results = CdtDecomposer.ConvexPartition(vertices);
141  break;
142  default:
143  throw new ArgumentOutOfRangeException(nameof(algorithm));
144  }
145 
146  if (discardAndFixInvalid)
147  {
148  for (int i = results.Count - 1; i >= 0; i--)
149  {
150  Vertices polygon = results[i];
151 
152  if (!ValidatePolygon(polygon))
153  {
154  results.RemoveAt(i);
155  }
156  }
157  }
158 
159  return results;
160  }

167  private static bool ValidatePolygon(Vertices polygon)
168  {
169  PolygonError errorCode = polygon.CheckPolygon();
170 
171  if (errorCode == PolygonError.InvalidAmountOfVertices || errorCode == PolygonError.AreaTooSmall ||
172  errorCode == PolygonError.SideTooSmall || errorCode == PolygonError.NotSimple)
173  {
174  return false;
175  }
176 
177  if (
178  errorCode ==
179  PolygonError
180  .NotCounterClockWise) //NotCounterCloseWise is the last check in CheckPolygon(), thus we don't need to call ValidatePolygon again.
181  {
182  polygon.Reverse();
183  }
184 
185  if (errorCode == PolygonError.NotConvex)
186  {
187  polygon = GiftWrap.GetConvexHull(polygon);
188  return ValidatePolygon(polygon);
189  }
190 
191  return true;
192  }
193  }
