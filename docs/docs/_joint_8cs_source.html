258  internal void Validate(float invDt)
259  {
260  if (!Enabled)
261  {
262  return;
263  }
264 
265  float jointErrorSquared = GetReactionForce(invDt).LengthSquared();
266 
267  if (Math.Abs(jointErrorSquared) <= breakpoint * breakpoint)
268  {
269  return;
270  }
271 
272  Enabled = false;
273 
274  Broke?.Invoke(this, (float) Math.Sqrt(jointErrorSquared));
275  }

309  case JointType.Wheel:

311  case JointType.Weld:
