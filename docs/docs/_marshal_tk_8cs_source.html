80  public static IntPtr MarshalStringToPtr(string str)
81  {
82  if (string.IsNullOrEmpty(str))
83  {
84  return IntPtr.Zero;
85  }
86 
87  // Allocate a buffer big enough to hold the marshaled string.
88  // GetMaxByteCount() appears to allocate space for the final NUL
89  // character, but allocate an extra one just in case (who knows
90  // what old Mono version would do here.)
91  int maxCount = Encoding.UTF8.GetMaxByteCount(str.Length) + 1;
92  IntPtr ptr = Marshal.AllocHGlobal(maxCount);
93  if (ptr == IntPtr.Zero)
94  {
95  throw new OutOfMemoryException();
96  }
97 
98  // Pin the managed string and convert it to UTF8 using
99  // the pointer overload of System.Encoding.UTF8.GetBytes().
100  unsafe
101  {
102  fixed (char* pstr = str)
103  {
104  int actualCount = Encoding.UTF8.GetBytes(pstr, str.Length, (byte*) ptr, maxCount);
105  Marshal.WriteByte(ptr, actualCount, 0); // Append '\0' at the end of the string
106  return ptr;
107  }
108  }
109  }

126  public static IntPtr MarshalStringArrayToPtr(string[] strArray)
127  {
128  IntPtr ptr = IntPtr.Zero;
129  if (strArray != null && strArray.Length != 0)
130  {
131  ptr = Marshal.AllocHGlobal(strArray.Length * IntPtr.Size);
132  if (ptr == IntPtr.Zero)
133  {
134  throw new OutOfMemoryException();
135  }
136 
137  int i = 0;
138  try
139  {
140  for (i = 0; i < strArray.Length; i++)
141  {
142  IntPtr str = MarshalStringToPtr(strArray[i]);
143  Marshal.WriteIntPtr(ptr, i * IntPtr.Size, str);
144  }
145  }
146  catch (OutOfMemoryException)
147  {
148  for (i = i - 1; i >= 0; --i)
149  {
150  Marshal.FreeHGlobal(Marshal.ReadIntPtr(ptr, i * IntPtr.Size));
151  }
152 
153  Marshal.FreeHGlobal(ptr);
154 
155  throw;
156  }
157  }
158 
159  return ptr;
160  }
