318  public static void CreateFromAxisAngle(Vector3d axis, double angle, out Matrix3x4d result)
319  {
320  axis.Normalize();
321  double axisX = axis.X, axisY = axis.Y, axisZ = axis.Z;
322 
323  double cos = Math.Cos(angle);
324  double sin = Math.Sin(angle);
325  double t = 1.0f - cos;
326 
327  double tXX = t * axisX * axisX;
328  double tXY = t * axisX * axisY;
329  double tXZ = t * axisX * axisZ;
330  double tYY = t * axisY * axisY;
331  double tYZ = t * axisY * axisZ;
332  double tZZ = t * axisZ * axisZ;
333 
334  double sinX = sin * axisX;
335  double sinY = sin * axisY;
336  double sinZ = sin * axisZ;
337 
338  result.Row0.X = tXX + cos;
339  result.Row0.Y = tXY - sinZ;
340  result.Row0.Z = tXZ + sinY;
341  result.Row0.W = 0;
342  result.Row1.X = tXY + sinZ;
343  result.Row1.Y = tYY + cos;
344  result.Row1.Z = tYZ - sinX;
345  result.Row1.W = 0;
346  result.Row2.X = tXZ - sinY;
347  result.Row2.Y = tYZ + sinX;
348  result.Row2.Z = tZZ + cos;
349  result.Row2.W = 0;
350  }

370  public static void CreateFromQuaternion(in Quaternion q, out Matrix3x4d result)
371  {
372  double x = q.X;
373  double y = q.Y;
374  double z = q.Z;
375  double w = q.W;
376  double tx = 2 * x;
377  double ty = 2 * y;
378  double tz = 2 * z;
379  double txx = tx * x;
380  double tyy = ty * y;
381  double tzz = tz * z;
382  double txy = tx * y;
383  double txz = tx * z;
384  double tyz = ty * z;
385  double txw = tx * w;
386  double tyw = ty * w;
387  double tzw = tz * w;
388 
389  result.Row0.X = 1f - (tyy + tzz);
390  result.Row0.Y = txy + tzw;
391  result.Row0.Z = txz - tyw;
392  result.Row0.W = 0f;
393  result.Row1.X = txy - tzw;
394  result.Row1.Y = 1f - (txx + tzz);
395  result.Row1.Z = tyz + txw;
396  result.Row1.W = 0f;
397  result.Row2.X = txz + tyw;
398  result.Row2.Y = tyz - txw;
399  result.Row2.Z = 1f - (txx + tyy);
400  result.Row2.W = 0f;
401  }

654  public static void Mult(in Matrix3x4d left, in Matrix4x3d right, out Matrix3d result)
655  {
656  double leftM11 = left.Row0.X;
657  double leftM12 = left.Row0.Y;
658  double leftM13 = left.Row0.Z;
659  double leftM14 = left.Row0.W;
660  double leftM21 = left.Row1.X;
661  double leftM22 = left.Row1.Y;
662  double leftM23 = left.Row1.Z;
663  double leftM24 = left.Row1.W;
664  double leftM31 = left.Row2.X;
665  double leftM32 = left.Row2.Y;
666  double leftM33 = left.Row2.Z;
667  double leftM34 = left.Row2.W;
668  double rightM11 = right.Row0.X;
669  double rightM12 = right.Row0.Y;
670  double rightM13 = right.Row0.Z;
671  double rightM21 = right.Row1.X;
672  double rightM22 = right.Row1.Y;
673  double rightM23 = right.Row1.Z;
674  double rightM31 = right.Row2.X;
675  double rightM32 = right.Row2.Y;
676  double rightM33 = right.Row2.Z;
677  double rightM41 = right.Row3.X;
678  double rightM42 = right.Row3.Y;
679  double rightM43 = right.Row3.Z;
680 
681  result.Row0.X = leftM11 * rightM11 + leftM12 * rightM21 + leftM13 * rightM31 + leftM14 * rightM41;
682  result.Row0.Y = leftM11 * rightM12 + leftM12 * rightM22 + leftM13 * rightM32 + leftM14 * rightM42;
683  result.Row0.Z = leftM11 * rightM13 + leftM12 * rightM23 + leftM13 * rightM33 + leftM14 * rightM43;
684  result.Row1.X = leftM21 * rightM11 + leftM22 * rightM21 + leftM23 * rightM31 + leftM24 * rightM41;
685  result.Row1.Y = leftM21 * rightM12 + leftM22 * rightM22 + leftM23 * rightM32 + leftM24 * rightM42;
686  result.Row1.Z = leftM21 * rightM13 + leftM22 * rightM23 + leftM23 * rightM33 + leftM24 * rightM43;
687  result.Row2.X = leftM31 * rightM11 + leftM32 * rightM21 + leftM33 * rightM31 + leftM34 * rightM41;
688  result.Row2.Y = leftM31 * rightM12 + leftM32 * rightM22 + leftM33 * rightM32 + leftM34 * rightM42;
689  result.Row2.Z = leftM31 * rightM13 + leftM32 * rightM23 + leftM33 * rightM33 + leftM34 * rightM43;
690  }

713  double leftM11 = left.Row0.X;
714  double leftM12 = left.Row0.Y;
715  double leftM13 = left.Row0.Z;
716  double leftM14 = left.Row0.W;
717  double leftM21 = left.Row1.X;
718  double leftM22 = left.Row1.Y;
719  double leftM23 = left.Row1.Z;
720  double leftM24 = left.Row1.W;
721  double leftM31 = left.Row2.X;
722  double leftM32 = left.Row2.Y;
723  double leftM33 = left.Row2.Z;
724  double leftM34 = left.Row2.W;
725  double rightM11 = right.Row0.X;
726  double rightM12 = right.Row0.Y;
727  double rightM13 = right.Row0.Z;
728  double rightM14 = right.Row0.W;
729  double rightM21 = right.Row1.X;
730  double rightM22 = right.Row1.Y;

736  double rightM34 = right.Row2.W;

738  result.Row0.X = leftM11 * rightM11 + leftM12 * rightM21 + leftM13 * rightM31;
739  result.Row0.Y = leftM11 * rightM12 + leftM12 * rightM22 + leftM13 * rightM32;
740  result.Row0.Z = leftM11 * rightM13 + leftM12 * rightM23 + leftM13 * rightM33;
741  result.Row0.W = leftM11 * rightM14 + leftM12 * rightM24 + leftM13 * rightM34 + leftM14;

747  result.Row2.Y = leftM31 * rightM12 + leftM32 * rightM22 + leftM33 * rightM32;
748  result.Row2.Z = leftM31 * rightM13 + leftM32 * rightM23 + leftM33 * rightM33;
749  result.Row2.W = leftM31 * rightM14 + leftM32 * rightM24 + leftM33 * rightM34 + leftM34;

849  public static void Invert(in Matrix3x4d mat, out Matrix3x4d result)
850  {
851  Matrix3d inverseRotation = new Matrix3d(mat.Column0, mat.Column1, mat.Column2);
852  inverseRotation.Row0 /= inverseRotation.Row0.LengthSquared;
853  inverseRotation.Row1 /= inverseRotation.Row1.LengthSquared;
854  inverseRotation.Row2 /= inverseRotation.Row2.LengthSquared;
855 
856  Vector3d translation = new Vector3d(mat.Row0.W, mat.Row1.W, mat.Row2.W);
857 
858  result.Row0 = new Vector4d(inverseRotation.Row0, -Vector3d.Dot(inverseRotation.Row0, translation));
859  result.Row1 = new Vector4d(inverseRotation.Row1, -Vector3d.Dot(inverseRotation.Row1, translation));
860  result.Row2 = new Vector4d(inverseRotation.Row2, -Vector3d.Dot(inverseRotation.Row2, translation));
861  }
