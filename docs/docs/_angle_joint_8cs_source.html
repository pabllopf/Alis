159  internal override void InitVelocityConstraints(ref SolverData data)
160  {
161  int indexA = BodyA.IslandIndex;
162  int indexB = BodyB.IslandIndex;
163 
164  float aW = data.Positions[indexA].A;
165  float bW = data.Positions[indexB].A;
166 
167  jointError = bW - aW - targetAngle;
168  bias = -biasFactor * data.Step.InvertedDeltaTime * jointError;
169  massFactor = (1 - softness) / (BodyA.InvI + BodyB.InvI);
170  }

176  internal override void SolveVelocityConstraints(ref SolverData data)
177  {
178  int indexA = BodyA.IslandIndex;
179  int indexB = BodyB.IslandIndex;
180 
181  float p = (bias - data.Velocities[indexB].W + data.Velocities[indexA].W) * massFactor;
182 
183  data.Velocities[indexA].W -=
184  BodyA.InvI * MathUtils.Sign(p) * MathUtils.Min(MathUtils.Abs(p), maxImpulse);
185  data.Velocities[indexB].W +=
186  BodyB.InvI * MathUtils.Sign(p) * MathUtils.Min(MathUtils.Abs(p), maxImpulse);
187  }
