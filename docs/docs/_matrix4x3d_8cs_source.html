331  public static void CreateFromAxisAngle(Vector3d axis, double angle, out Matrix4x3d result)
332  {
333  axis.Normalize();
334  double axisX = axis.X, axisY = axis.Y, axisZ = axis.Z;
335 
336  double cos = Math.Cos(-angle);
337  double sin = Math.Sin(-angle);
338  double t = 1.0f - cos;
339 
340  double tXX = t * axisX * axisX;
341  double tXY = t * axisX * axisY;
342  double tXZ = t * axisX * axisZ;
343  double tYY = t * axisY * axisY;
344  double tYZ = t * axisY * axisZ;
345  double tZZ = t * axisZ * axisZ;
346 
347  double sinX = sin * axisX;
348  double sinY = sin * axisY;
349  double sinZ = sin * axisZ;
350 
351  result.Row0.X = tXX + cos;
352  result.Row0.Y = tXY - sinZ;
353  result.Row0.Z = tXZ + sinY;
354  result.Row1.X = tXY + sinZ;
355  result.Row1.Y = tYY + cos;
356  result.Row1.Z = tYZ - sinX;
357  result.Row2.X = tXZ - sinY;
358  result.Row2.Y = tYZ + sinX;
359  result.Row2.Z = tZZ + cos;
360  result.Row3.X = 0;
361  result.Row3.Y = 0;
362  result.Row3.Z = 0;
363  }

383  public static void CreateFromQuaternion(in Quaternion q, out Matrix4x3d result)
384  {
385  double x = q.X;
386  double y = q.Y;
387  double z = q.Z;
388  double w = q.W;
389  double tx = 2 * x;
390  double ty = 2 * y;
391  double tz = 2 * z;
392  double txx = tx * x;
393  double tyy = ty * y;
394  double tzz = tz * z;
395  double txy = tx * y;
396  double txz = tx * z;
397  double tyz = ty * z;
398  double twx = w * tx;
399  double twy = w * ty;
400  double twz = w * tz;
401 
402  result.Row0.X = 1f - tyy - tzz;
403  result.Row0.Y = txy - twz;
404  result.Row0.Z = txz + twy;
405  result.Row1.X = txy + twz;
406  result.Row1.Y = 1f - txx - tzz;
407  result.Row1.Z = tyz - twx;
408  result.Row2.X = txz - twy;
409  result.Row2.Y = tyz + twx;
410  result.Row2.Z = 1f - txx - tyy;
411  result.Row3.X = 0;
412  result.Row3.Y = 0;
413  result.Row3.Z = 0;
414  }

669  public static void Mult(in Matrix4x3d left, in Matrix3x4d right, out Matrix4d result)
670  {
671  double leftM11 = left.Row0.X;
672  double leftM12 = left.Row0.Y;
673  double leftM13 = left.Row0.Z;
674  double leftM21 = left.Row1.X;
675  double leftM22 = left.Row1.Y;
676  double leftM23 = left.Row1.Z;
677  double leftM31 = left.Row2.X;
678  double leftM32 = left.Row2.Y;
679  double leftM33 = left.Row2.Z;
680  double leftM41 = left.Row3.X;
681  double leftM42 = left.Row3.Y;
682  double leftM43 = left.Row3.Z;
683  double rightM11 = right.Row0.X;
684  double rightM12 = right.Row0.Y;
685  double rightM13 = right.Row0.Z;
686  double rightM14 = right.Row0.W;
687  double rightM21 = right.Row1.X;
688  double rightM22 = right.Row1.Y;
689  double rightM23 = right.Row1.Z;
690  double rightM24 = right.Row1.W;
691  double rightM31 = right.Row2.X;
692  double rightM32 = right.Row2.Y;
693  double rightM33 = right.Row2.Z;
694  double rightM34 = right.Row2.W;
695 
696  result.Row0.X = leftM11 * rightM11 + leftM12 * rightM21 + leftM13 * rightM31;
697  result.Row0.Y = leftM11 * rightM12 + leftM12 * rightM22 + leftM13 * rightM32;
698  result.Row0.Z = leftM11 * rightM13 + leftM12 * rightM23 + leftM13 * rightM33;
699  result.Row0.W = leftM11 * rightM14 + leftM12 * rightM24 + leftM13 * rightM34;
700  result.Row1.X = leftM21 * rightM11 + leftM22 * rightM21 + leftM23 * rightM31;
701  result.Row1.Y = leftM21 * rightM12 + leftM22 * rightM22 + leftM23 * rightM32;
702  result.Row1.Z = leftM21 * rightM13 + leftM22 * rightM23 + leftM23 * rightM33;
703  result.Row1.W = leftM21 * rightM14 + leftM22 * rightM24 + leftM23 * rightM34;
704  result.Row2.X = leftM31 * rightM11 + leftM32 * rightM21 + leftM33 * rightM31;
705  result.Row2.Y = leftM31 * rightM12 + leftM32 * rightM22 + leftM33 * rightM32;
706  result.Row2.Z = leftM31 * rightM13 + leftM32 * rightM23 + leftM33 * rightM33;
707  result.Row2.W = leftM31 * rightM14 + leftM32 * rightM24 + leftM33 * rightM34;
708  result.Row3.X = leftM41 * rightM11 + leftM42 * rightM21 + leftM43 * rightM31;
709  result.Row3.Y = leftM41 * rightM12 + leftM42 * rightM22 + leftM43 * rightM32;
710  result.Row3.Z = leftM41 * rightM13 + leftM42 * rightM23 + leftM43 * rightM33;
711  result.Row3.W = leftM41 * rightM14 + leftM42 * rightM24 + leftM43 * rightM34;
712  }

736  double leftM12 = left.Row0.Y;

743  double leftM33 = left.Row2.Z;

761  result.Row0.Y = leftM11 * rightM12 + leftM12 * rightM22 + leftM13 * rightM32 + rightM42;

767  result.Row2.Y = leftM31 * rightM12 + leftM32 * rightM22 + leftM33 * rightM32 + rightM42;

770  result.Row3.Y = leftM41 * rightM12 + leftM42 * rightM22 + leftM43 * rightM32 + rightM42;

874  public static void Invert(in Matrix4x3d mat, out Matrix4x3d result)
875  {
876  Matrix3d inverseRotation = new Matrix3d(mat.Column0.Xyz, mat.Column1.Xyz, mat.Column2.Xyz);
877  inverseRotation.Row0 /= inverseRotation.Row0.LengthSquared;
878  inverseRotation.Row1 /= inverseRotation.Row1.LengthSquared;
879  inverseRotation.Row2 /= inverseRotation.Row2.LengthSquared;
880 
881  Vector3d translation = mat.Row3;
882 
883  result.Row0 = inverseRotation.Row0;
884  result.Row1 = inverseRotation.Row1;
885  result.Row2 = inverseRotation.Row2;
886  result.Row3 = new Vector3d
887  (
888  -Vector3d.Dot(inverseRotation.Row0, translation),
889  -Vector3d.Dot(inverseRotation.Row1, translation),
890  -Vector3d.Dot(inverseRotation.Row2, translation)
891  );
892  }
