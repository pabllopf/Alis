160  public void PrepareTriangulation(TriangulationContext tcx)
161  {
162  if (TrianglesPrivate == null)
163  {
164  TrianglesPrivate = new List<DelaunayTriangle>(PointsPrivate.Count);
165  }
166  else
167  {
168  TrianglesPrivate.Clear();
169  }
170 
171  // Outer constraints
172  for (int i = 0; i < PointsPrivate.Count - 1; i++)
173  {
174  tcx.NewConstraint(PointsPrivate[i], PointsPrivate[i + 1]);
175  }
176 
177  tcx.NewConstraint(PointsPrivate[0], PointsPrivate[PointsPrivate.Count - 1]);
178  tcx.Points.AddRange(PointsPrivate);
179 
180  // Hole constraints
181  if (HolesPrivate != null)
182  {
183  foreach (Polygon p in HolesPrivate)
184  {
185  for (int i = 0; i < p.PointsPrivate.Count - 1; i++)
186  {
187  tcx.NewConstraint(p.PointsPrivate[i], p.PointsPrivate[i + 1]);
188  }
189 
190  tcx.NewConstraint(p.PointsPrivate[0], p.PointsPrivate[p.PointsPrivate.Count - 1]);
191  tcx.Points.AddRange(p.PointsPrivate);
192  }
193  }
194 
195  if (SteinerPointsPrivate != null)
196  {
197  tcx.Points.AddRange(SteinerPointsPrivate);
198  }
199  }

258  public void InsertPointAfter(PolygonPoint point, PolygonPoint newPoint)
259  {
260  // Validate that
261  int index = PointsPrivate.IndexOf(point);
262  if (index == -1)
263  {
264  throw new ArgumentException(
265  "Tried to insert a point into a Polygon after a point not belonging to the Polygon", nameof(point));
266  }
267 
268  newPoint.Next = point.Next;
269  newPoint.Previous = point;
270  point.Next.Previous = newPoint;
271  point.Next = newPoint;
272  PointsPrivate.Insert(index + 1, newPoint);
273  }

277  public void AddPoints(IEnumerable<PolygonPoint> list)
278  {
279  foreach (PolygonPoint p in list)
280  {
281  p.Previous = LastPrivate;
282  if (LastPrivate != null)
283  {
284  p.Next = LastPrivate.Next;
285  LastPrivate.Next = p;
286  }
287 
288  LastPrivate = p;
289  PointsPrivate.Add(p);
290  }
291 
292  PolygonPoint first = (PolygonPoint) PointsPrivate[0];
293  LastPrivate.Next = first;
294  first.Previous = LastPrivate;
295  }
