237  internal override void InitVelocityConstraints(ref SolverData data)
238  {
239  indexA = BodyA.IslandIndex;
240  localCenterA = BodyA.Sweep.LocalCenter;
241  invMassA = BodyA.InvMass;
242  invIa = BodyA.InvI;
243 
244  Vector2 cA = data.Positions[indexA].C;
245  float aA = data.Positions[indexA].A;
246  Vector2 vA = data.Velocities[indexA].V;
247  float wA = data.Velocities[indexA].W;
248 
249  Rot qA = new Rot(aA);
250 
251  float d = damping;
252  float k = stiffness;
253 
254  // magic formulas
255  // gamma has units of inverse mass.
256  // beta has units of inverse time.
257  float h = data.Step.DeltaTime;
258  gamma = h * (d + h * k);
259  if (gamma != 0.0f)
260  {
261  gamma = 1.0f / gamma;
262  }
263 
264  beta = h * k * gamma;
265 
266  // Compute the effective mass matrix.
267  rA = MathUtils.Mul(qA, localAnchorA - localCenterA);
268 
269  // K = [(1/m1 + 1/m2) * eye(2) - skew(r1) * invI1 * skew(r1) - skew(r2) * invI2 * skew(r2)]
270  // = [1/m1+1/m2 0 ] + invI1 * [r1.y*r1.y -r1.x*r1.y] + invI2 * [r1.y*r1.y -r1.x*r1.y]
271  // [ 0 1/m1+1/m2] [-r1.x*r1.y r1.x*r1.x] [-r1.x*r1.y r1.x*r1.x]
272  Mat22 kk = new Mat22();
273  kk.Ex.X = invMassA + invIa * rA.Y * rA.Y + gamma;
274  kk.Ex.Y = -invIa * rA.X * rA.Y;
275  kk.Ey.X = kk.Ex.Y;
276  kk.Ey.Y = invMassA + invIa * rA.X * rA.X + gamma;
277 
278  mass = kk.Inverse;
279 
280  c = cA + rA - targetB;
281  c *= beta;
282 
283  // Cheat with some damping
284  wA *= 0.98f;
285 
286  if (data.Step.WarmStarting)
287  {
288  impulse *= data.Step.DeltaTimeRatio;
289  vA += invMassA * impulse;
290  wA += invIa * MathUtils.Cross(rA, impulse);
291  }
292  else
293  {
294  impulse = Vector2.Zero;
295  }
296 
297  data.Velocities[indexA].V = vA;
298  data.Velocities[indexA].W = wA;
299  }
300 
305  internal override void SolveVelocityConstraints(ref SolverData data)
306  {
307  Vector2 vA = data.Velocities[indexA].V;
308  float wA = data.Velocities[indexA].W;
309 
310  // Cdot = v + cross(w, r)
311  Vector2 cdot = vA + MathUtils.Cross(wA, rA);
312  Vector2 impulse = MathUtils.Mul(ref mass, -(cdot + c + gamma * this.impulse));
313 
314  Vector2 oldImpulse = this.impulse;
315  this.impulse += impulse;
316  float maxImpulse = data.Step.DeltaTime * maxForce;
317  if (this.impulse.LengthSquared() > maxImpulse * maxImpulse)
318  {
319  this.impulse *= maxImpulse / this.impulse.Length();
320  }
321 
322  impulse = this.impulse - oldImpulse;
323 
324  vA += invMassA * impulse;
325  wA += invIa * MathUtils.Cross(rA, impulse);
326 
327  data.Velocities[indexA].V = vA;
328  data.Velocities[indexA].W = wA;
329  }
