166  private void ParseReadString(char quote, ReadType readType)
167  {
168  SetPostValueState(true);
169 
170  switch (readType)
171  {
172  case ReadType.ReadAsBytes:
173  Guid g;
174  byte[] data;
175  if (_stringReference.Length == 0)
176  {
177  data = CollectionUtils.ArrayEmpty<byte>();
178  }
179  else if (_stringReference.Length == 36 && ConvertUtils.TryConvertGuid(_stringReference.ToString(), out g))
180  {
181  data = g.ToByteArray();
182  }
183  else
184  {
185  data = Convert.FromBase64CharArray(_stringReference.Chars, _stringReference.StartIndex, _stringReference.Length);
186  }
187 
188  SetToken(JsonToken.Bytes, data, false);
189  break;
190  case ReadType.ReadAsString:
191  string text = _stringReference.ToString();
192 
193  SetToken(JsonToken.String, text, false);
194  _quoteChar = quote;
195  break;
196  case ReadType.ReadAsInt32:
197  case ReadType.ReadAsDecimal:
198  case ReadType.ReadAsBoolean:
199  // caller will convert result
200  break;
201  default:
202  if (_dateParseHandling != DateParseHandling.None)
203  {
204  DateParseHandling dateParseHandling;
205  if (readType == ReadType.ReadAsDateTime)
206  {
207  dateParseHandling = DateParseHandling.DateTime;
208  }
209 #if HAVE_DATE_TIME_OFFSET
210  else if (readType == ReadType.ReadAsDateTimeOffset)
211  {
212  dateParseHandling = DateParseHandling.DateTimeOffset;
213  }
214 #endif
215  else

217  dateParseHandling = _dateParseHandling;
218  }
219

221  {

227  }

235  }

281  private void PrepareBufferForReadData(bool append, int charsRequired)
282  {
283  MiscellaneousUtils.Assert(_chars != null);
284 
285  // char buffer is full
286  if (_charsUsed + charsRequired >= _chars.Length - 1)
287  {
288  if (append)
289  {
290  int doubledArrayLength = _chars.Length * 2;
291 
292  // copy to new array either double the size of the current or big enough to fit required content
293  int newArrayLength = Math.Max(
294  doubledArrayLength < 0 ? int.MaxValue : doubledArrayLength, // handle overflow
295  _charsUsed + charsRequired + 1);
296 
297  // increase the size of the buffer
298  char[] dst = BufferUtils.RentBuffer(_arrayPool, newArrayLength);
299 
300  BlockCopyChars(_chars, 0, dst, 0, _chars.Length);
301 
302  BufferUtils.ReturnBuffer(_arrayPool, _chars);
303 
304  _chars = dst;
305  }
306  else
307  {
308  int remainingCharCount = _charsUsed - _charPos;
309 
310  if (remainingCharCount + charsRequired + 1 >= _chars.Length)
311  {
312  // the remaining count plus the required is bigger than the current buffer size
313  char[] dst = BufferUtils.RentBuffer(_arrayPool, remainingCharCount + charsRequired + 1);
314 
315  if (remainingCharCount > 0)
316  {
317  BlockCopyChars(_chars, _charPos, dst, 0, remainingCharCount);
318  }
319 
320  BufferUtils.ReturnBuffer(_arrayPool, _chars);
321 
322  _chars = dst;
323  }
324  else
325  {
326  // copy any remaining data to the beginning of the buffer if needed and reset positions
327  if (remainingCharCount > 0)
328  {
329  BlockCopyChars(_chars, _charPos, _chars, 0, remainingCharCount);
330  }
331  }
332 
333  _lineStartPos -= _charPos;
334  _charPos = 0;
335  _charsUsed = remainingCharCount;
336  }
337  }
338  }

375  private bool ReadChars(int relativePosition, bool append)
376  {
377  if (_isEndOfFile)
378  {
379  return false;
380  }
381 
382  int charsRequired = _charPos + relativePosition - _charsUsed + 1;
383 
384  int totalCharsRead = 0;
385 
386  // it is possible that the TextReader doesn't return all data at once
387  // repeat read until the required text is returned or the reader is out of content
388  do
389  {
390  int charsRead = ReadData(append, charsRequired - totalCharsRead);
391 
392  // no more content
393  if (charsRead == 0)
394  {
395  break;
396  }
397 
398  totalCharsRead += charsRead;
399  } while (totalCharsRead < charsRequired);
400 
401  if (totalCharsRead < charsRequired)
402  {
403  return false;
404  }
405  return true;

414  public override bool Read()
415  {
416  EnsureBuffer();
417  MiscellaneousUtils.Assert(_chars != null);
418 
419  while (true)
420  {
421  switch (_currentState)
422  {
423  case State.Start:
424  case State.Property:
425  case State.Array:
426  case State.ArrayStart:
427  case State.Constructor:
428  case State.ConstructorStart:
429  return ParseValue();
430  case State.Object:
431  case State.ObjectStart:
432  return ParseObject();
433  case State.PostValue:
434  // returns true if it hits
435  // end of object or array
436  if (ParsePostValue(false))
437  {
438  return true;
439  }
440  break;
441  case State.Finished:
442  if (EnsureChars(0, false))
443  {
444  EatWhitespace();
445  if (_isEndOfFile)
446  {
447  SetToken(JsonToken.None);
448  return false;
449  }
450  if (_chars[_charPos] == '/')
451  {
452  ParseComment(true);
453  return true;
454  }
455 
456  throw JsonReaderException.Create(this, "Additional text encountered after finished reading JSON content: {0}.".FormatWith(CultureInfo.InvariantCulture, _chars[_charPos]));
457  }
458  SetToken(JsonToken.None);
459  return false;
460  default:
461  throw JsonReaderException.Create(this, "Unexpected state: {0}.".FormatWith(CultureInfo.InvariantCulture, CurrentState));
462  }
463  }

498  {

501 
502  bool isWrapped = false;
503 
504  switch (_currentState)
505  {
506  case State.PostValue:

512  case State.Start:
513  case State.Property:
514  case State.Array:
515  case State.ArrayStart:
516  case State.Constructor:
517  case State.ConstructorStart:
518  while (true)
519  {
520  char currentChar = _chars[_charPos];
521 
522  switch (currentChar)
523  {
524  case '\0':
525  if (ReadNullChar())
526  {
527  SetToken(JsonToken.None, null, false);
528  return null;
529  }
530  break;
531  case '"':
532  case '\'':
533  ParseString(currentChar, ReadType.ReadAsBytes);
534  byte[]? data = (byte[]?)Value;
535  if (isWrapped)
536  {
537  ReaderReadAndAssert();

543  }

589  }

594  }
